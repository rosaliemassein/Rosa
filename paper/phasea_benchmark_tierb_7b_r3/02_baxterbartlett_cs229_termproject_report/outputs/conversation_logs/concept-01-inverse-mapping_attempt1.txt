from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Inverse Mapping Diagram
        # Left side: Desired Behavior (Input)
        input_rect = Rectangle(width=3.8, height=2.2, color=BLUE)
        input_title = Text("Desired Behavior", font_size=24).move_to(input_rect.get_top() + DOWN * 0.4)
        input_items = VGroup(
            Text("Rise Time", font_size=18),
            Text("Steady State Error", font_size=18),
            Text("Stability", font_size=18)
        ).arrange(DOWN, buff=0.2).move_to(input_rect.get_center() + DOWN * 0.2)
        input_group = VGroup(input_rect, input_title, input_items).to_edge(LEFT, buff=0.8)

        # Right side: PID Constants (Output)
        output_rect = Rectangle(width=3.8, height=2.2, color=GREEN)
        output_title = Text("PID Constants", font_size=24).move_to(output_rect.get_top() + DOWN * 0.4)
        output_items = VGroup(
            MathTex("K_p", font_size=24),
            MathTex("K_i", font_size=24),
            MathTex("K_d", font_size=24)
        ).arrange(DOWN, buff=0.2).move_to(output_rect.get_center() + DOWN * 0.2)
        output_group = VGroup(output_rect, output_title, output_items).to_edge(RIGHT, buff=0.8)

        # Large arrow between them
        connect_arrow = Arrow(input_group.get_right(), output_group.get_left(), buff=0.1, stroke_width=6)
        
        diagram = VGroup(input_group, connect_arrow, output_group).shift(UP * 1.5)
        
        # Formula Reference from prompt
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=22).to_edge(UP, buff=0.2)

        # Intro animation
        self.play(FadeIn(diagram), Write(formula))
        self.wait(1)
        
        # "Flips the script" - Rotate the entire diagram 180 degrees
        # Using .animate.rotate instead of Rotate class to ensure compatibility
        self.play(diagram.animate.rotate(PI))
        self.wait(1)

        # 2. Motor Velocity Plot (Manual implementation to avoid disallowed Axes/DashedLine classes)
        origin_x, origin_y = -3.5, -2.5
        plot_origin = [origin_x, origin_y, 0]
        
        x_axis = Line(plot_origin, [origin_x + 7, origin_y, 0]).add_tip()
        y_axis = Line(plot_origin, [origin_x, origin_y + 2.5, 0]).add_tip()
        x_label = Text("Time", font_size=18).next_to(x_axis, RIGHT, buff=0.1)
        y_label = Text("RPM", font_size=18).next_to(y_axis, UP, buff=0.1)
        
        # Curve generation (Underdamped motor response)
        curve_points = []
        res = 80
        for i in range(res + 1):
            t_val = (i / res) * 6.0
            # Function: 1 - exp(-0.7*t) * cos(3*t)
            v_val = 1.0 - np.exp(-0.7 * t_val) * np.cos(3 * t_val)
            # Map values to screen: scale height by 1.5
            curve_points.append([origin_x + t_val, origin_y + (v_val * 1.5), 0])
        
        curve = VMobject(color=YELLOW)
        curve.set_points_as_corners(curve_points)
        
        # Setpoint line at y=1.0 (scaled to 1.5 units high)
        set_y = origin_y + 1.5
        setpoint = Line([origin_x, set_y, 0], [origin_x + 6.5, set_y, 0], color=WHITE, stroke_width=2).set_opacity(0.5)
        
        plot_elements = VGroup(x_axis, y_axis, x_label, y_label, setpoint)
        self.play(Create(plot_elements))
        self.play(Create(curve), run_time=3)

        rise_t2 = Line([origin_x + rt_x, origin_y - 0.3, 0], [origin_x + rt_x, origin_y - 0.1, 0], color=BLUE)
        rise_label = Text("Rise Time", font_size=14, color=BLUE).next_to(rise_line, DOWN, buff=0.1)

        error_label = Text("Steady State Error", font_size=14, color=RED).next_to(error_line, RIGHT, buff=0.1)
        
        self.play(Create(VGroup(rise_line, rise_t1, rise_t2)), Write(rise_label))
        self.play(Create(error_line), Write(error_label))
        self.wait(2)
