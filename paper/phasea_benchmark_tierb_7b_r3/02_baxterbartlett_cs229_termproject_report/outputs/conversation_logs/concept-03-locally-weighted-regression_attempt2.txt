from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-2, 5, 1],
            axis_config={"color": BLUE},
            tips=False
        )
        
        # 2. Generate training data (a noisy curve)
        np.random.seed(42)
        x_data = np.linspace(-4, 4, 40)
        y_data = 0.5 * np.cos(x_data) + 0.1 * x_data**2 + np.random.normal(0, 0.2, 40)
        
        # Create visual dots for the scatter plot
        scatter_dots = VGroup(*[
            Dot(axes.c2p(x, y), radius=0.08, color=GRAY, fill_opacity=0.5) 
            for x, y in zip(x_data, y_data)
        ])
        
        # 3. Query Point and its ValueTracker
        query_x_tracker = ValueTracker(-3.5)
        
        # The query dot itself
        query_dot = Dot(color=WHITE, z_index=10).scale(1.3)
        query_dot.add_updater(lambda m: m.move_to(axes.c2p(query_x_tracker.get_value(), 0)))
        
        # 4. Local Regression Line
        reg_line = Line(color=RED, stroke_width=6, z_index=5)
        tau = 0.8  # Bandwidth parameter for LWR

        def update_line(line):
            q_x = query_x_tracker.get_value()
            
            # Weighted Least Squares Calculation
            # Weights: w_i = exp(-(x_i - x)^2 / (2 * tau^2))
            weights = np.exp(-(x_data - q_x)**2 / (2 * tau**2))
            
            # W is a diagonal matrix of weights
            W = np.diag(weights)
            # X is the design matrix (with intercept)
            X = np.vstack([np.ones_like(x_data), x_data]).T
            # Y is the target vector
            Y = y_data.reshape(-1, 1)
            
            # beta = (X^T W X)^-1 X^T W Y
            try:
                XTWX = X.T @ W @ X + np.eye(2) * 1e-6 # Add epsilon for stability
                beta = np.linalg.solve(XTWX, X.T @ W @ Y)
                intercept, slope = beta.flatten()
                
                # Update line position (show a local segment)
                x_start, x_end = q_x - 1.0, q_x + 1.0
                y_start = intercept + slope * x_start
                y_end = intercept + slope * x_end
                
                line.put_start_and_end_on(
                    axes.c2p(x_start, y_start),
                    axes.c2p(x_end, y_end)
                )
                
                # Move the query dot to the calculated y-value on the line
                query_dot.move_to(axes.c2p(q_x, intercept + slope * q_x))
            except:
                pass

        reg_line.add_updater(update_line)

        # 5. Visual "Weight" Field (Radial Glow)
        weight_field = Dot(radius=1.5, color=YELLOW, fill_opacity=0.2, z_index=-1)
        weight_field.add_updater(lambda m: m.move_to(query_dot.get_center()))

        # 6. Updater for scatter dots (glowing effect)
        def update_scatter(m):
            q_x = query_x_tracker.get_value()
            for i, dot in enumerate(m):
                dist = abs(x_data[i] - q_x)
                # Calculate weight for visual properties
                weight = np.exp(-(dist**2) / (2 * tau**2))
                # Interpolate color from GRAY to YELLOW based on weight
                dot.set_color(interpolate_color(GRAY, YELLOW, weight))
                dot.set_fill(opacity=interpolate(0.3, 1.0, weight))

        scatter_dots.add_updater(update_scatter)

        # 7. Animation Sequence
        self.add(axes)
        self.play(Create(scatter_dots), run_time=1.5)
        self.wait(0.5)
        
        self.play(FadeIn(query_dot), FadeIn(weight_field))
        self.play(Create(reg_line))
        self.wait(1)

        # Animate the query point moving across the data range
        self.play(
            query_x_tracker.animate.set_value(3.5),
            run_time=8,
            rate_func=linear
        )
        self.wait(1)
        
        self.play(
            query_x_tracker.animate.set_value(0),
            run_time=3
        )
        self.wait(2)

        # Final cleanup
        self.play(
            FadeOut(reg_line),
            FadeOut(query_dot),
            FadeOut(weight_field),
            FadeOut(scatter_dots),
            FadeOut(axes)
        )
        self.wait(1)
