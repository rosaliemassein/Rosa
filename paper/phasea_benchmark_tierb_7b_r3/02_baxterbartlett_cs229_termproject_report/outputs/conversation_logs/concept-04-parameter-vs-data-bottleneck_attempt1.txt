from manim import *
import numpy as np

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Title and Header
        title = Text("Parameter vs. Data Bottleneck", font_size=36)
        title.to_edge(UP, buff=0.5)
        self.add(title)

        # 2. Setup Manual Axes for Left (NN) and Right (LWR)
        # We avoid 'Axes' as it is disallowed in the constrained subset.
        def create_custom_axes(origin_point, x_len=4, y_len=3):
            x_axis = Line(origin_point, origin_point + RIGHT * x_len, color=WHITE)
            y_axis = Line(origin_point, origin_point + UP * y_len, color=WHITE)
            return VGroup(x_axis, y_axis)

        left_origin = LEFT * 5 + DOWN * 0.5
        right_origin = RIGHT * 1 + DOWN * 0.5
        
        axes_left = create_custom_axes(left_origin)
        axes_right = create_custom_axes(right_origin)
        
        nn_label = Text("Neural Network: High Variance", color=RED, font_size=20).next_to(axes_left, UP)
        lwr_label = Text("LWR: Robust Linear", color=BLUE, font_size=20).next_to(axes_right, UP)

        # 3. Data Points (27 points)
        # Using a deterministic pseudo-random distribution
        np.random.seed(42)
        raw_x = np.linspace(0.5, 3.5, 15) # Simplified to 15 points for clarity
        raw_y = 0.6 * raw_x + 0.5 + np.random.normal(0, 0.3, 15)

        dots_left = VGroup()
        dots_right = VGroup()
        for x, y in zip(raw_x, raw_y):
            dots_left.add(Dot(left_origin + RIGHT * x + UP * y, radius=0.04, color=WHITE))
            dots_right.add(Dot(right_origin + RIGHT * x + UP * y, radius=0.04, color=WHITE))

        # 4. Neural Network Curve (Overfitting / Wiggling)
        # High-frequency oscillation around the points
        nn_curve = VMobject(color=RED)
        nn_points = []
        for x_val in np.linspace(0.5, 3.5, 100):
            # Underlying trend + high frequency noise
            y_val = 0.6 * x_val + 0.5 + 0.4 * np.sin(x_val * 15) + 0.2 * np.cos(x_val * 30)
            nn_points.append(left_origin + RIGHT * x_val + UP * y_val)
        nn_curve.set_points_as_corners(nn_points)

        # 5. LWR Curve (Smooth / Piecewise Linear)
        lwr_curve = Line(
            right_origin + RIGHT * 0.5 + UP * (0.6 * 0.5 + 0.5),
            right_origin + RIGHT * 3.5 + UP * (0.6 * 3.5 + 0.5),
            color=BLUE
        )

        # 6. MSE Visualization (Manual Bar Chart)
        mse_group = VGroup()
        
        # NN Bar
        nn_bar_height = 1.21
        nn_bar = Rectangle(width=0.6, height=nn_bar_height, color=RED, fill_opacity=0.8)
        nn_bar.move_to(DOWN * 3 + LEFT * 1)
        nn_bar_label = Text("NN: 1.21", font_size=18).next_to(nn_bar, UP, buff=0.1)
        
        # LWR Bar
        lwr_bar_height = 0.85
        lwr_bar = Rectangle(width=0.6, height=lwr_bar_height, color=BLUE, fill_opacity=0.8)
        lwr_bar.move_to(DOWN * 3 + RIGHT * 1)
        lwr_bar_label = Text("LWR: 0.85", font_size=18).next_to(lwr_bar, UP, buff=0.1)
        
        mse_title = Text("Test MSE", font_size=22).next_to(nn_bar, LEFT, buff=1)
        mse_group.add(nn_bar, lwr_bar, nn_bar_label, lwr_bar_label, mse_title)

        # 7. Formula
        formula = MathTex(
            r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}",
            font_size=28, color=YELLOW
        ).to_edge(RIGHT, buff=1).shift(DOWN * 2.5)

        # --- Animation Sequence ---
        self.play(Create(axes_left), Create(axes_right))
        self.play(FadeIn(dots_left), FadeIn(dots_right))
        self.wait(1)
        
        self.play(Write(nn_label), Write(lwr_label))
        self.play(Create(nn_curve), run_time=3)
        self.play(Create(lwr_curve), run_time=2)
        self.wait(1)
        
        self.play(
            FadeIn(mse_group),
            Write(formula)
        )
        self.wait(3)
