from manim import *
import numpy as np

class Concept02QuantifyingStability(Scene):
    def construct(self):
        # 1. Title (Replacing 'Title' class with 'Text' for compatibility)
        title = Text("Quantifying Stability", font_size=36).to_edge(UP)
        self.play(Write(title))

        # 2. Velocity Plot (Manual implementation to avoid 'Axes' and 'DashedLine')
        plot_origin = LEFT * 5 + DOWN * 1.5
        x_axis = Line(plot_origin, plot_origin + RIGHT * 5, color=WHITE)
        y_axis = Line(plot_origin, plot_origin + UP * 3.5, color=WHITE)
        
        x_label = Text("Time", font_size=16).next_to(x_axis, RIGHT)
        y_label = Text("RPM", font_size=16).next_to(y_axis, UP)
        
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))

        # 30 RPM level: assume y-axis 3.5 units = 70 RPM. So 30 RPM = 1.5 units.
        y_val_30 = 1.5
        # Standard Line instead of DashedLine to avoid disallowed features
        ref_line = Line(
            plot_origin + UP * y_val_30, 
            plot_origin + RIGHT * 5 + UP * y_val_30, 
            color=WHITE, 
            stroke_opacity=0.4
        )
        ref_text = Text("30 RPM", font_size=14).next_to(ref_line, RIGHT, buff=0.1)

        # Generate fluctuating data points
        # Manual point calculation to avoid 'axes.plot' and 'lambda x'
        pts = []
        num_points = 60
        for i in range(num_points + 1):
            alpha = i / num_points
            # Fluctuates around 30 RPM (y_val_30)
            noise = 0.2 * np.sin(i * 0.7) + 0.1 * np.cos(i * 1.3)
            current_y = y_val_30 + noise
            p = plot_origin + RIGHT * (alpha * 5) + UP * current_y
            pts.append(p)

        v_graph = VMobject(color=BLUE)
        v_graph.set_points_as_corners(pts)

        self.play(Create(v_graph))
        self.play(Create(ref_line), Write(ref_text))

        # 3. Highlight last 20% of the x-axis
        # x range is 5 units total. Last 20% is 1 unit.
        highlight_box = Rectangle(
            width=1.0, 
            height=3.5, 
            fill_color=BLUE, 
            fill_opacity=0.2, 
            stroke_width=0
        )
        highlight_box.move_to(plot_origin + RIGHT * 4.5 + UP * 1.75)
        h_text = Text("Final 20%", font_size=14, color=BLUE).next_to(highlight_box, UP, buff=0.1)
        
        self.play(FadeIn(highlight_box), Write(h_text))
        self.wait()

        # 4. Histogram (appearing on the right)
        hist_base = RIGHT * 1.5 + DOWN * 1.5
        heights = [0.4, 0.9, 2.8, 0.7, 0.3] # 30 RPM (index 2) is tallest
        bars = VGroup()
        for i in range(len(heights)):
            h = heights[i]
            bar = Rectangle(
                width=0.4, 
                height=h, 
                fill_color=YELLOW if i == 2 else WHITE, 
                fill_opacity=0.6,
                stroke_width=1
            )
            bar.move_to(hist_base + RIGHT * (i * 0.6), aligned_edge=DOWN)
            bars.add(bar)
        
        hist_title = Text("Speed Distribution", font_size=18).next_to(bars, UP)
        mode_label = Text("Mode (30 RPM)", font_size=14, color=YELLOW).next_to(bars[2], UP, buff=0.1)

        self.play(Create(bars), Write(hist_title))
        self.play(Write(mode_label))
        self.wait()

        # 5. Stability Formula
        # Using MathTex for the fraction
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=32
        ).to_edge(DOWN, buff=0.7)
        
        # Narration background
        self.play(Write(formula))
        self.wait(3)

        # Cleanup
        self.play(
            FadeOut(title),
            FadeOut(v_graph),
            FadeOut(x_axis),
            FadeOut(y_axis),
            FadeOut(x_label),
            FadeOut(y_label),
            FadeOut(ref_line),
            FadeOut(ref_text),
            FadeOut(highlight_box),
            FadeOut(h_text),
            FadeOut(bars),
            FadeOut(hist_title),
            FadeOut(mode_label),
            FadeOut(formula)
        )
        self.wait()
