from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Coordinate System Helper (Manual substitute for Axes)
        def c2p(x, y):
            # Scale coordinates to fit screen: 1 unit x = 1.5 units screen, 1 unit y = 1.2 units screen
            return x * RIGHT * 1.5 + y * UP * 1.2

        # 2. Axes Construction
        x_axis = Line(LEFT * 5.5, RIGHT * 5.5, color=BLUE)
        y_axis = Line(DOWN * 3.5, UP * 3.5, color=BLUE)
        self.add(x_axis, y_axis)

        # 3. Data Generation (Non-linear pattern: sin(x) + noise)
        np.random.seed(42)
        x_data = np.linspace(-3.2, 3.2, 50)
        y_data = np.sin(x_data) + np.random.normal(0, 0.15, 50)
        
        dots = VGroup(*[
            Dot(c2p(x, y), radius=0.08, color=GREEN)
            for x, y in zip(x_data, y_data)
        ])
        self.play(Create(dots))

        # 4. State Tracker (Substitute for ValueTracker)
        # We use a hidden dot and its x-coordinate to track the current query point position
        tracker = Dot(LEFT * 4.8).hide()
        self.add(tracker)
        
        # Query point: follows the path of the underlying function
        query_point = Dot(color=WHITE, radius=0.12)
        query_point.add_updater(lambda m: m.move_to(
            c2p(tracker.get_center()[0] / 1.5, np.sin(tracker.get_center()[0] / 1.5))
        ))

        # Weight field: circular visual indicator
        weight_field = Circle(radius=1.0, color=YELLOW, fill_opacity=0.3, stroke_width=0)
        weight_field.add_updater(lambda m: m.move_to(query_point.get_center()))

        # Interaction: Data points glow when near the query point
        def update_dots_behavior(group):
            qx = tracker.get_center()[0] / 1.5
            tau = 0.65
            for i, dot in enumerate(group):
                dist_sq = (x_data[i] - qx)**2
                weight = np.exp(-dist_sq / (2 * tau**2))
                # Update visual properties based on weight
                dot.set_fill(opacity=0.2 + 0.8 * weight)
                if weight > 0.45:
                    dot.set_color(YELLOW)
                else:
                    dot.set_color(GREEN)

        dots.add_updater(update_dots_behavior)

        # 5. Local Regression Line Logic
        # This line calculates a weighted linear fit for points near the query
        local_reg_line = Line(color=RED, stroke_width=6)
        
        def update_regression_line(line):
            qx = tracker.get_center()[0] / 1.5
            tau = 0.65
            weights = np.exp(-(x_data - qx)**2 / (2 * tau**2))
            
            # Weighted Least Squares Formula:
            # slope = (sum(w)*sum(wxy) - sum(wx)*sum(wy)) / (sum(w)*sum(wxx) - sum(wx)^2)
            s_w = np.sum(weights)
            s_wx = np.sum(weights * x_data)
            s_wy = np.sum(weights * y_data)
            s_wxx = np.sum(weights * x_data * x_data)
            s_wxy = np.sum(weights * x_data * y_data)
            
            denominator = s_w * s_wxx - s_wx**2
            if abs(denominator) < 1e-8:
                return
                
            slope = (s_w * s_wxy - s_wx * s_wy) / denominator
            intercept = (s_wy - slope * s_wx) / s_w
            
            # Define a short segment around the query point
            x_start, x_end = qx - 0.8, qx + 0.8
            p1 = c2p(x_start, slope * x_start + intercept)
            p2 = c2p(x_end, slope * x_end + intercept)
            line.put_start_and_end_on(p1, p2)

        local_reg_line.add_updater(update_regression_line)

        # 6. Formula and Scene Execution
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)")
        formula.to_corner(UP + RIGHT).scale(0.8)

        self.play(FadeIn(query_point), FadeIn(weight_field))
        self.add(local_reg_line)
        self.play(Write(formula))
        self.wait(1)
        
        # Animate the query point moving across the data range
        # Shift 9.6 units total (from -4.8 to +4.8 screen units)
        self.play(
            tracker.animate.shift(RIGHT * 9.6),
            run_time=10,
            rate_func=lambda t: t # Linear movement
        )
        
        self.wait(2)
