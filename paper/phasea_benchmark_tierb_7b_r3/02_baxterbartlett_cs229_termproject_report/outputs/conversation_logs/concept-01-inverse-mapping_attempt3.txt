from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Formula Reference (Top)
        formula_tex = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}")
        formula_tex.scale(0.8).to_edge(UP, buff=0.4)
        self.add(formula_tex)

        # 2. Left Input Box: Desired Behavior
        rect_in = Rectangle(width=4.5, height=1.5, color=BLUE)
        label_in = Text("Desired Behavior", font_size=22, color=BLUE)
        sub_in = Text("(Rise Time, SS Error, Stability)", font_size=14).next_to(label_in, DOWN, buff=0.1)
        group_in = VGroup(rect_in, label_in, sub_in).move_to(LEFT * 3.5 + UP * 0.5)

        # 3. Right Output Box: PID Constants
        rect_out = Rectangle(width=3.5, height=1.5, color=GREEN)
        label_out = Text("PID Constants", font_size=22, color=GREEN)
        group_out = VGroup(rect_out, label_out).move_to(RIGHT * 3.5 + UP * 0.5)

        # 4. Manual Arrow Construction
        line_arrow = Line(group_in.get_right(), group_out.get_left(), buff=0.2)
        tip_arrow = Triangle(color=WHITE).scale(0.12).rotate(-PI/2).move_to(line_arrow.get_end())
        arrow_obj = VGroup(line_arrow, tip_arrow)

        # 5. Assemble Diagram and Animate
        full_diagram = VGroup(group_in, arrow_obj, group_out)
        self.play(Create(group_in), Create(group_out))
        self.play(Create(arrow_obj))
        self.wait(1)

        # 6. Flip the Script (Rotation of entire diagram)
        self.play(full_diagram.animate.rotate(PI), run_time=1.5)
        self.wait(1)

        # 7. Motor Velocity Plot (Manual implementation of Axes)
        graph_origin = DOWN * 3 + LEFT * 4
        x_line = Line(graph_origin, graph_origin + RIGHT * 8)
        y_line = Line(graph_origin, graph_origin + UP * 2.5)
        label_x = Text("Time", font_size=18).next_to(x_line, DOWN, buff=0.1)
        label_y = Text("RPM", font_size=18).next_to(y_line, LEFT, buff=0.1)
        self.play(Create(x_line), Create(y_line), Write(label_x), Write(label_y))

        # 8. Setpoint (Manual implementation of Dashed Line)
        setpoint_height = 1.8
        dashes = VGroup()
        for k in range(25):
            start_pt = graph_origin + UP * setpoint_height + RIGHT * (k * 0.32)
            end_pt = start_pt + RIGHT * 0.16
            dashes.add(Line(start_pt, end_pt, color=YELLOW, stroke_width=2))
        self.play(Create(dashes))

        # 9. Motor Velocity Curve (Oscillation)
        # Using VMobject with set_points_as_corners to avoid disallowed plotting functions
        pts = []
        for j in range(101):
            curr_x = j * 0.08 # x axis from 0 to 8
            # PID-style underdamped response curve: 1.8 * (1 - e^(-0.6x) * cos(3x))
            curr_y = setpoint_height * (1 - np.exp(-0.6 * curr_x) * np.cos(3.0 * curr_x))
            pts.append(graph_origin + RIGHT * curr_x + UP * curr_y)
        
        velocity_curve = VMobject(color=BLUE)
        velocity_curve.set_points_as_corners(pts)
        self.play(Create(velocity_curve), run_time=3)

        # 10. Performance Indicators (Manual implementation of Braces)
        # Rise Time bracket (horizontal indicator on x-axis)
        rt_x_pos = 0.6
        rise_bracket = VGroup(
            Line(graph_origin + DOWN * 0.3, graph_origin + RIGHT * rt_x_pos + DOWN * 0.3),
            Line(graph_origin + DOWN * 0.2, graph_origin + DOWN * 0.4),
            Line(graph_origin + RIGHT * rt_x_pos + DOWN * 0.2, graph_origin + RIGHT * rt_x_pos + DOWN * 0.4)
        )
        rise_label = Text("Rise Time", font_size=18).next_to(rise_bracket, DOWN, buff=0.1)
        self.play(Create(rise_bracket), Write(rise_label))

        # Steady State Error (Vertical indicator at the tail end)
        error_x_pos = 7.6
        # Calculate curve height at this specific x position
        error_y_pos = setpoint_height * (1 - np.exp(-0.6 * error_x_pos) * np.cos(3.0 * error_x_pos))
        error_line_obj = Line(
            graph_origin + RIGHT * error_x_pos + UP * setpoint_height, 
            graph_origin + RIGHT * error_x_pos + UP * error_y_pos, 
            color=RED
        )
        error_label_obj = Text("Steady State Error", font_size=16).next_to(error_line_obj, RIGHT, buff=0.1)
        self.play(Create(error_line_obj), Write(error_label_obj))

        self.wait(2)
