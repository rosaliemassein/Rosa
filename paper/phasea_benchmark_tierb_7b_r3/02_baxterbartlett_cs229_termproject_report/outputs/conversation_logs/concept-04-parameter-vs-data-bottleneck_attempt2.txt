from manim import *
import numpy as np

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Title and Formula
        title = Text("Parameter vs. Data Bottleneck", font_size=32).to_edge(UP, buff=0.3)
        formula = MathTex(
            r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}", 
            font_size=28
        ).next_to(title, DOWN, buff=0.2)
        
        self.add(title, formula)

        # 2. Coordinate System Helpers (Manual)
        # We'll create two areas for plots
        left_origin = LEFT * 4 + DOWN * 0.5
        right_origin = RIGHT * 1 + DOWN * 0.5
        
        # Draw Left Axes (Neural Network)
        ax_l_x = Line(left_origin, left_origin + RIGHT * 3, color=WHITE)
        ax_l_y = Line(left_origin, left_origin + UP * 3, color=WHITE)
        label_nn = Text("Neural Network", font_size=24, color=RED).next_to(ax_l_y, UP, buff=0.2)
        label_variance = Text("High Variance", font_size=18, color=RED).next_to(ax_l_x, DOWN, buff=0.1)
        
        # Draw Right Axes (LWR)
        ax_r_x = Line(right_origin, right_origin + RIGHT * 3, color=WHITE)
        ax_r_y = Line(right_origin, right_origin + UP * 3, color=WHITE)
        label_lwr = Text("LWR", font_size=24, color=BLUE).next_to(ax_r_y, UP, buff=0.2)
        label_robust = Text("Robust Approximation", font_size=18, color=BLUE).next_to(ax_r_x, DOWN, buff=0.1)

        # 3. Data Points (The same for both)
        # Using 10 points to represent the small dataset of 27
        px = [0.3, 0.6, 0.9, 1.2, 1.5, 1.8, 2.1, 2.4, 2.7]
        py = [0.5, 1.4, 0.8, 1.9, 1.3, 2.5, 1.7, 2.8, 2.1]
        
        dots_l = VGroup(*[Dot(left_origin + x*RIGHT + y*UP, radius=0.05, color=WHITE) for x, y in zip(px, py)])
        dots_r = VGroup(*[Dot(right_origin + x*RIGHT + y*UP, radius=0.05, color=WHITE) for x, y in zip(px, py)])

        # 4. Neural Network Overfitting Curve (Wiggling)
        # We manually create a path that hits the points but wiggles wildly between them
        nn_points = []
        for i in range(len(px) - 1):
            p_start = left_origin + px[i]*RIGHT + py[i]*UP
            p_end = left_origin + px[i+1]*RIGHT + py[i+1]*UP
            # Add a middle point that is offset to create "wiggles"
            mid_x = (px[i] + px[i+1]) / 2
            mid_y = (py[i] + py[i+1]) / 2 + (0.6 if i % 2 == 0 else -0.6)
            p_mid = left_origin + mid_x*RIGHT + mid_y*UP
            nn_points.extend([p_start, p_mid])
        nn_points.append(left_origin + px[-1]*RIGHT + py[-1]*UP)
        
        nn_curve = VMobject(color=RED)
        nn_curve.set_points_as_corners(nn_points)

        # 5. LWR Smooth Path
        # A smoother linear-ish trend through the data
        lwr_curve = Line(
            right_origin + 0.2*RIGHT + 0.6*UP,
            right_origin + 2.8*RIGHT + 2.6*UP,
            color=BLUE
        )

        # 6. MSE Bar Chart (Manual)
        bar_nn_height = 1.21 * 1.2
        bar_lwr_height = 0.85 * 1.2
        
        bar_base_y = -3.2
        bar_nn = Rectangle(height=bar_nn_height, width=0.8, fill_opacity=0.8, color=RED, stroke_width=2)
        bar_nn.move_to(LEFT * 1.5 + UP * (bar_base_y + bar_nn_height/2))
        
        bar_lwr = Rectangle(height=bar_lwr_height, width=0.8, fill_opacity=0.8, color=BLUE, stroke_width=2)
        bar_lwr.move_to(RIGHT * 1.5 + UP * (bar_base_y + bar_lwr_height/2))
        
        val_nn = Text("1.21", font_size=20).next_to(bar_nn, UP, buff=0.1)
        val_lwr = Text("0.85", font_size=20).next_to(bar_lwr, UP, buff=0.1)
        tag_nn = Text("NN MSE", font_size=18).next_to(bar_nn, DOWN, buff=0.1)
        tag_lwr = Text("LWR MSE", font_size=18).next_to(bar_lwr, DOWN, buff=0.1)

        # 7. Animation Sequence
        self.play(
            Create(ax_l_x), Create(ax_l_y),
            Create(ax_r_x), Create(ax_r_y),
            Write(label_nn), Write(label_lwr)
        )
        self.play(FadeIn(dots_l), FadeIn(dots_r))
        self.wait(1)
        
        self.play(Create(nn_curve), Write(label_variance), run_time=2)
        self.wait(0.5)
        self.play(Create(lwr_curve), Write(label_robust), run_time=2)
        self.wait(1)
        
        self.play(
            Create(bar_nn), Create(bar_lwr),
            Write(val_nn), Write(val_lwr),
            Write(tag_nn), Write(tag_lwr)
        )
        
        # Highlight lower MSE
        rect_highlight = SurroundingRectangle(VGroup(bar_lwr, val_lwr), color=YELLOW)
        self.play(Create(rect_highlight))
        self.wait(2)
