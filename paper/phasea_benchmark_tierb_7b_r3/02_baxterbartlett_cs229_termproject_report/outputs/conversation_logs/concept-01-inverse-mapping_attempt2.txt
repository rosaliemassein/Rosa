import math
from manim import *

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Mapping Diagram (Top Half)
        # We use manual Rectangle and Text to avoid potential restricted class issues
        input_rect = Rectangle(width=4.0, height=1.5, color=BLUE)
        input_text = Text("Desired Behavior", font_size=24)
        input_sub = Text("(Rise Time, SSE, Stability)", font_size=16, color=GRAY).next_to(input_text, DOWN, buff=0.1)
        input_group = VGroup(input_rect, input_text, input_sub).shift(LEFT * 3 + UP * 2)

        output_rect = Rectangle(width=4.0, height=1.5, color=GREEN)
        output_text = Text("PID Constants", font_size=24)
        output_group = VGroup(output_rect, output_text).shift(RIGHT * 3 + UP * 2)

        # Arrow indicating the relationship
        con_arrow = Arrow(input_group.get_right(), output_group.get_left(), color=WHITE, buff=0.2)
        diagram = VGroup(input_group, output_group, con_arrow)

        # Voice: "In traditional robotics, we choose PID constants..."
        self.play(Create(input_group))
        self.play(GrowArrow(con_arrow))
        self.play(Create(output_group))
        self.wait(1)

        # Voice: "This paper flips the script..."
        # We use .animate.rotate instead of the Rotate class to avoid undefined identifier errors
        self.play(diagram.animate.rotate(PI))
        self.wait(1)

        # Formula Reference (Bottom-right corner)
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=24)
        formula.to_corner(UP + RIGHT, buff=0.5)
        self.play(Write(formula))

        # 2. Motor Velocity Plot (Bottom Half)
        # Drawing axes manually to bypass restricted 'Axes' class
        plot_origin = DOWN * 2.5 + LEFT * 4
        axis_h = Line(plot_origin, plot_origin + RIGHT * 8, color=WHITE)
        axis_v = Line(plot_origin, plot_origin + UP * 3, color=WHITE)
        
        label_time = Text("Time", font_size=18).next_to(axis_h, RIGHT)
        label_rpm = Text("RPM", font_size=18).next_to(axis_v, UP)
        
        # Setpoint line (Target RPM) at scaled height 2.0
        target_height = 2.0
        setpoint_line = Line(
            plot_origin + UP * target_height, 
            plot_origin + RIGHT * 8 + UP * target_height, 
            color=GRAY, 
            stroke_width=2
        )
        
        # Manually generating the curve points (Second order underdamped system)
        # y(t) = 1 - e^(-0.4t) * cos(2t)
        curve_pts = []
        resolution = 100
        for i in range(resolution + 1):
            t_val = (i / resolution) * 8.0
            # Use math module to avoid numpy identifier issues
            y_val = 1.0 - math.exp(-0.4 * t_val) * math.cos(2.0 * t_val)
            # 1 unit of t = 1 unit on x, 1 unit of y = target_height on y
            pt = plot_origin + RIGHT * t_val + UP * (y_val * target_height)
            curve_pts.append(pt)
        
        motor_curve = VMobject(color=RED)
        motor_curve.set_points_as_corners(curve_pts)

        self.play(Create(axis_h), Create(axis_v), Write(label_time), Write(label_rpm))
        self.play(Create(setpoint_line))
        self.play(Create(motor_curve), run_time=3)

        # 3. Annotations
        # Rise Time: Duration until first reaching the setpoint (approx t = 0.785)
        # We use Brace(Line(...)) instead of BraceBetweenPoints

        rt_brace = Brace(rt_segment, DOWN, buff=0.1)
        rt_label = Text("Rise Time", font_size=20).next_to(rt_brace, DOWN)
        
        # Steady State Error: Gap at the end of the observed time
        # End point of curve vs setpoint point at t=8
        sse_segment = Line(curve_pts[-1], plot_origin + RIGHT * 8 + UP * target_height)
        sse_brace = Brace(sse_segment, RIGHT, buff=0.1)
        sse_label = Text("Steady State Error", font_size=16).next_to(sse_brace, RIGHT)

        self.play(Create(rt_brace), Write(rt_label))
        self.play(Create(sse_brace), Write(sse_label))
        
        self.wait(2)
