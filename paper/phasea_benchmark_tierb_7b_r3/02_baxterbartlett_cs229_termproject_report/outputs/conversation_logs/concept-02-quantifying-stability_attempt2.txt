from manim import *
import numpy as np

class Concept02QuantifyingStability(Scene):
    def construct(self):
        # 1. Manual creation of Plot Axes (since Axes is disallowed)
        origin = np.array([-6, -2, 0])
        x_axis = Line(origin, origin + RIGHT * 5, color=WHITE)
        y_axis = Line(origin, origin + UP * 4, color=WHITE)
        x_axis_label = Text("Time", font_size=18).next_to(x_axis, DOWN)
        y_axis_label = Text("RPM (relative)", font_size=18).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        # 2. Manual creation of Velocity Plot
        np.random.seed(42)
        num_points = 50
        # Time values from -6 to -1 (spanning the x_axis)
        x_vals = np.linspace(-6, -1, num_points)
        # Random data fluctuating around center y=0
        y_vals = np.array([0.5 + np.random.uniform(-0.5, 0.5) for _ in range(num_points)])
        # Ensure 30 RPM (center) is common
        y_vals[15:20] = 0.5
        y_vals[40:45] = 0.5
        
        line_points = [np.array([x, y, 0]) for x, y in zip(x_vals, y_vals)]
        velocity_line = VMobject(color=YELLOW)
        velocity_line.set_points_as_corners(line_points)

        # 3. Highlight Last 20% of the graph
        # X range is 5 units, 20% is 1.0 units.
        highlight_box = Rectangle(
            width=1.0, 
            height=4.0, 
            fill_color=BLUE, 
            fill_opacity=0.3, 
            stroke_width=0
        ).move_to(np.array([-1.5, 0, 0])) # Centered at the last 1.0 unit segment

        # 4. Manual creation of Histogram (since BarChart is disallowed)
        hist_origin = np.array([2, -2, 0])
        counts = [0.4, 0.8, 2.8, 1.2, 0.6] # Heights for 28, 29, 30, 31, 32
        bin_names = ["28", "29", "30", "31", "32"]
        bars = VGroup()
        labels = VGroup()
        
        for i, h in enumerate(counts):
            bar = Rectangle(
                width=0.6, 
                height=h, 
                fill_opacity=0.8, 
                stroke_width=1
            ).move_to(hist_origin + RIGHT * i * 0.8 + UP * h / 2)
            
            if bin_names[i] == "30":
                bar.set_color(GREEN)
            else:
                bar.set_color(WHITE)
                
            label = Text(bin_names[i], font_size=16).next_to(bar, DOWN, buff=0.1)
            bars.add(bar)
            labels.add(label)
            
        hist_title = Text("Histogram", font_size=22).next_to(bars, UP, buff=0.5)

        # 5. Formula
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=32
        ).to_edge(UP, buff=0.4)

        # Execution
        self.add(x_axis, y_axis, x_axis_label, y_axis_label)
        self.play(Create(velocity_line), run_time=2)
        self.play(FadeIn(highlight_box))
        self.wait(1)
        
        self.play(
            Create(bars),
            Write(labels),
            Write(hist_title)
        )
        self.wait(1)
        
        self.play(Write(formula))
        self.play(formula.animate.set_color_by_tex("Stability", YELLOW))
        
        self.wait(2)
