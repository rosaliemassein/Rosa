from manim import *
import numpy as np

class StabilityAnimation(Scene):
    def construct(self):
        # Coordinates and scale (manual axes implementation to avoid blocked classes)
        # x_range: -5 to 1, y_range: -2 to 2
        origin_x, origin_y = -5, -2
        
        # 1. Create manual axes for the Velocity Plot
        x_axis_line = Line([origin_x, origin_y, 0], [origin_x + 6, origin_y, 0], color=WHITE)
        y_axis_line = Line([origin_x, origin_y, 0], [origin_x, origin_y + 4, 0], color=WHITE)
        time_label = Text("Time", font_size=24).next_to(x_axis_line, DOWN)
        rpm_label = Text("RPM", font_size=24).next_to(y_axis_line, LEFT).rotate(90 * DEGREES)
        
        # 2. Plotting the motor velocity
        # The motor fluctuates around 30 RPM (mode). 
        # In our 4-unit high graph (0-60 RPM), 30 RPM is exactly 2 units from the bottom.
        plot_points = []
        point_count = 101
        for i in range(point_count):
            p_ratio = i / (point_count - 1)
            coord_x = origin_x + p_ratio * 6
            # Sine waves to simulate realistic motor fluctuations around the 30 RPM baseline
            fluctuation = 0.3 * np.sin(p_ratio * 15) + 0.2 * np.cos(p_ratio * 40)
            coord_y = origin_y + 2.0 + fluctuation
            plot_points.append([coord_x, coord_y, 0])
        
        velocity_curve = VMobject(color=YELLOW)
        velocity_curve.set_points_as_corners(plot_points)
        
        # 3. Highlight the last 20% of the timeline
        # Total x-width is 6 units, 20% is 1.2 units.
        highlight_box = Rectangle(
            width=1.2, 
            height=4.0, 
            fill_color=BLUE, 
            fill_opacity=0.3, 
            stroke_width=0
        )
        # Position at the end of the x-axis
        highlight_box.move_to([origin_x + 5.4, origin_y + 2.0, 0])
        highlight_text = Text("Final 20%", font_size=16, color=BLUE).next_to(highlight_box, UP, buff=0.1)
        
        # 4. Manual Histogram on the right side
        # Mode is at 30 RPM (the middle bar)
        hist_base_x = 2
        bar_heights = [0.6, 1.1, 3.4, 0.9, 0.5]
        histogram_bars = VGroup()
        for i in range(len(bar_heights)):
            h = bar_heights[i]
            # The mode (middle bar) is colored differently
            bar_color = RED if i == 2 else BLUE
            bar_rect = Rectangle(
                width=0.4, 
                height=h, 
                fill_color=bar_color, 
                fill_opacity=0.8, 
                stroke_width=1
            )
            # Align the bottom of the bar with the axes
            bar_rect.move_to([hist_base_x + i * 0.6, origin_y + h / 2, 0])
            histogram_bars.add(bar_rect)
            
        histogram_title = Text("RPM Frequency", font_size=24).next_to(histogram_bars, UP)
        
        # 5. Stability Formula definition
        # Stability = counts at mode / counts not at mode
        stability_formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=32
        ).to_edge(UP, buff=0.7)
        
        # --- Animation Sequence ---
        
        # Draw the graph area
        self.add(x_axis_line, y_axis_line, time_label, rpm_label)
        self.play(Create(velocity_curve), run_time=3)
        self.wait(0.5)
        
        # Highlight the steady-state period
        self.play(FadeIn(highlight_box), Write(highlight_text))
        self.wait(1)
        
        # Transition to the histogram analysis
        self.play(Create(histogram_bars), Write(histogram_title))
        self.wait(1)
        
        # Define the feature mathematically
        self.play(Write(stability_formula))
        self.wait(3)
