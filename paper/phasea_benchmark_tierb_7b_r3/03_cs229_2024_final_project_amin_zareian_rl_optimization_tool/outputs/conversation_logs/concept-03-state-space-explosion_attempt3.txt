from manim import *
import math

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Introduction and Formula
        voice_text = Text(
            "Standard Q-learning faces state space explosion.",
            font_size=24
        ).to_edge(UP)
        
        formula = MathTex(
            r"\text{Total States} = 2^{(N \times N)}", 
            color=YELLOW
        ).next_to(voice_text, DOWN)
        
        self.play(Write(voice_text))
        self.play(FadeIn(formula))
        self.wait(1)

        # 2. Manual Coordinate System (Replacement for disallowed 'Axes')
        # We place the origin in the lower right quadrant
        origin_point = [1.5, -2.5, 0]
        x_axis = Line(origin_point, [6, -2.5, 0], color=WHITE)
        y_axis = Line(origin_point, [1.5, 2.5, 0], color=WHITE)
        
        x_label = Text("Size (N)", font_size=18).next_to(x_axis, DOWN)
        y_label = Text("States (Log Scale)", font_size=18).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        axes_elements = VGroup(x_axis, y_axis, x_label, y_label)
        self.play(Create(axes_elements))

        # 3. Grid Visual (Top Left)
        grid_pos = [-4, 1.5, 0]
        def get_binary_grid(n_val):
            g = VGroup()
            for row_idx in range(n_val):
                for col_idx in range(n_val):
                    sq = Square(side_length=0.3, stroke_width=1)
                    # Checkerboard-like colors for visualization
                    sq.set_fill(RED if (row_idx + col_idx) % 2 == 0 else BLUE, opacity=0.8)
                    g.add(sq)
            return g.arrange_in_grid(rows=n_val, cols=n_val, buff=0.05).move_to(grid_pos)

        current_grid = get_binary_grid(2)
        grid_label = MathTex("N=2", font_size=24).next_to(current_grid, DOWN)
        
        # 4. Table Representation (Bottom Left)
        table_box = Rectangle(width=2.5, height=3.5).to_corner(DL, buff=0.5).shift(RIGHT * 0.5)
        table_label = Text("Q-Table Rows", font_size=22).next_to(table_box, UP)
        
        # Initial rows
        rows = VGroup(*[
            Line(LEFT, RIGHT).set_width(2.2).set_stroke(width=1, opacity=0.4) 
            for _ in range(12)
        ]).arrange(DOWN, buff=0.15).move_to(table_box)

        self.play(
            FadeIn(current_grid), 
            Write(grid_label),
            Create(table_box), 
            Write(table_label),
            Create(rows)
        )

        # 5. The Graph Curve (Manual Construction)
        # We plot 2^(N^2). On a log scale, this is N^2 * log10(2)
        graph_curve = VMobject(color=YELLOW)
        points = []
        for i in range(101):
            n_step = 2 + (i / 100) * 4 # N from 2 to 6
            # X mapping: N=2 -> 1.5, N=6 -> 6
            x_coord = 1.5 + (n_step - 2) * (4.5 / 4)
            # Y mapping: N=2 -> -2.5, N=6 -> 2.5 (representing ~10^11 states)
            # Using log value scaled: (n_step^2 * log10(2)) / 11 * 5 - 2.5
            y_val_log = (n_step * n_step * 0.301) 
            y_coord = -2.5 + (y_val_log / 11) * 5
            points.append([x_coord, y_coord, 0])
        
        graph_curve.set_points_as_corners(points)

        # 6. Growth and Explosion Animation
        # We iterate to show the complexity growing
        for n_size in [3, 4, 5]:
            new_grid = get_binary_grid(n_size)
            new_label = MathTex(f"N={n_size}", font_size=24).next_to(new_grid, DOWN)
            
            # Create more dense rows to simulate multiplication
            row_count = n_size * 20
            new_rows = VGroup(*[
                Line(LEFT, RIGHT).set_width(2.2).set_stroke(width=0.5, opacity=0.3) 
                for _ in range(row_count)
            ]).arrange(DOWN, buff=3.0/row_count).move_to(table_box)
            
            self.play(
                Transform(current_grid, new_grid),
                Transform(grid_label, new_label),
                Transform(rows, new_rows),
                run_time=0.7
            )

        # Final Blast: Everything turns into a blur and graph shoots up
        blur_rect = Rectangle(
            width=2.5, height=3.5, 
            fill_color=WHITE, fill_opacity=0.6, 
            stroke_width=0
        ).move_to(table_box)
        
        final_text = Text("33 Million+", color=RED, font_size=32).move_to(table_box)
        
        # Fast scrolling animation for the "blur" effect
        scrolling_rows = VGroup(*[
            Line(LEFT, RIGHT).set_width(2.2).set_stroke(width=0.2, opacity=0.2) 
            for _ in range(100)
        ]).arrange(DOWN, buff=0.02).move_to(table_box)

        self.play(
            Create(graph_curve),
            FadeIn(blur_rect),
            Write(final_text),
            scrolling_rows.animate.shift(UP * 5),
            run_time=3,
            rate_func=lambda t: t # Linear rate function replacement
        )
        
        self.play(graph_curve.animate.set_color(RED).set_stroke(width=5))
        self.wait(2)
