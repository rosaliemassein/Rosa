from manim import *

class RewardDynamicsScene(Scene):
    def construct(self):
        # 1. Setup Grid of "Pixels" (3x3)
        grid = VGroup(*[
            Square(side_length=1, fill_opacity=1, fill_color=RED, stroke_color=WHITE, stroke_width=2)
            for _ in range(9)
        ]).arrange_in_grid(rows=3, cols=3, buff=0).move_to(LEFT * 3)

        # Labels/Formula
        formula = MathTex(r"R_t = Border_{t} - Border_{t-1}").to_edge(UP)
        self.play(Write(formula))

        # Setup initial state: make top neighbor blue (index 1)
        # This creates 1 boundary (Red center vs Blue neighbor)
        grid[1].set_fill(BLUE)
        self.add(grid)

        # 2. Highlight focus area
        center = grid[4]
        # Neighbors: indices 1, 3, 5, 7
        neighbors = VGroup(grid[1], grid[3], grid[5], grid[7])
        
        self.play(center.animate.set_stroke(YELLOW, width=6))
        self.play(Indicate(neighbors, color=YELLOW))
        self.wait(0.5)

        # 3. Flip Animation
        # Before: 1 border (Top)
        # Flip center to Blue: 
        #   Lost border: Top (Blue-Blue)
        #   New borders: Left, Right, Bottom (Blue-Red)
        # Net: -1 + 3 = +2
        
        # Manually create "Cross" (X) for lost border
        top_pos = center.get_edge_center(UP)
        cross = VGroup(
            Line(top_pos + (UL+UR)*0.1, top_pos + (DL+DR)*0.1, color=RED, stroke_width=6),
            Line(top_pos + (UL+DL)*0.1, top_pos + (UR+DR)*0.1, color=RED, stroke_width=6)
        ).scale(0.8).move_to(top_pos)

        # New borders (Green lines)
        left_border = Line(center.get_corner(UL), center.get_corner(DL), color=GREEN, stroke_width=8)
        right_border = Line(center.get_corner(UR), center.get_corner(DR), color=GREEN, stroke_width=8)
        bottom_border = Line(center.get_corner(DL), center.get_corner(DR), color=GREEN, stroke_width=8)

        self.play(center.animate.set_fill(BLUE))
        self.play(Create(cross))
        self.play(Create(left_border), Create(right_border), Create(bottom_border))

        reward_text = Text("+2", color=GREEN).next_to(center, UR, buff=0.1)
        self.play(Write(reward_text))

        # 4. Manual Bar Chart (Using Axes and Rectangle to avoid undefined BarChart)
        axes = Axes(
            x_range=[0, 2, 1],
            y_range=[0, 5, 1],
            x_length=2.5,
            y_length=3.5,
            axis_config={"include_tip": False},
            tips=False
        ).to_edge(RIGHT, buff=1.5)
        
        chart_title = Text("Cumulative Reward", font_size=24).next_to(axes, DOWN)
        
        # Initial bar at 0 height
        bar = Rectangle(
            width=0.8, 
            height=0.01, 
            fill_opacity=1, 
            fill_color=GREEN, 
            stroke_width=0
        ).move_to(axes.c2p(1, 0), aligned_edge=DOWN)

        self.play(Create(axes), Write(chart_title), Create(bar))
        
        # Animate bar growing to height 2 (reward value)
        # In axes.c2p(1, 2), height corresponds to the y-axis scaling
        target_height = axes.c2p(0, 2)[1] - axes.c2p(0, 0)[1]
        self.play(
            bar.animate.stretch_to_fit_height(target_height).move_to(axes.c2p(1, 0), aligned_edge=DOWN)
        )
        
        self.wait(2)
