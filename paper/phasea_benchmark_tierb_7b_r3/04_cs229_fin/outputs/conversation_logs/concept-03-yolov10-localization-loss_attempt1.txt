from manim import *

class YOLOv10LocalizationLoss(Scene):
    def construct(self):
        # 1. Background Placeholder (Avoid potential ImageMobject issues in sandbox)
        background = Rectangle(width=12, height=7, color=GRAY, fill_opacity=0.1)
        bg_label = Text("Image: Trash Pile Grid", font_size=24).to_edge(DOWN)
        self.add(background, bg_label)

        # 2. Manual Grid Construction (Avoiding NumberPlane)
        grid = VGroup()
        for x in range(-3, 4):
            grid.add(Line(start=[x, -3.5, 0], end=[x, 3.5, 0], stroke_width=1, stroke_opacity=0.5))
        for y in range(-3, 4):
            grid.add(Line(start=[-3.5, y, 0], end=[3.5, y, 0], stroke_width=1, stroke_opacity=0.5))
        
        # Using FadeIn for the group to be safer than Create
        self.play(FadeIn(grid))

        # 3. Highlight a specific cell (Center at x=1, y=0)
        target_cell = Rectangle(width=1, height=1, color=YELLOW, stroke_width=4).move_to([1, 0, 0])
        self.play(Create(target_cell))

        # 4. Boxes
        # Ground Truth: Red (using DashedVMobject for compatibility)
        gt_rect = Rectangle(width=1.6, height=1.2, color=RED).move_to([1, 0, 0])
        gt_box = DashedVMobject(gt_rect, num_dashes=30)
        
        # Predicted Box: Blue
        pred_box = Rectangle(width=2.2, height=1.8, color=BLUE, fill_opacity=0.3).move_to([1.4, 0.4, 0])
        
        self.play(Create(gt_box), Create(pred_box))

        center_pred = pred_box.get_center()

                        buff=0, color=GREEN, stroke_width=4)

                             buff=0, color=YELLOW, stroke_width=3)

        self.play(Create(w_arrow))

        part3 = MathTex(r" + \sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}", font_size=32)
