from manim import *
import numpy as np

class SoftmaxClassification(Scene):
    def construct(self):
        # 1. Plastic Bottle Input Image (Placeholder)
        bottle_rect = Rectangle(height=2.5, width=1.8, color=BLUE, fill_opacity=0.2)
        bottle_label = Text("Plastic Bottle\n(Input)", font_size=20).move_to(bottle_rect)
        bottle_img = VGroup(bottle_rect, bottle_label).to_edge(LEFT, buff=0.8)
        
        # 2. CNN Layers
        cnn_layers = VGroup(*[
            Rectangle(height=1.5, width=0.4, fill_opacity=0.5, color=GREY_A) 
            for _ in range(3)
        ]).arrange(RIGHT, buff=0.2).next_to(bottle_img, RIGHT, buff=1)
        
        arrow_to_cnn = Arrow(bottle_img.get_right(), cnn_layers.get_left(), buff=0.1)
        
        # 3. Column Vector of 20 Raw Values
        # Using MathTex to construct a vector manually to avoid Matrix identifier issues
        x_vector = MathTex(
            r"\begin{bmatrix} x_{1} \\ x_{2} \\ \vdots \\ x_{20} \end{bmatrix}",
            font_size=36
        ).next_to(cnn_layers, RIGHT, buff=1)
        
        arrow_to_vector = Arrow(cnn_layers.get_right(), x_vector.get_left(), buff=0.1)

        # 4. Softmax Formula
        softmax_formula = MathTex(
            r"\text{softmax}(x_{i})=\frac{e^{x_{i}}}{\sum_{j=1}^{20}e^{x_{j}}}",
            font_size=42
        ).to_edge(UP, buff=0.5)

        # 5. Probability Bar Chart (Manual Construction)
        # Using Axes and Rectangles to ensure compatibility
        axes = Axes(
            x_range=[0, 20, 1],
            y_range=[0, 1, 0.5],
            x_length=5,
            y_length=3,
            axis_config={"include_tip": False},
            tips=False
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)
        
        probs = [0.03] * 20
        winner_index = 4 
        probs[winner_index] = 0.8
        
        bars = VGroup()
        for i, p in enumerate(probs):
            bar = Rectangle(
                width=axes.x_axis.get_unit_size() * 0.8,
                height=p * axes.y_axis.get_unit_size(),
                fill_opacity=0.8,
                color=BLUE
            )
            bar.move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN)
            bars.add(bar)
            
        chart_group = VGroup(axes, bars)
        chart_title = Text("Probabilities", font_size=20).next_to(axes, UP)

        # --- Animations ---
        self.play(FadeIn(bottle_img))
        self.play(GrowArrow(arrow_to_cnn), Create(cnn_layers))
        self.wait(0.5)
        
        self.play(GrowArrow(arrow_to_vector), Write(x_vector))
        self.wait(1)
        
        self.play(Write(softmax_formula))
        self.wait(1)
        
        # Transform vector into the bars of the chart
        self.play(
            Create(axes),
            FadeIn(chart_title),
            ReplacementTransform(x_vector.copy(), bars),
            run_time=2
        )
        self.wait(1)
        
        # 6. Indicate highest bar and label it
        highest_bar = bars[winner_index]
        prediction_label = Text("Plastic Bottle", color=YELLOW, font_size=24).next_to(highest_bar, UP, buff=0.2)
        
        self.play(Indicate(highest_bar, color=YELLOW), run_time=2)
        self.play(Write(prediction_label))
        
        self.wait(3)
