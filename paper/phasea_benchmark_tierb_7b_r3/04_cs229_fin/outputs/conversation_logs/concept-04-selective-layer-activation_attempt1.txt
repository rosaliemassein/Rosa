from manim import *

class SelectiveLayerActivation(Scene):
    def construct(self):
        # 1. Create a chain of 24 'Square' objects representing layers 0 to 23.
        # We reduce side_length to ensure 24 squares fit on the screen.
        n_layers = 24
        square_size = 0.4
        buff = 0.05
        layers = VGroup(*[Square(side_length=square_size, fill_opacity=0.5) for _ in range(n_layers)])
        layers.arrange(RIGHT, buff=buff)
        layers.center().to_edge(UP, buff=2)

        # 2. Color layers based on the strategy
        # Color layers 0-6 gray (Frozen Backbone)
        for i in range(0, 7):
            layers[i].set_color(GRAY)
            layers[i].set_fill(GRAY, opacity=0.3)

        # Color layers 7-10 bright blue (Active Semantics)
        # We use a stroke width increase to simulate a 'glow'
        for i in range(7, 11):
            layers[i].set_color(BLUE)
            layers[i].set_fill(BLUE, opacity=0.7)
            layers[i].set_stroke(width=6)

        # Color layers 11-22 gray
        for i in range(11, 23):
            layers[i].set_color(GRAY)
            layers[i].set_fill(GRAY, opacity=0.3)

        # Color layer 23 bright orange (Active Head)
        layers[23].set_color(ORANGE)
        layers[23].set_fill(ORANGE, opacity=0.7)
        layers[23].set_stroke(width=6)

        # 3. Create Labels
        # Using simple Text and positioning them relative to specific layer groups
        frozen_label = Text("Frozen Backbone", font_size=18, color=GRAY)
        frozen_label.next_to(layers[0:7], DOWN, buff=0.5)
        
        active_sem_label = Text("Active Semantics", font_size=18, color=BLUE)
        active_sem_label.next_to(layers[7:11], DOWN, buff=1.0)
        
        active_head_label = Text("Active Head", font_size=18, color=ORANGE)
        active_head_label.next_to(layers[23], DOWN, buff=0.5)

        # Simple lines as separators/indicators instead of Brace
        line_frozen = Line(layers[0].get_bottom() + DOWN*0.1, layers[6].get_bottom() + DOWN*0.1, color=GRAY)
        line_sem = Line(layers[7].get_bottom() + DOWN*0.1, layers[10].get_bottom() + DOWN*0.1, color=BLUE)

        # 4. Display Scene
        self.add(layers)
        self.play(
            Create(line_frozen),
            Create(line_sem),
            Write(frozen_label),
            Write(active_sem_label),
            Write(active_head_label),
            run_time=2
        )
        self.wait(1)

        # 5. Animation of pulses moving through blue and orange squares
        # We use a loop to create a succession of pulses (scale and stroke intensity)
        pulse_targets = [7, 8, 9, 10, 23]
        
        pulse_anims = []
        for idx in pulse_targets:
            # Each pulse is a small scale up and back down
            pulse_anims.append(layers[idx].animate(run_time=0.25).scale(1.3).set_fill(opacity=1))
            pulse_anims.append(layers[idx].animate(run_time=0.25).scale(1/1.3).set_fill(opacity=0.7))

        # Run the pulse sequence
        self.play(Succession(*pulse_anims))
        
        # Second pass pulse
        self.play(
            Succession(*[
                layers[i].animate(run_time=0.15, rate_func=there_and_back).scale(1.4) 
                for i in pulse_targets
            ])
        )

        self.wait(2)
