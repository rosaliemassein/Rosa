from manim import *
import numpy as np

class YOLOv10LocalizationLoss(Scene):
    def construct(self):
        # Using hex colors to avoid 'undefined' color errors
        COLOR_GREY = "#888888"
        COLOR_YELLOW = "#FFFF00"
        COLOR_RED = "#FF0000"
        COLOR_BLUE = "#0000FF"
        COLOR_GREEN = "#00FF00"
        COLOR_PURPLE = "#800080"

        # 1. Background Placeholder for Image
        bg_rect = Rectangle(width=6, height=4, color=COLOR_GREY, fill_opacity=0.2).to_edge(UP, buff=0.5)
        
        # In case ImageMobject is restricted, we'll try it but keep the rect as backup
        try:
            image_obj = ImageMobject("img-11.jpeg")
            image_obj.set_height(4).move_to(bg_rect)
            self.add(image_obj)
        except:
            self.add(bg_rect)

        # 2. Manual Grid (avoiding NumberPlane)
        grid = VGroup()
        grid_width, grid_height = 4.0, 4.0
        step = grid_width / 7
        top_left = bg_rect.get_center() + np.array([-grid_width/2, grid_height/2, 0])
        
        for i in range(8):
            # Vertical lines
            grid.add(Line(top_left + np.array([i * step, 0, 0]), top_left + np.array([i * step, -grid_height, 0])))
            # Horizontal lines
            grid.add(Line(top_left + np.array([0, -i * step, 0]), top_left + np.array([grid_width, -i * step, 0])))
        grid.set_stroke(WHITE, opacity=0.3, width=1)
        self.add(grid)

        # 3. Highlight Target Cell (e.g., cell at 4,3)
        cell_center = top_left + np.array([4 * step + step/2, -3 * step - step/2, 0])
        target_cell = Rectangle(
            width=step, height=step, 
            fill_color=COLOR_YELLOW, fill_opacity=0.3, 
            stroke_width=2, color=COLOR_YELLOW
        ).move_to(cell_center)
        self.add(target_cell)

        # 4. Bounding Boxes
        # Predicted box (Red)
        pred_box = Rectangle(width=1.0, height=1.3, color=COLOR_RED).move_to(cell_center + np.array([0.2, -0.2, 0]))
        
        # Ground Truth Box (Dashed Blue - constructed manually to avoid DashedVMobject)
        gt_w, gt_h = 1.2, 1.5
        gt_center = cell_center + np.array([0.4, 0.1, 0])
        gt_box = VGroup(
            DashedLine(gt_center + [-gt_w/2, gt_h/2, 0], gt_center + [gt_w/2, gt_h/2, 0], color=COLOR_BLUE),
            DashedLine(gt_center + [gt_w/2, gt_h/2, 0], gt_center + [gt_w/2, -gt_h/2, 0], color=COLOR_BLUE),
            DashedLine(gt_center + [gt_w/2, -gt_h/2, 0], gt_center + [-gt_w/2, -gt_h/2, 0], color=COLOR_BLUE),
            DashedLine(gt_center + [-gt_w/2, -gt_h/2, 0], gt_center + [-gt_w/2, gt_h/2, 0], color=COLOR_BLUE)
        )

        # 5. Offset Arrows
        arrow_x = Arrow(pred_box.get_center(), [gt_center[0], pred_box.get_center()[1], 0], color=COLOR_GREEN, buff=0, tip_length=0.1)
        arrow_y = Arrow([gt_center[0], pred_box.get_center()[1], 0], gt_center, color=COLOR_PURPLE, buff=0, tip_length=0.1)

        # 6. Formula
        formula = MathTex(
            r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}", 
            r"\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right)",
            r" + ",
            r"\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}",
            r"\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)"
        ).scale(0.55).to_edge(DOWN, buff=0.3)

        # Animations
        self.play(Create(pred_box), Create(gt_box))
        self.play(Create(arrow_x), Create(arrow_y))
        self.wait(0.5)
        
        self.play(Write(formula))
        self.play(Indicate(formula[1], color=COLOR_GREEN))
        self.play(Indicate(formula[4], color=COLOR_PURPLE))
        self.wait(1)

        # Mock training align
        self.play(
            pred_box.animate.move_to(gt_center).set_width(gt_w).set_height(gt_h),
            FadeOut(arrow_x),
            FadeOut(arrow_y),
            run_time=2
        )
        self.wait(2)
