from manim import *

class SoftmaxExplanation(Scene):
    def construct(self):
        # 1. Left side: Image of a plastic bottle
        # Use ImageMobject instead of Image (which is a PIL class)
        try:
            input_image = ImageMobject("img-12.jpeg").scale(0.6)
        except:
            # Fallback if image is missing in the environment
            input_image = Rectangle(height=2, width=1.6, color=BLUE)
            input_image.add(Text("Plastic Bottle", font_size=16))
        
        input_image.to_edge(LEFT, buff=0.5)
        self.play(FadeIn(input_image))

        # 2. Vertical rectangles representing CNN layers
        cnn_layers = VGroup(*[
            Rectangle(width=0.2, height=1.5, fill_opacity=0.3, color=WHITE) 
            for _ in range(5)
        ]).arrange(RIGHT, buff=0.2)
        cnn_layers.next_to(input_image, RIGHT, buff=0.8)
        
        flow_arrow = Arrow(input_image.get_right(), cnn_layers.get_left(), buff=0.1)
        self.play(Create(cnn_layers), GrowArrow(flow_arrow))

        # 3. Column vector of 20 raw values (x_i)
        # Manually constructing a vector-like structure since Matrix was undefined
        vector_rect = Rectangle(width=0.7, height=4).next_to(cnn_layers, RIGHT, buff=1.2)
        v_top = MathTex("x_1", font_size=24).move_to(vector_rect.get_top() + DOWN * 0.4)
        v_mid = MathTex("\\vdots", font_size=24).move_to(vector_rect.get_center())
        v_bot = MathTex("x_{20}", font_size=24).move_to(vector_rect.get_bottom() + UP * 0.4)
        vector_elements = VGroup(v_top, v_mid, v_bot)
        vector_label = MathTex("x_i").next_to(vector_rect, UP)
        
        vector_group = VGroup(vector_rect, vector_elements, vector_label)
        flow_to_vector = Arrow(cnn_layers.get_right(), vector_rect.get_left(), buff=0.1)
        
        self.play(Create(vector_group), GrowArrow(flow_to_vector))

        # 4. Softmax formula appearing at the top
        softmax_formula = MathTex(r"softmax(x_{i})=\frac{e^{x_{i}}}{\sum_{j=1}^{n}e^{x_{j}}}")
        softmax_formula.to_edge(UP, buff=0.5).shift(RIGHT * 2)
        self.play(Write(softmax_formula))

        # 5. Transform raw values into a BarChart
        # Manually building the chart using Axes and Rectangles since BarChart was undefined
        axes = Axes(
            x_range=[0, 20, 1],
            y_range=[0, 1, 0.5],
            x_length=6,
            y_length=3,
            axis_config={"include_tip": False, "include_numbers": False}
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)

        # Create 20 bars, one specifically high for 'Plastic Bottle'
        bars = VGroup()
        for i in range(20):
            # probability value
            h_val = 0.85 if i == 14 else 0.05 
            bar = Rectangle(
                width=axes.x_length / 25,
                height=h_val * (axes.y_length / 1.0),
                fill_opacity=0.8,
                color=BLUE,
                stroke_width=1
            )
            bar.move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN)
            bars.add(bar)

        # Transition
        self.play(
            FadeOut(cnn_layers),
            FadeOut(flow_arrow),
            FadeOut(flow_to_vector),
            ReplacementTransform(vector_group, axes),
            FadeIn(bars)
        )

        # 6. Indicate the highest bar and label it
        target_bar = bars[14]
        self.play(Indicate(target_bar, color=YELLOW))
        
        label = Text("Plastic Bottle", font_size=24, color=YELLOW).next_to(target_bar, UP, buff=0.2)
        self.play(Write(label))
        
        self.wait(2)
