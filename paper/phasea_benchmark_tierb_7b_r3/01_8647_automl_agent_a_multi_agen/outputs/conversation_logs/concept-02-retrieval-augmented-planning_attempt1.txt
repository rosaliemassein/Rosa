from manim import *

class RetrievalAugmentedPlanning(Scene):
    def construct(self):
        # 1. Formula at the top
        formula = MathTex(r"\mathbf{P} = \mathcal{A}_{mgr}(\mathrm{RAP}(R))").to_edge(UP)

        # 2. Central Agent Manager Icon
        am_box = RoundedRectangle(corner_radius=0.1, height=1.0, width=2.5, color=BLUE)
        am_label = Text("Agent Manager", font_size=20)
        agent_manager = VGroup(am_box, am_label).center()

        # 3. Knowledge Base Icon (Custom Cloud using Circles)
        c1 = Circle(radius=0.4, fill_opacity=1, color=GRAY_B).shift(LEFT * 0.3)
        c2 = Circle(radius=0.5, fill_opacity=1, color=GRAY_B).shift(RIGHT * 0.3)
        c3 = Circle(radius=0.4, fill_opacity=1, color=GRAY_B).shift(UP * 0.3)
        kb_icon = VGroup(c1, c2, c3).set_stroke(WHITE, width=1)
        kb_label = Text("Knowledge Base", font_size=14, color=WHITE).move_to(kb_icon.get_center())
        knowledge_base = VGroup(kb_icon, kb_label).to_edge(LEFT, buff=1.0)

        # 4. Plan Paths (Right side)
        # Plan A: Transformer Strategy
        plan_a_box = Rectangle(height=0.7, width=2.2, color=YELLOW)
        plan_a_text = Text("Plan A: Transformer", font_size=16)
        plan_a = VGroup(plan_a_box, plan_a_text).shift(RIGHT * 4 + UP * 1.8)

        # Plan B: CNN Strategy
        plan_b_box = Rectangle(height=0.7, width=2.2, color=GREEN)
        plan_b_text = Text("Plan B: CNN", font_size=16)
        plan_b = VGroup(plan_b_box, plan_b_text).shift(RIGHT * 4)

        # Plan C: Other Strategy
        plan_c_box = Rectangle(height=0.7, width=2.2, color=RED)
        plan_c_text = Text("Plan C: Other", font_size=16)
        plan_c = VGroup(plan_c_box, plan_c_text).shift(RIGHT * 4 + DOWN * 1.8)

        # 5. Connectors
        arrow_kb = Arrow(knowledge_base.get_right(), agent_manager.get_left(), color=WHITE)
        label_r = MathTex("R", font_size=32).next_to(arrow_kb, UP)

        arrow_a = Arrow(agent_manager.get_right(), plan_a.get_left(), buff=0.1)
        arrow_b = Arrow(agent_manager.get_right(), plan_b.get_left(), buff=0.1)
        arrow_c = Arrow(agent_manager.get_right(), plan_c.get_left(), buff=0.1)

        # --- ANIMATION SEQUENCE ---

        # Show Initial Elements
        self.play(Write(formula))
        self.play(FadeIn(knowledge_base), FadeIn(agent_manager))
        self.wait(0.5)

        # Animate Knowledge Retrieval
        self.play(GrowArrow(arrow_kb), Write(label_r))
        
        # Stream of data particles moving into Manager
        particles = VGroup(*[Dot(color=YELLOW, radius=0.06) for _ in range(5)])
        self.play(
            LaggedStart(
                *[p.animate.move_to(agent_manager.get_left()).set_opacity(0) for p in particles],
                lag_ratio=0.2
            ),
            knowledge_base.animate.set_color(YELLOW),
            run_time=1.5
        )
        self.play(knowledge_base.animate.set_color(GRAY_B))

        # Branching out into parallel plans
        # We use ReplacementTransform on copies of the manager to simulate the "branching" or "unfolding"
        self.play(
            LaggedStart(
                Create(arrow_a),
                Create(arrow_b),
                Create(arrow_c),
                lag_ratio=0.1
            )
        )

        self.play(
            ReplacementTransform(agent_manager.copy(), plan_a),
            ReplacementTransform(agent_manager.copy(), plan_b),
            ReplacementTransform(agent_manager.copy(), plan_c),
            run_time=2
        )

        # Highlight Parallelism
        self.play(
            plan_a.animate.scale(1.1),
            plan_b.animate.scale(1.1),
            plan_c.animate.scale(1.1),
            rate_func=there_and_back
        )

        self.wait(3)
