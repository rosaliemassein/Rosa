from manim import *
import numpy as np

class ObjectCondensation(Scene):
    def construct(self):
        # 1. Formula Display
        # q_i = arctanh(beta_i) + q_min
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula.to_edge(UP)
        self.play(Write(formula))

        # 2. Setup 2D Plane of Dots
        np.random.seed(123)
        dots = VGroup()
        num_dots = 40
        
        # Define positions and simulated beta values
        positions = [
            np.array([np.random.uniform(-6, 6), np.random.uniform(-3, 2), 0])
            for _ in range(num_dots)
        ]
        betas = [np.random.uniform(0.1, 0.99) for _ in range(num_dots)]
        
        # Force a few "condensation points" with very high beta
        betas[0], betas[1], betas[2] = 0.98, 0.95, 0.92
        
        for i in range(num_dots):
            # Start dots as small and purple
            d = Dot(point=positions[i], color=PURPLE, radius=0.06)
            d.beta = betas[i]
            dots.add(d)
            
        self.play(FadeIn(dots, lag_ratio=0.05))
        self.wait(0.5)
        
        # 3. Mechanism: Beta values grow, changing dot size and color
        # We manually map beta to colors to avoid potential interpolate_color issues
        growth_anims = []
        condensation_points = []
        for d in dots:
            if d.beta > 0.9:
                target_color = YELLOW
                condensation_points.append(d)
            elif d.beta > 0.5:
                target_color = PINK
            else:
                target_color = PURPLE_A
            
            # Animate growth and color transition
            growth_anims.append(
                d.animate.scale(1 + d.beta * 2).set_color(target_color)
            )
            
        self.play(*growth_anims, run_time=2)
        self.wait(0.5)
        
        # 4. Create Gravity Wells (Concentric Circles)
        wells = VGroup()
        for cp in condensation_points:
            well_group = VGroup()
            # Create a set of concentric circles to represent a potential well
            for r in [0.4, 0.8, 1.2, 1.6]:
                # Decreasing opacity for radial gradient effect
                op = max(0, 1.0 - r/1.8)
                c = Circle(radius=r, color=YELLOW, stroke_width=2).set_stroke(opacity=op)
                c.move_to(cp.get_center())
                well_group.add(c)
            wells.add(well_group)
            
        self.play(Create(wells, lag_ratio=0.2), run_time=2)
        
        # Show specific charge label for a condensation point
        q_label = MathTex(r"q_i \approx 2.5").scale(0.7)
        q_label.next_to(condensation_points[0], UP, buff=0.1)
        self.play(Write(q_label))
        self.wait(0.5)

        # 5. Clustering: Pull low-beta dots toward condensation points
        # Using .animate.move_to as a robust alternative to MoveAlongPath
        pull_anims = []
        for d in dots:
            if d not in condensation_points:
                # Identify the closest condensation point
                closest_cp = condensation_points[0]
                min_dist = np.linalg.norm(d.get_center() - closest_cp.get_center())
                for cp in condensation_points:
                    dist = np.linalg.norm(d.get_center() - cp.get_center())
                    if dist < min_dist:
                        min_dist = dist
                        closest_cp = cp
                
                # Pull dots that are within range
                if min_dist < 5.0:
                    pull_anims.append(d.animate.move_to(closest_cp.get_center()))
                    
        # Use a smooth rate function for the attraction movement
        self.play(*pull_anims, run_time=3, rate_func=smooth)
        self.wait(2)
