import numpy as np
from manim import *

class LearnClusterSpace(Scene):
    def construct(self):
        # Define center points for the two "spaces" to create a split-screen effect
        left_origin = LEFT * 3.5
        right_origin = RIGHT * 3.5
        
        # Create manual coordinate indicators (simple crosshairs)
        # We avoid Axes and NumberPlane because they are disallowed in this environment
        left_axes = VGroup(
            Line(left_origin + LEFT * 2, left_origin + RIGHT * 2, color=GRAY, stroke_width=2),
            Line(left_origin + UP * 2, left_origin + DOWN * 2, color=GRAY, stroke_width=2)
        )
        right_axes = VGroup(
            Line(right_origin + LEFT * 2, right_origin + RIGHT * 2, color=GRAY, stroke_width=2),
            Line(right_origin + UP * 2, right_origin + DOWN * 2, color=GRAY, stroke_width=2)
        )
        
        title_phys = Text("Physical Space", font_size=24).move_to(left_origin + UP * 2.8)
        title_learn = Text("Learned Space", font_size=24).move_to(right_origin + UP * 2.8)
        
        # Data point generation (simulating detector hits)
        # Using a fixed seed for reproducibility
        np.random.seed(42)
        n_hits = 15
        
        # Physical Space: High overlap to represent complex energy deposits
        # Cluster 1 (Red) and Cluster 2 (Blue) are centered near each other
        red_start_pos = [
            left_origin + np.array([np.random.normal(0.2, 0.5), np.random.normal(0.2, 0.5), 0]) 
            for _ in range(n_hits)
        ]
        blue_start_pos = [
            left_origin + np.array([np.random.normal(-0.2, 0.5), np.random.normal(-0.2, 0.5), 0]) 
            for _ in range(n_hits)
        ]
        
        # Learned Space: Distinct separated distributions
        # Cluster 1 moves to the left side of the right grid, Cluster 2 moves to the right
        red_end_pos = [
            right_origin + LEFT * 1.3 + np.array([np.random.normal(0, 0.3), np.random.normal(0, 0.3), 0]) 
            for _ in range(n_hits)
        ]
        blue_end_pos = [
            right_origin + RIGHT * 1.3 + np.array([np.random.normal(0, 0.3), np.random.normal(0, 0.3), 0]) 
            for _ in range(n_hits)
        ]
        
        dots_red = VGroup(*[Dot(pos, color=RED, radius=0.07) for pos in red_start_pos])
        dots_blue = VGroup(*[Dot(pos, color=BLUE, radius=0.07) for pos in blue_start_pos])
        
        # 1. Initial Setup
        self.add(left_axes, right_axes, title_phys, title_learn)
        self.play(FadeIn(dots_red), FadeIn(dots_blue))
        self.wait(1)
        
        # 2. Transformation: Move dots from physical space to learned space
        # We animate the dots individually to show the "unwarping" of the coordinates
        transform_animations = []
        for i in range(n_hits):
            transform_animations.append(dots_red[i].animate.move_to(red_end_pos[i]))
            transform_animations.append(dots_blue[i].animate.move_to(blue_end_pos[i]))
        
        self.play(*transform_animations, run_time=3, rate_func=slow_into)
        self.wait(0.5)
        
        # 3. Successful Classification Indicators
        # Using standard Circles as DashedLine/DashedVMobject are disallowed
        circle_red = Circle(radius=1.0, color=RED).move_to(right_origin + LEFT * 1.3)
        circle_blue = Circle(radius=1.0, color=BLUE).move_to(right_origin + RIGHT * 1.3)
        
        # Clustering loss formula from the prompt
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=24
        ).to_edge(DOWN, buff=0.5)
        
        self.play(Create(circle_red), Create(circle_blue))
        self.play(Write(formula))
        self.play(
            Indicate(circle_red, color=RED, scale_factor=1.1),
            Indicate(circle_blue, color=BLUE, scale_factor=1.1)
        )
        
        self.wait(2)
