from manim import *
import numpy as np

class LearnClusteringSpace(Scene):
    def construct(self):
        # 1. Manual Axes (Lines) to replace restricted NumberPlane and Axes
        # Left Side: Physical Space
        l_ax_h = Line(LEFT * 6, LEFT * 1, color=GRAY)
        l_ax_v = Line(UP * 2.5 + LEFT * 3.5, DOWN * 2.5 + LEFT * 3.5, color=GRAY)
        left_center = [-3.5, 0, 0]
        
        # Right Side: Learned Space
        r_ax_h = Line(RIGHT * 1, RIGHT * 6, color=GRAY)
        r_ax_v = Line(UP * 2.5 + RIGHT * 3.5, DOWN * 2.5 + RIGHT * 3.5, color=GRAY)
        right_center = [3.5, 0, 0]

        left_label = Text("Physical Space", font_size=24).move_to([-3.5, 3, 0])
        right_label = Text("Learned Space", font_size=24).move_to([3.5, 3, 0])

        # 2. Hits (Dots)
        num_dots = 10
        np.random.seed(123)
        
        red_group = VGroup()
        blue_group = VGroup()

        # Physical space: Centered and overlapping
        for _ in range(num_dots):
            # Red dots slightly offset
            p1 = [left_center[0] + np.random.uniform(-0.4, 0.4), left_center[1] + np.random.uniform(-0.4, 0.4), 0]
            red_group.add(Dot(point=p1, color=RED, radius=0.08))
            
            # Blue dots overlapping with red
            p2 = [left_center[0] + np.random.uniform(-0.4, 0.4), left_center[1] + np.random.uniform(-0.4, 0.4), 0]
            blue_group.add(Dot(point=p2, color=BLUE, radius=0.08))

        # Learned space: Two separated clusters
        # Cluster 1 (Red) Top Right of right plane
        target_red = [[right_center[0] + 1.2 + np.random.uniform(-0.4, 0.4), 1.2 + np.random.uniform(-0.4, 0.4), 0] for _ in range(num_dots)]
        # Cluster 2 (Blue) Bottom Left of right plane
        target_blue = [[right_center[0] - 1.2 + np.random.uniform(-0.4, 0.4), -1.2 + np.random.uniform(-0.4, 0.4), 0] for _ in range(num_dots)]

        # 3. Animation Sequence
        self.add(l_ax_h, l_ax_v, r_ax_h, r_ax_v)
        self.add(left_label, right_label)
        self.play(FadeIn(red_group), FadeIn(blue_group))
        self.wait(1)

        # Transformation to Learned Space
        move_anims = []
        for i in range(num_dots):
            move_anims.append(red_group[i].animate.move_to(target_red[i]))
            move_anims.append(blue_group[i].animate.move_to(target_blue[i]))
        
        self.play(*move_anims, run_time=2, rate_func=smooth)

        # Draw classification indicators (Circles)
        # Replacing DashedVMobject with standard Circle for compatibility
        c1 = Circle(radius=0.9, color=RED).move_to([right_center[0] + 1.2, 1.2, 0])
        c2 = Circle(radius=0.9, color=BLUE).move_to([right_center[0] - 1.2, -1.2, 0])
        
        self.play(Create(c1), Create(c2))

        # Formula
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=24
        ).to_edge(DOWN, buff=0.3)
        
        self.play(Write(formula))
        self.wait(2)
