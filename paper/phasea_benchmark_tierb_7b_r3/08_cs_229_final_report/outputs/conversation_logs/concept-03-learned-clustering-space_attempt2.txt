import numpy as np

class LearnedClusteringSpace(Scene):

            x_range=[-4, 4],

            y_length=5,
            axis_config={"include_tip": False, "stroke_opacity": 0.5}
        ).shift(LEFT * 3.5)
        
        learn_axes = Axes(
            x_range=[-4, 4], 
            y_range=[-4, 4], 
            x_length=5, 
            y_length=5,
            axis_config={"include_tip": False, "stroke_opacity": 0.5}
        ).shift(RIGHT * 3.5)
        
        phys_label = Text("Physical Space", font_size=24).next_to(phys_axes, UP)
        learn_label = Text("Learned Space", font_size=24).next_to(learn_axes, UP)
        
        # 2. Formula (from the prompt)
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",

        # 3. Create clusters of dots

        # Separated distributions for Learned Space
