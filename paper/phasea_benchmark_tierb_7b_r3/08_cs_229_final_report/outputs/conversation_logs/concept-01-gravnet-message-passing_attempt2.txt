from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Create a scatter plot of dots representing ECal cells
        # We use a 2D representation as 3D scenes are restricted
        dots = VGroup()
        for x in range(-4, 5):
            for y in range(-2, 3):
                # Add slight randomness for irregular grid
                pos = np.array([
                    x * 0.8 + np.random.uniform(-0.2, 0.2),
                    y * 0.8 + np.random.uniform(-0.2, 0.2),
                    0
                ])
                dots.add(Dot(point=pos, radius=0.08, color=BLUE))
        
        dots.move_to(ORIGIN)
        
        # 2. Highlight one central dot
        center_idx = len(dots) // 2
        central_cell = dots[center_idx]
        
        # 3. Formula
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_edge(UP)
        
        self.play(FadeIn(dots), Write(formula))
        self.play(central_cell.animate.set_color(YELLOW).scale(1.5))

        # 4. Arrows from central dot to neighbors with updaters
        # The weight V = exp(-d^2) controls opacity and thickness
        arrows = VGroup()
        
        # We'll pick a subset of dots as neighbors to keep the visual clean
        for i, dot in enumerate(dots):
            if i == center_idx:
                continue
            
            # Filter for initial nearby neighbors
            dist_init = np.linalg.norm(dot.get_center() - central_cell.get_center())
            if dist_init > 2.5:
                continue

            # Create an arrow that updates its position and style every frame
            arrow = Line(central_cell.get_center(), dot.get_center(), color=RED)
            
            def arrow_updater(obj, target_dot=dot):
                p1 = central_cell.get_center()
                p2 = target_dot.get_center()
                diff = p2 - p1
                dist = np.linalg.norm(diff)
                
                # Weighted connection: exp(-d^2)
                # We scale the distance for better visualization in coordinate space
                weight = np.exp(-(dist**2) / 2.0)
                
                # Update the line to act like an arrow
                obj.set_points_as_corners([p1, p2])
                obj.set_stroke(width=10 * weight, opacity=weight)

            arrow.add_updater(arrow_updater)
            arrows.add(arrow)

        self.play(FadeIn(arrows))
        self.wait(1)

        # 5. Update the dots' positions to show "learned relationships"
        # As dots move, the updaters will automatically adjust arrow appearance
        movements = []
        for dot in dots:
            # Generate a small random shift
            shift = np.array([
                np.random.uniform(-1.5, 1.5),
                np.random.uniform(-1.0, 1.0),
                0
            ])
            # Some move closer to center, some move further
            target_pos = dot.get_center() + shift
            movements.append(dot.animate.move_to(target_pos))

        self.play(*movements, run_time=5, rate_func=smooth)
        self.wait(2)

        # Fade out
        self.play(FadeOut(arrows), FadeOut(dots), FadeOut(formula))
