import numpy as np
from manim import *

class GravNetAnimation(Scene):
    def construct(self):
        # 1. Setup Data - Irregular hexagonal-like grid in 2D
        # We use a standard Scene to avoid disallowed 3D features
        hex_angles = [0, 60, 120, 180, 240, 300]
        neighbor_dots = VGroup()
        for angle_deg in hex_angles:
            angle_rad = angle_deg * DEGREES
            # Distance from center with some jitter
            distance_scalar = 1.5 + np.random.uniform(-0.3, 0.3)
            pos_val = np.array([
                distance_scalar * np.cos(angle_rad), 
                distance_scalar * np.sin(angle_rad), 
                0
            ])
            # Vary dot radius slightly to simulate 'depth' in feature space
            dot_radius = 0.1 + np.random.uniform(-0.02, 0.05)
            neighbor_dots.add(Dot(point=pos_val, color=BLUE, radius=dot_radius))
        
        central_dot = Dot(point=ORIGIN, color=GREEN, radius=0.2)

        # 2. Formula label
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # 3. Connection Builder (Non-dynamic, used for Transform)
        # This function calculates connections based on current positions
        def get_connection_group(center_m, neighbors_vg):
            connections = VGroup()
            p_start = center_m.get_center()
            for dot_obj in neighbors_vg:
                p_end = dot_obj.get_center()
                # Calculate distance squared: d^2
                dist_sq = np.sum((p_start - p_end)**2)
                # Weighted potential: V = exp(-d^2)
                # We apply a small multiplier to dist_sq to balance Manim's coordinate units
                weight_val = np.exp(-dist_sq * 0.5)
                
                # Create arrow with visual properties tied to weight
                # buff=0.1 avoids overlapping the dots
                conn_arrow = Arrow(p_start, p_end, buff=0.1, color=YELLOW)
                # Opacity and thickness scale with the potential
                conn_arrow.set_stroke(
                    width=weight_val * 15, 
                    opacity=max(weight_val, 0.15)
                )
                connections.add(conn_arrow)
            return connections

        # Initial Arrows
        arrows = get_connection_group(central_dot, neighbor_dots)

        # 4. Intro Animation
        self.add(central_dot, neighbor_dots)
        self.play(Create(arrows), run_time=1.5)
        self.wait(1)

        # 5. Dynamic Update Loop (Simulating "learned feature" updates)
        # We iterate to show how the graph structure shifts as features update
        for _ in range(3):
            # Define target state for dots
            target_neighbors = VGroup()
            for neighbor in neighbor_dots:
                # Random jitter in 2D space
                move_offset = np.array([
                    np.random.uniform(-0.8, 0.8),
                    np.random.uniform(-0.8, 0.8),
                    0
                ])
                new_pos = neighbor.get_center() + move_offset
                new_dot = Dot(point=new_pos, color=BLUE, radius=neighbor.radius)
                target_neighbors.add(new_dot)
            
            # Central dot also shifts slightly
            center_offset = np.array([np.random.uniform(-0.4, 0.4), np.random.uniform(-0.4, 0.4), 0])
            target_center_pos = central_dot.get_center() + center_offset
            target_center = Dot(point=target_center_pos, color=GREEN, radius=0.2)
            
            # Calculate new arrows for the target positions
            target_arrows = get_connection_group(target_center, target_neighbors)
            
            # Animate transition between the graph states
            # rate_func=lambda t: t is used to ensure linear movement as requested
            self.play(
                Transform(neighbor_dots, target_neighbors),
                central_dot.animate.move_to(target_center_pos),
                Transform(arrows, target_arrows),
                run_time=2,
                rate_func=lambda t: t
            )
            self.wait(0.2)

        self.wait(2)
