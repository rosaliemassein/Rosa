from manim import *
import numpy as np

class ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup MathTex for the formula
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{\text{min}}")
        formula.to_edge(UP)
        self.add(formula)

        # 2. Create a plane of dots with varying beta values
        # We use numpy for randomness to avoid the 'random' undefined error
        np.random.seed(42)
        num_dots = 25
        dots = VGroup()
        betas = []
        
        # Designate two indices as high-beta condensation points
        high_beta_indices = [7, 18]
        
        for i in range(num_dots):
            pos = np.array([
                np.random.uniform(-5, 5),
                np.random.uniform(-3, 1),
                0
            ])
            
            if i in high_beta_indices:
                beta = np.random.uniform(0.85, 0.98)
            else:
                beta = np.random.uniform(0.1, 0.45)
            betas.append(beta)
            
            # Determine color based on beta manually to avoid interpolate_color issues
            if beta < 0.3:
                color = PURPLE
            elif beta < 0.6:
                color = BLUE
            elif beta < 0.8:
                color = GREEN
            else:
                color = YELLOW
            
            # Radius scales with beta
            dot = Dot(point=pos, radius=0.06 * (1 + 2.5 * beta), color=color)
            dots.add(dot)

        self.play(LaggedStart(*[FadeIn(d) for d in dots], lag_ratio=0.05))
        self.wait(0.5)

        # 3. Draw "Gravity Wells" for high beta dots
        # Instead of Surface (which is disallowed), we use concentric rings
        wells = VGroup()
        for idx in high_beta_indices:
            center_pos = dots[idx].get_center()
            for r in np.linspace(0.2, 1.8, 8):
                # Gradient effect using varying opacity
                ring = Circle(radius=r, stroke_width=2, color=YELLOW)
                ring.set_stroke(opacity=max(0, 1 - r/1.8))
                ring.move_to(center_pos)
                wells.add(ring)
        
        self.play(Write(wells), run_time=1.5)
        self.wait(0.5)

        # 4. Pull low-beta dots toward the condensation points
        # Each low-beta dot is pulled toward its nearest high-beta dot
        pull_animations = []
        for i, dot in enumerate(dots):
            if i in high_beta_indices:
                continue
            
            # Find closest condensation point
            centers = [dots[idx].get_center() for idx in high_beta_indices]
            dists = [np.linalg.norm(dot.get_center() - c) for c in centers]
            nearest_center = centers[np.argmin(dists)]
            
            # Calculate destination (75% of the way to the center)
            # Use manual lerp: (1-t)*A + t*B
            start_pos = dot.get_center()
            end_pos = start_pos + 0.75 * (nearest_center - start_pos)
            
            pull_animations.append(
                dot.animate.move_to(end_pos)
            )

        # Use a lambda for rate_func to avoid undefined exponential_decay
        self.play(*pull_animations, run_time=3, rate_func=lambda t: t**2)
        self.wait(2)
