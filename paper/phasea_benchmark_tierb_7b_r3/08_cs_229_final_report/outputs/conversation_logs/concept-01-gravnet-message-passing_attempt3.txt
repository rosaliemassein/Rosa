from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # Create a group of dots representing ECal cells in an irregular grid
        dots = VGroup()
        for x in range(-4, 5):
            for y in range(-3, 4):
                # Add irregular hexagonal-like spacing
                h_offset = 0.4 if y % 2 == 0 else 0
                pos = np.array([x * 0.9 + h_offset, y * 0.8, 0])
                # Add slight random jitter
                pos += np.random.uniform(-0.15, 0.15, 3)
                dot = Dot(point=pos, radius=0.08, color=GRAY)
                dots.add(dot)

        # Formula at the top corner
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # Identify a central hit to focus on
        center_dot = dots[len(dots) // 2 + 3]
        
        self.add(dots)
        self.play(center_dot.animate.set_color(BLUE).scale(1.5))
        self.wait(0.5)

        # Function to generate lines weighted by distance
        def get_lines(c_dot, all_dots):
            lines = VGroup()
            c_pos = c_dot.get_center()
            for d in all_dots:
                if d == c_dot:
                    continue
                d_pos = d.get_center()
                # Calculate Euclidean distance
                dist = np.linalg.norm(d_pos - c_pos)
                # Apply the potential function V = exp(-d^2)
                weight = np.exp(-(dist**2))
                
                # Draw lines for significant connections
                if weight > 0.05:
                    line = Line(c_pos, d_pos, color=RED)
                    # Scale opacity and thickness with weight
                    line.set_stroke(width=weight * 12, opacity=weight)
                    lines.add(line)
            return lines

        # Initial connections
        connections = get_lines(center_dot, dots)
        self.play(Create(connections))
        self.wait(1)

        # Setup dynamic structure update
        # We simulate "learned features" updating by moving dots slightly
        # and re-calculating connections
        
        # We'll perform the update in a loop to show shifting relationships
        for _ in range(2):
            # Determine new random positions for dots
            move_anims = []
            for d in dots:
                target_pos = d.get_center() + np.random.uniform(-0.5, 0.5, 3)
                # Keep center centered for clarity, or let it move slightly
                move_anims.append(d.animate.move_to(target_pos))
            
            # Create the updated lines for the new positions
            # We use a ValueTracker-like logic or simply update every frame
            # Since we want to show the 'shifting' we use a Transform
            
            # 1. Start movement
            # 2. In the next step, create new lines for those targets
            # To animate smoothly, we can use an updater
            
            def connections_updater(mob):
                new_conn = get_lines(center_dot, dots)
                mob.become(new_conn)

            connections.add_updater(connections_updater)
            
            self.play(
                *move_anims,
                run_time=3,
                rate_func=smooth
            )
            
            connections.remove_updater(connections_updater)
            self.wait(0.5)

        # Highlight final state
        self.play(
            center_dot.animate.set_color(YELLOW),
            dots.animate.set_color(LIGHT_GRAY),
            connections.animate.set_color(ORANGE),
            run_time=1
        )
        self.wait(2)
