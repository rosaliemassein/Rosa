from manim import *
import numpy as np

class ObjectCondensation(Scene):
    def construct(self):
        # 1. Formula setup
        q_formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        q_formula.to_edge(UP)
        self.play(Write(q_formula))

        # 2. Create 2D plane of dots
        num_dots = 60
        dots = VGroup(*[
            Dot(point=[np.random.uniform(-5, 5), np.random.uniform(-3, 2), 0], radius=0.06, color=PURPLE) 
            for _ in range(num_dots)
        ])
        self.play(FadeIn(dots, shift=UP), run_time=1)
        self.wait(0.5)

        # 3. Simulate beta values and identify condensation points
        # Higher beta means center of a particle
        center_indices = np.random.choice(range(num_dots), size=4, replace=False)
        betas = np.random.uniform(0.1, 0.4, size=num_dots)
        for idx in center_indices:
            betas[idx] = 0.95

        # 4. Transform dots based on beta values (Size and Color)
        # Using a manually calculated color interpolation if interpolate_color is finicky
        transformation_anims = []
        for i, dot in enumerate(dots):
            # Manually interpolate between PURPLE (#8732a8) and YELLOW (#ffff00)
            # Or use the built-in interpolate_color if available, otherwise fallback
            target_color = interpolate_color(PURPLE, YELLOW, betas[i])
            scale_factor = 1.0 + (betas[i] * 1.8)
            transformation_anims.append(
                dot.animate.set_color(target_color).scale(scale_factor)
            )
        
        self.play(*transformation_anims, run_time=1.5)

        # 5. Draw 'gravity wells' for high-beta dots
        wells = VGroup()
        for idx in center_indices:
            center_pos = dots[idx].get_center()
            # Create a concentric 'gravity well' (gradient circles)
            well = VGroup(*[
                Circle(radius=r, stroke_width=2, stroke_opacity=1 - (r/1.5), color=YELLOW)
                for r in np.linspace(0.1, 1.5, 10)
            ]).move_to(center_pos)
            wells.add(well)
        
        self.play(FadeIn(wells, scale=0.5), run_time=1)

        # 6. Animation of dots sliding along gradient lines toward nearest high-beta dot
        movement_anims = []
        for i, dot in enumerate(dots):
            if i in center_indices:
                continue
            
            dot_pos = dot.get_center()
            # Find the closest high-beta condensation point
            distances = [np.linalg.norm(dot_pos - dots[c_idx].get_center()) for c_idx in center_indices]
            closest_idx = np.argmin(distances)
            
            # If reasonably close, pull it in
            if distances[closest_idx] < 4.0:
                target_center_pos = dots[center_indices[closest_idx]].get_center()
                # Move 70% of the way to the center
                move_vec = (target_center_pos - dot_pos) * 0.7
                movement_anims.append(dot.animate.shift(move_vec))
            else:
                # "Noise" dots push away or fade
                noise_shift = np.array([np.random.uniform(-0.3, 0.3), np.random.uniform(-0.3, 0.3), 0])
                movement_anims.append(dot.animate.shift(noise_shift).set_opacity(0.3))

        # Using standard 'smooth' rate function to avoid undefined identifier errors
        self.play(*movement_anims, run_time=2, rate_func=smooth)
        
        self.wait(2)

        # Clean up
        self.play(
            FadeOut(dots), 
            FadeOut(wells), 
            FadeOut(q_formula),
            run_time=1
        )
