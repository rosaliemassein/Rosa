from manim import *
import numpy as np

class KMeansClustering(Scene):
    def construct(self):
        # 1. Draw the outline of a country (Australia-like shape)
        australia_points = [
            [-3, 1, 0], [-2, 1.5, 0], [0, 1.2, 0], [2, 1.5, 0], [3, 0.5, 0],
            [3.5, -1, 0], [2.5, -2, 0], [1, -1.8, 0], [-1, -2.2, 0], [-3, -1.5, 0],
            [-3.5, -0.5, 0], [-3, 1, 0]
        ]
        country_outline = Polygon(*australia_points, color=WHITE)
        self.play(Create(country_outline))

        # 2. Populate the country with data points
        dots = VGroup()
        for _ in range(100):
            p = [np.random.uniform(-2.5, 2.5), np.random.uniform(-1.5, 1.0), 0]
            dots.add(Dot(point=p, radius=0.06, color=GRAY, fill_opacity=0.7))
        self.play(FadeIn(dots))

        # 3. Animate Centroid crosses (mu_j) appearing at random locations
        # Defining a custom Cross shape since 'Cross' might have version issues or require a target
        def create_centroid_cross(color):
            line1 = Line(DL, UR, color=color).scale(0.15)
            line2 = Line(UL, DR, color=color).scale(0.15)
            return VGroup(line1, line2)

        centroid_colors = [RED, BLUE, GREEN, YELLOW, PURPLE]
        centroids = VGroup()
        for i in range(len(centroid_colors)):
            c = create_centroid_cross(centroid_colors[i])
            start_pos = [np.random.uniform(-1.5, 1.5), np.random.uniform(-1, 0.5), 0]
            c.move_to(start_pos)
            centroids.add(c)
        
        self.play(LaggedStart(*[Create(c) for c in centroids], lag_ratio=0.2))
        self.wait(0.5)

        # 4. Formula display
        formula = MathTex(
            r"\mu_{j}:=\frac{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}x^{(i)}}{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}}",
            font_size=36
        ).to_edge(UP)
        self.play(Write(formula))

        # 5. Voronoi-like behavior: Change color based on nearest centroid
        def get_nearest_centroid_index(dot, centroids):
            distances = [np.linalg.norm(dot.get_center() - c.get_center()) for c in centroids]
            return np.argmin(distances)

        color_animations = []
        for dot in dots:
            idx = get_nearest_centroid_index(dot, centroids)
            color_animations.append(dot.animate.set_color(centroid_colors[idx]))
        
        self.play(*color_animations, run_time=1.5)
        self.wait(0.5)

        # 6. Show centroids moving to the center of their respective colored groups
        move_animations = []
        for i in range(len(centroids)):
            # Find all dots assigned to this centroid color
            assigned_dots = [d for d in dots if np.allclose(d.get_color().to_rgb(), color_to_rgb(centroid_colors[i]))]
            if assigned_dots:
                new_pos = np.mean([d.get_center() for d in assigned_dots], axis=0)
                move_animations.append(centroids[i].animate.move_to(new_pos))
        
        self.play(*move_animations, run_time=2)
        self.wait(1)

        # 7. Highlight final cluster centers
        self.play(*[Indicate(c, color=WHITE, scale_factor=2) for c in centroids])
        self.wait(2)

        # Cleanup
        self.play(FadeOut(dots), FadeOut(country_outline), FadeOut(formula), FadeOut(centroids))
        self.wait(1)
