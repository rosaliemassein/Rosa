from manim import *
import numpy as np

class ContrastiveAlignment(Scene):
    def construct(self):
        # 1. Title
        title = Text("Contrastive Alignment for Image-Text Matching", font_size=32).to_edge(UP)
        self.play(Write(title))
        
        # 2. Setup Encoders (Image and Text Neural Network Blocks)
        image_encoder = Rectangle(width=2.4, height=1.0, color=BLUE).shift(LEFT * 4 + UP * 1.2)
        text_encoder = Rectangle(width=2.4, height=1.0, color=GREEN).shift(LEFT * 4 + DOWN * 1.2)
        img_enc_label = Text("Image Encoder", font_size=20).move_to(image_encoder)
        txt_enc_label = Text("Text Encoder", font_size=20).move_to(text_encoder)
        
        encoder_group = VGroup(image_encoder, text_encoder, img_enc_label, txt_enc_label)
        self.play(Create(encoder_group))
        
        # 3. Embedding Space Coordinate System (Using Arrows instead of Axes)
        origin = RIGHT * 2 + DOWN * 0.5
        x_axis = Arrow(origin, origin + RIGHT * 3, buff=0, color=GRAY)
        y_axis = Arrow(origin, origin + UP * 3, buff=0, color=GRAY)
        space_label = Text("Shared Space", font_size=18).next_to(y_axis, UP)
        self.play(Create(x_axis), Create(y_axis), Write(space_label))
        
        # 4. Initial Vectors and Tags
        # Image vector (fixed direction at 45 degrees)
        img_dir = np.array([1.8, 1.8, 0])
        img_vec = Arrow(origin, origin + img_dir, buff=0, color=BLUE)
        img_tag = Text("Image", font_size=18, color=BLUE).next_to(origin + img_dir, UR, buff=0.1)
        
        # Text vector (starts pointing in a different direction)
        txt_start_dir = np.array([0.5, 2.2, 0])
        txt_vec = Arrow(origin, origin + txt_start_dir, buff=0, color=GREEN)
        txt_tag = Text("Text", font_size=18, color=GREEN)
        # Keeping the tag attached to the arrow tip
        txt_tag.add_updater(lambda m: m.next_to(txt_vec.get_end(), UP, buff=0.1))
        
        # Similarity Score (Using Text and animate.become to avoid DecimalNumber/ValueTracker)
        sim_label = Text("Similarity: ", font_size=24).to_edge(DOWN, buff=1.5).shift(LEFT * 0.3)
        sim_score = Text("0.45", font_size=24).next_to(sim_label, RIGHT)
        
        self.play(
            Create(img_vec), 
            Write(img_tag), 
            Create(txt_vec), 
            Write(txt_tag), 
            Write(sim_label), 
            Write(sim_score)
        )
        self.wait(1)
        
        # 5. Correct Alignment Animation (France)
        france_in = Text("'A photo in France'", font_size=22).next_to(text_encoder, LEFT, buff=0.5)
        self.play(FadeIn(france_in))
        
        # Rotate text vector to align with image vector (45 degrees)
        # Using Rotate to animate the orientation shift
        self.play(
            Rotate(txt_vec, angle=-32 * DEGREES, about_point=origin),
            sim_score.animate.become(Text("1.00", font_size=24, color=YELLOW).move_to(sim_score)),
            run_time=2
        )
        self.wait(1)
        
        # 6. Wrong Label Animation (Japan)
        japan_in = Text("'A photo in Japan'", font_size=22).next_to(text_encoder, LEFT, buff=0.5)
        self.play(Transform(france_in, japan_in))
        
        # Rotate text vector away from the image vector
        self.play(
            Rotate(txt_vec, angle=130 * DEGREES, about_point=origin),
            sim_score.animate.become(Text("-0.30", font_size=24, color=RED).move_to(sim_score)),
            run_time=2
        )
        self.wait(1)
        
        # 7. CLIP Loss Formula
        formula = MathTex(r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})").to_edge(DOWN, buff=0.5)
        self.play(Write(formula))
        self.wait(2)
