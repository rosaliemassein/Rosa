from manim import *

class HierarchicalMLPRefinement(Scene):
    def construct(self):
        # 1. Image and Image Vector
        try:
            image = ImageMobject("img-8.jpeg").scale(0.8)
        except:
            image_rect = Rectangle(height=2, width=3, color=BLUE).set_fill(GREY, opacity=0.5)
            image_label = Text("Image", font_size=24).move_to(image_rect)
            image = Group(image_rect, image_label)
        
        image.to_edge(UP + LEFT, buff=0.5)
        image_vector_text = MathTex(r"\mathbf{x}^{(i)}", color=BLUE).next_to(image, DOWN, buff=0.2)
        
        # 2. CLIP Block
        clip_box = Rectangle(width=2, height=1, color=WHITE)
        clip_text = Text("CLIP", font_size=30).move_to(clip_box)
        clip_group = VGroup(clip_box, clip_text).next_to(image, RIGHT, buff=1.0)
        arrow_to_clip = Arrow(image.get_right(), clip_box.get_left())

        # 3. Country MLP Block
        country_mlp_rect = Rectangle(width=3, height=1, color=BLUE)
        country_mlp_text = Text("Country MLP", font_size=20).move_to(country_mlp_rect)
        country_mlp_group = VGroup(country_mlp_rect, country_mlp_text).next_to(clip_group, DOWN, buff=1)
        arrow_to_country = Arrow(clip_group.get_bottom(), country_mlp_group.get_top())

        # 4. Detailed 3-Layer Neural Network
        layer_sizes = [4, 5, 3]
        mlp_layers = VGroup()
        for i, size in enumerate(layer_sizes):
            layer = VGroup(*[Circle(radius=0.12, color=BLUE, fill_opacity=0.3) for _ in range(size)])
            layer.arrange(DOWN, buff=0.15)
            layer.shift(RIGHT * i * 1.2)
            mlp_layers.add(layer)
        
        mlp_layers.next_to(country_mlp_group, DOWN, buff=0.8).shift(LEFT * 0.5)
        
        connections = VGroup()
        for i in range(len(layer_sizes) - 1):
            for node1 in mlp_layers[i]:
                for node2 in mlp_layers[i+1]:
                    line = Line(node1.get_right(), node2.get_left(), stroke_width=1, stroke_opacity=0.2)
                    connections.add(line)
        
        relu_label = MathTex(r"\text{ReLU}", font_size=24, color=YELLOW).next_to(mlp_layers, RIGHT, buff=0.2)

        # 5. Map and Centroids
        map_rect = Rectangle(height=4, width=4, color=WHITE).to_edge(RIGHT, buff=0.5)
        map_bg = Rectangle(height=4, width=4).set_fill(WHITE, opacity=0.1).move_to(map_rect)
        map_title = Text("Local Clusters", font_size=20).next_to(map_rect, UP)
        
        centroid_positions = [[-1, 1, 0], [1, 0.5, 0], [-0.5, -1, 0], [0.8, -0.8, 0], [0, 0, 0]]
        centroids = VGroup(*[
            Dot(map_rect.get_center() + np.array(pos), color=RED, radius=0.08) 
            for pos in centroid_positions
        ])
        target_centroid = centroids[4] 
        arrow_mlp_to_map = Arrow(mlp_layers.get_right(), target_centroid.get_center(), color=YELLOW, buff=0.1)

        # 6. GPS Pin
        pin_base = Dot(target_centroid.get_center(), color=GREEN)
        pin_head = Triangle(color=GREEN, fill_opacity=1).scale(0.15).rotate(180*DEGREES).next_to(pin_base, UP, buff=0)
        gps_pin = VGroup(pin_base, pin_head)

        # 7. Formulas & Result Labels
        formula = MathTex(r"c^{(i)}:=\arg\min_{j}\|x^{(i)}-\mu_{j}\|^{2}", font_size=32).to_edge(BOTTOM, buff=0.5)
        final_coord_text = MathTex(r"(\text{lat}, \text{lon})", color=GREEN, font_size=36).next_to(map_rect, DOWN, buff=0.2)

        # ANIMATION
        self.add(image, image_vector_text)
        self.play(Create(arrow_to_clip), FadeIn(clip_group))
        self.wait(0.5)
        
        self.play(Create(arrow_to_country), FadeIn(country_mlp_group))
        self.wait(0.5)
        
        self.play(Create(mlp_layers), Create(connections), Write(relu_label))
        
        # Signal flow animation
        all_nodes = VGroup(*[n for layer in mlp_layers for n in layer])
        self.play(LaggedStart(*[n.animate.set_color(YELLOW) for n in all_nodes], lag_ratio=0.05))
        
        self.play(Create(map_rect), Create(map_bg), Write(map_title))
        self.play(Create(centroids))
        
        self.play(Create(arrow_mlp_to_map))
        self.play(
            target_centroid.animate.scale(2).set_color(GREEN),
            FadeIn(gps_pin, shift=DOWN)
        )
        
        self.play(Write(formula))
        # ReplacementTransform is a robust alternative for TransformMatchingTex errors
        self.play(ReplacementTransform(image_vector_text.copy(), final_coord_text))
        
        self.wait(2)
