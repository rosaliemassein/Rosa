from manim import *
import numpy as np

class FeatureMatrixScene(Scene):
    def construct(self):
        # 1. Create silhouette of 15 basketball players
        # Using Dots to represent players
        players = VGroup(*[Dot(radius=0.15, color=BLUE) for _ in range(15)])
        players.arrange(RIGHT, buff=0.4).to_edge(UP, buff=1.0)
        
        # 2. Create minutes played filter
        filter_rect = Rectangle(width=0.6, height=1.2, color=YELLOW).move_to(players[0])
        
        self.add(players)
        self.play(Create(filter_rect))

        # 3. Animate minutes played filter passing over them, highlighting top 10
        for i in range(15):
            self.play(filter_rect.animate.move_to(players[i]), run_time=0.1)
            if i >= 10:
                players[i].set_color(DARK_GRAY)
            else:
                players[i].set_color(YELLOW)
        
        self.play(FadeOut(filter_rect))
        self.wait(0.5)

        # 4. Create 10x15 grid of squares representing a 'Team Matrix'
        grid = VGroup()
        for i in range(10):
            row = VGroup()
            for j in range(15):
                # Normalized data values represented by varying opacity
                intensity = np.random.uniform(0.2, 1.0)
                cell = Square(side_length=0.35, stroke_width=1, color=WHITE)
                cell.set_fill(BLUE, opacity=intensity)
                row.add(cell)
            row.arrange(RIGHT, buff=0.05)
            grid.add(row)
        grid.arrange(DOWN, buff=0.05).center().shift(DOWN * 0.5)

        # 5. Slide the top 10 players into the rows of the grid
        self.play(
            *[players[i].animate.move_to(grid[i][0].get_left() + LEFT * 0.5) for i in range(10)],
            run_time=1.5
        )
        self.play(FadeIn(grid))

        # 6. Labels for stats above the columns
        stat_labels_text = ["PTS", "REB", "AST", "TOV", "STL", "BLK", "PF", "FGM", "FGA", "3PM", "3PA", "FTM", "FTA", "OREB", "PIE"]
        labels = VGroup()
        for i, text in enumerate(stat_labels_text):
            label = Text(text, font_size=12).next_to(grid[0][i], UP, buff=0.1).rotate(45 * DEGREES)
            labels.add(label)
        
        self.play(Write(labels))

        # 7. Show the formula n = m x p
        formula = MathTex("n = m \\times p").to_corner(UL, buff=0.5)
        self.play(Write(formula))
        self.wait(1)

        # 8. Collapse 2D grid into a single horizontal vector of 150 elements
        # Flatten the grid into a list of cells
        all_cells = VGroup(*[cell for row in grid for cell in row])
        
        # Create a target arrangement for the vector
        # 14.0 is used as a standard frame width fallback
        vector_target = all_cells.generate_target()
        vector_target.arrange(RIGHT, buff=0.01)
        vector_target.scale_to_fit_width(13.5).to_edge(DOWN, buff=1)

        self.play(
            FadeOut(labels),
            FadeOut(players),
            FadeOut(formula),
            MoveToTarget(all_cells),
            run_time=2
        )
        
        # Final animation showing it "entering the model"
        self.play(all_cells.animate.shift(RIGHT * 12), rate_func=smooth, run_time=1.5)
        self.wait()
