from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Manual axes construction to avoid restricted Axes class
        # x-axis: [0, 10], y-axis: [0, 10]
        # Coordinates: Origin at (-3.5, -2). X-scale 0.7 per unit, Y-scale 0.35 per unit.
        x_axis_line = Line(LEFT * 3.5 + DOWN * 2, RIGHT * 3.5 + DOWN * 2)
        y_axis_line = Line(LEFT * 3.5 + DOWN * 2, LEFT * 3.5 + UP * 1.5)
        
        # Labels using MathTex for robust formatting
        label_n_players = MathTex(r"Number \ of \ Players").scale(0.6).next_to(x_axis_line, DOWN)
        label_mse = MathTex(r"MSE").scale(0.6).rotate(90 * DEGREES).next_to(y_axis_line, LEFT)
        
        self.add(x_axis_line, y_axis_line, label_n_players, label_mse)

        # Coordinate helper function
        def get_pos(val_x, val_y):
            return np.array([-3.5 + val_x * 0.7, -2.0 + val_y * 0.35, 0])

        # 2. Plotting Curves
        # KNN Curve (Red): Dips at first then rises sharply (overfitting)
        knn_curve_pts = []
        for i_step in range(101):
            curr_x = i_step * 0.1
            curr_y = 0.15 * (curr_x - 3.5)**2 + 2
            knn_curve_pts.append(get_pos(curr_x, curr_y))
        knn_graph = VMobject(color=RED).set_points_as_corners(knn_curve_pts)
        knn_tag = MathTex(r"\text{KNN}", color=RED).scale(0.5).move_to(get_pos(9, 7))

        # Lasso/Ridge Curve (Green): Steady decline
        reg_curve_pts = []
        for i_step in range(101):
            curr_x = i_step * 0.1
            curr_y = 7 * np.exp(-0.35 * curr_x) + 1
            reg_curve_pts.append(get_pos(curr_x, curr_y))
        reg_graph = VMobject(color=GREEN).set_points_as_corners(reg_curve_pts)
        reg_tag = MathTex(r"\text{Lasso/Ridge}", color=GREEN).scale(0.5).move_to(get_pos(9, 1.5))

        self.play(Create(knn_graph), Create(reg_graph), Write(knn_tag), Write(reg_tag))
        self.wait(1)

        # 3. Regularization formula
        # Segmented formula to highlight lambda later
        formula_tex = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 +", 
            r"\lambda", 
            r"\|\theta\|_1"
        )
        formula_tex.scale(0.8).to_edge(UP, buff=0.5)
        self.play(Write(formula_tex))

        # 4. Penalty visualization: coefficient bars
        # Representing 40 coefficients
        coeff_bars = VGroup()
        for i_idx in range(40):
            bar_height = 0.5 + (i_idx % 5) * 0.2
            bar_obj = Line(ORIGIN, UP * bar_height, color=BLUE, stroke_width=3)
            coeff_bars.add(bar_obj)
        coeff_bars.arrange(RIGHT, buff=0.1).shift(DOWN * 3)
        
        self.play(Create(coeff_bars))
        self.wait(0.5)

        # 5. Transition: Lambda increases, Lasso shrinks coefficients to zero
        target_coeff_bars = VGroup()
        for i_idx in range(len(coeff_bars)):
            base_p = coeff_bars[i_idx].get_start()
            # In Lasso, most coefficients hit exactly zero
            # We keep a few small non-zero bars to represent "important" features
            if i_idx % 8 == 0:
                new_h = 0.25
            else:
                new_h = 0.001 # Visual representation of vanishing
            
            target_bar = Line(base_p, base_p + UP * new_h, color=BLUE, stroke_width=3)
            target_coeff_bars.add(target_bar)

        self.play(
            Transform(coeff_bars, target_coeff_bars),
            formula_tex[1].animate.set_color(YELLOW).scale(1.2),
            run_time=3
        )
        self.wait(2)
