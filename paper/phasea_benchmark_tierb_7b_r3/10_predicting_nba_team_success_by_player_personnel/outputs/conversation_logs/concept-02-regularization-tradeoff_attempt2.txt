from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # Manual construction of Axes to avoid disallowed 'Axes' class
        origin_pos = np.array([-4.5, -2, 0])
        x_axis_line = Line(origin_pos, origin_pos + np.array([6, 0, 0]), color=WHITE)
        y_axis_line = Line(origin_pos, origin_pos + np.array([0, 5, 0]), color=WHITE)
        
        x_label = Text("Number of Players").scale(0.4).next_to(x_axis_line, DOWN)
        y_label = Text("Mean Squared Error").scale(0.4).rotate(90 * DEGREES).next_to(y_axis_line, LEFT)
        
        self.add(x_axis_line, y_axis_line, x_label, y_label)

        # Helper to map data coordinates to screen coordinates
        # Data X: 1 to 10 -> Screen X: 0 to 6
        # Data Y: 0 to 10 -> Screen Y: 0 to 5
        def to_screen(val_x, val_y):
            s_x = origin_pos[0] + (val_x - 1) * (6 / 9)
            s_y = origin_pos[1] + (val_y) * (5 / 10)
            return np.array([s_x, s_y, 0])

        # Plot KNN line (Red): Overfitting curve (dips then rises)
        knn_pts = []
        for i in range(51):
            curr_x = 1 + i * (9 / 50)
            curr_y = 0.25 * (curr_x - 4)**2 + 2
            knn_pts.append(to_screen(curr_x, curr_y))
        knn_line = VMobject(color=RED).set_points_as_corners(knn_pts)
        knn_tag = Text("KNN", color=RED).scale(0.4).move_to(to_screen(9, 8.5))

        # Plot Lasso/Ridge line (Green): Steady decline
        reg_pts = []
        for i in range(51):
            curr_x = 1 + i * (9 / 50)
            curr_y = 7.5 / (curr_x ** 0.5)
            reg_pts.append(to_screen(curr_x, curr_y))
        reg_line = VMobject(color=GREEN).set_points_as_corners(reg_pts)
        reg_tag = Text("Lasso/Ridge", color=GREEN).scale(0.4).move_to(to_screen(9, 1.5))

        self.play(Create(knn_line), Create(reg_line))
        self.play(Write(knn_tag), Write(reg_tag))
        self.wait(1)

        # Lasso Cost Function Formula
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            color=WHITE
        ).scale(0.8).to_corner(UP + RIGHT, buff=1)
        self.play(Write(formula))

        # Coefficients Visualization (150 bars)
        coeffs = VGroup()
        np.random.seed(42)
        for _ in range(150):
            height_val = np.random.uniform(0.1, 1.8)
            # Create small vertical lines to represent coefficients
            bar = Line(ORIGIN, UP * height_val, stroke_width=2, color=BLUE)
            coeffs.add(bar)
        
        coeffs.arrange(RIGHT, buff=0.01, aligned_edge=DOWN).next_to(formula, DOWN, buff=0.8)
        self.play(Create(coeffs), run_time=2)
        self.wait(1)

        # Visualization of the Penalty (Lambda increasing)
        # We target the lambda term in the MathTex (index 0 if it's one string, but we can highlight it)
        highlight_box = SurroundingRectangle(formula, color=YELLOW, buff=0.1)
        self.play(Create(highlight_box))

        # Animate bars shrinking and vanishing (feature selection)
        shrinkage_animations = []
        for bar in coeffs:
            # Probability that a coefficient goes to zero in Lasso
            if np.random.rand() < 0.7:
                shrinkage_animations.append(
                    bar.animate.scale(0, about_edge=DOWN).set_opacity(0)
                )
            else:
                # Shrink but don't vanish
                shrinkage_animations.append(
                    bar.animate.scale(0.2, about_edge=DOWN)
                )

        self.play(*shrinkage_animations, run_time=3)
        self.play(FadeOut(highlight_box))
        self.wait(2)
