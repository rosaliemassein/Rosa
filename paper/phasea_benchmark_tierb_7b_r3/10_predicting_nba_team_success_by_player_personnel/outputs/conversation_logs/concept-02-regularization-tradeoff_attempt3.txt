from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Setup Axes
        # Using a coordinate system to plot the MSE vs Players tradeoff
        ax = Axes(
            x_range=[0, 11, 1],
            y_range=[0, 2.5, 0.5],
            x_length=9,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(DOWN, buff=0.5)
        
        # Custom labels to avoid potential Axis label helper issues
        x_lab = Text("Number of Players", font_size=20).next_to(ax.x_axis, DOWN, buff=0.2)
        y_lab = Text("Mean Squared Error", font_size=20).rotate(90 * DEGREES).next_to(ax.y_axis, LEFT, buff=0.2)
        
        # 2. Plotting Curves
        # KNN curve: drops at first, then rises sharply due to overfitting (curse of dimensionality)
        # Using 't' as the lambda parameter to avoid reserved identifier issues
        knn_graph = ax.plot(
            lambda t: 0.1 * (t - 3)**2 + 0.5 if t < 3 else 0.2 * (t - 3)**2 + 0.5,
            x_range=[1, 8],
            color=RED
        )
        knn_tag = Text("KNN (Overfitting)", color=RED, font_size=16).next_to(knn_graph.get_end(), UR, buff=0.1)

        # Lasso/Ridge curve: steady decline as regularization handles noise
        lasso_graph = ax.plot(
            lambda t: 1.8 * np.exp(-0.3 * t) + 0.3,
            x_range=[1, 10],
            color=GREEN
        )
        lasso_tag = Text("Lasso/Ridge", color=GREEN, font_size=16).next_to(lasso_graph.get_end(), RIGHT, buff=0.1)

        self.play(Create(ax), Write(x_lab), Write(y_lab))
        self.play(Create(knn_graph), Write(knn_tag))
        self.play(Create(lasso_graph), Write(lasso_tag))
        self.wait(1)

        # 3. Regularization Formula
        # J(theta) = Loss + Penalty
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            font_size=40
        ).to_edge(UP, buff=0.5)
        
        self.play(Write(formula))
        self.wait()

        # 4. Coefficients Visualization (Bar Chart)
        # Recreate the effect of coefficients shrinking and vanishing (Lasso selection)
        num_coeffs = 150
        np.random.seed(42)
        initial_heights = np.random.uniform(0.1, 1.2, num_coeffs)
        
        coeffs_group = VGroup()
        for h in initial_heights:
            rect = Rectangle(
                width=0.03, 
                height=h, 
                fill_color=BLUE, 
                fill_opacity=0.7, 
                stroke_width=0
            )
            coeffs_group.add(rect)
        
        coeffs_group.arrange(RIGHT, buff=0.01).next_to(formula, DOWN, buff=0.5)
        
        # Move existing plot down slightly to make room for bars
        self.play(
            FadeIn(coeffs_group, shift=DOWN),
            VGroup(ax, x_lab, y_lab, knn_graph, lasso_graph, knn_tag, lasso_tag).animate.scale(0.7).to_edge(DOWN, buff=0.2)
        )
        self.wait(1)

        # 5. Shrinkage Animation
        # As lambda increases, coefficients shrink. In Lasso, many hit exactly zero.
        shrink_anims = []
        for i, bar in enumerate(coeffs_group):
            # Logic: Some coefficients stay (shrunk), many vanish (set to near zero height)
            if i % 5 == 0:
                # Kept features
                target_height = initial_heights[i] * 0.3
                target_opacity = 0.9
                target_color = BLUE
            else:
                # Vanished features (Feature Selection)
                target_height = 0.001
                target_opacity = 0.1
                target_color = GRAY
                
            shrink_anims.append(
                bar.animate.stretch_to_fit_height(target_height).set_fill(color=target_color, opacity=target_opacity)
            )

        self.play(*shrink_anims, run_time=3)
        self.wait(2)
