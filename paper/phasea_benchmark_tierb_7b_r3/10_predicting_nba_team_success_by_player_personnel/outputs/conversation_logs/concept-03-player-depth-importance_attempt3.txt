from manim import *

class PlayerDepthImportance(Scene):
    def construct(self):
        # 1. Formula reference
        formula = MathTex(
            r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})",
            font_size=36
        ).to_edge(UP, buff=0.7)

        # 2. Define data for the bars (Estimates to match the "star" and "depth" importance)
        # Higher values for 1st, 2nd and local peaks for 5th, 8th
        coeffs = [3.8, 3.2, 1.4, 1.3, 2.6, 1.1, 0.9, 2.1, 0.6, 0.5]
        
        bars = VGroup()
        icons = VGroup()
        
        for i, val in enumerate(coeffs):
            # Create the bar
            bar = Rectangle(
                width=0.5, 
                height=val, 
                fill_opacity=0.8,
                stroke_width=1
            )
            
            # Set initial colors
            if i == 0 or i == 1:
                bar.set_color(GOLD)
            elif i == 4 or i == 7:
                bar.set_color(BLUE)
            else:
                bar.set_color(GREY)
            
            # Create a simple "player icon" (Circle with a number)
            icon_circle = Circle(radius=0.2, color=WHITE, stroke_width=2)
            icon_label = Text(f"{i+1}", font_size=16).move_to(icon_circle.get_center())
            player_icon = VGroup(icon_circle, icon_label)
            
            bars.add(bar)
            icons.add(player_icon)

        # 3. Layout: Arrange players in a row
        all_elements = VGroup()
        for b, ic in zip(bars, icons):
            unit = VGroup(b, ic).arrange(DOWN, buff=0.2)
            all_elements.add(unit)
            
        all_elements.arrange(RIGHT, buff=0.3, aligned_edge=DOWN)
        all_elements.center().shift(DOWN * 0.5)

        # 4. Base Line
        base_line = Line(
            all_elements.get_left() + LEFT * 0.2,
            all_elements.get_right() + RIGHT * 0.2,
            color=GREY_B
        ).next_to(icons, UP, buff=0.1)

        # 5. Animation Sequence
        self.play(Write(formula))
        self.wait(0.5)
        
        # Display icons and base line
        self.play(
            Create(base_line),
            FadeIn(icons, shift=UP)
        )
        
        # Animate bars growing from the base
        self.play(
            AnimationGroup(
                *[GrowFromPoint(bar, bar.get_bottom()) for bar in bars],
                lag_ratio=0.1,
                run_time=2
            )
        )
        self.wait(1)

        # Highlight Stars (1st and 2nd) with a gold pulse/indication
        self.play(
            Indicate(bars[0], color=GOLD, scale_factor=1.2),
            Indicate(bars[1], color=GOLD, scale_factor=1.2),
            run_time=1.5
        )
        self.wait(0.5)

        # Highlight Depth (5th and 8th) with a pulsing blue light
        self.play(
            Indicate(bars[4], color=BLUE, scale_factor=1.3),
            Indicate(bars[7], color=BLUE, scale_factor=1.3),
            run_time=2
        )
        
        # Flash the specific values to emphasize depth findings
        self.play(
            Flash(bars[4], color=BLUE, flash_radius=0.5),
            Flash(bars[7], color=BLUE, flash_radius=0.5)
        )

        self.wait(3)
