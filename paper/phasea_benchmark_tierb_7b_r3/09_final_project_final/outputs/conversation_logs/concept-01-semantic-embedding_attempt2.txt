import numpy as np
from manim import *

class Concept01SemanticEmbedding(Scene):
    def construct(self):
        # 1. Setup Vector Space (2D Axes as NumberPlane is disallowed)
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6,
            axis_config={"color": WHITE}
        )
        self.add(axes)

        # 2. Labels and Positions
        # Deep Learning and Neural Networks are close, Robotics is far
        pos_dl = axes.c2p(-3, 1.5, 0)
        pos_nn = axes.c2p(-2.5, 1, 0)
        pos_ro = axes.c2p(3, -1.5, 0)

        label_dl = Text("Deep Learning", font_size=20).move_to(pos_dl + UP * 0.4)
        label_nn = Text("Neural Networks", font_size=20).move_to(pos_nn + DOWN * 0.4)
        label_ro = Text("Robotics", font_size=20).move_to(pos_ro + UP * 0.4)

        # 3. Animate text pop up
        self.play(Write(label_dl), Write(label_nn), Write(label_ro))
        self.wait(1)

        # 4. Transform into Dots
        dot_dl = Dot(pos_dl, color=BLUE)
        dot_nn = Dot(pos_nn, color=BLUE)
        dot_ro = Dot(pos_ro, color=RED)

        self.play(
            Transform(label_dl, dot_dl),
            Transform(label_nn, dot_nn),
            Transform(label_ro, dot_ro)
        )
        self.wait(1)

        # 5. Show distances (Using Line instead of DashedLine)
        line_close = Line(pos_dl, pos_nn, color=YELLOW, stroke_width=2)
        line_far = Line(pos_dl, pos_ro, color=WHITE, stroke_width=1, stroke_opacity=0.5)

        self.play(Create(line_close))
        self.play(Create(line_far))
        self.wait(2)

        # 6. Clear for Averaging Concept
        self.play(
            FadeOut(label_dl), FadeOut(label_nn), FadeOut(label_ro),
            FadeOut(line_close), FadeOut(line_far)
        )

        # 7. Machine and Learning vectors
        v_machine_coord = np.array([-2, -1, 0])
        v_learning_coord = np.array([1, 2, 0])
        v_avg_coord = (v_machine_coord + v_learning_coord) / 2

        pos_m = axes.c2p(*v_machine_coord)
        pos_l = axes.c2p(*v_learning_coord)
        pos_avg = axes.c2p(*v_avg_coord)

        word_m = Text("Machine", font_size=24).move_to(pos_m + LEFT * 0.7)
        vec_m = Arrow(axes.c2p(0, 0, 0), pos_m, buff=0, color=ORANGE)
        
        word_l = Text("Learning", font_size=24).move_to(pos_l + RIGHT * 0.7)
        vec_l = Arrow(axes.c2p(0, 0, 0), pos_l, buff=0, color=PURPLE)

        self.play(Create(vec_m), Write(word_m))
        self.play(Create(vec_l), Write(word_l))
        self.wait(1)

        # 8. Transform to merge into a single central vector
        vec_phrase = Arrow(axes.c2p(0, 0, 0), pos_avg, buff=0, color=GREEN)
        word_phrase = Text("Machine Learning", font_size=24).move_to(pos_avg + UP * 0.5)

        self.play(
            Transform(vec_m, vec_phrase),
            Transform(vec_l, vec_phrase),
            Transform(word_m, word_phrase),
            Transform(word_l, word_phrase)
        )
        self.wait(1)

        # 9. Display the formula
        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}")
        formula.to_edge(UP)
        self.play(Write(formula))
        
        self.wait(2)
