from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Background Setup: Manual Axes (Avoiding 'Axes' class)
        origin = LEFT * 3.5 + DOWN * 1.5
        x_axis = Line(origin, origin + RIGHT * 4, color=GRAY)
        y_axis = Line(origin, origin + UP * 4, color=GRAY)
        axes_labels = VGroup(
            MathTex("x").next_to(x_axis, RIGHT),
            MathTex("y").next_to(y_axis, UP)
        )
        self.add(x_axis, y_axis, axes_labels)

        # 2. Vector Logic: Using a hidden Dot to simulate a ValueTracker
        # X-coordinate of tracker_dot represents the angle theta
        tracker_dot = Dot(point=[PI/4, 0, 0]).hide()
        self.add(tracker_dot)

        # Vector A (Static)
        vec_a = Arrow(origin, origin + RIGHT * 3, buff=0, color=RED)
        label_a = MathTex(r"\mathbf{A}", color=RED).next_to(vec_a.get_end(), DOWN)
        
        # Vector B (Dynamic)
        vec_b = Arrow(origin, origin + RIGHT * 3, buff=0, color=BLUE)
        def b_updater(m):
            ang = tracker_dot.get_center()[0]
            # Handle small angles to avoid zero-length errors in some backends
            ang = max(0.001, ang)
            target = origin + np.array([np.cos(ang), np.sin(ang), 0]) * 3
            m.put_start_and_end_on(origin, target)
        vec_b.add_updater(b_updater)

        # Angle Arc (Avoiding 'Angle' class)
        arc = Arc(radius=0.8, start_angle=0, angle=PI/4, arc_center=origin, color=GREEN)
        def arc_updater(m):
            ang = tracker_dot.get_center()[0]
            ang = max(0.001, ang)
            m.become(Arc(radius=0.8, start_angle=0, angle=ang, arc_center=origin, color=GREEN))
        arc.add_updater(arc_updater)

        # Labels and Formula
        formula = MathTex(r"\text{similarity} = \cos(\theta)", font_size=36).to_edge(UP)
        sim_value = MathTex(r"0.71", color=YELLOW).next_to(formula, RIGHT)
        
        def sim_updater(m):
            ang = tracker_dot.get_center()[0]
            val = np.cos(ang)
            m.become(MathTex(f"{val:.2f}", color=YELLOW).next_to(formula, RIGHT))
        sim_value.add_updater(sim_updater)

        self.play(Create(vec_a), Create(vec_b), Create(arc), Write(label_a), Write(formula), Write(sim_value))

        # 3. Matrix Setup
        grid = VGroup()
        rows, cols = 5, 5
        for i in range(rows):
            for j in range(cols):
                sq = Square(side_length=0.6)
                sq.move_to(RIGHT * 3 + UP * (i - 2) * 0.7 + RIGHT * (j - 2) * 0.7)
                sq.set_fill(BLUE, opacity=0.1)
                sq.set_stroke(WHITE, width=1)
                grid.add(sq)
        
        self.play(Create(grid))

        # Target cell that glows based on alignment (e.g., cell 12 is the center)
        target_cell = grid[12]
        def cell_updater(m):
            ang = tracker_dot.get_center()[0]
            val = np.cos(ang)
            # Normalize and set brightness
            opacity = max(0.1, val)
            m.set_fill(YELLOW if val > 0.8 else BLUE, opacity=opacity)
            if val > 0.9:
                m.set_stroke(WHITE, width=4)
            else:
                m.set_stroke(WHITE, width=1)
        
        target_cell.add_updater(cell_updater)

        # 4. Animation Sequence
        # Rotate B to align with A (Angle -> 0, Similarity -> 1.0)
        self.play(tracker_dot.animate.move_to([0.001, 0, 0]), run_time=3)
        self.wait(1)

        # Rotate B away (Angle -> 90 degrees, Similarity -> 0.0)
        self.play(tracker_dot.animate.move_to([PI / 2, 0, 0]), run_time=3)
        self.wait(1)

        # Final move to a custom alignment
        self.play(tracker_dot.animate.move_to([PI / 6, 0, 0]), run_time=2)
        self.wait(2)

        # Cleanup
        vec_b.remove_updater(b_updater)
        arc.remove_updater(arc_updater)
        sim_value.remove_updater(sim_updater)
        target_cell.remove_updater(cell_updater)
        self.play(FadeOut(VGroup(x_axis, y_axis, axes_labels, vec_a, vec_b, arc, label_a, formula, sim_value, grid)))
