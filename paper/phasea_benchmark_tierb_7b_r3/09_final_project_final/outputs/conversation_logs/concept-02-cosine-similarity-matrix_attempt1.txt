from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Manual Axes (Replacing disallowed Axes)
        x_axis = Line(LEFT * 3, RIGHT * 3, color=WHITE).shift(LEFT * 3 + DOWN * 1)
        y_axis = Line(DOWN * 2, UP * 4, color=WHITE).shift(LEFT * 3 + DOWN * 1)
        origin = x_axis.get_start() + RIGHT * 3
        
        # 2. Angle Tracker using a hidden Dot (Replacing disallowed ValueTracker)
        # We will use the x-coordinate of this dot to store the angle in radians
        tracker_dot = Dot(point=[1.5, 0, 0], radius=0, fill_opacity=0)
        self.add(tracker_dot)

        # 3. Vectors A and B (Replacing disallowed Vector with Arrow)
        # Vector A is fixed at a 20 degree angle
        vec_a_end = origin + [2.5 * np.cos(20 * DEGREES), 2.5 * np.sin(20 * DEGREES), 0]
        vec_a = Arrow(origin, vec_a_end, buff=0, color=RED)
        label_a = MathTex(r"\mathbf{A}", color=RED).next_to(vec_a.get_end(), RIGHT, buff=0.1)
        
        # Vector B is dynamic based on tracker_dot's x-coordinate
        vec_b = Arrow(origin, origin + RIGHT, buff=0, color=BLUE)
        def update_vec_b(mob):
            angle = tracker_dot.get_center()[0]
            new_end = origin + [2.5 * np.cos(angle), 2.5 * np.sin(angle), 0]
            mob.put_start_and_end_on(origin, new_end)
        vec_b.add_updater(update_vec_b)
        
        label_b = MathTex(r"\mathbf{B}", color=BLUE)
        def update_label_b(mob):
            mob.next_to(vec_b.get_end(), UP + RIGHT, buff=0.1)
        label_b.add_updater(update_label_b)

        # 4. Angle Arc (Replacing disallowed Angle and always_redraw)
        arc_obj = Arc(radius=0.7, start_angle=0, angle=0.5, arc_center=origin, color=YELLOW)
        def update_arc(mob):
            angle_b = tracker_dot.get_center()[0]
            angle_a = 20 * DEGREES
            start = min(angle_a, angle_b)
            total = abs(angle_a - angle_b)
            # Reconstruct the arc manually
            mob.become(Arc(radius=0.7, start_angle=start, angle=total, arc_center=origin, color=YELLOW))
        arc_obj.add_updater(update_arc)

        # 5. Similarity Label
        # Replacing disallowed DecimalNumber with Text
        sim_label = MathTex(r"\text{Similarity: }", color=WHITE).to_edge(UP).shift(RIGHT * 2)
        sim_val_text = Text("0.00", color=YELLOW, font_size=36).next_to(sim_label, RIGHT)
        
        def update_sim_val(mob):
            angle_b = tracker_dot.get_center()[0]
            angle_a = 20 * DEGREES
            val = np.cos(angle_b - angle_a)
            mob.become(Text(f"{val:.2f}", color=YELLOW, font_size=36).next_to(sim_label, RIGHT))
        sim_val_text.add_updater(update_sim_val)

        # 6. Similarity Matrix (Grid of Squares)
        matrix_group = VGroup()
        for i in range(5):
            for j in range(5):
                sq = Square(side_length=0.5, stroke_width=2, color=WHITE)
                sq.set_fill(BLUE, opacity=0.2)
                matrix_group.add(sq)
        matrix_group.arrange_in_grid(rows=5, cols=5, buff=0.05).to_edge(RIGHT, buff=0.8)
        
        # Target cell in the matrix to glow
        target_cell = matrix_group[12] # Center cell (2,2)
        def update_target_cell(mob):
            angle_b = tracker_dot.get_center()[0]
            angle_a = 20 * DEGREES
            val = np.cos(angle_b - angle_a)
            # Map similarity 0..1 to color intensity manually
            # Using basic interpolation: Blue (0.2 opacity) to Yellow (1.0 opacity)
            alpha = max(0, min(1, val))
            mob.set_fill(color=YELLOW if alpha > 0.8 else BLUE, opacity=0.2 + 0.8 * alpha)
        target_cell.add_updater(update_target_cell)

        # 7. Animation Sequence
        self.play(Create(x_axis), Create(y_axis))
        self.play(Create(vec_a), Write(label_a))
        self.play(Create(vec_b), Write(label_b))
        self.add(arc_obj, sim_label, sim_val_text)
        self.play(Create(matrix_group))
        self.wait(1)

        # Move tracker dot to simulate rotation
        # Rotate B from 90 degrees (1.57 rad) to 20 degrees (0.35 rad)
        self.play(tracker_dot.animate.shift(LEFT * (1.57 - 0.35)), run_time=4)
        self.wait(1)
        
        # Diverge B away again
        self.play(tracker_dot.animate.shift(RIGHT * 1.0), run_time=2)
        
        # Final formula display
        formula = MathTex(r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}").scale(0.7)
        formula.to_edge(DOWN, buff=0.5)
        self.play(Write(formula))
        self.wait(2)
