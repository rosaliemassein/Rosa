from manim import *

class SemanticEmbedding(Scene):
    def construct(self):
        # 1. Background Grid representing the conceptual space
        plane = NumberPlane(
            x_range=[-7, 7, 1],
            y_range=[-4, 4, 1],
            background_line_style={
                "stroke_color": BLUE_D,
                "stroke_width": 1,
                "stroke_opacity": 0.3
            }
        )
        self.add(plane)

        # 2. Setup coordinates for concepts
        dl_pos = np.array([2, 1, 0])
        nn_pos = np.array([2.5, 0.2, 0])
        rb_pos = np.array([-3, -1.5, 0])

        # 3. Floating text labels
        dl_label = MathTex(r"\text{Deep Learning}").scale(0.8).set_color(BLUE).move_to(dl_pos + UP * 0.5)
        nn_label = MathTex(r"\text{Neural Networks}").scale(0.8).set_color(GREEN).move_to(nn_pos + DOWN * 0.5)
        rb_label = MathTex(r"\text{Robotics}").scale(0.8).set_color(RED).move_to(rb_pos + UP * 0.5)

        # 4. Animate text labels
        self.play(
            Write(dl_label),
            Write(nn_label),
            Write(rb_label),
            run_time=2
        )
        self.wait(1)

        # 5. Transform labels into Dots
        dl_dot = Dot(dl_pos, color=BLUE)
        nn_dot = Dot(nn_pos, color=GREEN)
        rb_dot = Dot(rb_pos, color=RED)

        self.play(
            ReplacementTransform(dl_label, dl_dot),
            ReplacementTransform(nn_label, nn_dot),
            ReplacementTransform(rb_label, rb_dot)
        )
        self.wait(0.5)

        # 6. Show distances using Lines (replacing DashedLine to ensure compatibility)
        # Related items have a shorter line
        line_close = Line(dl_dot.get_center(), nn_dot.get_center(), color=YELLOW)
        # Unrelated items have a dimmer/longer line
        line_far = Line(dl_dot.get_center(), rb_dot.get_center(), color=WHITE).set_opacity(0.3)

        self.play(Create(line_close))
        self.play(Create(line_far))
        self.wait(2)

        # 7. Transition to Formula
        self.play(
            FadeOut(line_close), FadeOut(line_far),
            FadeOut(dl_dot), FadeOut(nn_dot), FadeOut(rb_dot)
        )

        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}").to_edge(UP)
        self.play(Write(formula))
        self.wait(1)

        # 8. Vector merging for "Machine Learning"
        origin = ORIGIN + DOWN * 1
        m_vec_end = origin + LEFT * 2 + UP * 2
        l_vec_end = origin + RIGHT * 2 + UP * 1
        avg_vec_end = (m_vec_end + l_vec_end) / 2

        # Create arrows and labels
        v_machine = Arrow(origin, m_vec_end, buff=0, color=BLUE)
        v_learning = Arrow(origin, l_vec_end, buff=0, color=GREEN)
        v_avg = Arrow(origin, avg_vec_end, buff=0, color=YELLOW)

        t_machine = MathTex(r"\text{Machine}").next_to(m_vec_end, UP).scale(0.7)
        t_learning = MathTex(r"\text{Learning}").next_to(l_vec_end, UP).scale(0.7)
        t_avg = MathTex(r"\text{Machine Learning (Average)}").next_to(avg_vec_end, UP).scale(0.8).set_color(YELLOW)

        # Show individual components
        self.play(Create(v_machine), Write(t_machine))
        self.play(Create(v_learning), Write(t_learning))
        self.wait(1)

        # Merge them into the average
        self.play(
            ReplacementTransform(v_machine, v_avg),
            ReplacementTransform(v_learning, v_avg),
            ReplacementTransform(t_machine, t_avg),
            ReplacementTransform(t_learning, t_avg),
            run_time=2
        )
