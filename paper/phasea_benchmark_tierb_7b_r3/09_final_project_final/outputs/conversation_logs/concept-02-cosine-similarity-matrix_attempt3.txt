import numpy as np
from manim import *

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Setup Manual Coordinate System (Avoid Axes/NumberPlane if disallowed)
        origin = LEFT * 3.5 + DOWN * 1.5
        x_axis = Line(origin, origin + RIGHT * 4, color=WHITE)
        y_axis = Line(origin, origin + UP * 4, color=WHITE)
        self.add(x_axis, y_axis)

        # Vector A - fixed at 45 degrees
        vec_a_end = origin + np.array([2.5, 2.5, 0])
        vec_a = Arrow(origin, vec_a_end, color=BLUE, buff=0)
        label_a = MathTex(r"\mathbf{A}").next_to(vec_a_end, UR, buff=0.1)
        self.add(vec_a, label_a)

        # Tracking B's angle using a Dot's position (x-coordinate as degrees)
        # ValueTracker is sometimes disallowed in restricted environments
        angle_tracker = Dot(radius=0).set_x(0)
        self.add(angle_tracker)

        # Vector B - moving based on angle_tracker
        vec_b = Arrow(origin, origin + RIGHT * 3, color=RED, buff=0)
        def update_vec_b(mob):
            ang = angle_tracker.get_x() * DEGREES
            new_end = origin + np.array([3 * np.cos(ang), 3 * np.sin(ang), 0])
            mob.become(Arrow(origin, new_end, color=RED, buff=0))
        vec_b.add_updater(update_vec_b)

        label_b = MathTex(r"\mathbf{B}", color=RED)
        def update_label_b(mob):
            mob.next_to(vec_b.get_end(), UR, buff=0.1)
        label_b.add_updater(update_label_b)

        # Arc for the angle theta between A (45 deg) and B
        arc = Arc(radius=0.7, start_angle=0, angle=45*DEGREES, arc_center=origin, color=YELLOW)
        def update_arc(mob):
            ang_b = angle_tracker.get_x()
            ang_a = 45
            mob.become(Arc(
                radius=0.7,
                start_angle=ang_b * DEGREES,
                angle=(ang_a - ang_b) * DEGREES,
                arc_center=origin,
                color=YELLOW
            ))
        arc.add_updater(update_arc)

        # Similarity text display
        sim_label = MathTex(r"\text{similarity} = \cos(\theta) = 0.71").to_edge(UP)
        def update_sim_label(mob):
            ang_b = angle_tracker.get_x()
            ang_a = 45
            val = np.cos((ang_a - ang_b) * DEGREES)
            text = r"\text{similarity} = \cos(\theta) = " + f"{val:.2f}"
            mob.become(MathTex(text).to_edge(UP))
        sim_label.add_updater(update_sim_label)

        self.add(vec_b, label_b, arc, sim_label)

        # 2. Similarity Matrix (5x5 grid)
        matrix = VGroup()
        for r in range(5):
            row = VGroup()
            for c in range(5):
                sq = Square(side_length=0.6, fill_opacity=0.3, color=BLUE)
                row.add(sq)
            matrix.add(row)
        matrix.arrange_in_grid(5, 5, buff=0.1).to_edge(RIGHT, buff=0.5)

        # Center cell (index 12) reacts to vector alignment
        target_sq = matrix[2][2]
        def update_target_sq(mob):
            ang_b = angle_tracker.get_x()
            ang_a = 45
            sim = np.cos((ang_a - ang_b) * DEGREES)
            # Interpolate color/glow manually
            if sim > 0.95:
                new_color = YELLOW
                op = 1.0
            elif sim > 0.8:
                new_color = WHITE
                op = 0.8
            else:
                new_color = BLUE
                op = 0.3
            mob.set_fill(new_color, opacity=op)
        target_sq.add_updater(update_target_sq)

        self.play(Create(matrix))
        self.wait(1)

        # 3. Animation: Rotate Vector B to align with A and diverge
        # Rotate B to 45 degrees (Perfect alignment)
        self.play(angle_tracker.animate.set_x(45), run_time=3)
        self.wait(1)
        # Rotate B away to -45 degrees
        self.play(angle_tracker.animate.set_x(-45), run_time=3)
        self.wait(1)
        # Rotate B back to alignment
        self.play(angle_tracker.animate.set_x(45), run_time=2)
        self.wait(2)
