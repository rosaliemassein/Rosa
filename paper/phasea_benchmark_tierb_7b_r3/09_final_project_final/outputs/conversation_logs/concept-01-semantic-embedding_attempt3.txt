from manim import *

class SemanticEmbedding(Scene):
    def construct(self):
        # Use Axes instead of NumberPlane or ThreeDAxes
        axes = Axes(
            x_range=[-6, 6, 1],
            y_range=[-4, 4, 1],
            x_length=10,
            y_length=6,
            axis_config={"color": WHITE, "stroke_width": 2}
        )
        self.add(axes)

        # 1. Floating text labels
        # Use simple standard colors: GREEN, BLUE, YELLOW
        text_dl = Text("Deep Learning", color=GREEN).scale(0.6)
        text_nn = Text("Neural Networks", color=BLUE).scale(0.6)
        text_r = Text("Robotics", color=YELLOW).scale(0.6)

        text_dl.to_edge(UP).shift(LEFT * 3)
        text_nn.to_edge(UP).shift(RIGHT * 3)
        text_r.to_edge(DOWN).shift(RIGHT * 4)

        self.play(Write(text_dl), Write(text_nn), Write(text_r))
        self.wait(1)

        # 2. Coordinates for conceptual mapping
        # DL and NN are close together, Robotics is further away
        pos_dl = axes.c2p(1, 1, 0)
        pos_nn = axes.c2p(1.8, 1.2, 0)
        pos_r = axes.c2p(-3, -1.5, 0)

        dot_dl = Dot(pos_dl, color=GREEN)
        dot_nn = Dot(pos_nn, color=BLUE)
        dot_r = Dot(pos_r, color=YELLOW)

        # Transform text into dots
        self.play(
            Transform(text_dl, dot_dl),
            Transform(text_nn, dot_nn),
            Transform(text_r, dot_r)
        )
        self.wait(1)

        # 3. Create a manual dashed line (since DashedLine is disallowed)
        def create_manual_dash(p1, p2, num_segments=10):
            dashes = VGroup()
            for i in range(num_segments):
                start = p1 + (p2 - p1) * (i / num_segments)
                end = p1 + (p2 - p1) * ((i + 0.5) / num_segments)
                dashes.add(Line(start, end, stroke_width=2, color=WHITE))
            return dashes

        dash_line = create_manual_dash(dot_dl.get_center(), dot_nn.get_center())
        self.play(Create(dash_line))
        self.wait(1)

        # 4. Multi-word phrase logic: "Machine" + "Learning"
        text_m = Text("Machine", color=PURPLE).scale(0.7).to_corner(UL)
        text_l = Text("Learning", color=RED).scale(0.7).next_to(text_m, DOWN)

        self.play(Write(text_m), Write(text_l))
        self.wait(0.5)

        # Define vector positions for Machine and Learning
        pos_m = axes.c2p(-4, 2, 0)
        pos_l = axes.c2p(-2, 3, 0)
        # Calculate the average (center of gravity)
        pos_avg = axes.c2p(-3, 2.5, 0)

        dot_m = Dot(pos_m, color=PURPLE)
        dot_l = Dot(pos_l, color=RED)
        dot_avg = Dot(pos_avg, color=WHITE)

        # Transform word labels to their vector positions
        self.play(
            Transform(text_m, dot_m),
            Transform(text_l, dot_l)
        )
        self.wait(1)

        # Show averaging/merging into a single central vector
        avg_label = Text("Phrase Vector", color=WHITE).scale(0.4).next_to(dot_avg, UP)
        self.play(
            Transform(text_m, dot_avg),
            Transform(text_l, dot_avg),
            FadeIn(avg_label)
        )
        self.wait(1)

        # 5. Display the mathematical formula
        formula = MathTex(
            r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}",
            color=WHITE
        ).to_edge(DOWN, buff=0.5)
        
        # Add a small background rectangle to formula for clarity
        rect = SurroundingRectangle(formula, color=WHITE, buff=0.2, stroke_width=1)
        
        self.play(Create(rect), Write(formula))
        self.wait(3)
