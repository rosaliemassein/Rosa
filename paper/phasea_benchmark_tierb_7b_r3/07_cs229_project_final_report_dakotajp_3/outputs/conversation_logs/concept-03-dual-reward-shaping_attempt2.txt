from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Environment Setup
        board = Rectangle(width=9, height=5, color=GREY_B)
        target_zone = Circle(radius=0.7, color=BLUE_B, fill_opacity=0.3).shift(RIGHT * 3.5)
        target_label = Text("Target Zone", font_size=20).next_to(target_zone, UP)
        
        # 2. Pegs Setup
        pegs = VGroup(*[
            Dot(color=BLUE_D).move_to(LEFT * 3.5 + UP * (i - 2) * 0.8)
            for i in range(5)
        ])
        
        # 3. Centroid Marker
        # We calculate the average position of all pieces
        def get_centroid_pos():
            positions = [p.get_center() for p in pegs]
            return sum(positions) / len(positions)

        centroid = Dot(color=YELLOW_C)
        centroid.move_to(get_centroid_pos())
        centroid_text = Text("Centroid", font_size=16, color=YELLOW_C)
        
        # 4. Connecting Line (Replacing DashedLine which was disallowed)
        # Using a solid line with lower opacity to represent the gravitational pull
        line = Line(color=WHITE, stroke_width=1.5, stroke_opacity=0.6)
        
        # 5. Reward Tracking via Text and Updaters (Replacing DecimalNumber and ValueTracker)
        reward_label = Text("Total Reward:", font_size=24).to_edge(UP).shift(LEFT * 0.5)
        reward_val_text = Text("0.00", font_size=24).next_to(reward_label, RIGHT)
        
        # 6. Updaters for real-time visualization
        centroid.add_updater(lambda m: m.move_to(get_centroid_pos()))
        centroid_text.add_updater(lambda m: m.next_to(centroid, DOWN, buff=0.1))
        
        def update_line(l):
            l.put_start_and_end_on(centroid.get_center(), target_zone.get_center())
        line.add_updater(update_line)
        
        def update_reward_display(m):
            # Distance from Centroid to Target
            c_pos = centroid.get_center()
            t_pos = target_zone.get_center()
            dist = np.linalg.norm(c_pos - t_pos)
            
            # Count pegs inside target radius
            pegs_in = 0
            for p in pegs:
                if np.linalg.norm(p.get_center() - t_pos) < 0.7:
                    pegs_in += 1
            
            # Formula: R = w1 * pegs_in^2 + w2 * dist^-1
            score = (pegs_in ** 2) * 8 + (4.0 / (dist + 0.1))
            # Update text content by creating a new Text object and using become
            new_text = Text(f"{score:.2f}", font_size=24).move_to(m.get_center())
            m.become(new_text)
            
        reward_val_text.add_updater(update_reward_display)
        
        # 7. Add Mobjects to Scene
        self.add(board, target_zone, target_label, pegs, centroid, centroid_text, line, reward_label, reward_val_text)
        
        # 8. Animation Sequence
        for i in range(len(pegs)):
            # Define destination point inside target zone with some random offset
            dest_offset = np.array([
                np.random.uniform(-0.3, 0.3),
                np.random.uniform(-0.3, 0.3),
                0
            ])
            dest_point = target_zone.get_center() + dest_offset
            
            # Move peg (rate_func=lambda t: t replaces the undefined 'linear')
            self.play(
                pegs[i].animate.move_to(dest_point),
                run_time=1.4,
                rate_func=lambda t: t
            )
            
            # Squared Bonus Visual Pop-up
            bonus_val = (i + 1) ** 2
            bonus_pop = Text(f"{i+1}^2 Bonus!", font_size=28, color=GOLD).next_to(target_zone, DOWN)
            self.play(FadeIn(bonus_pop, shift=UP * 0.2), run_time=0.4)
            self.play(FadeOut(bonus_pop), run_time=0.3)
            
        self.wait(2)
