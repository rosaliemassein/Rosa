from manim import *

class OpponentAbstraction(Scene):
    def construct(self):
        # 1. Formula reference
        formula = MathTex(r"S \in \mathbb{B}^{n \times n \times 3}").to_edge(UP)
        self.add(formula)

        # 2. 2-Player Representation
        # Representing as layers/grids to honor the "3 layers" encoding concept
        layer_p1 = VGroup(*[Square(side_length=0.2) for _ in range(9)]).arrange_in_grid(3, 3, buff=0.05)
        layer_p1.set_fill(BLUE, opacity=0.7).set_stroke(BLUE_A, 1)
        
        layer_p2 = VGroup(*[Square(side_length=0.2) for _ in range(9)]).arrange_in_grid(3, 3, buff=0.05)
        layer_p2.set_fill(RED, opacity=0.7).set_stroke(RED_A, 1)

        p1_group = VGroup(layer_p1, Text("Me", font_size=20).next_to(layer_p1, DOWN)).shift(LEFT * 2)
        p2_group = VGroup(layer_p2, Text("Opponent", font_size=20).next_to(layer_p2, DOWN)).shift(RIGHT * 2)

        self.play(FadeIn(p1_group), FadeIn(p2_group))
        self.wait(1)

        # 3. 6-Player Representation
        # 5 different opponent colors (using safe standard Manim colors)
        colors = [GREEN, PURPLE, ORANGE, GOLD, TEAL]
        multi_opponents = VGroup()
        for i, color in enumerate(colors):
            opp = VGroup(*[Square(side_length=0.15) for _ in range(9)]).arrange_in_grid(3, 3, buff=0.02)
            opp.set_fill(color, opacity=0.7).set_stroke(WHITE, 0.5)
            # Stack them slightly for visual effect
            opp.shift(RIGHT * 2.5 + UP * (1.2 - i * 0.6))
            multi_opponents.add(opp)

        multi_label = Text("5 Different Opponents", font_size=18).next_to(multi_opponents, RIGHT)

        self.play(
            ReplacementTransform(p2_group, multi_opponents),
            Write(multi_label)
        )
        self.wait(1)

        # 4. Merging into a single 'Red' intensity
        # Create a manual cross-out animation (avoiding the 'Cross' identifier)
        cross_lines = VGroup()
        for opp in multi_opponents:
            l1 = Line(opp.get_corner(UL), opp.get_corner(DR), color=RED, stroke_width=4)
            l2 = Line(opp.get_corner(UR), opp.get_corner(DL), color=RED, stroke_width=4)
            cross_lines.add(l1, l2)

        self.play(Create(cross_lines))
        self.wait(0.5)

        # Unified Opponent Layer (Red)
        unified_layer = VGroup(*[Square(side_length=0.2) for _ in range(9)]).arrange_in_grid(3, 3, buff=0.05)
        unified_layer.set_fill(RED, opacity=0.9).set_stroke(RED_A, 1).shift(RIGHT * 2)
        unified_label = Text("Unified Opponent Layer", font_size=20).next_to(unified_layer, DOWN)

        self.play(
            FadeOut(multi_opponents),
            FadeOut(multi_label),
            FadeOut(cross_lines),
            FadeIn(unified_layer),
            FadeIn(unified_label)
        )

        # 5. Check mark over unified layer (Manual drawing to ensure compatibility)
        check_mark = VGroup(
            Line(ORIGIN, RIGHT * 0.3 + DOWN * 0.3, color=GREEN, stroke_width=8),
            Line(RIGHT * 0.3 + DOWN * 0.3, RIGHT * 0.8 + UP * 0.5, color=GREEN, stroke_width=8)
        ).move_to(unified_layer.get_center() + RIGHT * 1.5)

        self.play(Write(check_mark))

        # 6. Final message
        final_text = Text(
            "Strategy scales to any number of opponents", 
            font_size=24, 
            color=LIGHT_GREY
        ).to_edge(DOWN)
        self.play(Write(final_text))
        
        self.wait(3)
