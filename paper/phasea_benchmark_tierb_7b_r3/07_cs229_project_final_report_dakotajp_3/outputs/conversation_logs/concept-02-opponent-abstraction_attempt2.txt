from manim import *

class OpponentAbstraction(Scene):
    def construct(self):
        # Voice-over context/Formula
        formula = MathTex(r"S \in \mathbb{B}^{n \times n \times 3}").to_edge(UP)
        self.add(formula)

        # Helper function for grid since 'Grid' is undefined
        def create_grid(rows, cols, cell_size=0.6):
            grid = VGroup()
            for r in range(rows):
                for c in range(cols):
                    sq = Square(side_length=cell_size)
                    sq.move_to(np.array([c * cell_size, -r * cell_size, 0]))
                    grid.add(sq)
            return grid.center()

        # 1. 2-player board representation
        grid_2p = create_grid(3, 3).to_edge(LEFT, buff=1.5)
        label_2p = Text("2-Player Game", font_size=24).next_to(grid_2p, UP)
        blue_dot_2p = Dot(grid_2p[4].get_center(), color=BLUE, radius=0.2)
        red_dot_2p = Dot(grid_2p[2].get_center(), color=RED, radius=0.2)

        self.play(Create(grid_2p), Write(label_2p))
        self.play(FadeIn(blue_dot_2p), FadeIn(red_dot_2p))
        self.wait(1)

        # 2. 6-player board representation
        grid_6p = create_grid(3, 3).to_edge(RIGHT, buff=1.5)
        label_6p = Text("6-Player Game", font_size=24).next_to(grid_6p, UP)
        blue_dot_6p = Dot(grid_6p[4].get_center(), color=BLUE, radius=0.2)
        
        # Using colors that are standard and avoiding PINK/MAGENTA
        opp_colors = [GREEN, PURPLE, YELLOW, ORANGE, TEAL]
        opp_indices = [0, 1, 2, 5, 8]
        opponents = VGroup(*[
            Dot(grid_6p[idx].get_center(), color=opp_colors[i], radius=0.2)
            for i, idx in enumerate(opp_indices)
        ])

        self.play(Create(grid_6p), Write(label_6p))
        self.play(FadeIn(blue_dot_6p), FadeIn(opponents))
        self.wait(1)

        # 3. Animate the 5 opponent colors merging into a single 'Red' intensity
        # This represents the "Opponent Layer" abstraction
        self.play(opponents.animate.set_color(RED))
        self.wait(0.5)

        # 4. Cross-out animation over individual identities (Manual Cross)
        crosses = VGroup()
        for dot in opponents:
            p = dot.get_center()
            l1 = Line(p + UL*0.15, p + DR*0.15, color=WHITE, stroke_width=3)
            l2 = Line(p + UR*0.15, p + DL*0.15, color=WHITE, stroke_width=3)
            crosses.add(VGroup(l1, l2))
        
        self.play(Create(crosses))
        self.wait(1)

        # 5. Check mark over the unified 'Opponent Layer' (Manual Check)
        check_mark = VGroup(
            Line(ORIGIN, RIGHT*0.2 + DOWN*0.2, color=GREEN, stroke_width=6),
            Line(RIGHT*0.2 + DOWN*0.2, RIGHT*0.5 + UP*0.4, color=GREEN, stroke_width=6)
        ).next_to(grid_6p, DOWN, buff=0.5)
        
        check_text = Text("Unified Opponent Layer", font_size=22, color=GREEN).next_to(check_mark, RIGHT)
        
        self.play(Create(check_mark), Write(check_text))
        self.wait(2)

        # Cleanup
        self.play(
            FadeOut(grid_2p), FadeOut(blue_dot_2p), FadeOut(red_dot_2p), FadeOut(label_2p),
            FadeOut(grid_6p), FadeOut(blue_dot_6p), FadeOut(opponents), FadeOut(label_6p),
            FadeOut(crosses), FadeOut(check_mark), FadeOut(check_text), FadeOut(formula)
        )
        self.wait(1)
