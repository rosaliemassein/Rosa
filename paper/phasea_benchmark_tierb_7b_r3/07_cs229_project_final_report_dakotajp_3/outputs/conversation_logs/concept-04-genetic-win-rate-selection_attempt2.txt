from manim import *

class GeneticWinRateSelection(Scene):
    def construct(self):
        # Formula reference at the top
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)")
        formula.to_edge(UP)
        self.add(formula)

        # Central Parent Node
        parent_node = VGroup(
            Circle(radius=0.5, color=WHITE),
            Text("Parent", font_size=20)
        ).shift(UP * 2)
        
        self.play(Create(parent_node))
        self.wait(0.5)

        # Child Configuration
        # index 1 (GOLD) is the winner
        child_configs = [
            {"color": BLUE, "rate": 0.45, "x": -4},
            {"color": GOLD, "rate": 0.88, "x": 0},
            {"color": RED, "rate": 0.62, "x": 4}
        ]

        children = VGroup()
        lines = VGroup()
        bars = VGroup()
        labels = VGroup()

        for i, config in enumerate(child_configs):
            # Child Circle Model
            c_node = VGroup(
                Circle(radius=0.4, color=config["color"]),
                Text(f"Model {i+1}", font_size=16)
            ).move_to(RIGHT * config["x"] + DOWN * 0.5)
            
            # Line from parent to child
            c_line = Line(parent_node.get_bottom(), c_node.get_top(), buff=0.1)
            
            # Win Rate Bar (Manual construction to avoid BarChart identifier error)
            b_height = config["rate"] * 2.5
            b_rect = Rectangle(
                height=b_height, 
                width=0.8, 
                fill_opacity=0.8, 
                fill_color=config["color"],
                stroke_color=WHITE
            )
            b_rect.next_to(c_node, DOWN, buff=1.0)
            
            # Win Rate Percentage Text
            b_label = Text(f"{int(config['rate']*100)}%", font_size=20).next_to(b_rect, DOWN)
            
            children.add(c_node)
            lines.add(c_line)
            bars.add(b_rect)
            labels.add(b_label)

        # Animate children branching out
        self.play(
            LaggedStart(*[Create(l) for l in lines], lag_ratio=0.2),
            LaggedStart(*[Create(c) for c in children], lag_ratio=0.2)
        )
        self.wait(0.5)

        # Animate bars growing to show win rates
        self.play(
            LaggedStart(*[GrowFromEdge(b, DOWN) for b in bars], lag_ratio=0.3),
            LaggedStart(*[Write(l) for l in labels], lag_ratio=0.3),
            run_time=2
        )
        self.wait(1)

        # Highlight the child with the highest win rate
        selection_ring = SurroundingRectangle(bars[1], color=GOLD, buff=0.1)
        winner_tag = Text("WINNER", color=GOLD, font_size=24).next_to(selection_ring, UP)
        
        self.play(Create(selection_ring), Write(winner_tag))
        self.play(children[1].animate.scale(1.2), bars[1].animate.set_fill(opacity=1.0))
        self.wait(1)

        # Evolutionary Step: Fade out others and move winner to parent position
        others = VGroup(
            parent_node, lines, 
            children[0], children[2], 
            bars[0], bars[2], 
            labels[0], labels[2], 
            selection_ring, winner_tag
        )
        
        self.play(
            FadeOut(others),
            bars[1].animate.fade(1),
            labels[1].animate.fade(1),
            children[1].animate.move_to(parent_node.get_center()).scale(1/1.2)
        )
        
        # Quick indication of next generation cycle
        new_parent = children[1]
        cycle_text = Text("Next Generation...", font_size=24, color=GOLD).next_to(new_parent, DOWN)
        self.play(Write(cycle_text))
        
        # Fast branch out to show recursion
        dots = VGroup(*[Dot(new_parent.get_bottom() + DOWN + RIGHT * (i-1) * 2.5) for i in range(3)])
        new_lines = VGroup(*[Line(new_parent.get_bottom(), d.get_center()) for d in dots])
        
        self.play(Create(new_lines), Create(dots), run_time=0.8)
        self.wait(2)
