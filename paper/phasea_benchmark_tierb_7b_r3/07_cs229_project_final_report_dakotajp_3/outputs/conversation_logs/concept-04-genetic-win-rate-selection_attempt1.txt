from manim import *
import numpy as np

class GeneticSelection(Scene):
    def construct(self):
        # Formula Reference from JSON
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)")
        formula.to_edge(UP)
        self.add(formula)

        # 1. Central 'Parent' node
        parent_circle = Circle(radius=0.5, color=BLUE, fill_opacity=0.5)
        parent_label = Text("Parent", font_size=24).next_to(parent_circle, UP)
        parent = VGroup(parent_circle, parent_label).move_to(UP * 1.5)
        self.play(Create(parent))
        self.wait(0.5)

        # 2. Three 'Child' nodes branching out
        win_rates = [0.25, 0.85, 0.45]
        children = VGroup()
        child_labels = VGroup()
        branch_lines = VGroup()
        
        positions = [[-4, -0.5, 0], [0, -0.5, 0], [4, -0.5, 0]]
        
        for i in range(3):
            child = Square(side_length=0.8, color=BLUE, fill_opacity=0.3)
            child.move_to(positions[i])
            label = Text(f"Child {i+1}", font_size=20).next_to(child, UP, buff=0.1)
            line = Line(parent_circle.get_bottom(), child.get_top(), buff=0.1)
            children.add(child)
            child_labels.add(label)
            branch_lines.add(line)

        self.play(
            Create(branch_lines),
            Create(children),
            Write(child_labels)
        )

        # 3. Representing 'Win Rate' bars below each child
        bars = VGroup()
        for i, rate in enumerate(win_rates):
            # Create a bar that represents the win rate
            bar = Rectangle(
                width=0.6, 
                height=rate * 2.5, 
                color=GREEN, 
                fill_opacity=0.8
            )
            # Position bar base relative to child
            bar.move_to(children[i].get_bottom() + DOWN * 2.5, DOWN)
            
            rate_text = Text(f"{int(rate*100)}%", font_size=18).next_to(bar, DOWN, buff=0.1)
            bars.add(VGroup(bar, rate_text))

        # Animate the bars growing to different heights
        # Using GrowFromPoint since GrowFromEdge was flagged
        self.play(
            *[GrowFromPoint(b[0], b[0].get_bottom()) for b in bars],
            *[Write(b[1]) for b in bars],
            run_time=1.5
        )
        self.wait(1)

        # 4. Highlight the child with the highest bar in gold
        best_idx = 1 # 85% is highest
        self.play(
            bars[best_idx][0].animate.set_color(GOLD),
            children[best_idx].animate.set_color(GOLD),
            Indicate(children[best_idx], color=GOLD, scale_factor=1.2)
        )
        self.wait(1)

        # 5. FadeOut on parent and other children
        others = VGroup(
            parent, branch_lines,
            children[0], children[2],
            child_labels[0], child_labels[1], child_labels[2],
            bars[0], bars[2], bars[1][1]
        )
        
        # 6. Move the gold child to the center to become the new 'Parent'
        new_parent_center = [0, 1.5, 0]
        self.play(
            FadeOut(others),
            children[best_idx].animate.move_to(new_parent_center),
            bars[best_idx][0].animate.scale(0).move_to(new_parent_center)
        )
        
        # Transitioning to the next generation
        final_circle = Circle(radius=0.5, color=BLUE, fill_opacity=0.5).move_to(new_parent_center)
        final_label = Text("New Parent", font_size=24).next_to(final_circle, UP)
        
        self.play(
            Transform(children[best_idx], final_circle),
            Write(final_label)
        )
        
        # Quick indication of repeated evolution
        flash = SurroundingRectangle(VGroup(children[best_idx], final_label), color=GOLD, buff=0.2)
        self.play(Create(flash))
        self.play(FadeOut(flash))
        
        self.wait(2)
