from manim import *
import math

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Custom Coordinate Mapping (to avoid Axes)
        def get_coords(z, p):
            # z logic from -6 to 6 maps to scene x from 0 to 6
            # p logic from 0 to 1 maps to scene y from -2 to 2
            return [3.0 + z * 0.5, -2.0 + p * 4.0, 0]

        def lin_rate(t):
            return t

        # 2. Manual Axes
        x_ax = Line(start=[0, -2, 0], end=[6, -2, 0], color=WHITE)
        y_ax = Line(start=[3, -2, 0], end=[3, 2, 0], color=WHITE)
        self.add(x_ax, y_ax)
        
        # 3. Sigmoid Curve (constructed manually)
        sig_curve = VMobject(color=YELLOW)
        pts = []
        for i in range(51):
            zv = -6.0 + i * 12.0 / 50.0
            pv = 1.0 / (1.0 + math.exp(-zv))
            pts.append(get_coords(zv, pv))
        sig_curve.set_points_as_corners(pts)
        self.add(sig_curve)
        
        # 4. Feature Trackers (using invisible dots to avoid ValueTracker)
        val1 = Dot(point=[0, 0, 0], fill_opacity=0, stroke_opacity=0)
        val2 = Dot(point=[0, 0, 0], fill_opacity=0, stroke_opacity=0)
        self.add(val1, val2)
        
        # 5. Bars (Left side)
        b1 = Rectangle(width=0.6, height=1.0, fill_opacity=0.7, color=RED)
        b1.move_to([-4, -2, 0], aligned_edge=DOWN)
        b2 = Rectangle(width=0.6, height=1.0, fill_opacity=0.7, color=BLUE)
        b2.move_to([-1.5, -2, 0], aligned_edge=DOWN)
        
        def up_b1(m):
            h = val1.get_center()[1] + 1.5
            m.stretch_to_fit_height(max(0.1, h))
            m.move_to([-4, -2, 0], aligned_edge=DOWN)
            
        def up_b2(m):
            h = val2.get_center()[1] + 1.5
            m.stretch_to_fit_height(max(0.1, h))
            m.move_to([-1.5, -2, 0], aligned_edge=DOWN)
            
        b1.add_updater(up_b1)
        b2.add_updater(up_b2)
        self.add(b1, b2)
        
        # 6. Prediction Dot on Curve
        on_curve_dot = Dot(color=WHITE)
        def up_dot(m):
            v1_v = val1.get_center()[1]
            v2_v = val2.get_center()[1]
            # Map tracker movement to logic Z values
            z_sum = (v1_v * 2.5) + (v2_v * 2.5)
            p_val = 1.0 / (1.0 + math.exp(-z_sum))
            m.move_to(get_coords(z_sum, p_val))
            
        on_curve_dot.add_updater(up_dot)
        self.add(on_curve_dot)
        
        # 7. Labels and Formula
        l1 = Text("1-Day T-CSP", font_size=20).next_to(b1, DOWN)
        l2 = Text("30-Day T-CSP", font_size=20).next_to(b2, DOWN)
        form = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=32
        ).to_edge(UP)
        self.add(l1, l2, form)
        
        # 8. Animation Sequence
        self.play(val1.animate.shift(UP * 1.0), run_time=2, rate_func=lin_rate)
        self.wait(0.5)
        self.play(val2.animate.shift(DOWN * 0.5), run_time=1, rate_func=lin_rate)
        self.wait(0.5)
        self.play(
            val1.animate.shift(DOWN * 1.5), 
            val2.animate.shift(UP * 2.0), 
            run_time=2, 
            rate_func=lin_rate
        )
        self.wait(1)
        
        # 9. Formula Vector Transformations
        f_vec = MathTex(r"\mathbf{X} = \begin{pmatrix} X_{1d} \\ X_{30d} \end{pmatrix}").move_to([-2.7, 1, 0])
        w_vec = MathTex(r"\mathbf{\beta} = \begin{pmatrix} \beta_1 \\ \beta_{30} \end{pmatrix}").move_to([-2.7, 1, 0])
        
        self.play(FadeIn(f_vec))
        self.wait(1)
        self.play(ReplacementTransform(f_vec, w_vec))
        self.wait(2)
