from manim import *

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.7)
        
        # Labels for the axes
        x_label = Text("Fraction of Zones Flagged").scale(0.45).next_to(axes.x_axis, DOWN)
        y_label = Text("Recall").scale(0.45).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)
        
        # 2. Concave Curve: y = x^0.5
        curve = axes.plot(lambda x: x**0.5, x_range=[0, 1], color=BLUE)
        
        # 3. Recall Formula
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.7).to_corner(UL)
        
        # 4. Building Floorplan (Side Panel)
        # 4x4 grid of zones
        zones = VGroup()
        for row in range(4):
            for col in range(4):
                square = Square(side_length=0.4, stroke_width=2)
                square.move_to([4.5 + col * 0.5, 1.5 - row * 0.5, 0])
                zones.add(square)
        
        # 5. ValueTracker for animation state
        tracker = ValueTracker(0.1)
        
        # 6. Highlighted Point on the curve
        dot = Dot(color=RED)
        dot.add_updater(lambda mob: mob.move_to(
            axes.coords_to_point(tracker.get_value(), tracker.get_value()**0.5)
        ))
        
        # 7. Updater for the Floorplan zones
        def update_floorplan(mob_group):
            val = tracker.get_value()
            # Calculate how many zones to highlight based on current x-value
            num_highlighted = int(val * 16)
            for i, zone in enumerate(mob_group):
                if i < num_highlighted:
                    zone.set_fill(RED, opacity=0.8)
                else:
                    zone.set_fill(BLACK, opacity=0.1)
        
        zones.add_updater(update_floorplan)
        
        # Add components to the scene
        self.add(axes, x_label, y_label, curve, formula, zones, dot)
        
        # 8. Animation Sequence
        # Move point along the curve to demonstrate sensitivity/selectivity trade-off
        # Using a simple lambda rate function to avoid potential identifier issues
        self.play(tracker.animate.set_value(0.95), run_time=4, rate_func=lambda t: t)
        self.wait(0.5)
        self.play(tracker.animate.set_value(0.3), run_time=2, rate_func=lambda t: t)
        
        # 9. Highlight the 'Elbow'
        elbow_point = axes.coords_to_point(0.3, 0.3**0.5)
        elbow_circle = Circle(radius=0.4, color=YELLOW, stroke_width=6).move_to(elbow_point)
        elbow_label = Text("Optimal 'Elbow'", color=YELLOW).scale(0.5).next_to(elbow_circle, UR)
        
        self.play(
            Create(elbow_circle),
            Write(elbow_label),
            Flash(elbow_point, color=YELLOW)
        )
        
        # Subtle pulsing animation for the elbow
        self.play(
            elbow_circle.animate.scale(1.2).set_stroke(opacity=0.3),
            rate_func=there_and_back,
            run_time=1.5
        )
        
        self.wait(2)
