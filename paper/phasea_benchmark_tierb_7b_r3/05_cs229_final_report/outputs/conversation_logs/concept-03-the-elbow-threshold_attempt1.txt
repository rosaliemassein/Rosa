from manim import *
import math

class ElbowThreshold(Scene):
    def construct(self):
        # Setup Axes following Pattern #3
        # Using specific values to avoid potential linter issues with ranges
        axes = Axes(
            x_range=[0, 1, 0.2],
            y_range=[0, 1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"color": BLUE}
        ).shift(LEFT * 2.5)
        
        # Labels for the axes
        x_lab = Text("Fraction of Zones Flagged", font_size=20).next_to(axes.x_axis, DOWN)
        y_lab = Text("Recall", font_size=20).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)
        
        # Plotting the concave curve following Pattern #3
        # Using a named function to avoid lambda identifier issues
        def curve_logic(x_val):
            # A simple concave function: 1 - e^(-5x)
            return 1 - math.exp(-5 * x_val)
            
        curve = axes.plot(curve_logic, x_range=[0, 1], color=YELLOW)

        # ValueTracker for movement following Pattern #2
        # We use a tracker to move the point and update the floorplan
        threshold_tracker = ValueTracker(0.1)

        # The Moving Point
        dot = Dot(color=RED)
        
        # Defining updaters as named functions to avoid 'd' or 'x' undefined errors
        def update_dot_position(mobject):
            current_x = threshold_tracker.get_value()
            current_y = curve_logic(current_x)
            mobject.move_to(axes.c2p(current_x, current_y))

        dot.add_updater(update_dot_position)

        # Floorplan: A 4x4 grid of zones
        zones = VGroup()
        for i in range(16):
            sq = Square(side_length=0.7, stroke_width=2, color=WHITE)
            zones.add(sq)
        zones.arrange_in_grid(rows=4, cols=4, buff=0.1).to_edge(RIGHT, buff=1)
        
        plan_label = Text("Zone Status", font_size=24).next_to(zones, UP)

        # Updater for the zones to highlight them based on the tracker
        def update_zones(mobject_group):
            val = threshold_tracker.get_value()
            # As val goes 0 to 1, highlight 0 to 16 zones
            num_to_highlight = int(val * 16)
            for index, square in enumerate(mobject_group):
                if index < num_to_highlight:
                    square.set_fill(GREEN, opacity=0.8)
                else:
                    square.set_fill(BLACK, opacity=0)

        zones.add_updater(update_zones)

        # Elbow Highlight (Visual 'Sweet Spot')
        elbow_val = 0.35
        elbow_coord = axes.c2p(elbow_val, curve_logic(elbow_val))
        elbow_circle = Circle(radius=0.4, color=GOLD).move_to(elbow_coord)
        elbow_tag = Text("Elbow", font_size=20, color=GOLD).next_to(elbow_circle, UR)

        # Animation Sequence
        self.add(axes, x_lab, y_lab, curve)
        self.play(Create(zones), Write(plan_label))
        self.add(dot)
        
        # 1. Show low threshold (Few zones flagged, low recall)
        self.play(threshold_tracker.animate.set_value(0.1), run_time=1)
        self.wait(1)
        
        # 2. Show high threshold (Many zones flagged, high recall, inefficient)
        self.play(threshold_tracker.animate.set_value(0.9), run_time=3)
        self.wait(1)
        
        # 3. Move to the Elbow (The Sweet Spot)
        self.play(threshold_tracker.animate.set_value(elbow_val), run_time=2)
        self.play(Create(elbow_circle), Write(elbow_tag))
        self.play(Flash(elbow_circle, color=GOLD))
        
        # Formula display
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=30).to_corner(UL)
        self.play(Write(formula))
        
        self.wait(2)

        # Remove updaters for clean exit
        dot.remove_updater(update_dot_position)
        zones.remove_updater(update_zones)
