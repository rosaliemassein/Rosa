from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Setup Trackers for inputs (X_1d and X_30d)
        # Using the ValueTracker pattern from Example 2
        x1_track = ValueTracker(0.0)
        x30_track = ValueTracker(0.0)
        
        # Logistic weights
        b0 = -3.0
        b1 = 0.8
        b30 = 0.4

        # 2. Coordinate System and Curve
        # Using the Axes and plot pattern from Example 3
        axes = Axes(
            x_range=[-6, 6, 2],
            y_range=[0, 1.1, 0.5],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5)
        
        # Define the sigmoid function for the plot
        def sigmoid_func(x):
            return 1 / (1 + np.exp(-x))
            
        curve = axes.plot(sigmoid_func, color=BLUE)
        curve_label = MathTex(r"P(y=1)").scale(0.8).next_to(axes, UP, buff=0.2)

        # 3. Sliding Bars (Left side)
        # Rectangles to represent features
        bar_bg_1 = Rectangle(width=0.6, height=2.0, color=GRAY, fill_opacity=0.2)
        bar_bg_1.to_edge(LEFT, buff=1.0).shift(UP * 0.5)
        
        bar_bg_30 = Rectangle(width=0.6, height=2.0, color=GRAY, fill_opacity=0.2)
        bar_bg_30.next_to(bar_bg_1, RIGHT, buff=1.0)
        
        label_1 = MathTex(r"X_{1d}").scale(0.8).next_to(bar_bg_1, DOWN)
        label_30 = MathTex(r"X_{30d}").scale(0.8).next_to(bar_bg_30, DOWN)
        
        # The moving bars
        bar_1 = Rectangle(width=0.6, height=0.1, color=BLUE, fill_opacity=0.8)
        bar_1.move_to(bar_bg_1.get_bottom(), aligned_edge=DOWN)
        
        bar_30 = Rectangle(width=0.6, height=0.1, color=GREEN, fill_opacity=0.8)
        bar_30.move_to(bar_bg_30.get_bottom(), aligned_edge=DOWN)

        # 4. Dot on the curve
        dot = Dot(color=RED)

        # 5. Updaters (Replacing always_redraw with explicit updaters as in Example 2)
        def update_bar1(mob):
            val = x1_track.get_value()
            new_height = max(0.1, 0.1 + (val / 5.0) * 1.9)
            mob.stretch_to_fit_height(new_height)
            mob.move_to(bar_bg_1.get_bottom(), aligned_edge=DOWN)

        def update_bar30(mob):
            val = x30_track.get_value()
            new_height = max(0.1, 0.1 + (val / 5.0) * 1.9)
            mob.stretch_to_fit_height(new_height)
            mob.move_to(bar_bg_30.get_bottom(), aligned_edge=DOWN)

        def update_dot(mob):
            z = b0 + b1 * x1_track.get_value() + b30 * x30_track.get_value()
            p = sigmoid_func(z)
            mob.move_to(axes.c2p(z, p))

        bar_1.add_updater(update_bar1)
        bar_30.add_updater(update_bar30)
        dot.add_updater(update_dot)

        # 6. Formula
        # Logistic formula displayed at the bottom
        formula = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=34
        ).to_edge(DOWN, buff=0.5)

        # 7. Animation Sequence
        self.add(axes, curve, curve_label, bar_bg_1, bar_bg_30, bar_1, bar_30, label_1, label_30, dot)
        self.play(Write(formula))
        self.wait(1)
        
        # Animate the sliding bars and the dot on the sigmoid
        self.play(
            x1_track.animate.set_value(5.0),
            run_time=2,
            rate_func=linear
        )
        self.wait(0.5)
        
        self.play(
            x30_track.animate.set_value(5.0),
            run_time=2,
            rate_func=linear
        )
        self.wait(1)
        
        # Show scaling by transforming the formula
        formula_values = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(-3 + 0.8(5) + 0.4(5))}}",
            font_size=34
        ).move_to(formula)
        
        self.play(Transform(formula, formula_values))
        self.wait(2)
        
        # Move them back
        self.play(
            x1_track.animate.set_value(1.0),
            x30_track.animate.set_value(2.0),
            run_time=2
        )
        self.wait(2)
