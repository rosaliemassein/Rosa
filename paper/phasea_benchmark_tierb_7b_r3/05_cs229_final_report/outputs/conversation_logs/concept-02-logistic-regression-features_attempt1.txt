import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Manual setup for Axes (avoiding the Axes class)
        graph_origin = RIGHT * 3 + DOWN * 0.5
        x_axis = Line(graph_origin + LEFT * 3, graph_origin + RIGHT * 3, color=WHITE)
        y_axis = Line(graph_origin, graph_origin + UP * 3, color=WHITE)
        
        x_label = MathTex("z", font_size=24).next_to(x_axis, RIGHT)
        y_label = MathTex("P", font_size=24).next_to(y_axis, UP)
        
        # 2. Manual Sigmoid Curve (avoiding ParametricFunction)
        sigmoid_curve = VMobject(color=YELLOW)
        points = []
        # Generate points for x from -4 to 4
        for val in np.linspace(-4, 4, 50):
            # Sigmoid formula: 1 / (1 + exp(-x))
            # Map math x [-4, 4] to screen x [-3, 3]
            # Map math y [0, 1] to screen y [0, 3]
            screen_x = (val / 4.0) * 3.0
            prob_y = 1.0 / (1.0 + np.exp(-val))
            screen_y = prob_y * 3.0
            points.append(graph_origin + screen_x * RIGHT + screen_y * UP)
        
        sigmoid_curve.set_points_as_corners(points)
        
        # 3. Sliders constructed manually (avoiding Slider class)
        # Slider for 1-Day T-CSP
        s1_line = Line(LEFT * 5 + DOWN * 1.5, LEFT * 5 + UP * 1.5, color=GRAY)
        s1_knob = Dot(LEFT * 5 + ORIGIN, color=RED)
        s1_text = Text("1-Day T-CSP", font_size=16).next_to(s1_line, DOWN)
        
        # Slider for 30-Day T-CSP
        s2_line = Line(LEFT * 3 + DOWN * 1.5, LEFT * 3 + UP * 1.5, color=GRAY)
        s2_knob = Dot(LEFT * 3 + ORIGIN, color=BLUE)
        s2_text = Text("30-Day T-CSP", font_size=16).next_to(s2_line, DOWN)
        
        # 4. Prob Dot on the curve
        prob_dot = Dot(color=GREEN)
        
        def update_prob_dot(mobject):
            # Calculate values based on knob positions relative to their slider lines
            # Mapping knob Y from [-1.5, 1.5] to feature values [-2, 2]
            v1 = s1_knob.get_center()[1] * 1.33
            v2 = s2_knob.get_center()[1] * 1.33
            
            # Simple model: z = v1 + v2
            z_val = v1 + v2
            # Clamp z for display
            display_z = max(min(z_val, 4), -4)
            
            # Math y = 1 / (1 + exp(-z))
            prob = 1.0 / (1.0 + np.exp(-z_val))
            
            # Map back to screen coordinates
            screen_x = (display_z / 4.0) * 3.0
            screen_y = prob * 3.0
            mobject.move_to(graph_origin + screen_x * RIGHT + screen_y * UP)

        prob_dot.add_updater(update_prob_dot)
        
        # 5. Formulas (avoiding TransformMatchingTex)

            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",

            Create(s1_line), Create(s1_knob), Write(s1_text),
            Create(s2_line), Create(s2_knob), Write(s2_text)
        )
        self.add(prob_dot)

        self.play(s2_knob.animate.shift(UP * 1.0), run_time=2)

        self.play(s2_knob.animate.shift(DOWN * 1.5), run_time=2)
        self.play(s1_knob.animate.move_to(LEFT * 5 + ORIGIN), s2_knob.animate.move_to(LEFT * 3 + ORIGIN), run_time=2)
