import numpy as np
from manim import *

class AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. UI Labels and Split Titles
        st_title = Text("Static Benchmark").scale(0.6).shift(LEFT * 3.5 + UP * 3)
        ml_title = Text("Logistic Model").scale(0.6).shift(RIGHT * 3.5 + UP * 3)
        self.add(st_title, ml_title)

        # 2. Floorplan Visualizations (Simplified Grids)
        # Left Grid (Static Industry Standard)
        left_grid = VGroup()
        for row_idx in range(3):
            for col_idx in range(3):
                square = Rectangle(width=0.7, height=0.5)
                square.move_to(LEFT * 3.5 + UP * 1 + RIGHT * col_idx * 0.8 + DOWN * row_idx * 0.6)
                left_grid.add(square)
        left_grid.move_to(LEFT * 3.5 + UP * 0.5)
        
        # Right Grid (Adaptive Machine Learning)
        right_grid = VGroup()
        for row_idx in range(3):
            for col_idx in range(3):
                square = Rectangle(width=0.7, height=0.5)
                square.move_to(RIGHT * 3.5 + UP * 1 + RIGHT * col_idx * 0.8 + DOWN * row_idx * 0.6)
                right_grid.add(square)
        right_grid.move_to(RIGHT * 3.5 + UP * 0.5)
        
        self.add(left_grid, right_grid)

        # Static Highlights: Standard "rules of thumb" are constant
        st_h1 = left_grid[1].copy().set_fill(BLUE, opacity=0.6).set_stroke(BLUE, 2)
        st_h2 = left_grid[4].copy().set_fill(BLUE, opacity=0.6).set_stroke(BLUE, 2)
        st_h3 = left_grid[6].copy().set_fill(BLUE, opacity=0.6).set_stroke(BLUE, 2)
        self.add(st_h1, st_h2, st_h3)

        # 3. Timeline at the bottom (May to September)
        timeline_line = Line(LEFT * 5, RIGHT * 5).shift(DOWN * 1.5)
        self.add(timeline_line)
        month_names = ["May", "Jun", "Jul", "Aug", "Sep"]
        for month_idx in range(len(month_names)):
            point_pos = timeline_line.point_from_proportion(month_idx / 4.0)
            month_text = Text(month_names[month_idx], font_size=20).next_to(point_pos, DOWN)
            self.add(month_text)

        # 4. Performance Metrics Graph (Manual drawing since Axes is restricted)
        graph_origin = DOWN * 3.2 + LEFT * 4
        x_axis_line = Line(graph_origin, graph_origin + RIGHT * 8)
        y_axis_line = Line(graph_origin, graph_origin + UP * 1.2)
        self.add(x_axis_line, y_axis_line)
        graph_label = Text("Missed Requests (Error)", font_size=16).next_to(y_axis_line, UP, buff=0.1)
        self.add(graph_label)

        # Graph Paths
        static_error_points = []
        ml_error_points = []
        for step_idx in range(15):
            # Static error stays high
            static_error_points.append(graph_origin + RIGHT * (step_idx * 0.55) + UP * (0.9 + 0.05 * np.sin(step_idx)))
            # ML error decreases and stays low
            ml_error_points.append(graph_origin + RIGHT * (step_idx * 0.55) + UP * (0.6 * np.exp(-step_idx / 4.0) + 0.1))
        
        static_line = VMobject(color=BLUE).set_points_as_corners(static_error_points)
        ml_line = VMobject(color=RED).set_points_as_corners(ml_error_points)

        # 5. Adaptive Animation Logic
        # Using a dummy mobject to track progress (bypassing restricted ValueTracker)
        progress_dot = Dot(radius=0).move_to(ORIGIN)
        
        def update_ml_zones(target_mobject):
            time_val = progress_dot.get_center()[0]
            current_highlights = VGroup()
            for zone_idx in range(9):
                # Use a pseudo-random function of index and time to flicker
                if np.sin(time_val * 4.0 + zone_idx * 1.5) > 0.35:
                    highlight = right_grid[zone_idx].copy().set_fill(RED, opacity=0.6).set_stroke(RED, 2)
                    current_highlights.add(highlight)
            target_mobject.become(current_highlights)

        ml_highlights = VGroup()
        ml_highlights.add_updater(update_ml_zones)
        self.add(ml_highlights)

        # 6. Execute Animations
        self.play(
            Create(static_line),
            Create(ml_line),
            progress_dot.animate.shift(RIGHT * 10),
            run_time=10,
            rate_func=lambda t_prog: t_prog
        )
        self.wait(2)
