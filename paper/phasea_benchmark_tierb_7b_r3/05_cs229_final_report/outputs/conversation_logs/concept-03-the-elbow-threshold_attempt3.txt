from manim import *
import numpy as np

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Manual Axes Construction (Avoiding the disallowed Axes class)
        # Origin at lower left
        ox, oy = -4.5, -2.5
        origin = np.array([ox, oy, 0])
        
        x_axis = Line(origin, origin + RIGHT * 6, color=WHITE)
        y_axis = Line(origin, origin + UP * 5, color=WHITE)
        
        x_label = Text("Fraction of Zones Flagged", font_size=18).next_to(x_axis, DOWN)
        y_label = Text("Recall", font_size=18).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        # 2. Formula (Objective)
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_corner(UL, buff=0.5)
        
        # 3. Curve Construction (Manual VMobject points to avoid Axes.plot)
        # Curve function: y = x^0.4 (Concave, rises sharp then levels)
        curve = VMobject(color=BLUE)
        curve_pts = []
        for i in range(51):
            t = i / 50.0  # Normalized progress 0 to 1
            px = ox + t * 5.5
            py = oy + (t ** 0.4) * 4.5
            curve_pts.append([px, py, 0])
        curve.set_points_as_corners(curve_pts)
        
        # 4. Replacement for ValueTracker (Using a hidden Mobject's coordinate)
        # The x-coordinate of this dot represents our progress 't' along the x-axis
        tracker = Dot(radius=0).move_to(origin)
        
        # 5. Point on the curve
        moving_dot = Dot(color=RED, radius=0.1)
        
        def update_moving_dot(mob):
            # Calculate current progress t based on tracker position
            current_x = tracker.get_center()[0]
            t = (current_x - ox) / 5.5
            t = max(0.0001, min(t, 1.0)) # Clamp and avoid zero
            new_px = ox + t * 5.5
            new_py = oy + (t ** 0.4) * 4.5
            mob.move_to([new_px, new_py, 0])
            
        moving_dot.add_updater(update_moving_dot)
        
        # 6. Building Floorplan Grid
        fp_title = Text("Building Floorplan", font_size=22).shift(RIGHT * 3.5 + UP * 2.5)
        
        grid = VGroup()
        for r in range(4):
            for c in range(4):
                sq = Square(side_length=0.5, stroke_width=2, color=WHITE, fill_opacity=0.1)
                sq.move_to([2.5 + c * 0.65, -0.5 + r * 0.65, 0])
                grid.add(sq)
        
        def update_grid(g):
            current_x = tracker.get_center()[0]
            t = (current_x - ox) / 5.5
            t = max(0, min(t, 1))
            num_flagged = int(t * 16)
            for i, zone in enumerate(g):
                if i < num_flagged:
                    zone.set_fill(GREEN, opacity=0.8)
                else:
                    zone.set_fill(GRAY, opacity=0.1)
                    
        grid.add_updater(update_grid)
        
        # 7. Animation Execution
        self.add(x_axis, y_axis, x_label, y_label, formula, curve, moving_dot, fp_title, grid, tracker)
        
        # Move right (lowering threshold, flagging more zones)
        self.play(tracker.animate.shift(RIGHT * 5.0), run_time=3, rate_func=linear)
        self.wait(0.5)
        
        # Move left (raising threshold, flagging fewer zones)
        self.play(tracker.animate.move_to(origin + RIGHT * 0.3), run_time=2, rate_func=smooth)
        self.wait(0.5)
        
        # Find the 'Elbow' (Sweet spot)
        elbow_t = 0.22 # Approximate point where slope changes significantly
        self.play(tracker.animate.move_to(origin + RIGHT * (elbow_t * 5.5)), run_time=1.5)
        
        # Highlight the elbow
        elbow_pos = [ox + elbow_t * 5.5, oy + (elbow_t ** 0.4) * 4.5, 0]
        glowing_circle = Circle(radius=0.3, color=YELLOW).move_to(elbow_pos)
        self.play(Create(glowing_circle))
        self.play(Flash(glowing_circle, color=YELLOW, flash_radius=0.5))
        
        self.wait(2)
