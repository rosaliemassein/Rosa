from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Axes (Standard Manim)
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 100, 20],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        x_label = axes.get_x_axis_label("Number of Players").scale(0.6)
        y_label = axes.get_y_axis_label("MSE").scale(0.6)

        # 2. Plotting curves
        # KNN curve: Overfitting behavior (dips then rises)
        knn_plot = axes.plot(
            lambda x: 5.0 * (x - 3.0)**2 + 20.0, 
            x_range=[0.5, 7.0], 
            color=RED
        )
        knn_label = Text("KNN", color=RED).scale(0.5).next_to(knn_plot, UR, buff=0.1)
        
        # Lasso curve: Steady decline as regularization helps
        lasso_plot = axes.plot(
            lambda x: 70.0 * (0.7**x) + 10.0, 
            x_range=[0.5, 9.5], 
            color=GREEN
        )
        lasso_label = Text("Lasso", color=GREEN).scale(0.5).next_to(lasso_plot, RIGHT, buff=0.1)

        # 3. Formula
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1"
        ).scale(0.8).to_edge(UP)

        # 4. Coefficients visualization (Bar chart)
        coeffs = VGroup()
        for i in range(60):
            # Generate deterministic heights for bars
            h_val = (i % 5) * 0.2 + 0.4
            bar = Line(ORIGIN, UP * h_val, stroke_width=2, color=BLUE)
            coeffs.add(bar)
        
        coeffs.arrange(RIGHT, buff=0.03).to_edge(RIGHT, buff=0.5).shift(DOWN)
        coeffs_text = Text("Coefficients", font_size=24).next_to(coeffs, UP)

        # 5. Animation Sequence
        self.play(Create(axes), Write(x_label), Write(y_label))
        self.play(Create(knn_plot), Write(knn_label))
        self.play(Create(lasso_plot), Write(lasso_label))
        self.wait(1)
        
        self.play(Write(formula))
        self.play(Create(coeffs), Write(coeffs_text))
        self.wait(1)

        # Visualize penalty effect (Lambda increasing)
        lambda_indicator = MathTex(r"\lambda \rightarrow \infty", color=YELLOW).next_to(formula, DOWN)
        
        # Many coefficients shrink to zero in Lasso
        shrink_animations = []
        for i, bar in enumerate(coeffs):
            if (i % 4) != 0:
                # 75% of coefficients vanish
                shrink_animations.append(bar.animate.scale(0, about_edge=DOWN))
            else:
                # 25% stay but small
                shrink_animations.append(bar.animate.scale(0.2, about_edge=DOWN))
        
        self.play(
            Write(lambda_indicator),
            *shrink_animations,
            run_time=3
        )
        self.wait(2)
