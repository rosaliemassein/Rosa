from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Setup Manual Axes (Avoiding 'Axes' class if restricted)
        origin_point = 3 * LEFT + 2 * DOWN
        x_axis_line = Line(origin_point, origin_point + 6 * RIGHT, color=WHITE)
        y_axis_line = Line(origin_point, origin_point + 4 * UP, color=WHITE)
        
        # Labels
        x_label = Tex("Number of Players").scale(0.6).next_to(x_axis_line, DOWN)
        y_label = Tex("MSE").scale(0.6).next_to(y_axis_line, LEFT).rotate(90 * DEGREES)
        
        self.add(x_axis_line, y_axis_line, x_label, y_label)

        # 2. Curves (Using ParametricFunction with 't' to avoid 'x')
        def get_knn_y(t_val):
            # Dips then rises (overfitting)
            return 0.15 * (t_val - 3)**2 + 0.5
            
        def get_reg_y(t_val):
            # Steady decline
            return 3.0 / (t_val + 1) + 0.2

        # Mapping functions to coordinates
        # t ranges from 0 to 5.5 for display
        knn_curve = ParametricFunction(
            lambda t: origin_point + t * RIGHT + get_knn_y(t) * UP,
            t_range=[0.2, 5.5],
            color=RED
        )
        
        reg_curve = ParametricFunction(
            lambda t: origin_point + t * RIGHT + get_reg_y(t) * UP,
            t_range=[0.2, 5.5],
            color=GREEN
        )
        
        knn_text = Tex("KNN", color=RED).scale(0.5).next_to(knn_curve.get_end(), UR, buff=0.1)
        reg_text = Tex("Lasso/Ridge", color=GREEN).scale(0.5).next_to(reg_curve.get_end(), RIGHT, buff=0.1)

        self.play(Create(knn_curve), Write(knn_text))
        self.play(Create(reg_curve), Write(reg_text))

        # 3. Lasso Formula
        # J(theta) = ||y - X*theta||^2 + lambda * ||theta||_1
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            color=YELLOW
        ).to_edge(UP, buff=0.5)
        
        self.play(Write(formula))
        self.wait(1)

        # 4. Coefficients Visualization (using Rectangle instead of BarChart)
        bars = VGroup()
        np.random.seed(42)
        bar_count = 35
        initial_heights = np.random.uniform(0.2, 1.4, bar_count)
        
        for h_val in initial_heights:
            bar_obj = Rectangle(
                width=0.08, 
                height=h_val, 
                fill_opacity=0.8, 
                fill_color=BLUE, 
                stroke_width=0
            )
            bars.add(bar_obj)
        
        bars.arrange(RIGHT, buff=0.03, aligned_edge=DOWN).to_edge(DR, buff=0.8).shift(UP * 0.5)
        bar_title = Tex("Coefficients", color=BLUE).scale(0.5).next_to(bars, UP, buff=0.2)
        
        self.play(Create(bars), Write(bar_title))
        self.wait(1)

        # 5. Shrinkage Animation
        # We simulate increasing lambda by shrinking bars; many hit 0 (Lasso)
        target_bars = bars.copy()
        for i, b_item in enumerate(target_bars):
            # Lasso effect: subtract penalty and floor at small value
            shrunk_h = initial_heights[i] - 0.8
            if shrunk_h < 0.05:
                shrunk_h = 0.001
                b_item.set_fill(color=WHITE, opacity=0.2)
            b_item.stretch_to_fit_height(shrunk_h)
            
        # Highlight lambda and animate bars
        self.play(
            Transform(bars, target_bars),
            formula.animate.set_color_by_tex(r"\lambda", ORANGE),
            run_time=3
        )
        
        selection_note = Tex("Feature Selection: Some $\\theta \\to 0$", color=ORANGE).scale(0.5).next_to(bars, DOWN, buff=0.3)
        self.play(Write(selection_note))
        
        self.wait(2)
