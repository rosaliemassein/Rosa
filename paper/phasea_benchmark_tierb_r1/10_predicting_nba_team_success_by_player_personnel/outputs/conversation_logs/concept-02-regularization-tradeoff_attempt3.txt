from manim import *
import numpy as np

class Concept02RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Manual Axes Construction (since Axes is disallowed)
        origin_point = LEFT * 3.5 + DOWN * 2
        x_axis = Line(origin_point, origin_point + RIGHT * 6, color=WHITE)
        y_axis = Line(origin_point, origin_point + UP * 4, color=WHITE)
        x_label = Text("Number of Players", font_size=16).next_to(x_axis, DOWN)
        y_label = Text("MSE", font_size=16).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        axes_group = VGroup(x_axis, y_axis, x_label, y_label)

        # 2. Curves as VMobjects
        # Map x from [0, 10] to [0, 6], y from [0, 10] to [0, 4]
        def to_coord(x_val, y_val):
            return origin_point + RIGHT * (x_val * 0.6) + UP * (y_val * 0.4)

        # KNN Curve (U-shaped: overfitting)
        knn_points = [to_coord(x, 0.2 * (x - 3)**2 + 2) for x in np.linspace(0.5, 9.5, 20)]
        knn_curve = VMobject(color=RED).set_points_as_corners(knn_points)
        knn_tag = Text("KNN", color=RED, font_size=18).move_to(to_coord(9.5, 8.5))

        # Lasso Curve (Steady decline)
        lasso_points = [to_coord(x, 8 * np.exp(-0.3 * x) + 1) for x in np.linspace(0.5, 9.5, 20)]
        lasso_curve = VMobject(color=GREEN).set_points_as_corners(lasso_points)
        lasso_tag = Text("Lasso/Ridge", color=GREEN, font_size=18).move_to(to_coord(9.5, 2))

        # 3. Formula and Titles
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            font_size=36
        ).to_edge(UP, buff=0.5)

        voiceover_1 = Text("As complexity grows, models risk noise (Curse of Dimensionality).", font_size=20).next_to(formula, DOWN)
        
        # 4. Bar Chart for coefficients
        bar_origin = RIGHT * 3 + DOWN * 1
        num_bars = 40
        bar_heights = np.random.uniform(-1.5, 1.5, num_bars)
        bars = VGroup(*[
            Line(bar_origin + RIGHT * (i * 0.1), bar_origin + RIGHT * (i * 0.1) + UP * h, stroke_width=3, color=BLUE)
            for i, h in enumerate(bar_heights)
        ])
        bar_title = Text("Coefficients", font_size=20).next_to(bars, UP)

        # --- ANIMATION ---

        # Phase 1: Curves
        self.play(Create(axes_group))
        self.play(Write(voiceover_1))
        self.play(Create(knn_curve), Write(knn_tag))
        self.play(Create(lasso_curve), Write(lasso_tag))
        self.wait(1)

        # Phase 2: Penalty and Coefficients
        self.play(
            FadeOut(voiceover_1),
            VGroup(axes_group, knn_curve, knn_tag, lasso_curve, lasso_tag).animate.scale(0.7).to_edge(LEFT, buff=0.5),
            Write(formula)
        )
        self.play(Create(bars), Write(bar_title))
        self.wait(1)

        # Phase 3: Manual Shrinking Animation (avoiding ValueTracker/always_redraw)
        def shrink_bars(mobject, alpha):
            # alpha goes 0 to 1
            # penalty lambda increases with alpha
            lam = alpha * 1.5 
            for i, bar in enumerate(mobject):
                h_orig = bar_heights[i]
                # Lasso shrinkage formula: sign(x) * max(0, |x| - lambda)
                h_new = np.sign(h_orig) * max(0, abs(h_orig) - lam)
                start = bar_origin + RIGHT * (i * 0.1)
                end = start + UP * h_new
                bar.set_points_as_corners([start, end])
                if abs(h_new) < 0.001:
                    bar.set_stroke(opacity=0.2)
                else:
                    bar.set_stroke(opacity=1.0)

        lambda_label = MathTex(r"\lambda \text{ increasing...}", font_size=24, color=YELLOW).next_to(bar_title, UP)
        self.play(Write(lambda_label))
        self.play(
            UpdateFromAlphaFunc(bars, shrink_bars),
            run_time=4,
            rate_func=lambda t: t # Linear
        )
        
        conclusion = Text("Lasso forces noise to zero.", font_size=24, color=YELLOW).next_to(bars, DOWN)
        self.play(Write(conclusion))
        self.wait(2)
