from manim import *

class Concept02OpponentAbstraction(Scene):
    def construct(self):
        # 1. Formula display
        formula = MathTex(r"S \in \mathbb{B}^{n \times n \times 3}")
        formula.to_edge(UP)
        self.play(Write(formula))
        self.wait(1)

        # 2. 2-player board representation
        title_2p = Text("2-Player Game", font_size=24).shift(LEFT * 3 + UP * 1.5)
        board_2p = VGroup(*[
            Square(side_length=0.5, fill_opacity=0.5).set_fill(BLUE if i < 2 else RED)
            for i in range(4)
        ]).arrange_in_grid(rows=2, cols=2, buff=0.1).next_to(title_2p, DOWN)
        
        self.play(Create(title_2p), FadeIn(board_2p))
        self.wait(1)

        # 3. 6-player board representation
        title_6p = Text("6-Player Game", font_size=24).shift(RIGHT * 3 + UP * 1.5)
        
        # Colors for 5 different opponents (using standard Manim colors)
        opp_colors = [RED, GREEN, YELLOW, PURPLE, ORANGE]
        
        # Create a grid where one cell is "Me" (Blue) and others are different opponents
        board_6p_me = Square(side_length=0.5, fill_opacity=0.8, color=BLUE).set_fill(BLUE)
        board_6p_opps = VGroup(*[
            Square(side_length=0.5, fill_opacity=0.8, color=c).set_fill(c)
            for c in opp_colors
        ]).arrange_in_grid(rows=1, cols=5, buff=0.1)
        
        group_6p = VGroup(board_6p_me, board_6p_opps).arrange(DOWN, buff=0.5).next_to(title_6p, DOWN)
        
        self.play(Create(title_6p), FadeIn(group_6p))
        self.wait(2)

        # 4. Abstraction Process (Merging colors)
        # Use Cross instead of CrossOut
        crosses = VGroup(*[Cross(sq) for sq in board_6p_opps])
        
        self.play(Create(crosses))
        self.wait(1)

        # Transform the diverse opponent squares into a single unified "Red" layer
        unified_opp_layer = Square(side_length=0.5, fill_opacity=0.8, color=RED).set_fill(RED)
        unified_opp_layer.move_to(board_6p_opps.get_center())
        
        label_unified = Text("Unified Opponent Layer", font_size=20, color=RED).next_to(unified_opp_layer, DOWN)
        check_mark = MathTex(r"\checkmark", color=GREEN).scale(1.5).next_to(label_unified, RIGHT)

        self.play(
            FadeOut(crosses),
            ReplacementTransform(board_6p_opps, unified_opp_layer),
            Write(label_unified)
        )
        self.play(Write(check_mark))
        self.wait(2)

        # 5. Concept Text
        concept = Text("Strategy scales without changing architecture", font_size=28)
        concept.to_edge(DOWN, buff=0.8)
        self.play(Write(concept))
        self.play(Indicate(concept))
        
        self.wait(3)
