from manim import *
import numpy as np

class GeneticWinRateSelection(Scene):
    def construct(self):
        # 1. Formula Header
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)")
        formula.to_edge(UP)
        self.play(Write(formula))

        # 2. Initial Parent
        parent = Circle(radius=0.6, color=YELLOW)
        parent_text = Text("Parent", font_size=24).next_to(parent, UP)
        self.play(Create(parent), Write(parent_text))
        
        current_parent_node = parent
        current_parent_label = parent_text

        # 3. Evolution Cycle (Run twice to show evolution)
        for cycle in range(2):
            # Define Children positions
            child_positions = [3 * LEFT + DOWN, DOWN, 3 * RIGHT + DOWN]
            children = VGroup(*[Circle(radius=0.5, color=GREEN).move_to(pos) for pos in child_positions])
            child_labels = VGroup(*[Text(f"Child {i+1}", font_size=20).next_to(children[i], UP) for i in range(3)])
            
            # Connect Parent to Children
            lines = VGroup(*[Line(current_parent_node.get_bottom(), child.get_top(), buff=0.1) for child in children])
            
            self.play(
                Create(lines),
                Create(children),
                Write(child_labels)
            )

            # Win Rates (Iteration 1: 2nd wins, Iteration 2: 1st wins)
            if cycle == 0:
                rates = [0.45, 0.73, 0.48]
            else:
                rates = [0.88, 0.62, 0.55]
            
            # 4. BarChart Representation
            bars = VGroup()
            rate_texts = VGroup()
            for i, rate in enumerate(rates):
                # Create a bar that grows from its bottom
                bar = Rectangle(width=0.6, height=rate * 2, fill_opacity=0.8, color=BLUE)
                bar.next_to(children[i], DOWN, buff=0.5)
                # Align the bottom of the bar correctly
                bar.shift(UP * (bar.height / 2))
                
                label = DecimalNumber(rate, num_decimal_places=2).scale(0.6).next_to(bar, DOWN)
                bars.add(bar)
                rate_texts.add(label)

            self.play(
                *[Create(bar) for bar in bars],
                *[Write(rt) for rt in rate_texts],
                run_time=1
            )
            self.wait(0.5)

            # 5. Highlight best child
            best_idx = int(np.argmax(rates))
            winner_child = children[best_idx]
            winner_bar = bars[best_idx]
            winner_label = child_labels[best_idx]
            
            highlight = SurroundingRectangle(VGroup(winner_child, winner_bar), color=GOLD, buff=0.2)
            self.play(
                Create(highlight),
                winner_child.animate.set_color(GOLD),
                winner_bar.animate.set_color(GOLD)
            )
            self.wait(1)

            # 6. Fade out others and move gold child to center
            to_fade = VGroup(
                current_parent_node, current_parent_label, lines, highlight,
                *[children[i] for i in range(3) if i != best_idx],
                *[child_labels[i] for i in range(3) if i != best_idx],
                *[bars[i] for i in range(3) if i != best_idx],
                *[rate_texts[i] for i in range(3) if i != best_idx],
                winner_bar, rate_texts[best_idx], winner_label
            )

            # Transition winner to be the new parent
            self.play(
                FadeOut(to_fade),
                winner_child.animate.move_to(ORIGIN).scale(1.2).set_color(YELLOW),
                run_time=1.5
            )
            
            # Reset references for next loop
            current_parent_node = winner_child
            current_parent_label = Text("Parent", font_size=24).next_to(current_parent_node, UP)
            self.play(Write(current_parent_label))
            self.wait(0.5)

        self.wait(2)
