from manim import *
import numpy as np

class IterativeClustering(Scene):
    def construct(self):
        # 1. Generate a cloud of 27 Dot objects scattered randomly
        np.random.seed(42)
        dots = VGroup(*[
            Dot(point=[np.random.uniform(-5, 5), np.random.uniform(-3, 3), 0], radius=0.08) 
            for _ in range(27)
        ])
        self.add(dots)

        unassigned_indices = list(range(27))
        cluster_colors = [BLUE, GREEN, RED]

        for cluster_idx in range(3):
            current_color = cluster_colors[cluster_idx]
            
            # 2. Highlight the two dots with the shortest distance (highest similarity)
            min_dist = float('inf')
            seed_pair = (None, None)
            
            for i in range(len(unassigned_indices)):
                for j in range(i + 1, len(unassigned_indices)):
                    idx1 = unassigned_indices[i]
                    idx2 = unassigned_indices[j]
                    dist = np.linalg.norm(dots[idx1].get_center() - dots[idx2].get_center())
                    if dist < min_dist:
                        min_dist = dist
                        seed_pair = (idx1, idx2)
            
            p1_idx, p2_idx = seed_pair
            cluster_indices = [p1_idx, p2_idx]
            unassigned_indices.remove(p1_idx)
            unassigned_indices.remove(p2_idx)
            
            # Highlighting seed pair with bright rings
            seed_highlight = VGroup(
                Circle(radius=0.15, color=YELLOW, stroke_width=4).move_to(dots[p1_idx]),
                Circle(radius=0.15, color=YELLOW, stroke_width=4).move_to(dots[p2_idx])
            )
            self.play(
                Create(seed_highlight), 
                dots[p1_idx].animate.set_color(current_color), 
                dots[p2_idx].animate.set_color(current_color)
            )
            
            # 3. Animate a 'search' circle expanding from this cluster
            # We'll expand the circle from the centroid of the current cluster
            search_circle = Circle(radius=0, color=YELLOW, stroke_width=2)
            self.add(search_circle)

            while len(cluster_indices) < 9:
                centroid = np.mean([dots[i].get_center() for i in cluster_indices], axis=0)
                search_circle.move_to(centroid)
                
                # Find the next most similar (closest) unassigned dot
                closest_dist = float('inf')
                next_dot_idx = None
                
                for idx in unassigned_indices:
                    dist = np.linalg.norm(dots[idx].get_center() - centroid)
                    if dist < closest_dist:
                        closest_dist = dist
                        next_dot_idx = idx
                
                # Search circle expanding until it touches the next closest dot
                self.play(
                    search_circle.animate.set_radius(closest_dist),
                    run_time=0.4,
                    rate_func=rate_functions.linear
                )
                
                # When it touches, that dot changes color and flashes
                self.play(
                    dots[next_dot_idx].animate.set_color(current_color),
                    Flash(dots[next_dot_idx], color=current_color, flash_radius=0.2),
                    run_time=0.3
                )
                
                cluster_indices.append(next_dot_idx)
                unassigned_indices.remove(next_dot_idx)
            
            # Fade out tools before starting the next cluster
            self.play(FadeOut(search_circle), FadeOut(seed_highlight))
            
        self.wait(2)
