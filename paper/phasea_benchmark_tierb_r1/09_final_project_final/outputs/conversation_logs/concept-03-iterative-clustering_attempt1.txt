from manim import *
import numpy as np

class IterativeClustering(Scene):
    def construct(self):
        # 1. Setup: Create 27 randomly scattered Dot objects
        np.random.seed(42)  # For reproducible random positions
        dots = VGroup(*[
            Dot(point=[np.random.uniform(-5, 5), np.random.uniform(-2, 3), 0]) 
            for _ in range(27)
        ])
        
        # Narration Text at the bottom
        narrator_text = Text(
            "Standard clustering methods like K-Means often fail on sparse data.",
            font_size=24
        ).to_edge(DOWN).set_color(GRAY_A)
        
        self.add(dots)
        self.play(Write(narrator_text))
        self.wait(1)

        unassigned_indices = list(range(27))
        cluster_colors = [RED, BLUE, GREEN]
        
        # We need to create 3 clusters of 9 dots each
        for cluster_idx in range(3):
            # Update Narration for the seed step
            if cluster_idx == 0:
                self.play(narrator_text.animate.become(
                    Text("We start by finding the two most similar proposals to 'seed' a cluster.", font_size=24).to_edge(DOWN).set_color(GRAY_A)
                ))
            
            # Step A: Find the two most similar (closest) dots among unassigned
            min_dist = float('inf')
            seed_pair = (None, None)
            
            for i in range(len(unassigned_indices)):
                for j in range(i + 1, len(unassigned_indices)):
                    idx1 = unassigned_indices[i]
                    idx2 = unassigned_indices[j]
                    dist = np.linalg.norm(dots[idx1].get_center() - dots[idx2].get_center())
                    if dist < min_dist:
                        min_dist = dist
                        seed_pair = (idx1, idx2)
            
            # Highlight the seed pair with a bright ring
            seed_group = VGroup(dots[seed_pair[0]], dots[seed_pair[1]])
            seed_center = seed_group.get_center()
            ring = Circle(radius=0.4, color=YELLOW, stroke_width=5).move_to(seed_center)
            
            self.play(Create(ring))
            self.play(
                dots[seed_pair[0]].animate.set_color(cluster_colors[cluster_idx]),
                dots[seed_pair[1]].animate.set_color(cluster_colors[cluster_idx])
            )
            
            current_cluster = [seed_pair[0], seed_pair[1]]
            unassigned_indices.remove(seed_pair[0])
            unassigned_indices.remove(seed_pair[1])
            
            # Update Narration for expansion
            if cluster_idx == 0:
                self.play(narrator_text.animate.become(
                    Text("A search circle expands until it hits the next closest proposal.", font_size=24).to_edge(DOWN).set_color(GRAY_A)
                ))

            # Step B: Greedily pull in dots until cluster size is 9
            while len(current_cluster) < 9:
                best_unassigned = -1
                closest_dist = float('inf')
                
                # Find unassigned dot closest to the current cluster's centroid
                cluster_vgroup = VGroup(*[dots[i] for i in current_cluster])
                center = cluster_vgroup.get_center()
                
                for u_idx in unassigned_indices:
                    dist = np.linalg.norm(dots[u_idx].get_center() - center)
                    if dist < closest_dist:
                        closest_dist = dist
                        best_unassigned = u_idx
                
                # Animate search circle expanding from center
                # Using lambda t: t for linear expansion to avoid undefined identifier issues
                search_circle = Circle(radius=0.01, color=YELLOW, stroke_opacity=0.4).move_to(center)
                target_radius = np.linalg.norm(dots[best_unassigned].get_center() - center)
                
                self.play(
                    search_circle.animate.set_width(max(0.1, target_radius * 2)),
                    run_time=0.25,
                    rate_func=lambda t: t
                )
                self.play(
                    dots[best_unassigned].animate.set_color(cluster_colors[cluster_idx]),
                    FadeOut(search_circle),
                    run_time=0.1
                )
                
                current_cluster.append(best_unassigned)
                unassigned_indices.remove(best_unassigned)
            
            # Clear the seed ring
            self.play(FadeOut(ring))

        # Final Narration
        self.play(narrator_text.animate.become(
            Text("This ensures each group has the highest possible thematic cohesion.", font_size=24).to_edge(DOWN).set_color(GRAY_A)
        ))
        self.wait(2)
        self.play(narrator_text.animate.become(
            Text("Reviewers can now cover a single batch with ease.", font_size=24).to_edge(DOWN).set_color(GRAY_A)
        ))
        self.wait(2)
