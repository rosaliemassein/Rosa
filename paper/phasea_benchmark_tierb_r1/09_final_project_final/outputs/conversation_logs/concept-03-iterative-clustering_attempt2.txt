from manim import *
import numpy as np

class ConceptIterativeClustering(Scene):
    def construct(self):
        # 1. Generate a cloud of 27 Dot objects scattered randomly
        np.random.seed(42)
        dots = VGroup(*[
            Dot(point=[np.random.uniform(-5, 5), np.random.uniform(-3, 2.5), 0], radius=0.08)
            for _ in range(27)
        ])
        self.play(Create(dots), run_time=1.5)
        self.wait(0.5)

        unassigned_indices = list(range(27))
        colors = [RED, GREEN, BLUE]
        
        # Text for narration
        narration_label = Text(
            "Finding the two most similar proposals...",
            font_size=24
        ).to_edge(DOWN).shift(UP * 0.3)
        self.play(Write(narration_label))

        for cluster_idx in range(3):
            cluster_color = colors[cluster_idx]
            cluster_indices = []

            # Step A: Find the two closest unassigned dots (highest similarity)
            min_dist = float('inf')
            seed_pair = (unassigned_indices[0], unassigned_indices[1])
            
            for i in range(len(unassigned_indices)):
                for j in range(i + 1, len(unassigned_indices)):
                    idx1 = unassigned_indices[i]
                    idx2 = unassigned_indices[j]
                    dist = np.linalg.norm(dots[idx1].get_center() - dots[idx2].get_center())
                    if dist < min_dist:
                        min_dist = dist
                        seed_pair = (idx1, idx2)
            
            idx1, idx2 = seed_pair
            cluster_indices.extend([idx1, idx2])
            unassigned_indices.remove(idx1)
            unassigned_indices.remove(idx2)

            # Highlight the seeds with a ring as requested
            ring = Circle(radius=0.15, color=YELLOW, stroke_width=4).move_to(dots[idx1])
            self.play(Create(ring), run_time=0.5)
            self.play(
                dots[idx1].animate.set_color(cluster_color),
                dots[idx2].animate.set_color(cluster_color),
                ring.animate.move_to(dots[idx2]),
                run_time=0.5
            )
            self.play(FadeOut(ring))

            if cluster_idx == 0:
                new_text = Text("Greedily pulling in dots until target size (9) is reached...", font_size=24).to_edge(DOWN).shift(UP * 0.3)
                self.play(Transform(narration_label, new_text))

            # Step B: Animate a 'search' circle expanding to find 7 more dots
            search_circle = Circle(radius=0, color=cluster_color, stroke_width=2, fill_opacity=0.1)
            self.add(search_circle)

            for _ in range(7):
                current_cluster_mobjects = VGroup(*[dots[i] for i in cluster_indices])
                center = current_cluster_mobjects.get_center()
                
                # Find the closest unassigned dot to the cluster center
                closest_idx = -1
                dist_to_closest = float('inf')
                for u_idx in unassigned_indices:
                    d = np.linalg.norm(dots[u_idx].get_center() - center)
                    if d < dist_to_closest:
                        dist_to_closest = d
                        closest_idx = u_idx
                
                # Expansion animation (using lambda t: t for linear interpolation)
                self.play(
                    search_circle.animate.set_width(dist_to_closest * 2).move_to(center),
                    run_time=0.3,
                    rate_func=lambda t: t
                )
                
                self.play(dots[closest_idx].animate.set_color(cluster_color), run_time=0.1)
                
                cluster_indices.append(closest_idx)
                unassigned_indices.remove(closest_idx)

            self.play(FadeOut(search_circle))
            
            if cluster_idx < 2:
                next_cluster_text = Text(f"Seeding cluster {cluster_idx + 2}...", font_size=24).to_edge(DOWN).shift(UP * 0.3)
                self.play(Transform(narration_label, next_cluster_text))

        # Final reveal: thematic cohesion
        final_text = Text(
            "Thematic cohesion ensures panels can cover the whole batch.",
            font_size=24
        ).to_edge(DOWN).shift(UP * 0.3)
        self.play(Transform(narration_label, final_text))

        # Group dots by color for final separation
        group1 = VGroup(*[d for d in dots if d.get_color() == RED])
        group2 = VGroup(*[d for d in dots if d.get_color() == GREEN])
        group3 = VGroup(*[d for d in dots if d.get_color() == BLUE])

        self.play(
            group1.animate.shift(LEFT * 2.5 + UP * 0.5),
            group2.animate.shift(RIGHT * 2.5 + UP * 0.5),
            group3.animate.shift(DOWN * 1.5),
            run_time=2
        )
        self.wait(2)
