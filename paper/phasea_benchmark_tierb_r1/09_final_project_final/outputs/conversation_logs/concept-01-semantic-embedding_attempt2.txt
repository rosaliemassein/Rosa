from manim import *
import numpy as np

class SemanticEmbedding(Scene):
    def construct(self):
        # 1. Setup 2D Environment (Axes instead of NumberPlane as NumberPlane is restricted)
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-4, 4, 1],
            x_length=10,
            y_length=6,
            axis_config={"color": GREY, "include_tip": True}
        )
        self.add(axes)

        # 2. Text Labels
        dl_label = Text("Deep Learning", font_size=24).move_to(axes.c2p(-3, 2))
        nn_label = Text("Neural Networks", font_size=24).move_to(axes.c2p(-1, 3))
        robotics_label = Text("Robotics", font_size=24).move_to(axes.c2p(3, -2))

        self.play(FadeIn(dl_label), FadeIn(nn_label), FadeIn(robotics_label))
        self.wait(1)

        # 3. Transform labels into Dot objects (2D Dot)
        # Deep Learning and Neural Networks are close together
        dl_dot = Dot(axes.c2p(-3.5, 2.5), color=BLUE)
        nn_dot = Dot(axes.c2p(-2.8, 2.2), color=GREEN)
        robotics_dot = Dot(axes.c2p(4, -2.5), color=RED)

        self.play(
            Transform(dl_label, dl_dot),
            Transform(nn_label, nn_dot),
            Transform(robotics_label, robotics_dot),
            run_time=2
        )
        self.wait(0.5)

        # 4. Show distances (Using Line instead of DashedLine as DashedLine is restricted)
        # Highlight small distance between DL and NN
        dist_line_near = Line(dl_dot.get_center(), nn_dot.get_center(), color=YELLOW, stroke_width=2)
        dist_label = Text("Similar", font_size=16).next_to(dist_line_near, UP, buff=0.1)
        
        self.play(Create(dist_line_near), Write(dist_label))
        self.wait(1)

        # 5. Show Formula
        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}", font_size=36)
        formula.to_edge(UP, buff=0.5)
        self.play(Write(formula))
        self.wait(1)

        # 6. Phrase Averaging: 'Machine' and 'Learning'
        # Define positions
        m_pos = axes.c2p(1, -1)
        l_pos = axes.c2p(3, 1)
        avg_pos = (m_pos + l_pos) / 2

        m_arrow = Arrow(axes.get_origin(), m_pos, color=PURPLE, buff=0)
        m_text = Text("Machine", font_size=20).next_to(m_pos, DOWN)
        
        l_arrow = Arrow(axes.get_origin(), l_pos, color=ORANGE, buff=0)
        l_text = Text("Learning", font_size=20).next_to(l_pos, RIGHT)

        self.play(Create(m_arrow), Write(m_text))
        self.play(Create(l_arrow), Write(l_text))
        self.wait(1)

        # 7. Merge into single central vector (Average)
        avg_arrow = Arrow(axes.get_origin(), avg_pos, color=WHITE, buff=0)
        avg_text = Text("Machine Learning (Average)", font_size=20).next_to(avg_pos, UR)

        self.play(
            ReplacementTransform(m_arrow, avg_arrow),
            ReplacementTransform(l_arrow, avg_arrow),
            ReplacementTransform(m_text, avg_text),
            ReplacementTransform(l_text, avg_text),
            run_time=2
        )
        self.wait(1)

        # 8. Narration Summary
        summary = Text(
            "Keywords are mapped to a vector space where similar concepts\nare close. Phrases are the average of their words.",
            font_size=18,
            line_spacing=1.5
        ).to_edge(DOWN, buff=0.5)
        
        background_rect = SurroundingRectangle(summary, color=BLACK, fill_opacity=0.8, fill_color=BLACK, buff=0.2)
        self.play(FadeIn(background_rect), Write(summary))
        self.wait(3)
