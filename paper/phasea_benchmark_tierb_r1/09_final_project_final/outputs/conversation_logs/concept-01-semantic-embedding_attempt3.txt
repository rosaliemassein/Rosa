from manim import *
import numpy as np

class SemanticEmbedding(Scene):
    def construct(self):
        # Use Axes as NumberPlane is disallowed in this environment
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-4, 4, 1],
            x_length=10,
            y_length=6,
            axis_config={"color": BLUE, "include_tip": False}
        )
        self.add(axes)

        # Defining positions for the concepts in vector space
        pos_dl = np.array([-3, 2, 0])
        pos_nn = np.array([-2, 1, 0])
        pos_rb = np.array([3, -2, 0])

        # Text labels positioned near their vector coordinates
        text_dl = Text("Deep Learning").scale(0.5).move_to(pos_dl + np.array([0, 0.5, 0]))
        text_nn = Text("Neural Networks").scale(0.5).move_to(pos_nn + np.array([0, -0.5, 0]))
        text_rb = Text("Robotics").scale(0.5).move_to(pos_rb + np.array([0, 0.5, 0]))

        # Mobjects representing points (vectors)
        dot_dl = Dot(point=pos_dl, color=RED)
        dot_nn = Dot(point=pos_nn, color=BLUE)
        dot_rb = Dot(point=pos_rb, color=YELLOW)

        # Step 1: Words appear and transform into dots
        self.play(Write(text_dl), Write(text_nn), Write(text_rb))
        self.wait(1)
        self.play(
            Transform(text_dl, dot_dl),
            Transform(text_nn, dot_nn),
            Transform(text_rb, dot_rb)
        )

        # Step 2: Illustrate similarity/distance with a line
        # Using Line instead of DashedLine as DashedLine is disallowed
        dist_line = Line(pos_dl, pos_nn, color=WHITE)
        self.play(Create(dist_line))
        self.wait(1)

        # Transition to averaging demonstration
        self.play(
            FadeOut(text_dl), 
            FadeOut(text_nn), 
            FadeOut(text_rb),
            FadeOut(dist_line), 
            FadeOut(axes)
        )

        # Step 3: Show the averaging formula
        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}").to_edge(UP)
        self.play(Write(formula))

        # Step 4: Demonstrate averaging (Machine + Learning)
        # Using Arrow instead of Vector as Vector is disallowed
        v_m_end = np.array([-2, 1, 0])
        v_l_end = np.array([2, 1, 0])
        v_avg_end = (v_m_end + v_l_end) / 2

        label_m = Text("Machine", color=RED).scale(0.6).move_to([-3, -1, 0])
        arrow_m = Arrow(start=ORIGIN, end=v_m_end, buff=0, color=RED)
        
        label_l = Text("Learning", color=YELLOW).scale(0.6).move_to([3, -1, 0])
        arrow_l = Arrow(start=ORIGIN, end=v_l_end, buff=0, color=YELLOW)

        self.play(Write(label_m), Create(arrow_m))
        self.play(Write(label_l), Create(arrow_l))
        self.wait(1)

        # Step 5: Final transformation into the average vector
        arrow_avg = Arrow(start=ORIGIN, end=v_avg_end, buff=0, color=GREEN)
        label_avg = Text("Machine Learning", color=GREEN).scale(0.8).move_to([0, -2, 0])

        self.play(
            Transform(arrow_m, arrow_avg),
            Transform(arrow_l, arrow_avg),
            Transform(label_m, label_avg),
            Transform(label_l, label_avg)
        )
        self.wait(2)
