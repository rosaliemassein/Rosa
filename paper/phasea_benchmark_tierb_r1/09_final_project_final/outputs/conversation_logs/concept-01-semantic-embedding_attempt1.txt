from manim import *

class SemanticEmbedding(Scene):
    def construct(self):
        # 1. Create a simulated coordinate system (using Lines since Axes/ThreeDAxes are restricted)
        x_axis = Line(LEFT * 5, RIGHT * 5, color=GRAY)
        y_axis = Line(DOWN * 3.5, UP * 3.5, color=GRAY)
        grid_lines = VGroup()
        for i in range(-5, 6):
            grid_lines.add(Line([i, -3.5, 0], [i, 3.5, 0], color=GRAY, stroke_width=1, stroke_opacity=0.3))
        for i in range(-3, 4):
            grid_lines.add(Line([-5, i, 0], [5, i, 0], color=GRAY, stroke_width=1, stroke_opacity=0.3))
        
        plane = VGroup(grid_lines, x_axis, y_axis)
        self.play(Create(plane))

        # 2. Text Labels for concepts
        dl_label = Text("Deep Learning", font_size=20, color=YELLOW).move_to([2, 1.5, 0])
        nn_label = Text("Neural Networks", font_size=20, color=YELLOW).move_to([2.8, 1.8, 0])
        rob_label = Text("Robotics", font_size=20, color=YELLOW).move_to([-3, -2, 0])

        self.play(Write(dl_label), Write(nn_label), Write(rob_label))
        self.wait(1)

        # 3. Transform labels into Dot objects
        dl_dot = Dot(point=[2, 1, 0], color=BLUE)
        nn_dot = Dot(point=[2.5, 1.2, 0], color=BLUE)
        rob_dot = Dot(point=[-2.5, -1.5, 0], color=RED)

        self.play(
            ReplacementTransform(dl_label, dl_dot),
            ReplacementTransform(nn_label, nn_dot),
            ReplacementTransform(rob_label, rob_dot)
        )

        # 4. Show distance (Using regular Line since DashedLine is restricted)
        # Using a thin line with low opacity to simulate the distance measurement
        dist_line = Line(dl_dot.get_center(), nn_dot.get_center(), color=YELLOW, stroke_width=2)
        dist_label = Text("Small Distance", font_size=16).next_to(dist_line, UP, buff=0.1)
        
        self.play(Create(dist_line), Write(dist_label))
        self.wait(1)
        self.play(FadeOut(dist_line), FadeOut(dist_label))

        # 5. Formula
        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}").scale(0.8)
        formula.to_corner(UL).set_backstroke(color=BLACK, width=3)
        self.play(Write(formula))

        # 6. Averaging Multi-word Phrases
        # Define "Machine" and "Learning" vectors
        m_pos = np.array([-3, 1, 0])
        l_pos = np.array([-1, 2, 0])
        avg_pos = (m_pos + l_pos) / 2

        m_word = Text("Machine", font_size=18).move_to(m_pos + UP * 0.3)
        l_word = Text("Learning", font_size=18).move_to(l_pos + UP * 0.3)
        m_dot = Dot(point=m_pos, color=GREEN)
        l_dot = Dot(point=l_pos, color=GREEN)

        self.play(
            FadeIn(m_word), FadeIn(l_word),
            Create(m_dot), Create(l_dot)
        )
        self.wait(1)

        # Line connecting the two to show the space between them
        connector = Line(m_pos, l_pos, color=WHITE, stroke_width=1, stroke_opacity=0.5)
        self.play(Create(connector))

        # Final average dot and phrase label
        phrase_dot = Dot(point=avg_pos, color=GOLD)
        phrase_label = Text("Machine Learning (Average)", font_size=20, color=GOLD).move_to(avg_pos + DOWN * 0.4)

        self.play(
            ReplacementTransform(VGroup(m_dot, l_dot), phrase_dot),
            ReplacementTransform(VGroup(m_word, l_word), phrase_label),
            FadeOut(connector),
            run_time=2
        )

        self.wait(2)
