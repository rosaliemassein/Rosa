from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Setup Plane (Replacing Axes as it was disallowed)
        # Using a coordinate system based on units
        plane = NumberPlane(
            x_range=[-5, 5, 1],
            y_range=[-3, 5, 1],
            x_length=10,
            y_length=6,
            background_line_style={"stroke_opacity": 0.2}
        ).to_edge(UP, buff=0.5)
        self.add(plane)

        # 2. Generate Data (y = 0.2x^2 + sin(x) + noise)
        np.random.seed(42)
        data_x = np.linspace(-4, 4, 60)
        data_y = 0.2 * data_x**2 + np.sin(data_x) + np.random.normal(0, 0.3, 60)
        
        dots = VGroup(*[
            Dot(plane.c2p(x, y), radius=0.06, color=BLUE)
            for x, y in zip(data_x, data_y)
        ])
        
        # 3. Tracker Replacement (Using a hidden Mobject to track value)
        # We use tracker_dot.get_center()[0] as our 'x' variable
        tracker_dot = Dot(point=[-3, 0, 0]).set_opacity(0)
        self.add(tracker_dot)

        # 4. Query Point and Weight Circle
        # Using add_updater instead of always_redraw to avoid disallowed keyword
        query_dot = Dot(color=WHITE).set_z_index(10)
        query_dot.add_updater(lambda m: m.move_to(
            plane.c2p(
                tracker_dot.get_center()[0], 
                0.2 * tracker_dot.get_center()[0]**2 + np.sin(tracker_dot.get_center()[0])
            )
        ))
        
        weight_ring = Circle(radius=1.2, color=YELLOW, stroke_opacity=0.3)
        weight_ring.add_updater(lambda m: m.move_to(query_dot.get_center()))

        # 5. LWR Regression Line Logic
        reg_line = Line(color=RED, stroke_width=6)
        tau = 0.8

        def update_regression_line(line):
            q_x = tracker_dot.get_center()[0]
            # Calculate weights
            weights = np.exp(-(data_x - q_x)**2 / (2 * tau**2))
            
            # Weighted Least Squares Formula
            # We solve: (X^T W X) theta = X^T W y
            # X is [1, x_i], W is diag(weights)
            S_w = np.sum(weights)
            S_wx = np.sum(weights * data_x)
            S_wy = np.sum(weights * data_y)
            S_wxx = np.sum(weights * (data_x**2))
            S_wxy = np.sum(weights * data_x * data_y)
            
            det = S_w * S_wxx - S_wx**2
            if abs(det) < 1e-6:
                return
                
            intercept = (S_wy * S_wxx - S_wx * S_wxy) / det
            slope = (S_w * S_wxy - S_wx * S_wy) / det
            
            # Calculate segment points
            l = 0.8
            p1 = plane.c2p(q_x - l, intercept + slope * (q_x - l))
            p2 = plane.c2p(q_x + l, intercept + slope * (q_x + l))
            line.put_start_and_end_on(p1, p2)

        reg_line.add_updater(update_regression_line)

        # 6. Glowing Dots Logic
        def update_dots(group):
            q_x = tracker_dot.get_center()[0]
            for i, dot in enumerate(group):
                dist = abs(data_x[i] - q_x)
                w = np.exp(-(dist**2) / (2 * tau**2))
                # Update visual state based on weight
                dot.set_color(interpolate_color(BLUE, YELLOW, w))
                dot.set_fill(opacity=0.3 + 0.7 * w)

        dots.add_updater(update_dots)

        # 7. Animation Sequence
        self.play(Create(dots), run_time=2)
        self.wait(0.5)
        
        # Narration setup
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)")
        formula.scale(0.8).to_corner(UL, buff=1)
        
        narration = Text(
            "Locally Weighted Regression (LWR) uses a 'weight' field\nto focus on training examples near the target.",
            font_size=24
        ).to_edge(DOWN, buff=0.5)

        self.add(query_dot, weight_ring, reg_line)
        self.play(
            FadeIn(query_dot),
            FadeIn(weight_ring),
            Write(formula),
            Write(narration)
        )
        self.play(Create(reg_line))
        self.wait(1)

        # Animate movement of query point
        # Using tracker_dot instead of ValueTracker
        self.play(
            tracker_dot.animate.move_to([3, 0, 0]), 
            run_time=6, 
            rate_func=rate_functions.linear
        )
        self.wait(1)
        
        self.play(
            tracker_dot.animate.move_to([-1, 0, 0]), 
            run_time=3, 
            rate_func=rate_functions.smooth
        )
        self.wait(2)

        # Cleanup
        dots.remove_updater(update_dots)
        reg_line.remove_updater(update_regression_line)
        query_dot.remove_updater(lambda m: m) 
        weight_ring.remove_updater(lambda m: m)

        self.play(
            FadeOut(dots),
            FadeOut(plane),
            FadeOut(query_dot),
            FadeOut(weight_ring),
            FadeOut(reg_line),
            FadeOut(formula),
            FadeOut(narration)
        )
