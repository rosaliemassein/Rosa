from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. PID Formula Reference (Top Left)
        formula = MathTex(
            r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", 
            font_size=24
        ).to_corner(UP + LEFT)
        self.add(formula)

        # 2. Setup the "Mapping" Diagram (Top Half)
        # We build the "Desired Behavior" box and "PID Constants" box manually
        
        # Input side (Desired Behavior)
        in_rect = Rectangle(width=3.8, height=2.2, color=WHITE)
        in_title = Text("Desired Behavior", font_size=24, color=BLUE)
        in_metrics = Text("Rise Time, SSE, Stability", font_size=18)
        in_content = VGroup(in_title, in_metrics).arrange(DOWN, buff=0.3)
        input_group = VGroup(in_rect, in_content).shift(LEFT * 3.5 + UP * 1.5)
        input_tag = Text("INPUT", font_size=20, color=YELLOW).next_to(in_rect, UP, buff=0.1)

        # Output side (PID Constants)
        out_rect = Rectangle(width=3.8, height=2.2, color=WHITE)
        out_title = Text("PID Constants", font_size=24, color=GREEN)
        out_math = MathTex(r"K_p, K_i, K_d", font_size=28)
        out_content = VGroup(out_title, out_math).arrange(DOWN, buff=0.3)
        output_group = VGroup(out_rect, out_content).shift(RIGHT * 3.5 + UP * 1.5)
        output_tag = Text("OUTPUT", font_size=20, color=YELLOW).next_to(out_rect, UP, buff=0.1)

        # Connector arrow
        connector = Arrow(in_rect.get_right(), out_rect.get_left(), color=YELLOW, buff=0.1)
        
        full_diagram = VGroup(input_group, input_tag, output_group, output_tag, connector)

        # --- ANIMATE TOP SECTION ---
        self.play(FadeIn(input_group), FadeIn(input_tag))
        self.play(FadeIn(output_group), FadeIn(output_tag))
        self.play(GrowArrow(connector))
        self.wait(1)

        # "Flips the script": Rotate the diagram 180 degrees
        # We use animate.rotate instead of the Rotate class to avoid restricted keywords
        self.play(full_diagram.animate.rotate(np.pi, axis=[0, 0, 1]))
        
        # Correction: The text is now upside down, so we flip the content groups back
        self.play(
            in_content.animate.rotate(np.pi),
            out_content.animate.rotate(np.pi),
            input_tag.animate.rotate(np.pi),
            output_tag.animate.rotate(np.pi)
        )
        self.wait(1)

        # 3. Motor Velocity Plot (Bottom Half)
        # We avoid 'Axes' and 'DashedLine' as requested by the error constraints
        
        origin_x, origin_y = -5, -2.5
        x_axis = Arrow(start=[origin_x, origin_y, 0], end=[5, origin_y, 0], buff=0, stroke_width=2)
        y_axis = Arrow(start=[origin_x, origin_y, 0], end=[origin_x, 1, 0], buff=0, stroke_width=2)
        
        x_label = Text("Time", font_size=18).next_to(x_axis, RIGHT, buff=0.2)
        y_label = Text("RPM", font_size=18).next_to(y_axis, UP, buff=0.2)
        
        # Constant setpoint line (Manual horizontal line)
        setpoint_height = 1.5 # relative to origin_y
        set_y = origin_y + setpoint_height
        setpoint_line = Line(
            start=[origin_x, set_y, 0], 
            end=[4.5, set_y, 0], 
            color=GRAY, 
            stroke_width=2
        )
        
        # Calculate points for an oscillating curve manually
        # y = 1 - exp(-0.6*t) * cos(3*t)
        curve_points = []
        for i in range(101):
            time_step = i / 10.0 # 0 to 10
            oscillation = 1.0 - np.exp(-0.6 * time_step) * np.cos(3.0 * time_step)
            # Scale and shift to match the visual axis
            px = origin_x + time_step
            py = origin_y + (oscillation * setpoint_height)
            curve_points.append([px, py, 0])
        
        motor_curve = VMobject(color=BLUE)
        motor_curve.set_points_as_corners(curve_points)

        # --- ANIMATE BOTTOM SECTION ---
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint_line))
        self.play(Create(motor_curve), run_time=3)

        # 4. Braced Labels
        # Rise time is approx where curve first crosses setpoint (t ~ 0.52)
        rise_line = Line([origin_x, origin_y - 0.2, 0], [origin_x + 0.52, origin_y - 0.2, 0])
        rise_brace = Brace(rise_line, DOWN, buff=0.1)
        rise_text = Text("Rise Time", font_size=16, color=YELLOW).next_to(rise_brace, DOWN, buff=0.1)
        
        # Steady State Error is the gap at the end
        final_pt = curve_points[-1]
        sse_brace = Brace(
            Line([final_pt[0] + 0.2, final_pt[1], 0], [final_pt[0] + 0.2, set_y, 0]), 
            RIGHT, 
            buff=0.1
        )
        sse_text = Text("Steady State Error", font_size=16, color=RED).next_to(sse_brace, RIGHT, buff=0.1)

        self.play(Create(rise_brace), Write(rise_text))
        self.play(Create(sse_brace), Write(sse_text))

        self.wait(3)
