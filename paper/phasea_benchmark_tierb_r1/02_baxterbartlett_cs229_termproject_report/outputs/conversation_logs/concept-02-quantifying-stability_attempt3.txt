from manim import *

class QuantifyStability(Scene):
    def construct(self):
        # 1. Coordinate System (Manual creation to avoid Axes restriction)
        origin = LEFT * 4.5 + DOWN * 1.5
        x_axis = Line(origin, origin + RIGHT * 5, color=WHITE)
        y_axis = Line(origin, origin + UP * 4, color=WHITE)
        x_label = Text("Time", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("Velocity (RPM)", font_size=20).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        # 2. Velocity Plot (Fluctuating around 30 RPM)
        # We manually map y=2.0 as the 30 RPM mark
        y_vals = [2.1, 1.8, 2.3, 1.9, 2.2, 1.7, 2.0, 2.4, 2.1, 1.8, 2.2, 1.9, 2.0, 2.1, 1.8, 2.3, 1.9, 2.2, 1.7, 2.0]
        points = [origin + RIGHT * (i * 5 / 19) + UP * y_vals[i] for i in range(len(y_vals))]
        
        graph = VGroup()
        for i in range(len(points)-1):
            graph.add(Line(points[i], points[i+1], color=RED, stroke_width=2))
        
        # 3. Highlight Last 20%
        # Total width is 5 units, so last 20% is 1.0 unit
        highlight_rect = Rectangle(
            width=1.0, height=4.0, 
            fill_color=BLUE, fill_opacity=0.3, 
            stroke_width=0
        ).move_to(origin + RIGHT * 4.5 + UP * 2)
        highlight_label = Text("Last 20%", font_size=14, color=BLUE).next_to(highlight_rect, UP)

        # 4. Histogram (on the right side)
        hist_center = RIGHT * 3 + DOWN * 0.5
        bar_heights = [0.4, 0.6, 2.8, 0.7, 0.3]
        bars = VGroup()
        for i, h in enumerate(bar_heights):
            # The middle bar represents the mode (30 RPM)
            bar_color = YELLOW if i == 2 else BLUE
            b = Rectangle(width=0.4, height=h, fill_color=bar_color, fill_opacity=0.8)
            bars.add(b)
        bars.arrange(RIGHT, buff=0.1, aligned_edge=DOWN).move_to(hist_center)
        
        hist_title = Text("Histogram", font_size=20).next_to(bars, UP)
        mode_tag = Text("30 RPM", font_size=14, color=YELLOW).next_to(bars[2], DOWN)

        # 5. Stability Formula
        stability_formula = MathTex(
            r"Stability = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=28
        ).to_edge(DOWN, buff=0.4)

        # --- Animation Sequence ---
        
        # Draw the axes and velocity plot
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(graph), run_time=2)
        
        # Highlight the last 20% (Steady State Error concept)
        self.play(FadeIn(highlight_rect), Write(highlight_text))
        self.wait(0.5)
        
        # Show histogram and mode
        self.play(Create(bars), Write(hist_title), Write(mode_tag))
        self.wait(0.5)
        
        # Show the resulting mathematical definition
        self.play(Write(stability_formula))
        self.wait(3)
