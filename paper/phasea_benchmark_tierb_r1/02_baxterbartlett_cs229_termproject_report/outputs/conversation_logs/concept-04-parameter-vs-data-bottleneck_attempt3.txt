from manim import *

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Formula at the top
        formula = MathTex(r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}").scale(0.8).to_edge(UP, buff=0.3)
        self.add(formula)

        # 2. Manual Plotting Setup (Avoids Axes/BarChart disallowed identifiers)
        def get_pt(origin, x, y):
            return origin + x * 0.7 * RIGHT + y * 0.5 * UP

        def create_custom_plot(origin, title_str, label_str, color):
            # Axes
            xaxis = Line(origin, origin + RIGHT * 4)
            yaxis = Line(origin, origin + UP * 3)
            # Labels
            title = Text(title_str, font_size=24).next_to(yaxis, UP, buff=0.3)
            label = Text(label_str, font_size=20, color=color).next_to(xaxis, DOWN, buff=0.3)
            # Data points
            x_vals = [0.5, 1.5, 2.5, 3.5, 4.5]
            y_vals = [1.0, 3.5, 1.5, 4.5, 2.5]
            dots = VGroup(*[Dot(get_pt(origin, x, y), color=BLUE, radius=0.06) for x, y in zip(x_vals, y_vals)])
            return VGroup(xaxis, yaxis, title, label, dots), x_vals, y_vals

        # Origins for side-by-side plots
        origin_l = LEFT * 5 + DOWN * 0.5
        origin_r = LEFT * 0.3 + DOWN * 0.5

        plot_l_base, x_vals, y_vals = create_custom_plot(origin_l, "Neural Network", "High Variance", RED)
        plot_r_base, _, _ = create_custom_plot(origin_r, "LWR", "Robust / Local Linear", GREEN)

        # 3. Neural Network 'Erratic' Curve
        nn_curve = VMobject(color=RED)
        nn_pts = []
        for i in range(len(x_vals) - 1):
            p1 = get_pt(origin_l, x_vals[i], y_vals[i])
            p2 = get_pt(origin_l, x_vals[i+1], y_vals[i+1])
            # Add wiggles between points
            mid = (p1 + p2) / 2 + (UP if i % 2 == 0 else DOWN) * 1.2
            nn_pts.extend([p1, mid])
        nn_pts.append(get_pt(origin_l, x_vals[-1], y_vals[-1]))
        nn_curve.set_points_smoothly(nn_pts)

        # 4. LWR Piecewise-Linear Curve
        lwr_curve = VMobject(color=GREEN)
        lwr_pts = [get_pt(origin_r, x, y) for x, y in zip(x_vals, y_vals)]
        lwr_curve.set_points_as_corners(lwr_pts)

        # 5. Animation - Plots
        self.play(
            Create(plot_l_base[0]), Create(plot_l_base[1]),
            Create(plot_r_base[0]), Create(plot_r_base[1])
        )
        self.play(Write(plot_l_base[2]), Write(plot_r_base[2]))
        self.play(FadeIn(plot_l_base[4]), FadeIn(plot_r_base[4]))
        self.wait(0.5)
        
        self.play(Create(nn_curve), Write(plot_l_base[3]), run_time=2)
        self.play(Create(lwr_curve), Write(plot_r_base[3]), run_time=2)
        self.wait(1)

        # 6. Manual Bar Chart (at the bottom/right)
        bar_origin = RIGHT * 5 + DOWN * 2.5
        
        # NN Bar
        bar_nn = Rectangle(width=0.8, height=1.21 * 1.5, fill_opacity=0.8, color=RED)
        bar_nn.move_to(bar_origin + LEFT * 0.6, aligned_edge=DOWN)
        val_nn = Text("1.21", font_size=18).next_to(bar_nn, UP, buff=0.1)
        tag_nn = Text("NN", font_size=18).next_to(bar_nn, DOWN, buff=0.1)
        
        # LWR Bar
        bar_lwr = Rectangle(width=0.8, height=0.85 * 1.5, fill_opacity=0.8, color=GREEN)
        bar_lwr.move_to(bar_origin + RIGHT * 0.6, aligned_edge=DOWN)
        val_lwr = Text("0.85", font_size=18).next_to(bar_lwr, UP, buff=0.1)
        tag_lwr = Text("LWR", font_size=18).next_to(bar_lwr, DOWN, buff=0.1)
        
        chart_title = Text("Test MSE", font_size=22).next_to(VGroup(bar_nn, bar_lwr), UP, buff=0.6)

        # 7. Animation - Bars
        self.play(Write(chart_title))
        self.play(
            Create(bar_nn), Write(val_nn), Write(tag_nn),
            Create(bar_lwr), Write(val_lwr), Write(tag_lwr),
            run_time=1.5
        )
        
        # Narration Context
        narration = Text(
            "Neural Network memorizes noise (Overfitting)\nLWR handles small data (Robustness)",
            font_size=18, color=GRAY
        ).to_edge(DOWN, buff=0.2)
        self.play(FadeIn(narration))
        self.wait(3)
