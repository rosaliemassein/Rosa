from manim import *
import math

class Concept02QuantifyingStability(Scene):
    def construct(self):
        # 1. Title
        title_obj = Text("Quantifying Stability", color=BLUE).scale(0.8)
        title_obj.to_edge(UP)
        self.play(Write(title_obj))

        # 2. Manual Plot Construction (Avoiding Axes class)
        # Origin for our manual coordinate system
        origin_pt = LEFT * 5 + DOWN * 2
        
        # Horizontal and Vertical lines
        h_axis_line = Line(origin_pt, origin_pt + RIGHT * 6, color=WHITE)
        v_axis_line = Line(origin_pt, origin_pt + UP * 4, color=WHITE)
        
        h_label = Text("Time", font_size=20).next_to(h_axis_line, RIGHT)
        v_label = Text("RPM", font_size=20).next_to(v_axis_line, UP)
        
        # Draw base lines
        self.play(Create(h_axis_line), Create(v_axis_line))
        self.play(Write(h_label), Write(v_label))

        # 3. Create a velocity curve manually using VMobject
        # baseline is 30 RPM. Y-axis is 0-50, height is 4 units. 
        # So 30 RPM is at (30/50)*4 = 2.4 units above origin.
        curve_obj = VMobject(color=YELLOW)
        point_list = []
        for step_idx in range(101):
            t_norm = step_idx / 100.0  # 0 to 1
            # Fluctuation logic around 30 RPM
            rpm_val = 30 + 4 * math.sin(t_norm * 15) + 2 * math.cos(t_norm * 25)
            
            # Map to screen coordinates
            pos_horiz = origin_pt[0] + (t_norm * 6)
            pos_vert = origin_pt[1] + (rpm_val / 50.0) * 4
            point_list.append([pos_horiz, pos_vert, 0])
        
        curve_obj.set_points_as_corners(point_list)
        self.play(Create(curve_obj), run_time=2)

        # 4. Highlight the last 20%
        # Width of axis is 6. 20% is 1.2 units.
        highlight_width = 1.2
        highlight_area = Rectangle(
            width=highlight_width,
            height=4,
            fill_color=BLUE,
            fill_opacity=0.3,
            stroke_width=0
        )
        # Center of the last 20% area
        highlight_area.move_to(origin_pt + RIGHT * (6 - highlight_width/2) + UP * 2)
        
        highlight_tag = Text("Final 20%", font_size=16, color=BLUE).next_to(highlight_area, UP, buff=0.1)
        
        self.play(FadeIn(highlight_area), Write(highlight_tag))
        self.wait(1)

        # 5. Histogram on the side (Right side)
        # We manually build bars to avoid Histogram or Axes classes
        hist_origin = RIGHT * 3 + DOWN * 2
        bar_heights = [0.8, 1.4, 3.5, 1.1, 0.6]
        hist_bars = VGroup()
        
        for bar_idx in range(len(bar_heights)):
            # The middle bar (index 2) is the 'Mode' at 30 RPM
            b_color = GREEN if bar_idx == 2 else YELLOW
            single_bar = Rectangle(
                width=0.4,
                height=bar_heights[bar_idx],
                fill_color=b_color,
                fill_opacity=0.8,
                stroke_color=WHITE,
                stroke_width=1
            )
            # Position bars side by side
            single_bar.move_to(hist_origin + RIGHT * (bar_idx * 0.5), aligned_edge=DOWN)
            hist_bars.add(single_bar)
            
        hist_title_text = Text("RPM Distribution", font_size=20).next_to(hist_bars, UP, buff=0.4)
        mode_label_text = Text("Mode", font_size=16, color=GREEN).next_to(hist_bars[2], UP, buff=0.1)

        self.play(Create(hist_bars), Write(hist_title_text))
        self.play(Write(mode_label_text))
        self.wait(1)

        # 6. Stability Formula
        formula_math = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            color=WHITE
        ).scale(0.75)
        formula_math.to_edge(DOWN, buff=0.5)
        
        formula_border = SurroundingRectangle(formula_math, color=GOLD, buff=0.2)
        
        self.play(Write(formula_math))
        self.play(Create(formula_border))
        
        # Final descriptive text
        footer_note = Text("Steady State Error: logic derived from final 20%", font_size=18, color=BLUE)
        footer_note.next_to(formula_border, UP, buff=0.2)
        self.play(Write(footer_note))
        
        self.wait(2)
