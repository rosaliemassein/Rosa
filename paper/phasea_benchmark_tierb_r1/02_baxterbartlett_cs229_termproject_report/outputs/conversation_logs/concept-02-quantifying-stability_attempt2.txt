from manim import *
import numpy as np

class ConceptQuantifyingStability(Scene):
    def construct(self):
        # 1. Setup Axes for Velocity Plot
        # Positioned to the left to make room for the histogram
        axes = Axes(
            x_range=[0, 50, 10],
            y_range=[0, 60, 10],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        # 2. Velocity Plot
        # Create a function fluctuating around 30 RPM
        # Use 't' as the variable name to avoid common identifier issues
        def velocity_func(t):
            return 30 + 4 * np.sin(t) + 2 * np.cos(1.2 * t) + 0.5 * np.sin(3 * t)

        velocity_curve = axes.plot(
            velocity_func,
            x_range=[0, 50],
            color=BLUE
        )

        # 3. Highlight the last 20% of the x-axis (from 40 to 50)
        # Calculate positions using the axes' coordinate-to-point method
        p_start = axes.c2p(40, 0)
        p_end = axes.c2p(50, 60)
        
        highlight_rect = Rectangle(
            width=p_end[0] - p_start[0],
            height=p_end[1] - p_start[1],
            fill_color=BLUE,
            fill_opacity=0.2,
            stroke_width=0
        ).move_to(axes.c2p(45, 30))
        
        highlight_label = Text("Final 20%", font_size=18).next_to(highlight_rect, UP)

        # 4. Histogram (Manual Construction with Rectangles)
        # significantly taller bin at 30 RPM (the mode)
        bar_heights = [0.5, 1.2, 3.8, 1.4, 0.6]
        bars = VGroup()
        for height in bar_heights:
            # Create individual bar
            bar_obj = Rectangle(
                width=0.4, 
                height=height, 
                fill_color=BLUE, 
                fill_opacity=0.7,
                stroke_color=WHITE,
                stroke_width=1
            )
            bars.add(bar_obj)
        
        # Arrange bars and position on the right side
        bars.arrange(RIGHT, buff=0.1, aligned_edge=DOWN)
        bars.to_edge(RIGHT, buff=1.0).shift(UP * 0.5)
        
        # Highlight the mode bin (the middle one representing 30 RPM)
        bars[2].set_color(YELLOW)
        
        chart_label = Text("Histogram (RPM Counts)", font_size=20).next_to(bars, UP)

        # 5. Stability Formula
        # Definition: Stability = counts at mode / counts not at mode
        stability_formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=28
        ).next_to(bars, DOWN, buff=1.0)

        # 6. Narration / Voice-over Text
        # Positioned at the bottom
        narration = Text(
            "Stability is the ratio: how often the motor stays at its most common speed\n"
            "versus how often it deviates.",
            font_size=20,
            line_spacing=1.5
        ).to_edge(DOWN, buff=0.4)

        # Animation Sequence
        self.play(Create(axes), Create(velocity_curve))
        self.wait(0.5)
        
        # Highlight last 20%
        self.play(FadeIn(highlight_rect), Write(highlight_label))
        self.wait(0.5)
        
        # Show Histogram
        self.play(Create(bars), Write(chart_label))
        self.play(Indicate(bars[2], color=YELLOW))
        self.wait(0.5)
        
        # Show Formula and Narration
        self.play(Write(stability_formula))
        self.play(Write(narration))
        
        # Closing Wait
        self.wait(3)
