from manim import *
import numpy as np

class InverseMapping(Scene):
    def construct(self):
        # 1. Formula Reference (Top)
        formula = MathTex(
            r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}",
            font_size=36
        ).to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Input and Output Boxes (Middle)
        # Input: Desired Behavior
        input_title = Text("Desired Behavior", font_size=24, color=BLUE)
        input_metrics = VGroup(
            Text("Rise Time", font_size=18),
            Text("Steady State Error", font_size=18),
            Text("Stability", font_size=18)
        ).arrange(DOWN, aligned_edge=LEFT, buff=0.15)
        
        input_content = VGroup(input_title, input_metrics).arrange(DOWN, buff=0.3)
        input_rect = Rectangle(width=3.5, height=2.0, color=WHITE)
        input_group = VGroup(input_rect, input_content)

        # Output: PID Constants
        output_title = Text("PID Constants", font_size=24, color=GREEN)
        output_metrics = MathTex(r"K_p, K_i, K_d", font_size=28)
        
        output_content = VGroup(output_title, output_metrics).arrange(DOWN, buff=0.3)
        output_rect = Rectangle(width=3.0, height=2.0, color=WHITE)
        output_group = VGroup(output_rect, output_content)

        # Layout
        diagram_center = UP * 0.5
        input_group.move_to(LEFT * 3.5 + diagram_center)
        output_group.move_to(RIGHT * 3.5 + diagram_center)
        
        # Connection Arrow
        arrow = Arrow(input_group.get_right(), output_group.get_left(), buff=0.2)
        
        mapping_diagram = VGroup(input_group, output_group, arrow)

        # 3. Motor Velocity Plot (Bottom)
        plot_origin = DOWN * 2.5 + LEFT * 3
        x_axis = Line(plot_origin, plot_origin + RIGHT * 6)
        y_axis = Line(plot_origin, plot_origin + UP * 2.5)
        
        x_label = Text("Time", font_size=16).next_to(x_axis, RIGHT)
        y_label = Text("RPM", font_size=16).next_to(y_axis, UP)
        
        # Setpoint line (manual dashed effect using color/opacity)
        setpoint = Line(
            plot_origin + UP * 1.5, 
            plot_origin + UP * 1.5 + RIGHT * 6, 
            color=RED, 
            stroke_opacity=0.5
        )

        # Generate Curve Points (Underdamped system: 1 - e^-t * cos(3t))
        curve_pts = []
        for step in range(61):
            t_val = step / 10.0
            # Scaling: x by 1, y by 1.5
            y_val = 1.0 - np.exp(-t_val) * np.cos(3.0 * t_val)
            curve_pts.append(plot_origin + RIGHT * t_val + UP * (y_val * 1.5))
        
        curve = VMobject(color=YELLOW).set_points_as_corners(curve_pts)

        # Labels for plot using Brace
        # Rise Time: 0 to ~1.0 on x-axis
        rise_brace = Brace(Line(plot_origin, plot_origin + RIGHT * 1.0), DOWN, buff=0.1)
        rise_text = Text("Rise Time", font_size=14).next_to(rise_brace, DOWN, buff=0.05)
        
        # Steady State Error: end of curve vs setpoint
        sse_line = Line(plot_origin + RIGHT * 5.8 + UP * 1.5, plot_origin + RIGHT * 5.8 + UP * 1.4, color=WHITE)
        sse_brace = Brace(sse_line, RIGHT, buff=0.1)
        sse_text = Text("Steady State Error", font_size=14).next_to(sse_brace, RIGHT, buff=0.05)

        plot_group = VGroup(x_axis, y_axis, x_label, y_label, setpoint, curve, rise_brace, rise_text, sse_brace, sse_text)

        # --- ANIMATIONS ---

        # Phase 1: Show initial setup
        self.play(FadeIn(input_group), FadeIn(output_group))
        self.play(Create(arrow))
        self.wait(1)

        # Phase 2: "Flip the script" - Rotate the mapping diagram 180 degrees
        # We rotate everything, then fix the text orientation for readability
        self.play(
            mapping_diagram.animate.rotate(PI),
            run_time=2
        )
        # Flip internal text back to readable
        self.play(
            input_title.animate.rotate(PI),
            input_metrics.animate.rotate(PI),
            output_title.animate.rotate(PI),
            output_metrics.animate.rotate(PI),
            run_time=0.5
        )
        self.wait(1)

        # Phase 3: Animate the motor plot
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint))
        self.play(Create(curve), run_time=3)
        
        self.play(
            Create(rise_brace), 
            Write(rise_text)
        )
        self.play(
            Create(sse_brace), 
            Write(sse_text)
        )
        
        self.wait(2)
