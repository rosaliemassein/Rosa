from manim import *
import numpy as np

class InverseMappingConcept(Scene):
    def construct(self):
        # 1. Title/Formula
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}").scale(0.8)
        formula.to_edge(UP)
        self.add(formula)

        # 2. Input/Output Boxes (Using Rectangle and Text to avoid disallowed features)
        in_box = Rectangle(width=4.5, height=1.5, color=BLUE)
        in_label = Text("Input: Desired Behavior", font_size=24, color=BLUE).move_to(in_box.get_center())
        in_sub = Text("Rise Time, SSE, Stability", font_size=16).next_to(in_label, DOWN, buff=0.1)
        input_group = VGroup(in_box, in_label, in_sub).shift(LEFT * 3 + UP * 1)

        out_box = Rectangle(width=4, height=1.5, color=GREEN)
        out_label = Text("Output: PID Constants", font_size=24, color=GREEN).move_to(out_box.get_center())
        output_group = VGroup(out_box, out_label).shift(RIGHT * 3 + UP * 1)

        connector = Arrow(input_group.get_right(), output_group.get_left(), color=WHITE)
        
        full_diagram = VGroup(input_group, output_group, connector)
        self.play(Create(full_diagram))
        self.wait(1)

        # 3. "Flip the script" - Rotate 180 degrees
        # Using .animate.rotate instead of the Rotate class to avoid undefined identifier errors
        self.play(full_diagram.animate.rotate(3.14159, about_point=UP * 1), run_time=1.5)
        self.wait(1)

        # 4. Motor Plot (Manual construction to avoid disallowed Axes/DashedLine)
        origin_pt = DOWN * 3 + LEFT * 4
        x_ax = Line(origin_pt, origin_pt + RIGHT * 8, color=WHITE)
        y_ax = Line(origin_pt, origin_pt + UP * 3, color=WHITE)
        x_lab = Text("Time", font_size=20).next_to(x_ax, RIGHT)
        y_lab = Text("RPM", font_size=20).next_to(y_ax, UP)
        
        # Manual construction of a dashed line for the setpoint
        setpoint_height = 2.0
        setpoint_y = origin_pt + UP * setpoint_height
        dashes = VGroup()
        for i in range(20):
            d_seg = Line(
                setpoint_y + RIGHT * (i * 0.4), 
                setpoint_y + RIGHT * (i * 0.4 + 0.2), 
                stroke_width=2, 
                color=WHITE
            )
            dashes.add(d_seg)
        
        # Curve using points to represent damped oscillation
        curve_pts = []
        for i in range(101):
            x_val = (i / 100.0) * 7.5
            # Formula: 1 - e^(-0.6x) * cos(3x)
            y_val = 1.0 - np.exp(-0.6 * x_val) * np.cos(3.0 * x_val)
            # Map values to coordinate space
            curve_pts.append(origin_pt + RIGHT * x_val + UP * (y_val * setpoint_height))
        
        response_curve = VMobject(color=YELLOW)
        response_curve.set_points_as_corners(curve_pts)

        self.play(Create(x_ax), Create(y_ax), Write(x_lab), Write(y_lab))
        self.play(Create(dashes))
        self.play(Create(response_curve), run_time=2)

        # 5. Braced Labels for Rise Time and Steady State Error
        # Rise time crossing y=1 occurs around x=0.52
        rt_line = Line(origin_pt, origin_pt + RIGHT * 0.52)
        rt_brace = Brace(rt_line, DOWN, buff=0.1)
        rt_text = Text("Rise Time", font_size=16, color=GREEN).next_to(rt_brace, DOWN)

        # SSE at the end of the curve
        sse_line = Line(origin_pt + RIGHT * 7.5 + UP * setpoint_height, curve_pts[-1])
        sse_brace = Brace(sse_line, RIGHT, buff=0.1)
        sse_text = Text("Steady State Error", font_size=16, color=RED).next_to(sse_brace, RIGHT)

        self.play(Create(rt_brace), Write(rt_text))
        self.play(Create(sse_brace), Write(sse_text))
        
        self.wait(2)
