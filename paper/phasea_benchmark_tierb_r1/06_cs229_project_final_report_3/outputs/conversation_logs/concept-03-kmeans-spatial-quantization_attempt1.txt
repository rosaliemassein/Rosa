from manim import *
import numpy as np
import random

class Concept03KMeansSpatialQuantization(Scene):
    def construct(self):
        # Create Axes to represent the map (Avoiding NumberPlane as per constraints)
        axes = Axes(
            x_range=[-7, 7, 1],
            y_range=[-4, 4, 1],
            x_length=12,
            y_length=7,
            axis_config={"include_tip": False, "stroke_opacity": 0.3}
        )
        self.add(axes)

        # Simplified outline of Australia using a Polygon
        australia_points = [
            [-3, 1, 0], [-1, 1.5, 0], [1, 1, 0], [2.5, 0.5, 0], 
            [3.5, -1, 0], [3, -2, 0], [1, -2.5, 0], [-1, -2.5, 0], 
            [-2.5, -2, 0], [-3.5, -1, 0], [-4, 0, 0]
        ]
        australia = Polygon(*australia_points, color=BLUE, stroke_width=3).scale(1.2)
        self.add(australia)

        # Generate points inside the region
        num_points = 80
        dots_list = []
        for _ in range(num_points):
            p = [random.uniform(-4, 4), random.uniform(-2.5, 1.5), 0]
            dots_list.append(Dot(p, color=GRAY, radius=0.06))
        
        dots = VGroup(*dots_list)
        self.play(Create(dots))
        self.wait(0.5)

        # Define colors for clusters
        cluster_colors = [RED, YELLOW, GREEN, PINK]
        num_clusters = len(cluster_colors)

        # Create custom Centroids (Crosses made of two Lines)
        def create_centroid_marker(color):
            line1 = Line(0.15 * UL, 0.15 * DR, color=color, stroke_width=5)
            line2 = Line(0.15 * UR, 0.15 * DL, color=color, stroke_width=5)
            return VGroup(line1, line2)

        centroids = VGroup(*[
            create_centroid_marker(cluster_colors[i]).move_to([
                random.uniform(-3, 3), random.uniform(-2, 2), 0
            ])
            for i in range(num_clusters)
        ])

        self.play(LaggedStart(*[Create(c) for c in centroids], lag_ratio=0.2))
        self.wait(1)

        # Animation Loop for K-Means logic
        # 1. Assignment Step (Coloring dots)
        color_anims = []
        for dot in dots:
            dot_pos = dot.get_center()
            distances = [np.linalg.norm(dot_pos - c.get_center()) for c in centroids]
            best_idx = np.argmin(distances)
            color_anims.append(dot.animate.set_color(cluster_colors[best_idx]))
        
        self.play(*color_anims, run_time=1.5)
        self.wait(0.5)

        # 2. Update Step (Moving centroids to mean)
        move_anims = []
        for i in range(num_clusters):
            # Collect points assigned to this cluster
            points_in_cluster = [
                d.get_center() for d in dots 
                if np.array_equal(d.get_color(), Color(cluster_colors[i]).to_rgb())
            ]
            if points_in_cluster:
                target_pos = np.mean(points_in_cluster, axis=0)
                move_anims.append(centroids[i].animate.move_to(target_pos))
        
        # Display the formula
        formula = MathTex(
            r"\mu_{j}:=\frac{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}x^{(i)}}{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}}",
            font_size=34
        ).to_edge(UP).add_background_rectangle()

        self.play(Write(formula))
        self.play(*move_anims, run_time=2)
        self.wait(1)

        # Final Highlight
        highlights = VGroup(*[
            SurroundingRectangle(c, color=WHITE, buff=0.1, stroke_width=2) 
            for c in centroids
        ])
        self.play(Create(highlights))
        self.wait(2)
