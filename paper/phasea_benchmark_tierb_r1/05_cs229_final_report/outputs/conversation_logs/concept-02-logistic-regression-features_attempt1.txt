from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Manual Coordinate System (Replacing Axes/NumberPlane)
        # We'll build a custom grid area for the graph on the right side
        x_start, x_end = 0, 5
        y_start, y_end = -2, 2
        
        x_axis = Line([x_start, y_start, 0], [x_end, y_start, 0], color=WHITE)
        y_axis = Line([x_start, y_start, 0], [x_start, y_end, 0], color=WHITE)
        
        y_label = MathTex("P(y=1)", font_size=24).next_to(y_axis, UP)
        x_label = MathTex("Z", font_size=24).next_to(x_axis, RIGHT)
        
        graph_elements = VGroup(x_axis, y_axis, y_label, x_label)
        self.add(graph_elements)

        # 2. Create Sigmoid Curve using VMobject
        def sigmoid_calc(z_input):
            return 1 / (1 + np.exp(-z_input))

        curve_points = []
        # Map z from -4 to 4 onto the x-coordinate range [0, 5]
        # Map probability 0 to 1 onto the y-coordinate range [-2, 2]
        for z_val in np.linspace(-4, 4, 100):
            mapped_x = (z_val + 4) * (5/8) + x_start
            mapped_y = sigmoid_calc(z_val) * (y_end - y_start) + y_start
            curve_points.append([mapped_x, mapped_y, 0])
        
        sigmoid_curve = VMobject(color=GREEN)
        sigmoid_curve.set_points_as_corners(curve_points)
        self.add(sigmoid_curve)

        # 3. Sliders (Left Side) using Dots and Lines instead of ValueTrackers
        slider1_x = -5
        slider2_x = -3
        
        slider1_line = Line([slider1_x, -2, 0], [slider1_x, 1, 0], color=GRAY)
        slider2_line = Line([slider2_x, -2, 0], [slider2_x, 1, 0], color=GRAY)
        
        # Handles are dots we can animate
        handle1 = Dot(point=[slider1_x, -0.5, 0], color=BLUE)
        handle2 = Dot(point=[slider2_x, -0.5, 0], color=BLUE)
        
        label1 = Text("1-Day T-CSP", font_size=18).next_to(slider1_line, DOWN)
        label2 = Text("30-Day T-CSP", font_size=18).next_to(slider2_line, DOWN)
        
        self.add(slider1_line, slider2_line, handle1, handle2, label1, label2)

        # 4. Logistic Formula
        formula = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=32
        ).to_edge(UP, buff=0.5)
        self.add(formula)

        # 5. Moving Dot on Curve
        # The dot's position is updated every frame based on handle heights
        moving_dot = Dot(color=YELLOW)
        
        def update_dot_position(mobj):
            # Calculate feature values from handle heights (mapped from range -2 to 1)
            # Center of sliders is -0.5. Let's say range is +/- 1.5 units
            h1_val = handle1.get_center()[1] + 0.5
            h2_val = handle2.get_center()[1] + 0.5
            
            # Simple linear sum for Z calculation
            z_sum = (h1_val + h2_val) * 2 
            z_sum = max(-4, min(4, z_sum)) # Clamp to curve range
            
            new_x = (z_sum + 4) * (5/8) + x_start
            new_y = sigmoid_calc(z_sum) * (y_end - y_start) + y_start
            mobj.move_to([new_x, new_y, 0])

        moving_dot.add_updater(update_dot_position)
        
        # Horizontal and Vertical tracking lines
        h_line = Line(color=YELLOW, stroke_width=2)
        v_line = Line(color=YELLOW, stroke_width=2)
        
        h_line.add_updater(lambda m: m.become(Line(
            [x_start, moving_dot.get_center()[1], 0],
            moving_dot.get_center(),
            color=YELLOW, stroke_width=2
        )))
        v_line.add_updater(lambda m: m.become(Line(
            [moving_dot.get_center()[0], y_start, 0],
            moving_dot.get_center(),
            color=YELLOW, stroke_width=2
        )))
        
        self.add(h_line, v_line, moving_dot)

        # 6. Animations
        self.play(handle1.animate.shift(UP * 1.2), run_time=2)
        self.wait(0.5)
        self.play(handle2.animate.shift(UP * 1.2), run_time=2)
        self.wait(0.5)
        self.play(
            handle1.animate.shift(DOWN * 2.0),
            handle2.animate.shift(DOWN * 0.5),
            run_time=2.5
        )
        self.wait(1)

        # 7. Final transformation of formula
        formula_scaled = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \sum \beta_i X_i)}}",
            font_size=32
        ).move_to(formula)
        
        self.play(Transform(formula, formula_scaled))
        self.wait(2)
