from manim import *
import numpy as np

class AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. Setup Layout and Titles
        title_static = Text("Static Benchmark (ASHRAE)").scale(0.6).to_edge(UP).shift(LEFT * 3.5)
        title_adaptive = Text("Logistic Model (Adaptive ML)").scale(0.6).to_edge(UP).shift(RIGHT * 3.5)
        divider = Line(UP * 4, DOWN * 1.5, color=GRAY)
        self.add(title_static, title_adaptive, divider)

        # 2. Floor Plan Grids (Manual building as VGroups)
        def create_floor_grid(center_pos):
            grid = VGroup()
            for row in range(3):
                for col in range(3):
                    cell = Rectangle(width=0.8, height=0.6)
                    cell.move_to(center_pos + np.array([col * 0.9 - 0.9, row * 0.7 - 0.7, 0]))
                    grid.add(cell)
            return grid

        fp_left = create_floor_grid(LEFT * 3.5 + UP * 1)
        fp_right = create_floor_grid(RIGHT * 3.5 + UP * 1)
        self.add(fp_left, fp_right)

        # 3. Static Highlighting (Left side: frozen indices)
        static_indices = [1, 4, 6]
        for idx in static_indices:
            fp_left[idx].set_fill(BLUE, opacity=0.5)
            fp_left[idx].set_stroke(BLUE, 2)

        # 4. Manual Timeline (Manual implementation of NumberLine)
        timeline_start_pt = LEFT * 5 + DOWN * 3.2
        timeline_end_pt = RIGHT * 5 + DOWN * 3.2
        timeline_path = Line(timeline_start_pt, timeline_end_pt, color=WHITE)
        self.add(timeline_path)

        months = ["May", "Jun", "Jul", "Aug", "Sep"]
        for i in range(len(months)):
            pos = timeline_path.point_from_proportion(i / (len(months) - 1))
            tick = Line(pos + UP * 0.1, pos + DOWN * 0.1)
            label = Text(months[i]).scale(0.35).next_to(tick, DOWN)
            self.add(tick, label)

        # Use a dot as a pointer/tracker since ValueTracker is disallowed
        time_pointer = Dot(timeline_start_pt, color=RED, radius=0.1)
        self.add(time_pointer)

        # 5. Manual Graphs (Manual implementation of Axes)
        graph_origin = DOWN * 2.3 + LEFT * 2.5
        x_axis = Line(graph_origin, graph_origin + RIGHT * 5, color=GRAY)
        y_axis = Line(graph_origin, graph_origin + UP * 1.3, color=GRAY)
        graph_label = Text("Missed Requests (Error Rate)").scale(0.3).next_to(y_axis, UP, buff=0.1)
        self.add(x_axis, y_axis, graph_label)

        static_line = VMobject(color=BLUE)
        adaptive_line = VMobject(color=YELLOW)
        # Initialize with origin point
        static_line.set_points_as_corners([graph_origin, graph_origin])
        adaptive_line.set_points_as_corners([graph_origin, graph_origin])
        self.add(static_line, adaptive_line)

        # Storage for graph history for the updater
        self.history_s = [[graph_origin[0], graph_origin[1], 0]]
        self.history_a = [[graph_origin[0], graph_origin[1], 0]]

        # 6. Animation Updaters
        def adaptive_fp_logic(m):
            # Calculate progress [0, 1]
            progress = (time_pointer.get_center()[0] - timeline_start_pt[0]) / 10.0
            
            # Clear all current highlights
            for cell in m:
                cell.set_fill(opacity=0)
                cell.set_stroke(WHITE, 1)
            
            # Change number of active zones based on "Heatwave" (sine curve)
            # More zones in the middle (July), fewer at ends
            intensity = np.sin(progress * PI)
            count = int(2 + 4 * intensity)
            
            # Use pseudo-random flickering
            seed = int(progress * 100)
            for j in range(count):
                idx = (seed + j * 3) % 9
                flicker_opacity = 0.3 + 0.4 * np.abs(np.sin(progress * 40 + j))
                m[idx].set_fill(YELLOW, opacity=flicker_opacity)
                m[idx].set_stroke(YELLOW, 2)

        fp_right.add_updater(adaptive_fp_logic)

        def graph_drawing_logic(m):
            progress = (time_pointer.get_center()[0] - timeline_start_pt[0]) / 10.0
            if progress <= 0: return
            
            curr_x = graph_origin[0] + progress * 5
            
            # Static line: high error, fluctuates slightly
            s_y = graph_origin[1] + 0.7 + 0.15 * np.sin(progress * 8)
            # Adaptive line: lower error, optimized
            a_y = graph_origin[1] + 0.2 + 0.1 * np.cos(progress * 15)
            
            self.history_s.append([curr_x, s_y, 0])
            self.history_a.append([curr_x, a_y, 0])
            
            static_line.set_points_as_corners(self.history_s)
            adaptive_line.set_points_as_corners(self.history_a)

        static_line.add_updater(graph_drawing_logic)

        # Line Labels
        label_s = Text("Static").scale(0.25).set_color(BLUE)
        label_a = Text("ML Model").scale(0.25).set_color(YELLOW)
        
        def update_labels(m):
            m[0].next_to(static_line.get_end(), RIGHT, buff=0.1)
            m[1].next_to(adaptive_line.get_end(), RIGHT, buff=0.1)
        
        labels_group = VGroup(label_s, label_a)
        labels_group.add_updater(update_labels)
        self.add(labels_group)

        # 7. Narration Text
        narration = Text(
            "While static rules stay frozen, the adaptive model recognizes shifts in load.",
            t2c={"static": BLUE, "adaptive": YELLOW}
        ).scale(0.4).to_edge(DOWN, buff=0.1)

        # 8. Execution
        self.play(FadeIn(narration))
        self.play(
            time_pointer.animate.move_to(timeline_end_pt),
            run_time=12,
            rate_func=lambda t: t
        )
        self.wait(2)
