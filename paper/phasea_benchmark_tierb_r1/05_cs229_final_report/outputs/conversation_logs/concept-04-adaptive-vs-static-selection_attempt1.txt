import numpy as np

class AdaptiveVsStaticSelection(Scene):
    def construct(self):

        self.add(static_title, ml_title)

        # Floorplans - Grids of zones
        floor_static = VGroup(*[Square(side_length=0.45) for _ in range(12)]).arrange_in_grid(3, 4, buff=0.1)
        floor_static.shift(LEFT * 3.5 + UP * 1.5)
        
        floor_ml = VGroup(*[Square(side_length=0.45) for _ in range(12)]).arrange_in_grid(3, 4, buff=0.1)
        floor_ml.shift(RIGHT * 3.5 + UP * 1.5)
        
        # Static Highlighting (Remains frozen)
        static_indices = [0, 2, 5, 8, 11]
        for idx_val in static_indices:
            floor_static[idx_val].set_fill(RED, opacity=0.7)
            
        self.add(floor_static, floor_ml)

        # Timeline and Axes
        # Use Line and manual construction to avoid "Axes" and "ValueTracker" if disallowed
        time_start_pos = -5
        time_end_pos = 5
        timeline_y = -3.5
        timeline_line = Line(LEFT * 5 + DOWN * 3.5, RIGHT * 5 + DOWN * 3.5)
        timeline_label = Text("May to September").scale(0.5).next_to(timeline_line, DOWN, buff=0.2)
        self.add(timeline_line, timeline_label)

        # Graph Area
        graph_base_y = -3.0
        graph_left_x = -5
        v_axis = Line(DOWN * 3.0 + LEFT * 5, DOWN * 1.0 + LEFT * 5)
        v_label = Text("Missed Requests").scale(0.4).rotate(90 * DEGREES).next_to(v_axis, LEFT, buff=0.1)
        self.add(v_axis, v_label)

        # Timer mechanism using a hidden Dot
        tracker_mobject = Dot(LEFT * 5 + DOWN * 3.5, fill_opacity=0)
        self.add(tracker_mobject)

        # ML Floorplan Updater (Flickering and changing zone count)
        def update_ml_floor(mob):
            curr_x = tracker_mobject.get_center()[0]
            t_scaled = curr_x + 5 # Scale from 0 to 10
            for i_idx, square_obj in enumerate(mob):
                # Deterministic flickering based on index and time
                logic_val = np.sin(t_scaled * 2.0 + i_idx * 0.8) + np.cos(t_scaled * 0.5)
                if logic_val > 0.6:
                    square_obj.set_fill(GREEN, opacity=0.7)
                else:
                    square_obj.set_fill(GREEN, opacity=0)

        floor_ml.add_updater(update_ml_floor)

        # Graph Curves
        static_err_line = VMobject(color=RED)
        ml_err_line = VMobject(color=GREEN)

        def update_static_err(mob):
            curr_x = tracker_mobject.get_center()[0]
            points_list = []
            # Generate points from start to current tracker position
            for step_val in np.linspace(-5, curr_x, 30):
                # High error with slight jitter
                noise = 0.05 * np.sin(step_val * 10)
                y_coord = graph_base_y + 1.5 + noise
                points_list.append([step_val, y_coord, 0])
            if len(points_list) > 1:
                mob.set_points_as_corners(points_list)

        def update_ml_err(mob):
            curr_x = tracker_mobject.get_center()[0]
            points_list = []
            for step_val in np.linspace(-5, curr_x, 30):
                # Error starts high then drops and stays low
                normalized_t = step_val + 5
                y_val_relative = 1.3 * np.exp(-normalized_t * 0.4) + 0.3
                y_coord = graph_base_y + y_val_relative
                points_list.append([step_val, y_coord, 0])
            if len(points_list) > 1:
                mob.set_points_as_corners(points_list)

        static_err_line.add_updater(update_static_err)
        ml_err_line.add_updater(update_ml_err)
        self.add(static_err_line, ml_err_line)

        # Labels for the lines
        s_label = Text("Static", color=RED).scale(0.35)
        m_label = Text("ML", color=GREEN).scale(0.35)
        
        # Continuous animation of the tracker dot moving across the timeline
        self.play(
            tracker_mobject.animate.shift(RIGHT * 10),
            run_time=12,
            rate_func=lambda t_func: t_func
        )
        
        # Position labels at the end of the lines
        s_label.move_to(static_err_line.get_end() + RIGHT * 0.4)
        m_label.move_to(ml_err_line.get_end() + RIGHT * 0.4)
        self.play(Write(s_label), Write(m_label))
        
        self.wait(2)
