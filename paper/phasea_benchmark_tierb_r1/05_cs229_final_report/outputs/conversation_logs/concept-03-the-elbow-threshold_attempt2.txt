from manim import *

class TheElbowThreshold(Scene):
    def construct(self):
        # 1. Create Axes and Labels
        # We use the standard Axes class as seen in the reference examples.
        main_axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5).shift(DOWN * 0.5)

        x_label_text = Text("Fraction of Zones Flagged", font_size=20)
        x_label_text.next_to(main_axes.x_axis, DOWN)
        y_label_text = Text("Recall", font_size=20).rotate(90 * DEGREES)
        y_label_text.next_to(main_axes.y_axis, LEFT)

        # 2. Define the Recall Curve
        # A concave curve: y = 1 - (1-x)^3.
        def recall_function(x_val):
            return 1.0 - (1.0 - x_val)**3
        
        curve = main_axes.plot(recall_function, x_range=[0, 1], color=RED)

        # 3. Floorplan Simulation
        # Represented as a grid of squares.
        zone_grid = VGroup()
        for row in range(5):
            for col in range(5):
                zone_square = Square(side_length=0.4, fill_opacity=0.2, fill_color=GRAY, stroke_width=1)
                zone_grid.add(zone_square)
        zone_grid.arrange_in_grid(5, 5, buff=0.1).to_edge(RIGHT, buff=1)
        grid_title = Text("Building Floorplan", font_size=24).next_to(zone_grid, UP)

        # 4. Value Tracker for Animation
        # We use ValueTracker to sync the point on the curve with the highlighted zones.
        progress_tracker = ValueTracker(0.0)

        # 5. Moving Point on Curve
        # Highlighted point that follows the recall curve.
        active_point = Dot(color=YELLOW)
        
        def update_active_point(mob):
            current_progress = progress_tracker.get_value()
            target_pos = main_axes.c2p(current_progress, recall_function(current_progress))
            mob.move_to(target_pos)
            
        active_point.add_updater(update_active_point)

        # 6. Floorplan Highlighter Logic
        # Highlights squares based on the fraction flagged.
        def update_zone_grid(mob):
            current_progress = progress_tracker.get_value()
            total_zones = len(mob)
            threshold_index = int(current_progress * total_zones)
            for index, zone in enumerate(mob):
                if index < threshold_index:
                    zone.set_fill(YELLOW, opacity=0.8)
                else:
                    zone.set_fill(GRAY, opacity=0.2)
        
        zone_grid.add_updater(update_zone_grid)

        # 7. Text Elements
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_corner(UL)
        narration = Text(
            "Balancing sensitivity and selectivity.",
            font_size=22
        ).to_edge(UP)

        # 8. Animation Sequence
        self.add(main_axes, x_label_text, y_label_text, formula, grid_title)
        self.play(Create(curve), Write(narration))
        self.add(active_point, zone_grid)
        
        # Animate the threshold moving right (more zones flagged, higher recall)
        self.play(progress_tracker.animate.set_value(0.9), run_time=4)
        self.wait(0.5)
        
        # Animate back to the 'elbow' (around 0.25)
        self.play(progress_tracker.animate.set_value(0.25), run_time=2)
        
        # 9. Highlight the Elbow
        elbow_coord = main_axes.c2p(0.25, recall_function(0.25))
        elbow_highlight = Circle(radius=0.4, color=WHITE, stroke_width=2).move_to(elbow_coord)
        elbow_label = Text("The Elbow", font_size=18).next_to(elbow_highlight, UR)
        
        self.play(Create(elbow_highlight), Write(elbow_label))
        # Simple glow effect using scale
        self.play(elbow_highlight.animate.scale(1.2), run_time=0.4)
        self.play(elbow_highlight.animate.scale(1.0/1.2), run_time=0.4)
        
        self.wait(2)

        # Remove updaters to prevent errors on cleanup
        active_point.remove_updater(update_active_point)
        zone_grid.remove_updater(update_zone_grid)
