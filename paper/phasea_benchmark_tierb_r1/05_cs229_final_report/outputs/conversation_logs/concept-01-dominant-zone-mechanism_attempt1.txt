from manim import *

class DominantZoneMechanism(Scene):
    def construct(self):
        # Define colors manually to ensure compatibility
        COLOR_BLUE_A = "#C7E9F1"
        COLOR_BLUE_B = "#9CDAD6"
        COLOR_BLUE_C = "#58C4DD"
        COLOR_BLUE_E = "#236B8E"
        COLOR_GREEN_B = "#A6CF8C"

        # 1. Create the AHU (Air Handling Unit)
        ahu_rect = Rectangle(width=3, height=1.5, color=COLOR_BLUE_B)
        ahu_rect.set_fill(COLOR_BLUE_B, opacity=0.2)
        ahu_label = Text("AHU", font_size=24).move_to(ahu_rect.get_center())
        ahu = VGroup(ahu_rect, ahu_label).shift(UP * 2)

        # 2. Create five zone rectangles
        zone_groups = []
        for i in range(5):
            z_rect = Rectangle(width=1.2, height=1, color=COLOR_GREEN_B)
            z_rect.set_fill(COLOR_GREEN_B, opacity=0.3)
            z_label = Text(f"Zone {i+1}", font_size=18).next_to(z_rect, DOWN, buff=0.1)
            zone_groups.append(VGroup(z_rect, z_label))
        
        zones = VGroup(*zone_groups).arrange(RIGHT, buff=0.4).shift(DOWN * 1.5)

        self.add(ahu, zones)
        self.wait(1)

        # 3. Animate heat gradient (orange) growing in two dominant zones
        dominant_indices = [1, 3] # Indices for the "dominant" zones
        other_indices = [0, 2, 4]

        self.play(*[
            zones[i][0].animate.set_fill(ORANGE, opacity=0.8).set_color(ORANGE)
            for i in dominant_indices
        ], run_time=2)
        self.wait(1)

        # 4. Dashed arrows: Cooling Request from orange zones to AHU
        req_arrows = VGroup()
        for i in dominant_indices:
            # Using DashedLine with a tip since DashedArrow is not a standard primitive
            d_line = DashedLine(zones[i].get_top(), ahu.get_bottom(), color=WHITE)
            d_line.add_tip(tip_length=0.2)
            req_arrows.add(d_line)
        
        req_label = Text("Cooling Request", font_size=20).next_to(req_arrows, LEFT, buff=0.5)

        self.play(Create(req_arrows), Write(req_label))
        self.wait(1)

        # 5. Change AHU color to deep blue as it responds
        self.play(
            ahu[0].animate.set_fill(COLOR_BLUE_E, opacity=1).set_color(COLOR_BLUE_E),
            FadeOut(req_arrows),
            FadeOut(req_label)
        )
        self.wait(0.5)

        # 6. Supply air arrows (Deep Blue) flowing back to ALL zones
        supply_arrows = VGroup(*[
            Arrow(ahu.get_bottom(), zones[i].get_top(), color=COLOR_BLUE_C, buff=0.1)
            for i in range(5)
        ])
        supply_label = Text("Supply Air (Cold)", font_size=20, color=COLOR_BLUE_C).next_to(supply_arrows, RIGHT, buff=0.2)

        self.play(Create(supply_arrows), Write(supply_label))
        self.wait(1)

        # 7. Green zones turn a 'chilled' white-blue color to signify overcooling
        self.play(*[
            zones[i][0].animate.set_fill(COLOR_BLUE_A, opacity=0.7).set_color(COLOR_BLUE_A)
            for i in other_indices
        ])
        
        # Dominant zones reach desired state
        self.play(*[
            zones[i][0].animate.set_fill(COLOR_BLUE_C, opacity=0.5).set_color(COLOR_BLUE_C)
            for i in dominant_indices
        ])

        self.wait(2)
