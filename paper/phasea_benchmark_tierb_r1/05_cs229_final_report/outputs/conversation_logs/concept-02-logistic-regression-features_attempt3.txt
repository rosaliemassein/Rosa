import math
from manim import *

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Custom Axes (replacing disallowed Axes/NumberPlane)
        # Position axes on the right side of the screen
        origin_point = RIGHT * 3 + DOWN * 1.5
        x_axis_line = Line(origin_point + LEFT * 2, origin_point + RIGHT * 2, color=WHITE)
        y_axis_line = Line(origin_point, origin_point + UP * 3, color=WHITE)
        x_label = MathTex("X_{sum}").next_to(x_axis_line, DOWN)
        y_label = MathTex("P(y=1)").next_to(y_axis_line, LEFT)
        axes_elements = VGroup(x_axis_line, y_axis_line, x_label, y_label)
        self.add(axes_elements)

        # 2. Sigmoid Curve (drawn manually using VMobject points)
        sigmoid_curve = VMobject(color=ORANGE)
        curve_points = []
        # Calculate points for x from -2.5 to 2.5
        for i in range(-25, 26):
            val_x = i * 0.1
            # Standard sigmoid: 1 / (1 + exp(-x))
            # Scaled for the visual: y is 0 to 1, mapped to 0 to 3 units high
            val_y = 1.0 / (1.0 + math.exp(-val_x))
            curve_points.append(origin_point + val_x * RIGHT + val_y * 3.0 * UP)
        
        sigmoid_curve.set_points_as_corners(curve_points)
        self.add(sigmoid_curve)

        # 3. Sliding Bars on the left
        # Backgrounds for bars
        bar1_bg = Rectangle(width=0.6, height=3, color=WHITE, stroke_opacity=0.3).move_to(LEFT * 4 + DOWN * 0.5)
        bar2_bg = Rectangle(width=0.6, height=3, color=WHITE, stroke_opacity=0.3).move_to(LEFT * 2.5 + DOWN * 0.5)
        self.add(bar1_bg, bar2_bg)

        # The actual interactive bars (using height to represent feature value)
        # Initial height 1.5 (middle of the 0-3 range)
        bar1 = Rectangle(width=0.6, height=1.5, fill_opacity=0.8, fill_color=YELLOW, stroke_width=0)
        bar1.move_to(bar1_bg.get_bottom(), aligned_edge=DOWN)
        
        bar2 = Rectangle(width=0.6, height=1.5, fill_opacity=0.8, fill_color=YELLOW, stroke_width=0)
        bar2.move_to(bar2_bg.get_bottom(), aligned_edge=DOWN)
        self.add(bar1, bar2)

        # Labels for the features
        label1 = MathTex("X_{1d}").next_to(bar1_bg, DOWN)
        label2 = MathTex("X_{30d}").next_to(bar2_bg, DOWN)
        desc1 = Text("1-Day T-CSP", font_size=16).next_to(label1, DOWN, buff=0.1)
        desc2 = Text("30-Day T-CSP", font_size=16).next_to(label2, DOWN, buff=0.1)
        self.add(label1, label2, desc1, desc2)

        # 4. Moving Dot on the Curve
        moving_dot = Dot(color=BLUE, radius=0.12)
        self.add(moving_dot)

        # Updater to link bar heights to dot position on curve
        def update_dot(mobj):
            # Calculate feature values from bar heights (0 to 3 units)
            # Map height [0, 3] to value [-2.5, 2.5]
            f1 = bar1.get_height() - 1.5
            f2 = bar2.get_height() - 1.5
            # Simplified sum for visualization
            total_sum = f1 + f2
            # Clamp sum to curve bounds
            if total_sum > 2.5: total_sum = 2.5
            if total_sum < -2.5: total_sum = -2.5
            
            # Re-calculate sigmoid probability
            prob_val = 1.0 / (1.0 + math.exp(-total_sum))
            # Move dot: origin + x_offset + y_offset
            mobj.move_to(origin_point + total_sum * RIGHT + prob_val * 3.0 * UP)

        moving_dot.add_updater(update_dot)

        # 5. Formulas and Text Transitions
        formula_initial = MathTex("X = [X_{1d}, X_{30d}]").to_edge(UP, buff=0.5)
        formula_logistic = MathTex(
            "P(y=1) = \\frac{1}{1 + e^{-(\\beta_0 + \\beta_1 X_{1d} + \\beta_{30} X_{30d})}}"
        ).scale(0.9).to_edge(UP, buff=0.5)

        self.play(Write(formula_initial))
        self.wait(1)
        self.play(Transform(formula_initial, formula_logistic))
        self.wait(0.5)

        # 6. Interactive Animation: Sliding Bars
        # Scenario A: High 1-day gap, Low 30-day gap
        self.play(
            bar1.animate.stretch_to_fit_height(2.8, about_edge=DOWN),
            bar2.animate.stretch_to_fit_height(0.5, about_edge=DOWN),
            run_time=3
        )
        self.wait(1)

        # Scenario B: Low 1-day, High 30-day
        self.play(
            bar1.animate.stretch_to_fit_height(0.4, about_edge=DOWN),
            bar2.animate.stretch_to_fit_height(2.6, about_edge=DOWN),
            run_time=3
        )
        self.wait(1)

        # Scenario C: Both high (Strong cooling load signal)
        self.play(
            bar1.animate.stretch_to_fit_height(2.2, about_edge=DOWN),
            bar2.animate.stretch_to_fit_height(2.2, about_edge=DOWN),
            run_time=2
        )
        self.wait(2)
