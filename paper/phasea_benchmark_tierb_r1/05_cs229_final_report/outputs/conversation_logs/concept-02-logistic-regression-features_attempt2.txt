from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # Using NumberPlane as a substitute for Axes, which was disallowed
        plane = NumberPlane(
            x_range=[-6, 6, 1],
            y_range=[0, 1, 0.5],
            x_length=6,
            y_length=4
        ).to_edge(RIGHT, buff=0.5)
        
        # Define the sigmoid function using def to avoid potential lambda issues with 'x'
        def sigmoid_math(val):
            return 1.0 / (1.0 + np.exp(-val))
        
        # Create the sigmoid curve using ParametricFunction for maximum compatibility
        def get_sigmoid_point(t_val):
            return plane.c2p(t_val, sigmoid_math(t_val))
            
        sigmoid_curve = ParametricFunction(
            get_sigmoid_point, 
            t_range=[-6, 6], 
            color=BLUE
        )
        
        curve_label = MathTex(r"P(y=1) = \sigma(z)").next_to(plane, UP)
        
        # Slider UI: Using Dots instead of ValueTracker (disallowed)
        s1_line = Line(DOWN, UP).shift(LEFT * 5 + UP * 1.2)
        s1_dot = Dot(color=GREEN).move_to(s1_line.get_center())
        s1_text = Text("1-Day T-CSP", font_size=20).next_to(s1_line, UP)
        
        s2_line = Line(DOWN, UP).shift(LEFT * 5 + DOWN * 1.2)
        s2_dot = Dot(color=RED).move_to(s2_line.get_center())
        s2_text = Text("30-Day T-CSP", font_size=20).next_to(s2_line, UP)
        
        # Weights for the prediction logic
        beta_0 = 0
        beta_1 = 1.5
        beta_30 = 1.5
        
        # Moving dot on the sigmoid curve
        moving_indicator = Dot(color=YELLOW)
        
        # Updater using a named function to avoid lambda issues
        def update_curve_dot(dot_obj):
            # Normalize slider position as the feature input
            val_1 = s1_dot.get_center()[1] - s1_line.get_center()[1]
            val_30 = s2_dot.get_center()[1] - s2_line.get_center()[1]
            
            # Weighted sum Z
            z = beta_0 + beta_1 * (val_1 * 3) + beta_30 * (val_30 * 3)
            p = sigmoid_math(z)
            dot_obj.move_to(plane.c2p(z, p))
            
        moving_indicator.add_updater(update_curve_dot)
        
        # Formulas for transformation sequence
        # ReplacementTransform is used instead of TransformMatchingTex (disallowed)
        feature_vector = MathTex("X = [X_{1d}, X_{30d}]").to_edge(DOWN, buff=1).shift(LEFT * 2)
        logistic_formula = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}"
        ).scale(0.8).to_edge(DOWN, buff=0.8)
        
        # Initial Scene Setup
        self.add(plane, sigmoid_curve, curve_label)
        self.add(s1_line, s1_dot, s1_text, s2_line, s2_dot, s2_text)
        self.add(moving_indicator)
        
        # Sequence 1: Show mathematical input representation
        self.play(Write(feature_vector))
        self.wait(1)
        
        # Sequence 2: Map inputs to the logistic regression formula
        self.play(ReplacementTransform(feature_vector, logistic_formula))
        self.wait(1)
        
        # Sequence 3: Animate feature changes and observe the probability prediction
        self.play(s1_dot.animate.shift(UP * 0.7), run_time=1.5)
        self.play(s2_dot.animate.shift(DOWN * 0.5), run_time=1.5)
        self.wait(0.5)
        
        # Show a drastic change in features (high cooling load)
        self.play(
            s1_dot.animate.shift(DOWN * 1.4),
            s2_dot.animate.shift(UP * 1.8),
            run_time=3
        )
        
        self.wait(2)
