import numpy as np
from manim import *

class AdaptiveVSStaticSelection(Scene):
    def construct(self):
        # Define Basic Colors
        color_benchmark = BLUE
        color_adaptive = GREEN
        color_highlight = RED
        color_base = WHITE
        color_time = YELLOW

        # 1. Header Labels
        text_static_header = Text("Static (ASHRAE)", font_size=24).shift(LEFT * 3.5 + UP * 3.5)
        text_adaptive_header = Text("Logistic Model (Adaptive)", font_size=24).shift(RIGHT * 3.5 + UP * 3.5)
        self.add(text_static_header, text_adaptive_header)

        # 2. Floorplan Construction
        def create_floor_grid():
            grid_group = VGroup()
            for _ in range(3):
                row_group = VGroup(*[Square(side_length=0.5, fill_opacity=0.2, color=color_base) for _ in range(3)])
                row_group.arrange(RIGHT, buff=0.1)
                grid_group.add(row_group)
            grid_group.arrange(DOWN, buff=0.1)
            return grid_group

        floor_left_grid = create_floor_grid().shift(LEFT * 3.5 + UP * 1.5)
        floor_right_grid = create_floor_grid().shift(RIGHT * 3.5 + UP * 1.5)
        
        # Static highlights: Middle column remains frozen as "problem zones"
        for row_idx in range(3):
            floor_left_grid[row_idx][1].set_fill(color_highlight, opacity=0.8)
        
        self.add(floor_left_grid, floor_right_grid)

        # 3. Timeline Setup
        timeline_path = Line(LEFT * 5, RIGHT * 5).shift(DOWN * 0.5)
        text_may = Text("May", font_size=20).next_to(timeline_path, DOWN, aligned_edge=LEFT)
        text_sep = Text("September", font_size=20).next_to(timeline_path, DOWN, aligned_edge=RIGHT)
        timeline_cursor = Dot(color=color_time).move_to(timeline_path.get_start())
        self.add(timeline_path, text_may, text_sep, timeline_cursor)

        # 4. Manual Graph (Avoiding Axes class)
        axis_x = Line(LEFT * 4, RIGHT * 4).shift(DOWN * 3.0)
        axis_y = Line(DOWN * 3.0, DOWN * 1.5).shift(LEFT * 4)
        label_recall = Text("Missed Requests (Recall Error)", font_size=18).next_to(axis_y, UP, buff=0.2).shift(RIGHT * 2)
        self.add(axis_x, axis_y, label_recall)

        graph_path_static = VMobject(color=color_benchmark)
        graph_path_ml = VMobject(color=color_adaptive)
        self.add(graph_path_static, graph_path_ml)

        # 5. Animation Logic and Updaters
        def get_current_progress():
            # Maps cursor X position [-5, 5] to [0, 1]
            return (timeline_cursor.get_center()[0] + 5) / 10.0

        def update_adaptive_floor(mob_obj):
            prog_val = get_current_progress()
            for r_idx in range(3):
                for c_idx in range(3):
                    # Pseudo-random logic based on time and index to flicker
                    noise_val = np.sin(prog_val * 25 + r_idx * 3 + c_idx)
                    # Threshold lowers in the middle of the timeline (heatwaves/stress)
                    # to show more zones flagged, then increases at the ends
                    dynamic_threshold = 0.7 - 0.5 * np.sin(prog_val * np.pi)
                    if noise_val > dynamic_threshold:
                        mob_obj[r_idx][c_idx].set_fill(color_highlight, opacity=0.8)
                    else:
                        mob_obj[r_idx][c_idx].set_fill(color_base, opacity=0.2)
        
        floor_right_grid.add_updater(update_adaptive_floor)

        def update_path_static(mob_obj):
            prog_val = get_current_progress()
            if prog_val < 0.01:
                mob_obj.set_points_as_corners([axis_y.get_start(), axis_y.get_start() + [0.01, 0, 0]])
                return
            points_list = []
            steps_count = int(prog_val * 50) + 1
            for i_step in range(steps_count):
                step_p = i_step / 50.0
                pos_x_coord = -4.0 + (step_p * 8.0)
                # Static error is high and fluctuates slightly
                pos_y_coord = -1.8 + 0.05 * np.sin(step_p * 12)
                points_list.append([pos_x_coord, pos_y_coord, 0])
            mob_obj.set_points_as_corners(points_list)

        def update_path_ml(mob_obj):
            prog_val = get_current_progress()
            if prog_val < 0.01:
                mob_obj.set_points_as_corners([axis_y.get_start(), axis_y.get_start() + [0.01, 0, 0]])
                return
            points_list = []
            steps_count = int(prog_val * 50) + 1
            for i_step in range(steps_count):
                step_p = i_step / 50.0
                pos_x_coord = -4.0 + (step_p * 8.0)
                # ML error starts lower and decreases as it learns/adapts
                pos_y_coord = -2.1 - (0.7 * step_p) + 0.05 * np.cos(step_p * 15)
                points_list.append([pos_x_coord, pos_y_coord, 0])
            mob_obj.set_points_as_corners(points_list)

        graph_path_static.add_updater(update_path_static)
        graph_path_ml.add_updater(update_path_ml)

        # 6. Animation Execution
        # Move timeline across (May to September)
        self.play(
            timeline_cursor.animate(rate_func=lambda t_val: t_val).move_to(timeline_path.get_end()),
            run_time=12
        )
        
        # Cleanup updaters for static frame
        floor_right_grid.remove_updater(update_adaptive_floor)
        graph_path_static.remove_updater(update_path_static)
        graph_path_ml.remove_updater(update_path_ml)

        # 7. Final Summary
        text_conclusion = Text("Adaptive ML: Higher performance, lower energy waste.", 
                               font_size=22, color=color_time).to_edge(UP, buff=0.8)
        self.play(Write(text_conclusion))
        self.wait(2)
