from manim import *

class DominantZoneMechanism(Scene):
    def construct(self):
        # 1. Setup AHU and Zones
        # Use standard names or hex codes to ensure compatibility
        ahu = Rectangle(height=1.5, width=3, color=BLUE, fill_opacity=0.8).shift(UP * 2)
        ahu_label = Text("AHU", font_size=24).move_to(ahu.get_center())
        ahu_group = VGroup(ahu, ahu_label)

        # Create 5 zones
        zones = VGroup(*[
            Rectangle(height=1, width=1.5, color=GREEN, fill_opacity=0.5) for _ in range(5)
        ]).arrange(RIGHT, buff=0.5).shift(DOWN * 1.5)
        
        zone_labels = VGroup(*[
            Text(f"Zone {i+1}", font_size=18).move_to(zones[i].get_center())
            for i in range(5)
        ])

        self.play(FadeIn(ahu_group), FadeIn(zones), FadeIn(zone_labels))
        self.wait(1)

        # 2. Heat gradient growing in two dominant zones (Indices 1 and 3)
        dominant_indices = [1, 3]
        self.play(
            *[zones[idx].animate.set_fill(ORANGE, opacity=0.8).set_stroke(ORANGE) for idx in dominant_indices],
            run_time=2
        )
        self.wait(0.5)

        # 3. Cooling Request Arrows (Dashed)
        # We use DashedLine + add_tip since DashedArrow is not standard in all versions
        request_arrows = VGroup(*[
            DashedLine(
                start=zones[idx].get_top(), 
                end=ahu.get_bottom(), 
                color=ORANGE,
                dash_length=0.1
            ).add_tip()
            for idx in dominant_indices
        ])
        request_label = Text("Cooling Request", font_size=20, color=ORANGE).move_to(ORIGIN)

        self.play(Create(request_arrows), Write(request_label))
        self.wait(1)

        # 4. AHU changes to deep blue
        # Using hex for deep blue: #00008B
        self.play(
            ahu.animate.set_color("#00008B").set_fill("#00008B"),
            FadeOut(request_label),
            run_time=1
        )

        # 5. Supply Air arrows flowing back to ALL zones
        supply_arrows = VGroup(*[
            Arrow(
                start=ahu.get_bottom(), 
                end=zones[i].get_top(), 
                buff=0.1, 
                color=BLUE
            )
            for i in range(5)
        ])
        supply_label = Text("Supply Air (Colder)", font_size=20, color=BLUE).move_to(ORIGIN)

        self.play(Create(supply_arrows), Write(supply_label))
        self.wait(1)

        # 6. Overcooling effect: Green zones turn chilled white-blue
        # White-blue hex: #E0FFFF
        overcooling_anims = []
        for i in range(5):
            if i not in dominant_indices:
                overcooling_anims.append(
                    zones[i].animate.set_fill("#E0FFFF", opacity=0.8).set_stroke(WHITE)
                )
            else:
                # Dominant zones return to a satisfied state (green-ish)
                overcooling_anims.append(
                    zones[i].animate.set_fill(GREEN, opacity=0.5).set_stroke(GREEN)
                )

        self.play(
            *overcooling_anims, 
            FadeOut(supply_arrows), 
            FadeOut(request_arrows), 
            FadeOut(supply_label)
        )
        self.wait(2)
