from manim import *
import random

class ConceptPileGeneration(Scene):
    def construct(self):
        # 1. Canvas Setup
        canvas = Rectangle(color=WHITE, width=8, height=6)
        canvas_label = Text("Synthetic Canvas", font_size=24).next_to(canvas, UP)
        self.add(canvas, canvas_label)

        # 2. Define Segmented Trash Icons (Templates)
        # Using basic colors to avoid undefined identifier errors
        bottle = RoundedRectangle(corner_radius=0.1, height=0.7, width=0.3, color=BLUE, fill_opacity=0.8)
        can = Circle(radius=0.25, color=GRAY, fill_opacity=0.8)
        paper = Square(side_length=0.4, color=WHITE, fill_opacity=0.5)
        scrap = Triangle(color=GREEN, fill_opacity=0.8).scale(0.3)
        cardboard = Rectangle(width=0.6, height=0.4, color=BROWN, fill_opacity=0.8)
        
        # Display templates on the left side
        templates = VGroup(bottle, can, paper, scrap, cardboard).arrange(DOWN, buff=0.3).to_edge(LEFT, buff=0.5)
        self.add(templates)

        # 3. Random Number Display Labels (Alternatives to ValueTracker/always_redraw)
        scale_label = Text("Scale: 1.00", font_size=20).to_corner(UR, buff=0.5)
        rot_label = Text("Rotation: 0", font_size=20).next_to(scale_label, DOWN, align_edge=LEFT)
        self.add(scale_label, rot_label)

        # 4. Generate the Pile (9 objects total)
        # Sequence of indices to sample from the 5 templates
        sequence = [0, 1, 2, 3, 4, 0, 1, 2, 3]
        
        for idx in sequence:
            # Generate random transformation values
            s_val = random.uniform(0.6, 1.4)
            r_val = random.uniform(0, 360)
            # Define coordinates within the 8x6 canvas area
            target_x = random.uniform(-3.0, 3.0)
            target_y = random.uniform(-2.0, 2.0)
            
            # Create a copy of the icon from templates
            obj = templates[idx].copy()
            
            # Create updated text labels for the current transformation
            new_scale_text = Text(f"Scale: {s_val:.2f}", font_size=20).move_to(scale_label)
            new_rot_text = Text(f"Rotation: {int(r_val)}", font_size=20).move_to(rot_label)
            
            # Animate the movement and transformation onto the canvas
            # Objects added later will naturally have higher Z-indices (overlap previous ones)
            self.play(
                obj.animate.move_to([target_x, target_y, 0]).scale(s_val).rotate(r_val * 3.1415 / 180.0),
                ReplacementTransform(scale_label, new_scale_text),
                ReplacementTransform(rot_label, new_rot_text),
                run_time=0.4
            )
            
            # Update label references for the next iteration
            scale_label = new_scale_text
            rot_label = new_rot_text

        self.wait(2)
