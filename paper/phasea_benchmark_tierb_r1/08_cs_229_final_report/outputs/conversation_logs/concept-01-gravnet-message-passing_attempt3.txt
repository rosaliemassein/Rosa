from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Formula reference: V_jk = exp(-d_jk^2)
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # 2. Create nodes (ECal hits) in an irregular 2D grid
        # Central node to be highlighted
        center_node = Dot(point=ORIGIN, color=WHITE)
        
        # Neighboring nodes representing other hits
        neighbor_positions = [
            [2.2, 1.2, 0],
            [-1.5, 2.3, 0],
            [-2.5, -0.8, 0],
            [0.2, -2.5, 0],
            [2.8, -1.2, 0],
            [-1.0, 1.8, 0]
        ]
        neighbors = VGroup(*[Dot(point=p, color=BLUE) for p in neighbor_positions])
        
        # All nodes group
        all_nodes = VGroup(center_node, neighbors)

        # 3. Create connections with updaters to respond to distance
        # The connection strength is weighted by the potential exp(-d^2)
        connections = VGroup()
        for neighbor in neighbors:
            # We use a Line to represent the dynamic connection
            line = Line(center_node.get_center(), neighbor.get_center(), color=BLUE)
            
            def update_connection(l, target=neighbor):
                start = center_node.get_center()
                end = target.get_center()
                # Calculate Euclidean distance
                dist = np.linalg.norm(start - end)
                # Calculate strength: V = exp(-d^2). 
                # We scale d by 1.5 so the visual effect is clear in the coordinate space.
                strength = np.exp(-(dist / 1.5)**2)
                
                # Update line position and visual properties
                l.put_start_and_end_on(start, end)
                # Strength controls thickness and opacity
                l.set_stroke(width=strength * 15, opacity=strength)
            
            line.add_updater(update_connection)
            connections.add(line)

        # 4. Animation Sequence
        # Initial display of hits
        self.play(FadeIn(all_nodes))
        self.wait(0.5)
        
        # Highlight the central node and show initial connections
        self.play(Indicate(center_node, color=WHITE))
        self.play(Create(connections))
        self.wait(1)

        # 5. Dynamic Graph Structure: Move nodes in "learned feature space"
        # As nodes move, the updaters will automatically adjust the connection weights
        new_positions = [
            [1.0, 0.4, 0],    # Neighbor 1 moves closer
            [-3.5, 3.0, 0],   # Neighbor 2 moves further
            [-0.8, -0.6, 0],  # Neighbor 3 moves closer
            [0.5, -3.5, 0],   # Neighbor 4 moves further
            [0.7, -0.3, 0],   # Neighbor 5 moves closer
            [-2.5, 1.0, 0]    # Neighbor 6 moves further
        ]
        
        move_actions = [
            neighbors[i].animate.move_to(new_positions[i]) 
            for i in range(len(neighbors))
        ]
        # Also drift the center node slightly
        move_actions.append(center_node.animate.shift(RIGHT * 0.4 + UP * 0.2))
        
        self.play(*move_actions, run_time=5, rate_func=smooth)
        self.wait(2)

        # Final jitter to emphasize the "fluid" nature of the graph connections
        jitter_positions = [
            n.animate.shift(np.array([np.random.uniform(-0.5, 0.5), np.random.uniform(-0.5, 0.5), 0]))
            for n in neighbors
        ]
        self.play(*jitter_positions, run_time=2, rate_func=smooth)
        self.wait(2)
