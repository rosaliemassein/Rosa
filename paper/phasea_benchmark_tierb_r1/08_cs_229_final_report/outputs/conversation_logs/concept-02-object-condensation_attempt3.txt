from manim import *
import numpy as np
import random

class ObjectCondensation(Scene):
    def construct(self):
        # Set seed for reproducibility
        random.seed(42)

        # 1. 2D plane of dots
        # Create a grid of dots representing particle hits
        dots = VGroup(*[
            Dot(radius=0.08, color=PURPLE)
            for _ in range(40)
        ])
        dots.arrange_in_grid(rows=5, cols=8, buff=0.7)
        
        # Randomize positions slightly to look like real hit data
        for dot in dots:
            dot.shift(random.uniform(-0.2, 0.2) * RIGHT + random.uniform(-0.2, 0.2) * UP)
        
        self.add(dots)
        self.wait(1)

        # 2. Beta-charge mechanism: Grow and change color
        # Select specific indices to be high-beta "condensation points"
        condensation_indices = [11, 22, 28]
        
        growth_animations = []
        # We'll store beta values to use for the pull animation later
        beta_values = []
        
        for i, dot in enumerate(dots):
            if i in condensation_indices:
                beta = random.uniform(0.85, 0.98)
            else:
                beta = random.uniform(0.1, 0.4)
            beta_values.append(beta)
            
            # Map beta to color: Purple (low beta) -> Yellow (high beta)
            target_color = interpolate_color(PURPLE, YELLOW, beta)
            # Map beta to size scaling
            target_scale = interpolate(0.8, 2.4, beta)
            
            growth_animations.append(
                dot.animate.set_color(target_color).scale(target_scale)
            )
        
        self.play(AnimationGroup(*growth_animations, lag_ratio=0.02), run_time=2)
        self.wait(1)

        # 3. Gravity Wells and Attraction
        # Create gradient-like circles for the high-beta condensation points
        wells = VGroup()
        pull_animations = []
        
        for idx in condensation_indices:
            center_pos = dots[idx].get_center()
            
            # Draw a concentric gravity well using multiple circles with decreasing opacity
            well = VGroup(*[
                Circle(
                    radius=r, 
                    stroke_width=0, 
                    fill_opacity=0.2 * (1 - r/2), 
                    fill_color=YELLOW
                ).move_to(center_pos)
                for r in np.linspace(0.2, 2.0, 8)
            ])
            wells.add(well)
            
            # For each other dot, if it is close enough, pull it toward this center
            for j, other_dot in enumerate(dots):
                if j == idx:
                    continue
                
                dist = np.linalg.norm(other_dot.get_center() - center_pos)
                if dist < 2.5:
                    # Move towards the condensation point center
                    # Higher beta condensation points pull more effectively
                    strength = 0.6
                    target_pos = interpolate(other_dot.get_center(), center_pos, strength)
                    pull_animations.append(other_dot.animate.move_to(target_pos))

        self.play(FadeIn(wells, lag_ratio=0.1))
        self.play(*pull_animations, run_time=2, rate_func=ease_in_out_expo)
        self.wait(1)

        # 4. MathTex label for the charge calculation
        # q_i = arctanh(beta_i) + q_min
        formula = MathTex(
            r"q_i = \text{arctanh}(\beta_i) + q_{\min}",
            color=WHITE
        ).scale(1.2)
        
        # Add background rectangle for clarity
        formula_bg = SurroundingRectangle(
            formula, 
            color=BLACK, 
            fill_opacity=0.8, 
            fill_color=BLACK, 
            buff=0.3
        )
        formula_group = VGroup(formula_bg, formula).to_edge(DOWN, buff=0.5)
        
        self.play(FadeIn(formula_bg), Write(formula))
        self.wait(3)
