from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # Formula display
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_edge(UP + RIGHT)
        self.add(formula)

        # Create a scatter plot of dots representing ECal cells in an irregular grid
        # Using 2D dots as 3D features are restricted
        dots = VGroup()
        np.random.seed(42)
        for x in range(-3, 4):
            for y in range(-2, 3):
                # Add some jitter to make it "irregular"
                pos = np.array([x * 1.2, y * 1.2, 0]) + np.random.uniform(-0.4, 0.4, 3)
                # Ensure it's not too close to the center
                if np.linalg.norm(pos) < 0.5:
                    pos += np.array([0.6, 0.6, 0])
                dot = Dot(point=pos, radius=0.08, color=BLUE)
                dots.add(dot)

        # Highlight one central dot
        center_dot = Dot(point=ORIGIN, radius=0.12, color=RED)
        
        # Define the connection logic using a helper function for the updater
        connections = VGroup()
        
        def get_distance_weight(p1, p2):
            dist = np.linalg.norm(p1 - p2)
            # Scaling distance so the effect is visible in screen units
            return np.exp(-(dist * 0.7)**2)

        def update_connection(line, target_dot):
            start = center_dot.get_center()
            end = target_dot.get_center()
            weight = get_distance_weight(start, end)
            line.put_start_and_end_on(start, end)
            line.set_stroke(opacity=weight, width=weight * 8)

        for dot in dots:
            line = Line(center_dot.get_center(), dot.get_center(), buff=0, color=WHITE)
            # Using a closure to capture the specific dot
            line.add_updater(lambda mob, d=dot: update_connection(mob, d))
            connections.add(line)

        # Initial layout
        self.add(dots, center_dot)
        self.play(FadeIn(dots), FadeIn(center_dot))
        self.wait(0.5)
        
        # Show connections
        self.play(Create(connections), run_time=2)
        self.wait(1)

        # Animate dots moving to represent "learned features" updating
        # Connections update automatically via their updaters
        move_animations = []
        for dot in dots:
            # Move dots to new random "feature space" positions
            new_pos = dot.get_center() + np.random.uniform(-1.5, 1.5, 3)
            new_pos[2] = 0 # Keep in 2D plane
            move_animations.append(dot.animate.move_to(new_pos))
        
        # Also move the center dot slightly
        move_animations.append(center_dot.animate.shift(RIGHT * 0.5))

        self.play(
            *move_animations,
            run_time=5,
            rate_func=rate_functions.linear
        )
        
        self.wait(1)
        
        # Emphasize the dynamic shift by oscillating positions
        vibration = []
        for dot in dots:
            vibration.append(dot.animate.shift(np.random.uniform(-0.3, 0.3, 3) * np.array([1, 1, 0])))
            
        self.play(
            *vibration,
            run_time=2,
            rate_func=rate_functions.there_and_back
        )
        
        self.wait(2)
