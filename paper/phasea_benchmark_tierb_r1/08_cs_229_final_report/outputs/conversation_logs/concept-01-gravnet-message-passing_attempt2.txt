from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # Create a pseudo-hexagonal grid of dots
        dots = VGroup()
        rows, cols = 5, 7
        for r in range(rows):
            for c in range(cols):
                # Hexagonal grid logic: shift every other row
                x_pos = c - cols / 2 + (0.5 if r % 2 == 1 else 0)
                y_pos = (r - rows / 2) * 0.866
                # Add slight irregularity
                pos = np.array([x_pos, y_pos, 0]) + np.random.uniform(-0.15, 0.15, 3)
                pos[2] = 0 # Ensure it stays 2D
                dots.add(Dot(point=pos, radius=0.08, color=BLUE))

        # Identify central dot
        central_idx = len(dots) // 2
        central_dot = dots[central_idx]
        central_dot.set_color(RED)

        # Formula display
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").scale(1.0)
        formula.to_corner(UL)
        
        # Explain the concept (Goal)
        goal_text = Text("Distance-Weighted Connections", font_size=20)
        goal_text.next_to(formula, DOWN, buff=0.2).align_to(formula, LEFT)

        # Create connections (Arrows) with dynamic updaters
        # We only connect dots within a certain radius for visual clarity
        arrows = VGroup()
        for dot in dots:
            if dot == central_dot:
                continue
            
            # Initial arrow
            arrow = Arrow(
                start=central_dot.get_center(), 
                end=dot.get_center(), 
                buff=0.1, 
                max_tip_length_to_length_ratio=0.2,
                stroke_width=2
            )
            
            # Use an updater to change opacity and thickness based on distance
            def update_arrow(obj, target_dot=dot):
                start = central_dot.get_center()
                end = target_dot.get_center()
                dist = np.linalg.norm(end - start)
                # Apply the potential formula V = exp(-d^2)
                # We scale the distance (dist * 0.7) so the weights look good on screen
                weight = np.exp(-(dist * 0.7)**2)
                
                # Update arrow geometry and style
                obj.put_start_and_end_on(start, end)
                obj.set_stroke(width=weight * 8, opacity=weight)
                # Update tip size relative to weight
                if weight > 0.1:
                    obj.get_tip().set_opacity(weight)
                else:
                    obj.get_tip().set_opacity(0)

            arrow.add_updater(update_arrow)
            arrows.add(arrow)

        # Intro Animation
        self.play(Create(dots), run_time=1.5)
        self.play(central_dot.animate.scale(1.5), Write(formula))
        self.play(FadeIn(goal_text))
        self.wait(0.5)
        
        # Add connections
        self.play(Create(arrows), run_time=2)
        self.wait(1)

        # Movement Phase: "Learned features update"
        # Dots move slightly, and the updaters will recalculate the weights (arrows)
        movements = []
        for dot in dots:
            # Random slight shift to simulate feature space updates
            target_pos = dot.get_center() + np.random.uniform(-1.2, 1.2, 3)
            target_pos[2] = 0 # Keep in 2D
            movements.append(dot.animate.move_to(target_pos))

        self.play(*movements, run_time=5, rate_func=smooth)
        self.wait(2)

        # Highlight the formula one last time
        self.play(Indicate(formula, color=YELLOW, scale_factor=1.2))
        self.wait(2)

        # Cleanup
        self.play(FadeOut(VGroup(dots, arrows, formula, goal_text)))
