from manim import *
import numpy as np

class Concept02ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup background and title/formula
        # We avoid 'Axes' and 'NumberPlane' due to environment restrictions
        x_line = Line(LEFT * 6, RIGHT * 6, color=BLUE, stroke_opacity=0.2)
        y_line = Line(DOWN * 3.5, UP * 3.5, color=BLUE, stroke_opacity=0.2)
        self.add(x_line, y_line)

        # q_i = arctanh(beta_i) + q_min
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula.to_edge(UP)
        self.play(Write(formula))

        # 2. Create Hits (Dots)
        # We simulate the beta-charge mapping visually
        num_dots = 20
        dots = VGroup()
        for _ in range(num_dots):
            # Start dots at random locations
            start_pos = np.array([
                np.random.uniform(-5, 5),
                np.random.uniform(-3, 3),
                0
            ])
            # Initial color is purple (low beta)
            dot = Dot(point=start_pos, radius=0.08, color=PURPLE_E)
            dots.add(dot)

        self.play(FadeIn(dots, lag_ratio=0.1))
        self.wait(0.5)

        # 3. Define Condensation Points (Gravity Wells)
        # These are dots with high beta
        well_centers = [
            np.array([-3, 1, 0]),
            np.array([2, -1.5, 0]),
            np.array([4, 2, 0])
        ]
        
        wells = VGroup()
        center_dots = VGroup()
        
        for pos in well_centers:
            # The "Condensation Point" dot (High Beta = Yellow)
            c_dot = Dot(point=pos, radius=0.2, color=YELLOW)
            center_dots.add(c_dot)
            
            # The "Gravity Well" effect using concentric circles
            well_effect = VGroup()
            for r in [0.5, 1.0, 1.5]:
                circle = Circle(radius=r, color=YELLOW)
                circle.move_to(pos)
                circle.set_stroke(width=2, opacity=1.0 - (r/1.8))
                well_effect.add(circle)
            
            wells.add(well_effect)
            
            # Animate the appearance of the condensation point
            self.play(
                FadeIn(c_dot, scale=0.5),
                FadeIn(well_effect, scale=0.2),
                run_time=0.8
            )

        # 4. Pulling Mechanism
        # Move dots toward their nearest condensation point
        pull_animations = []
        for d in dots:
            d_pos = d.get_center()
            
            # Find closest center
            closest_center = well_centers[0]
            min_dist = np.linalg.norm(d_pos - closest_center)
            for cp in well_centers:
                dist = np.linalg.norm(d_pos - cp)
                if dist < min_dist:
                    min_dist = dist
                    closest_center = cp
            
            # Attraction logic: if within reach, pull it in
            if min_dist < 5.0:
                # Calculate path: move 70% of the way to the center
                target_pos = d_pos + (closest_center - d_pos) * 0.7
                
                # As it moves, color changes toward yellow (increasing beta)
                pull_animations.append(
                    d.animate.move_to(target_pos).set_color(YELLOW_A).set_scale(1.2)
                )

        # Run the "condensation" animation
        if pull_animations:
            self.play(*pull_animations, run_time=3)

        # 5. Final pulsating effect to show potential stability
        self.play(
            *[w.animate.scale(1.1) for w in wells],
            rate_func=lambda t: np.sin(t * PI),
            run_time=2
        )

        self.wait(2)
