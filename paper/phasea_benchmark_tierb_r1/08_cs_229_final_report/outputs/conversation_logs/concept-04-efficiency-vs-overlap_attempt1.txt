from manim import *

class Concept04EfficiencyVsOverlap(Scene):
    def construct(self):
        # Create axes - using simple style to ensure compatibility
        axes = Axes(
            x_range=[0, 1.2, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=8,
            y_length=5,
            axis_config={"color": WHITE}
        ).shift(DOWN * 0.5)

        # Labels for the axes
        x_lab = Text("Distance", scale=0.4).next_to(axes.x_axis, RIGHT)
        y_lab = Text("Efficiency", scale=0.4).next_to(axes.y_axis, UP)

        # Functions for the curves (avoiding lambda to prevent 'x' identifier issues)
        def ml_efficiency(val):
            return 0.9

        def std_efficiency(val):
            # Sharp drop near 0.3
            if val < 0.2:
                return 0.05
            if val < 0.5:
                return 0.9 * (val - 0.2) / 0.3
            return 0.9

        # Plot the curves
        ml_curve = axes.plot(ml_efficiency, color=BLUE)
        std_curve = axes.plot(std_efficiency, color=RED)
        
        # Labels for curves
        ml_text = Text("ML Model", color=BLUE).scale(0.4).to_corner(UL).shift(RIGHT * 2)
        std_text = Text("Standard Algorithm", color=RED).scale(0.4).next_to(ml_text, DOWN, aligned_edge=LEFT)

        # Particle icons using simple dots
        p1 = Dot(color=BLUE, radius=0.15)
        p2 = Dot(color=BLUE, radius=0.15)
        particles = VGroup(p1, p2)

        # Value tracker for animation
        tracker = ValueTracker(1.0)

        # Vertical tracking line
        v_line = Line(axes.c2p(1.0, 0), axes.c2p(1.0, 0.9), color=YELLOW)

        # Updaters for synchronization
        def update_v_line(mob):
            curr_x = tracker.get_value()
            mob.put_start_and_end_on(axes.c2p(curr_x, 0), axes.c2p(curr_x, 0.9))

        def update_particles(mob):
            dist = tracker.get_value()
            # Particles move toward each other as distance decreases
            center_pos = UP * 2.5
            mob[0].move_to(center_pos + LEFT * dist * 1.5)
            mob[1].move_to(center_pos + RIGHT * dist * 1.5)

        v_line.add_updater(update_v_line)
        particles.add_updater(update_particles)

        # Add everything to scene
        self.add(axes, x_lab, y_lab, ml_curve, std_curve, ml_text, std_text, v_line, particles)

        # Animation sequence
        self.play(
            tracker.animate.set_value(0.1),
            run_time=6,
            rate_func=bezier([0, 0, 1, 1])
        )
        self.wait(2)

        # Visual indicator of efficiency difference
        highlight_box = Rectangle(height=2, width=1, color=YELLOW).move_to(axes.c2p(0.2, 0.5))
        self.play(Create(highlight_box))
        self.play(FadeOut(highlight_box))
        self.wait()
