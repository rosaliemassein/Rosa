from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # Display Narration Text and Formula
        voice_text = Text("Mapping hits to a learned coordinate space", font_size=24).to_edge(UP)
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=24
        ).to_edge(DOWN)
        self.add(voice_text, formula)

        # Create the Split-Screen View using Axes (NumberPlane was disallowed)
        left_axes = Axes(
            x_range=[-3, 3, 1], y_range=[-3, 3, 1], 
            x_length=5, y_length=5,
            axis_config={"include_tip": False, "stroke_opacity": 0.5}
        ).shift(LEFT * 3.5)
        
        right_axes = Axes(
            x_range=[-3, 3, 1], y_range=[-3, 3, 1], 
            x_length=5, y_length=5,
            axis_config={"include_tip": False, "stroke_opacity": 0.5}
        ).shift(RIGHT * 3.5)

        left_title = Text("Physical Space", font_size=24).next_to(left_axes, UP)
        right_title = Text("Learned Space", font_size=24).next_to(right_axes, UP)
        
        self.add(left_axes, right_axes, left_title, right_title)

        # Generate dots for physical space (overlapping clusters)
        dots_phys_1 = VGroup()
        dots_phys_2 = VGroup()
        for _ in range(20):
            # Cluster 1 centered near (0,0)
            p1 = left_axes.c2p(np.random.normal(0, 0.5), np.random.normal(0, 0.5))
            # Cluster 2 centered near (0.5, 0.5), creating overlap
            p2 = left_axes.c2p(np.random.normal(0.5, 0.5), np.random.normal(0.5, 0.5))
            dots_phys_1.add(Dot(p1, color=YELLOW, radius=0.06))
            dots_phys_2.add(Dot(p2, color=BLUE, radius=0.06))

        # Define targets for learned space (distinct, separated clusters)
        targets_1 = VGroup()
        targets_2 = VGroup()
        for _ in range(20):
            # Cluster 1 moved to the left in the new space
            t1 = right_axes.c2p(np.random.normal(-1.5, 0.4), np.random.normal(0, 0.4))
            # Cluster 2 moved to the right in the new space
            t2 = right_axes.c2p(np.random.normal(1.5, 0.4), np.random.normal(0, 0.4))
            targets_1.add(Dot(t1, color=YELLOW, radius=0.06))
            targets_2.add(Dot(t2, color=BLUE, radius=0.06))

        # Initial state
        self.add(dots_phys_1, dots_phys_2)
        self.wait(1)

        # Transformation: Move dots from Physical to Learned distributions
        self.play(
            Transform(dots_phys_1, targets_1),
            Transform(dots_phys_2, targets_2),
            run_time=3,
            rate_func=slow_into
        )
        self.wait(0.5)

        # Draw classification circles around the separated clusters
        # Using simple Circle as DashedVMobject was reported as undefined
        circle1 = Circle(radius=1.2, color=WHITE, stroke_width=2).move_to(right_axes.c2p(-1.5, 0))
        circle2 = Circle(radius=1.2, color=WHITE, stroke_width=2).move_to(right_axes.c2p(1.5, 0))
        
        self.play(Create(circle1), Create(circle2))
        self.wait(2)
