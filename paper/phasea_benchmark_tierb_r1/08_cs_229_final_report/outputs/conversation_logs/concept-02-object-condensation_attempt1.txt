from manim import *
import numpy as np

class ObjectCondensationAnimation(Scene):
    def construct(self):
        # 1. Setup the formula for q_i
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula.to_edge(UP, buff=0.5)
        
        # 2. Create a 2D plane of dots
        # We'll use a seed for reproducibility
        np.random.seed(42)
        dots = VGroup()
        num_dots = 40
        
        # We designate 3 specific condensation points (high beta)
        cp_indices = [5, 18, 32]
        
        for i in range(num_dots):
            pos = [np.random.uniform(-5, 5), np.random.uniform(-3, 1), 0]
            # Initial state: small and purple
            dot = Dot(point=pos, radius=0.08, color=PURPLE)
            dots.add(dot)
            
        self.add(dots)
        self.play(Write(formula))
        self.wait(1)
        
        # 3. Simulate Beta-Charge (growth and color change)
        # Based on remarks: change color from purple to bright yellow based on beta
        growth_animations = []
        for i, dot in enumerate(dots):
            if i in cp_indices:
                beta = 0.9
                target_color = YELLOW
                target_scale = 2.5
            else:
                beta = np.random.uniform(0.1, 0.5)
                # Simple color transition logic to avoid interpolate_color if it's missing
                if beta < 0.3:
                    target_color = PURPLE_A
                else:
                    target_color = BLUE_B
                target_scale = 1.0 + beta
            
            growth_animations.append(
                dot.animate.set_color(target_color).scale(target_scale)
            )
            
        self.play(*growth_animations, run_time=2)
        self.wait(1)
        
        # 4. Draw a concentric 'gravity well' (gradient circle effect)
        # Since Surface is disallowed, we use concentric circles to visualize the well
        wells = VGroup()
        for idx in cp_indices:
            center = dots[idx].get_center()
            # Create a set of concentric circles with decreasing opacity
            well = VGroup()
            for r in np.linspace(0.3, 1.8, 8):
                circle = Circle(radius=r, color=YELLOW, stroke_width=2)
                circle.set_stroke(opacity=max(0, 0.5 - (r/4)))
                well.add(circle)
            well.move_to(center)
            wells.add(well)
            
        self.play(FadeIn(wells, scale=0.5), run_time=1.5)
        self.wait(1)
        
        # 5. Pulling low-beta dots toward the center (condensation)
        # Show neighboring dots sliding toward the condensation points
        attraction_animations = []
        for i, dot in enumerate(dots):
            if i not in cp_indices:
                # Find the center of the nearest gravity well
                dot_pos = dot.get_center()
                distances = [np.linalg.norm(dot_pos - dots[idx].get_center()) for idx in cp_indices]
                closest_idx = cp_indices[np.argmin(distances)]
                closest_pos = dots[closest_idx].get_center()
                
                # If dot is close enough to the well, pull it in
                if np.min(distances) < 3.0:
                    # Move closer to the center (70% of the distance)
                    new_pos = dot_pos + 0.7 * (closest_pos - dot_pos)
                    attraction_animations.append(dot.animate.move_to(new_pos))
        
        # Using a standard rate function like smooth to ensure compatibility
        self.play(*attraction_animations, run_time=2, rate_func=smooth)
        self.wait(2)
        
        # 6. Final Highlight
        label = Text("Clustered Particles", font_size=24, color=YELLOW).to_edge(DOWN)
        self.play(Write(label))
        self.play(Indicate(wells), color=YELLOW)
        self.wait(2)
