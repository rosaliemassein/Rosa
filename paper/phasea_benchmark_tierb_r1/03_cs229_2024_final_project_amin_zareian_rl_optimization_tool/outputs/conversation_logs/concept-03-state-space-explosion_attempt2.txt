from manim import *

class ConceptStateSpaceExplosion(Scene):
    def construct(self):
        # 1. Manual Axis Construction
        # Using Line because 'Axes' is restricted in this environment
        x_axis = Line(start=[-5, -2, 0], end=[0, -2, 0], color=WHITE)
        y_axis = Line(start=[-5, -2, 0], end=[-5, 3, 0], color=WHITE)
        
        # Axis Labels
        x_label = Text("Matrix Size (N)", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("States (log10)", font_size=20).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        # 2. Manual Curve Construction
        # Formula: States = 2^(N*N) => log10(States) = N*N * log10(2)
        # N goes from 2 to 6
        # log10(States) goes from 1.2 (2^4) to 10.8 (2^36)
        # We'll map N=[2,6] to X=[-5,0] and log=[0,11] to Y=[-2,3]
        
        curve_points = []
        # log10(2) is approx 0.30103
        log10_2 = 0.30103
        
        for i in range(41):
            n_val = 2.0 + (i * 0.1)
            log_val = (n_val * n_val) * log10_2
            
            # Map n_val (2 to 6) to x_coord (-5 to 0)
            # x = -5 + (n_val - 2) * (5 / 4)
            curr_x = -5.0 + (n_val - 2.0) * 1.25
            
            # Map log_val (0 to 11) to y_coord (-2 to 3)
            # y = -2 + log_val * (5 / 11)
            curr_y = -2.0 + (log_val * 0.4545)
            
            curve_points.append([curr_x, curr_y, 0])
            
        curve = VMobject(color=YELLOW)
        curve.set_points_as_corners(curve_points)
        
        # 3. Formula Display
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").scale(0.8).to_corner(UR, buff=0.5)
        
        # 4. Table Representation
        # A rectangle with internal lines representing rows
        table_center = [3.5, 0, 0]
        table_outline = Rectangle(height=5, width=3, color=WHITE).move_to(table_center)
        table_title = Text("Q-Table", font_size=24).next_to(table_outline, UP)
        
        # Initial sparse lines (representing a small Q-table)
        lines_group = VGroup()
        for i in range(12):
            y_offset = -2.3 + (i * 4.6 / 11.0)
            row_line = Line(
                start=[3.5 - 1.4, y_offset, 0],
                end=[3.5 + 1.4, y_offset, 0],
                stroke_width=1,
                stroke_opacity=0.8
            )
            lines_group.add(row_line)
            
        # 5. Animations
        self.play(
            Create(x_axis), 
            Create(y_axis), 
            Write(x_label), 
            Write(y_label)
        )
        self.play(Write(formula))
        self.play(
            Create(table_outline), 
            Write(table_title), 
            Create(lines_group)
        )
        self.wait(1)
        
        # Animate graph growth
        self.play(Create(curve), run_time=4)
        
        # 6. Representing the State Space Explosion (Blurring effect)
        # We create a solid fill and a dense pack of lines
        blur_fill = Rectangle(
            height=4.8, 
            width=2.8, 
            fill_opacity=0.8, 
            fill_color=BLUE_E, 
            stroke_width=0
        ).move_to(table_center)
        
        dense_lines = VGroup()
        for i in range(100):
            y_offset = -2.4 + (i * 4.8 / 99.0)
            dense_line = Line(
                start=[3.5 - 1.4, y_offset, 0],
                end=[3.5 + 1.4, y_offset, 0],
                stroke_width=0.5,
                stroke_opacity=0.3
            )
            dense_lines.add(dense_line)
            
        self.play(
            Transform(lines_group, dense_lines),
            FadeIn(blur_fill),
            curve.animate.set_stroke(width=8, color=RED),
            run_time=2
        )
        
        # Highlight final point
        final_dot = Dot(curve_points[-1], color=RED, radius=0.1)
        explosion_text = Text("Explosion!", font_size=32, color=RED).next_to(final_dot, UP)
        
        self.play(FadeIn(final_dot), Write(explosion_text))
        self.wait(2)
