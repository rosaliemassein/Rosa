from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Setup Axes (Matrix Size N vs log2 of States, which is N^2)
        ax = Axes(
            x_range=[2, 6, 1],
            y_range=[0, 40, 10],
            x_length=5,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=1)
        
        ax_labels = ax.get_axis_labels(
            x_label=Tex("Size $N$").scale(0.8), 
            y_label=Tex("$\log_2(\text{States})$").scale(0.8)
        )
        
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP)
        self.add(ax, ax_labels, formula)

        # 2. Grid Visualization
        n_tracker = ValueTracker(2)
        
        def get_grid():
            n = int(n_tracker.get_value())
            # Limit n to prevent rendering thousands of squares
            display_n = min(n, 5)
            grid = VGroup(*[
                Square(side_length=1.5/display_n, fill_opacity=0.6, stroke_width=1)
                .set_color(BLUE if (i+j) % 2 == 0 else RED)
                for i in range(display_n) for j in range(display_n)
            ]).arrange_in_grid(rows=display_n, cols=display_n, buff=0.02)
            grid.to_corner(UR, buff=1)
            return grid

        grid_mobject = always_redraw(get_grid)
        grid_label = always_redraw(lambda: 
            Tex(f"${int(n_tracker.get_value())}\\times{int(n_tracker.get_value())}$ Grid")
            .scale(0.8).next_to(grid_mobject, UP)
        )

        # 3. Table/Rows Visualization (Simplified "scrolling" representation)
        table_area = Rectangle(height=3, width=2.5, color=WHITE).next_to(grid_mobject, DOWN, buff=0.8)
        table_title = Tex("Q-Table Rows").scale(0.7).next_to(table_area, UP, buff=0.1)
        
        # Create a group of lines to simulate table rows
        rows = VGroup(*[
            Line(LEFT, RIGHT, stroke_width=1, color=GRAY_A).set_width(2.3)
            for _ in range(20)
        ]).arrange(DOWN, buff=0.15).move_to(table_area)
        
        def update_rows(m, dt):
            # Speed increases with n_tracker
            val = n_tracker.get_value()
            speed = (val ** 2) * 0.2
            m.shift(UP * speed * dt)
            for row in m:
                if row.get_center()[1] > table_area.get_top()[1]:
                    row.move_to(table_area.get_bottom() + UP * 0.1)

        rows.add_updater(update_rows)
        
        # 4. Graphing (y = x^2 represents log2 of states)
        graph = ax.plot(lambda x: x**2, x_range=[2, 6], color=YELLOW)
        dot = always_redraw(lambda: Dot(color=YELLOW).move_to(
            ax.c2p(n_tracker.get_value(), n_tracker.get_value()**2)
        ))

        # 5. Animations
        self.add(grid_mobject, grid_label, table_area, table_title, rows, dot)
        
        # Move from 2x2 to 5x5
        self.play(
            Create(graph),
            n_tracker.animate.set_value(5),
            run_time=5,
            rate_func=linear
        )
        
        # Blur Effect for "Explosion"
        blur = Rectangle(
            height=3, width=2.5, 
            fill_color=BLUE_E, fill_opacity=0.9, 
            stroke_width=0
        ).move_to(table_area)
        
        explosion_text = Text("33 Million+ States", color=RED).scale(0.6).move_to(table_area)
        
        # Final push to 6x6 where it shoots off
        self.play(
            FadeIn(blur),
            Write(explosion_text),
            n_tracker.animate.set_value(6),
            run_time=2
        )
        
        self.wait(2)
