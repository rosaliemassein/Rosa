from manim import *

class Concept02RewardDynamics(Scene):
    def construct(self):
        # 1. Setup Grid (3x3)
        # Pixel arrangement for VGroup.arrange_in_grid:
        # 0 1 2
        # 3 4 5
        # 6 7 8
        # Center is index 4.
        grid = VGroup(*[
            Square(side_length=1).set_fill(RED, opacity=0.7).set_stroke(WHITE, 1)
            for _ in range(9)
        ]).arrange_in_grid(rows=3, cols=3, buff=0.1)
        
        # Initial Logic: 
        # Center (4) is RED.
        # Neighbor (7) is BLUE (creates 1 initial border).
        # Neighbors (1, 3, 5) are RED (no initial border).
        grid[7].set_fill(BLUE)
        grid_group = VGroup(grid).to_edge(LEFT, buff=1.0)
        
        center_pixel = grid[4]
        neighbors_indices = [1, 3, 5, 7]
        neighbors_group = VGroup(*[grid[i] for i in neighbors_indices])
        
        # 2. Formula and Title
        formula = MathTex(r"R_t = \text{Border}_{t} - \text{Border}_{t-1}").to_edge(UP, buff=0.5)
        
        # 3. BarChart Setup
        # Displaying the cumulative reward over time
        chart = BarChart(
            values=[0, 0, 0, 0, 0],
            y_range=[0, 10, 2],
            x_length=5,
            y_length=3,
            y_axis_config={"font_size": 24},
            x_axis_config={"font_size": 24},
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)
        chart_label = Text("Cumulative Reward", font_size=24).next_to(chart, UP)

        # Presentation start
        self.add(grid_group, formula, chart, chart_label)
        self.play(FadeIn(grid_group))
        
        # Highlight center pixel and its 4 neighbors as per remarks
        self.play(center_pixel.animate.set_stroke(YELLOW, 6))
        neighbor_highlighter = SurroundingRectangle(neighbors_group, color=YELLOW, buff=0.05)
        self.play(Create(neighbor_highlighter))
        self.wait(0.5)

        # 4. Define visual border feedback
        # Lost Border: between center (4) and neighbor (7) - will disappear when 4 flips to Blue
        lost_border = Line(
            grid[4].get_bottom() + LEFT * 0.5, 
            grid[4].get_bottom() + RIGHT * 0.5, 
            color=RED, stroke_width=8
        )
        x_mark = Cross(lost_border, stroke_color=RED).scale(0.2)
        
        # New Borders: between center (4) and neighbors (1, 3, 5) - created when 4 flips to Blue
        gain1 = Line(grid[4].get_top() + LEFT * 0.5, grid[4].get_top() + RIGHT * 0.5)
        gain2 = Line(grid[4].get_left() + UP * 0.5, grid[4].get_left() + DOWN * 0.5)
        gain3 = Line(grid[4].get_right() + UP * 0.5, grid[4].get_right() + DOWN * 0.5)
        gained_borders = VGroup(gain1, gain2, gain3).set_color(GREEN).set_stroke(width=8)

        # 5. Animate Flip (Red to Blue)
        # Final Calculation: 3 new borders, 1 lost border. Net change = +2.
        self.play(center_pixel.animate.set_fill(BLUE))
        
        # Show lost border logic
        self.play(Create(lost_border))
        self.play(Create(x_mark))
        self.play(FadeOut(lost_border, x_mark))
        
        # Show gained border highlights
        self.play(Create(gained_borders))
        
        # Floating reward text feedback
        reward_float = Text("+2", color=GREEN).scale(1.2).next_to(center_pixel, UP, buff=0.1)
        self.play(Write(reward_float))
        self.play(reward_float.animate.shift(UP * 1.0).set_opacity(0), run_time=1)
        
        # 6. Update BarChart showing growth
        # Initial step
        self.play(chart.animate.change_bar_values([2, 0, 0, 0, 0]))
        self.wait(0.5)
        
        # Sequence of values growing over time
        self.play(chart.animate.change_bar_values([2, 4, 6, 9, 10]), run_time=2)
        
        self.wait(2)
