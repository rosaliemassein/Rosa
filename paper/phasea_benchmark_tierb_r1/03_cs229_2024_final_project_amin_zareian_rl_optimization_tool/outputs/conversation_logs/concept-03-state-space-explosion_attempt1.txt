from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Setup Axes
        # Since 2^(N^2) grows too fast for a linear scale, we plot y = N^2 
        # which represents the exponent in the Total States formula.
        axes = Axes(
            x_range=[2, 6, 1],
            y_range=[0, 40, 10],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.7)

        x_label = axes.get_x_axis_label("N", edge=DOWN, direction=DOWN).scale(0.8)
        y_label = axes.get_y_axis_label(MathTex(r"\log_2(\text{States})"), edge=LEFT, direction=LEFT).scale(0.7)
        
        # Plotting N^2 curve
        curve = axes.plot(lambda x: x**2, x_range=[2, 6], color=BLUE)
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}", color=YELLOW).to_corner(UL).scale(0.9)

        # 2. Q-Table Visual (2x2 Grid -> 16 states)
        # Using Lines/Rects for better performance than Text
        table_rows = VGroup(*[
            Rectangle(
                height=0.08, 
                width=2.5, 
                fill_opacity=0.8, 
                fill_color=BLUE if i % 2 == 0 else RED,
                stroke_width=0
            ) for i in range(16)
        ]).arrange(DOWN, buff=0.02).shift(RIGHT * 3.5 + UP * 0.5)
        
        table_label = Text("Q-Table (2x2)", font_size=24).next_to(table_rows, UP)
        
        # 3. Animation
        self.add(axes, x_label, y_label, formula)
        self.play(Create(table_rows), Write(table_label))
        self.wait(1)

        # 4. The "Explosion" - Transforming rows into a solid block of color
        # representing the 5x5 grid (2^25 states)
        explosion_block = Rectangle(
            height=5.0,
            width=2.5,
            fill_opacity=1,
            stroke_width=0
        ).move_to(table_rows)
        # Use a gradient to represent a "blur" of countless rows
        explosion_block.set_fill(color=[RED, BLUE, YELLOW, RED], opacity=1)
        
        explosion_label = Text("Q-Table (5x5)\n33M+ Rows", font_size=24, color=RED).next_to(explosion_block, UP)

        # Animate the graph and the table expansion simultaneously
        self.play(
            Create(curve),
            Transform(table_rows, explosion_block),
            Transform(table_label, explosion_label),
            run_time=4,
            rate_func=smooth
        )

        # 5. Final effect: A scrolling/vibrating "blur" to show density
        def vibrate_updater(m):
            m.set_x(3.5 + np.random.uniform(-0.02, 0.02))
            
        table_rows.add_updater(vibrate_updater)
        self.wait(2)
        table_rows.remove_updater(vibrate_updater)
        
        # Show that the curve goes off-screen (Exponential growth)
        extra_curve = axes.plot(lambda x: x**2, x_range=[6, 7], color=RED, stroke_width=6)
        self.play(Create(extra_curve), run_time=1)
        self.wait(1)
