from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup Formula and Title
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}"
        ).to_edge(UP)
        self.add(formula)

        # 2. Create Planning and Deployment zones
        planning_label = Text("Planning", font_size=24).shift(LEFT * 5.5 + UP * 0.5)
        deployment_label = Text("Deployment", font_size=24, color=GOLD).shift(RIGHT * 5.5 + UP * 0.5)
        deployment_icon = Star(color=GOLD, fill_opacity=1).scale(0.3).next_to(deployment_label, DOWN)
        
        # 3. Create three vertical lines for the gates
        gate_positions = [-2.5, 0, 2.5]
        gate_names = ["ReqVer", "ExecVer", "ImpVer"]
        gate_lines = VGroup()
        gate_labels = VGroup()

        for pos, name in zip(gate_positions, gate_names):
            line = Line(UP * 1.5, DOWN * 1.5).move_to(RIGHT * pos)
            label = Text(name, font_size=20).next_to(line, UP)
            gate_lines.add(line)
            gate_labels.add(label)

        self.add(planning_label, deployment_label, deployment_icon, gate_lines, gate_labels)

        # 4. Create the 'Plan' object (a small square)
        plan = Square(side_length=0.4, color=WHITE, fill_opacity=0.5).move_to(LEFT * 5.5)
        self.play(FadeIn(plan))

        # 5. Animation logic to show the loop
        
        # Step 1: Pass first gate
        self.play(plan.animate.move_to(RIGHT * gate_positions[0]))
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.2)

        # Step 2: Fail second gate
        self.play(plan.animate.move_to(RIGHT * gate_positions[1]))
        self.play(plan.animate.set_color(RED))
        
        # Create a curved loop-back arrow using ArcBetweenPoints
        # This is more compatible across Manim versions than CurvedArrow
        start_pt = plan.get_bottom()
        end_pt = planning_label.get_bottom() + DOWN * 0.2
        fail_arrow = ArcBetweenPoints(start_pt, end_pt, angle=PI/2, color=RED).add_tip()
        
        self.play(Create(fail_arrow))
        # Loop back to start
        self.play(
            plan.animate.move_to(LEFT * 5.5).set_color(WHITE),
            run_time=1.5
        )
        self.play(FadeOut(fail_arrow))
        self.wait(0.5)

        # Step 3: Success run through all gates
        for pos in gate_positions:
            self.play(plan.animate.move_to(RIGHT * pos).set_color(GREEN), run_time=0.7)
        
        # Step 4: Reach Deployment
        self.play(plan.animate.move_to(deployment_icon.get_center()))
        self.play(
            Indicate(deployment_icon, color=YELLOW, scale_factor=1.5),
            plan.animate.scale(0).set_opacity(0)
        )
        
        # Highlight completion
        self.play(formula.animate.set_color(YELLOW))
        self.wait(2)
