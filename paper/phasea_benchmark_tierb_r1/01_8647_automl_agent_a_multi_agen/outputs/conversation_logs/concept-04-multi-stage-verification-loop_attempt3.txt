from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup Layout
        # Start Stage (Agent Manager)
        planning_box = RoundedRectangle(height=1.5, width=2.5, corner_radius=0.1, color=BLUE)
        planning_text = Text("Agent Manager\n(Planning)", font_size=20)
        planning_stage = VGroup(planning_box, planning_text).shift(LEFT * 5)
        
        # End Stage (Deployment)
        deployment_hex = RegularPolygon(n=6, color=YELLOW, fill_opacity=0.3)
        deployment_text = Text("Deployment", font_size=20).next_to(deployment_hex, DOWN, buff=0.1)
        deployment_stage = VGroup(deployment_hex, deployment_text).shift(RIGHT * 5.5)

        # Verification Gates (Vertical Lines)
        gate_x_coords = [-2.5, 0, 2.5]
        gate_names = ["ReqVer", "ExecVer", "ImpVer"]
        gates_group = VGroup()
        for x, name in zip(gate_x_coords, gate_names):
            line = Line(UP * 1.5, DOWN * 1.5, color=GREY)
            line.move_to(RIGHT * x)
            label = Text(name, font_size=22).next_to(line, UP)
            gates_group.add(VGroup(line, label))

        self.add(planning_stage, deployment_stage, gates_group)

        # 2. Create the Plan Object (Small Square)
        plan_square = Square(side_length=0.6, fill_color=WHITE, fill_opacity=1, stroke_color=BLUE)
        plan_text = Text("Plan", font_size=16, color=BLACK).move_to(plan_square)
        plan_obj = VGroup(plan_square, plan_text).move_to(planning_stage.get_center())

        # 3. Animation: Gate Failure Loop
        self.play(FadeIn(plan_obj))
        self.wait(0.5)
        
        # Move to first gate and fail
        self.play(plan_obj.animate.move_to(RIGHT * gate_x_coords[0]))
        self.play(plan_square.animate.set_fill(RED))
        
        # Draw feedback loop back to Agent Manager using ArcBetweenPoints + tip
        # This replaces CurvedArrow to avoid undefined identifier issues
        fail_path = ArcBetweenPoints(
            start=RIGHT * gate_x_coords[0] + UP * 0.7,
            end=planning_stage.get_center() + UP * 0.7,
            angle=-PI/2,
            color=RED
        ).add_tip()
        
        feedback_label = Text("Feedback / Revision", font_size=18, color=RED).next_to(fail_path, UP)
        
        self.play(Create(fail_path), Write(feedback_label))
        self.play(plan_obj.animate.move_to(planning_stage.get_center()))
        self.play(FadeOut(fail_path), FadeOut(feedback_label), plan_square.animate.set_fill(WHITE))
        self.wait(0.5)

        # 4. Animation: Successful Path
        # Pass through all gates
        for x in gate_x_coords:
            self.play(plan_obj.animate.move_to(RIGHT * x))
            self.play(plan_square.animate.set_fill(GREEN), run_time=0.3)
        
        # Reach Deployment
        self.play(plan_obj.animate.move_to(deployment_stage.get_center()))
        self.play(FadeOut(plan_obj), deployment_stage.animate.set_color(GREEN).scale(1.1))
        self.wait(0.5)

        # 5. Formula and Goal Display
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}"
        ).scale(0.85).to_edge(DOWN, buff=1.0)
        
        goal_display = Text(
            "Goal: Understand the iterative feedback loop and verification stages.", 
            font_size=20, 
            color=BLUE
        ).to_edge(DOWN, buff=0.4)
        
        self.play(Write(formula))
        self.play(FadeIn(goal_display))
        self.wait(2)
