from manim import *

class Concept02RetrievalAugmentedPlanning(Scene):
    def construct(self):
        # 1. Setup Agent Manager (Central)
        manager_rect = RoundedRectangle(corner_radius=0.1, height=1.5, width=2.5, color=BLUE)
        manager_text = Text("Agent Manager", font_size=24).move_to(manager_rect.get_center())
        agent_manager = VGroup(manager_rect, manager_text)
        agent_manager.center()

        # 2. Setup Knowledge Base (Left)
        # Creating a cloud-like shape with circles
        cloud = VGroup(
            Circle(radius=0.5).shift(LEFT * 0.3),
            Circle(radius=0.6).shift(ORIGIN),
            Circle(radius=0.5).shift(RIGHT * 0.3),
            Circle(radius=0.4).shift(UP * 0.4)
        ).set_fill(WHITE, opacity=0.3).set_stroke(WHITE, 2)
        kb_text = Text("Knowledge Base", font_size=20).next_to(cloud, DOWN)
        knowledge_base = VGroup(cloud, kb_text).to_edge(LEFT, buff=1)

        # 3. Knowledge Retrieval Animation
        knowledge_particle = Text("Knowledge", font_size=18, color=YELLOW)
        knowledge_particle.move_to(knowledge_base.get_center())
        
        # 4. Define Branching Plans (Right)
        start_pt = agent_manager.get_right()
        
        # Plan A - Transformer
        end_a = start_pt + RIGHT * 3 + UP * 2
        path_a = ArcBetweenPoints(start_pt, end_a, angle=-PI/4).add_tip()
        icon_a = Triangle(color=BLUE).scale(0.3).next_to(end_a, RIGHT)
        label_a = Text("Plan A: Transformer", font_size=18).next_to(icon_a, RIGHT)
        plan_group_a = VGroup(path_a, icon_a, label_a)

        # Plan B - CNN
        end_b = start_pt + RIGHT * 3
        path_b = Arrow(start_pt, end_b, buff=0)
        icon_b = Square(color=GREEN).scale(0.3).next_to(end_b, RIGHT)
        label_b = Text("Plan B: CNN", font_size=18).next_to(icon_b, RIGHT)
        plan_group_b = VGroup(path_b, icon_b, label_b)

        # Plan C - Evidence-based
        end_c = start_pt + RIGHT * 3 + DOWN * 2
        path_c = ArcBetweenPoints(start_pt, end_c, angle=PI/4).add_tip()
        icon_c = Circle(color=RED).scale(0.3).next_to(end_c, RIGHT)
        label_c = Text("Plan C: Evidence", font_size=18).next_to(icon_c, RIGHT)
        plan_group_c = VGroup(path_c, icon_c, label_c)

        # 5. Formula
        formula = MathTex(r"\mathbf{P} = \mathcal{A}_{mgr}(\mathrm{RAP}(R))")
        formula.to_edge(UP, buff=0.7)

        # ANIMATION SEQUENCE
        # Step 1: Show Manager and Knowledge Base
        self.play(FadeIn(agent_manager), FadeIn(knowledge_base))
        self.wait(0.5)

        # Step 2: Animate knowledge flow
        self.play(FadeIn(knowledge_particle))
        self.play(
            knowledge_particle.animate.move_to(agent_manager.get_center()),
            path_arc=PI/3,
            run_time=1.5
        )
        self.play(FadeOut(knowledge_particle), agent_manager.animate.set_color(YELLOW))
        self.play(agent_manager.animate.set_color(BLUE))

        # Step 3: Branching into multiple plans
        self.play(Write(formula))
        
        # Create copies of manager to transform into paths
        mgr_copy_a = agent_manager.copy()
        mgr_copy_b = agent_manager.copy()
        mgr_copy_c = agent_manager.copy()

        self.play(
            TransformMatchingShapes(mgr_copy_a, plan_group_a),
            TransformMatchingShapes(mgr_copy_b, plan_group_b),
            TransformMatchingShapes(mgr_copy_c, plan_group_c),
            run_time=2
        )
        
        self.wait(1)

        # Step 4: Highlight parallel exploration
        self.play(
            Indicate(icon_a, scale_factor=1.5),
            Indicate(icon_b, scale_factor=1.5),
            Indicate(icon_c, scale_factor=1.5),
            run_time=2
        )

        self.wait(2)
