from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Formula reference at the top
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}"
        ).scale(0.8).to_edge(UP)
        self.add(formula)

        # 2. Layout Constants
        planning_pos = LEFT * 5
        gate_x_coords = [-2.5, 0, 2.5]
        deployment_pos = RIGHT * 5

        # 3. Static Elements
        planning_label = Text("Planning", font_size=24).move_to(planning_pos + UP * 2)
        deployment_label = Text("Deployment", font_size=24).move_to(deployment_pos + UP * 2)
        deployment_icon = Star(color=GOLD, fill_opacity=1).scale(0.4).move_to(deployment_pos)

        gate_names = ["ReqVer", "ExecVer", "ImpVer"]
        gates = VGroup()
        gate_labels = VGroup()
        for x, name in zip(gate_x_coords, gate_names):
            gate = Line(UP * 1.5, DOWN * 1.5, color=YELLOW, stroke_width=4).move_to(RIGHT * x)
            label = Text(name, font_size=20).next_to(gate, UP)
            gates.add(gate)
            gate_labels.add(label)

        self.add(planning_label, deployment_label, gates, gate_labels, deployment_icon)

        # 4. The Plan Object
        plan = Square(side_length=0.4, fill_opacity=1, color=WHITE).move_to(planning_pos)
        self.play(FadeIn(plan))

        # 5. First Attempt: Hit ImpVer and Fail
        # Move through first two gates
        for i in range(2):
            self.play(plan.animate.move_to(RIGHT * gate_x_coords[i]), run_time=0.8)
            self.play(plan.animate.set_color(GREEN), run_time=0.3)
        
        # Hit third gate and fail
        self.play(plan.animate.move_to(RIGHT * gate_x_coords[2]), run_time=0.8)
        self.play(plan.animate.set_color(RED), run_time=0.4)

        # 6. Feedback Loop (Using ArcBetweenPoints as a robust alternative to CurvedArrow)
        start_point = plan.get_bottom() + DOWN * 0.2
        end_point = planning_pos + DOWN * 0.5
        
        # Create a curved arrow manually using Arc and add_tip
        feedback_path = ArcBetweenPoints(start_point, end_point, angle=-PI/2, color=RED)
        feedback_path.add_tip()
        
        feedback_label = Text("Feedback to Agent Manager", font_size=18, color=RED).next_to(feedback_path, DOWN)
        
        self.play(Create(feedback_path), Write(feedback_label))
        self.wait(0.5)
        
        # Reset to start
        self.play(
            plan.animate.move_to(planning_pos).set_color(WHITE),
            FadeOut(feedback_path),
            FadeOut(feedback_label),
            run_time=1.5
        )

        # 7. Second Attempt: Pass all gates
        for x in gate_x_coords:
            self.play(plan.animate.move_to(RIGHT * x), run_time=0.6)
            self.play(plan.animate.set_color(GREEN), run_time=0.2)

        # 8. Success: Reach Deployment
        self.play(plan.animate.move_to(deployment_pos), run_time=0.8)
        self.play(
            FadeOut(plan),
            Flash(deployment_icon, color=GREEN, line_length=0.3),
            deployment_icon.animate.scale(1.2).set_color(GREEN)
        )
        self.wait(2)
