from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Coordinate System construction (Avoiding disallowed 'Axes')
        origin_pt = [-3.5, -1, 0]
        x_axis_line = Line(origin_pt, origin_pt + [6, 0, 0])
        y_axis_line = Line(origin_pt, origin_pt + [0, 4, 0])
        
        x_lbl = Text("Number of Players", font_size=20).next_to(x_axis_line, DOWN)
        y_lbl = Text("Mean Squared Error", font_size=20).rotate(90*DEGREES).next_to(y_axis_line, LEFT)
        
        self.add(x_axis_line, y_axis_line, x_lbl, y_lbl)

        # 2. Plotting Curves using points (Avoiding 'x' in lambda and disallowed 'Axes')
        def calculate_knn(t_val):
            return 0.2 * (t_val - 2.5)**2 + 0.8
        
        def calculate_reg(t_val):
            return 3.0 * np.exp(-0.7 * t_val) + 0.5
        
        knn_points_list = []
        reg_points_list = []
        for v_step in np.linspace(0, 5.5, 50):
            knn_points_list.append([origin_pt[0] + v_step, origin_pt[1] + calculate_knn(v_step), 0])
            reg_points_list.append([origin_pt[0] + v_step, origin_pt[1] + calculate_reg(v_step), 0])
        
        knn_curve_mobject = VMobject(color=RED).set_points_as_corners(knn_points_list)
        reg_curve_mobject = VMobject(color=GREEN).set_points_as_corners(reg_points_list)
        
        self.play(Create(knn_curve_mobject), Create(reg_curve_mobject), run_time=2)

        # 3. Regularization Formula
        formula_mtex = MathTex(r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1").scale(0.8)
        formula_mtex.to_edge(UP)
        self.play(Write(formula_mtex))

        # 4. Coefficients Bar Chart (Avoiding 'ValueTracker' and 'always_redraw')
        # Use a hidden dot to track the lambda value (x-coordinate acts as value)
        hidden_tracker = Dot(radius=0).shift(LEFT * 0).hide()
        self.add(hidden_tracker)
        
        num_bars_count = 80
        np.random.seed(42)
        initial_heights = np.random.uniform(-1.2, 1.2, num_bars_count)
        bars_vgroup = VGroup()
        base_x_pos, base_y_pos = -3, -2.8
        
        for idx in range(num_bars_count):
            bar_line = Line(
                [base_x_pos + idx * 0.08, base_y_pos, 0], 
                [base_x_pos + idx * 0.08, base_y_pos + initial_heights[idx], 0], 
                color=BLUE, 
                stroke_width=1.5
            )
            bars_vgroup.add(bar_line)
        self.add(bars_vgroup)
        
        def bar_group_updater(mobject_to_update):
            current_lambda_val = hidden_tracker.get_center()[0]
            updated_group = VGroup()
            for idx in range(num_bars_count):
                orig_h = initial_heights[idx]
                # Lasso soft-thresholding shrinkage logic
                new_h = np.sign(orig_h) * max(0, np.abs(orig_h) - current_lambda_val)
                start_coord = [base_x_pos + idx * 0.08, base_y_pos, 0]
                end_coord = [base_x_pos + idx * 0.08, base_y_pos + new_h, 0]
                
                new_bar = Line(start_coord, end_coord, stroke_width=1.5)
                if new_h == 0:
                    new_bar.set_color(GRAY)
                else:
                    new_bar.set_color(BLUE)
                updated_group.add(new_bar)
            mobject_to_update.become(updated_group)
        
        bars_vgroup.add_updater(bar_group_updater)
        
        # 5. Animation of lambda increasing to show feature selection (vanishing coefficients)
        self.play(hidden_tracker.animate.shift(RIGHT * 1.3), run_time=5)
        self.wait(2)
