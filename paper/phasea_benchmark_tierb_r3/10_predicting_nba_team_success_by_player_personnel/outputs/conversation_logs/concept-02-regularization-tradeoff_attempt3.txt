from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Background / Title / Narration Setup
        title = Text("Regularization Trade-off", font_size=32).to_edge(UP)
        narrator_text = Text(
            "Lasso and Ridge handle high dimensionality better than KNN.",
            font_size=18
        ).to_edge(DOWN, buff=0.5)
        self.add(title, narrator_text)

        # 2. Manual Axes Construction 
        # (Using basic Line mobjects to avoid 'Axes' which was flagged as disallowed)
        origin_pt = LEFT * 5 + DOWN * 2.5
        x_axis = Line(origin_pt, origin_pt + RIGHT * 6)
        y_axis = Line(origin_pt, origin_pt + UP * 5)
        x_axis_label = Text("Number of Players", font_size=18).next_to(x_axis, DOWN)
        y_axis_label = Text("Mean Squared Error", font_size=18).next_to(y_axis, LEFT).rotate(90)
        
        self.play(Create(x_axis), Create(y_axis), Write(x_axis_label), Write(y_axis_label))

        # Helper function for coordinate mapping
        def get_screen_pt(xv, yv):
            # x mapping: 0-10 -> 0-5.5 units
            # y mapping: 0-100 -> 0-4.5 units
            return origin_pt + RIGHT * (xv * 0.55) + UP * (yv * 0.045)

        # 3. KNN Curve (Red): Dipping then rising sharply (Overfitting)
        knn_pts_data = [
            get_screen_pt(1, 90), 
            get_screen_pt(2, 60), 
            get_screen_pt(3, 40), 
            get_screen_pt(4, 30), 
            get_screen_pt(5, 45), 
            get_screen_pt(6, 75), 
            get_screen_pt(7, 105)
        ]
        knn_curve = VMobject(color=RED)
        knn_curve.set_points_as_corners(knn_pts_data)
        knn_txt = Text("KNN", color=RED, font_size=16).move_to(get_screen_pt(7, 112))

        # 4. Lasso/Ridge Curve (Green): Steady decline
        lasso_pts_data = [
            get_screen_pt(1, 90), 
            get_screen_pt(2, 75), 
            get_screen_pt(3, 65), 
            get_screen_pt(4, 58), 
            get_screen_pt(5, 52), 
            get_screen_pt(6, 48), 
            get_screen_pt(7, 45), 
            get_screen_pt(8, 43), 
            get_screen_pt(10, 40)
        ]
        lasso_curve = VMobject(color=GREEN)
        lasso_curve.set_points_as_corners(lasso_pts_data)
        lasso_txt = Text("Lasso/Ridge", color=GREEN, font_size=16).move_to(get_screen_pt(9, 50))

        self.play(Create(knn_curve), Write(knn_txt), run_time=2)
        self.play(Create(lasso_curve), Write(lasso_txt), run_time=2)
        self.wait(1)

        # 5. Lasso Cost Function Formula
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1", 
            font_size=30
        ).shift(RIGHT * 3 + UP * 2)
        self.play(Write(formula))

        # 6. Coefficient Bar Chart
        # Create a group of rectangles representing coefficients
        bars = VGroup()
        for i in range(50):
            h_val = np.random.uniform(0.2, 1.4)
            b = Rectangle(
                width=0.06, 
                height=h_val, 
                fill_opacity=0.8, 
                fill_color=YELLOW, 
                stroke_width=0
            )
            bars.add(b)
        
        bars.arrange(RIGHT, buff=0.04).next_to(formula, DOWN, buff=1.0)
        bar_tag = Text("Coefficients", font_size=18).next_to(bars, UP, buff=0.2)
        self.play(FadeIn(bars), Write(bar_tag))
        self.wait(1)

        # 7. Shrinkage Animation (Simulating Increasing Lambda)
        lambda_msg = Text("Lambda Increasing...", color=YELLOW, font_size=20).next_to(formula, RIGHT, buff=0.3)
        
        # Prepare list of animations for coefficients
        shrink_actions = []
        for idx, bar_obj in enumerate(bars):
            # In Lasso, many coefficients vanish (set to 0 height/opacity)
            if idx % 6 == 0:
                # Features that drive wins stay but shrink slightly
                new_h = bar_obj.get_height() * 0.2
                shrink_actions.append(bar_obj.animate.set_height(new_h, stretch=True).set_color(GREEN))
            else:
                # Noise features vanish
                shrink_actions.append(bar_obj.animate.set_height(0.01, stretch=True).set_opacity(0))
        
        self.play(Write(lambda_msg), *shrink_actions, run_time=4)
        self.wait(2)
