from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Setup Narrator Text
        voice_txt = (
            "As we add more players and stats, the model risks noise. "
            "Lasso and Ridge apply a penalty to focus on the stats that drive wins."
        )
        narrator = Text(voice_txt, font_size=18).to_edge(DOWN)

        # 2. Setup Plot (MSE vs Number of Players)
        # Using Axes as shown in Reference Example 3
        axes = Axes(
            x_range=[0, 11, 1],
            y_range=[0, 10, 2],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": False}
        ).shift(LEFT * 3 + UP * 0.5)
        
        x_lab = Text("Number of Players", font_size=18).next_to(axes.x_axis, DOWN)
        y_lab = Text("Mean Squared Error", font_size=18).rotate(90*DEGREES).next_to(axes.y_axis, LEFT)

        # KNN overfitting curve: dips then rises
        knn_curve = axes.plot(lambda t: 0.3 * (t - 3.5)**2 + 1.2, color=RED, x_range=[1, 10])
        knn_label = Text("KNN", color=RED, font_size=20).next_to(knn_curve.point_from_proportion(0.9), UP)

        # Lasso/Ridge stable curve: steady decline
        lasso_curve = axes.plot(lambda t: 7 * np.exp(-0.35 * t) + 0.5, color=GREEN, x_range=[1, 10])
        lasso_label = Text("Lasso/Ridge", color=GREEN, font_size=20).next_to(lasso_curve.point_from_proportion(0.9), UR)

        # 3. Penalty Equation
        # J(theta) = ||y - X*theta||^2 + lambda * ||theta||_1
        formula = MathTex(r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1").scale(0.7)
        formula.to_edge(UP).shift(RIGHT * 3)

        # 4. Visualization of Coefficient Shrinkage (Feature Selection)
        bar_axes = Axes(
            x_range=[0, 61, 1],
            y_range=[0, 4, 1],
            x_length=5,
            y_length=3,
            axis_config={"include_tip": False}
        ).shift(RIGHT * 3.5 + DOWN * 1.5)
        
        coeff_label = Text("Coefficients Shrinking", font_size=20).next_to(bar_axes, UP)

        # Creating many thin bars to represent high dimensionality
        np.random.seed(42)
        initial_heights = np.random.uniform(0.1, 3.8, 60)
        bars = VGroup()
        for i in range(60):
            # Using basic Rectangle as in Reference Example 1 patterns
            rect = Rectangle(
                width=0.06, height=initial_heights[i], 
                fill_opacity=0.8, fill_color=BLUE, stroke_width=0
            ).move_to(bar_axes.c2p(i+1, 0), aligned_edge=DOWN)
            bars.add(rect)

        # ValueTracker for lambda penalty, as in Reference Example 2
        lambda_val = ValueTracker(0.0)

        # Manual Updater for the Lasso shrinking effect
        # Many coefficients hit zero as lambda increases
        def update_bars_lasso(mob):
            current_lambda = lambda_val.get_value()
            for idx, b in enumerate(mob):
                # Lasso soft-thresholding simulation
                new_h = initial_heights[idx] - current_lambda
                if new_h < 0.001:
                    new_h = 0.001
                    b.set_fill(opacity=0)
                else:
                    b.set_fill(opacity=0.8)
                b.stretch_to_fit_height(new_h, about_edge=DOWN)

        bars.add_updater(update_bars_lasso)

        # Lambda value display using an updater for numeric feedback
        l_sym = MathTex(r"\lambda = ").scale(0.8).next_to(formula, DOWN)
        l_num = Text("0.00", font_size=20).next_to(l_sym, RIGHT)
        l_num.add_updater(lambda m: m.become(
            Text(f"{lambda_val.get_value():.2f}", font_size=20).next_to(l_sym, RIGHT)
        ))

        # 5. Animation Sequence
        self.add(narrator)
        self.play(Create(axes), Write(x_lab), Write(y_lab))
        self.play(Create(knn_curve), Write(knn_label))
        self.play(Create(lasso_curve), Write(lasso_label))
        self.wait(1)
        
        self.play(Write(formula), Create(bar_axes), Create(bars))
        self.play(Write(l_sym), Write(l_num), Write(coeff_label))
        
        # Animate the penalty increasing, causing coefficients to vanish
        # rate_func=lambda t: t serves as a linear progression
        self.play(
            lambda_val.animate.set_value(3.8), 
            run_time=6, 
            rate_func=lambda t: t
        )
        self.wait(2)
