import numpy as np
from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Define data and strings
        formula_str = r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})"
        # Representative heights for coefficients
        height_values = [3.5, 3.0, 1.2, 1.1, 2.0, 1.0, 0.9, 1.7, 0.6, 0.5]

        # 2. Create 10 player icons in a row
        players = VGroup(*[
            VGroup(
                Circle(radius=0.25, color=WHITE, fill_opacity=0.2),
                Text(str(i+1), font_size=20)
            ) for i in range(10)
        ]).arrange(RIGHT, buff=0.5).to_edge(DOWN, buff=1.5)

        # 3. Create vertical bars representing 'Average Coefficient'
        bars = VGroup()
        for i, h in enumerate(height_values):
            bar = Rectangle(
                width=0.4, 
                height=h, 
                fill_opacity=0.7, 
                color=BLUE,
                stroke_width=1
            )
            # Position bar above player icon
            bar.next_to(players[i], UP, buff=0.1)
            # Align bottom of bar to its position so it grows upward
            bar.align_to(players[i].get_top() + UP*0.1, DOWN)
            bars.add(bar)

        # 4. Animation: Bars growing
        self.add(players)
        self.play(
            AnimationGroup(
                *[GrowFromPoint(bar, bar.get_bottom()) for bar in bars],
                lag_ratio=0.1
            ),
            run_time=2
        )
        self.wait(1)

        # 5. Highlight 1st and 2nd (Stars) with Gold glow
        stars_indices = [0, 1]
        star_rects = VGroup(*[
            SurroundingRectangle(bars[i], color=GOLD, buff=0.05, stroke_width=4)
            for i in stars_indices
        ])
        
        self.play(
            *[bars[i].animate.set_color(GOLD).set_fill(GOLD, opacity=0.9) for i in stars_indices],
            Create(star_rects),
            run_time=1
        )
        self.wait(0.5)

        # 6. Highlight 5th and 8th (Depth) with pulsing light
        depth_indices = [4, 7] # 5th man (index 4) and 8th man (index 7)
        depth_rects = VGroup(*[
            SurroundingRectangle(bars[i], color=BLUE_B, buff=0.05, stroke_width=6)
            for i in depth_indices
        ])

        self.play(
            *[bars[i].animate.set_color(BLUE_B).set_fill(BLUE_B, opacity=0.9) for i in depth_indices],
            Create(depth_rects),
            run_time=1
        )

        # Pulse updater function for the highlight
        def pulse_effect(m, dt):
            # Using self.time (Scene clock) for sine wave
            val = np.sin(self.time * 7)
            m.set_stroke(opacity=0.5 + 0.5 * val)
            m.scale(1.002 + 0.005 * val)

        for rect in depth_rects:
            rect.add_updater(pulse_effect)

        # 7. Display Formula and title/summary
        formula = MathTex(formula_str, font_size=34).to_edge(UP, buff=0.8)
        summary_text = Text("Depth is as measurable as stardom", font_size=28, color=YELLOW).next_to(formula, DOWN, buff=0.4)

        self.play(Write(formula))
        self.play(FadeIn(summary_text))
        
        self.wait(5)
        
        # Cleanup updaters
        for rect in depth_rects:
            rect.remove_updater(pulse_effect)
