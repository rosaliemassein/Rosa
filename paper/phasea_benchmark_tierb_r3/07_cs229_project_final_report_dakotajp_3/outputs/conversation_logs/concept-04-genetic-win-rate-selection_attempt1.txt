from manim import *

class GeneticWinRateSelection(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)")
        formula.to_edge(UP, buff=0.5)
        self.play(Write(formula))
        self.wait(0.5)

        # Helper to create a node
        def create_node(label, color=BLUE):
            circle = Circle(radius=0.5, color=color, fill_opacity=0.2)
            text = Text(label, font_size=18)
            return VGroup(circle, text)

        # Helper to create a manual bar for win rate
        def create_bar(value, color=WHITE):
            # Scale value (0-100) to height (0-2)
            height = (value / 100) * 1.5
            bar = Rectangle(width=0.4, height=max(height, 0.1), fill_color=color, fill_opacity=0.8, stroke_width=1)
            bar.align_to(ORIGIN, DOWN) # Align bottom to origin
            label = Text(f"{value}%", font_size=14).next_to(bar, UP, buff=0.1)
            return VGroup(bar, label)

        # 2. Initial Parent
        current_parent = create_node("Parent", BLUE).move_to(UP * 1.2)
        self.play(FadeIn(current_parent))

        # Evolution Cycles
        for cycle in range(2):
            # 3. Create Children nodes branching out
            child_positions = [LEFT * 3.5 + DOWN * 0.5, DOWN * 0.5, RIGHT * 3.5 + DOWN * 0.5]
            children = VGroup()
            lines = VGroup()
            
            for i, pos in enumerate(child_positions):
                child = create_node(f"Child {i+1}", GREEN).move_to(pos)
                line = Line(current_parent.get_bottom(), child.get_top(), stroke_width=2, color=GRAY)
                children.add(child)
                lines.add(line)

            self.play(Create(lines), FadeIn(children))

            # 4. Create Win Rate Bars below children
            win_rates = [45, 88, 62] if cycle == 0 else [92, 54, 71]
            bars_group = VGroup()
            for i, rate in enumerate(win_rates):
                bar_obj = create_bar(rate).next_to(children[i], DOWN, buff=0.4)
                bars_group.add(bar_obj)

            # Animate bars growing (using scale from bottom)
            self.play(
                *[FadeIn(b[1]) for b in bars_group],
                *[GrowFromEdge(b[0], DOWN) for b in bars_group],
                run_time=1.5
            )
            self.wait(1)

            # 5. Identify and highlight winner
            winner_idx = win_rates.index(max(win_rates))
            winner_rect = SurroundingRectangle(VGroup(children[winner_idx], bars_group[winner_idx]), color=GOLD, buff=0.2)
            
            self.play(
                Create(winner_rect),
                children[winner_idx][0].animate.set_color(GOLD).set_fill(opacity=0.4),
                bars_group[winner_idx][0].animate.set_color(GOLD),
                run_time=1
            )
            self.wait(0.5)

            # 6. Selection: Keep winner, discard others
            others = VGroup(
                current_parent, lines, winner_rect,
                *[children[i] for i in range(3) if i != winner_idx],
                *[bars_group[i] for i in range(3) if i != winner_idx]
            )
            
            winner_mobjects = VGroup(children[winner_idx], bars_group[winner_idx])
            
            # Move winner to top center to become new parent
            new_parent_node = create_node("Parent", BLUE).move_to(UP * 1.2)
            
            self.play(FadeOut(others))
            self.play(
                winner_mobjects.animate.move_to(UP * 1.2).scale(0.8),
                run_time=1.5
            )
            
            # Finalize transformation into a clean Parent node
            self.play(ReplacementTransform(winner_mobjects, new_parent_node))
            current_parent = new_parent_node
            self.wait(0.5)

        # Conclusion
        final_text = Text("Direct Evolutionary Gradient for Victory", font_size=24, color=GOLD).to_edge(DOWN)
        self.play(Write(final_text))
        self.play(Indicate(current_parent))
        self.wait(2)
