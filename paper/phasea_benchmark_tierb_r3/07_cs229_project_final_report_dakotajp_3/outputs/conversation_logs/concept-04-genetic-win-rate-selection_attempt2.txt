from manim import *

class GeneticWinRateSelection(Scene):
    def construct(self):
        # Header Formula
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)").to_edge(UP)
        self.add(formula)

        # 1. Central 'Parent' node
        parent_node = Circle(radius=0.4, fill_color=GOLD, fill_opacity=1, stroke_color=GOLD)
        parent_text = Text("Parent", font_size=24).next_to(parent_node, DOWN, buff=0.1)
        parent_group = VGroup(parent_node, parent_text).move_to(UP * 1.5)

        # 2. 'Child' nodes
        child_colors = [YELLOW, GREEN, RED]
        child_groups = VGroup()
        for i in range(3):
            c_node = Circle(radius=0.3, fill_color=child_colors[i], fill_opacity=1, stroke_color=child_colors[i])
            c_text = Text(f"Child {i+1}", font_size=20).next_to(c_node, DOWN, buff=0.1)
            child_groups.add(VGroup(c_node, c_text))
        
        child_groups.arrange(RIGHT, buff=1.5).move_to(ORIGIN)
        
        # Branching lines
        lines = VGroup(*[Line(parent_group.get_bottom(), c.get_top(), stroke_width=2) for c in child_groups])

        # 3. Custom Bar Chart Representation (using Axes and Rectangles for compatibility)
        charts = VGroup()
        bars_vgroup = VGroup()
        
        final_heights = [
            [0.3, 0.6, 0.5], # Child 1
            [0.5, 0.8, 1.1], # Child 2 (Winner)
            [0.2, 0.4, 0.3]  # Child 3
        ]

        for i in range(3):
            # Create a small coordinate system for each child
            ax = Axes(
                x_range=[0, 3], y_range=[0, 1.2], 
                x_length=1.2, y_length=1.0,
                axis_config={"include_tip": False, "include_ticks": False, "stroke_width": 1}
            ).next_to(child_groups[i], DOWN, buff=0.4)
            
            # Create 3 bars per chart
            bars = VGroup(*[
                Rectangle(
                    width=0.2, height=0.01, 
                    fill_opacity=0.8, fill_color=child_colors[i],
                    stroke_width=1
                ).move_to(ax.c2p(j + 0.5, 0), DOWN)
                for j in range(3)
            ])
            
            label = Text("Win Rate", font_size=14).next_to(ax, DOWN, buff=0.1)
            charts.add(VGroup(ax, label))
            bars_vgroup.add(bars)

        # Animation Sequence
        self.play(FadeIn(parent_group))
        self.play(Create(lines), FadeIn(child_groups))
        self.play(FadeIn(charts), FadeIn(bars_vgroup))
        self.wait(0.5)

        # 4. Animate the bars growing
        growth_anims = []
        for i in range(3):
            for j in range(3):
                growth_anims.append(
                    bars_vgroup[i][j].animate.stretch_to_fit_height(final_heights[i][j], about_edge=DOWN)
                )
        self.play(*growth_anims, run_time=2)
        self.wait(0.5)

        # 5. Highlight the winner (Child 2)
        winner_idx = 1
        highlight_box = SurroundingRectangle(VGroup(child_groups[winner_idx], charts[winner_idx], bars_vgroup[winner_idx]), color=GOLD, buff=0.2)
        
        self.play(
            Create(highlight_box),
            bars_vgroup[winner_idx].animate.set_color(GOLD),
            child_groups[winner_idx].animate.scale(1.1)
        )
        self.wait(1)

        # 6. Evolutionary Step: Winner becomes the new Parent
        others = VGroup(
            parent_group, lines, highlight_box,
            child_groups[0], child_groups[2],
            charts[0], charts[2],
            bars_vgroup[0], bars_vgroup[2],
            charts[winner_idx], bars_vgroup[winner_idx]
        )

        self.play(
            FadeOut(others),
            child_groups[winner_idx].animate.move_to(parent_group.get_center()).scale(1/1.1),
            run_time=1.5
        )

        # 7. Quick visual to show cycle
        next_gen_indicator = Text("Evolutionary Step Complete", font_size=24, color=GOLD).move_to(DOWN*2)
        self.play(Write(next_gen_indicator))
        self.wait(1)
        self.play(FadeOut(next_gen_indicator), FadeOut(child_groups[winner_idx]))
        self.wait(0.5)
