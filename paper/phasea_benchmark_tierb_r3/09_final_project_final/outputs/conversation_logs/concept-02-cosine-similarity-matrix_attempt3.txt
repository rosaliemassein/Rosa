from manim import *
import math

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Setup Manual Axes (replacing Axes)
        origin = LEFT * 3 + DOWN * 1.5
        x_axis = Line(origin, origin + RIGHT * 4, color=WHITE)
        y_axis = Line(origin, origin + UP * 4, color=WHITE)
        self.add(x_axis, y_axis)

        # 2. Formula (from prompt)
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=32
        ).to_edge(UP, buff=0.5)
        self.add(formula)

        # 3. Vectors (using Line + add_tip instead of Vector)
        vec_a = Line(origin, origin + RIGHT * 3, color=RED, buff=0)
        vec_a.add_tip()
        label_a = MathTex(r"\mathbf{A}", color=RED).next_to(vec_a.get_end(), RIGHT)
        
        vec_b = Line(origin, origin + RIGHT * 3, color=YELLOW, buff=0)
        vec_b.add_tip()
        # Rotate B to initial 60 degrees
        vec_b.rotate(60 * DEGREES, about_point=origin)
        
        # 4. Angle and Similarity Display
        # Angle from Example 2
        angle_arc = Angle(x_axis, vec_b, radius=0.6, other_angle=False)
        angle_arc.add_updater(lambda x: x.become(Angle(x_axis, vec_b, radius=0.6, other_angle=False)))
        
        sim_label = MathTex(r"\text{Similarity: }", font_size=24).to_corner(UL, buff=1.5).shift(DOWN * 1)
        sim_val = MathTex("0.50", font_size=24).next_to(sim_label, RIGHT)
        
        def update_sim_text(x):
            # Use vec_b.get_angle() to find the current rotation
            current_angle = vec_b.get_angle()
            val = math.cos(current_angle)
            x.become(MathTex(f"{val:.2f}", font_size=24, color=YELLOW).next_to(sim_label, RIGHT))
            
        sim_val.add_updater(update_sim_text)
        
        self.add(vec_a, label_a, vec_b, angle_arc, sim_label, sim_val)

        # 5. Matrix Display
        matrix = VGroup(*[Square(side_length=0.4, fill_opacity=0.1, color=BLUE) for _ in range(25)])
        matrix.arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=1)
        matrix_title = MathTex(r"\text{Similarity Matrix}", font_size=24).next_to(matrix, UP)
        self.add(matrix, matrix_title)

        # 6. Linking Vector Alignment to Matrix
        # We simulate the center cell (12) reflecting the state of our two vectors
        center_cell = matrix[12]
        
        # Animation sequence
        self.wait(1)
        
        # Step 1: Align Vectors (theta -> 0)
        self.play(
            Rotate(vec_b, angle=-60 * DEGREES, about_point=origin),
            center_cell.animate.set_fill(YELLOW, opacity=0.9),
            run_time=3
        )
        self.play(Indicate(center_cell))
        self.wait(1)
        
        # Step 2: Diverge Vectors (theta -> 90)
        self.play(
            Rotate(vec_b, angle=90 * DEGREES, about_point=origin),
            center_cell.animate.set_fill(BLUE, opacity=0.1),
            run_time=3
        )
        self.wait(2)
