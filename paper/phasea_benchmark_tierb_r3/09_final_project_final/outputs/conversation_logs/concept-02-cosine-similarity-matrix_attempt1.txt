from manim import *

class CosineSimilarity(Scene):
    def construct(self):
        # 1. Coordinate system setup
        ax = Axes(
            x_range=[-1, 5, 1], 
            y_range=[-1, 5, 1], 
            x_length=5, 
            y_length=5
        ).to_edge(LEFT, buff=0.5)
        
        # 2. Angle tracker for animation (Reference Example 2)
        vt = ValueTracker(60)
        
        # 3. Vectors representing proposals and interests
        # We use Line + add_tip for maximum compatibility
        v1 = Line(ax.c2p(0, 0), ax.c2p(3.5, 0.5), color=BLUE)
        v1.add_tip()
        
        v2_ref = Line(ax.c2p(0, 0), ax.c2p(3.5, 0.5), color=YELLOW)
        v2_ref.add_tip()
        
        # v2 will be our moving vector
        v2 = v2_ref.copy()
        
        def update_v2(mob):
            # Rotate v2 based on the current tracker value
            angle_val = vt.get_value() * DEGREES
            mob.become(v2_ref.copy().rotate(angle_val, about_point=ax.c2p(0, 0)))
        
        v2.add_updater(update_v2)
        
        # 4. Angle Arc (Reference Example 2)
        angle_arc = Angle(v1, v2, radius=0.8, other_angle=False)
        
        def update_arc(mob):
            mob.become(Angle(v1, v2, radius=0.8, other_angle=False))
        
        angle_arc.add_updater(update_arc)
        
        # 5. Display Similarity value (cos theta)
        # Using MathTex updated by a function
        sim_label = MathTex(r"\text{Similarity } \cos(\theta) =").shift(UP * 2.5 + RIGHT * 2)
        sim_val = MathTex("0.50").next_to(sim_label, RIGHT)
        
        def update_val(mob):
            # cos(theta) calculation
            current_cos = np.cos(vt.get_value() * DEGREES)
            mob.become(MathTex("{:.2f}".format(current_cos)).next_to(sim_label, RIGHT))
            
        sim_val.add_updater(update_val)
        
        # 6. Similarity Matrix (5x5 grid of squares)
        matrix_group = VGroup(*[
            Square(side_length=0.7, fill_opacity=0.3, color=BLUE)
            for _ in range(25)
        ]).arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=0.5)
        
        # Heatmap effect: The center cell glows based on alignment
        def update_cell(mob):
            current_cos = np.cos(vt.get_value() * DEGREES)
            # Map cos_val (0 to 1) to opacity
            opac = max(0.0, min(1.0, current_cos))
            mob.set_fill(WHITE, opacity=opac)
        
        matrix_group[12].add_updater(update_cell)
        
        # 7. Rendering the Scene
        self.add(ax, v1, v2, angle_arc, sim_label, sim_val, matrix_group)
        
        # Animation: Vectors aligning (Similarity increases to 1.0)
        self.play(vt.animate.set_value(0), run_time=3)
        self.wait(1)
        
        # Animation: Vectors diverging (Similarity decreases)
        self.play(vt.animate.set_value(90), run_time=3)
        self.wait(1)
        
        # Return to a middle state
        self.play(vt.animate.set_value(45), run_time=2)
        self.wait(2)
