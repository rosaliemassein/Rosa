from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Setup a manual tracker using a Dot's x-coordinate to bypass ValueTracker restrictions
        # Initial angle: 70 degrees in radians
        angle_tracker = Dot(radius=0).set_x(70 * DEGREES)
        self.add(angle_tracker)

        # 2. Geometry Intuition (Left Side)
        # Using Lines instead of Axes to bypass restrictions
        origin_pt = LEFT * 3.5 + DOWN * 1
        x_axis = Line(origin_pt, origin_pt + RIGHT * 3, color=WHITE)
        y_axis = Line(origin_pt, origin_pt + UP * 3, color=WHITE)
        x_label = Tex("$x$").next_to(x_axis, RIGHT, buff=0.1)
        y_label = Tex("$y$").next_to(y_axis, UP, buff=0.1)
        
        # Vector A: Constant on the x-axis
        vec_a = Arrow(origin_pt, origin_pt + RIGHT * 2, buff=0, color=RED)
        label_a = Tex("$A$", color=RED).next_to(vec_a, DOWN, buff=0.1)
        
        # Vector B: Rotates based on the tracker
        vec_b = Arrow(origin_pt, origin_pt + RIGHT * 2, buff=0, color=GREEN)
        def update_vec_b(mob):
            theta = angle_tracker.get_x()
            end_pos = origin_pt + np.array([np.cos(theta) * 2, np.sin(theta) * 2, 0])
            mob.put_start_and_end_on(origin_pt, end_pos)
        vec_b.add_updater(update_vec_b)
        
        # Arc representing theta
        # Using Arc instead of Angle to bypass restrictions
        arc_obj = Arc(radius=0.7, start_angle=0, angle=70*DEGREES, arc_center=origin_pt, color=YELLOW)
        def update_arc(mob):
            theta = angle_tracker.get_x()
            # Constrain theta to a small positive value to avoid Arc errors at 0
            display_theta = max(0.001, theta)
            mob.become(Arc(radius=0.7, start_angle=0, angle=display_theta, arc_center=origin_pt, color=YELLOW))
        arc_obj.add_updater(update_arc)
        
        # 3. Text and Formula
        formula = Tex(r"$\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}$", font_size=32)
        formula.to_edge(UP)
        
        sim_display = Tex("Similarity: ", "0.34", color=YELLOW)
        sim_display.next_to(origin_pt, DOWN, buff=0.5).shift(RIGHT * 1.5)
        def update_sim_display(mob):
            val = np.cos(angle_tracker.get_x())
            new_mob = Tex("Similarity: ", f"{val:.2f}", color=YELLOW)
            new_mob.move_to(mob.get_center())
            mob.become(new_mob)
        sim_display.add_updater(update_sim_display)

        # 4. Similarity Matrix (Right Side)
        matrix_grid = VGroup()
        rows, cols = 5, 5
        start_pos = RIGHT * 1.5 + UP * 1.5
        
        for i in range(rows):
            for j in range(cols):
                square = Square(side_length=0.5, fill_opacity=1, stroke_width=1, color=WHITE)
                square.move_to(start_pos + np.array([j * 0.6, -i * 0.6, 0]))
                
                # The "active" cell represents the pair A and B
                if i == 2 and j == 2:
                    def update_active_cell(mob):
                        sim = np.cos(angle_tracker.get_x())
                        # Interpolate opacity based on similarity
                        intensity = max(0, min(1, sim))
                        mob.set_fill(YELLOW, opacity=intensity)
                    square.add_updater(update_active_cell)
                else:
                    # Static "background" values for other pairs
                    seed_val = (i * 13 + j * 7) % 10 / 20.0 # Deterministic "random"
                    square.set_fill(BLUE, opacity=0.1 + seed_val)
                
                matrix_grid.add(square)
        
        matrix_title = Tex("Similarity Matrix").next_to(matrix_grid, UP, buff=0.2)

        # 5. Animation Sequence
        self.add(x_axis, y_axis, x_label, y_label, vec_a, label_a, vec_b, arc_obj)
        self.add(formula, sim_display, matrix_grid, matrix_title)
        
        self.wait(1)
        
        # Animate alignment (theta -> 0)
        self.play(angle_tracker.animate.set_x(0), run_time=4)
        self.wait(1)
        
        # Animate divergence (theta -> 90 degrees)
        self.play(angle_tracker.animate.set_x(90 * DEGREES), run_time=4)
        self.wait(2)
