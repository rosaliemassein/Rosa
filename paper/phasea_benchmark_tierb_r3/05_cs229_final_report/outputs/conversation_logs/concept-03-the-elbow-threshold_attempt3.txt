from manim import *

class TheElbowThreshold(Scene):
    def construct(self):
        # 1. Create Axes for plotting
        # x-axis: Fraction of Zones Flagged, y-axis: Recall
        ax = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=5,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=1)
        
        # Axis labels
        x_label = Text("Fraction of Zones Flagged", font_size=20).next_to(ax.x_axis, DOWN)
        y_label = Text("Recall", font_size=20).rotate(90 * DEGREES).next_to(ax.y_axis, LEFT)
        
        # 2. Define the concave curve (represents validation results)
        # Using y = x^0.3 to create a curve that rises sharply then levels off
        curve_graph = ax.plot(lambda x: x**0.3, x_range=[0, 1], color=BLUE)
        
        # 3. Setup ValueTracker for interactive movement
        # tracker_val controls the progress along the curve
        tracker_val = ValueTracker(0.05)
        
        # 4. Create a highlighted point that moves along the curve
        moving_dot = Dot(color=YELLOW)
        moving_dot.add_updater(
            lambda mob: mob.move_to(
                ax.c2p(tracker_val.get_value(), tracker_val.get_value()**0.3)
            )
        )
        
        # 5. Side-panel Floorplan (Grid of zones)
        # Squares light up as more of the fraction is flagged
        zone_grid = VGroup(*[
            Square(side_length=0.4, color=WHITE, stroke_width=2) 
            for i in range(16)
        ]).arrange_in_grid(rows=4, cols=4, buff=0.1).to_edge(RIGHT, buff=1)
        
        def update_floorplan(mobs):
            current_x = tracker_val.get_value()
            # Calculate how many zones to highlight out of 16
            highlight_count = int(current_x * 16)
            for i, square in enumerate(mobs):
                if i < highlight_count:
                    square.set_fill(RED, opacity=0.8)
                else:
                    square.set_fill(BLACK, opacity=0)
        
        zone_grid.add_updater(update_floorplan)
        
        # 6. Formula reference
        recall_formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=30).to_corner(UL)
        
        # Add static elements
        self.add(ax, x_label, y_label, curve_graph, moving_dot, zone_grid, recall_formula)
        
        # 7. Animation sequence
        # Move right: increasing flagged zones (increasing recall)
        self.play(
            tracker_val.animate.set_value(0.9), 
            run_time=4, 
            rate_func=lambda t: t
        )
        self.wait(0.5)
        
        # Move back to the 'elbow' of the curve (approx 0.2)
        elbow_point_x = 0.2
        self.play(tracker_val.animate.set_value(elbow_point_x), run_time=2)
        
        # Highlight the 'elbow' with a glowing circle and label
        elbow_coord = ax.c2p(elbow_point_x, elbow_point_x**0.3)
        glow_circle = Circle(radius=0.4, color=GREEN, stroke_width=4).move_to(elbow_coord)
        elbow_text = Text("The Elbow", color=GREEN, font_size=24).next_to(glow_circle, UR)
        
        self.play(Create(glow_circle), Write(elbow_text))
        
        # Visual pulse effect
        self.play(glow_circle.animate.scale(1.2), rate_func=there_and_back, run_time=0.5)
        
        self.wait(2)
