from manim import *
import numpy as np

class Concept03TheElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Manual Coordinate System (Avoiding Axes class)
        origin = LEFT * 5 + DOWN * 2
        x_end = origin + RIGHT * 6
        y_end = origin + UP * 5
        
        x_axis = Arrow(origin, x_end, buff=0, color=BLUE)
        y_axis = Arrow(origin, y_end, buff=0, color=BLUE)
        
        x_label = Text("Fraction of Zones Flagged", font_size=24).next_to(x_axis, DOWN)
        y_label = Text("Recall", font_size=24).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        # 2. Formula
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=30).to_corner(UL)
        
        # 3. Create Curve (Concave curve: 1 - e^-5x)
        curve_points = []
        for val in np.linspace(0, 1, 50):
            px = val * 6
            py = (1 - np.exp(-5 * val)) * 5
            curve_points.append(origin + px * RIGHT + py * UP)
        
        curve = VMobject(color=RED)
        curve.set_points_as_corners(curve_points)
        
        # 4. Side Panel and Zones (Floorplan)
        side_rect = Rectangle(height=4.5, width=4.5, color=WHITE, fill_opacity=0.05)
        side_rect.to_edge(RIGHT, buff=0.5)
        panel_title = Text("Zone Map", font_size=24).next_to(side_rect, UP)
        
        zones = VGroup()
        for i in range(25):
            s = Square(side_length=0.7, stroke_width=1, fill_opacity=0.2, color=BLUE)
            zones.add(s)
        zones.arrange_in_grid(5, 5, buff=0.1).move_to(side_rect.get_center())
        
        # 5. Interaction Tracking (Using a dot instead of ValueTracker)
        tracker_dot = Dot(radius=0).move_to(origin)
        
        moving_dot = Dot(color=YELLOW)
        v_line = Line(origin, origin, color=YELLOW, stroke_width=2)
        h_line = Line(origin, origin, color=YELLOW, stroke_width=2)
        
        # Update functions (Avoiding lambda)
        def update_moving_dot(mob):
            # Calculate t (0 to 1) based on tracker_dot's X position relative to origin
            t = (tracker_dot.get_center()[0] - origin[0]) / 6.0
            # Ensure t is clamped
            if t < 0: t = 0
            if t > 1: t = 1
            y_val = (1 - np.exp(-5 * t)) * 5
            mob.move_to(origin + t * 6 * RIGHT + y_val * UP)

        def update_v_line(mob):
            p = moving_dot.get_center()
            mob.put_start_and_end_on([p[0], origin[1], 0], p)

        def update_h_line(mob):
            p = moving_dot.get_center()
            mob.put_start_and_end_on([origin[0], p[1], 0], p)
            
        def update_zones_colors(z_group):
            t = (tracker_dot.get_center()[0] - origin[0]) / 6.0
            if t < 0: t = 0
            if t > 1: t = 1
            num_active = int(t * 25)
            for idx, zone in enumerate(z_group):
                if idx < num_active:
                    zone.set_fill(YELLOW, opacity=0.8)
                else:
                    zone.set_fill(BLUE, opacity=0.2)

        moving_dot.add_updater(update_moving_dot)
        v_line.add_updater(update_v_line)
        h_line.add_updater(update_h_line)
        zones.add_updater(update_zones_colors)
        
        # 6. Animation Sequence
        self.add(x_axis, y_axis, x_label, y_label, formula)
        self.play(Create(curve))
        self.add(side_rect, zones, panel_title)
        self.add(tracker_dot, moving_dot, v_line, h_line)
        
        # Move forward (Increasing threshold/fraction)
        self.play(tracker_dot.animate.move_to(origin + 0.8 * 6 * RIGHT), run_time=4)
        self.wait(0.5)
        # Move back to find the "Elbow"
        self.play(tracker_dot.animate.move_to(origin + 0.3 * 6 * RIGHT), run_time=2)
        
        # Highlight the Elbow

        elbow_y = (1 - np.exp(-5 * elbow_t)) * 5
        elbow_point = origin + elbow_t * 6 * RIGHT + elbow_y * UP
        
        highlight_circle = Circle(radius=0.4, color=WHITE).move_to(elbow_point)
        elbow_text = Text("The Elbow", font_size=24, color=YELLOW).next_to(highlight_circle, UR)
        
        self.play(Create(highlight_circle), Write(elbow_text))
        self.play(highlight_circle.animate.scale(1.2), rate_func=there_and_back)
        
        self.wait(2)
