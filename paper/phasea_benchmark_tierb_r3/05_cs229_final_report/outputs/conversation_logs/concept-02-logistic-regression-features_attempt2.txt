from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Background Plane and Sigmoid Curve
        # Using NumberPlane as requested in the remarks
        plane = NumberPlane(
            x_range=[-6, 6, 1],
            y_range=[0, 1.1, 0.5],
            x_length=6,
            y_length=4,
            background_line_style={"stroke_opacity": 0.4}
        ).to_edge(RIGHT, buff=0.5)

        sigmoid_curve = plane.plot(
            lambda x_val: 1 / (1 + np.exp(-x_val)),
            color=RED
        )
        
        # 2. Features / Sliding Bars UI
        # We track feature values using ValueTrackers
        x1_tracker = ValueTracker(0)
        x30_tracker = ValueTracker(0)
        
        slider_1_bg = Line(DOWN * 1.5, UP * 1.5).shift(LEFT * 5)
        slider_30_bg = Line(DOWN * 1.5, UP * 1.5).shift(LEFT * 3)
        
        knob_1 = Dot(color=GOLD).move_to(slider_1_bg.get_center())
        knob_30 = Dot(color=GOLD).move_to(slider_30_bg.get_center())
        
        # Updaters for the knobs (using 'mob' to avoid 'd', 'm', 'x' naming issues)
        knob_1.add_updater(lambda mob: mob.move_to(
            slider_1_bg.point_from_proportion(np.clip((x1_tracker.get_value() + 3) / 6, 0, 1))
        ))
        knob_30.add_updater(lambda mob: mob.move_to(
            slider_30_bg.point_from_proportion(np.clip((x30_tracker.get_value() + 3) / 6, 0, 1))
        ))
        
        label_1 = MathTex("X_{1d}").next_to(slider_1_bg, UP)
        label_30 = MathTex("X_{30d}").next_to(slider_30_bg, UP)
        
        # 3. Dot on Sigmoid curve
        # Prediction point depends on the sum of inputs (simplified logistic z)
        moving_dot = Dot(color=BLUE)
        moving_dot.add_updater(lambda mob: mob.move_to(
            plane.c2p(
                x1_tracker.get_value() + x30_tracker.get_value(),
                1 / (1 + np.exp(-(x1_tracker.get_value() + x30_tracker.get_value())))
            )
        ))

        # 4. Formulas and Transformation
        # Initial feature vector
        formula_x = MathTex("X = [X_{1d}, X_{30d}]").to_edge(DOWN, buff=1.2).shift(LEFT * 3)
        
        # The main logistic formula
        logistic_formula = MathTex(
            "P(y=1)", "=", "{1", "\\over", "1 + e^{-(\\beta_0 + \\beta_1 X_{1d} + \\beta_{30} X_{30d})}}"
        ).scale(0.85).to_edge(DOWN, buff=0.4)

        # 5. Animation Sequence
        self.add(plane, sigmoid_curve)
        self.add(slider_1_bg, slider_30_bg, knob_1, knob_30, label_1, label_30)
        self.add(moving_dot)
        self.play(Write(formula_x))
        self.wait(0.5)
        
        # Transform the feature vector logic into the full formula
        self.play(
            TransformMatchingTex(formula_x, logistic_formula)
        )
        
        # Animate feature changes
        # Slide 1: High positive correlation
        self.play(
            x1_tracker.animate.set_value(2),
            x30_tracker.animate.set_value(2),
            run_time=2,
            rate_func=lambda t: t  # Replacement for 'linear'
        )
        self.wait(0.5)
        
        # Slide 2: Low negative correlation
        self.play(
            x1_tracker.animate.set_value(-3),
            x30_tracker.animate.set_value(-1),
            run_time=2,
            rate_func=lambda t: t
        )
        
        # Final adjustment
        self.play(
            x1_tracker.animate.set_value(0.5),
            x30_tracker.animate.set_value(0.5),
            run_time=1.5
        )
        
        self.wait(2)
