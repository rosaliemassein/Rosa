from manim import *
import numpy as np

class Concept03TheElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 1.05, 0.2],
            y_range=[0, 1.05, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": False}
        ).to_edge(LEFT, buff=0.7)
        
        x_label = Text("Fraction of Zones Flagged", font_size=20).next_to(axes.x_axis, DOWN)
        y_label = Text("Recall", font_size=20).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)
        
        # 2. Plot Concave Elbow Curve
        # Using a function that rises sharply: y = 1 - (1-x)^4
        recall_curve = axes.plot(
            lambda x: 1 - (1 - x)**4,
            x_range=[0, 1],
            color=BLUE
        )
        
        # 3. Floorplan side-panel
        grid_size = 6
        zones = VGroup(*[
            Square(side_length=0.35, stroke_width=1, color=WHITE)
            for _ in range(grid_size * grid_size)
        ]).arrange_in_grid(rows=grid_size, cols=grid_size, buff=0.05)
        
        side_panel_title = Text("Building Floorplan", font_size=24)
        side_panel = VGroup(side_panel_title, zones).arrange(DOWN, buff=0.4).to_edge(RIGHT, buff=0.5)
        
        # 4. ValueTracker for animation
        tracker = ValueTracker(0.0)
        
        # 5. Dynamic Elements (Redrawers)
        # Moving dot on curve
        dot = always_redraw(lambda: Dot(
            point=axes.c2p(tracker.get_value(), 1 - (1 - tracker.get_value())**4),
            color=YELLOW,
            radius=0.08
        ))
        
        # Dynamic zone highlighting
        def update_zones(z_group):
            val = tracker.get_value()
            total = len(z_group)
            num_on = int(val * total)
            for i, zone in enumerate(z_group):
                if i < num_on:
                    zone.set_fill(RED, opacity=0.8)
                else:
                    zone.set_fill(BLACK, opacity=0)
        
        zones.add_updater(update_zones)
        
        # 6. Formula
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_corner(UL)
        
        # 7. Animation Sequence
        self.add(axes, x_label, y_label, formula)
        self.play(Create(recall_curve), FadeIn(side_panel))
        self.add(dot)
        
        # Move from low threshold to high fraction
        self.play(tracker.animate.set_value(0.95), run_time=4, rate_func=linear)
        self.wait(0.5)
        
        # Move back to find the elbow
        elbow_x = 0.35
        self.play(tracker.animate.set_value(elbow_x), run_time=2)
        
        # Highlight the Elbow
        elbow_point = axes.c2p(elbow_x, 1 - (1 - elbow_x)**4)
        elbow_circle = Circle(radius=0.4, color=GOLD).move_to(elbow_point)
        elbow_label = Text("Elbow", color=GOLD, font_size=30).next_to(elbow_circle, UR, buff=0.1)
        
        # Glowing effect logic
        self.play(
            Create(elbow_circle),
            Write(elbow_label),
            elbow_circle.animate.set_stroke(width=10),
            flash_radius=0.5
        )
        self.play(Flash(elbow_point, color=GOLD))
        
        self.wait(2)
        
        # Cleanup updaters for safety
        zones.remove_updater(update_zones)
