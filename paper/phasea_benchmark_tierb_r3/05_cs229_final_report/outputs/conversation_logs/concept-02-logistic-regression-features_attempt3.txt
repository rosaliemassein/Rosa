from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # Sigmoid math helper function
        def get_sig_val(z_input):
            return 1.0 / (1.0 + np.exp(-z_input))

        # --- 1. Manual Coordinate System (Built from primitives to avoid disallowed Axes) ---
        # origin for the graph positioned on the right
        start_x_coord = 1.5
        start_y_coord = -1.5
        graph_base_point = [start_x_coord, start_y_coord, 0]
        
        horiz_axis_line = Line(graph_base_point, graph_base_point + [4, 0, 0], color=WHITE)
        vert_axis_line = Line(graph_base_point, graph_base_point + [0, 3, 0], color=WHITE)
        
        label_z_axis = MathTex("z").next_to(horiz_axis_line, RIGHT, buff=0.1)
        label_p_axis = MathTex("P(y=1)").next_to(vert_axis_line, UP, buff=0.1).scale(0.8)
        
        top_ref_line = DashedLine(graph_base_point + [0, 3, 0], graph_base_point + [4, 3, 0], color=GRAY)
        
        self.add(horiz_axis_line, vert_axis_line, label_z_axis, label_p_axis, top_ref_line)

        # --- 2. Logistic Curve ---
        sig_curve_points = []
        # Create 40 segments to approximate the curve
        for step_idx in range(41):
            z_val_calc = -5.0 + step_idx * (10.0 / 40.0)
            # Map z range (-5 to 5) to axis length (0 to 4)
            point_x = start_x_coord + (z_val_calc + 5.0) * 0.4
            # Map Sigmoid range (0 to 1) to axis height (0 to 3)
            point_y = start_y_coord + get_sig_val(z_val_calc) * 3.0
            sig_curve_points.append([point_x, point_y, 0])
        
        logistic_sigmoid_curve = VMobject(color=BLUE)
        logistic_sigmoid_curve.set_points_as_corners(sig_curve_points)
        self.play(Create(logistic_sigmoid_curve))

        # --- 3. Formula Display ---
        # Displaying the logistic regression probability formula at the top
        math_formula = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
        ).scale(0.7).to_edge(UP)
        self.play(Write(math_formula))

        # --- 4. Feature Sliders (Custom built with Line and Dot) ---
        slider_1_x = -4.5
        slider_2_x = -2.0
        
        track_line_1 = Line([slider_1_x, -1.5, 0], [slider_1_x, 1.5, 0], color=GRAY)
        track_line_2 = Line([slider_2_x, -1.5, 0], [slider_2_x, 1.5, 0], color=GRAY)
        
        txt_label_1 = Text("1-Day T-CSP", font_size=18).next_to(track_line_1, DOWN)
        txt_label_2 = Text("30-Day T-CSP", font_size=18).next_to(track_line_2, DOWN)
        
        knob_handle_1 = Dot(color=YELLOW).move_to(track_line_1.get_center())
        knob_handle_2 = Dot(color=YELLOW).move_to(track_line_2.get_center())
        
        self.add(track_line_1, track_line_2, txt_label_1, txt_label_2, knob_handle_1, knob_handle_2)

        # --- 5. Interactive Dot on Curve with Updater ---
        prediction_dot_indicator = Dot(color=RED)
        
        def move_dot_on_sigmoid(mob_obj):
            # Simulation weights for the features
            beta_bias = -2.0
            beta_weight_1 = 3.0
            beta_weight_2 = 3.0
            
            # Extract normalized slider percentages (0.0 to 1.0)
            y_min_pos = track_line_1.get_bottom()[1]
            total_h = track_line_1.get_length()
            
            val_p1 = (knob_handle_1.get_center()[1] - y_min_pos) / total_h
            val_p2 = (knob_handle_2.get_center()[1] - y_min_pos) / total_h
            
            # Logistic linear component: z = b0 + b1*x1 + b2*x2
            z_score_res = beta_bias + (beta_weight_1 * val_p1) + (beta_weight_2 * val_p2)
            
            # Clamp the score to the visible graph range of -5 to 5
            z_score_res = max(-5.0, min(5.0, z_score_res))
            
            # Map the calculated score back to screen coordinates
            screen_pos_x = start_x_coord + (z_score_res + 5.0) * 0.4
            screen_pos_y = start_y_coord + get_sig_val(z_score_res) * 3.0
            mob_obj.move_to([screen_pos_x, screen_pos_y, 0])

        prediction_dot_indicator.add_updater(move_dot_on_sigmoid)
        self.add(prediction_dot_indicator)

        # --- 6. Animation Sequence ---
        # Moving sliders to demonstrate the mapping of feature inputs to probability
        self.play(knob_handle_1.animate.move_to([slider_1_x, 1.2, 0]), run_time=2)
        self.wait(0.5)
        self.play(knob_handle_2.animate.move_to([slider_2_x, 1.0, 0]), run_time=2)
        self.wait(0.5)
        self.play(knob_handle_1.animate.move_to([slider_1_x, -1.3, 0]), run_time=2)
        self.wait(0.5)
        self.play(
            knob_handle_1.animate.move_to(track_line_1.get_center()),
            knob_handle_2.animate.move_to(track_line_2.get_center()),
            run_time=2
        )
        self.wait(2)
