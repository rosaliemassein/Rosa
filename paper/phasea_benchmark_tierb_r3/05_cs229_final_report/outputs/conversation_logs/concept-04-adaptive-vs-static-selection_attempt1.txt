class AdaptiveVsStaticSelection(Scene):

        # 1. Screen Layout and Labels
        line_divider = Line([0, 2.5, 0], [0, -1.5, 0], color=GRAY)
        title_static = Text("Static Benchmark", color=YELLOW).scale(0.7).move_to([-3.5, 3.2, 0])
        title_ml = Text("Logistic Model", color=BLUE).scale(0.7).move_to([3.5, 3.2, 0])
        self.add(line_divider, title_static, title_ml)

        # 2. Floorplan Grids
        # Manual construction of grids using Square and VGroup
        grid_static = VGroup()
        for y in range(4):
            for x in range(5):
                s = Square(side_length=0.4, fill_opacity=0.1, stroke_width=2, color=WHITE)
                s.move_to([-3.5 + x*0.6 - 1.2, y*0.6 - 0.9, 0])
                grid_static.add(s)
        
        grid_ml = VGroup()
        for y in range(4):
            for x in range(5):
                s = Square(side_length=0.4, fill_opacity=0.1, stroke_width=2, color=WHITE)
                s.move_to([3.5 + x*0.6 - 1.2, y*0.6 - 0.9, 0])
                grid_ml.add(s)

        self.add(grid_static, grid_ml)

        # Static zones are "frozen" and highlighted in yellow
        static_indices = [1, 6, 11, 16]
        for i in static_indices:
            grid_static[i].set_fill(YELLOW, opacity=0.7)

        # 3. Timeline Setup (May to September)
        timeline_line = Line([-5, -2, 0], [5, -2, 0], color=WHITE)
        self.add(timeline_line)
        month_names = ["May", "Jun", "Jul", "Aug", "Sep"]
        for i in range(5):
            m_label = Text(month_names[i], font_size=20).move_to([-5 + i*2.5, -2.4, 0])
            self.add(m_label)
        
        progress_dot = Dot([-5, -2, 0], color=WHITE)
        self.add(progress_dot)

        # 4. Graph Labels (Simulating a line chart for recall error)
        graph_label = Text("Missed Requests (Recall Error)", font_size=18).move_to([0, -1.5, 0])
        self.add(graph_label)
        
        # 5. Animation Data and Iteration
        # Pre-defined indices for ML flickering to simulate adaptive recognition
        ml_indices = [
            [2, 3], [1, 2, 7], [0, 4, 5, 9, 14], [6, 7, 8, 11, 12, 13, 17, 18], 
            [0, 1, 2, 10, 11, 12, 15], [14, 15, 16, 17], [1, 6, 11], [5]
        ]
        # Simulated Y values for the lines (Static stays high, ML stays low)
        static_err_y = [0.8, 0.75, 0.9, 0.85, 0.95, 0.8, 0.85, 0.9]
        ml_err_y = [0.3, 0.25, 0.15, 0.2, 0.1, 0.05, 0.1, 0.15]
        
        # Track previous points to draw segments
        graph_y_base = -3.5
        prev_static_p = [-5, graph_y_base + 0.7, 0]
        prev_ml_p = [-5, graph_y_base + 0.4, 0]
        
        # Animation loop to progress across the timeline
        steps = len(ml_indices)
        for i in range(steps):
            new_x = -5 + (i + 1) * (10 / steps)
            target_static_p = [new_x, graph_y_base + static_err_y[i], 0]
            target_ml_p = [new_x, graph_y_base + ml_err_y[i], 0]
            
            # Create graph line segments
            s_seg = Line(prev_static_p, target_static_p, color=YELLOW, stroke_width=2)
            m_seg = Line(prev_ml_p, target_ml_p, color=BLUE, stroke_width=2)
            
            # Prepare ML grid updates (flickering and count changes)
            ml_updates = []
            current_ml_set = ml_indices[i]
            for idx in range(20):
                if idx in current_ml_set:

            # Play one step of the timeline progression

                progress_dot.animate.move_to([new_x, -2, 0]),
                Create(s_seg),
                Create(m_seg),
                *ml_updates,
                run_time=0.7,
                rate_func=lambda t: t # Linear progression
            )
            
            # Move segment pointers forward
            prev_static_p = target_static_p
            prev_ml_p = target_ml_p

        self.wait(2)
