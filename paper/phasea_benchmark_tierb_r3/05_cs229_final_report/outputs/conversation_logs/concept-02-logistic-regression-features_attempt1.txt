from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Setup Axes (Replacement for NumberPlane)
        axes = Axes(
            x_range=[-6, 6, 1],
            y_range=[0, 1.2, 0.5],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(RIGHT, buff=0.5)
        
        y_label = MathTex("P(y=1)", font_size=24).next_to(axes.y_axis, UP)
        x_label = MathTex("z", font_size=24).next_to(axes.x_axis, RIGHT)
        self.add(axes, y_label, x_label)

        # 2. Sigmoid Curve
        def sigmoid_func(val):
            return 1 / (1 + np.exp(-val))

        sigmoid_graph = axes.plot(sigmoid_func, color=BLUE)
        self.play(Create(sigmoid_graph))

        # 3. Sliders (Manual implementation without ValueTracker)
        # We use the y-position of a dot to track values
        slider1_x = -5
        slider2_x = -3.5
        
        line1 = Line(start=[slider1_x, -1.5, 0], end=[slider1_x, 1.5, 0], color=GRAY)
        line2 = Line(start=[slider2_x, -1.5, 0], end=[slider2_x, 1.5, 0], color=GRAY)
        
        # Dots acting as value holders
        dot_1 = Dot(point=[slider1_x, -1.5, 0], color=YELLOW)
        dot_2 = Dot(point=[slider2_x, -1.5, 0], color=YELLOW)
        
        label_1d = Text("1-Day T-CSP", font_size=18).next_to(line1, DOWN)
        label_30d = Text("30-Day T-CSP", font_size=18).next_to(line2, DOWN)
        
        self.add(line1, line2, dot_1, dot_2, label_1d, label_30d)

        # 4. Result Dot and Math logic
        # z = beta0 + beta1 * X1 + beta2 * X2
        # beta0 = -3, beta1 = 2, beta2 = 1.5
        # We map dot y-coord [-1.5, 1.5] to feature values [0, 3]
        
        def get_current_z():
            x1 = dot_1.get_center()[1] + 1.5
            x2 = dot_2.get_center()[1] + 1.5
            return -3.5 + (1.5 * x1) + (1.2 * x2)

        result_dot = Dot(color=RED)
        
        def update_result_dot(mob):
            z_val = get_current_z()
            p_val = sigmoid_func(z_val)
            mob.move_to(axes.c2p(z_val, p_val))

        result_dot.add_updater(update_result_dot)
        
        # 5. Formula and Probability Text
        formula = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=28
        ).to_edge(UP).shift(LEFT * 1.5)
        
        # Redraw text to avoid DecimalNumber
        prob_display = always_redraw(lambda: 
            Text(f"P: {sigmoid_func(get_current_z()):.2f}", font_size=20, color=RED)
            .next_to(result_dot, UR, buff=0.1)
        )

        self.play(Write(formula))
        self.add(result_dot, prob_display)

        # 6. Animation
        # Move sliders and observe the dot
        self.play(
            dot_1.animate.move_to([slider1_x, 1.0, 0]),
            run_time=2,
            rate_func=rate_functions.linear
        )
        self.wait(0.5)
        
        self.play(
            dot_2.animate.move_to([slider2_x, 0.5, 0]),
            run_time=2,
            rate_func=rate_functions.linear
        )
        self.wait(0.5)
        
        # Feature Vector labels
        x_vec = MathTex(r"X = [X_{1d}, X_{30d}]^T", font_size=24, color=GREEN).next_to(label_30d, RIGHT, buff=0.8)
        self.play(FadeIn(x_vec))
        
        self.play(
            dot_1.animate.move_to([slider1_x, -1.0, 0]),
            dot_2.animate.move_to([slider2_x, -1.0, 0]),
            run_time=2,
            rate_func=rate_functions.smooth
        )
        
        self.wait(2)
