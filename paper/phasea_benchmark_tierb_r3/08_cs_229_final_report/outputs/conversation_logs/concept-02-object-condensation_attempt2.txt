from manim import *
import numpy as np

class ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup Title
        title = Text("Object Condensation", font_size=36).to_edge(UP)
        self.add(title)
        
        # 2. Create Grid of Hits (Dots)
        # We use a distribution of dots across the screen
        dots = VGroup(*[
            Dot(point=[x, y, 0], radius=0.08, color=PURPLE)
            for x in np.linspace(-4, 4, 12)
            for y in np.linspace(-2.2, 1.8, 8)
        ])
        self.play(FadeIn(dots))
        self.wait(0.5)

        # 3. Define Condensation Points (High Beta)
        # Choosing two specific indices to act as particle centers
        cp_indices = [35, 62] 
        condensation_points = [dots[cp_indices[0]], dots[cp_indices[1]]]
        
        # 4. Animate Charge Activation (Color and Scale)
        animations = []
        for i, dot in enumerate(dots):
            if i in cp_indices:
                # High Beta: Becomes a large yellow charge
                animations.append(dot.animate.set_color(YELLOW).scale(2.5))
            else:
                # Others: Become slightly blue/smaller
                animations.append(dot.animate.set_color(BLUE_B).scale(0.8))
        
        self.play(*animations, run_time=2)
        self.wait(0.5)

        # 5. Show Formula
        formula = MathTex(
            r"q_i = \text{arctanh}(\beta_i) + q_{min}",
            color=WHITE
        ).to_edge(DOWN, buff=0.5)
        self.play(Write(formula))
        self.wait(0.5)

        # 6. Gravity Wells (Gradient Circles)
        # Creating a visual representation of the potential field
        wells = VGroup()
        for cp in condensation_points:
            center_pos = cp.get_center()
            # Layers of circles to simulate a gradient/well
            well = VGroup(*[
                Circle(radius=r, stroke_width=0, fill_opacity=0.15 - (r * 0.06))
                .set_fill(YELLOW, opacity=0.15 - (r * 0.06))
                .move_to(center_pos)
                for r in np.linspace(0.1, 2.5, 8)
            ])
            wells.add(well)
        
        self.play(FadeIn(wells))
        self.wait(0.5)

        # 7. Pulling hits into wells
        pull_animations = []
        for i, dot in enumerate(dots):
            if i in cp_indices:
                continue
            
            current_pos = dot.get_center()
            cp1_pos = condensation_points[0].get_center()
            cp2_pos = condensation_points[1].get_center()
            
            dist1 = np.linalg.norm(current_pos - cp1_pos)
            dist2 = np.linalg.norm(current_pos - cp2_pos)
            
            # Determine nearest condensation point
            if dist1 < dist2:
                nearest_cp_pos = cp1_pos
                min_dist = dist1
            else:
                nearest_cp_pos = cp2_pos
                min_dist = dist2
            
            # If hit is close enough, pull it toward the center
            if min_dist < 3.0:
                # Calculate target position (moving 70% toward the center)
                target_pos = current_pos + 0.7 * (nearest_cp_pos - current_pos)
                pull_animations.append(dot.animate.move_to(target_pos).set_color(YELLOW))
            else:
                # Distant dots fade out slightly as "noise"
                random_offset = np.array([np.random.uniform(-0.1, 0.1), np.random.uniform(-0.1, 0.1), 0])
                pull_animations.append(dot.animate.shift(random_offset).set_opacity(0.3))

        # Using standard smooth rate function for the movement
        self.play(*pull_animations, run_time=3, rate_func=smooth)
        self.wait(2)
