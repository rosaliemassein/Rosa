from manim import *
import numpy as np

class ConceptLearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Labels
        physical_title = Text("Physical Space", color=WHITE).scale(0.6).shift(LEFT * 3.5 + UP * 3.2)
        learned_title = Text("Learned Space", color=WHITE).scale(0.6).shift(RIGHT * 3.5 + UP * 3.2)

        def create_custom_axes(center):
            h_line = Line(center + LEFT * 2.5, center + RIGHT * 2.5, color=GRAY, stroke_opacity=0.5)
            v_line = Line(center + DOWN * 2.5, center + UP * 2.5, color=GRAY, stroke_opacity=0.5)
            return VGroup(h_line, v_line)
        
        phys_axes = create_custom_axes(LEFT * 3.5)
        learn_axes = create_custom_axes(RIGHT * 3.5)
        
        # 3. Formula
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            color=YELLOW
        ).scale(0.6).to_edge(DOWN, buff=0.3)
        
        self.add(physical_title, learned_title, phys_axes, learn_axes, formula)

        # 4. Create Overlapping Dots in Physical Space
        np.random.seed(42)
        dots_blue = VGroup()
        dots_red = VGroup()
        
        phys_center = LEFT * 3.5
        for _ in range(25):
            # Blue cluster center slightly offset from Red center, but heavily overlapping
            pos_b = phys_center + np.array([np.random.normal(0.1, 0.4), np.random.normal(0.1, 0.4), 0])
            pos_r = phys_center + np.array([np.random.normal(-0.1, 0.4), np.random.normal(-0.1, 0.4), 0])
            dots_blue.add(Dot(pos_b, radius=0.05, color=BLUE))
            dots_red.add(Dot(pos_r, radius=0.05, color=RED))
            
        self.play(FadeIn(dots_blue), FadeIn(dots_red))
        self.wait(1)

        # 5. Define Target Positions in Learned Space (Separated)
        learn_center = RIGHT * 3.5
        target_blue_group = VGroup()
        target_red_group = VGroup()
        
        # Cluster centers in learned space are far apart
        cluster_a_learned = learn_center + LEFT * 1.2 + UP * 1.2
        cluster_b_learned = learn_center + RIGHT * 1.2 + DOWN * 1.2
        
        for i in range(25):
            pos_bt = cluster_a_learned + np.array([np.random.normal(0, 0.4), np.random.normal(0, 0.4), 0])
            pos_rt = cluster_b_learned + np.array([np.random.normal(0, 0.4), np.random.normal(0, 0.4), 0])
            target_blue_group.add(dots_blue[i].animate.move_to(pos_bt))
            target_red_group.add(dots_red[i].animate.move_to(pos_rt))

        # 6. Transformation Animation
        self.play(
            *target_blue_group,
            *target_red_group,
            run_time=2.5,
            rate_func=smooth
        )
        self.wait(0.5)

        # 7. Manual Dashed Circles (Avoiding DashedVMobject as it was undefined)
        def get_dashed_circle(center, radius, color):
            dashes = VGroup()
            num_dashes = 16
            for i in range(num_dashes):
                start_angle = i * (2 * PI / num_dashes)
                dash_angle = PI / num_dashes # half the segment
                dashes.add(Arc(radius=radius, start_angle=start_angle, angle=dash_angle, arc_center=center, color=color))
            return dashes

        circle_blue = get_dashed_circle(cluster_a_learned, 0.9, BLUE)
        circle_red = get_dashed_circle(cluster_b_learned, 0.9, RED)

        self.play(Create(circle_blue), Create(circle_red))
        self.wait(2)
