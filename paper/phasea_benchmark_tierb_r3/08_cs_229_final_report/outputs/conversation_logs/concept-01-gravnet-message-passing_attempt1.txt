from manim import *
import numpy as np

class GravNetGraphAnimation(Scene):
    def construct(self):
        # 1. Setup formula and text based on the prompt
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)", color=BLUE).to_edge(UP)
        
        # Voiceover/Goal description at the bottom
        voiceover_text = Text(
            "GravNet: Connection strength weighted by feature distance", 
            font_size=20
        ).to_edge(DOWN)

        # 2. Create the central hit and its neighbors in a 2D "feature space"
        # We use a 2D Scene to avoid disallowed 3D features
        central_dot = Dot(point=ORIGIN, color=RED, radius=0.15)
        
        neighbors = VGroup()
        num_neighbors = 10
        for i in range(num_neighbors):
            angle = (i / num_neighbors) * 2 * np.pi
            # Irregular hexagonal-like starting points
            dist = 2.0 + np.random.uniform(-0.4, 0.4)
            pos = np.array([
                dist * np.cos(angle),
                dist * np.sin(angle),
                0
            ])
            neighbors.add(Dot(point=pos, color=WHITE, radius=0.1))

        # 3. Create connections with dynamic updaters
        # The connection strength (thickness/opacity) follows exp(-d^2)
        connections = VGroup()

        def update_connection(line):
            # Helper to calculate distance-based potential
            start = central_dot.get_center()
            end = line.target_dot.get_center()
            dist = np.linalg.norm(end - start)
            
            # Apply formula: V = exp(-d^2). 
            # We divide dist by a factor to make the visual effect clear in the 2D frame
            scaled_dist = dist / 2.0
            strength = np.exp(-(scaled_dist**2))
            
            line.put_start_and_end_on(start, end)
            line.set_stroke(width=strength * 15, opacity=strength)

        for neighbor in neighbors:
            conn = Line(central_node=central_dot, target_node=neighbor, color=YELLOW)
            # Attach the specific neighbor to the line object for the updater to reference
            conn.target_dot = neighbor
            conn.add_updater(update_connection)
            connections.add(conn)

        # 4. Display initial state
        self.add(formula, voiceover_text, central_dot, neighbors, connections)
        self.play(
            FadeIn(central_dot),
            FadeIn(neighbors),
            FadeIn(connections),
            Write(formula)
        )
        self.wait(1)

        # 5. Animate "Learned Features" update
        # We move the dots to simulate the network learning a new embedding space
        new_positions_1 = [
            neighbor.get_center() + np.array([np.random.uniform(-1.5, 1.5), np.random.uniform(-1.5, 1.5), 0])
            for neighbor in neighbors
        ]

        self.play(
            *[neighbors[i].animate.move_to(new_positions_1[i]) for i in range(num_neighbors)],
            central_dot.animate.shift(RIGHT * 0.5 + UP * 0.2),
            run_time=4,
            rate_func=rate_functions.smooth
        )
        self.wait(0.5)

        # 6. Second movement phase to show dynamic re-weighting
        final_positions = [
            np.array([np.random.uniform(-4, 4), np.random.uniform(-2, 2), 0])
            for _ in range(num_neighbors)
        ]

        self.play(
            *[neighbors[i].animate.move_to(final_positions[i]) for i in range(num_neighbors)],
            central_dot.animate.move_to(ORIGIN),
            run_time=4,
            rate_func=rate_functions.smooth
        )

        # Keep scene open to observe final structure
        self.wait(2)
