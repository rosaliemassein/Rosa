from manim import *
import numpy as np

class Concept02ObjectCondensation(Scene):
    def construct(self):
        # 1. Create a 2D plane of dots
        # Using a grid of dots to represent the initial hits
        dots = VGroup()
        rows, cols = 9, 9
        for i in range(-(rows // 2), rows // 2 + 1):
            for j in range(-(cols // 2), cols // 2 + 1):
                # Purple is the starting color for low beta
                dot = Dot(point=[i * 0.8, j * 0.8, 0], radius=0.08, color=PURPLE)
                dots.add(dot)

        self.add(dots)

        # 2. Assign beta values and update visuals
        # Choose specific indices to act as condensation points (high beta)
        condensation_indices = [12, 40, 65]
        
        def get_beta(point):
            # Simulated beta value based on proximity to condensation points
            dists = [np.linalg.norm(point - dots[idx].get_center()) for idx in condensation_indices]
            min_dist = min(dists)
            # Beta closer to 1 near centers, closer to 0 far away
            return np.exp(-min_dist * 1.2)

        growth_anims = []
        for dot in dots:
            beta = get_beta(dot.get_center())
            # Use interpolate_color from manim.utils.color (imported via manim *)
            # Color change: purple (low beta) to yellow (high beta)
            new_color = interpolate_color(PURPLE, YELLOW, beta)
            # Size change: dots with high beta grow significantly
            growth_anims.append(dot.animate.scale(1 + 2.5 * beta).set_color(new_color))

        self.play(AnimationGroup(*growth_anims), run_time=2)
        self.wait(0.5)

        # 3. Draw 'gravity wells' around high-beta dots
        # Visualized as concentric rings with fading opacity
        wells = VGroup()
        for idx in condensation_indices:
            center_pos = dots[idx].get_center()
            well = VGroup(*[
                Circle(radius=r, stroke_width=1.5, stroke_opacity=1 - (r / 1.5), color=YELLOW)
                for r in np.linspace(0.1, 1.5, 6)
            ]).move_to(center_pos)
            wells.add(well)

        self.play(FadeIn(wells, lag_ratio=0.1))

        # 4. Animate neighboring dots being pulled toward the center
        pull_anims = []
        for dot in dots:
            pos = dot.get_center()
            # Find the nearest condensation center to determine pull direction
            dists = [np.linalg.norm(pos - dots[idx].get_center()) for idx in condensation_indices]
            nearest_idx = condensation_indices[np.argmin(dists)]
            target_pos = dots[nearest_idx].get_center()
            
            # Only pull dots that aren't already the centers
            dist = np.linalg.norm(pos - target_pos)
            if 0.1 < dist < 2.5:
                # Manual linear interpolation for position to avoid undefined 'interpolate' identifier
                # Move the dot 60% of the way toward the condensation point
                alpha = 0.6
                new_pos = pos * (1 - alpha) + target_pos * alpha
                pull_anims.append(dot.animate.move_to(new_pos))

        # 5. Label the charge calculation formula
        # q_i = arctanh(beta_i) + q_min
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula.to_edge(UP).shift(RIGHT * 3)
        background_rect = SurroundingRectangle(formula, color=BLACK, fill_opacity=0.8, buff=0.1)
        
        self.play(
            FadeIn(background_rect),
            Write(formula),
            AnimationGroup(*pull_anims, lag_ratio=0.02),
            run_time=3
        )
        self.wait(2)
