from manim import *
import numpy as np

class ConceptObjectCondensation(Scene):
    def construct(self):
        # 1. Setup Configuration
        num_dots = 20
        q_min = 0.1
        np.random.seed(42)
        
        # 2. Create Initial State
        # A 2D plane of dots, starting as small purple hits
        dots = VGroup()
        beta_values = np.random.uniform(0.1, 0.9, num_dots)
        positions = [
            np.array([np.random.uniform(-5, 5), np.random.uniform(-3, 2), 0])
            for _ in range(num_dots)
        ]
        
        for pos in positions:
            dot = Dot(point=pos, radius=0.08, color=PURPLE)
            dots.add(dot)
            
        self.add(dots)
        self.wait(1)

        # 3. Formula Display
        # Show the charge calculation formula
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}", font_size=36)
        formula.to_edge(UP)
        self.play(Write(formula))
        
        # 4. Animate Beta Growth (Color and Size)
        # Dots grow and change color from purple to yellow based on beta
        # We use a manual index into a color gradient to ensure compatibility
        colors_list = color_gradient([PURPLE, YELLOW], 100)
        
        animations = []
        for i, dot in enumerate(dots):
            beta = beta_values[i]
            color_idx = int(beta * 99)
            target_color = colors_list[color_idx]
            # Scale the dot based on the beta value (simulated confidence)
            target_scale = 1 + beta * 2.5
            animations.append(
                dot.animate.set_color(target_color).scale(target_scale)
            )
        
        self.play(*animations, run_time=2)
        self.wait(1)

        # 5. Create Gravity Wells
        # Identify high-beta dots as condensation points (top 3)
        high_beta_indices = np.argsort(beta_values)[-3:]
        wells = VGroup()
        
        well_fade_ins = []
        for idx in high_beta_indices:
            center = dots[idx].get_center()
            # Create a concentric "gravity well" using circles with decreasing opacity
            well = VGroup(*[
                Circle(
                    radius=r, 
                    color=YELLOW, 
                    stroke_width=0, 
                    fill_opacity=0.2 * (1 - r/1.5)
                ).move_to(center)
                for r in np.linspace(0.1, 1.5, 8)
            ])
            wells.add(well)
            well_fade_ins.append(FadeIn(well, scale=0.8))
            
        self.play(*well_fade_ins)
        self.wait(0.5)

        # 6. Physical Pull Animation
        # Neighboring low-beta dots are pulled toward the condensation points
        pull_animations = []
        for i, dot in enumerate(dots):
            if i in high_beta_indices:
                continue
            
            # Find the nearest condensation point
            well_centers = [dots[idx].get_center() for idx in high_beta_indices]
            dists = [np.linalg.norm(dot.get_center() - wc) for wc in well_centers]
            nearest_well_center = well_centers[np.argmin(dists)]
            
            # Move the dot 75% of the way to the center of the well
            # Use 'smooth' as the rate function
            target_pos = dot.get_center() + (nearest_well_center - dot.get_center()) * 0.75
            pull_animations.append(dot.animate.move_to(target_pos))

        self.play(*pull_animations, run_time=2, rate_func=smooth)
        self.wait(2)

        # Final cleanup
        self.play(FadeOut(wells), FadeOut(dots), FadeOut(formula))
        self.wait(1)
