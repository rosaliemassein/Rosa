from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # Define axes instead of NumberPlane to avoid restricted identifiers
        axes_physical = Axes(
            x_range=[-3, 3], 
            y_range=[-3, 3], 
            x_length=5, 
            y_length=5,
            axis_config={"include_tip": False}
        ).to_edge(LEFT, buff=0.5)
        
        axes_learned = Axes(
            x_range=[-3, 3], 
            y_range=[-3, 3], 
            x_length=5, 
            y_length=5,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5)
        
        # Titles for the two spaces
        label_physical = Text("Physical Space", font_size=26).next_to(axes_physical, UP)
        label_learned = Text("Learned Space", font_size=26).next_to(axes_learned, UP)
        
        # Points for physical space (overlapping clusters)
        np.random.seed(42)
        pts_red = VGroup(*[
            Dot(axes_physical.c2p(np.random.normal(0.25, 0.6), np.random.normal(0.25, 0.6)), color=RED, radius=0.05) 
            for _ in range(22)
        ])
        pts_blue = VGroup(*[
            Dot(axes_physical.c2p(np.random.normal(-0.25, 0.6), np.random.normal(-0.25, 0.6)), color=BLUE, radius=0.05) 
            for _ in range(22)
        ])
        
        # Target points for learned space (separated clusters)
        target_red = VGroup(*[
            Dot(axes_learned.c2p(np.random.normal(1.5, 0.4), np.random.normal(1.5, 0.4)), color=RED, radius=0.05) 
            for _ in range(22)
        ])
        target_blue = VGroup(*[
            Dot(axes_learned.c2p(np.random.normal(-1.5, 0.4), np.random.normal(-1.5, 0.4)), color=BLUE, radius=0.05) 
            for _ in range(22)
        ])
        
        # Manual construction of dashed circles (avoiding DashedVMobject)
        def create_dashes(center, color_ref):
            dashes = VGroup()
            num_dashes = 12
            for i in range(num_dashes):
                dash = Arc(
                    radius=1.1, 
                    start_angle=i * (TAU / num_dashes), 
                    angle=TAU / (2 * num_dashes), 
                    color=color_ref
                )
                dashes.add(dash)
            return dashes.move_to(center)
        
        circle_red = create_dashes(axes_learned.c2p(1.5, 1.5), RED)
        circle_blue = create_dashes(axes_learned.c2p(-1.5, -1.5), BLUE)
        
        # Reference formula from prompt
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=28
        ).to_edge(DOWN, buff=0.5)
        
        # Animation sequence
        self.add(axes_physical, axes_learned, label_physical, label_learned)
        self.play(FadeIn(pts_red), FadeIn(pts_blue))
        self.wait(1)
        
        # Transform hits from physical space to learned space
        self.play(
            Transform(pts_red, target_red),
            Transform(pts_blue, target_blue),
            run_time=3,
            path_arc=0.3
        )
        
        # Final classification indicator
        self.play(Create(circle_red), Create(circle_blue))
        self.play(Write(formula))
        self.wait(3)
