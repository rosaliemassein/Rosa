from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Formula setup
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_edge(UP)
        
        # 2. Irregular hexagonal-like grid setup
        # We use standard Scene (2D) to avoid compatibility issues with 3D features
        center_pos = ORIGIN
        neighbor_offsets = [
            np.array([2.0, 0.8, 0]), 
            np.array([0.3, 2.2, 0]), 
            np.array([-1.8, 1.5, 0]),
            np.array([-2.2, -0.6, 0]), 
            np.array([-0.5, -2.0, 0]), 
            np.array([1.7, -1.6, 0])
        ]
        
        # Create the dots
        central_dot = Dot(center_pos, color=YELLOW, radius=0.15)
        neighbors = VGroup(*[Dot(pos, color=BLUE, radius=0.1) for pos in neighbor_offsets])
        all_dots = VGroup(central_dot, *neighbors)
        
        # 3. Dynamic Connections (Arrows with Updaters)
        # The remarks specify arrows' opacity and thickness scaling with exp(-d^2)
        connections = VGroup()
        for neighbor in neighbors:
            # We initialize with a dummy arrow
            arrow = Arrow(central_dot.get_center(), neighbor.get_center(), buff=0.1)
            
            def arrow_updater(obj, target=neighbor):
                p1 = central_dot.get_center()
                p2 = target.get_center()
                # Calculate distance in the 'feature space'
                dist = np.linalg.norm(p1 - p2)
                # Apply the formula V = exp(-d^2). Scaling distance for visual clarity.
                weight = np.exp(-(dist**2) / 4.0)
                
                # Update the arrow's appearance
                new_arrow = Arrow(
                    p1, p2, 
                    buff=0.1, 
                    stroke_width=weight * 14, # Thickness scales with distance
                    max_tip_length_to_length_ratio=0.2,
                    color=GREEN
                )
                new_arrow.set_opacity(weight) # Opacity scales with distance
                obj.become(new_arrow)
                
            arrow.add_updater(arrow_updater)
            connections.add(arrow)

        # 4. Animation Sequence
        self.play(Write(formula))
        self.play(FadeIn(all_dots, shift=UP))
        self.play(Create(connections))
        self.wait(1)

        # 5. Dynamic movement (Update Learned Features)
        # Defining new positions to show how the graph structure shifts
        new_learned_positions = [
            np.array([0.5, -0.4, 0]),    # Central dot moves
            np.array([2.8, 1.2, 0]),     # N1 moves further away
            np.array([0.6, 0.9, 0]),     # N2 moves significantly closer
            np.array([-3.0, 1.8, 0]),    # N3 moves further away
            np.array([-0.8, -0.8, 0]),   # N4 moves closer
            np.array([0.0, -3.0, 0]),    # N5 moves further away
            np.array([1.8, -0.2, 0])     # N6 moves closer
        ]

        # Prepare animations for dots. The connections will update automatically via updaters.
        move_anims = []
        move_anims.append(central_dot.animate.move_to(new_learned_positions[0]))
        for i in range(len(neighbors)):
            move_anims.append(neighbors[i].animate.move_to(new_learned_positions[i+1]))

        # Using standard 'smooth' rate function to avoid undefined identifier errors
        self.play(*move_anims, run_time=4, rate_func=smooth)
        self.wait(1)

        # 6. Final jitter to emphasize the dynamic nature of the graph
        jitter_offsets = [
            np.array([0.15, -0.1, 0]), np.array([-0.1, 0.15, 0]),
            np.array([0.1, 0.1, 0]), np.array([-0.15, -0.1, 0]),
            np.array([0.1, 0.0, 0]), np.array([-0.1, -0.15, 0]),
            np.array([0.0, 0.15, 0])
        ]
        
        # Jitter out
        self.play(
            *[all_dots[i].animate.shift(jitter_offsets[i]) for i in range(len(all_dots))],
            run_time=0.7, 
            rate_func=smooth
        )
        # Jitter back
        self.play(
            *[all_dots[i].animate.shift(-jitter_offsets[i]) for i in range(len(all_dots))],
            run_time=0.7, 
            rate_func=smooth
        )
        
        self.wait(2)
