from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Define screen sections manually using Lines to avoid forbidden Axes/NumberPlane
        l_origin = LEFT * 3.5
        r_origin = RIGHT * 3.5
        
        # Physical Space Grid/Axes
        axes_l = VGroup(
            Line(l_origin + LEFT * 2.5, l_origin + RIGHT * 2.5, color=GRAY, stroke_opacity=0.5),
            Line(l_origin + DOWN * 2.5, l_origin + UP * 2.5, color=GRAY, stroke_opacity=0.5)
        )
        
        # Learned Space Grid/Axes
        axes_r = VGroup(
            Line(r_origin + LEFT * 2.5, r_origin + RIGHT * 2.5, color=GRAY, stroke_opacity=0.5),
            Line(r_origin + DOWN * 2.5, r_origin + UP * 2.5, color=GRAY, stroke_opacity=0.5)
        )
        
        title_l = Text("Physical Space", font_size=24).next_to(axes_l, UP)
        title_r = Text("Learned Space", font_size=24).next_to(axes_r, UP)
        
        # 2. Setup Dots for Physical Space (Overlapping Clusters)
        np.random.seed(42)
        dots_red = VGroup()
        dots_green = VGroup()
        for _ in range(25):
            # Random points clustered near origin with high overlap
            p_red = l_origin + np.array([np.random.normal(0, 0.45), np.random.normal(0, 0.45), 0])
            p_green = l_origin + np.array([np.random.normal(0.3, 0.5), np.random.normal(-0.3, 0.5), 0])
            dots_red.add(Dot(p_red, color=RED, radius=0.07))
            dots_green.add(Dot(p_green, color=GREEN, radius=0.07))
            
        # 3. Setup Target Positions for Learned Space (Separated Clusters)
        target_red = VGroup()
        target_green = VGroup()
        for _ in range(25):
            # Separated centers in the learned coordinate system
            tr = r_origin + LEFT * 1.2 + UP * 1.2 + np.array([np.random.normal(0, 0.35), np.random.normal(0, 0.35), 0])
            tg = r_origin + RIGHT * 1.2 + DOWN * 1.2 + np.array([np.random.normal(0, 0.35), np.random.normal(0, 0.35), 0])
            target_red.add(Dot(tr, color=RED, radius=0.07))
            target_green.add(Dot(tg, color=GREEN, radius=0.07))
            
        # 4. Manual Dashed Circles using Arc segments (avoiding forbidden DashedVMobject)
        def get_custom_dashed_circle(center, radius, color):
            dashed_group = VGroup()
            for i in range(20):
                segment = Arc(
                    radius=radius, 
                    start_angle=i * 18 * DEGREES, 
                    angle=9 * DEGREES, 
                    color=color,
                    stroke_width=2
                ).shift(center)
                dashed_group.add(segment)
            return dashed_group

        circle_r = get_custom_dashed_circle(r_origin + LEFT * 1.2 + UP * 1.2, 1.1, RED)
        circle_g = get_custom_dashed_circle(r_origin + RIGHT * 1.2 + DOWN * 1.2, 1.1, GREEN)
        
        # 5. Formula and Narration
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=26
        ).to_edge(DOWN, buff=0.5)
        
        narration_text = Text(
            "Mapping complex physical hits into separable blobs in learned space.",
            font_size=18,
            slant=ITALIC
        ).to_edge(UP, buff=0.4)

        # 6. Execute Animation
        self.add(axes_l, axes_r, title_l, title_r)
        self.play(FadeIn(dots_red), FadeIn(dots_green))
        self.wait(1)
        
        # Transform the dots across the screen
        self.play(
            Transform(dots_red, target_red),
            Transform(dots_green, target_green),
            run_time=3,
            rate_func=slow_into
        )
        
        # Highlight classification
        self.play(Create(circle_r), Create(circle_g))
        self.play(Write(formula), Write(narration_text))
        self.wait(3)
