from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Setup: Create an irregular grid of dots representing ECal cells
        # We use a 2D plane to adhere to constraints while representing the concept
        dots = VGroup()
        for x in range(-5, 6):
            for y in range(-3, 4):
                # Position with randomness to create an "irregular hexagonal-like grid"
                pos = np.array([x * 1.1, y * 1.1, 0]) + np.random.uniform(-0.4, 0.4, 3)
                pos[2] = 0  # Force coordinate to be 2D
                dots.add(Dot(pos, radius=0.08, color=WHITE))
        
        dots.move_to(ORIGIN)
        
        # 2. Highlight a central dot
        center_dot = dots[len(dots)//2]
        
        # 3. Create connections with dynamic weight-based styling
        # Connection strength V_jk = exp(-d_jk^2)
        edges = VGroup()
        
        # Create a helper function for the updater to avoid closure/scoping issues
        def get_edge_updater(target_node):
            def updater(mob):
                p1 = center_dot.get_center()
                p2 = target_node.get_center()
                distance = np.linalg.norm(p2 - p1)
                # Apply the potential formula: V = exp(-d^2)
                # A scaling factor (0.7) is used for better visual dynamics
                weight = np.exp(-(distance * 0.7)**2)
                mob.put_start_and_end_on(p1, p2)
                # Scale thickness and opacity based on the learned potential
                mob.set_stroke(width=weight * 12, opacity=weight)
            return updater

        for d in dots:
            if d == center_dot:
                continue
            
            # Only connect to dots within a specific range for the "local neighborhood"
            initial_dist = np.linalg.norm(d.get_center() - center_dot.get_center())
            if initial_dist < 3.2:
                edge = Line(center_dot.get_center(), d.get_center(), color=YELLOW)
                edge.add_updater(get_edge_updater(d))
                edges.add(edge)
        
        # 4. Math and UI elements
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_edge(UP, buff=0.5)
        
        # 5. Animation Sequence
        self.play(FadeIn(dots))
        self.wait(0.5)
        
        # Highlight the central node
        self.play(
            center_dot.animate.set_color(RED).scale(1.5),
            Flash(center_dot, color=RED, line_length=0.2)
        )
        
        # Show graph connections and the governing formula
        self.play(
            Create(edges),
            Write(formula),
            run_time=2
        )
        self.wait(1)
        
        # 6. Dynamic Motion: Simulating the update of "learned features"
        # As the hits move in feature space, the graph edges update dynamically
        move_anims = []
        for d in dots:
            # Shift each dot by a random vector to simulate feature space updates
            shift_vector = np.random.uniform(-1.3, 1.3, 3)
            shift_vector[2] = 0
            move_anims.append(d.animate.shift(shift_vector))
        
        # Execute the movement; the edge updaters will handle visual updates automatically
        self.play(*move_anims, run_time=5, rate_func=smooth)
        self.wait(2)
        
        # 7. Scene Cleanup
        self.play(
            FadeOut(dots),
            FadeOut(edges),
            FadeOut(formula)
        )
        self.wait(1)
