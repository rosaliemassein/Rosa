from manim import *
import numpy as np

class ZeroShotClassification(Scene):
    def construct(self):
        # 1. Setup - Vector v and Image representation
        image_box = Rectangle(width=3.5, height=2.5, color=WHITE).to_edge(UL, buff=1)
        image_label = Text("New Image", font_size=24).move_to(image_box)
        
        # Representing the vector v
        v_vec = MathTex(
            "\\mathbf{v} = \\begin{bmatrix} v_1 \\\\ v_2 \\\\ \\vdots \\\\ v_d \\end{bmatrix}"
        ).scale(0.8).next_to(image_box, RIGHT, buff=1)
        v_label = MathTex("\\text{Image Vector}").scale(0.6).next_to(v_vec, UP)

        self.play(Create(image_box), Write(image_label))
        self.play(Write(v_vec), Write(v_label))
        self.wait(0.5)

        # 2. Matrix X (Text Embeddings for Countries)
        country_labels_str = ["France", "Japan", "Brazil", "...", "India", "Zambia"]
        x_rows = VGroup(*[
            Text(name, font_size=18) for name in country_labels_str
        ]).arrange(DOWN, buff=0.2)
        
        x_rect = SurroundingRectangle(x_rows, color=BLUE, buff=0.4)
        x_name = MathTex("\\mathbf{X}").next_to(x_rect, UP)
        x_desc = Text("140 Country Embeddings", font_size=16, color=BLUE).next_to(x_rect, DOWN)
        x_group = VGroup(x_rect, x_rows, x_name, x_desc).next_to(v_vec, RIGHT, buff=1.2)

        self.play(Create(x_rect), Write(x_rows), Write(x_name), FadeIn(x_desc))
        self.wait(1)

        # 3. Dot Product pulsing highlight
        highlight = Rectangle(
            width=x_rect.width - 0.2, 
            height=0.3, 
            color=YELLOW, 
            fill_opacity=0.4, 
            stroke_width=0
        )
        highlight.move_to(x_rows[0])
        
        dot_product_tex = MathTex("\\mathbf{X}\\mathbf{v}").next_to(v_vec, DOWN, buff=1).shift(RIGHT*2)
        
        self.play(FadeIn(highlight))
        for i in [0, 1, 2, 4, 5]: 
            self.play(
                highlight.animate.move_to(x_rows[i]),
                Indicate(x_rows[i], color=YELLOW, scale_factor=1.1),
                run_time=0.4
            )
        self.play(FadeOut(highlight), Write(dot_product_tex))
        self.wait(1)

        # 4. Mathematical Transformation to Softmax
        formula = MathTex(
            "h(\\mathbf{v};\\mathbf{X}) = \\frac{\\exp(\\mathbf{X}\\mathbf{v})}{\\sum_{j=1}^{N}\\exp(\\mathbf{X}_{j}^{T}\\mathbf{v})}",
            font_size=42
        ).to_edge(UP, buff=0.8)

        self.play(
            FadeOut(image_box), FadeOut(image_label), 
            FadeOut(v_vec), FadeOut(v_label),
            FadeOut(x_group),
            ReplacementTransform(dot_product_tex, formula)
        )
        self.wait(1)

        # 5. Manual Bar Chart implementation (to avoid undefined BarChart)
        axes = Axes(
            x_range=[0, 5, 1],
            y_range=[0, 1, 0.2],
            x_length=8,
            y_length=4,
            axis_config={"include_tip": False}
        ).shift(DOWN)
        
        chart_labels = ["USA", "France", "Japan", "Brazil", "India"]
        x_labels = VGroup()
        for i, name in enumerate(chart_labels):
            lab = Text(name, font_size=20).next_to(axes.c2p(i + 0.5, 0), DOWN)
            x_labels.add(lab)

        def get_bars(values, colors=[BLUE, GREEN, BLUE, BLUE, BLUE]):
            bars = VGroup()
            for i, val in enumerate(values):
                # Create a bar using Rectangle
                p1 = axes.c2p(i + 0.2, 0)
                p2 = axes.c2p(i + 0.8, val)
                rect = Rectangle(
                    color=colors[i],
                    fill_opacity=0.8,
                    stroke_width=1
                )
                rect.replace(Line(p1, p2), stretch=True)
                # Ensure the bar sits on the axis
                rect.move_to(axes.c2p(i + 0.5, val/2))
                bars.add(rect)
            return bars

        initial_probs = [0.15, 0.2, 0.15, 0.15, 0.15]
        bars = get_bars(initial_probs)

        self.play(Create(axes), Write(x_labels))
        self.play(Create(bars))
        self.wait(1)

        # 6. Animation: France (index 1) shoots up
        final_probs = [0.05, 0.8, 0.05, 0.05, 0.05]
        final_bars = get_bars(final_probs)
        # Highlight France in Green
        final_bars[1].set_color(GREEN)

        self.play(
            Transform(bars, final_bars),
            run_time=2,
            rate_func=rate_functions.smooth
        )
        
        prediction_text = Text("Prediction: France", font_size=36, color=GREEN).next_to(axes, DOWN, buff=0.7)
        self.play(Write(prediction_text), Flash(bars[1], color=GREEN))
        
        self.wait(2)
