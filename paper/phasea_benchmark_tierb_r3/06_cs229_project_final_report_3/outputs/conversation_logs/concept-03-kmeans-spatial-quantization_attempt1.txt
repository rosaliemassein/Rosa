from manim import *
import numpy as np

class KMeansSpatialQuantization(Scene):
    def construct(self):
        # 1. Setup Axes instead of NumberPlane
        axes = Axes(
            x_range=[-7, 7, 1],
            y_range=[-4, 4, 1],
            axis_config={"include_tip": False, "stroke_opacity": 0.5}
        )
        self.add(axes)

        # 2. Draw Australia-like outline
        australia_points = [
            [2, 1, 0], [4, 0.5, 0], [5, -1, 0], [4.5, -2.5, 0], 
            [2, -3, 0], [0, -2.5, 0], [-1, -1, 0], [0, 0.5, 0]
        ]
        australia_outline = Polygon(*australia_points, color=WHITE, stroke_width=4)
        australia_outline.scale(1.2).shift(LEFT * 0.5)
        country_label = Text("Australia", font_size=24).next_to(australia_outline, UP)
        
        self.play(Create(australia_outline), Write(country_label))

        # 3. Generate random data points
        np.random.seed(42)
        points_list = []
        for _ in range(60):
            pt = australia_outline.get_center() + np.array([
                np.random.uniform(-2.2, 2.2),
                np.random.uniform(-1.8, 1.2),
                0
            ])
            points_list.append(Dot(pt, radius=0.06, color=GRAY))
        
        country_points = VGroup(*points_list)
        self.play(FadeIn(country_points))
        self.wait(0.5)

        # 4. Show Centroid Formula
        formula = MathTex(
            r"\mu_{j}:=\frac{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}x^{(i)}}{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}}",
            font_size=32
        ).to_edge(UP, buff=0.5)
        formula_bg = Rectangle(
            width=formula.get_width() + 0.5, 
            height=formula.get_height() + 0.3,
            fill_color=BLACK, 
            fill_opacity=0.8, 
            stroke_color=YELLOW
        ).move_to(formula)
        
        self.play(FadeIn(formula_bg), Write(formula))

        # 5. Create manual Centroid crosses (Cross replacement)
        centroid_colors = [RED, GREEN, YELLOW, BLUE, ORANGE]
        K = 5
        centroids = VGroup()
        for i in range(K):
            pos = australia_outline.get_center() + np.array([
                np.random.uniform(-1.5, 1.5),
                np.random.uniform(-1, 1),
                0
            ])
            # Manual cross shape
            c_cross = VGroup(
                Line(0.15 * LEFT, 0.15 * RIGHT),
                Line(0.15 * UP, 0.15 * DOWN)
            ).set_stroke(color=centroid_colors[i], width=6).move_to(pos)
            centroids.add(c_cross)

        self.play(Create(centroids))
        self.wait(0.5)

        # 6. K-means Iteration Logic
        def get_assignment_anims():
            anims = []
            for p in country_points:
                dists = [np.linalg.norm(p.get_center() - c.get_center()) for c in centroids]
                nearest_idx = np.argmin(dists)
                anims.append(p.animate.set_color(centroid_colors[nearest_idx]))
            return anims

        def get_update_anims():
            anims = []
            for i in range(K):
                assigned_pts = [p.get_center() for p in country_points if p.get_color() == centroid_colors[i]]
                if len(assigned_pts) > 0:
                    new_pos = np.mean(assigned_pts, axis=0)
                    anims.append(centroids[i].animate.move_to(new_pos))
            return anims

        # Perform 2 iterations
        self.play(*get_assignment_anims(), run_time=1.5)
        self.play(*get_update_anims(), run_time=1.5)
        self.wait(0.5)
        self.play(*get_assignment_anims(), run_time=1.5)
        self.play(*get_update_anims(), run_time=1.5)

        # 7. Final highlight
        narration_text = Text(
            "Discretizing space into predictable regions.",
            font_size=24,
            color=YELLOW
        ).to_edge(DOWN, buff=0.5)
        
        self.play(Write(narration_text))
        self.play(
            *[Indicate(c, scale_factor=1.5) for c in centroids],
            country_points.animate.set_opacity(0.4)
        )
        self.wait(2)
