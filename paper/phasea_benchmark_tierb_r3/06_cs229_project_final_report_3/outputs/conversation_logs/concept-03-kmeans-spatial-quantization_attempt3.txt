from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # Helper to create a Cross since 'Cross' might be restricted/undefined
        def create_cross_marker(color=WHITE):
            return VGroup(
                Line(0.15 * LEFT + 0.15 * UP, 0.15 * RIGHT + 0.15 * DOWN, color=color, stroke_width=4),
                Line(0.15 * LEFT + 0.15 * DOWN, 0.15 * RIGHT + 0.15 * UP, color=color, stroke_width=4)
            )

        # 1. Use Axes instead of NumberPlane to avoid restricted features
        axes = Axes(
            x_range=[-6, 6, 1],
            y_range=[-4, 4, 1],
            x_length=10,
            y_length=6,
            axis_config={"color": BLUE, "stroke_opacity": 0.5},
            tips=False
        )
        self.add(axes)

        # 2. Draw an outline of a country (Australia-like shape)
        australia_points = [
            [-3, 1.5, 0], [-1, 1.8, 0], [1, 1.5, 0], [3, 0.5, 0], 
            [3.5, -1, 0], [2, -2, 0], [0, -2.5, 0], [-2, -2, 0], 
            [-3.5, -0.5, 0], [-3, 1, 0]
        ]
        country_outline = Polygon(*australia_points, color=BLUE, fill_opacity=0.1)
        self.play(Create(country_outline))

        # 3. Populate with many small gray dots
        np.random.seed(42)
        dots = VGroup()
        for _ in range(70):
            # Simple rejection sampling for the shape
            px = np.random.uniform(-3.5, 3.5)
            py = np.random.uniform(-2.5, 2.0)
            dots.add(Dot(point=[px, py, 0], radius=0.05, color=GREY))
        
        self.play(FadeIn(dots, lag_ratio=0.1))
        self.wait(0.5)

        # 4. Animate Centroid crosses appearing
        colors = [RED, YELLOW, GREEN, PURPLE]
        centroid_starts = [[-1.5, 1, 0], [1.5, 1, 0], [-1, -1, 0], [2, -1, 0]]
        centroids = VGroup()
        for i in range(len(colors)):
            c = create_cross_marker(color=colors[i]).move_to(centroid_starts[i])
            centroids.add(c)
        
        self.play(AnimationGroup(*[Create(c) for c in centroids], lag_ratio=0.3))
        self.wait(1)

        # 5. Animate points changing color based on nearest centroid
        def update_dot_colors():
            anims = []
            for dot in dots:
                dists = [np.linalg.norm(dot.get_center() - c.get_center()) for c in centroids]
                nearest_idx = np.argmin(dists)
                anims.append(dot.animate.set_color(colors[nearest_idx]))
            return anims

        self.play(*update_dot_colors())
        self.wait(0.5)

        # 6. Formula display
        formula = MathTex(
            r"\mu_{j}:=\frac{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}x^{(i)}}{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}}",
            font_size=34
        ).to_edge(UP).add_background_rectangle()
        self.play(Write(formula))

        # 7. Move centroids to the center of their respective groups
        def get_move_anims():
            m_anims = []
            for i in range(len(centroids)):
                # Find dots assigned to this centroid color
                group_points = [d.get_center() for d in dots if d.get_color() == colors[i]]
                if group_points:
                    new_pos = np.mean(group_points, axis=0)
                    m_anims.append(centroids[i].animate.move_to(new_pos))
            return m_anims

        self.play(*get_move_anims())
        self.wait(0.5)

        # 8. Second iteration for convergence look
        self.play(*update_dot_colors())
        self.play(*get_move_anims())
        self.wait(1)

        # 9. Final goal text and highlighting targets
        goal_text = Text(
            "Discretizing space into predictable regions",
            font_size=28
        ).to_edge(DOWN).add_background_rectangle()
        
        self.play(
            FadeIn(goal_text),
            *[c.animate.scale(1.5) for c in centroids]
        )
        self.wait(2)
