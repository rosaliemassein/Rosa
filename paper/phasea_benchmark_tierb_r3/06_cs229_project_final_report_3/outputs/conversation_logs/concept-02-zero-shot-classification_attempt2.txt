from manim import *

class ZeroShotClassification(Scene):
    def construct(self):
        # 1. Create the matrix X with country names
        countries = ["France", "Germany", "Italy", "Spain", "Japan", "Brazil", "Canada", "Australia", "India", "USA"]
        X_labels = VGroup(*[
            Text(name, font_size=20) for name in countries
        ]).arrange(DOWN, aligned_edge=LEFT, buff=0.3)
        
        X_box = SurroundingRectangle(X_labels, color=BLUE, buff=0.3)
        X_title = Text("Matrix X (Embeddings)", font_size=24).next_to(X_box, UP)
        X_group = VGroup(X_labels, X_box, X_title).to_edge(LEFT, buff=0.5)

        # 2. Create the vector v representing the image
        v_rect = Rectangle(height=4, width=0.6, color=RED, fill_opacity=0.2)
        v_label = MathTex(r"\mathbf{v}").next_to(v_rect, UP)
        image_label = Text("Image Vector", font_size=20).next_to(v_label, UP)
        v_group = VGroup(v_rect, v_label, image_label).next_to(X_group, RIGHT, buff=1.5)

        self.play(FadeIn(X_group), FadeIn(v_group))
        self.wait(1)

        # 3. Pulsing highlight for dot product operation
        highlight = Rectangle(
            width=X_box.get_width(), 
            height=0.4, 
            color=YELLOW, 
            fill_opacity=0.3, 
            stroke_width=0
        ).move_to(X_labels[0].get_center())
        
        dot_product_label = MathTex(r"\mathbf{X}_j^T \mathbf{v}", color=YELLOW).next_to(v_group, RIGHT, buff=1)
        
        self.play(FadeIn(highlight), Write(dot_product_label))
        
        for i in range(1, 5):
            self.play(highlight.animate.move_to(X_labels[i].get_center()), run_time=0.4)
        
        self.play(FadeOut(highlight))
        self.wait(0.5)

        # 4. Manual Bar Chart construction for better compatibility
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 1, 0.5],
            x_length=6,
            y_length=3,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)

        final_probs = [0.8, 0.05, 0.02, 0.02, 0.02, 0.02, 0.02, 0.02, 0.02, 0.01]
        bars = VGroup()
        for i, p in enumerate(final_probs):
            bar = Rectangle(
                width=0.4,
                height=0.01,
                fill_opacity=0.8,
                color=BLUE,
                stroke_width=1
            ).move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN)
            bars.add(bar)

        self.play(
            X_group.animate.scale(0.6).to_corner(UL),
            v_group.animate.scale(0.6).next_to(X_group, DOWN, buff=0.5),
            ReplacementTransform(dot_product_label, axes),
            FadeIn(bars)
        )

        growth_anims = []
        for i, bar in enumerate(bars):
            new_height = axes.y_axis.get_unit_size() * final_probs[i]
            growth_anims.append(
                bar.animate.stretch_to_fit_height(new_height).move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN)
            )

        self.play(*growth_anims, run_time=2, rate_func=smooth)
        self.play(Indicate(bars[0], color=GREEN))

        # 5. Formula
        formula = MathTex(
            r"h(\mathbf{v};\mathbf{X})=\frac{\exp(\mathbf{X}\mathbf{v})}{\sum_{j=1}^{N}\exp(\mathbf{X}_{j}^{T}\mathbf{v})}",
            font_size=38
        ).to_edge(UP, buff=0.5).shift(RIGHT * 1.5)

        self.play(Write(formula))
        self.play(formula.animate.set_color(YELLOW), run_time=1)
        
        # Result Label
        pred_text = Text("Prediction: France", color=GREEN, font_size=30).next_to(axes, UP)
        self.play(Write(pred_text))
        
        self.wait(3)
