from manim import *
import numpy as np

class KMeansQuantization(Scene):
    def construct(self):
        # 1. Setup Axes and Country Outline
        # Using Axes instead of NumberPlane as per constraints
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6,
            axis_config={"include_tip": False, "stroke_opacity": 0.3}
        )
        
        # Approximate Australia shape with a simple Polygon
        australia_points = [
            [2, 1, 0], [3, 0.5, 0], [3.2, -1, 0], [2.5, -2, 0],
            [1, -2.2, 0], [-1, -2, 0], [-2.5, -1, 0], [-2.8, 0.5, 0],
            [-2, 1.2, 0], [0, 1.5, 0]
        ]
        country_outline = Polygon(*australia_points, color=BLUE, fill_opacity=0.1)
        country_label = Text("Spatial Data (e.g., Australia)", font_size=24).next_to(country_outline, UP)

        # 2. Populate with data points
        num_points = 80
        point_positions = []
        dots = VGroup()
        for _ in range(num_points):
            # Generate points within a rough bounding box of the polygon
            p = [np.random.uniform(-2.2, 2.8), np.random.uniform(-1.8, 1.2), 0]
            point_positions.append(p)
            dots.add(Dot(point=p, radius=0.05, color=GRAY))

        # 3. Define Centroids and Colors
        cluster_colors = [RED, GREEN, YELLOW, GOLD]
        initial_centroid_pos = [
            [1.5, 0.8, 0], 
            [-1.5, -0.5, 0], 
            [0.5, -1.5, 0], 
            [-1, 1, 0]
        ]
        
        # Custom cross creation since 'Cross' is not allowed
        def create_cross(pos, color):
            size = 0.15
            l1 = Line(pos + np.array([-size, -size, 0]), pos + np.array([size, size, 0]), color=color, stroke_width=5)
            l2 = Line(pos + np.array([-size, size, 0]), pos + np.array([size, -size, 0]), color=color, stroke_width=5)
            return VGroup(l1, l2)

        centroids = VGroup(*[create_cross(initial_centroid_pos[i], cluster_colors[i]) for i in range(len(cluster_colors))])

        # 4. Text and Formulas
        narration = Text(
            "Discretizing space into predictable regions using K-means clustering.",
            font_size=20
        ).to_edge(DOWN, buff=0.5)
        
        formula = MathTex(
            r"\mu_{j}:=\frac{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}x^{(i)}}{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}}",
            font_size=30
        ).to_corner(UL).add_background_rectangle()

        # --- Animation Sequence ---
        
        # Step 1: Initialize Scene
        self.play(Create(axes), Create(country_outline), Write(country_label))
        self.play(FadeIn(dots, lag_ratio=0.02), run_time=2)
        self.wait(0.5)

        # Step 2: Show initial centroids and narration
        self.play(Write(narration))
        self.play(AnimationGroup(*[Create(c) for c in centroids], lag_ratio=0.2))
        self.wait(1)

        # Step 3: Assignment Step (Color points by nearest centroid)
        assignments = []
        target_dots = dots.copy()
        for i, d in enumerate(target_dots):
            d_pos = d.get_center()
            # Find nearest centroid index
            dists = [np.linalg.norm(d_pos - np.array(cp)) for cp in initial_centroid_pos]
            nearest_idx = np.argmin(dists)
            assignments.append(nearest_idx)
            d.set_color(cluster_colors[nearest_idx])

        self.play(Transform(dots, target_dots), Write(formula))
        self.wait(1)

        # Step 4: Update Step (Move centroids to mean of assigned points)
        move_animations = []
        for i in range(len(cluster_colors)):
            # Calculate mean of points assigned to this cluster
            assigned_points = [
                point_positions[j] for j, idx in enumerate(assignments) if idx == i
            ]
            if assigned_points:
                new_mean = np.mean(assigned_points, axis=0)
                move_animations.append(centroids[i].animate.move_to(new_mean))

        self.play(*move_animations, run_time=2)
        self.wait(1)

        # Step 5: Final highlight of discrete 'targets'
        rects = VGroup(*[
            SurroundingRectangle(centroids[i], color=cluster_colors[i], buff=0.2)
            for i in range(len(cluster_colors))
        ])
        
        self.play(Create(rects))
        self.wait(2)
