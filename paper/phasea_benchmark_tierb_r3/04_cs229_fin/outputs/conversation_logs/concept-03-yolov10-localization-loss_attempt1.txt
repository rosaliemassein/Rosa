from manim import *

class YoloLocalizationLoss(Scene):
    def construct(self):
        # 1. Background Setup (Image or Fallback)
        try:
            img = ImageMobject("img-11.jpeg")
            img.scale_to_fit_height(6)
            self.add(img)
        except:
            bg_rect = Rectangle(width=8, height=6, fill_color=GREY_E, fill_opacity=1, stroke_width=0)
            self.add(bg_rect)

        # 2. 7x7 Grid Construction
        grid_dim = 6
        cells = 7
        step = grid_dim / cells
        grid = VGroup()
        
        # Horizontal lines
        for i in range(cells + 1):
            grid.add(Line(
                start=[-grid_dim/2, -grid_dim/2 + i * step, 0],
                end=[grid_dim/2, -grid_dim/2 + i * step, 0],
                stroke_width=1, stroke_opacity=0.4, color=WHITE
            ))
        # Vertical lines
        for i in range(cells + 1):
            grid.add(Line(
                start=[-grid_dim/2 + i * step, -grid_dim/2, 0],
                end=[-grid_dim/2 + i * step, grid_dim/2, 0],
                stroke_width=1, stroke_opacity=0.4, color=WHITE
            ))
        self.add(grid)

        # 3. Highlight a specific cell (Row 3, Col 4)
        target_cell_center = [-grid_dim/2 + 3.5 * step, -grid_dim/2 + 2.5 * step, 0]
        highlight_cell = Square(side_length=step, color=YELLOW, fill_opacity=0.3, stroke_width=2)
        highlight_cell.move_to(target_cell_center)
        self.play(Create(highlight_cell))

        # 4. Predicted Box and Ground Truth Box
        # Ground Truth (Dashed Red)
        gt_box = Rectangle(width=1.2 * step, height=1.8 * step, color=RED)
        gt_box.move_to(target_cell_center + UP * 0.15 + RIGHT * 0.1)
        gt_dashed = DashedVMobject(gt_box, num_dashes=20)
        
        # Predicted Box (Green)
        pred_box = Rectangle(width=1.6 * step, height=1.4 * step, color=GREEN)
        pred_box.move_to(target_cell_center - DOWN * 0.2 - LEFT * 0.1)
        
        self.play(Create(gt_dashed), Create(pred_box))

        # 5. Coordinate and Dimension Arrows
        arrow_xy = Arrow(start=pred_box.get_center(), end=gt_box.get_center(), buff=0, color=WHITE, tip_length=0.1)
        
        # Representing width/height differences
        arrow_w = DoubleArrow(
            start=pred_box.get_left() + UP * 0.2, 
            end=gt_box.get_left() + UP * 0.2, 
            buff=0, color=BLUE, tip_length=0.1
        )
        arrow_h = DoubleArrow(
            start=pred_box.get_top() + RIGHT * 0.2, 
            end=gt_box.get_top() + RIGHT * 0.2, 
            buff=0, color=PURPLE, tip_length=0.1
        )
        
        self.play(GrowFromCenter(arrow_xy))
        self.play(GrowFromCenter(arrow_w), GrowFromCenter(arrow_h))

        # 6. Formula Display
        formula = MathTex(
            r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right)",
            r" + \sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)",
            font_size=24
        ).to_edge(DOWN, buff=0.3).set_background_stroke(color=BLACK, width=4)
        
        self.play(Write(formula))

        # 7. Mock Training Animation
        # Part 1: Align Centers (x, y)

            formula[0].animate.set_color(YELLOW),
            pred_box.animate.move_to(gt_box.get_center()),
            FadeOut(arrow_xy),
            run_time=2
        )
        
        # Part 2: Align Dimensions (w, h)

            formula[0].animate.set_color(WHITE),
            formula[1].animate.set_color(YELLOW),

            FadeOut(arrow_w),
            FadeOut(arrow_h),
            run_time=2
        )
