from manim import *

class GranularSoftmaxClassification(Scene):
    def construct(self):
        # 1. Input Image on the left
        # Using a rectangle placeholder if the image is missing to ensure compilation
        try:
            image = ImageMobject("img-12.jpeg").scale(0.7)
        except:
            image = Rectangle(height=2, width=1.5, color=BLUE).add(Text("Plastic Bottle", font_size=16))
        
        image.to_edge(LEFT, buff=0.8)
        image_label = Text("Input Image", font_size=20).next_to(image, DOWN)

        # 2. CNN Layers (series of vertical rectangles)
        cnn_layers = VGroup(*[
            Rectangle(width=0.3, height=2.0, fill_opacity=0.5, color=GREY_A, fill_color=GREY_A)
            for _ in range(4)
        ]).arrange(RIGHT, buff=0.15)
        cnn_layers.next_to(image, RIGHT, buff=0.8)
        cnn_label = Text("CNN Layers", font_size=18).next_to(cnn_layers, UP)

        # 3. Column vector of 20 raw values (x_i)
        raw_values = VGroup(*[
            MathTex(rf"x_{{{i}}}", font_size=14) for i in range(1, 21)
        ]).arrange(DOWN, buff=0.05)
        
        vector_box = SurroundingRectangle(raw_values, color=WHITE, buff=0.1)
        vector_label = MathTex("x_i").next_to(vector_box, UP, buff=0.1)
        vector_group = VGroup(vector_box, raw_values, vector_label).next_to(cnn_layers, RIGHT, buff=0.8)

        # Arrows for data flow
        arrow1 = Arrow(image.get_right(), cnn_layers.get_left(), buff=0.1)
        arrow2 = Arrow(cnn_layers.get_right(), vector_box.get_left(), buff=0.1)

        # Initial animations
        self.play(FadeIn(image), Write(image_label))
        self.play(Create(cnn_layers), Write(cnn_label), GrowArrow(arrow1))
        self.play(Create(vector_box), Write(raw_values), Write(vector_label), GrowArrow(arrow2))
        self.wait(1)

        # 4. Softmax formula appearing at the top
        softmax_formula = MathTex(
            r"softmax(x_{i})=\frac{e^{x_{i}}}{\sum_{j=1}^{n}e^{x_{j}}}",
            font_size=36
        ).to_edge(UP, buff=0.4)
        
        self.play(Write(softmax_formula))
        self.wait(1)

        # 5. Transform vector into a Bar Chart (Manual implementation for robustness)
        axes = Axes(
            x_range=[0, 20, 5],
            y_range=[0, 1, 0.5],
            x_length=6,
            y_length=3,
            axis_config={"include_tip": False, "font_size": 18}
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)
        
        y_axis_label = Text("Probability", font_size=16).next_to(axes.y_axis, UP, buff=0.1)
        
        bars = VGroup()
        probs = [0.03] * 20
        probs[14] = 0.85 # High probability for the target class
        
        bar_width = (axes.x_length / 20) * 0.7
        for i in range(20):
            bar_height = probs[i] * axes.y_length
            bar = Rectangle(
                width=bar_width,
                height=bar_height,
                fill_color=BLUE,
                fill_opacity=0.7,
                stroke_width=1,
                color=BLUE
            )
            # Position the bottom of the bar on the x-axis
            bar.move_to(axes.c2p(i + 0.5, probs[i] / 2))
            bars.add(bar)

        # Transformation to the probability view
        self.play(
            ReplacementTransform(raw_values, bars),
            Create(axes),
            Write(y_axis_label),
            FadeOut(vector_box),
            FadeOut(vector_label),
            FadeOut(cnn_layers),
            FadeOut(cnn_label),
            FadeOut(arrow1),
            FadeOut(arrow2),
            image.animate.scale(0.6).to_corner(DL),
            image_label.animate.scale(0.6).next_to(image, DOWN, buff=0.1)
        )
        self.wait(1)

        # 6. Highlight the highest bar and label it
        target_bar = bars[14]
        self.play(Indicate(target_bar, color=YELLOW, scale_factor=1.3))
        
        final_label = Text("Plastic Bottle", color=YELLOW, font_size=24)
        final_label.next_to(target_bar, UP, buff=0.2)
        
        self.play(Write(final_label))
        self.wait(3)
