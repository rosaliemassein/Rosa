from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Background Image
        try:
            img = ImageMobject("img-11.jpeg")
            img.scale_to_fit_height(config.frame_height)
            self.add(img)
        except:
            # Fallback placeholder
            img = Rectangle(width=10, height=6, color=GRAY, fill_opacity=0.2)
            self.add(img)

        # 2. 7x7 Grid (Manual construction to avoid NumberPlane restrictions)
        grid_size = 7
        grid_width = 6
        step = grid_width / grid_size
        grid = VGroup()
        for i in range(grid_size + 1):
            # Vertical lines
            grid.add(Line([-3 + i * step, -3, 0], [-3 + i * step, 3, 0], stroke_width=1, stroke_opacity=0.4))
            # Horizontal lines
            grid.add(Line([-3, -3 + i * step, 0], [3, -3 + i * step, 0], stroke_width=1, stroke_opacity=0.4))
        
        self.play(Create(grid))

        # 3. Highlight specific cell
        # Let's pick a cell near the center-right (index 4, 3)
        cell_x, cell_y = 4, 3
        cell_center = [-3 + cell_x * step + step/2, -3 + cell_y * step + step/2, 0]
        highlighted_cell = Square(side_length=step, color=YELLOW, fill_opacity=0.4, stroke_width=2).move_to(cell_center)
        
        self.play(FadeIn(highlighted_cell))
        self.wait(1)

        # 4. Bounding Boxes
        # Predicted box (Red)
        pred_box = Rectangle(width=1.4, height=1.0, color=RED, stroke_width=4).move_to(cell_center + [0.3, 0.2, 0])
        # Ground Truth (Dashed Green)
        gt_box_base = Rectangle(width=1.2, height=1.2, color=GREEN, stroke_width=4).move_to(cell_center)
        gt_box = DashedVMobject(gt_box_base)
        
        self.play(Create(gt_box), Create(pred_box))
        self.wait(1)

        # 5. Show Coordinate and Dimension errors (Arrows)
        # Center difference
        center_arrow = Arrow(gt_box.get_center(), pred_box.get_center(), buff=0, color=WHITE, stroke_width=3)
        # Dimension difference (width example)
        width_arrow = DoubleArrow(
            [gt_box.get_right()[0], gt_box.get_y(), 0],
            [pred_box.get_right()[0], gt_box.get_y(), 0],
            buff=0, color=BLUE, stroke_width=3
        )
        
        self.play(Create(center_arrow), Create(width_arrow))
        self.wait(1)

        # 6. Loss Formula
        # Split formula into parts for better handling
        formula_part1 = r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right)"
        formula_part2 = r" + \sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)"
        
        loss_formula = MathTex(formula_part1, formula_part2, font_size=28)
        loss_formula.to_edge(UP, buff=0.4)
        
        # Add a dark background for formula visibility
        formula_bg = BackgroundRectangle(loss_formula, color=BLACK, fill_opacity=0.8, buff=0.1)
        
        self.play(FadeIn(formula_bg), Write(loss_formula))
        self.wait(2)

        # 7. Mock Training Animation
        # Highlight coordinate terms as we align boxes
        self.play(loss_formula[0].animate.set_color(YELLOW))
        self.play(
            pred_box.animate.move_to(gt_box.get_center()).set_width(1.2).set_height(1.2),
            FadeOut(center_arrow),
            FadeOut(width_arrow),
            run_time=2
        )
        self.play(loss_formula[0].animate.set_color(WHITE), loss_formula[1].animate.set_color(YELLOW))
