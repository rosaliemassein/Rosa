from manim import *

class Concept02RewardDynamics(Scene):
    def construct(self):
        # 1. Setup Formula and Title
        formula = MathTex(r"R_t = \text{Border}_{t} - \text{Border}_{t-1}").to_edge(UP)
        self.play(Write(formula))

        # 2. Create the 3x3 Grid
        grid = VGroup(*[
            Square(fill_opacity=1, stroke_width=2, stroke_color=WHITE) 
            for _ in range(9)
        ]).arrange_in_grid(rows=3, cols=3, buff=0.1).scale(0.8).shift(LEFT * 3)

        # Initialize colors: Center (4) is Red, Neighbors 1,3,5 are Red, 7 is Blue
        for i in [0, 1, 2, 3, 4, 5, 6, 8]:
            grid[i].set_fill(RED)
        grid[7].set_fill(BLUE)

        self.play(Create(grid))
        self.wait(0.5)

        # 3. Focus on center pixel and its 4 neighbors
        center_index = 4
        neighbor_indices = [1, 3, 5, 7] 
        
        self.play(grid[center_index].animate.set_stroke(YELLOW, 6))
        
        neighbor_group = VGroup(*[grid[i] for i in neighbor_indices])
        self.play(Indicate(neighbor_group, color=YELLOW))
        self.wait(0.5)

        # 4. Animate the center pixel flipping from Red to Blue
        # Before flip: 1 border (4-7). After flip: 3 borders (4-1, 4-3, 4-5). Net: +2
        self.play(grid[center_index].animate.set_fill(BLUE), run_time=1)

        # 5. Visualize border changes
        # Green lines for new boundaries (Top, Left, Right)
        new_borders = VGroup(
            Line(grid[center_index].get_corner(UL), grid[center_index].get_corner(UR), color=GREEN, stroke_width=10),
            Line(grid[center_index].get_corner(UL), grid[center_index].get_corner(DL), color=GREEN, stroke_width=10),
            Line(grid[center_index].get_corner(UR), grid[center_index].get_corner(DR), color=GREEN, stroke_width=10)
        )

        # Red X for border lost (Bottom - now both Blue)
        lost_loc = grid[center_index].get_edge_center(DOWN)
        x_mark = VGroup(
            Line(UL, DR), Line(UR, DL)
        ).scale(0.15).set_color(RED).move_to(lost_loc)

        self.play(Create(new_borders))
        self.play(Create(x_mark))
        self.wait(0.5)

        # 6. Floating reward text
        reward_text = Text("+2", color=GREEN, weight="BOLD").scale(1.2)
        reward_text.next_to(grid[center_index], RIGHT, buff=0.5)
        self.play(Write(reward_text), reward_text.animate.shift(UP * 0.5))

        # 7. Manual Bar Chart (Replacing BarChart with Axes + Rectangles for compatibility)
        axes = Axes(
            x_range=[0, 4, 1],
            y_range=[0, 5, 1],
            x_length=4,
            y_length=3,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)
        
        chart_title = Text("Cumulative Reward", font_size=24).next_to(axes, UP)
        
        # Creating bars manually
        def get_bar(x, val, color):
            h = val * (axes.y_length / 5)
            rect = Rectangle(width=0.6, height=h, fill_opacity=0.8, fill_color=color, stroke_width=1)
            rect.move_to(axes.c2p(x, 0), aligned_edge=DOWN)
            return rect

        bar1 = get_bar(1, 1, BLUE)
        bar2 = get_bar(2, 2, BLUE)
        bar3 = get_bar(3, 4, GREEN) # The growth step

        self.play(Create(axes), Write(chart_title))
        self.play(FadeIn(bar1))
        self.play(FadeIn(bar2))
        self.play(FadeIn(bar3))
        
        self.wait(2)
