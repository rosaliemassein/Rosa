from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Header Information
        title = Text("State Space Explosion", color=BLUE).scale(0.8).to_edge(UP)
        voice_txt = Text(
            "Standard Q-learning vs. Exponential States", 
            font_size=20
        ).next_to(title, DOWN)
        
        # 2. Manual Plotting Construction (Avoids 'Axes' which was disallowed)
        origin_coord = [2, -2.5, 0]
        xaxis = Line(origin_coord, [6.5, -2.5, 0], color=WHITE)
        yaxis = Line(origin_coord, [2, 3.5, 0], color=WHITE)
        xlab = Text("N (Grid Size)", font_size=18).next_to(xaxis, DOWN)
        # Use simple multiplication for degrees to avoid potential issues
        ylab = Text("States", font_size=18).rotate(1.5708).next_to(yaxis, LEFT)
        plot_frame = VGroup(xaxis, yaxis, xlab, ylab)

        # 3. Functional Curve Construction
        state_curve = VMobject(color=RED)
        pts = []
        for idx in range(31):
            n_val = 2 + (idx / 30) * 4.5 
            # Log-ish visual scale: x position from N, y position from N^2
            pos_x = origin_coord[0] + (n_val - 2) * 0.8
            pos_y = origin_coord[1] + (n_val * n_val - 4) * 0.15
            pts.append([pos_x, pos_y, 0])
        state_curve.set_points_as_corners(pts)

        # 4. Grid Display Logic
        def build_grid(side):
            group = VGroup()
            for r in range(side):
                for c in range(side):
                    sq = Square(side_length=0.3, stroke_width=1, color=BLUE)
                    sq.move_to([r * 0.35 - 5.5, c * 0.35 - 0.5, 0])
                    group.add(sq)
            return group

        active_grid = build_grid(2)
        grid_5x5 = build_grid(5)
        grid_lbl = Text("2x2 Grid", font_size=24).next_to(active_grid, UP)
        lbl_5x5 = Text("5x5 Grid", font_size=24).next_to(grid_5x5, UP)

        # 5. Q-Table and Scrolling Row Logic
        table_boundary = Rectangle(width=2.2, height=3.5, color=WHITE).next_to(active_grid, DOWN, buff=1.0)
        rows = VGroup(*[
            Line([-1, 0, 0], [1, 0, 0], stroke_width=1, color=BLUE).move_to(
                table_boundary.get_center() + [0, 1.6 - k_idx * 0.25, 0]
            )
            for k_idx in range(14)
        ])
        table_lbl = Text("Q-Table Rows", font_size=20).next_to(table_boundary, UP, buff=0.1)
        
        # spd_tracker (a hidden Mobject) to control animation speed manually
        spd_tracker = Dot(radius=0).move_to([0.5, 0, 0])

        def row_scroller(mobj, dt):
            # Speed is derived from the dummy tracker's x position
            v = spd_tracker.get_x()
            for row_line in mobj:
                row_line.shift(DOWN * v * dt)
                # Wrap rows back to the top of the rectangle
                if row_line.get_y() < table_boundary.get_bottom()[1]:
                    row_line.set_y(table_boundary.get_top()[1])

        rows.add_updater(row_scroller)

        # 6. Formula
        math_formula = MathTex(r"States = 2^{N \times N}").scale(0.8).to_edge(LEFT, buff=0.5).shift(UP * 2.5)

        # 7. Scene Execution
        self.add(title, voice_txt, plot_frame, math_formula, active_grid, grid_lbl, table_boundary, rows, table_lbl, spd_tracker)
        self.wait(1)

        # Main animation sequence
        self.play(
            Create(state_curve),
            Transform(active_grid, grid_5x5),
            Transform(grid_lbl, lbl_5x5),
            spd_tracker.animate.set_x(12),
            run_time=6,
            rate_func=lambda t: t
        )

        # Explosion and final state representation
        solid_blur = Rectangle(width=2.2, height=3.5, fill_color=BLUE, fill_opacity=0.9).move_to(table_boundary)
        boom_text = Text("EXPLOSION", color=RED).scale(0.8).move_to(table_boundary)
        
        self.play(
            FadeIn(solid_blur),
            Write(boom_text),
            spd_tracker.animate.set_x(60),
            run_time=2
        )
        self.wait(2)
