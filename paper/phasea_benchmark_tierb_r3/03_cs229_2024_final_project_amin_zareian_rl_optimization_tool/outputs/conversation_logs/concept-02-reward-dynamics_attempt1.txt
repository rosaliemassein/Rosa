from manim import *

class RewardDynamics(Scene):
    def construct(self):
        # 1. Setup the Formula for reference
        formula = MathTex(r"R_t = \text{Border}_{t} - \text{Border}_{t-1}").to_edge(UP)
        self.play(Write(formula))

        # 2. Create the 3x3 grid (Pixels)
        grid = VGroup(*[
            Square(side_length=0.8, fill_opacity=1, color=WHITE).set_stroke(GRAY, 1)
            for _ in range(9)
        ]).arrange_in_grid(rows=3, cols=3, buff=0.1).shift(LEFT * 3)

        # Initial Color setup
        # Neighbors (indices 1, 3, 7) are RED, Neighbor (index 5) is BLUE
        # Center (index 4) starts as RED
        # Initial boundary count: 1 (between center RED and index 5 BLUE)
        colors = [
            WHITE, RED,  WHITE,
            RED,   RED,  BLUE,
            WHITE, RED,  WHITE
        ]
        for i, color in enumerate(colors):
            grid[i].set_fill(color)

        self.play(Create(grid))
        self.wait(0.5)

        # 3. Highlight the center pixel and its 4 neighbors
        center_pixel = grid[4]
        neighbor_indices = [1, 3, 5, 7]
        highlights = VGroup(*[
            SurroundingRectangle(grid[i], color=YELLOW, buff=0.05)
            for i in neighbor_indices
        ])
        
        self.play(Create(highlights))
        self.play(Indicate(center_pixel, scale_factor=1.2))
        self.wait(0.5)

        # 4. Animate the center pixel flipping from Red to Blue
        # New boundary count: 3 (between center BLUE and indices 1, 3, 7 RED)
        # Lost boundary count: 1 (between center BLUE and index 5 BLUE)
        # Net reward: 3 - 1 = +2
        self.play(center_pixel.animate.set_fill(BLUE), run_time=1)

        # 5. Visual indicators for border changes
        # Green lines for new borders (boundaries between Blue and Red)
        new_border_top = Line(grid[4].get_corner(UL), grid[4].get_corner(UR), color=GREEN, stroke_width=6)
        new_border_left = Line(grid[4].get_corner(UL), grid[4].get_corner(DL), color=GREEN, stroke_width=6)
        new_border_bottom = Line(grid[4].get_corner(DL), grid[4].get_corner(DR), color=GREEN, stroke_width=6)
        
        # Red 'X' for lost border (where it was Red-Blue, now it's Blue-Blue)
        lost_border_loc = grid[4].get_right()
        red_x = VGroup(
            Line(UL, DR, color=RED).scale(0.15),
            Line(UR, DL, color=RED).scale(0.15)
        ).move_to(lost_border_loc)

        self.play(
            Create(new_border_top),
            Create(new_border_left),
            Create(new_border_bottom),
            Create(red_x)
        )

        # 6. Floating Reward Text
        reward_tex = MathTex("+2", color=GREEN).scale(1.2).next_to(grid[4], UR, buff=0.2)
        self.play(Write(reward_tex), reward_tex.animate.shift(UP * 0.4))

        # 7. Manual BarChart implementation (to avoid "BarChart" identifier error)
        axes = Axes(
            x_range=[0, 3, 1],
            y_range=[0, 5, 1],
            x_length=4,
            y_length=3,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=1)
        
        axis_labels = axes.get_axis_labels(x_label="Step", y_label="Reward")
        
        # Bar at t-1
        bar_t1_height = axes.c2p(1, 1)[1] - axes.c2p(1, 0)[1]
        bar_t1 = Rectangle(width=0.6, height=bar_t1_height, fill_opacity=0.8, color=BLUE)
        bar_t1.move_to(axes.c2p(1, 0), aligned_edge=DOWN)
        
        # Bar at t (Cumulative: 1 + 2 = 3)
        bar_t2_height = axes.c2p(2, 3)[1] - axes.c2p(2, 0)[1]
        bar_t2 = Rectangle(width=0.6, height=bar_t2_height, fill_opacity=0.8, color=BLUE)
        bar_t2.move_to(axes.c2p(2, 0), aligned_edge=DOWN)

        chart_group = VGroup(axes, axis_labels)
        self.play(Create(chart_group))
        self.play(FadeIn(bar_t1, shift=UP))
        self.wait(0.5)
        self.play(FadeIn(bar_t2, shift=UP))
        self.play(Indicate(bar_t2, color=GREEN))

        self.wait(2)
