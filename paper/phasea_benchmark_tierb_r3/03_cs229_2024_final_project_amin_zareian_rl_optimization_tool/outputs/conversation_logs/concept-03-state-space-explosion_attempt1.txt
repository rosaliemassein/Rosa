from manim import *

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Setup - Formula and Manual Coordinate System
        formula_display = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP)
        self.add(formula_display)

        # Manual coordinate system using Line objects
        # We avoid the 'Axes' class as per the environment constraints.
        chart_origin = 4 * LEFT + 2 * DOWN
        horiz_axis = Line(chart_origin, chart_origin + 6 * RIGHT, color=WHITE)
        vert_axis = Line(chart_origin, chart_origin + 5 * UP, color=WHITE)
        
        horiz_label = Tex("Matrix Size $N$").scale(0.8).next_to(horiz_axis, DOWN)
        vert_label = Tex("Possible States (log scale)").scale(0.8).rotate(90 * DEGREES).next_to(vert_axis, LEFT)
        
        self.add(horiz_axis, vert_axis, horiz_label, vert_label)

        # 2. Q-Table Graphic Construction
        table_boundary = Rectangle(width=2.5, height=4.5, color=WHITE).shift(3 * RIGHT + 0.5 * DOWN)
        table_header_tex = Tex("Q-Table").next_to(table_boundary, UP)
        
        # Start with 16 discrete rows to represent a 2x2 grid (2^4 = 16 states)
        table_rows = VGroup()
        for row_num in range(16):
            row_line = Line(LEFT, RIGHT).scale(1.1)
            table_rows.add(row_line)
        table_rows.arrange(DOWN, buff=0.1).move_to(table_boundary)
        
        self.play(Create(table_boundary), Write(table_header_tex), FadeIn(table_rows))
        self.wait(0.5)

        # 3. Exponential Growth Curve (Logarithmic Representation)
        # On a log2 scale, the function 2^(N^2) appears as N^2.
        state_graph = VMobject(color=BLUE)
        points_array = []
        # Calculate 40 points for a smooth curve from N=2 to N=6
        for point_idx in range(41):
            n_val_calc = 2 + (point_idx * 0.1)
            # Growth is N^2 relationship
            growth_val = n_val_calc * n_val_calc
            # Position mapping: N=2 -> 0 shift, N=6 -> 4 shift. growth_val=4 -> 0.4 shift, 36 -> 3.6 shift.
            target_pos = chart_origin + (n_val_calc - 2) * RIGHT + (growth_val * 0.1) * UP
            points_array.append(target_pos)
        
        state_graph.set_points_as_corners(points_array)

        # 4. Table Explosion Visualization Setup
        # Create dense lines for high-speed scrolling effect
        dense_rows = VGroup()
        for dense_idx in range(80):
            single_row = Line(table_boundary.get_left() + 0.1*RIGHT, table_boundary.get_right() - 0.1*RIGHT, stroke_width=0.5)
            dense_rows.add(single_row)
        dense_rows.arrange(DOWN, buff=0.04).move_to(table_boundary)
        
        # Blur overlay to simulate "solid blur of color"
        blur_rect = Rectangle(width=2.5, height=4.5, fill_color=BLUE, fill_opacity=0, stroke_width=0).move_to(table_boundary)

        # Scrolling logic for the dense rows
        def scroll_lines_action(mobj_to_scroll):
            mobj_to_scroll.shift(UP * 0.25)
            for line_unit in mobj_to_scroll:
                if line_unit.get_center()[1] > table_boundary.get_top()[1]:
                    line_unit.move_to(table_boundary.get_bottom() + UP * 0.04)

        # 5. The Animation Sequence
        narration_tex = Tex("States grow exponentially...", font_size=32).to_edge(DOWN)
        self.play(Write(narration_tex))
        self.play(Create(state_graph), run_time=2)
        
        # Transition Q-table into a blurred, high-speed scrolling state
        self.play(
            FadeOut(table_rows),
            FadeIn(dense_rows),
            blur_rect.animate.set_fill(opacity=0.85),
            run_time=1
        )
        
        dense_rows.add_updater(scroll_lines_action)
        
        # Final explosion: The graph shoots off the top of the screen and color changes to red
        warning_tex = Tex("Training time skyrockets!", color=RED).scale(1.2).move_to(ORIGIN)
        
        self.play(
            state_graph.animate.scale(5, about_point=chart_origin).set_color(RED),
            blur_rect.animate.set_fill(color=RED, opacity=0.95),
            Transform(narration_tex, warning_tex),
            run_time=3
        )
        
        self.wait(2)
