from manim import *

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Setup Axes
        # Matrix size on x, number of states (represented as log values) on y
        ax = Axes(
            x_range=[2, 7, 1],
            y_range=[0, 12, 2],
            x_length=7,
            y_length=5,
            axis_config={"color": BLUE, "include_tip": True}
        ).to_edge(LEFT, buff=0.5)

        # Labels for axes (using Text to avoid potential LaTeX issues in some environments)
        x_label = Text("Matrix Size (N)", font_size=20).next_to(ax.x_axis, DOWN)
        y_label = Text("States (Log Scale)", font_size=20).rotate(90 * DEGREES).next_to(ax.y_axis, LEFT)

        # 2. Function Plotting
        # We plot log10(2^(t*t)) which is t*t * log10(2)
        # log10(2) is approximately 0.301
        def log_growth(t):
            return t * t * 0.301

        curve = ax.plot(log_growth, x_range=[2, 5.5], color=GREEN)

        # 3. Q-Table Representation
        # A rectangle to represent the boundaries of the Q-Table
        table_rect = Rectangle(height=4, width=2.2, color=WHITE).to_edge(RIGHT, buff=1)
        table_title = Text("Q-Table", font_size=20).next_to(table_rect, UP)

        # Initial table rows for small N (e.g., N=2)
        rows = VGroup(*[
            Line(LEFT, RIGHT, stroke_width=1).set_width(2.0)
            for _ in range(16)
        ]).arrange(DOWN, buff=0.1).move_to(table_rect.get_top() + DOWN * 0.5)

        # Reference formula
        formula = MathTex(r"States = 2^{N \times N}", font_size=36).to_edge(UP)

        # 4. Animation Sequence
        self.add(ax, x_label, y_label)
        self.play(Create(curve), run_time=2)
        self.play(Create(table_rect), Write(table_title))
        self.play(Create(rows))
        self.wait(1)

        # "Explosion" effect
        # Create a semi-transparent blue block to represent millions of overlapping rows
        blur_effect = Rectangle(
            height=3.8, width=2.0,
            fill_color=BLUE, fill_opacity=0.8,
            stroke_width=0
        ).move_to(table_rect)

        # Text to show the resulting explosion
        explosion_stats = Text("33 Million+", font_size=24, color=YELLOW).move_to(table_rect)

        self.play(
            Write(formula),
            # Rows scroll up and fade out to represent rapid multiplication
            rows.animate.shift(UP * 4).set_opacity(0),
            FadeIn(blur_effect),
            # The graph curve shoots off the top of the screen
            Create(ax.plot(log_growth, x_range=[5.5, 6.8], color=GREEN)),
            run_time=3,
            rate_func=smooth
        )

        self.play(Write(explosion_stats))
        self.wait(2)
