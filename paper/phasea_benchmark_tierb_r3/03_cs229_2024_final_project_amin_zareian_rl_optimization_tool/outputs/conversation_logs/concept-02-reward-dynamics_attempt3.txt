from manim import *

class RewardDynamics(Scene):
    def construct(self):
        # 1. Setup the 3x3 Grid
        # Initial state: Center is RED, neighbors are RED.
        # Flipping RED to BLUE surrounded by RED creates 4 new boundaries.
        colors = [
            BLUE, RED,  BLUE,
            RED,  RED,  RED,
            BLUE, RED,  BLUE
        ]
        
        grid = VGroup(*[
            Square(side_length=1).set_fill(colors[i], opacity=1).set_stroke(WHITE, 1)
            for i in range(9)
        ]).arrange_in_grid(3, 3, buff=0).shift(LEFT * 3)

        grid_title = Text("Environment State", font_size=24).next_to(grid, UP)
        
        self.add(grid, grid_title)
        self.wait(0.5)

        # 2. Highlight the center pixel and its 4 neighbors
        center_idx = 4
        neighbor_indices = [1, 3, 5, 7] # North, West, East, South
        
        center_pixel = grid[center_idx]
        
        self.play(Indicate(center_pixel, color=YELLOW, scale_factor=1.1))
        self.play(
            *[Indicate(grid[i], color=YELLOW, scale_factor=1.1) for i in neighbor_indices],
            run_time=1.5
        )

        # 3. Animate the center pixel flipping from Red to Blue
        # This increases the boundary length because neighbors are RED and center becomes BLUE
        self.play(center_pixel.animate.set_fill(BLUE), run_time=1)

        # 4. Show new borders and the reward
        # Create lines on the edges of the center square
        boundaries = VGroup()
        edge_positions = [UP*0.5, LEFT*0.5, RIGHT*0.5, DOWN*0.5]
        edge_rotations = [0, PI/2, PI/2, 0]
        
        for pos, rot in zip(edge_positions, edge_rotations):
            line = Line(
                LEFT*0.5, RIGHT*0.5, 
                color=GREEN, 
                stroke_width=6
            ).move_to(center_pixel.get_center() + pos).rotate(rot)
            boundaries.add(line)

        reward_val = Text("+4", color=GREEN).next_to(center_pixel, UR, buff=0.1)
        
        self.play(Create(boundaries), Write(reward_val))
        self.wait(0.5)

        # 5. Formula and Manual Bar Chart (Replacing BarChart class for compatibility)
        formula = MathTex("R_t = Border_{t} - Border_{t-1}").to_edge(UP).shift(RIGHT * 3)
        self.play(Write(formula))

        # Build a chart using Axes and Rectangles
        axes = Axes(
            x_range=[0, 4, 1],
            y_range=[0, 10, 2],
            x_length=4,
            y_length=3,
            axis_config={"include_tip": False}
        ).next_to(formula, DOWN, buff=1)
        
        chart_label = Text("Cumulative Reward", font_size=20).next_to(axes, UP)
        
        # Create initial bars
        bar1 = Rectangle(width=0.5, height=axes.c2p(0, 1)[1] - axes.c2p(0, 0)[1], fill_opacity=0.8, fill_color=BLUE, stroke_width=1).move_to(axes.c2p(1, 0.5))
        bar2 = Rectangle(width=0.5, height=axes.c2p(0, 3)[1] - axes.c2p(0, 0)[1], fill_opacity=0.8, fill_color=BLUE, stroke_width=1).move_to(axes.c2p(2, 1.5))
        bar3 = Rectangle(width=0.5, height=axes.c2p(0, 3)[1] - axes.c2p(0, 0)[1], fill_opacity=0.8, fill_color=BLUE, stroke_width=1).move_to(axes.c2p(3, 1.5))

        self.play(Create(axes), Write(chart_label))
        self.play(Create(bar1), Create(bar2), Create(bar3))

        # 6. Animate cumulative reward growing (updating the 3rd bar)
        new_height = axes.c2p(0, 7)[1] - axes.c2p(0, 0)[1]
        self.play(
            bar3.animate.stretch_to_fit_height(new_height).move_to(axes.c2p(3, 3.5)),
            reward_val.animate.scale(1.5).set_opacity(0),
            run_time=1.5
        )

        self.wait(2)
