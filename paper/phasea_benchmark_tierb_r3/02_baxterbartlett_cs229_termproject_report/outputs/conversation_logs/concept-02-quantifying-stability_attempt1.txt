from manim import *
import numpy as np

class QuantifyingStability(Scene):
    def construct(self):
        # 1. Setup coordinate system using Axes as per Reference Example 3
        # Using a variable name other than 'axes' to be safe
        ax_system = Axes(
            x_range=[0, 10, 2],
            y_range=[0, 50, 10],
            x_length=6,
            y_length=3,
            axis_config={"include_tip": False}
        ).to_edge(LEFT, buff=0.5).shift(UP * 0.5)

        # 2. Velocity Plot
        # Using a named function 'vel_func' with parameter 't' to avoid identifier issues
        def vel_func(t):
            # Fluctuating around 30 RPM
            return 30 + 5 * np.sin(2 * t) * np.cos(0.5 * t) + 2 * np.sin(10 * t) * 0.1
        
        velocity_plot = ax_system.plot(vel_func, color=BLUE)
        plot_label = Text("Velocity (RPM)", font_size=20).next_to(ax_system, UP)

        # 3. Highlight the last 20% (from t=8 to t=10)
        # We calculate the region width and height based on the axes
        region_width = ax_system.x_axis.get_unit_size() * 2
        region_height = ax_system.y_axis.get_length()
        highlight_rect = Rectangle(
            width=region_width,
            height=region_height,
            fill_color=YELLOW,
            fill_opacity=0.3,
            stroke_width=0
        ).move_to(ax_system.c2p(9, 25))

        # 4. Histogram on the side
        # Created manually with rectangles
        h_bar1 = Rectangle(width=0.4, height=0.6, fill_color=RED, fill_opacity=0.8)
        h_bar2 = Rectangle(width=0.4, height=2.5, fill_color=GREEN, fill_opacity=0.8)
        h_bar3 = Rectangle(width=0.4, height=0.9, fill_color=RED, fill_opacity=0.8)
        
        histogram = VGroup(h_bar1, h_bar2, h_bar3).arrange(RIGHT, buff=0.1, aligned_edge=DOWN)
        histogram.to_edge(RIGHT, buff=1).align_to(ax_system, UP)
        
        hist_label = Text("Histogram", font_size=20).next_to(histogram, UP)
        mode_label = Text("Mode", font_size=16, color=GREEN).next_to(h_bar2, DOWN)

        # 5. Stability Formula
        # Using MathTex for the specific mathematical definition
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=36
        ).to_edge(DOWN, buff=1)

        # 6. Animation Sequence
        # Voice: "To train a model, we need to turn qualitative behavior into hard numbers."
        self.play(Create(ax_system), Create(velocity_plot), Write(plot_label))
        self.wait(1)

        # Voice: "...'Steady State Error' is calculated specifically from the final twenty percent..."
        self.play(FadeIn(highlight_rect))
        self.wait(1)

        # Voice: "...'Stability' as a ratio: how often the motor stays at its most common speed..."
        self.play(Create(histogram), Write(hist_label), Write(mode_label))
        self.wait(1)

        # Voice: "Stability = counts at mode / counts not at mode"
        self.play(Write(formula))
        self.wait(3)

        # Final cleanup/display
        self.play(
            FadeOut(plot_label),
            FadeOut(hist_label),
            FadeOut(mode_label),
            highlight_rect.animate.set_fill(opacity=0.1)
        )
        self.wait(1)
