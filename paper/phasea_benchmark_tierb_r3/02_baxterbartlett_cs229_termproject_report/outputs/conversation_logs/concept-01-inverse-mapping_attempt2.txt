import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):

        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}").scale(0.7).to_edge(UP, buff=0.3)
        self.add(formula)

        # 2. Split Screen Diagram
        # Left box: Desired Behavior
        box_left_rect = Rectangle(width=4.0, height=2.5, color=WHITE)
        box_left_title = Text("Desired Behavior", color=BLUE).scale(0.45).move_to(box_left_rect.get_top() + DOWN * 0.3)
        box_left_content = VGroup(
            Text("Rise Time").scale(0.35),
            Text("Steady State Error").scale(0.35),
            Text("Stability").scale(0.35)
        ).arrange(DOWN, buff=0.2).next_to(box_left_title, DOWN, buff=0.2)
        box_left = VGroup(box_left_rect, box_left_title, box_left_content).shift(LEFT * 3.5 + UP * 0.5)
        label_in = Text("Input", color=GREY).scale(0.3).next_to(box_left, UP, buff=0.1)

        # Right box: PID Constants
        box_right_rect = Rectangle(width=4.0, height=2.5, color=WHITE)
        box_right_title = Text("PID Constants", color=RED).scale(0.45).move_to(box_right_rect.get_top() + DOWN * 0.3)
        box_right_content = VGroup(
            MathTex("K_p").scale(0.6),
            MathTex("K_i").scale(0.6),
            MathTex("K_d").scale(0.6)
        ).arrange(DOWN, buff=0.15).next_to(box_right_title, DOWN, buff=0.2)
        box_right = VGroup(box_right_rect, box_right_title, box_right_content).shift(RIGHT * 3.5 + UP * 0.5)
        label_out = Text("Output", color=GREY).scale(0.3).next_to(box_right, UP, buff=0.1)

        # Connect with Arrow
        main_arrow = Arrow(box_left.get_right(), box_right.get_left(), buff=0.2)
        
        diagram_group = VGroup(box_left, box_right, label_in, label_out, main_arrow)
        
        # Display diagram
        self.play(FadeIn(diagram_group))
        self.wait(1)

        # "Flips the script" - rotate diagram 180 degrees
        # Using animate.rotate to avoid the Rotate class which might be restricted
        self.play(diagram_group.animate.rotate(PI), run_time=1.5)
        self.wait(1)

        # 3. Motor Velocity Plot (Bottom)
        # Constructing axes manually to avoid 'Axes' and 'DashedLine'
        origin_point = DOWN * 2.5 + LEFT * 3.5
        x_axis = Line(origin_point, origin_point + RIGHT * 7)
        y_axis = Line(origin_point, origin_point + UP * 2)
        
        label_x = Text("Time").scale(0.35).next_to(x_axis, DOWN, buff=0.1)
        label_y = Text("RPM").scale(0.35).next_to(y_axis, LEFT, buff=0.1)
        
        # Setpoint line (manual dash pattern)
        setpoint_height = 1.4
        set_start = origin_point + UP * setpoint_height
        set_end = set_start + RIGHT * 7
        setpoint_line = VGroup()
        for i in range(25):
            alpha_1 = i / 25
            alpha_2 = (i + 0.5) / 25
            setpoint_line.add(Line(
                set_start + (set_end - set_start) * alpha_1,
                set_start + (set_end - set_start) * alpha_2,
                color=YELLOW, stroke_width=1
            ))

        # Motor performance curve
        motor_curve = VMobject(color=BLUE)
        curve_points = []
        # Using a loop to avoid lambda functions and the 'x' identifier

            time_prog = step_idx / 100.0
            time_val = time_prog * 7.0
            # Formula for a damped oscillation approaching 1.25 (below setpoint 1.4)
            val_y = 1.2 * (1.0 - np.exp(-0.6 * time_val) * np.cos(2.0 * time_val))
            curve_points.append(origin_point + RIGHT * time_val + UP * val_y)
        
        motor_curve.set_points_as_corners(curve_points)

        # Animate the plot
        self.play(Create(x_axis), Create(y_axis), Write(label_x), Write(label_y))
        self.play(FadeIn(setpoint_line))
        self.play(Create(motor_curve), run_time=2)
        self.play(FadeIn(brace_rise), Write(text_rise))
        self.play(FadeIn(brace_sse), Write(text_sse))

        self.wait(2)
