from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Manual Axes for Neural Network (Left)
        nn_origin = 3.5 * LEFT + 0.5 * DOWN
        nn_x_axis = Line(nn_origin, nn_origin + 3 * RIGHT, color=WHITE)
        nn_y_axis = Line(nn_origin, nn_origin + 3 * UP, color=WHITE)
        nn_axes = VGroup(nn_x_axis, nn_y_axis)
        nn_label = Text("Neural Network", font_size=24).next_to(nn_y_axis, UP)
        nn_var_label = Text("High Variance", color=RED, font_size=20).next_to(nn_label, DOWN, buff=0.1)

        # 2. Setup Manual Axes for LWR (Right)
        lwr_origin = 0.5 * RIGHT + 0.5 * DOWN
        lwr_x_axis = Line(lwr_origin, lwr_origin + 3 * RIGHT, color=WHITE)
        lwr_y_axis = Line(lwr_origin, lwr_origin + 3 * UP, color=WHITE)
        lwr_axes = VGroup(lwr_x_axis, lwr_y_axis)
        lwr_label = Text("LWR", font_size=24).next_to(lwr_y_axis, UP)

        # 3. Generate Data Points
        np.random.seed(42)
        x_data = np.linspace(0.3, 2.7, 10)
        # Linear trend with noise
        y_data = 0.8 * x_data + np.random.normal(0, 0.3, len(x_data))
        
        nn_dots = VGroup(*[Dot(nn_origin + x * RIGHT + y * UP, radius=0.05) for x, y in zip(x_data, y_data)])
        lwr_dots = VGroup(*[Dot(lwr_origin + x * RIGHT + y * UP, radius=0.05) for x, y in zip(x_data, y_data)])

        # 4. Generate Curves
        # NN: Erratic curve hitting points (overfit)
        nn_curve_points = []
        for x_val in np.linspace(0.3, 2.7, 50):
            # Base linear + high frequency oscillation to mimic overfitting noise
            y_val = 0.8 * x_val + 0.6 * np.sin(12 * x_val)
            nn_curve_points.append(nn_origin + x_val * RIGHT + y_val * UP)
        nn_curve = VMobject(color=RED).set_points_as_corners(nn_curve_points)

        # LWR: Smooth local linear path
        lwr_curve_points = [
            lwr_origin + 0.3 * RIGHT + 0.4 * UP,
            lwr_origin + 1.5 * RIGHT + 1.2 * UP,
            lwr_origin + 2.7 * RIGHT + 2.1 * UP
        ]
        lwr_curve = VMobject(color=BLUE).set_points_as_corners(lwr_curve_points)

        # 5. MSE Bar Chart (Manual Construction)
        bar_nn = Rectangle(height=1.21 * 1.2, width=0.6, fill_opacity=0.8, fill_color=RED, stroke_color=WHITE)
        bar_lwr = Rectangle(height=0.85 * 1.2, width=0.6, fill_opacity=0.8, fill_color=BLUE, stroke_color=WHITE)
        
        bar_nn.to_edge(DOWN, buff=0.5).shift(LEFT * 1.0)
        bar_lwr.align_to(bar_nn, BOTTOM).shift(RIGHT * 1.0)
        
        val_nn = Text("1.21", font_size=16).next_to(bar_nn, UP, buff=0.1)
        val_lwr = Text("0.85", font_size=16).next_to(bar_lwr, UP, buff=0.1)
        label_nn_bar = Text("NN", font_size=16).next_to(bar_nn, DOWN, buff=0.1)
        label_lwr_bar = Text("LWR", font_size=16).next_to(bar_lwr, DOWN, buff=0.1)
        mse_chart = VGroup(bar_nn, bar_lwr, val_nn, val_lwr, label_nn_bar, label_lwr_bar)

        # 6. Formula
        mse_formula = MathTex(r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}", font_size=28)
        mse_formula.to_corner(DR, buff=0.5).shift(UP * 0.5)

        # Animation Sequence
        self.play(Create(nn_axes), Create(lwr_axes), Write(nn_label), Write(lwr_label))
        self.play(FadeIn(nn_dots), FadeIn(lwr_dots))
        self.wait(1)
        
        self.play(Create(nn_curve), Write(nn_var_label), run_time=2)
        self.play(Create(lwr_curve), run_time=1.5)
        self.wait(1)
        
        self.play(Write(mse_formula))
        self.play(FadeIn(mse_chart, shift=UP))
        self.wait(3)
