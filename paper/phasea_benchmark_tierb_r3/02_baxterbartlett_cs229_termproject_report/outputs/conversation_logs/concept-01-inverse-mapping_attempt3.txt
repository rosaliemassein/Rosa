from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # --- TOP DIAGRAM: INPUT/OUTPUT BOXES ---
        # Define colors and standard sizes
        box_width, box_height = 4.0, 1.2
        
        # Left Side: Desired Behavior
        input_rect = Rectangle(width=box_width, height=box_height, color=BLUE)
        input_text = Text("Desired Behavior", font_size=24)
        input_sub = Text("(Rise Time, SS Error, Stability)", font_size=16, color=BLUE).next_to(input_rect, DOWN, buff=0.1)
        input_box = VGroup(input_rect, input_text, input_sub).shift(LEFT * 3.5 + UP * 2)

        # Right Side: PID Constants
        output_rect = Rectangle(width=box_width, height=box_height, color=GREEN)
        output_text = Text("PID Constants", font_size=24)
        output_box = VGroup(output_rect, output_text).shift(RIGHT * 3.5 + UP * 2)

        # Arrow
        connection_arrow = Arrow(input_box.get_right(), output_box.get_left(), buff=0.2, color=WHITE)
        
        # Combine into a single diagram group
        diagram = VGroup(input_box, connection_arrow, output_box)

        # --- BOTTOM DIAGRAM: MOTOR PLOT ---
        # Create axes manually using Line to avoid 'Axes' restriction
        origin = DOWN * 2.5 + LEFT * 5
        x_axis = Line(origin, origin + RIGHT * 10, color=WHITE)
        y_axis = Line(origin, origin + UP * 3, color=WHITE)
        x_label = Text("Time", font_size=20).next_to(x_axis, DOWN, buff=0.2)
        y_label = Text("RPM", font_size=20).next_to(y_axis, LEFT, buff=0.2).rotate(90 * DEGREES)
        
        # Setpoint line
        setpoint_y = origin[1] + 2.0
        setpoint_line = Line(
            [origin[0], setpoint_y, 0], 
            [origin[0] + 9, setpoint_y, 0], 
            color=YELLOW
        )
        setpoint_label = Text("Setpoint", font_size=16, color=YELLOW).next_to(setpoint_line, RIGHT, buff=0.1)

        # Motor velocity curve (Underdamped calculation)
        curve_points = []
        steps = 100
        x_range = 9.0
        for i in range(steps + 1):
            t_val = (i / steps) * 6.0 # Logic time 0 to 6
            # Simplified underdamped response: 1 - exp(-0.8t) * cos(4t)
            # Scaled to fit screen coordinates
            y_val_logic = (1 - np.exp(-0.8 * t_val) * (np.cos(4 * t_val) + 0.2 * np.sin(4 * t_val))) * 0.95
            
            px = origin[0] + (t_val / 6.0) * x_range
            py = origin[1] + y_val_logic * 2.0
            curve_points.append([px, py, 0])
            
        motor_curve = VMobject(color=WHITE)
        motor_curve.set_points_as_corners(curve_points)

        # Rise Time annotation (Manual)
        rt_start = origin[0]
        rt_end = origin[0] + (1.1 / 6.0) * x_range
        rt_line = Line([rt_start, origin[1] - 0.2, 0], [rt_end, origin[1] - 0.2, 0], color=BLUE)
        rt_tick1 = Line([rt_start, origin[1] - 0.3, 0], [rt_start, origin[1] - 0.1, 0], color=BLUE)
        rt_tick2 = Line([rt_end, origin[1] - 0.3, 0], [rt_end, origin[1] - 0.1, 0], color=BLUE)
        rt_label = Text("Rise Time", font_size=18, color=BLUE).next_to(rt_line, DOWN, buff=0.1)

        # Steady State Error annotation (Manual)
        ss_x = origin[0] + 8.5
        ss_top = setpoint_y
        ss_bottom = origin[1] + 0.95 * 2.0
        ss_err_line = Line([ss_x, ss_top, 0], [ss_x, ss_bottom, 0], color=RED)
        ss_err_label = Text("Steady State Error", font_size=16, color=RED).next_to(ss_err_line, RIGHT, buff=0.1)

        # --- ANIMATION ---
        # Part 1: Initial Relationship
        self.play(Create(input_box), Create(output_box))
        self.play(GrowFromPoint(connection_arrow, connection_arrow.get_start()))
        self.wait(1)

        # Part 2: "Flips the Script" (Rotate Diagram)
        # Note: We rotate the group, then fix the text orientation
        self.play(diagram.animate.rotate(PI))
        self.play(
            input_text.animate.rotate(PI),
            input_sub.animate.rotate(PI),
            output_text.animate.rotate(PI)
        )
        self.wait(1)

        # Part 3: Motor Plot
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint_line), Write(setpoint_label))
        self.play(Create(motor_curve), run_time=2.5)
        
        # Part 4: Annotations
        self.play(Create(rt_line), Create(rt_tick1), Create(rt_tick2), Write(rt_label))
        self.play(Create(ss_err_line), Write(ss_err_label))
        
        self.wait(2)
