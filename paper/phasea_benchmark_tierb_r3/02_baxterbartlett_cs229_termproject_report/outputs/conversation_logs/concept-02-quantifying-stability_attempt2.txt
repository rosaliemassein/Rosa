from manim import *
import numpy as np

class StabilityConcept(Scene):
    def construct(self):
        # Define a helper function for coordinate mapping to avoid using Axes
        # Time range: 0 to 40 -> mapped to x: -5 to 1 (width 6)
        # RPM range: 0 to 60 -> mapped to y: -2 to 2 (height 4)
        def get_point(time_val, rpm_val):
            x_coord = -5 + (time_val / 40.0) * 6
            y_coord = -2 + (rpm_val / 60.0) * 4
            return np.array([x_coord, y_coord, 0])

        # 1. Create the velocity plot manually using Lines
        horizontal_axis = Line(get_point(0, 0), get_point(40, 0), color=WHITE)
        vertical_axis = Line(get_point(0, 0), get_point(0, 60), color=WHITE)
        
        # Add simple tick-like markings or labels
        h_label = Text("Time", font_size=20).next_to(horizontal_axis, DOWN)
        v_label = Text("RPM", font_size=20).rotate(90 * DEGREES).next_to(vertical_axis, LEFT)
        
        self.add(horizontal_axis, vertical_axis, h_label, v_label)

        # 2. Fluctuating motor velocity curve around 30 RPM
        velocity_curve = VMobject(color=YELLOW)
        curve_points = []
        # Generate 100 points for a smooth fluctuating curve
        for step in range(101):
            t_val = step * 0.4  # Ranges from 0 to 40
            # Function: 30 + some sine waves to simulate fluctuation
            r_val = 30 + 6 * np.sin(0.6 * t_val) + 3 * np.cos(1.2 * t_val)
            curve_points.append(get_point(t_val, r_val))
        
        velocity_curve.set_points_as_corners(curve_points)
        
        self.play(Create(velocity_curve), run_time=2)

        # 3. Highlight the last 20% of the x-axis (from time 32 to 40)
        # Background highlight area
        p_start_highlight = get_point(32, 0)
        p_end_highlight = get_point(40, 60)
        
        highlight_box = Rectangle(
            width = p_end_highlight[0] - p_start_highlight[0],
            height = p_end_highlight[1] - p_start_highlight[1],
            fill_color = BLUE,
            fill_opacity = 0.3,
            stroke_width = 0
        ).move_to(get_point(36, 30)) # Center of the last 20% area
        
        self.play(FadeIn(highlight_box))

        # 4. Create a Histogram on the right side
        # Positions: centered around x=4
        hist_center_x = 4
        bar_heights = [0.5, 1.2, 3.5, 0.9, 0.4] # The middle bar is significantly taller
        bars = VGroup()
        
        for idx in range(len(bar_heights)):
            h = bar_heights[idx]
            # Use GREEN for the mode (tallest bin), WHITE for others
            b_color = GREEN if idx == 2 else WHITE
            rect = Rectangle(
                width=0.45, 
                height=h, 
                fill_color=b_color, 
                fill_opacity=0.8,
                stroke_color=WHITE,
                stroke_width=1
            )
            # Align bottom of bars to y=-2 (baseline)
            rect.move_to(np.array([hist_center_x + (idx - 2) * 0.55, -2 + h/2, 0]))
            bars.add(rect)
            
        hist_title = Text("Histogram", font_size=22).next_to(bars, UP)
        self.play(Create(bars), Write(hist_title))

        # 5. Animate the Stability Formula
        # Formula: Stability = \frac{\text{counts at mode}}{\text{counts not at mode}}
        stability_formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            color=WHITE
        ).scale(0.8).to_edge(DOWN, buff=0.8)
        
        self.play(Write(stability_formula))
        self.wait(3)
