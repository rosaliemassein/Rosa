from manim import *
import numpy as np

class Concept04ParameterVSDataBottleneck(Scene):
    def construct(self):
        # 1. Setup Manual Layout
        left_center = LEFT * 3.5 + UP * 1
        right_center = RIGHT * 3.5 + UP * 1
        
        # Helper to create manual axes
        def create_axes(center):
            h_line = Line(LEFT * 2, RIGHT * 2, color=GRAY).move_to(center + DOWN * 1.5)
            v_line = Line(DOWN * 1.5, UP * 1.5, color=GRAY).move_to(center + LEFT * 2)
            return VGroup(h_line, v_line)

        axes_l = create_axes(left_center)
        axes_r = create_axes(right_center)
        
        # 2. Titles and Labels
        title_nn = Text("Neural Network", color=BLUE).scale(0.6).next_to(axes_l, UP, buff=0.5)
        label_nn = Text("High Variance", color=RED).scale(0.4).next_to(title_nn, DOWN, buff=0.1)
        
        title_lwr = Text("LWR", color=GREEN).scale(0.6).next_to(axes_r, UP, buff=0.5)
        label_lwr = Text("Robust/Linear", color=BLUE).scale(0.4).next_to(title_lwr, DOWN, buff=0.1)

        # 3. Data Points (Relative to center)
        # Using a few points to represent "small dataset"
        rel_pts = [
            [-1.5, -0.8, 0], [-0.8, 0.5, 0], [0, -0.3, 0], [0.7, 0.9, 0], [1.5, 0.1, 0]
        ]
        
        dots_l = VGroup(*[Dot(left_center + p, radius=0.08, color=WHITE) for p in rel_pts])
        dots_r = VGroup(*[Dot(right_center + p, radius=0.08, color=WHITE) for p in rel_pts])

        # 4. Neural Network Curve (Wiggling erratic path)
        nn_path = VMobject(color=BLUE)
        nn_points = []
        for i in range(len(rel_pts)-1):
            p_start = np.array(rel_pts[i])
            p_end = np.array(rel_pts[i+1])
            # Add a wiggle in between points
            mid = (p_start + p_end) / 2
            wiggle = np.array([0, 0.8 * (-1)**i, 0])
            nn_points.extend([left_center + p_start, left_center + mid + wiggle, left_center + p_end])
        nn_path.set_points_as_corners(nn_points)

        # 5. LWR Curve (Smooth piecewise linear)
        lwr_path = VMobject(color=GREEN)
        lwr_points = [right_center + np.array(p) for p in rel_pts]
        lwr_path.set_points_as_corners(lwr_points)

        # 6. Formula (Bottom Left)
        formula = MathTex(r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}").scale(0.7)
        formula.to_edge(DOWN, buff=1).to_edge(LEFT, buff=1.5)

        # 7. Manual Bar Chart (Bottom Right)
        bar_origin = DOWN * 2.5 + RIGHT * 3
        nn_bar_height = 1.21 * 1.5
        lwr_bar_height = 0.85 * 1.5
        
        nn_bar = Rectangle(height=nn_bar_height, width=0.6, fill_color=BLUE, fill_opacity=0.8)
        nn_bar.move_to(bar_origin + LEFT * 0.5 + UP * (nn_bar_height/2))
        
        lwr_bar = Rectangle(height=lwr_bar_height, width=0.6, fill_color=GREEN, fill_opacity=0.8)
        lwr_bar.move_to(bar_origin + RIGHT * 0.5 + UP * (lwr_bar_height/2))
        
        bar_label_nn = Text("1.21", color=BLUE).scale(0.4).next_to(nn_bar, UP, buff=0.1)
        bar_label_lwr = Text("0.85", color=GREEN).scale(0.4).next_to(lwr_bar, UP, buff=0.1)
        
        bar_name_nn = Text("NN", color=WHITE).scale(0.3).next_to(nn_bar, DOWN, buff=0.1)
        bar_name_lwr = Text("LWR", color=WHITE).scale(0.3).next_to(lwr_bar, DOWN, buff=0.1)
        
        mse_title = Text("MSE Comparison").scale(0.4).next_to(nn_bar, UP, buff=0.8).shift(RIGHT*0.5)

        # 8. Animation Sequence
        self.play(Create(axes_l), Create(axes_r))
        self.play(FadeIn(dots_l), FadeIn(dots_r))
        self.wait(0.5)
        
        self.play(
            Create(nn_path),
            Write(title_nn),
            Write(label_nn),
            run_time=2
        )
        self.play(
            Create(lwr_path),
            Write(title_lwr),
            Write(label_lwr),
            run_time=2
        )
        self.wait(1)
        
        self.play(Write(formula))
        self.play(
            FadeIn(mse_title),
            Create(nn_bar),
            Create(lwr_bar),
            Write(bar_label_nn),
            Write(bar_label_lwr),
            Write(bar_name_nn),
            Write(bar_name_lwr)
        )
        
        self.wait(3)
