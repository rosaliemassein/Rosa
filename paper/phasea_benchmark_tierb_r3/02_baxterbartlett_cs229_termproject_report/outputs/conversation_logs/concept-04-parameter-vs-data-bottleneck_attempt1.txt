from manim import *
import numpy as np

class Concept04ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Formula Reference
        formula = MathTex(r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}").scale(0.8).to_edge(UP, buff=0.3)
        self.add(formula)

        # 2. Setup Manual Coordinate Systems (Avoiding 'Axes' keyword)
        # Left Plot (Neural Network)
        origin_l = np.array([-5, -0.5, 0])
        x_axis_l = Line(origin_l, origin_l + 4 * RIGHT, color=WHITE)
        y_axis_l = Line(origin_l, origin_l + 3 * UP, color=WHITE)
        title_l = Text("Neural Network", color=BLUE).scale(0.5).next_to(y_axis_l, UP)
        subtitle_l = Text("High Variance", color=RED).scale(0.4).next_to(title_l, DOWN, buff=0.1)

        # Right Plot (LWR)
        origin_r = np.array([1, -0.5, 0])
        x_axis_r = Line(origin_r, origin_r + 4 * RIGHT, color=WHITE)
        y_axis_r = Line(origin_r, origin_r + 3 * UP, color=WHITE)
        title_r = Text("LWR", color=GOLD).scale(0.5).next_to(y_axis_r, UP)
        subtitle_r = Text("Robust / Smooth", color=GREEN).scale(0.4).next_to(title_r, DOWN, buff=0.1)

        # 3. Data Points (27 points)
        np.random.seed(42)
        x_data = np.linspace(0.4, 3.6, 27)
        y_data = 0.5 * x_data + 0.8 + np.random.normal(0, 0.25, 27)
        
        dots_l = VGroup(*[Dot(origin_l + x*RIGHT + y*UP, radius=0.04, color=WHITE) for x, y in zip(x_data, y_data)])
        dots_r = VGroup(*[Dot(origin_r + x*RIGHT + y*UP, radius=0.04, color=WHITE) for x, y in zip(x_data, y_data)])

        # 4. Fit Curves
        # Neural Network: Highly erratic (Overfitting)
        nn_curve = VMobject(color=BLUE)
        nn_curve_points = []
        for x_val in np.linspace(0.4, 3.6, 120):
            # Complex oscillation to represent overfitting the noise
            y_val = 0.5 * x_val + 0.8 + 0.4 * np.sin(10 * x_val) + 0.2 * np.cos(18 * x_val)
            nn_curve_points.append(origin_l + x_val * RIGHT + y_val * UP)
        nn_curve.set_points_as_corners(nn_curve_points)

        # LWR: Piecewise-linear / Smooth Path
        lwr_curve = Line(
            origin_r + 0.4 * RIGHT + (0.5 * 0.4 + 0.8) * UP,
            origin_r + 3.6 * RIGHT + (0.5 * 3.6 + 0.8) * UP,
            color=GOLD,
            stroke_width=4
        )

        # 5. MSE Bar Chart (Bottom)
        bar_base_y = -3.2
        nn_bar = Rectangle(height=1.21, width=0.8, fill_opacity=0.8, color=BLUE).move_to(np.array([-1.5, bar_base_y, 0]), aligned_edge=DOWN)
        lwr_bar = Rectangle(height=0.85, width=0.8, fill_opacity=0.8, color=GOLD).move_to(np.array([1.5, bar_base_y, 0]), aligned_edge=DOWN)
        
        nn_mse_text = Text("1.21", font_size=20).next_to(nn_bar, UP, buff=0.1)
        lwr_mse_text = Text("0.85", font_size=20).next_to(lwr_bar, UP, buff=0.1)
        
        nn_label = Text("NN Error", font_size=18).next_to(nn_bar, DOWN, buff=0.1)
        lwr_label = Text("LWR Error", font_size=18).next_to(lwr_bar, DOWN, buff=0.1)

        # 6. Animations
        self.play(
            Create(x_axis_l), Create(y_axis_l),
            Create(x_axis_r), Create(y_axis_r),
            Write(title_l), Write(title_r)
        )
        self.play(FadeIn(dots_l), FadeIn(dots_r))
        self.wait(0.5)
        
        self.play(Create(nn_curve), Write(subtitle_l))
        self.play(Create(lwr_curve), Write(subtitle_r))
        self.wait(1)
        
        self.play(
            FadeIn(nn_bar, shift=UP),
            FadeIn(lwr_bar, shift=UP),
            Write(nn_mse_text),
            Write(lwr_mse_text),
            Write(nn_label),
            Write(lwr_label)
        )
        self.wait(3)
