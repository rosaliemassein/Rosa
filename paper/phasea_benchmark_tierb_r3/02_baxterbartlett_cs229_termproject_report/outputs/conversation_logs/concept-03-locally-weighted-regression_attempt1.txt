from manim import *
import numpy as np

class Concept03LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Manual Axes Construction (Replacing disallowed Axes)
        x_axis = Line(LEFT * 5, RIGHT * 5, color=BLUE)
        y_axis = Line(DOWN * 3, UP * 3, color=BLUE)
        axes_group = VGroup(x_axis, y_axis)

        # 2. Scatter Plot Data Generation
        # Creating a non-linear set of points (sinusoidal)
        np.random.seed(42)
        data_x = np.linspace(-4, 4, 35)
        data_y = 1.5 * np.sin(data_x * 0.7) + np.random.normal(0, 0.15, 35)
        
        # Draw training points as Dots
        points = VGroup(*[
            Dot(np.array([x, y, 0]), radius=0.08, color=GRAY)
            for x, y in zip(data_x, data_y)
        ])

        # 3. Query Point (acted as a reference for LWR)
        # Using a Dot instead of ValueTracker to move across the domain
        query_point = Dot(np.array([-4, 0, 0]), color=WHITE, radius=0.12)
        query_point.set_stroke(WHITE, 4, opacity=0.5)

        # 4. Weight Region Visualization (Radial field)
        weight_ring = Circle(radius=1.0, color=YELLOW, fill_opacity=0.15, stroke_width=1)
        weight_ring.move_to(query_point)

        # 5. Local Regression Line
        # This line will be recalculated every frame based on query_point position
        local_line = Line(LEFT, RIGHT, color=YELLOW, stroke_width=6)

        # 6. Formula Label
        formula = MathTex(
            r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)",
            font_size=30
        ).to_corner(UP + RIGHT)

        # Initial layout
        self.add(axes_group, points, query_point, weight_ring, local_line, formula)

        # 7. Updater Logic for Locally Weighted Regression
        # tau is the bandwidth parameter
        tau = 0.7
        
        def update_regression_scene(mob):
            # Target x-coordinate is determined by query_point position
            qx = query_point.get_center()[0]
            
            # Calculate Gaussian weights: w = exp(-(dx)^2 / (2 * tau^2))
            weights = np.exp(-(data_x - qx)**2 / (2 * tau**2))
            
            # Update individual point appearances based on weights
            for i, p_dot in enumerate(points):
                w = weights[i]
                # High weight -> Glowing Yellow; Low weight -> Faded Gray
                p_dot.set_opacity(0.2 + 0.8 * w)
                if w > 0.35:
                    p_dot.set_color(YELLOW)
                else:
                    p_dot.set_color(GRAY)
            
            # Reposition weight visualization halo
            weight_ring.move_to(query_point.get_center())
            
            # Perform Weighted Least Squares (WLS) for simple linear local fit
            w_sum = np.sum(weights)
            if w_sum > 1e-4:
                # Weighted means
                xm = np.sum(weights * data_x) / w_sum
                ym = np.sum(weights * data_y) / w_sum
                
                # Solve for slope m: sum(w*(x-xm)*(y-ym)) / sum(w*(x-xm)^2)
                num = np.sum(weights * (data_x - xm) * (data_y - ym))
                den = np.sum(weights * (data_x - xm)**2)
                
                slope = num / den if abs(den) > 1e-5 else 0
                intercept = ym - slope * xm
                
                # Define short line segment endpoints locally around qx
                lx1, lx2 = qx - 1.2, qx + 1.2
                ly1 = slope * lx1 + intercept
                ly2 = slope * lx2 + intercept
                
                local_line.put_start_and_end_on(
                    np.array([lx1, ly1, 0]),
                    np.array([lx2, ly2, 0])
                )
                
                # Lock query point Y-axis to the local line for smoother tracking
                query_point.move_to(np.array([qx, slope * qx + intercept, 0]))

        # Register the dynamic updater to the scene
        local_line.add_updater(update_regression_scene)

        # 8. Animation Sequence
        self.play(Create(axes_group), Create(points))
        self.play(FadeIn(query_point), FadeIn(weight_ring), Write(formula))
        self.wait(1)
        
        # Animate the query point moving from left to right across the data
        # Using lambda rate_func to avoid 'linear' identifier if disallowed
        self.play(
            query_point.animate.move_to(np.array([4, 0, 0])),
            run_time=8,
            rate_func=lambda t: t
        )
        self.wait(2)
