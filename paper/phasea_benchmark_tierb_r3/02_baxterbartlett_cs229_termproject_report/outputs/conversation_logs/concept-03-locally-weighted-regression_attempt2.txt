from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Setup Manual Coordinate System (Since Axes is disallowed)
        x_axis = Line(LEFT * 5, RIGHT * 5, color=GREY)
        y_axis = Line(DOWN * 3, UP * 3, color=GREY)
        self.add(x_axis, y_axis)

        # 2. Data Generation
        np.random.seed(42)
        x_vals = np.linspace(-4, 4, 30)
        y_vals = np.sin(x_vals) + np.random.normal(0, 0.2, 30)
        
        dots = VGroup()
        for x, y in zip(x_vals, y_vals):
            dots.add(Dot(np.array([x, y, 0]), radius=0.08, color=BLUE))
        self.add(dots)

        # 3. Query Point and Weight Circle
        # We use a Dot as a tracker since ValueTracker is disallowed
        query_point = Dot(np.array([-4, 0, 0]), color=WHITE)
        weight_circle = Circle(radius=1.0, color=WHITE, stroke_opacity=0.3)
        weight_circle.add_updater(lambda m: m.move_to(query_point.get_center()))
        
        self.add(query_point, weight_circle)

        # 4. Math Logic for Local Regression
        tau = 0.6
        
        # We define a line that will be updated every frame
        local_line = Line(ORIGIN, RIGHT, color=YELLOW, stroke_width=6)
        
        def update_local_line(line):
            qx = query_point.get_center()[0]
            
            # Calculate weights based on query point x
            weights = np.exp(-(x_vals - qx)**2 / (2 * tau**2))
            
            # Update dot appearances based on weights
            for i, dot in enumerate(dots):
                w = weights[i]
                dot.set_opacity(0.2 + 0.8 * w)
                if w > 0.5:
                    dot.set_color(YELLOW)
                else:
                    dot.set_color(BLUE)

            # Weighted Linear Regression: Solve (X.T * W * X) * theta = X.T * W * Y
            # For 1D: A = [[sum w, sum w*x], [sum w*x, sum w*x^2]], B = [sum w*y, sum w*x*y]
            s_w = np.sum(weights)
            s_wx = np.sum(weights * x_vals)
            s_wxx = np.sum(weights * (x_vals**2))
            s_wy = np.sum(weights * y_vals)
            s_wxy = np.sum(weights * x_vals * y_vals)

            det = s_w * s_wxx - s_wx * s_wx
            if abs(det) < 0.0001:
                return

            # theta = [intercept, slope]
            intercept = (s_wy * s_wxx - s_wxy * s_wx) / det
            slope = (s_w * s_wxy - s_wx * s_wy) / det

            # Define line segment around the query point
            x_start = qx - 1.0
            x_end = qx + 1.0
            y_start = slope * x_start + intercept
            y_end = slope * x_end + intercept
            
            line.set_points_as_corners([
                np.array([x_start, y_start, 0]),
                np.array([x_end, y_end, 0])
            ])

        local_line.add_updater(update_local_line)
        self.add(local_line)

        # 5. Narration and Formula
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)")
        formula.to_corner(UL).scale(0.8)
        
        label = Text("Locally Weighted Regression", font_size=24).to_edge(UP)
        
        self.play(Write(formula), Write(label))
        self.wait(1)

        # 6. Animation: Move query point across the data
        # Using a simple rate_func=None or lambda t: t instead of 'linear'
        self.play(
            query_point.animate.move_to(np.array([4, 0, 0])),
            run_time=8,
            rate_func=lambda t: t
        )
        
        self.wait(1)
        
        # Reset position
        self.play(
            query_point.animate.move_to(np.array([0, 0, 0])),
            run_time=2
        )
        
        conclusion = Text("The local expert ignores distant data.", font_size=24, color=YELLOW)
        conclusion.to_edge(UP)
        self.play(Transform(label, conclusion))
        self.wait(2)
