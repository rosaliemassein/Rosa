            Text("Stability", font_size=18)
        ).arrange(DOWN, buff=0.2)
        
        left_box = Rectangle(width=4, height=2.5, color=WHITE).move_to(behavior_text)
        left_side = VGroup(left_box, behavior_text).shift(LEFT * 3.5 + UP * 1.5)

        # Right side: PID Constants
        pid_text = VGroup(
            Text("PID Constants", font_size=24, color=GREEN),

        ).arrange(DOWN, buff=0.3)
        
        right_box = Rectangle(width=4, height=2.5, color=WHITE).move_to(pid_text)
        right_side = VGroup(right_box, pid_text).shift(RIGHT * 3.5 + UP * 1.5)

        # Connector
        mid_arrow = Arrow(left_side.get_right(), right_side.get_left(), color=WHITE)
        
        full_diagram = VGroup(left_side, right_side, mid_arrow)

        self.play(GrowArrow(mid_arrow))
