class ConceptExecutionHandoff(Scene):
    def construct(self):
        # 1. Setup Layout with a simple Line instead of DashedLine
        div_line = Line(LEFT * 7, RIGHT * 7, color=GRAY)
        self.add(div_line)
        
        title_d = Text("Data Agent (A_d)", color=BLUE, font_size=32).to_edge(UP).shift(LEFT * 3)
        title_m = Text("Model Agent (A_m)", color=GREEN, font_size=32).to_edge(DOWN).shift(LEFT * 3)
        self.add(title_d, title_m)

        # 2. Search Efficiency Bar (Animated manually without ValueTracker)
        eff_label = Text("Search Efficiency", font_size=24).to_edge(UR, buff=0.5).shift(LEFT * 2.2)
        bar_bg = Rectangle(width=3, height=0.3, color=WHITE, stroke_width=1).next_to(eff_label, RIGHT)
        # Start with a very thin bar to represent 0
        eff_bar = Rectangle(
            width=0.01, 
            height=0.2, 
            fill_opacity=1, 
            fill_color=YELLOW, 
            stroke_width=0
        ).align_to(bar_bg, LEFT).shift(RIGHT * 0.05)
        
        self.add(eff_label, bar_bg, eff_bar)

        # 3. Data Agent Section
        data_block = VGroup(
            Rectangle(width=2.5, height=1.5, color=BLUE, fill_opacity=0.2),
            Text("Data Block", font_size=20)
        ).move_to(UP * 1.8 + LEFT * 3)
        
        self.play(Create(data_block))
        self.wait(0.5)

        # Processing step - increase efficiency bar
        insight_od = MathTex("O_d", color=YELLOW).scale(1.5).move_to(data_block)
        insight_label = Text("Data Insights", font_size=18, color=YELLOW).next_to(insight_od, DOWN)
        insights_group = VGroup(insight_od, insight_label)

        self.play(
            Indicate(data_block, color=BLUE_A),
            eff_bar.animate.stretch_to_fit_width(1.0, about_edge=LEFT),
            run_time=1.5
        )
        self.play(ReplacementTransform(data_block, insights_group))
        self.wait(0.5)

        # 4. Handoff Animation
        handoff_arrow = Arrow(UP, DOWN, color=WHITE).move_to(ORIGIN)
        self.play(Create(handoff_arrow))
        
        model_pos = DOWN * 1.8 + LEFT * 3
        self.play(
            insights_group.animate.move_to(model_pos),
            handoff_arrow.animate.shift(DOWN * 1).set_opacity(0),
            eff_bar.animate.stretch_to_fit_width(2.0, about_edge=LEFT),
            run_time=1.5
        )

        # 5. Model Agent Expansion
        # Formula: s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)
        formula = MathTex(
            "s_i^m = \\mathrm{PD}(R, \\mathcal{A}_m, \\mathbf{p}_i, O_i^d)", 
            font_size=32
        ).move_to(model_pos)
        
        task_box = Rectangle(width=6, height=1.2, color=GREEN).move_to(model_pos)
        
        self.play(
            ReplacementTransform(insights_group, formula),
            Create(task_box),
            eff_bar.animate.stretch_to_fit_width(2.9, about_edge=LEFT),
            run_time=2
        )

        # Final visual feedback
        self.play(Indicate(formula, color=GOLD))
        
        training_free_tag = Text("TRAINING-FREE SEARCH", color=GOLD, font_size=20).next_to(bar_bg, DOWN, buff=0.3)
        self.play(Write(training_free_tag))
        
        self.wait(2)
