from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Coordinate system for plotting Matrix Size vs Total States
        # Log10 scale used for Y-axis (e.g., 2^25 states is ~7.5 on log10 scale)
        axes = Axes(
            x_range=[2, 7, 1],
            y_range=[0, 12, 2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(RIGHT, buff=0.5)
        
        # Define the function f(x) = log10(2^(x^2)) = x^2 * log10(2)
        # We use log10 to represent the exponential growth on a log-scale Y-axis
        log10_2 = 0.30103
        graph = axes.plot(lambda x: x * x * log10_2, color=BLUE)
        
        # Formula and Labels
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP)
        x_label = axes.get_x_axis_label("N", edge=DOWN, direction=DOWN)
        y_label = axes.get_y_axis_label("10^k", edge=LEFT, direction=LEFT)
        
        # 2. Initial State: 2x2 Grid
        grid = VGroup(*[Square(side_length=0.4) for i in range(4)]).arrange_in_grid(rows=2, cols=2, buff=0.1)
        grid.to_edge(LEFT, buff=1.5).shift(UP * 0.5)
        grid_label = Text("2x2 Grid", font_size=24).next_to(grid, UP)
        
        # Q-Table Representation (Initial rows)
        table_box = Rectangle(width=2, height=3).next_to(grid, DOWN, buff=0.5)
        table_title = Text("Q-Table", font_size=24).next_to(table_box, UP, buff=0.1)
        rows = VGroup(*[Line(LEFT, RIGHT).set_width(1.7) for i in range(10)]).arrange(DOWN, buff=0.15).move_to(table_box)
        
        # Setup the Scene
        self.add(axes, x_label, y_label, formula, grid, grid_label, table_box, table_title, rows)
        self.play(Create(graph), run_time=2)
        self.wait(1)
        
        # 3. Animation: Explosion of States
        # Transition to 5x5 Grid (N=5)
        grid_5x5 = VGroup(*[Square(side_length=0.12) for i in range(25)]).arrange_in_grid(rows=5, cols=5, buff=0.04).move_to(grid)
        grid_5x5_label = Text("5x5 Grid", font_size=24).move_to(grid_label)
        
        # Table rows multiply into a "solid blur" of color (many thin lines)
        blur_rows = VGroup(*[
            Line(LEFT, RIGHT, stroke_width=0.5, color=BLUE).set_width(1.8) 
            for i in range(100)
        ]).arrange(DOWN, buff=0.01).move_to(table_box)
        
        self.play(
            Transform(grid, grid_5x5),
            Transform(grid_label, grid_5x5_label),
            Transform(rows, blur_rows),
            graph.animate.set_color(RED),
            run_time=3
        )
        
        # 4. Scrolling Animation
        # Representing high-speed processing of millions of states
        self.play(rows.animate.shift(UP * 0.8), rate_func=lambda t: t, run_time=0.4)
        self.play(rows.animate.shift(DOWN * 1.6), rate_func=lambda t: t, run_time=0.6)
        self.play(rows.animate.shift(UP * 0.8), rate_func=lambda t: t, run_time=0.4)
        
        # 5. Final: Graph shoots off the top of the screen
        # We simulate this by extending the plot range further
        graph_shoot = axes.plot(lambda x: x * x * log10_2, x_range=[2, 8], color=RED)
        self.play(Transform(graph, graph_shoot), run_time=1)
        
        # Conclusion text
        explosion_text = Text("State Space Explosion", color=RED).scale(0.8).next_to(axes, DOWN)
        self.play(Write(explosion_text))
        self.wait(2)
