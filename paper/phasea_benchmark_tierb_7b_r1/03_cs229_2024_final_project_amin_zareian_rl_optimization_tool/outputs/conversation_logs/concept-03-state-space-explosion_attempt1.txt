from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Formula Header
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}")
        formula.to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Axes Setup (mimicking Reference Example 3)
        # N: 2 to 6. Y-axis represents log2(States) = N^2.
        # This makes the exponential growth visible on a manageable scale.
        axes = Axes(
            x_range=[2, 7, 1],
            y_range=[0, 45, 10],
            x_length=6,
            y_length=4
        ).shift(LEFT * 1.5 + DOWN * 0.5)
        
        # Plotting y = x^2 using 'x' as used in Reference Example 3
        # This represents the log-scale growth of states.
        graph = axes.plot(lambda x: x**2, color=YELLOW)
        
        x_label = MathTex("N").next_to(axes.x_axis, RIGHT)
        y_label = MathTex(r"\log_2(S)").next_to(axes.y_axis, UP)
        
        self.add(axes, x_label, y_label)
        self.play(Create(graph), run_time=2)

        # 3. Initial 2x2 Grid and State Rows
        grid_2x2 = VGroup(
            Square(side_length=1.5, color=BLUE),
            Line(LEFT * 0.75, RIGHT * 0.75, color=BLUE),
            Line(UP * 0.75, DOWN * 0.75, color=BLUE)
        ).shift(RIGHT * 3.5 + UP * 0.5)
        
        grid_lbl = MathTex(r"2 \times 2").next_to(grid_2x2, UP)
        
        # 16 states represented by distinct rows
        rows = VGroup(*[
            Line(LEFT * 0.5, RIGHT * 0.5, color=BLUE, stroke_width=1)
            for _ in range(16)
        ]).arrange(DOWN, buff=0.04).next_to(grid_2x2, DOWN, buff=0.5)
        
        self.add(grid_2x2, grid_lbl, rows)
        self.wait(1)

        # 4. Transition to 5x5 Grid
        grid_5x5 = VGroup(Square(side_length=1.5, color=BLUE))
        for i in range(1, 5):
            d = -0.75 + i * 0.3
            grid_5x5.add(Line(LEFT * 0.75, RIGHT * 0.75, color=BLUE).shift(UP * d))
            grid_5x5.add(Line(UP * 0.75, DOWN * 0.75, color=BLUE).shift(RIGHT * d))
        grid_5x5.move_to(grid_2x2)
        
        # Solid blur representing the overwhelming number of states (2^25)
        blur = Rectangle(width=1.2, height=2, fill_opacity=0.8, color=BLUE)
        blur.set_fill(BLUE, opacity=0.8)
        blur.next_to(grid_2x2, DOWN, buff=0.5)
        
        new_lbl = MathTex(r"5 \times 5").next_to(grid_2x2, UP)
        
        # 5. State Space Explosion Animation
        self.play(
            Transform(grid_2x2, grid_5x5),
            ReplacementTransform(rows, blur),
            Transform(grid_lbl, new_lbl),
            graph.animate.set_color(RED).scale(1.5, about_point=axes.c2p(2,4)),
            run_time=2
        )
        
        # Display the exponential result
        res = MathTex(r"2^{25} \approx 3.3 \times 10^7", color=RED).next_to(blur, DOWN)
        self.play(Write(res))
        self.wait(2)
