from manim import *

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Custom Axes (Avoiding Axes class)
        origin = LEFT * 5 + DOWN * 3
        x_axis = Line(origin, origin + RIGHT * 6, color=WHITE)
        y_axis = Line(origin, origin + UP * 5.5, color=WHITE)
        
        # Labels for axes
        x_label = Text("Matrix Size (N)", font_size=20).next_to(origin + RIGHT * 3, DOWN)
        y_label = Text("States (log scale)", font_size=20).rotate(90 * DEGREES).next_to(origin + UP * 2.75, LEFT)
        
        # Ticks for Matrix Size 2 to 6
        x_ticks = VGroup()
        for i in range(5):
            n_val = i + 2
            tick = Line(DOWN * 0.1, UP * 0.1).move_to(origin + RIGHT * i * 1.5)
            label = Text(str(n_val), font_size=16).next_to(tick, DOWN)
            x_ticks.add(tick, label)
            
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label), Create(x_ticks))

        # 2. Formula
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP, buff=0.5)
        self.play(Write(formula))

        # 3. Small table graphic for 2x2 grid
        # 2x2 grid visual
        grid_2x2 = VGroup()
        for i in range(2):
            for j in range(2):
                sq = Square(side_length=0.4, stroke_width=2).move_to(RIGHT * 3.5 + UP * 2 + RIGHT * i * 0.45 + DOWN * j * 0.45)
                grid_2x2.add(sq)
        
        grid_label = Text("2x2 Grid", font_size=20).next_to(grid_2x2, UP)
        self.play(Create(grid_2x2), Write(grid_label))

        # Table rows representation
        rows = VGroup(*[
            Rectangle(width=2.5, height=0.08, fill_opacity=0.4, fill_color=BLUE, stroke_width=1)
            for _ in range(10)
        ]).arrange(DOWN, buff=0.05).next_to(grid_2x2, DOWN, buff=0.4)
        
        rows_label = Text("Q-Table Rows", font_size=18).next_to(rows, UP, buff=0.1)
        self.play(Create(rows), Write(rows_label))

        # 4. Plot the growth function (log scale of 2^(N^2) is N^2)
        # We plot y = N^2, mapping N=[2,6] to x=[0,6] and N^2=[4,36] to y=[0,5]
        def get_graph_point(n):
            x_coord = (n - 2) * 1.5
            y_coord = (n**2 - 4) * (5 / 32)
            return origin + RIGHT * x_coord + UP * y_coord

        points = [get_graph_point(2 + i * 3 / 10) for i in range(11)]
        graph = VMobject(color=YELLOW).set_points_as_corners(points)
        
        self.play(Create(graph), run_time=2)

        # 5. Animate to 5x5 Grid and State Explosion
        grid_5x5 = VGroup()
        for i in range(5):
            for j in range(5):
                sq = Square(side_length=0.15, stroke_width=1).move_to(RIGHT * 3.5 + UP * 2 + RIGHT * i * 0.18 + DOWN * j * 0.18)
                grid_5x5.add(sq)
        
        # Rapidly multiplying rows effect
        blurred_rows = Rectangle(
            width=2.5, height=3, 
            fill_opacity=0.9, 
            fill_color=BLUE, 
            stroke_width=0
        ).move_to(rows).shift(DOWN * 0.5)
        
        explosion_text = Text("33.5 MILLION STATES", color=RED, font_size=24).move_to(blurred_rows)
        
        # Extending the graph off-screen
        ext_points = [get_graph_point(5 + i * 2 / 10) for i in range(6)]
        graph_ext = VMobject(color=YELLOW).set_points_as_corners(ext_points)

        self.play(
            Transform(grid_2x2, grid_5x5),
            grid_label.animate.become(Text("5x5 Grid", font_size=20).move_to(grid_label)),
            rows.animate.shift(UP * 2).set_opacity(0),
            FadeIn(blurred_rows),
            Create(graph_ext),
            formula.animate.scale(1.1).set_color(RED),
            run_time=2
        )
        
        self.play(Write(explosion_text))
        self.wait(2)
