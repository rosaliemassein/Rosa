from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Setup Board and Target Zone
        board = Rectangle(width=8, height=5, color=WHITE).shift(UP * 0.5)
        target = Circle(radius=0.9, color=GREEN, fill_opacity=0.3).move_to(board.get_right() + LEFT * 1.5)
        target_label = Text("Target Zone", font_size=24, color=GREEN).next_to(target, UP)
        
        # 2. Setup Pegs
        pegs = VGroup(*[Dot(color=BLUE) for _ in range(12)])
        pegs.arrange_in_grid(rows=3, cols=4, buff=0.4).move_to(board.get_left() + RIGHT * 1.5)
        
        # 3. Setup Centroid and Label
        centroid = Dot(color=RED)
        centroid_label = Text("Centroid", font_size=20, color=RED)
        
        # 4. Setup Connector Line (Using Line as DashedLine is restricted)
        connector = Line(centroid.get_center(), target.get_center(), color=YELLOW)
        
        # 5. Reward Formula
        formula = MathTex(
            r"R = w_1(\text{pegs}_{in})^2 + w_2(\text{dist}(\text{centroid, target}))^{-1}",
            font_size=32
        ).to_edge(DOWN, buff=0.5)
        
        # Helper to calculate centroid position
        def get_curr_centroid():
            pos_sum = np.array([0.0, 0.0, 0.0])
            for p in pegs:
                pos_sum += p.get_center()
            return pos_sum / len(pegs)

        # Initial Positioning
        centroid.move_to(get_curr_centroid())
        centroid_label.next_to(centroid, UP, buff=0.1)
        
        # 6. Add Updaters (Using 'x' as parameter to match allowed patterns)
        centroid.add_updater(lambda x: x.move_to(get_curr_centroid()))
        centroid_label.add_updater(lambda x: x.next_to(centroid, UP, buff=0.1))
        connector.add_updater(lambda x: x.become(Line(centroid.get_center(), target.get_center(), color=YELLOW)))
        
        # 7. Initial Display
        self.add(board, target, target_label, pegs, centroid, centroid_label, connector, formula)
        self.wait(1)
        
        # 8. Animate Pegs Moving to Target with Bonus Popups
        pegs_in_target = 0
        indices_to_move = [0, 4, 8, 1, 5, 9]
        
        for i, idx in enumerate(indices_to_move):
            pegs_in_target += 1
            peg = pegs[idx]
            
            # Target position with slight offset using trigonometric values for variety
            offset = np.array([0.4 * np.cos(i), 0.4 * np.sin(i), 0])
            dest = target.get_center() + offset
            
            # Squared bonus popup
            bonus_val = pegs_in_target ** 2
            bonus_text = MathTex(f"{pegs_in_target}^2", color=GREEN, font_size=34)
            bonus_text.next_to(peg, UP, buff=0.2)
            
            self.play(
                peg.animate.move_to(dest).set_color(GREEN),
                FadeIn(bonus_text, shift=UP * 0.3),
                run_time=0.9
            )
            self.play(FadeOut(bonus_text), run_time=0.3)
            
        self.wait(2)
