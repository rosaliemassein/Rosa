from manim import *

class OpponentAbstraction(Scene):
    def construct(self):
        # 0. Formula Reference
        formula = MathTex(r"S \in \mathbb{B}^{n \times n \times 3}").to_edge(UP)
        self.add(formula)

        # 1. 2-player board representation
        # Creating a simple grid background
        grid = Square(side_length=4).set_stroke(GRAY, 1)
        blue_player = Circle(radius=0.5, color=BLUE, fill_opacity=0.8).shift(LEFT)
        red_player = Circle(radius=0.5, color=RED, fill_opacity=0.8).shift(RIGHT)
        
        label_2p = Text("2-Player Representation", font_size=24).next_to(grid, UP)
        me_text = Text("Me", color=BLUE, font_size=20).next_to(blue_player, DOWN)
        opp_text = Text("Opponent", color=RED, font_size=20).next_to(red_player, DOWN)

        self.play(Create(grid), Write(label_2p))
        self.play(FadeIn(blue_player), FadeIn(red_player), Write(me_text), Write(opp_text))
        self.wait(1)

        # 2. Show 6-player board with 5 different opponent colors
        # Removing 2-player labels to transition
        self.play(FadeOut(red_player), FadeOut(opp_text), FadeOut(label_2p))

        # 5 different colors for opponents (avoiding PINK as it caused errors)
        opp_colors = [RED, GREEN, PURPLE, ORANGE, MAGENTA]
        opponents_6p = VGroup(*[
            Circle(radius=0.3, color=c, fill_opacity=0.8) for c in opp_colors
        ])
        # Arrange opponents in a circular or grid fashion on the right
        opponents_6p.arrange_in_grid(rows=3, cols=2, buff=0.3).move_to(RIGHT * 1.5)
        
        label_6p = Text("6-Player Game", font_size=24).next_to(grid, UP)
        
        self.play(Write(label_6p), FadeIn(opponents_6p))
        self.wait(1)

        # 3. Cross-out individual identities
        # Manually drawing crosses to avoid "Cross" identifier issues
        crosses = VGroup()
        for opp in opponents_6p:
            center = opp.get_center()
            size = 0.2
            l1 = Line(center + UL*size, center + DR*size, color=WHITE, stroke_width=4)
            l2 = Line(center + UR*size, center + DL*size, color=WHITE, stroke_width=4)
            crosses.add(VGroup(l1, l2))
        
        self.play(Create(crosses))
        self.wait(0.5)

        # 4. Merge into a single 'Red' intensity (Opponent Layer)
        unified_opponent = Circle(radius=0.6, color=RED, fill_opacity=0.8).move_to(RIGHT * 1.5)
        unified_label = Text("Unified 'Opponent Layer'", color=RED, font_size=24).next_to(unified_opponent, DOWN)

        self.play(
            FadeOut(opponents_6p),
            FadeOut(crosses),
            FadeIn(unified_opponent),
            Write(unified_label),
            run_time=1.5
        )

        # 5. Check mark over the unified layer
        # Constructing a check mark manually or with Tex
        check_mark = MathTex(r"\checkmark", color=GREEN).scale(3).next_to(unified_opponent, RIGHT, buff=0.5)
        
        self.play(Write(check_mark))
        self.wait(1)

        # Final cleanup and emphasis on scaling
        final_text = Text("Strategy scales to any N", font_size=28).to_edge(DOWN)
        self.play(Write(final_text), formula.animate.scale(1.2).set_color(YELLOW))
        self.wait(2)
