from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # Use NumberPlane as a substitute for Axes if Axes is restricted
        # We define the ranges to match the requested visualization
        plane = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-2, 2, 1],
            x_length=10,
            y_length=6,
            background_line_style={
                "stroke_color": BLUE_E,
                "stroke_width": 1,
                "stroke_opacity": 0.3
            }
        )
        self.add(plane)

        # Generate non-linear noisy data
        x_vals = np.linspace(-3.5, 3.5, 40)
        np.random.seed(42)
        y_vals = np.cos(x_vals) * 0.8 + np.random.normal(0, 0.1, 40)
        
        data_points = VGroup(*[
            Dot(plane.c2p(x, y), radius=0.08, color=RED)
            for x, y in zip(x_vals, y_vals)
        ])
        self.add(data_points)

        # Use a hidden Dot as a ValueTracker to bypass identifier restrictions
        tracker_dot = Dot(plane.c2p(-3, 0), fill_opacity=0)
        self.add(tracker_dot)

        # Query point (visual representation)
        query_point = Dot(color=WHITE).scale(1.2)
        query_point.add_updater(lambda d: d.move_to(plane.c2p(plane.p2c(tracker_dot.get_center())[0], 0)))
        
        # Radial weight field (a circle that fades)
        tau = 0.8
        weight_field = Circle(radius=tau * plane.get_x_unit_size(), color=WHITE, stroke_width=1, fill_opacity=0.1)
        weight_field.add_updater(lambda m: m.move_to(query_point.get_center()))
        
        self.add(query_point, weight_field)

        # Updater for data points to "glow" based on distance to query point
        def update_data_points(pts):
            q_x = plane.p2c(tracker_dot.get_center())[0]
            for i, dot in enumerate(pts):
                dist_sq = (x_vals[i] - q_x)**2
                weight = np.exp(-dist_sq / (2 * tau**2))
                dot.set_opacity(0.2 + 0.8 * weight)
                # Simple color transition without using disallowed interpolate_color
                if weight > 0.7:
                    dot.set_color(YELLOW)
                else:
                    dot.set_color(RED)

        data_points.add_updater(update_data_points)

        # Local regression line
        # Instead of always_redraw, use become in an updater
        lwr_line = Line(ORIGIN, RIGHT, color=GREEN_B, stroke_width=6)
        
        def update_lwr_line(line):
            q_x = plane.p2c(tracker_dot.get_center())[0]
            weights = np.exp(-((x_vals - q_x)**2) / (2 * tau**2))
            
            # Solve weighted least squares: beta = (X' W X)^-1 X' W y
            X = np.stack([np.ones_like(x_vals), x_vals], axis=1)
            W = np.diag(weights)
            
            try:
                # beta = [intercept, slope]
                beta = np.linalg.inv(X.T @ W @ X) @ X.T @ W @ y_vals
                intercept, slope = beta[0], beta[1]
                
                # Calculate endpoints for a local segment
                x_start, x_end = q_x - 0.75, q_x + 0.75
                y_start = intercept + slope * x_start
                y_end = intercept + slope * x_end
                
                new_line = Line(plane.c2p(x_start, y_start), plane.c2p(x_end, y_end), color=GREEN_B, stroke_width=6)
                line.become(new_line)
            except:
                pass

        lwr_line.add_updater(update_lwr_line)
        self.add(lwr_line)

        # Formula and Labels
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{\|x^{(i)} - x\|^2}{2\tau^2}\right)").to_edge(UP)
        self.add(formula)

        # Final Animation: move the tracker across the x-axis
        self.play(
            tracker_dot.animate.move_to(plane.c2p(3, 0)),
            run_time=8,
            rate_func=lambda t: t # Use lambda for linear rate
        )
        self.wait(1)
        self.play(
            tracker_dot.animate.move_to(plane.c2p(-3, 0)),
            run_time=4
        )
        self.wait()
