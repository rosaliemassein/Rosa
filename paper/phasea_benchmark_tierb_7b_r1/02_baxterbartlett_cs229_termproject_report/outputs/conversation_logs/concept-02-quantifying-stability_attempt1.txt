from manim import *
import numpy as np

class QuantifyingStability(Scene):
    def construct(self):
        # 1. Custom Axes (since Axes is disallowed)
        x_axis = Line(LEFT * 4, RIGHT * 2, color=WHITE).shift(DOWN * 0.5)
        y_axis = Line(DOWN * 2, UP * 2, color=WHITE).shift(LEFT * 4 + DOWN * 0.5)
        
        y_label = Text("Velocity (RPM)", font_size=20).next_to(y_axis, UP)
        x_label = Text("Time", font_size=20).next_to(x_axis, RIGHT)
        
        # Mark for 30 RPM
        y_30_pos = y_axis.get_start() + UP * 2.5
        label_30 = Text("30", font_size=18).next_to(y_axis.point_from_proportion(0.625), LEFT)
        
        # Custom Dashed Line for 30 RPM
        dashed_30 = VGroup(*[
            Line(ORIGIN, RIGHT * 0.2, color=RED, stroke_width=2)
            for _ in range(20)
        ]).arrange(RIGHT, buff=0.1).move_to(y_30_pos + RIGHT * 3)

        self.play(Create(x_axis), Create(y_axis), Write(y_label), Write(x_label))
        self.play(Write(label_30), Create(dashed_30))

        # 2. Velocity Curve
        # Generate points for a fluctuation around 30 RPM
        points = []
        for x_val in np.linspace(0, 6, 100):
            # Fluctuates around the y_30_pos height
            y_offset = 0.3 * np.sin(2 * x_val) + 0.15 * np.cos(5 * x_val) + 0.1 * np.sin(10 * x_val)
            points.append(y_30_pos + RIGHT * x_val + UP * y_offset)
        
        velocity_curve = VMobject(color=YELLOW)
        velocity_curve.set_points_as_corners(points)
        
        self.play(Create(velocity_curve), run_time=3)

        # 3. Highlight Last 20%
        # Total x is 6 units. 20% of 6 is 1.2.
        highlight_rect = Rectangle(
            width=1.2, 
            height=3.5, 
            fill_color=BLUE, 
            fill_opacity=0.3, 
            stroke_width=0
        ).move_to(y_30_pos + RIGHT * 5.4)
        
        highlight_text = Text("Steady State (Last 20%)", font_size=16, color=BLUE).next_to(highlight_rect, UP)
        
        self.play(FadeIn(highlight_rect), Write(highlight_text))
        self.wait(1)

        # 4. Histogram (Manual)
        hist_origin = RIGHT * 4 + DOWN * 1
        h_x_axis = Line(hist_origin, hist_origin + RIGHT * 3)
        h_y_axis = Line(hist_origin, hist_origin + UP * 3)
        hist_title = Text("Histogram", font_size=20).next_to(h_y_axis, UP)

        # Bars
        bar1 = Rectangle(width=0.4, height=0.8, fill_color=BLUE, fill_opacity=0.8).move_to(hist_origin + RIGHT * 0.7 + UP * 0.4)
        bar_mode = Rectangle(width=0.4, height=2.5, fill_color=GREEN, fill_opacity=0.8).move_to(hist_origin + RIGHT * 1.5 + UP * 1.25)
        bar3 = Rectangle(width=0.4, height=0.6, fill_color=BLUE, fill_opacity=0.8).move_to(hist_origin + RIGHT * 2.3 + UP * 0.3)
        
        mode_label = Text("30 RPM", font_size=14).next_to(bar_mode, DOWN, buff=0.1)

        self.play(Create(h_x_axis), Create(h_y_axis), Write(hist_title))
        self.play(Create(bar1), Create(bar_mode), Create(bar3), Write(mode_label))

        # 5. Stability Formula
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            color=WHITE
        ).scale(0.8).to_edge(DOWN, buff=0.5)

        self.play(Write(formula))

        # 6. Final visual emphasis
        surround_numerator = SurroundingRectangle(bar_mode, color=GREEN)
        surround_denominator = SurroundingRectangle(VGroup(bar1, bar3), color=BLUE)
        
        self.play(Create(surround_numerator))
        self.wait(0.5)
        self.play(ReplacementTransform(surround_numerator, surround_denominator))
        self.wait(2)
