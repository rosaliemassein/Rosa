from manim import *
import numpy as np

class InverseMappingConcept(Scene):
    def construct(self):
        # 1. Formula reference at the top
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}")
        formula.to_edge(UP, buff=0.5).scale(0.8)
        self.add(formula)

        # 2. Input/Output Boxes (Top Section)
        # Left Group: Desired Behavior
        input_rect = Rectangle(width=4.0, height=1.8, color=BLUE)
        input_title = Text("Desired Behavior", font_size=22, color=BLUE).shift(UP * 0.4)
        input_info = Text("Rise Time, SS Error", font_size=16).shift(DOWN * 0.4)
        input_group = VGroup(input_rect, input_title, input_info).shift(LEFT * 3.5 + UP * 1.5)

        # Right Group: PID Constants
        output_rect = Rectangle(width=3.5, height=1.8, color=GREEN)
        output_title = Text("PID Constants", font_size=22, color=GREEN)
        output_group = VGroup(output_rect, output_title).shift(RIGHT * 3.5 + UP * 1.5)

        # Large arrow between input and output
        arrow = Arrow(input_group.get_right(), output_group.get_left(), color=WHITE)
        
        # Group diagram for rotation
        diagram = VGroup(input_group, arrow, output_group)

        self.play(Create(input_group), Create(output_group))
        self.play(GrowArrow(arrow))
        self.wait(1)

        # "Flips the script": Rotate the entire diagram 180 degrees
        # Using .animate.rotate instead of the Rotate animation class to ensure compatibility
        self.play(diagram.animate.rotate(3.14159, axis=[0, 0, 1]))
        self.wait(1)

        # 3. Motor Velocity Plot (Bottom Section)
        # Creating a manual coordinate system since Axes is restricted
        origin = DOWN * 2.5 + LEFT * 4
        x_axis = Line(origin, origin + RIGHT * 8, color=WHITE)
        y_axis = Line(origin, origin + UP * 2.5, color=WHITE)
        x_label = Text("Time", font_size=18).next_to(x_axis, RIGHT, buff=0.1)
        y_label = Text("RPM", font_size=18).next_to(y_axis, UP, buff=0.1)
        
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))

        # Target Setpoint Line (Manual dashed line)
        target_height = 1.8
        setpoint_y = origin[1] + target_height
        dashes = VGroup()
        for i in range(25):
            d_start = [origin[0] + i * 0.32, setpoint_y, 0]
            d_end = [origin[0] + i * 0.32 + 0.16, setpoint_y, 0]
            dashes.add(Line(d_start, d_end, color=RED, stroke_width=2))
        
        setpoint_label = Text("Setpoint", font_size=14, color=RED).next_to(dashes, RIGHT, buff=0.1)
        self.play(Create(dashes), Write(setpoint_label))

        # Motor response curve points
        # Simulation: 1 - exp(-0.6t) * cos(4t)
        curve_points = []
        for i in range(101):
            t_val = i * 8.0 / 100.0 # Time from 0 to 8
            # Response goes from 0 to approx 1.2
            response = 1.0 - np.exp(-0.6 * t_val) * np.cos(4.0 * t_val)
            
            # Map values to screen coordinates
            px = origin[0] + t_val
            py = origin[1] + (response * target_height)
            curve_points.append([px, py, 0])

        motor_curve = VMobject(color=BLUE)
        motor_curve.set_points_as_corners(curve_points)
        self.play(Create(motor_curve), run_time=3)

        # 4. Indicators (Using Line and Text as Brace is restricted)
        # Rise Time indicator (First time curve reaches setpoint)
        # Occurs roughly at t = 0.45
        rt_x_coord = origin[0] + 0.45
        rise_line = Line([origin[0], origin[1] - 0.3, 0], [rt_x_coord, origin[1] - 0.3, 0], color=PURPLE)
        rise_text = Text("Rise Time", font_size=16, color=PURPLE).next_to(rise_line, DOWN, buff=0.1)
        
        self.play(Create(rise_line), Write(rise_text))
        self.wait(0.5)

        # Steady State Error indicator (difference at the end of the graph)
        final_x = origin[0] + 8.0
        final_y = curve_points[-1][1]
        error_line = Line([final_x, setpoint_y, 0], [final_x, final_y, 0], color=PURPLE)
        error_text = Text("Steady State Error", font_size=14, color=PURPLE).next_to(error_line, RIGHT, buff=0.1)
        
        self.play(Create(error_line), Write(error_text))
        
        self.wait(2)
