from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. SETUP UI COMPONENTS (Manual creation to avoid disallowed Axes/DashedLine)
        # --------------------------------------------------------------------------
        
        # Reference Formula
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=32).to_edge(UP, buff=0.2)
        
        # Input Box (Left side initially)
        input_rect = Rectangle(width=4.5, height=2.2, color=GREEN)
        input_title = Text("Desired Behavior", font_size=24, color=GREEN).move_to(input_rect.get_top() + DOWN * 0.4)
        input_list = VGroup(
            Text("• Rise Time", font_size=18),
            Text("• Steady State Error", font_size=18),
            Text("• Stability", font_size=18)
        ).arrange(DOWN, aligned_edge=LEFT, buff=0.2).next_to(input_title, DOWN, buff=0.3)
        input_group = VGroup(input_rect, input_title, input_list).shift(LEFT * 3.5 + UP * 1.5)

        # Output Box (Right side initially)
        output_rect = Rectangle(width=4.5, height=2.2, color=BLUE)
        output_title = Text("PID Constants", font_size=24, color=BLUE).move_to(output_rect.get_top() + DOWN * 0.4)
        output_params = MathTex("K_p, K_i, K_d", font_size=36, color=BLUE).move_to(output_rect.get_center() + DOWN * 0.2)
        output_group = VGroup(output_rect, output_title, output_params).shift(RIGHT * 3.5 + UP * 1.5)

        # Mapping Arrow
        arrow = Arrow(start=input_group.get_right(), end=output_group.get_left(), color=WHITE, stroke_width=4)
        
        # Combined Diagram for rotation
        diagram = VGroup(input_group, output_group, arrow)

        # 2. SETUP PLOT (Manual creation to avoid disallowed Axes)
        # --------------------------------------------------------
        
        plot_origin = DOWN * 3 + LEFT * 4
        x_axis = Line(plot_origin, plot_origin + RIGHT * 8, color=WHITE)
        y_axis = Line(plot_origin, plot_origin + UP * 2.5, color=WHITE)
        x_label = Text("Time", font_size=16).next_to(x_axis, RIGHT, buff=0.2)
        y_label = Text("Motor Velocity (RPM)", font_size=16).next_to(y_axis, UP, buff=0.2)
        
        # Manual Dashed Line for Setpoint
        target_height = 1.8
        setpoint = VGroup()
        for i in range(30):
            start_p = plot_origin + RIGHT * (i * 0.27) + UP * target_height
            end_p = plot_origin + RIGHT * (i * 0.27 + 0.15) + UP * target_height
            setpoint.add(Line(start_p, end_p, color=RED, stroke_width=2))
        
        # Motor curve: y = 1.8 * (1 - exp(-0.8t) * cos(4t))
        curve = VMobject(color=YELLOW)
        points = []
        for t_val in np.linspace(0, 7, 100):
            y_val = target_height * (1 - np.exp(-0.8 * t_val) * np.cos(4 * t_val))
            points.append(plot_origin + t_val * 1.1 * RIGHT + y_val * UP)
        curve.set_points_as_corners(points)

        # Manual Braced labels (since Brace/BraceBetweenPoints were undefined/disallowed)
        # Rise Time indicator
        rise_line = Line(plot_origin + DOWN * 0.2, plot_origin + RIGHT * 0.5 + DOWN * 0.2, color=WHITE)
        rise_tick_l = Line(rise_line.get_left() + UP * 0.1, rise_line.get_left() + DOWN * 0.1)
        rise_tick_r = Line(rise_line.get_right() + UP * 0.1, rise_line.get_right() + DOWN * 0.1)
        rise_label = Text("Rise Time", font_size=16).next_to(rise_line, DOWN, buff=0.1)
        
        # SSE indicator at the tail
        end_curve_point = points[-1]
        sse_line = Line(end_curve_point, plot_origin + RIGHT * (7 * 1.1) + UP * target_height, color=WHITE)
        sse_label = Text("Steady State Error", font_size=14).next_to(sse_line, RIGHT, buff=0.1)

        # 3. ANIMATION SEQUENCE
        # ---------------------

        # Step 1: Show initial relationship
        self.add(formula)
        self.play(Create(input_group), Create(output_group))
        self.play(GrowArrow(arrow))
        self.wait(1)

        # Step 2: "Flip the script" - Rotate the entire diagram
        # Using .animate.rotate as Rotate was reported as undefined
        self.play(diagram.animate.rotate(PI, about_point=diagram.get_center()), run_time=1.5)
        
        # Correct the orientation of the text so it's not upside down
        self.play(
            input_title.animate.rotate(PI),
            input_list.animate.rotate(PI),
            output_title.animate.rotate(PI),
            output_params.animate.rotate(PI)
        )
        self.wait(1)

        # Step 3: Animate the motor plot
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint))
        self.play(Create(curve), run_time=3)
        
        # Step 4: Show metrics
        self.play(Create(rise_line), Create(rise_tick_l), Create(rise_tick_r), Write(rise_label))
        self.play(Create(sse_line), Write(sse_label))
        
        self.wait(2)
