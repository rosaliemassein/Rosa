from manim import *
import numpy as np

class InverseMapping(Scene):
    def construct(self):
        # 1. Header Formula
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}").scale(0.8)
        formula.to_edge(UP, buff=0.3)
        self.add(formula)

        # 2. Split Screen Diagram (Manual approach to avoid disallowed Axes/Brace)
        # Input Section
        input_rect = Rectangle(width=4.2, height=2.4, color=BLUE)
        input_title = Text("Desired Behavior", font_size=22, color=BLUE).move_to(input_rect.get_top() + DOWN * 0.4)
        input_content = VGroup(
            Text("Rise Time", font_size=18),
            Text("Steady State Error", font_size=18),
            Text("Stability", font_size=18)
        ).arrange(DOWN, buff=0.15).next_to(input_title, DOWN, buff=0.3)
        input_group = VGroup(input_rect, input_title, input_content).shift(LEFT * 3 + UP * 0.5)

        # Output Section
        output_rect = Rectangle(width=4.2, height=2.4, color=GREEN)
        output_title = Text("PID Constants", font_size=22, color=GREEN).move_to(output_rect.get_top() + DOWN * 0.4)
        output_content = VGroup(
            MathTex("K_p", font_size=28),
            MathTex("K_i", font_size=28),
            MathTex("K_d", font_size=28)
        ).arrange(DOWN, buff=0.15).next_to(output_title, DOWN, buff=0.3)
        output_group = VGroup(output_rect, output_title, output_content).shift(RIGHT * 3 + UP * 0.5)

        # Large Arrow
        main_arrow = Line(input_group.get_right(), output_group.get_left(), buff=0.2).add_tip()
        
        diagram = VGroup(input_group, output_group, main_arrow)
        
        self.play(FadeIn(diagram))
        self.wait(1)

        # "Flipping the script" - Rotate the entire diagram
        self.play(diagram.animate.rotate(PI), run_time=1.5)
        self.wait(1)

        # 3. Motor Velocity Plot (Constructed with Lines to avoid Axes class)
        origin = DOWN * 2.8 + LEFT * 4
        x_axis = Line(origin, origin + RIGHT * 8, color=WHITE)
        y_axis = Line(origin, origin + UP * 2.2, color=WHITE)
        x_label = Text("Time", font_size=16).next_to(x_axis, RIGHT, buff=0.2)
        y_label = Text("RPM", font_size=16).next_to(y_axis, UP, buff=0.2)
        
        # Plot curve manually using a sequence of points
        curve_points = []
        for step in np.linspace(0, 7.5, 100):
            # y = 1.0 - exp(-0.5x) * cos(3x)
            rel_x = step
            rel_y = (1.0 - np.exp(-0.5 * step) * np.cos(3.0 * step)) * 1.3
            curve_points.append(origin + RIGHT * rel_x + UP * rel_y)
        
        motor_curve = VMobject(color=YELLOW).set_points_as_corners(curve_points)
        
        # Setpoint line (using opacity to simulate a lighter/dashed look)
        setpoint_line = Line(origin + UP * 1.3, origin + RIGHT * 7.5 + UP * 1.3, color=RED, stroke_width=2)
        setpoint_line.set_stroke(opacity=0.6)
        setpoint_label = Text("Setpoint", font_size=14, color=RED).next_to(setpoint_line, RIGHT, buff=0.1)

        # Manual indicators for labels (avoiding Brace class)
        # Rise Time marker (horizontal span)
        rise_marker = Line(origin + DOWN * 0.2, origin + RIGHT * 1.0 + DOWN * 0.2, color=BLUE)
        rise_tick1 = Line(origin + DOWN * 0.1, origin + DOWN * 0.3, color=BLUE)
        rise_tick2 = Line(origin + RIGHT * 1.0 + DOWN * 0.1, origin + RIGHT * 1.0 + DOWN * 0.3, color=BLUE)
        rise_text = Text("Rise Time", font_size=14, color=BLUE).next_to(rise_marker, DOWN, buff=0.1)
        rise_indicator = VGroup(rise_marker, rise_tick1, rise_tick2, rise_text)

        # Steady State Error marker (vertical gap at end)
        error_x = 7.0
        error_line = Line(origin + RIGHT * error_x + UP * 1.3, origin + RIGHT * error_x + UP * 1.15, color=ORANGE)
        error_text = Text("Steady State Error", font_size=14, color=ORANGE).next_to(error_line, RIGHT, buff=0.1)
        error_indicator = VGroup(error_line, error_text)

        # Animate plot section
        self.play(
            Create(x_axis), Create(y_axis), 
            Write(x_label), Write(y_label),
            Create(setpoint_line), Write(setpoint_label)
        )
        self.play(Create(motor_curve), run_time=2.5)
        self.play(FadeIn(rise_indicator), FadeIn(error_indicator))
        
        self.wait(2)
