import numpy as np
from manim import *

class LWRConcept(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6
        )
        self.play(Create(axes))

        # 2. Data points
        np.random.seed(42)
        x_vals = np.linspace(-4.5, 4.5, 30)
        y_vals = np.sin(x_vals) + np.random.normal(0, 0.2, 30)
        
        dots = VGroup()
        for x, y in zip(x_vals, y_vals):
            dots.add(Dot(axes.c2p(x, y), radius=0.08, color=GRAY))
        
        self.play(Create(dots))

        # 3. Query Point (tracker)
        # Using a Dot as a tracker in case ValueTracker is restricted
        tracker = Dot(axes.c2p(-4, np.sin(-4)), color=WHITE, radius=0.12)
        
        # 4. Weight Field (Radial Gradient Intuition)
        # We'll use a circle that changes size or just follows
        weight_circle = Circle(radius=1.5, color=BLUE, fill_opacity=0.1, stroke_width=1)
        weight_circle.move_to(tracker.get_center())

        self.play(Create(tracker), FadeIn(weight_circle))

        # 5. LWR Line calculation and updater
        def get_regression_line():
            q_x = axes.p2c(tracker.get_center())[0]
            tau = 0.7
            
            # Weighted Least Squares
            weights = np.exp(-(x_vals - q_x)**2 / (2 * tau**2))
            
            # W is diagonal matrix
            W = np.diag(weights)
            X = np.column_stack([np.ones(len(x_vals)), x_vals])
            Y = y_vals
            
            # Solve (X^T W X) beta = X^T W Y
            try:
                XtW = X.T @ W
                beta = np.linalg.solve(XtW @ X, XtW @ Y)
                
                # Draw local line segment
                x_start, x_end = q_x - 0.8, q_x + 0.8
                y_start = beta[0] + beta[1] * x_start
                y_end = beta[0] + beta[1] * x_end
                
                return Line(
                    axes.c2p(x_start, y_start),
                    axes.c2p(x_end, y_end),
                    color=YELLOW,
                    stroke_width=6
                )
            except:
                return Line(ORIGIN, ORIGIN)

        local_line = always_redraw(get_regression_line)

        # 6. Dot glowing updater
        def update_dots_color(mobs):
            q_x = axes.p2c(tracker.get_center())[0]
            tau = 0.7
            for i, dot in enumerate(mobs):
                dist = abs(x_vals[i] - q_x)
                weight = np.exp(-dist**2 / (2 * tau**2))
                # Manual color interpolation to be safe
                # Gray is roughly [0.5, 0.5, 0.5], Yellow is [1, 1, 0]
                c = interpolate_color(GRAY, YELLOW, weight)
                dot.set_color(c)

        dots.add_updater(update_dots_color)
        weight_circle.add_updater(lambda m: m.move_to(tracker.get_center()))

        self.add(local_line)
        
        # 7. Animation: Move query point along the curve
        # We simulate the motion by moving the tracker dot
        path = axes.plot(lambda x: np.sin(x), x_range=[-4, 4])
        
        self.play(
            MoveAlongPath(tracker, path),
            run_time=10,
            rate_func=linear
        )
        
        self.wait(1)
        
        # Cleanup updaters for safety
        dots.remove_updater(update_dots_color)
        self.wait()
