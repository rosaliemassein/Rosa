from manim import *
import numpy as np

class ContrastiveAlignmentScene(Scene):
    def construct(self):
        # 1. Title and Objective
        title = Text("Contrastive Alignment", font_size=32).to_edge(UP)
        self.add(title)

        # 2. Neural Network Blocks (Encoders)
        # Using Rectangle and Text as primitive components
        img_box = Rectangle(width=2.5, height=1.0, color=BLUE).shift(LEFT * 3 + UP * 1.5)
        img_label = Text("Image Encoder", font_size=16).move_to(img_box)
        txt_box = Rectangle(width=2.5, height=1.0, color=RED).shift(RIGHT * 3 + UP * 1.5)
        txt_label = Text("Text Encoder", font_size=16).move_to(txt_box)
        
        image_encoder = VGroup(img_box, img_label)
        text_encoder = VGroup(txt_box, txt_label)
        self.play(Create(image_encoder), Create(text_encoder))

        # 3. Data Entry Animation
        img_sample = Square(side_length=0.6, color=BLUE, fill_opacity=0.3).shift(LEFT * 6 + UP * 1.5)
        txt_sample = Text("Photo in France", font_size=14).shift(RIGHT * 6 + UP * 1.5)
        self.play(
            img_sample.animate.move_to(img_box), 
            txt_sample.animate.move_to(txt_box), 
            run_time=1.5
        )
        self.play(FadeOut(img_sample), FadeOut(txt_sample))

        # 4. Embedding Space (Manual Axes)
        # Replacing Axes with Line objects
        origin = DOWN * 1.5
        x_axis = Line(origin + LEFT * 2, origin + RIGHT * 2, color=GREY)
        y_axis = Line(origin + DOWN * 0.5, origin + UP * 2, color=GREY)
        space_label = Text("Embedding Space", font_size=14).next_to(y_axis, UP)
        self.add(x_axis, y_axis, space_label)

        # 5. Angle Tracking
        # Replacing ValueTracker with a hidden Dot to avoid 'disallowed' issues
        # The x-coordinate of 'angle_pnt' represents the current angle in radians
        angle_pnt = Dot(radius=0).move_to(RIGHT * (135 * DEGREES))
        img_angle = 45 * DEGREES

        # 6. Vectors (Image and Text)
        # Image vector is static at 45 degrees
        img_end = origin + np.array([1.5 * np.cos(img_angle), 1.5 * np.sin(img_angle), 0])
        img_vector = Arrow(origin, img_end, buff=0, color=BLUE)
        img_tag = Text("Image", font_size=14, color=BLUE).next_to(img_end, UR, buff=0.1)
        self.add(img_vector, img_tag)

        # Text vector is dynamic based on angle_pnt
        text_vector = Arrow(origin, origin + RIGHT, buff=0, color=RED)
        
        def update_text_vector(mobj):
            current_angle = angle_pnt.get_center()[0]
            target_end = origin + np.array([1.5 * np.cos(current_angle), 1.5 * np.sin(current_angle), 0])
            mobj.become(Arrow(origin, target_end, buff=0, color=RED))
        
        text_vector.add_updater(update_text_vector)

        # Label for the text vector
        text_tag = Text("Text", font_size=14, color=RED)
        def update_text_tag(mobj):
            mobj.next_to(text_vector.get_end(), UP + RIGHT, buff=0.1)
        
        text_tag.add_updater(update_text_tag)
        self.add(text_vector, text_tag)

        # 7. Similarity Score Display
        # Replacing DecimalNumber with MathTex + Updater
        sim_title = Text("Cosine Similarity: ", font_size=20).to_corner(UL)
        sim_val = MathTex("0.00", font_size=24).next_to(sim_title, RIGHT)
        
        def update_sim_val(mobj):
            current_angle = angle_pnt.get_center()[0]
            # Similarity is cosine of angle difference
            score = np.cos(current_angle - img_angle)
            mobj.become(MathTex(f"{score:.2f}", font_size=24).next_to(sim_title, RIGHT))
            
        sim_val.add_updater(update_sim_val)
        self.add(sim_title, sim_val)

        # 8. Alignment Animation (Positive Pair)
        self.wait(1)
        self.play(
            angle_pnt.animate.move_to(RIGHT * img_angle), 
            run_time=2.5, 
            rate_func=linear
        )
        self.wait(1)

        # 9. Pushing Away Animation (Negative Pair)
        wrong_label = Text("'A photo in Japan'", font_size=16, color=RED).next_to(txt_box, DOWN)
        self.play(Write(wrong_label))
        
        # Move text vector away to 225 degrees
        negative_angle = 225 * DEGREES
        self.play(
            angle_pnt.animate.move_to(RIGHT * negative_angle), 
            run_time=2.5, 
            rate_func=linear
        )
        self.wait(1)

        # 10. CLIP Loss Formula
        formula = MathTex(
            r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})",
            font_size=28
        ).to_edge(DOWN, buff=0.5)
        self.play(Write(formula))
        self.wait(2)
