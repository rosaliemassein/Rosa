from manim import *
import numpy as np

def linear_rate_func(t):
    return t

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Setup Manual Axes
        # Drawing axes with Line objects because Axes is restricted
        origin_pt = LEFT * 5 + DOWN * 2.5
        x_line = Line(origin_pt, origin_pt + RIGHT * 6, color=WHITE)
        y_line = Line(origin_pt, origin_pt + UP * 5, color=WHITE)
        
        # Axis Labels
        x_label = Text("Number of Players", font_size=20).next_to(x_line, DOWN)
        y_label = Text("Mean Squared Error", font_size=20).rotate(90 * DEGREES).next_to(y_line, LEFT)
        
        self.add(x_line, y_line, x_label, y_label)

        # 2. KNN Curve Construction (Red, dips then overfits)
        # Using a collection of points to create a smooth-looking curve
        knn_pts = []
        num_points = 41
        for i_val in range(num_points):
            t_param = i_val / 40.0
            x_coord = t_param * 5.5
            # Parabolic shape: starts high, dips, then rises sharply
            y_coord = 0.7 * (x_coord - 2.2)**2 + 0.8
            knn_pts.append(origin_pt + RIGHT * x_coord + UP * y_coord)
        
        knn_curve = VMobject().set_points_as_corners(knn_pts).set_color(RED)
        knn_tag = Text("K-Nearest Neighbors", color=RED, font_size=16).next_to(knn_pts[-1], UP)

        # 3. Lasso/Ridge Curve Construction (Green, steady decline)
        reg_pts = []
        for i_val in range(num_points):
            t_param = i_val / 40.0
            x_coord = t_param * 5.5
            # Exponential decay representing effective regularization
            y_coord = 4.5 * np.exp(-0.6 * x_coord) + 0.4
            reg_pts.append(origin_pt + RIGHT * x_coord + UP * y_coord)

        reg_curve = VMobject().set_points_as_corners(reg_pts).set_color(GREEN)
        reg_tag = Text("Lasso / Ridge", color=GREEN, font_size=16).next_to(reg_pts[-1], RIGHT)

        self.play(Create(knn_curve), Write(knn_tag))
        self.play(Create(reg_curve), Write(reg_tag))
        self.wait(1)

        # 4. Formula Display
        formula = MathTex(r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1")
        formula.scale(0.7).to_corner(UR, buff=0.5)
        self.play(Write(formula))

        # 5. Coefficient Shrinkage Visualization (Bar Chart)
        # Creating a group of 150 vertical lines to represent model coefficients
        num_bars = 150
        np.random.seed(42)
        initial_heights = np.random.uniform(0.1, 2.5, num_bars)
        
        bars = VGroup(*[
            Line(ORIGIN, UP * h, stroke_width=1, color=BLUE)
            for h in initial_heights
        ]).arrange(RIGHT, buff=0.01).shift(RIGHT * 3.5 + DOWN * 1.5)
        
        lambda_val_label = Text("lambda = 0.00", font_size=20).next_to(bars, DOWN, buff=0.4)
        
        self.add(bars, lambda_val_label)

        # Custom updater function for the animation
        # alpha_val represents the progression of the animation from 0 to 1
        def update_visualization(mobject_group, alpha_val):
            current_lambda = alpha_val * 2.5
            
            # Extract bars and text from the container VGroup
            bars_vgroup = mobject_group[0]
            label_mobject = mobject_group[1]
            
            # Logic for Lasso: Soft-thresholding (shrinking coefficients to zero)
            for idx in range(len(bars_vgroup)):
                # Calculate new height based on lambda penalty
                new_h = initial_heights[idx] - current_lambda
                if new_h < 0:
                    new_h = 0
                
                # Apply new height to the Line object
                # We use a very small value (0.0001) for height 0 to avoid scaling errors
                bars_vgroup[idx].set_height(max(new_h, 0.0001), stretch=True, about_edge=DOWN)
                
                # If the coefficient hits zero, make it invisible (vanishing)
                if new_h <= 0:
                    bars_vgroup[idx].set_stroke(opacity=0)
                else:
                    bars_vgroup[idx].set_stroke(opacity=1)
            
            # Update the numerical string for lambda
            text_string = "lambda = " + "{:.2f}".format(current_lambda)
            updated_text = Text(text_string, font_size=20).next_to(bars_vgroup, DOWN, buff=0.4)
            label_mobject.become(updated_text)

        # 6. Execute shrinkage animation
        viz_container = VGroup(bars, lambda_val_label)
        self.play(
            UpdateFromAlphaFunc(viz_container, update_visualization),
            run_time=8,
            rate_func=linear_rate_func
        )
        self.wait(2)
