from manim import *

class FeatureMatrixAnimation(Scene):
    def construct(self):
        # 1. Setup Grid (10 rows x 15 columns)
        # Using a flat list for easier indexing
        grid_cells = [Square(side_length=0.3, stroke_width=1) for _ in range(150)]
        grid = VGroup(*grid_cells)
        grid.arrange_in_grid(rows=10, cols=15, buff=0.05)
        grid.shift(RIGHT * 1.5)
        grid.set_stroke(GRAY, opacity=0.5)

        # 2. Setup Player Silhouettes (15 silhouettes)
        def get_silhouette():
            head = Circle(radius=0.1, stroke_width=2)
            body = Line(DOWN * 0.1, DOWN * 0.35, stroke_width=2)
            arms = Line(LEFT * 0.15 + DOWN * 0.15, RIGHT * 0.15 + DOWN * 0.15, stroke_width=2)
            legs = VGroup(
                Line(DOWN * 0.35, LEFT * 0.1 + DOWN * 0.5, stroke_width=2),
                Line(DOWN * 0.35, RIGHT * 0.1 + DOWN * 0.5, stroke_width=2)
            )
            return VGroup(head, body, arms, legs)

        players = VGroup(*[get_silhouette() for _ in range(15)])
        players.arrange(DOWN, buff=0.1).to_edge(LEFT, buff=0.5)

        # 3. Animate Creation
        self.play(LaggedStart(*[Create(p) for p in players], lag_ratio=0.05), run_time=1.5)
        self.wait(0.5)

        # 4. 'Minutes Played' filter scan
        filter_rect = Rectangle(height=0.5, width=1.2, color=YELLOW, stroke_width=3)
        filter_rect.move_to(players[0])
        
        self.play(Create(filter_rect))
        
        for i in range(15):
            target_color = GREEN if i < 10 else RED
            self.play(
                filter_rect.animate.move_to(players[i]),
                players[i].animate.set_color(target_color),
                run_time=0.15
            )
        
        self.play(FadeOut(filter_rect))
        self.wait(0.5)

        # 5. Top 10 players move to the grid rows
        # Move players to align with the start of each row
        row_alignments = []
        for i in range(10):
            row_alignments.append(players[i].animate.scale(0.6).move_to(grid[i*15].get_left() + LEFT * 0.5))

        self.play(
            *row_alignments,
            FadeOut(players[10:]),
            Create(grid)
        )
        self.wait(0.5)

        # 6. Column Labels
        metrics_list = ["PTS", "REB", "AST", "STL", "BLK", "TOV", "PF", "FG%", "FT%", "3P%", "PIE", "MIN", "ORB", "DRB", "+/-"]
        labels = VGroup(*[Text(m, font_size=10) for m in metrics_list])
        for i, label in enumerate(labels):
            label.next_to(grid[i], UP, buff=0.2)
        
        self.play(Write(labels))
        self.wait(0.5)

        # 7. Fill grid cells with intensities
        fill_anims = []
        for i in range(150):
            # Using a pseudo-random value based on position
            val = (np.sin(i) + 1) / 2.0 
            fill_anims.append(grid[i].animate.set_fill(BLUE, opacity=val))
        
        self.play(LaggedStart(*fill_anims, lag_ratio=0.005), run_time=1.5)
        self.wait(1)

        # 8. Collapse to single horizontal vector
        formula = MathTex("n = m \\times p").to_edge(UP, buff=0.5)
        self.play(Write(formula))

        # Re-arrange all grid cells into a single row
        # Scaling down significantly to fit 150 items
        self.play(
            grid.animate.arrange(RIGHT, buff=0.01).scale(0.15).move_to(DOWN * 1.5),
            FadeOut(labels),
            FadeOut(players[:10]),
            run_time=2
        )

        # Custom visual instead of Brace to ensure compatibility
        indicator_line = Line(grid.get_corner(DL) + DOWN*0.2, grid.get_corner(DR) + DOWN*0.2)
        tick_l = Line(indicator_line.get_start() + UP*0.1, indicator_line.get_start() + DOWN*0.1)
        tick_r = Line(indicator_line.get_end() + UP*0.1, indicator_line.get_end() + DOWN*0.1)
        vector_label = Text("150-element Input Vector (m x p)", font_size=20).next_to(indicator_line, DOWN, buff=0.3)
        
        self.play(
            Create(indicator_line),
            Create(tick_l),
            Create(tick_r),
            Write(vector_label)
        )
        self.wait(2)
