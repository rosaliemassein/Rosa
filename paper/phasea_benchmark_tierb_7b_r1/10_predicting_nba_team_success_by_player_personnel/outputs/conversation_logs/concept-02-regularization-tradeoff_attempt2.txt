from manim import *
import numpy as np

class Concept02RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Manual Axis construction to avoid disallowed Axes class
        orig_p = LEFT * 5 + DOWN * 2
        h_line = Line(orig_p, orig_p + RIGHT * 5)
        v_line = Line(orig_p, orig_p + UP * 5)
        h_lbl = Text("Number of Players", font_size=20).next_to(h_line, DOWN)
        v_lbl = Text("Mean Squared Error", font_size=20).rotate(90 * DEGREES).next_to(v_line, LEFT)
        self.add(h_line, v_line, h_lbl, v_lbl)

        # 2. Manual KNN Curve construction (Dips then rises sharply)
        knn_pts_list = []
        for i_v in range(51):
            t_c = 0.1 + i_v * (4.5 / 50)
            y_c = 0.5 * (t_c - 2)**2 + 1
            knn_pts_list.append(orig_p + RIGHT * t_c + UP * y_c)
        knn_obj = VMobject(color=RED)
        knn_obj.set_points_as_corners(knn_pts_list)
        knn_label_obj = Text("KNN", color=RED, font_size=24).next_to(knn_pts_list[-1], UP)

        # 3. Manual Regularization Curve construction (Steady decline)
        reg_pts_list = []
        for i_v in range(51):
            t_c = 0.1 + i_v * (4.8 / 50)
            y_c = 4.5 / (t_c**0.5 + 0.5)
            reg_pts_list.append(orig_p + RIGHT * t_c + UP * y_c)
        reg_obj = VMobject(color=GREEN)
        reg_obj.set_points_as_corners(reg_pts_list)
        reg_label_obj = Text("Lasso/Ridge", color=GREEN, font_size=24).next_to(reg_pts_list[-1], RIGHT)

        self.play(Create(knn_obj), Write(knn_label_obj))
        self.play(Create(reg_obj), Write(reg_label_obj))

        # 4. Formula display
        formula_item = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            font_size=32
        ).to_edge(UP, buff=0.5)
        self.play(Write(formula_item))

        # 5. Coefficient bars construction (Visualizing penalty)
        bar_count = 60
        bars_start_p = RIGHT * 1.5 + DOWN * 2
        h_init = [np.random.uniform(0.5, 3.5) for _ in range(bar_count)]
        rates_sh = [np.random.uniform(0.2, 1.8) for _ in range(bar_count)]
        
        bars_group_obj = VGroup()
        for i_idx in range(bar_count):
            bar_item = Line(
                bars_start_p + RIGHT * i_idx * 0.08,
                bars_start_p + RIGHT * i_idx * 0.08 + UP * h_init[i_idx],
                stroke_width=2,
                color=BLUE
            )
            bars_group_obj.add(bar_item)
        
        bars_title_obj = Text("Coefficients", font_size=24).next_to(bars_group_obj, UP)
        self.play(Create(bars_group_obj), Write(bars_title_obj))

        # 6. Logic for shrinking bars using a Dot as a tracker
        track_dot_obj = Dot(radius=0).move_to(ORIGIN)
        def update_bars_process(m_v):
            # Using X coordinate of dot as a custom tracker value
            current_lambda_val = track_dot_obj.get_center()[0]
            for i_idx, b_item in enumerate(m_v):
                h_new_val = h_init[i_idx] - current_lambda_val * rates_sh[i_idx]
                if h_new_val < 0: h_new_val = 0
                p_s = bars_start_p + RIGHT * i_idx * 0.08
                p_e = p_s + UP * h_new_val
                b_item.put_start_and_end_on(p_s, p_e)
                # Visual feedback for feature selection (vanishing coefficients)
                if h_new_val <= 0:
                    b_item.set_color(GRAY_E)
                else:
                    b_item.set_color(BLUE)

        bars_group_obj.add_updater(update_bars_process)
        label_lambda_info = Text("Increasing Lambda...", font_size=24).next_to(bars_group_obj, DOWN, buff=0.5)
        self.add(label_lambda_info)
        
        # Animate the shrinkage
        self.play(track_dot_obj.animate.shift(RIGHT * 3), run_time=5)
        self.wait(2)
        
        # Cleanup
        self.play(FadeOut(VGroup(
            h_line, v_line, h_lbl, v_lbl, 
            knn_obj, knn_label_obj, reg_obj, reg_label_obj, 
            formula_item, bars_group_obj, bars_title_obj, label_lambda_info
        )))
        self.wait(1)
