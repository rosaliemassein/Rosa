from manim import *
import numpy as np

class TeamMatrixAnimation(Scene):
    def construct(self):
        # 1. Create Silhouette function and 15 player silhouettes
        def get_silhouette():
            head = Circle(radius=0.12, fill_opacity=1)
            body = Polygon(
                [-0.15, -0.1, 0], [0.15, -0.1, 0], 
                [0.2, -0.5, 0], [-0.2, -0.5, 0],
                fill_opacity=1
            ).next_to(head, DOWN, buff=0.05)
            return VGroup(head, body)

        players = VGroup(*[get_silhouette() for _ in range(15)])
        players.arrange_in_grid(rows=5, cols=3, buff=0.4).to_edge(LEFT, buff=1)
        player_labels = VGroup(*[
            Text(f"P{i+1}", font_size=14).next_to(players[i], DOWN, buff=0.05) 
            for i in range(15)
        ])
        
        player_group = VGroup(players, player_labels)
        self.play(Create(player_group))
        self.wait(0.5)

        # 2. Minutes Played Filter - Highlight top 10
        self.play(
            players[:10].animate.set_color(BLUE),
            players[10:].animate.set_opacity(0.2),
            player_labels[10:].animate.set_opacity(0.2)
        )
        self.wait(1)

        # 3. Create 10x15 grid of squares
        grid = VGroup(*[
            VGroup(*[
                Square(side_length=0.35, stroke_width=1) for _ in range(15)
            ]).arrange(RIGHT, buff=0.02)
            for _ in range(10)
        ]).arrange(DOWN, buff=0.02).shift(RIGHT * 2.5)

        # 4. Slide top 10 players into rows of the grid
        self.play(
            LaggedStart(*[
                players[i].animate.scale(0.6).next_to(grid[i], LEFT, buff=0.2)
                for i in range(10)
            ]),
            FadeOut(players[10:]),
            FadeOut(player_labels),
            Create(grid)
        )
        self.wait(0.5)

        # 5. Column Labels
        stat_names = ["PTS", "REB", "AST", "STL", "BLK", "TOV", "FGM", "FGA", "3PM", "3PA", "FTM", "FTA", "OREB", "DREB", "PIE"]
        labels = VGroup(*[
            Text(stat_names[i], font_size=12).next_to(grid[0][i], UP, buff=0.2).rotate(45*DEGREES)
            for i in range(15)
        ])
        self.play(Write(labels))

        # 6. Fill grid cells with varying intensities
        fill_anims = []
        for row in grid:
            for cell in row:
                intensity = np.random.uniform(0.1, 0.9)
                fill_anims.append(cell.animate.set_fill(BLUE, opacity=intensity))
        
        self.play(LaggedStart(*fill_anims, lag_ratio=0.005), run_time=2)
        self.wait(1)

        # 7. Collapse grid into a single long horizontal vector
        all_cells = VGroup(*[cell for row in grid for cell in row])
        formula = MathTex("n = m \\times p", color=YELLOW).to_edge(UP)
        
        self.play(
            FadeOut(players[:10]),
            FadeOut(labels),
            Write(formula)
        )
        
        # Transform the 2D grid structure into a 1D vector line
        self.play(
            all_cells.animate.arrange(RIGHT, buff=0.01).set_width(13.5).move_to(ORIGIN),
            run_time=2.5,
            rate_func=smooth
        )
        
        # Final highlight
        self.play(all_cells.animate.set_color(YELLOW), run_time=0.5)
        self.play(all_cells.animate.set_color(BLUE), run_time=0.5)
        
        self.wait(2)
