from manim import *

class SoftmaxAnimation(Scene):
    def construct(self):
        # 1. Input Image (Using a labeled rectangle as a placeholder for the plastic bottle image)
        input_image = Rectangle(height=2.5, width=2, color=BLUE)
        image_text = Text("Plastic Bottle", font_size=20).move_to(input_image.get_center())
        input_group = VGroup(input_image, image_text).to_edge(LEFT, buff=0.5)
        
        self.play(FadeIn(input_group))
        
        # 2. CNN Layers (Represented by a series of vertical rectangles)
        # Use hex codes to avoid identifier issues with specific color names
        cnn_layers = VGroup(*[
            Rectangle(width=0.4, height=2.0, fill_opacity=0.6, color="#555555", stroke_color=WHITE)
            for _ in range(4)
        ]).arrange(RIGHT, buff=0.15).next_to(input_group, RIGHT, buff=0.8)
        
        cnn_label = Text("CNN Layers", font_size=20).next_to(cnn_layers, UP)
        arrow1 = Arrow(input_group.get_right(), cnn_layers.get_left(), buff=0.1)
        
        self.play(Create(arrow1), Create(cnn_layers), Write(cnn_label))
        
        # 3. Column Vector of 20 raw values (x_i)
        # We represent this as a mathematical vector with ellipsis
        raw_vector = VGroup(
            MathTex("x_1"),
            MathTex("x_2"),
            MathTex(r"\vdots"),
            MathTex("x_{20}")
        ).arrange(DOWN, buff=0.25).scale(0.8).next_to(cnn_layers, RIGHT, buff=1.0)
        
        vector_brackets = SurroundingRectangle(raw_vector, color=WHITE, buff=0.2)
        vector_label = MathTex("x_i").next_to(vector_brackets, UP)
        arrow2 = Arrow(cnn_layers.get_right(), vector_brackets.get_left(), buff=0.1)
        
        self.play(Create(arrow2), Create(raw_vector), Create(vector_brackets), Write(vector_label))
        self.wait(1)
        
        # 4. Softmax Formula appearing
        formula = MathTex(
            r"softmax(x_{i})=\frac{e^{x_{i}}}{\sum_{j=1}^{20}e^{x_{j}}}",
            font_size=34
        ).to_edge(UP, buff=0.4).shift(RIGHT * 2)
        
        self.play(Write(formula))
        self.wait(1)
        
        # 5. Transform raw values into a Horizontal Bar Chart
        # We build the chart manually to ensure compatibility
        axes = Axes(
            x_range=[0, 1, 0.5],
            y_range=[0, 20, 5],
            x_length=4,
            y_length=4.5,
            axis_config={"include_tip": False, "font_size": 18}
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)
        
        # Probabilities: mostly low, one high peak for "Plastic Bottle"
        probs = [0.03] * 20
        target_idx = 14 # Index for the highest bar
        probs[target_idx] = 0.85
        
        bars = VGroup()
        for i in range(20):
            bar_w = axes.x_axis.get_unit_size() * probs[i]
            bar_h = axes.y_axis.get_unit_size() * 0.7
            # Create rectangle and position its left edge on the Y-axis
            bar = Rectangle(
                width=bar_w,
                height=bar_h,
                fill_opacity=0.8,
                color=BLUE,
                stroke_width=1
            )
            bar.move_to(axes.c2p(probs[i]/2, i + 0.5))
            bars.add(bar)
            
        # Animation: Transition from raw values to the chart
        self.play(
            FadeOut(raw_vector),
            FadeOut(vector_brackets),
            FadeOut(vector_label),
            FadeOut(arrow2),
            Create(axes),
            Create(bars)
        )
        
        # 6. Indicate the highest bar and label it
        max_bar = bars[target_idx]
        self.play(Indicate(max_bar, color=YELLOW, scale_factor=1.2))
        
        result_label = Text("Plastic Bottle", color=YELLOW, font_size=24).next_to(max_bar, RIGHT, buff=0.2)
        self.play(Write(result_label))
        
        self.wait(2)
