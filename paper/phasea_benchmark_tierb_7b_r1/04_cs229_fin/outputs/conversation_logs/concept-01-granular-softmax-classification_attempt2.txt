from manim import *

class GranularSoftmaxClassification(Scene):
    def construct(self):
        # 1. Input Image (Plastic Bottle)
        # We'll use a Square with text as a fallback if the image is missing.
        try:
            plastic_bottle = ImageMobject("img-12.jpeg").scale(1.2)
        except:
            plastic_bottle = VGroup(
                Rectangle(height=2, width=1.5, color=WHITE),
                Text("Bottle", font_size=20)
            )
        plastic_bottle.to_edge(LEFT, buff=0.5)

        # 2. CNN Layers
        cnn_layers = VGroup(*[
            Rectangle(height=h, width=0.3, fill_opacity=0.5, color=BLUE)
            for h in [3.0, 2.4, 1.8, 1.2]
        ]).arrange(RIGHT, buff=0.3).next_to(plastic_bottle, RIGHT, buff=0.8)

        # 3. Data flow arrows
        arrow1 = Arrow(plastic_bottle.get_right(), cnn_layers.get_left(), buff=0.1)

        # 4. Raw Values Vector (Column vector of 20 x_i values)
        raw_values_boxes = VGroup(*[
            Rectangle(height=0.15, width=0.4, color=BLUE, fill_opacity=0.3)
            for _ in range(20)
        ]).arrange(DOWN, buff=0.05)
        
        x_i_label = MathTex(r"x_i", font_size=30).next_to(raw_values_boxes, LEFT, buff=0.2)
        raw_vector_group = VGroup(raw_values_boxes, x_i_label).next_to(cnn_layers, RIGHT, buff=0.8)
        
        arrow2 = Arrow(cnn_layers.get_right(), raw_vector_group.get_left(), buff=0.1)

        # 5. Softmax Formula
        softmax_eq = MathTex(
            r"\text{softmax}(x_{i}) = \frac{e^{x_{i}}}{\sum_{j=1}^{20}e^{x_{j}}}",
            font_size=36
        ).to_edge(UP, buff=0.5)

        # 6. Manual Horizontal Bar Chart Implementation
        # Using Axes and Rectangles to ensure compatibility without BarChart class
        axes = Axes(
            x_range=[0, 1, 0.5],
            y_range=[0, 21, 1],
            x_length=5,
            y_length=5,
            tips=False,
            axis_config={"include_numbers": False}
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)

        probs = [0.03] * 20
        probs[14] = 0.85 # The "Plastic Bottle" category index
        
        bars = VGroup()
        for i, p in enumerate(probs):
            bar = Rectangle(
                width=p * 4.5, # Scale width based on probability
                height=0.15,
                fill_color=BLUE,
                fill_opacity=0.8,
                stroke_width=1
            )
            # Align bar to the Y-axis of the axes
            bar.align_to(axes.c2p(0, i + 1, 0), LEFT)
            bar.shift(UP * 0.05) # Centering tweak
            bars.add(bar)

        chart_group = VGroup(axes, bars)

        # --- ANIMATION SEQUENCE ---

        # Step 1: Input and CNN
        self.play(FadeIn(plastic_bottle))
        self.play(Create(arrow1), Create(cnn_layers))
        self.wait(1)

        # Step 2: Extracting raw values
        self.play(Create(arrow2), FadeIn(raw_vector_group))
        self.wait(1)

        # Step 3: Softmax formula appearance
        self.play(Write(softmax_eq))
        self.wait(1)

        # Step 4: Transform to Probability Distribution (Bar Chart)
        # Move objects to make room for chart
        self.play(
            FadeOut(plastic_bottle),
            FadeOut(cnn_layers),
            FadeOut(arrow1),
            FadeOut(arrow2),
            raw_vector_group.animate.scale(0.7).to_edge(LEFT, buff=1)
        )
        
        self.play(Create(axes))
        self.play(Create(bars, lag_ratio=0.1))
        self.wait(1)

        # Step 5: Highlight results
        max_bar = bars[14]
        self.play(Indicate(max_bar, color=YELLOW))
        
        label = Text("Plastic Bottle", font_size=24, color=YELLOW).next_to(max_bar, RIGHT, buff=0.2)
        self.play(Write(label))

        self.wait(2)
