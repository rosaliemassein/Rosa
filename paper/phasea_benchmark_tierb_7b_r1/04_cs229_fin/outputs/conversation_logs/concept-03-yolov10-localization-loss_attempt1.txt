from manim import *

class YOLOv10LocalizationLoss(Scene):
    def construct(self):
        # 1. Grid Construction (Manual to avoid NumberPlane restrictions)
        grid = VGroup()
        cell_size = 0.7
        n_cells = 7
        grid_side = n_cells * cell_size
        
        for i in range(n_cells + 1):
            # Horizontal lines
            grid.add(Line(
                start=[-grid_side/2, -grid_side/2 + i*cell_size, 0],
                end=[grid_side/2, -grid_side/2 + i*cell_size, 0],
                stroke_width=1, color=GRAY
            ))
            # Vertical lines
            grid.add(Line(
                start=[-grid_side/2 + i*cell_size, -grid_side/2, 0],
                end=[-grid_side/2 + i*cell_size, grid_side/2, 0],
                stroke_width=1, color=GRAY
            ))
        
        grid.to_edge(LEFT, buff=0.8)

        # 2. Image Placeholder and Grid Background
        img_placeholder = Rectangle(width=grid_side, height=grid_side, color=WHITE, fill_opacity=0.1).move_to(grid)
        try:
            image_mob = ImageMobject("img-11.jpeg").scale_to_fit_width(grid_side).move_to(grid)
            self.add(image_mob)
        except:
            pass
        self.add(img_placeholder, grid)

        # 3. Highlight Target Cell (e.g., cell at 4th column, 4th row)
        cell_origin = grid.get_corner(DL)
        # Center of the cell at (3, 3) 0-indexed
        cell_center = cell_origin + RIGHT * (3.5 * cell_size) + UP * (3.5 * cell_size)
        target_cell = Rectangle(width=cell_size, height=cell_size, color=YELLOW, stroke_width=4).move_to(cell_center)
        soda_center_dot = Dot(cell_center, color=ORANGE, radius=0.04)

        # 4. Bounding Boxes
        # Ground Truth: Green, dashed
        gt_box = Rectangle(width=cell_size*2.2, height=cell_size*1.6, color=GREEN).set_stroke(dash_length=0.1)
        gt_box.move_to(cell_center + RIGHT * 0.15 + DOWN * 0.1)
        
        # Predicted: Red, solid
        pred_box = Rectangle(width=cell_size*2.8, height=cell_size*2.1, color=RED)
        pred_box.move_to(cell_center + LEFT * 0.2 + UP * 0.2)

        # 5. Coordinate and Dimension Difference Arrows
        # x, y distance arrows (from centers)
        c_gt = gt_box.get_center()
        c_pr = pred_box.get_center()
        arrow_x = Arrow(start=c_pr, end=[c_gt[0], c_pr[1], 0], color=YELLOW, buff=0, tip_length=0.1)
        arrow_y = Arrow(start=[c_gt[0], c_pr[1], 0], end=c_gt, color=YELLOW, buff=0, tip_length=0.1)
        
        # w, h distance arrows (from edges)
        arrow_w = Arrow(start=gt_box.get_right(), end=[pred_box.get_right()[0], gt_box.get_y(), 0], color=BLUE, buff=0, tip_length=0.1)
        arrow_h = Arrow(start=gt_box.get_top(), end=[gt_box.get_x(), pred_box.get_top()[1], 0], color=BLUE, buff=0, tip_length=0.1)

        # 6. Loss Formula (Split into parts for highlighting)
        formula = MathTex(
            r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right)",
            r"+",
            r"\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)",
            font_size=18
        ).to_edge(RIGHT, buff=0.3).shift(UP * 2)

        # 7. Animation Sequence
        self.play(Create(grid), FadeIn(img_placeholder))
        self.play(Create(target_cell), FadeIn(soda_center_dot))
        self.wait(0.5)
        
        self.play(Create(gt_box), Create(pred_box))
        self.play(Write(formula))
        self.wait(1)

        # Highlight (x, y) errors
        self.play(
            formula[0].animate.set_color(YELLOW),
            Create(arrow_x), 
            Create(arrow_y)
        )
        self.wait(1.5)

        # Highlight (w, h) errors
        self.play(
            formula[0].animate.set_color(WHITE),
            formula[2].animate.set_color(BLUE),
            FadeOut(arrow_x), FadeOut(arrow_y),
            Create(arrow_w), Create(arrow_h)
        )
        self.wait(1.5)

        # Mock training: Boxes align
        self.play(
            pred_box.animate.move_to(gt_box.get_center()).set_width(gt_box.get_width()).set_height(gt_box.get_height()),
            formula[2].animate.set_color(WHITE),
            FadeOut(arrow_w), FadeOut(arrow_h),
            run_time=2.5
        )
        self.wait(2)
