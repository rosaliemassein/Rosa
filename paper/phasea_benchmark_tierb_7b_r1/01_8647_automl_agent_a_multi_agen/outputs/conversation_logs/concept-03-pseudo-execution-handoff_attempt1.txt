from manim import *

class PseudoExecutionHandoff(Scene):
    def construct(self):
        # 1. Setup Layout (Replacing Title with Text)
        header = Text("Pseudo-Execution Handoff", font_size=36).to_edge(UP, buff=0.5)
        self.add(header)

        # Positioning Agents
        data_agent_label = Text("Data Agent (A_d)", color=BLUE, font_size=28).shift(UP * 2 + LEFT * 3.5)
        model_agent_label = Text("Model Agent (A_m)", color=GREEN, font_size=28).shift(DOWN * 1.5 + LEFT * 3.5)
        
        # Data block
        data_box = RoundedRectangle(corner_radius=0.1, width=3, height=1.2, color=BLUE).next_to(data_agent_label, RIGHT, buff=1)
        data_text = Text("Pseudo-Analysis", font_size=20).move_to(data_box.get_center())
        data_group = VGroup(data_box, data_text)

        # Model task block (starts small)
        model_box = RoundedRectangle(corner_radius=0.1, width=3, height=1.2, color=GREEN).next_to(model_agent_label, RIGHT, buff=1)
        model_text = Text("s_m (Task Block)", font_size=20).move_to(model_box.get_center())
        model_group = VGroup(model_box, model_text)

        self.play(FadeIn(data_agent_label), FadeIn(model_agent_label))
        self.play(Create(data_group), Create(model_group))
        self.wait(1)

        # 2. Data Agent processing and output O_d
        o_d_circle = Circle(radius=0.35, color=YELLOW, fill_opacity=0.5).move_to(data_box.get_center())
        o_d_label = MathTex("O_d", color=WHITE, font_size=30).move_to(o_d_circle.get_center())
        o_d_bundle = VGroup(o_d_circle, o_d_label)
        
        insight_label = Text("Data Insights", font_size=16, color=YELLOW).next_to(o_d_circle, RIGHT)

        self.play(Write(insight_label))
        self.play(FadeIn(o_d_bundle))
        self.wait(0.5)

        # 3. Handoff to Model Agent (Arrow carry)
        handoff_arrow = Arrow(data_box.get_bottom(), model_box.get_top(), buff=0.2, color=WHITE)
        
        self.play(Create(handoff_arrow))
        self.play(
            o_d_bundle.animate.move_to(model_box.get_center() + UP * 0.7),
            FadeOut(insight_label),
            run_time=1.5
        )

        # 4. Model Agent expansion (Visualizing Formula)
        # s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)
        formula = MathTex(
            "s_i^m = \\mathrm{PD}(R, \\mathcal{A}_m, \\mathbf{p}_i, O_i^d)",
            font_size=28
        ).move_to(model_box.get_center())

        expanded_model_box = RoundedRectangle(corner_radius=0.1, width=5.5, height=1.8, color=GREEN).move_to(model_box.get_center())

        self.play(
            ReplacementTransform(model_text, formula),
            ReplacementTransform(model_box, expanded_model_box),
            o_d_bundle.animate.scale(0.5).next_to(expanded_model_box, RIGHT, buff=0.2),
            FadeOut(handoff_arrow),
            run_time=1.2
        )
        self.wait(1)

        # 5. Search Efficiency Bar (Without ValueTracker/always_redraw)
        efficiency_label = Text("Search Efficiency:", font_size=24).to_edge(DOWN, buff=0.8).shift(LEFT * 2.5)

        # We start with a tiny width bar
        bar_fill = Rectangle(
            width=0.05, 
            height=0.3, 
            fill_color=GOLD, 
            fill_opacity=1, 
            stroke_width=0
        ).align_to(bar_bg, LEFT).shift(RIGHT * 0.05)
        
        self.add(efficiency_label, bar_bg, bar_fill)

        perc_50 = Text("50%", font_size=20).next_to(bar_bg, RIGHT)
        perc_95 = Text("95%", font_size=20).next_to(bar_bg, RIGHT)
        
        self.add(perc_0)
