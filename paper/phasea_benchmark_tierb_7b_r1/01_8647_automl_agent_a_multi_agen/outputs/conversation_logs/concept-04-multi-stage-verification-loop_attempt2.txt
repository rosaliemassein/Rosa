from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Configuration & Positions
        planning_pos = LEFT * 5
        gate1_pos = LEFT * 2
        gate2_pos = ORIGIN
        gate3_pos = RIGHT * 2
        deploy_pos = RIGHT * 5

        # 2. UI Elements (Gates, Labels, Icon)
        def create_gate(label_text, pos):
            line = Line(UP * 1.2, DOWN * 1.2, color=GRAY_A).move_to(pos)
            label = Text(label_text, font_size=18).next_to(line, UP)
            return VGroup(line, label)

        req_ver = create_gate("ReqVer", gate1_pos)
        exec_ver = create_gate("ExecVer", gate2_pos)
        imp_ver = create_gate("ImpVer", gate3_pos)
        
        planning_label = Text("Planning", font_size=24, color=BLUE).move_to(planning_pos + UP * 1.8)
        
        # Using a VGroup placeholder for the deployment icon to avoid file-loading errors
        deploy_icon = VGroup(
            RoundedRectangle(height=1.2, width=1.8, corner_radius=0.2, color=GOLD),
            Text("DEPLOY", font_size=18, color=GOLD)
        ).move_to(deploy_pos)

        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=32
        ).to_edge(DOWN, buff=0.6)

        plan = Square(side_length=0.4, fill_opacity=1, color=WHITE).move_to(planning_pos)

        self.add(req_ver, exec_ver, imp_ver, planning_label, deploy_icon, formula)
        self.add(plan)

        # 3. Animation: Failure Loop
        # Move to Gate 1 (Success)
        self.play(plan.animate.move_to(gate1_pos), run_time=1)
        self.play(plan.animate.set_color(GREEN), run_time=0.2)
        
        # Move to Gate 2 (Failure)
        self.play(plan.animate.move_to(gate2_pos), run_time=1)
        self.play(plan.animate.set_color(RED), run_time=0.2)
        
        # Looping back using ArcBetweenPoints as a fallback for CurvedArrow
        loop_path = ArcBetweenPoints(
            start=gate2_pos + DOWN * 0.5,
            end=planning_pos + DOWN * 0.5,
            angle=PI / 3,
            color=RED
        ).add_tip()
        
        self.play(Create(loop_path))
        self.play(
            plan.animate.move_to(planning_pos).set_color(WHITE),
            FadeOut(loop_path),
            run_time=1.5
        )

        # 4. Animation: Successful Pass
        # Pass ReqVer
        self.play(plan.animate.move_to(gate1_pos), run_time=0.8)
        self.play(plan.animate.set_color(GREEN), run_time=0.2)
        
        # Pass ExecVer
        self.play(plan.animate.move_to(gate2_pos), run_time=0.8)
        self.play(plan.animate.set_color(GREEN), run_time=0.2)
        
        # Pass ImpVer
        self.play(plan.animate.move_to(gate3_pos), run_time=0.8)
        self.play(plan.animate.set_color(GREEN), run_time=0.2)
        
        # Reach Deployment
        self.play(plan.animate.move_to(deploy_pos), run_time=0.8)
        self.play(
            deploy_icon.animate.set_color(GREEN).scale(1.1),
            plan.animate.scale(0),
            run_time=0.5
        )
        
        success_msg = Text("Process Complete", font_size=20, color=GREEN).next_to(deploy_icon, DOWN)
        self.play(Write(success_msg))
        self.wait(2)
