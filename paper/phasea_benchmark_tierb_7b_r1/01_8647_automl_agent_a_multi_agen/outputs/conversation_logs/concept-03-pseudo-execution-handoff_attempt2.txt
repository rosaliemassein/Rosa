from manim import *

class Concept3PseudoExecutionHandoff(Scene):
    def construct(self):
        # 1. Setup Data and Model Agents
        data_label = Text("Data Agent (A_d)", font_size=24)
        data_box = Square(side_length=1.5, color=BLUE).next_to(data_label, UP)
        data_group = VGroup(data_box, data_label).shift(UP * 1.5 + LEFT * 3)

        model_label = Text("Model Agent (A_m)", font_size=24)
        model_box = Square(side_length=1.5, color=RED).next_to(model_label, DOWN)
        model_group = VGroup(model_box, model_label).shift(DOWN * 1.5 + LEFT * 3)

        # 2. Objects for Handoff
        data_block = Rectangle(height=0.8, width=1.2, fill_opacity=0.5, color=BLUE).move_to(data_box)
        insights = MathTex("O_d", color=YELLOW).move_to(data_box)
        
        # 3. Model side objects
        formula = MathTex(r"s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)", font_size=28).shift(RIGHT * 3 + UP * 1)
        
        # 4. Search Efficiency Bar (Avoid ValueTracker/always_redraw)
        eff_label = Text("Search Efficiency", font_size=24)
        eff_bg = Rectangle(width=4, height=0.3, color=WHITE)
        eff_fill = Rectangle(width=4, height=0.3, fill_color=GREEN, fill_opacity=0.8, stroke_width=0)
        
        # Position UI elements
        eff_ui = VGroup(eff_label, eff_bg).arrange(DOWN, buff=0.2)
        eff_group = VGroup(eff_ui, eff_fill).shift(RIGHT * 3 + DOWN * 1.5)
        eff_fill.move_to(eff_bg.get_left(), aligned_edge=LEFT)
        
        # Initial state: make the bar fill effectively invisible (very small scale)
        eff_fill.scale(0.001, about_edge=LEFT)

        # --- ANIMATION SEQUENCE ---

        # Introduction of Agents
        self.play(Create(data_group), Create(model_group))
        self.wait(0.5)

        # Data Processing (Pseudo-analysis)
        self.play(FadeIn(data_block))
        self.play(Write(insights))
        self.play(data_block.animate.set_fill(BLUE, opacity=0.8))
        self.wait(0.5)

        # Handoff Animation
        arrow = Arrow(data_box.get_bottom(), model_box.get_top(), color=YELLOW)
        self.play(Create(arrow))
        self.play(
            insights.animate.move_to(model_box.get_center()),
            FadeOut(arrow),
            run_time=1.5
        )

        # Model expansion (sub-tasks depending on data agent output)
        task_label = MathTex("s_m", color=WHITE, font_size=36).move_to(model_box.get_center())
        self.play(
            model_box.animate.scale(1.4).set_color(ORANGE),
            insights.animate.scale(0.7).next_to(task_label, RIGHT, buff=0.1),
            FadeIn(task_label),
            Write(formula)
        )
        self.wait(0.5)

        # Training-free search Efficiency Increase
        # Animate the bar width using standard scale animation (linear progress)
        self.play(FadeIn(eff_ui), FadeIn(eff_fill))
        self.play(
            eff_fill.animate.scale(1000, about_edge=LEFT),
            run_time=3,
            rate_func=lambda t: t
        )

        self.wait(2)
