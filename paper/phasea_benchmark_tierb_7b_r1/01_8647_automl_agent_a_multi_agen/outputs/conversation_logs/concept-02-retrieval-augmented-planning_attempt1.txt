import numpy as np
from manim import *

class RetrievalAugmentedPlanning(Scene):
    def construct(self):
        # 1. Icons and Labels
        # Agent Manager (Central)
        am_box = RoundedRectangle(height=1.5, width=2.5, color=BLUE, fill_opacity=0.2)
        am_text = Text("Agent Manager", font_size=24)
        agent_manager = VGroup(am_box, am_text).center()

        # Knowledge Base (Left Cloud)
        kb_cloud = VGroup(
            Circle(radius=0.5).shift(LEFT * 0.4),
            Circle(radius=0.6).shift(UP * 0.2),
            Circle(radius=0.5).shift(RIGHT * 0.4),
            Circle(radius=0.4).shift(DOWN * 0.1)
        ).set_fill(WHITE, opacity=0.3).set_stroke(WHITE, 2)
        kb_text = Text("Knowledge\nBase", font_size=20)
        knowledge_base = VGroup(kb_cloud, kb_text).to_edge(LEFT, buff=1)

        # Formula (Bottom)
        formula = MathTex(r"\mathbf{P} = \mathcal{A}_{mgr}(\mathrm{RAP}(R))", font_size=36).to_edge(DOWN, buff=0.8)

        # Plan Paths (Right side destinations)
        # Plan A: Transformer
        plan_a_icon = Star(n=5, color=YELLOW, fill_opacity=0.8).scale(0.4)
        plan_a_label = Text("Plan A: Transformer", font_size=20)
        plan_a = VGroup(plan_a_icon, plan_a_label).arrange(RIGHT, buff=0.3).shift(RIGHT * 4 + UP * 2)

        # Plan B: CNN
        plan_b_icon = Square(side_length=0.7, color=GREEN, fill_opacity=0.8)
        plan_b_label = Text("Plan B: CNN", font_size=20)
        plan_b = VGroup(plan_b_icon, plan_b_label).arrange(RIGHT, buff=0.3).shift(RIGHT * 4)

        # Plan C: LSTM
        plan_c_icon = Triangle(color=RED, fill_opacity=0.8).scale(0.4)
        plan_c_label = Text("Plan C: LSTM", font_size=20)
        plan_c = VGroup(plan_c_icon, plan_c_label).arrange(RIGHT, buff=0.3).shift(RIGHT * 4 + DOWN * 2)

        # 2. Animation Sequence
        # Initial Fade In
        self.play(FadeIn(knowledge_base), FadeIn(agent_manager))
        self.wait(0.5)

        # Knowledge Retrieval (Arrows)
        arrow_top = Arrow(knowledge_base.get_top(), agent_manager.get_top() + LEFT * 0.5, color=YELLOW, buff=0.1)
        arrow_mid = Arrow(knowledge_base.get_right(), agent_manager.get_left(), color=YELLOW, buff=0.1)
        arrow_bot = Arrow(knowledge_base.get_bottom(), agent_manager.get_bottom() + LEFT * 0.5, color=YELLOW, buff=0.1)
        
        self.play(GrowArrow(arrow_mid))
        self.play(GrowArrow(arrow_top), GrowArrow(arrow_bot))
        self.play(agent_manager.animate.scale(1.1).set_color(BLUE_B), run_time=0.5)
        self.play(agent_manager.animate.scale(1/1.1).set_color(BLUE), run_time=0.5)
        self.wait(0.5)

        # Branching out into parallel plans
        # Create lines for the branch paths
        line_a = Line(agent_manager.get_right(), plan_a.get_left(), color=GRAY, stroke_width=2)
        line_b = Line(agent_manager.get_right(), plan_b.get_left(), color=GRAY, stroke_width=2)
        line_c = Line(agent_manager.get_right(), plan_c.get_left(), color=GRAY, stroke_width=2)

        # Create branches from Agent Manager
        am_copy_a = agent_manager.copy()
        am_copy_b = agent_manager.copy()
        am_copy_c = agent_manager.copy()

        self.play(
            Create(line_a), Create(line_b), Create(line_c),
            ReplacementTransform(am_copy_a, plan_a),
            ReplacementTransform(am_copy_b, plan_b),
            ReplacementTransform(am_copy_c, plan_c),
            run_time=2
        )
        self.wait(1)

        # Formula Reveal
        self.play(Write(formula))
        self.wait(3)

        # Final Clean up
        self.play(
            FadeOut(knowledge_base),
            FadeOut(agent_manager),
            FadeOut(arrow_top), FadeOut(arrow_mid), FadeOut(arrow_bot),
            FadeOut(line_a), FadeOut(line_b), FadeOut(line_c),
            FadeOut(plan_a), FadeOut(plan_b), FadeOut(plan_c),
            FadeOut(formula)
        )
        self.wait(1)
