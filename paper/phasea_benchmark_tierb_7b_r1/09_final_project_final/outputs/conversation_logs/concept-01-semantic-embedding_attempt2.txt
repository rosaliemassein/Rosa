from manim import *
import numpy as np

class SemanticEmbedding(Scene):
    def construct(self):
        # 1. Setup a coordinate plane to represent the semantic vector space
        plane = NumberPlane(
            x_range=[-7, 7, 1],
            y_range=[-4, 4, 1],
            background_line_style={
                "stroke_color": BLUE_E,
                "stroke_width": 1,
                "stroke_opacity": 0.2
            }
        )
        self.add(plane)

        # 2. Define coordinates for keywords
        pos_dl = np.array([2.5, 1.5, 0])
        pos_nn = np.array([3.5, 0.5, 0])
        pos_r = np.array([-3.5, -2.0, 0])

        # 3. Create initial floating text labels
        label_dl = MathTex(r"\text{Deep Learning}", font_size=32).move_to(pos_dl + UP * 0.5)
        label_nn = MathTex(r"\text{Neural Networks}", font_size=32).move_to(pos_nn + DOWN * 0.5)
        label_r = MathTex(r"\text{Robotics}", font_size=32).move_to(pos_r + UP * 0.5)

        # 4. Create Dot objects for the vector points
        dot_dl = Dot(point=pos_dl, color=GREEN)
        dot_nn = Dot(point=pos_nn, color=BLUE)
        dot_r = Dot(point=pos_r, color=RED)

        # 5. Animation: Show text, then transform to dots
        self.play(Write(label_dl), Write(label_nn), Write(label_r))
        self.wait(1)
        self.play(
            ReplacementTransform(label_dl, dot_dl),
            ReplacementTransform(label_nn, dot_nn),
            ReplacementTransform(label_r, dot_r)
        )
        self.wait(1)

        # 6. Distance line (Manual dashed implementation to avoid restricted symbols)
        dashed_line = VGroup()
        num_dashes = 15
        for i in range(num_dashes):
            start_p = pos_dl + (pos_nn - pos_dl) * (i / num_dashes)
            end_p = pos_dl + (pos_nn - pos_dl) * ((i + 0.5) / num_dashes)
            dashed_line.add(Line(start_p, end_p, color=YELLOW, stroke_width=2))
        
        self.play(Create(dashed_line))
        self.wait(1)
        self.play(FadeOut(dashed_line), FadeOut(dot_r))

        # 7. Formula display
        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}").to_edge(UP)
        bg_rect = SurroundingRectangle(formula, color=BLACK, fill_opacity=0.8).set_fill(BLACK)
        formula_group = VGroup(bg_rect, formula)
        self.play(FadeIn(formula_group))
        self.wait(1)

        # 8. Phrase Averaging: 'Machine' and 'Learning'
        pos_m = np.array([-2.5, -1.0, 0])
        pos_l = np.array([1.5, -1.0, 0])
        pos_avg = (pos_m + pos_l) / 2

        word_m = MathTex(r"\text{Machine}", font_size=30).next_to(pos_m, UP)
        dot_m = Dot(pos_m, color=ORANGE)
        word_l = MathTex(r"\text{Learning}", font_size=30).next_to(pos_l, UP)
        dot_l = Dot(pos_l, color=PURPLE)

        self.play(
            FadeIn(dot_m), Write(word_m),
            FadeIn(dot_l), Write(word_l)
        )
        self.wait(1)

        # Transform both into a single central vector (average)
        dot_avg = Dot(pos_avg, color=YELLOW)
        phrase_label = MathTex(r"\text{Machine Learning}", color=YELLOW, font_size=34).next_to(pos_avg, DOWN)

        self.play(
            ReplacementTransform(dot_m, dot_avg),
            ReplacementTransform(dot_l, dot_avg),
            ReplacementTransform(word_m, phrase_label),
            ReplacementTransform(word_l, phrase_label)
        )
        self.wait(2)
