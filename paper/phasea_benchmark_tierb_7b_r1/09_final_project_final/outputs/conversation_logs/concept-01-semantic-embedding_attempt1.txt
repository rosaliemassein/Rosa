from manim import *

class SemanticEmbedding(Scene):
    def construct(self):
        # 1. Setup 2D Axes (NumberPlane and 3D features are disallowed)
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-4, 4, 1],
            x_length=10,
            y_length=7,
            axis_config={"include_tip": True}
        )
        self.add(axes)

        # 2. Floating text labels
        dl_text = Text("Deep Learning", color=BLUE).scale(0.6).move_to(axes.c2p(2, 1))
        nn_text = Text("Neural Networks", color=GREEN).scale(0.6).move_to(axes.c2p(2.5, 0.3))
        rb_text = Text("Robotics", color=RED).scale(0.6).move_to(axes.c2p(-3, -2))

        self.play(Write(dl_text), Write(nn_text), Write(rb_text))
        self.wait(1)

        # 3. Animate text labels transforming into Dot objects
        dl_dot = Dot(axes.c2p(2, 1), color=BLUE)
        nn_dot = Dot(axes.c2p(2.5, 0.3), color=GREEN)
        rb_dot = Dot(axes.c2p(-3, -2), color=RED)

        self.play(
            ReplacementTransform(dl_text, dl_dot),
            ReplacementTransform(nn_text, nn_dot),
            ReplacementTransform(rb_text, rb_dot)
        )
        self.wait(1)

        # 4. Show distance between related concepts (using Line instead of DashedLine)
        dist_line = Line(dl_dot.get_center(), nn_dot.get_center(), color=YELLOW, stroke_width=2)
        dist_label = Text("Small Distance", color=YELLOW).scale(0.4).next_to(dist_line, UP, buff=0.1)
        
        self.play(Create(dist_line), Write(dist_label))
        self.wait(1)
        self.play(FadeOut(dist_line), FadeOut(dist_label))

        # 5. Transition to formula and averaging concept
        # Clear the board
        self.play(FadeOut(dl_dot), FadeOut(nn_dot), FadeOut(rb_dot))
        
        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}").to_edge(UP)
        self.play(Write(formula))
        self.wait(1)

        # 6. Show keywords as vectors
        # "Machine" vector
        m_pos = axes.c2p(1, 3)
        m_arrow = Arrow(axes.c2p(0, 0), m_pos, buff=0, color=BLUE)
        m_label = Text("Machine", color=BLUE).scale(0.5).next_to(m_pos, UR, buff=0.1)
        
        # "Learning" vector
        l_pos = axes.c2p(3, 1)
        l_arrow = Arrow(axes.c2p(0, 0), l_pos, buff=0, color=GREEN)
        l_label = Text("Learning", color=GREEN).scale(0.5).next_to(l_pos, UR, buff=0.1)

        self.play(Create(m_arrow), Write(m_label))
        self.play(Create(l_arrow), Write(l_label))
        self.wait(1)

        # 7. Merge them into a single central vector representing the average
        avg_pos = axes.c2p(2, 2)
        avg_arrow = Arrow(axes.c2p(0, 0), avg_pos, buff=0, color=YELLOW)
        phrase_label = Text("Machine Learning", color=YELLOW).scale(0.6).move_to(axes.c2p(2, 2.5))

        self.play(
            ReplacementTransform(VGroup(m_arrow, l_arrow), avg_arrow),
            ReplacementTransform(VGroup(m_label, l_label), phrase_label)
        )
        self.wait(2)

        # Cleanup
        self.play(FadeOut(VGroup(axes, formula, avg_arrow, phrase_label)))
        self.wait(1)
