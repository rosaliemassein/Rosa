from manim import *

class DominantZoneMechanism(Scene):
    def construct(self):
        # 1. Setup AHU and Zones
        # Use basic BLUE and hex for maximum compatibility
        ahu = Rectangle(width=3, height=1.5, color=BLUE, fill_opacity=0.8).to_edge(UP, buff=1)
        ahu_text = Text("AHU", font_size=24).move_to(ahu)
        ahu_group = VGroup(ahu, ahu_text)

        # 5 Zones starting as green
        zones = VGroup(*[
            Rectangle(width=1.8, height=1.2, color=GREEN, fill_opacity=0.8)
            for _ in range(5)
        ]).arrange(RIGHT, buff=0.4).to_edge(DOWN, buff=1)

        zone_labels = VGroup(*[
            Text(f"Zone {i+1}", font_size=18).move_to(zones[i])
            for i in range(5)
        ])

        self.add(ahu_group, zones, zone_labels)
        self.wait(1)

        # 2. Animate heat gradient growing in two zones (Zones 1 and 2)
        self.play(
            zones[0].animate.set_fill(ORANGE),
            zones[1].animate.set_fill(ORANGE),
            run_time=2
        )
        self.wait(0.5)

        # 3. Dashed arrows from orange zones to AHU labeled 'Cooling Request'
        # Fix: DashedLine with added tip replaces non-existent DashedArrow
        req_arrow1 = DashedLine(zones[0].get_top(), ahu.get_bottom(), color=ORANGE).add_tip()
        req_arrow2 = DashedLine(zones[1].get_top(), ahu.get_bottom(), color=ORANGE).add_tip()
        req_label = Text("Cooling Request", font_size=20, color=ORANGE).move_to(UP * 0.5)

        self.play(
            Create(req_arrow1), 
            Create(req_arrow2), 
            Write(req_label)
        )
        self.wait(1)

        # 4. As arrows hit the AHU, change its color to a deeper blue
        # Use hex for DEEP_BLUE/BLUE_E: #1C758A or standard DARK_BLUE
        self.play(
            ahu.animate.set_fill("#1C758A").set_color("#1C758A"),
            FadeOut(req_arrow1),
            FadeOut(req_arrow2),
            FadeOut(req_label),
            run_time=1
        )
        self.wait(0.5)

        # 5. Show deep blue 'Supply Air' arrows flowing back to ALL zones
        supply_arrows = VGroup(*[
            Arrow(ahu.get_bottom(), zones[i].get_top(), color="#1C758A", buff=0.1)
            for i in range(5)
        ])
        supply_label = Text("Supply Air (Colder)", font_size=20, color="#1C758A").next_to(ahu, DOWN, buff=0.2)

        self.play(
            Create(supply_arrows),
            Write(supply_label)
        )
        self.wait(1)

        # 6. Green zones turn a 'chilled' white-blue color to signify overcooling
        # Chilled color hex: #ADD8E6 (LightBlue)
        self.play(
            zones[2].animate.set_fill("#ADD8E6"),
            zones[3].animate.set_fill("#ADD8E6"),
            zones[4].animate.set_fill("#ADD8E6"),
            # The orange zones cool back down to green
            zones[0].animate.set_fill(GREEN),
            zones[1].animate.set_fill(GREEN),
            run_time=2
        )
        
        overcool_text = Text("Overcooled", font_size=20, color=WHITE).move_to(zones[3].get_center())
        self.play(Write(overcool_text))
        
        self.wait(3)
