from manim import *
import math

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Setup Coordinate System (Manual replacement for Axes/NumberPlane)
        # Shifted to the right to leave space for the UI
        center_x = 3
        axis_x = Line(LEFT, RIGHT).scale(2).shift(RIGHT * center_x)
        axis_y = Line(DOWN, UP).scale(1.5).shift(RIGHT * (center_x - 2))
        self.add(axis_x, axis_y)

        # 2. Sigmoid Curve
        # Building the curve using VMobject and manual point calculation
        sigmoid_path = VMobject(color=RED)
        points_list = []
        for i in range(-20, 21):
            coord_val = i / 10.0 # From -2.0 to 2.0
            prob_val = 1.0 / (1.0 + math.exp(-coord_val)) # Sigmoid formula
            # Map coord_val to axis_x and prob_val to axis_y
            points_list.append([center_x + coord_val, (prob_val * 3.0) - 1.5, 0])
        sigmoid_path.set_points_as_corners(points_list)
        self.add(sigmoid_path)

        # 3. Sliding Bar UI (Left side)
        # Bar 1: 1-Day T-CSP
        box_1 = Rectangle(height=2, width=0.5, color=WHITE).shift(LEFT * 4 + UP * 1)
        label_1 = MathTex("1-Day", font_size=24).next_to(box_1, LEFT)
        tracker_1 = Dot(box_1.get_bottom()).set_opacity(0) # Invisible proxy for value

        # Bar 2: 30-Day T-CSP
        box_2 = Rectangle(height=2, width=0.5, color=WHITE).shift(LEFT * 4 + DOWN * 1.5)
        label_2 = MathTex("30-Day", font_size=24).next_to(box_2, LEFT)
        tracker_2 = Dot(box_2.get_bottom()).set_opacity(0) # Invisible proxy for value

        self.add(box_1, label_1, box_2, label_2, tracker_1, tracker_2)

        # 4. Fill Updaters (Replacing always_redraw)
        fill_1 = VMobject().set_fill(BLUE, 0.8).set_stroke(width=0)
        def update_fill_1(mob):
            height_val = tracker_1.get_center()[1] - box_1.get_bottom()[1]
            height_val = max(0.01, min(height_val, 2.0))
            mob.become(
                Rectangle(height=height_val, width=0.5, fill_color=BLUE, fill_opacity=0.8, stroke_width=0)
                .move_to(box_1.get_bottom(), aligned_edge=DOWN)
            )
        fill_1.add_updater(update_fill_1)

        fill_2 = VMobject().set_fill(GREEN, 0.8).set_stroke(width=0)
        def update_fill_2(mob):
            height_val = tracker_2.get_center()[1] - box_2.get_bottom()[1]
            height_val = max(0.01, min(height_val, 2.0))
            mob.become(
                Rectangle(height=height_val, width=0.5, fill_color=GREEN, fill_opacity=0.8, stroke_width=0)
                .move_to(box_2.get_bottom(), aligned_edge=DOWN)
            )
        fill_2.add_updater(update_fill_2)
        
        self.add(fill_1, fill_2)

        # 5. Tracking Dot on Curve
        active_dot = Dot(color=YELLOW)
        def update_active_dot(mob):
            # Calculate input values (0.0 to 1.0)
            val_1 = (tracker_1.get_center()[1] - box_1.get_bottom()[1]) / 2.0
            val_2 = (tracker_2.get_center()[1] - box_2.get_bottom()[1]) / 2.0
            # Map combined input to the sigmoid x-range (-2 to 2)
            combined_input = (val_1 + val_2) * 2.0 - 2.0
            # Calculate sigmoid probability
            res_prob = 1.0 / (1.0 + math.exp(-combined_input))
            # Move dot to the corresponding point on the screen
            mob.move_to([center_x + combined_input, (res_prob * 3.0) - 1.5, 0])
        
        active_dot.add_updater(update_active_dot)
        self.add(active_dot)

        # 6. Animations
        self.play(tracker_1.animate.shift(UP * 1.5), run_time=1.5)
        self.wait(0.2)
        self.play(tracker_2.animate.shift(UP * 1.8), run_time=1.5)
        self.wait(0.2)
        self.play(
            tracker_1.animate.shift(DOWN * 1.0),
            tracker_2.animate.shift(DOWN * 0.5),
            run_time=2
        )

        # 7. Mathematical Formula Transformation
        formula_main = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=32
        ).to_edge(UP)
        
        self.play(Write(formula_main))
        self.wait(2)

        # Cleanup
        self.play(FadeOut(VGroup(
            axis_x, axis_y, sigmoid_path, active_dot,
            box_1, box_2, label_1, label_2, fill_1, fill_2, formula_main
        )))
