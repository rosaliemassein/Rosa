from manim import *

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Manual Coordinate System setup (Avoids Axes)
        origin_point = LEFT * 4 + DOWN * 2
        x_axis_line = Line(origin_point, origin_point + RIGHT * 6, color=WHITE)
        y_axis_line = Line(origin_point, origin_point + UP * 5, color=WHITE)
        
        # Labels using Text (more robust than MathTex for long strings)
        x_label = Text("Fraction of Zones Flagged", font_size=20).next_to(x_axis_line, DOWN)
        y_label = Text("Recall", font_size=20).rotate(90 * DEGREES).next_to(y_axis_line, LEFT)
        
        self.add(x_axis_line, y_axis_line, x_label, y_label)

        # 2. Reference Formula
        recall_formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_corner(UL)
        self.add(recall_formula)

        # 3. Create Curve using segments (Avoids Axes.plot and VMobject class keyword)
        # Using y = x^0.4 curve for concave "elbow" shape
        curve_points_list = []
        for i in range(51):
            alpha = i / 50.0
            x_coord = alpha * 6
            y_coord = (alpha**0.4) * 4.5
            curve_points_list.append(origin_point + RIGHT * x_coord + UP * y_coord)
            
        curve_path = VGroup(*[
            Line(curve_points_list[j], curve_points_list[j+1], color=BLUE, stroke_width=4)
            for j in range(len(curve_points_list) - 1)
        ])
        self.add(curve_path)

        # 4. Floorplan Visual (Zone Grid)
        zone_squares = VGroup(*[
            Square(side_length=0.4).set_fill(GRAY, opacity=0.2).set_stroke(WHITE, 1)
            for _ in range(16)
        ]).arrange_in_grid(4, 4, buff=0.1).shift(RIGHT * 3.5)
        
        floorplan_title = Text("Building Floorplan", font_size=22).next_to(zone_squares, UP)
        self.add(zone_squares, floorplan_title)

        # 5. Dynamic Tracking (Avoids ValueTracker)
        # We use a dummy dot's x-coordinate to drive the logic
        control_dot = Dot(radius=0.01).move_to(origin_point).set_opacity(0)
        moving_highlight = Dot(color=YELLOW)

        def scene_update_function(mobject_to_update):
            # Calculate current progress based on control_dot's movement
            current_x = control_dot.get_center()[0]
            progress_ratio = (current_x - origin_point[0]) / 6.0
            progress_ratio = max(0, min(1, progress_ratio))
            
            # Position highlight dot on the curve
            point_index = int(progress_ratio * 50)
            moving_highlight.move_to(curve_points_list[point_index])
            
            # Update grid square colors
            threshold_count = int(progress_ratio * 16)
            for index, square in enumerate(zone_squares):
                if index < threshold_count:
                    square.set_fill(RED, opacity=0.8)
                    square.set_stroke(RED, opacity=1)
                else:
                    square.set_fill(GRAY, opacity=0.2)
                    square.set_stroke(WHITE, opacity=1)

        # Attach updater to the highlight dot
        moving_highlight.add_updater(scene_update_function)
        self.add(control_dot, moving_highlight)

        # 6. Animation Sequence
        # Sweep from 0 to 100% (Too low threshold to too high)
        # Use lambda t: t for linear rate
        self.play(control_dot.animate.move_to(origin_point + RIGHT * 6), run_time=4, rate_func=lambda t: t)
        self.wait(0.5)
        
        # Sweep back to find the "Elbow" (roughly 25% across)
        self.play(control_dot.animate.move_to(origin_point + RIGHT * 1.5), run_time=2, rate_func=lambda t: t)
        self.wait(0.5)

        # 7. Final Highlight of the Elbow point
        elbow_marker = Circle(radius=0.5, color=GOLD).move_to(moving_highlight.get_center())
        elbow_caption = Text("The Elbow", color=GOLD, font_size=24).next_to(elbow_marker, UR)
        
        self.play(Create(elbow_marker), Write(elbow_caption))
        self.play(Flash(elbow_marker, color=GOLD, line_length=0.3))
        
        self.wait(3)
