from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Setup Axes and Sigmoid Curve
        # Using Axes as NumberPlane was disallowed in the previous log
        axes = Axes(
            x_range=[-4, 4, 1],
            y_range=[0, 1, 0.5],
            x_length=5,
            y_length=3,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5)
        
        sigmoid_graph = axes.plot(
            lambda x: 1 / (1 + np.exp(-x)),
            color=BLUE
        )
        
        axes_labels = axes.get_axis_labels(x_label="z", y_label="P(y=1)")
        
        # 2. Trackers for the features (1-day and 30-day memory)
        x1_tracker = ValueTracker(0)
        x30_tracker = ValueTracker(0)
        
        # 3. Create sliding bars on the left
        bar_width = 0.8
        
        bar1_label = MathTex("1\\text{-Day T-CSP}").scale(0.6).move_to(LEFT * 5.2 + UP * 2)
        bar30_label = MathTex("30\\text{-Day T-CSP}").scale(0.6).move_to(LEFT * 3.2 + UP * 2)
        
        # Base horizontal lines for bars
        base1 = Line(LEFT * 5.7, LEFT * 4.7).shift(DOWN * 1.5)
        base30 = Line(LEFT * 3.7, LEFT * 2.7).shift(DOWN * 1.5)
        
        # Visual bars (Rectangles) using updaters to change height
        bar1 = Rectangle(width=bar_width, height=0.1, fill_opacity=0.8, color=PINK, stroke_width=0)
        bar1.move_to(LEFT * 5.2 + DOWN * 1.5, aligned_edge=DOWN)
        
        bar30 = Rectangle(width=bar_width, height=0.1, fill_opacity=0.8, color=TEAL, stroke_width=0)
        bar30.move_to(LEFT * 3.2 + DOWN * 1.5, aligned_edge=DOWN)
        
        # 4. Dot on the sigmoid curve
        dot = Dot(color=RED)
        
        # Function to calculate z-value based on tracker values
        def get_current_z():
            # z = beta0 + beta1*x1 + beta2*x30. Assuming betas = 1 for visualization
            return x1_tracker.get_value() + x30_tracker.get_value()
        
        def get_current_p(z):
            return 1 / (1 + np.exp(-z))
            
        # Add updaters for reactive animation
        dot.add_updater(lambda d: d.move_to(axes.c2p(get_current_z(), get_current_p(get_current_z()))))
        
        # Stretch height based on trackers (offset by +2 to allow positive height for visualization)
        bar1.add_updater(lambda b: b.stretch_to_fit_height(max(0.1, x1_tracker.get_value() + 2), about_edge=DOWN))
        bar30.add_updater(lambda b: b.stretch_to_fit_height(max(0.1, x30_tracker.get_value() + 2), about_edge=DOWN))
        
        # 5. Formulas
        formula = MathTex(
            "P(y=1) = \\frac{1}{1 + e^{-(\\beta_0 + \\beta_1 X_{1d} + \\beta_{30} X_{30d})}}"
        ).scale(0.75).to_edge(DOWN, buff=0.4)
        
        formula_simple = MathTex(
            "P(y=1) = \\frac{1}{1 + e^{-z}}"
        ).scale(0.75).move_to(formula.get_center())
        
        # 6. Animation sequence
        self.add(axes, sigmoid_graph, axes_labels)
        self.add(bar1_label, bar30_label, base1, base30, bar1, bar30)
        self.add(dot)
        
        self.play(Write(formula))
        self.wait(1)
        
        # Animate bars sliding up
        self.play(
            x1_tracker.animate.set_value(1.5),
            x30_tracker.animate.set_value(0.5),
            run_time=2
        )
        self.wait(0.5)
        
        # Animate bars sliding down
        self.play(
            x1_tracker.animate.set_value(-1.5),
            x30_tracker.animate.set_value(-1.0),
            run_time=2
        )
        self.wait(0.5)
        
        # Transform formula to simplified version and move to target probability
        self.play(
            TransformMatchingTex(formula, formula_simple),
            x1_tracker.animate.set_value(1.2),
            x30_tracker.animate.set_value(0.8),
            run_time=2
        )
        
        self.wait(2)
