from manim import *

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Coordinate System Primitives (Replacing Axes)
        # origin_point set to (-4, -2, 0)
        origin_pt = LEFT * 4 + DOWN * 2
        x_axis_line = Line(origin_pt, origin_pt + RIGHT * 6, color=WHITE)
        y_axis_line = Line(origin_pt, origin_pt + UP * 4.5, color=WHITE)
        
        x_label = Text("Fraction of Zones Flagged", font_size=18).next_to(x_axis_line, DOWN)
        y_label = Text("Recall", font_size=18).rotate(1.5708).next_to(y_axis_line, LEFT)
        
        self.add(x_axis_line, y_axis_line, x_label, y_label)

        # 2. Formula Reference
        formula_tex = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=28).to_corner(UL, buff=0.5)
        self.add(formula_tex)

        # 3. Concave Curve (Manual Point Generation)
        # Using y = 1 - (1-x)^2.5 for a sharp elbow shape
        points_list = []
        for i in range(26):
            x_f = i / 25.0
            y_f = 1.0 - (1.0 - x_f)**2.5
            points_list.append(origin_pt + x_f * 6 * RIGHT + y_f * 4 * UP)
        
        graph_line = VMobject(color=BLUE)
        graph_line.set_points_as_corners(points_list)
        self.play(Create(graph_line))

        # 4. State Tracker (Replacing ValueTracker)
        # We use a hidden Dot and track its x-coordinate
        # Initial position is at the origin (x = -4)
        state_tracker = Dot(radius=0, fill_opacity=0).move_to(origin_pt)
        self.add(state_tracker)

        # Dynamic Point on the curve
        moving_dot = Dot(color=YELLOW, radius=0.12)
        
        def update_moving_dot(mobj):
            # Calculate current fraction based on state_tracker's x position
            # origin_x = -4, width = 6
            current_x = (state_tracker.get_center()[0] - (-4.0)) / 6.0
            if current_x < 0: current_x = 0
            if current_x > 1: current_x = 1
            
            # y = 1 - (1-x)^2.5
            current_y = 1.0 - (1.0 - current_x)**2.5
            mobj.move_to(origin_pt + current_x * 6 * RIGHT + current_y * 4 * UP)

        moving_dot.add_updater(update_moving_dot)
        self.add(moving_dot)

        # 5. Side Panel Floorplan
        zones_grid = VGroup(*[Square(side_length=0.45, stroke_width=2) for _ in range(25)])
        zones_grid.arrange_in_grid(5, 5, buff=0.1).to_edge(RIGHT, buff=0.7).shift(UP * 0.5)
        
        grid_label = Text("Zones Highlighted", font_size=20).next_to(zones_grid, UP)
        self.add(zones_grid, grid_label)

        def update_zones_color(group):
            current_x = (state_tracker.get_center()[0] - (-4.0)) / 6.0
            if current_x < 0: current_x = 0
            if current_x > 1: current_x = 1
            
            num_to_light = int(current_x * 25)
            for idx, square in enumerate(group):
                if idx < num_to_light:
                    square.set_fill(RED, opacity=0.7)
                else:
                    square.set_fill(BLACK, opacity=0)

        zones_grid.add_updater(update_zones_color)

        # 6. Animation Sequence
        # Move from 0.0 to 1.0 (tracker moves 6 units RIGHT)
        self.play(
            state_tracker.animate.move_to(origin_pt + RIGHT * 6),
            run_time=4,
            rate_func=lambda t: t
        )
        self.wait(0.5)
        
        # Move back to identify the 'Elbow' (around x=0.3)
        # Target x = origin_x + 0.3 * width = -4 + 1.8 = -2.2
        self.play(
            state_tracker.animate.move_to(origin_pt + RIGHT * 1.8),
            run_time=2
        )
        
        # 7. Highlight the Elbow (Replacing Arc with Circle)
        elbow_x = 0.3
        elbow_y = 1.0 - (1.0 - elbow_x)**2.5
        elbow_pos = origin_pt + (elbow_x * 6 * RIGHT) + (elbow_y * 4 * UP)
        
        elbow_circle = Circle(radius=0.4, color=YELLOW, stroke_width=5).move_to(elbow_pos)
        elbow_text = Text("The Elbow", font_size=24, color=YELLOW).next_to(elbow_circle, UR, buff=0.2)
        
        self.play(
            Create(elbow_circle),
            Write(elbow_text)
        )
        self.wait(2)
