from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Manual Axes for Physical Space (Left)
        left_center = LEFT * 3.5
        l_x_axis = Line(left_center + LEFT * 2, left_center + RIGHT * 2, color=GRAY)
        l_y_axis = Line(left_center + DOWN * 2, left_center + UP * 2, color=GRAY)
        l_title = Text("Physical Space", font_size=24).next_to(l_y_axis, UP)
        left_group = VGroup(l_x_axis, l_y_axis, l_title)

        # 2. Setup Manual Axes for Learned Space (Right)
        right_center = RIGHT * 3.5
        r_x_axis = Line(right_center + LEFT * 2, right_center + RIGHT * 2, color=GRAY)
        r_y_axis = Line(right_center + DOWN * 2, right_center + UP * 2, color=GRAY)
        r_title = Text("Learned Space", font_size=24).next_to(r_y_axis, UP)
        right_group = VGroup(r_x_axis, r_y_axis, r_title)

        # 3. Formula at the bottom
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=30
        ).to_edge(DOWN, buff=0.5)

        self.add(left_group, right_group, formula)

        # 4. Create Overlapping Dots in Physical Space
        np.random.seed(42)
        num_dots = 15
        
        # Red cluster (Electron 1)
        dots_red = VGroup(*[
            Dot(left_center + np.array([np.random.normal(0.2, 0.4), np.random.normal(0.1, 0.4), 0]), 
                color=RED, radius=0.07) 
            for _ in range(num_dots)
        ])
        
        # Blue cluster (Electron 2) - heavily overlapping with red
        dots_blue = VGroup(*[
            Dot(left_center + np.array([np.random.normal(-0.2, 0.4), np.random.normal(-0.1, 0.4), 0]), 
                color=BLUE, radius=0.07) 
            for _ in range(num_dots)
        ])

        self.play(FadeIn(dots_red), FadeIn(dots_blue))
        self.wait(1)

        # 5. Define Targeted Positions in Learned Space (Separated)
        red_targets = [
            right_center + LEFT * 1.2 + np.array([np.random.normal(0, 0.35), np.random.normal(0, 0.35), 0])
            for _ in range(num_dots)
        ]
        blue_targets = [
            right_center + RIGHT * 1.2 + np.array([np.random.normal(0, 0.35), np.random.normal(0, 0.35), 0])
            for _ in range(num_dots)
        ]

        # 6. Transform Animation
        self.play(
            *[dots_red[i].animate.move_to(red_targets[i]) for i in range(num_dots)],
            *[dots_blue[i].animate.move_to(blue_targets[i]) for i in range(num_dots)],
            run_time=2,
            rate_func=bezier([0, 0, 1, 1])
        )
        self.wait(0.5)

        # 7. Draw Circles around separated clusters
        # Using standard Circle as DashedVMobject is restricted
        circle_red = Circle(radius=1.0, color=RED).move_to(right_center + LEFT * 1.2)
        circle_blue = Circle(radius=1.0, color=BLUE).move_to(right_center + RIGHT * 1.2)
        
        # Simulate dashed look with a loop of lines if needed, 
        # but standard Circle is safer for compilation.
        self.play(Create(circle_red), Create(circle_blue))
        
        self.wait(2)
