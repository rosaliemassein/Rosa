from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # Display the formula
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # Create dots for the irregular geometry
        central_dot = Dot(point=ORIGIN, color=GREEN, radius=0.15)
        
        # Initial positions for neighbors
        neighbor_positions = [
            [2, 1, 0], [-1.5, 1.5, 0], [0.5, -2, 0],
            [2.2, -1.2, 0], [-1.8, -0.5, 0], [-0.5, 2.2, 0],
            [1.2, 0.2, 0], [-1.2, -1.8, 0]
        ]
        
        neighbors = VGroup(*[
            Dot(point=np.array(pos), color=BLUE, radius=0.1) 
            for pos in neighbor_positions
        ])
        
        # Helper function to create arrows with style based on distance
        def get_weighted_arrows(center_dot, neighbor_group):
            new_arrows = VGroup()
            c_pos = center_dot.get_center()
            for dot in neighbor_group:
                n_pos = dot.get_center()
                dist_sq = np.sum((c_pos - n_pos)**2)
                # Weight formula: exp(-d^2)
                weight = np.exp(-dist_sq / 2.0) # Adjusted scale for visual clarity
                
                # Thickness and opacity scale with weight
                thickness = 1 + 10 * weight
                opac = 0.1 + 0.9 * weight
                
                arr = Arrow(
                    start=c_pos,
                    end=n_pos,
                    buff=0.1,
                    color=BLUE,
                    stroke_width=thickness,
                    max_tip_length_to_length_ratio=0.2
                ).set_opacity(opac)
                new_arrows.add(arr)
            return new_arrows

        # Initial Arrows
        arrows = get_weighted_arrows(central_dot, neighbors)

        # Animation 1: Create the graph
        self.play(Create(central_dot))
        self.play(Create(neighbors), run_time=1)
        self.play(Create(arrows), run_time=1.5)
        self.wait(1)

        # Updated positions to show "learned feature space" shifting
        new_positions = [
            [0.8, 0.4, 0],    # Closer
            [-2.5, 2.5, 0],   # Further
            [0.3, -0.7, 0],   # Closer
            [3.5, -2.5, 0],   # Further
            [-0.7, -0.2, 0],  # Closer
            [-2.0, 3.2, 0],   # Further
            [0.6, 0.1, 0],    # Closer
            [-0.5, -0.9, 0]   # Closer
        ]

        # Target neighbor group for transformation
        target_neighbors = VGroup(*[
            neighbors[i].copy().move_to(np.array(new_positions[i]))
            for i in range(len(neighbors))
        ])
        
        # Target arrows based on new positions
        target_arrows = get_weighted_arrows(central_dot, target_neighbors)

        # Animation 2: Dynamic graph structure update
        self.play(
            Transform(neighbors, target_neighbors),
            Transform(arrows, target_arrows),
            run_time=3,
            rate_func=slow_into
        )
        self.wait(2)

        # Final Fade Out
        self.play(
            FadeOut(arrows),
            FadeOut(neighbors),
            FadeOut(central_dot),
            FadeOut(formula)
        )
        self.wait(1)
