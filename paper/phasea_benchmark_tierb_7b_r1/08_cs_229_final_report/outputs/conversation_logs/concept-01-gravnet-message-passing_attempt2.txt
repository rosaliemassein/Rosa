from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Background Formula
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # 2. Setup Central Hit (The node we are focusing on)
        central_dot = Dot(point=ORIGIN, color=RED, radius=0.15)
        self.add(central_dot)

        # 3. Initial positions of neighboring calorimeter hits
        # Creating an irregular distribution to simulate detector cells
        neighbor_positions = [
            [2.0, 1.0, 0],
            [1.2, 1.8, 0],
            [-1.5, 1.2, 0],
            [-1.8, -0.8, 0],
            [-0.5, -2.0, 0],
            [1.5, -1.5, 0]
        ]
        
        neighbors = VGroup(*[
            Dot(point=pos, color=BLUE, radius=0.1) 
            for pos in neighbor_positions
        ])
        self.play(Create(neighbors))
        self.wait(0.5)

        # 4. Connection Arrow Logic
        # We define a helper that creates an arrow whose visual weight 
        # is determined by the distance d between the central dot and a neighbor.
        def get_weighted_connection(target_mobject):
            start_pos = central_dot.get_center()
            end_pos = target_mobject.get_center()
            dist_value = np.linalg.norm(start_pos - end_pos)
            
            # The core GravNet potential: V = exp(-d^2)
            # We scale the distance slightly for screen-space aesthetics
            weight = np.exp(-(dist_value**2) / 2.0)
            
            # Visual mapping:
            # Thickness scales with weight
            # Opacity scales with weight
            return Arrow(
                start_pos, end_pos, 
                buff=0.1, 
                stroke_width=max(0.5, weight * 15), 
                max_tip_length_to_length_ratio=min(0.2, weight + 0.1),
                color=WHITE
            ).set_opacity(max(0.1, weight))

        # 5. Create Connections with Updaters
        # always_redraw ensures the arrow is recalculated every frame as dots move
        connections = VGroup()
        for neighbor in neighbors:
            conn = always_redraw(lambda n=neighbor: get_weighted_connection(n))
            connections.add(conn)

        self.play(FadeIn(connections))
        self.wait(1)

        # 6. "Learning" Animation - Dynamic Feature Space
        # As the network "learns", the hits move in a learned feature space.
        # This movement causes the graph connections to shift dynamically.
        learned_positions = [
            [0.7, 0.4, 0],    # Moves closer -> Connection strengthens
            [1.5, 2.5, 0],    # Moves farther -> Connection weakens
            [-0.6, 0.5, 0],   # Moves closer -> Connection strengthens
            [-3.0, -1.5, 0],  # Moves farther -> Connection weakens
            [0.2, -0.7, 0],   # Moves closer -> Connection strengthens
            [2.8, -2.8, 0]    # Moves farther -> Connection weakens
        ]

        # Animate neighbors to their new feature-space positions
        # The connection arrows will update automatically because of always_redraw
        self.play(
            *[neighbors[i].animate.move_to(learned_positions[i]) 
              for i in range(len(neighbors))],
            run_time=4,
            rate_func=smooth
        )

        self.wait(3)
