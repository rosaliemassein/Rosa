from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Coordinate Systems manually using Lines since Axes is disallowed
        # Physical Space (Left)
        p_origin = LEFT * 3.5
        p_axis_h = Line(p_origin + LEFT * 2, p_origin + RIGHT * 2, color=GRAY)
        p_axis_v = Line(p_origin + DOWN * 2, p_origin + UP * 2, color=GRAY)
        physical_space = VGroup(p_axis_h, p_axis_v)
        
        physical_title = Text("Physical Space", font_size=24).next_to(p_axis_v, UP)

        # Learned Space (Right)
        l_origin = RIGHT * 3.5
        l_axis_h = Line(l_origin + LEFT * 2, l_origin + RIGHT * 2, color=GRAY)
        l_axis_v = Line(l_origin + DOWN * 2, l_origin + UP * 2, color=GRAY)
        learned_space = VGroup(l_axis_h, l_axis_v)
        
        learned_title = Text("Learned Space", font_size=24).next_to(l_axis_v, UP)

        # 2. Setup Dots (Clusters)
        np.random.seed(42)
        num_dots = 12
        
        # Physical: Heavy overlap
        dots_p1 = VGroup()
        dots_p2 = VGroup()
        for _ in range(num_dots):
            # Cluster 1 (Red)
            p1 = p_origin + np.array([np.random.normal(0.2, 0.4), np.random.normal(0.2, 0.4), 0])
            dots_p1.add(Dot(p1, color=RED, radius=0.08))
            # Cluster 2 (Blue)
            p2 = p_origin + np.array([np.random.normal(-0.2, 0.4), np.random.normal(-0.2, 0.4), 0])
            dots_p2.add(Dot(p2, color=BLUE, radius=0.08))

        # Learned: Clearly separated
        l1_center = l_origin + UP * 1.2 + RIGHT * 0.8
        l2_center = l_origin + DOWN * 1.2 + LEFT * 0.8
        
        dots_l1_targets = VGroup()
        dots_l2_targets = VGroup()
        for _ in range(num_dots):
            t1 = l1_center + np.array([np.random.normal(0, 0.3), np.random.normal(0, 0.3), 0])
            dots_l1_targets.add(Dot(t1, color=RED, radius=0.08))
            t2 = l2_center + np.array([np.random.normal(0, 0.3), np.random.normal(0, 0.3), 0])
            dots_l2_targets.add(Dot(t2, color=BLUE, radius=0.08))

        # 3. Formula
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=30
        ).to_edge(DOWN, buff=0.5)

        # 4. Helper for dashed circles (since DashedVMobject is disallowed)
        def create_dashed_circle(center, radius, color):
            circle_group = VGroup()
            num_segments = 16
            for i in range(num_segments):
                start_angle = i * (2 * PI / num_segments)
                arc = Arc(
                    radius=radius,
                    start_angle=start_angle,
                    angle=PI / num_segments,
                    arc_center=center,
                    color=color,
                    stroke_width=2
                )
                circle_group.add(arc)
            return circle_group

        # 5. Animation Sequence
        self.add(physical_space, learned_space, physical_title, learned_title)
        self.play(FadeIn(dots_p1), FadeIn(dots_p2))
        self.play(Write(formula))
        self.wait(1)

        # Transformation to learned space
        # We transform the physical dots to the target positions in the learned space
        self.play(
            Transform(dots_p1, dots_l1_targets),
            Transform(dots_p2, dots_l2_targets),
            run_time=2.5
        )
        self.wait(0.5)

        # Show classification success with dashed circles
        c1 = create_dashed_circle(l1_center, 0.8, RED)
        c2 = create_dashed_circle(l2_center, 0.8, BLUE)
        
        self.play(Create(c1), Create(c2))
        self.wait(2)
