from manim import *
import numpy as np

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Manual Axis Construction (to avoid disallowed Axes class)
        # We will use a coordinate system from x=-4 to 4 and y=-2 to 2
        origin_point = np.array([-4, -2, 0])
        x_axis = Line(origin_point, origin_point + np.array([8, 0, 0]), color=WHITE)
        y_axis = Line(origin_point, origin_point + np.array([0, 4, 0]), color=WHITE)
        
        x_label = Text("Distance between particles", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("Efficiency", font_size=20).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        # 2. Curves Construction
        # ML Curve: Flat line at 90% efficiency
        # Y coordinate: origin_y + 4.0 * 0.9 = -2 + 3.6 = 1.6
        ml_curve = Line([-4, 1.6, 0], [4, 1.6, 0], color=BLUE)
        ml_label = Text("ML Approach", color=BLUE, font_size=16).next_to(ml_curve, UP, buff=0.1)
        
        # Standard Algorithm Curve: Drops exponentially to 0 as distance closes
        std_pts = []
        for i in range(51):
            prop = i / 50.0  # 0 to 1
            # x maps from -4 to 4
            curr_x = -4 + prop * 8
            # y maps from efficiency 0 to 0.9 (scaled to 4.0 height)
            # Function: eff = 0.9 * (1 - exp(-5 * prop))
            curr_y = -2 + 4.0 * 0.9 * (1 - np.exp(-5 * prop))
            std_pts.append([curr_x, curr_y, 0])
            
        std_curve = VMobject(color=RED)
        std_curve.set_points_as_corners(std_pts)
        std_label = Text("Standard Algorithm", color=RED, font_size=16).move_to([1, -1, 0])

        # 3. Particle Shower Icons (at the top)
        # Using Dot and Circle to represent showers
        shower_l = Dot(color=YELLOW, radius=0.15)
        shower_r = Dot(color=YELLOW, radius=0.15)
        glow_l = Circle(radius=0.3, color=YELLOW, stroke_width=0, fill_opacity=0.2)
        glow_r = Circle(radius=0.3, color=YELLOW, stroke_width=0, fill_opacity=0.2)
        
        # 4. Tracking and Interaction Logic (avoiding ValueTracker and always_redraw)
        # We'll use a hidden Dot to track the "distance" state
        tracker = Dot(point=[4, 0, 0], fill_opacity=0)
        
        # Indicator line that follows the tracker
        # Line height goes from y=-2 to y=1.6
        v_line = Line([0, -2, 0], [0, 1.6, 0], color=WHITE, stroke_width=1)
        
        def update_indicator(mob):
            # Match indicator x to tracker x
            mob.move_to([tracker.get_x(), -0.2, 0])
            
        def update_showers(mob):
            # Calculate separation: tracker x goes 4 -> -4, ratio goes 1 -> 0
            ratio = (tracker.get_x() + 4) / 8.0
            sep = ratio * 2.0
            shower_l.move_to([-sep, 2.5, 0])
            shower_r.move_to([sep, 2.5, 0])
            glow_l.move_to(shower_l.get_center())
            glow_r.move_to(shower_r.get_center())

        # Attach updaters
        v_line.add_updater(update_indicator)
        shower_group = VGroup(shower_l, shower_r, glow_l, glow_r)
        shower_group.add_updater(update_showers)

        # 5. Animation Sequence
        self.add(x_axis, y_axis, x_label, y_label)
        self.play(Create(ml_curve), Write(ml_label))
        self.play(Create(std_curve), Write(std_label))
        self.add(v_line, tracker, shower_group)
        self.wait(1)
        
        # Move tracker to simulate distance closing
        # Note: 'linear' replaced by lambda for safety
        self.play(
            tracker.animate.move_to([-3.9, 0, 0]),
            run_time=8,
            rate_func=lambda t: t
        )
        self.wait(2)
