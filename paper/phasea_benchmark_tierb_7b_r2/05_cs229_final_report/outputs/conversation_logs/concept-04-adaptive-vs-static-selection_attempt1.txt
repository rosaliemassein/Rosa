from manim import *

class AdaptiveVSStaticSelection(Scene):
    def construct(self):
        # Layout: Labels for both sides
        left_title = Text("Static Benchmark", font_size=26).to_edge(UP).shift(LEFT * 3.2)
        right_title = Text("Logistic Model", font_size=26).to_edge(UP).shift(RIGHT * 3.2)
        
        # Floorplan frames (Rectangles as placeholders for the floorplan images)
        rect_left = Rectangle(width=4.5, height=3.2, color=GREY).shift(LEFT * 3.2 + UP * 0.4)
        rect_right = Rectangle(width=4.5, height=3.2, color=GREY).shift(RIGHT * 3.2 + UP * 0.4)
        
        self.add(left_title, right_title, rect_left, rect_right)

        # Timeline at the bottom (May to September)
        time_start_pos = LEFT * 5.5 + DOWN * 1.8
        time_end_pos = RIGHT * 5.5 + DOWN * 1.8
        timeline_path = Line(time_start_pos, time_end_pos, color=WHITE)
        self.add(timeline_path)
        
        months_labels = ["May", "Jun", "Jul", "Aug", "Sep"]
        for i in range(5):
            alpha_val = i / 4.0
            tick_pos = timeline_path.point_from_proportion(alpha_val)
            tick_line = Line(tick_pos + UP * 0.1, tick_pos + DOWN * 0.1)
            txt = Text(months_labels[i], font_size=18).next_to(tick_line, DOWN)
            self.add(tick_line, txt)

        # Static side: Highlighting zones that remain frozen
        frozen_dots = VGroup(
            Dot(rect_left.get_center() + UP * 0.8 + LEFT * 0.5, color=RED),
            Dot(rect_left.get_center() + DOWN * 0.6 + RIGHT * 0.9, color=RED),
            Dot(rect_left.get_center() + LEFT * 1.2, color=RED)
        )
        self.add(frozen_dots)

        # ML side: Groups of dots for varying count/adaptive behavior
        ml_center = rect_right.get_center()
        ml_group_early = VGroup(
            Dot(ml_center + UP * 0.5, color=BLUE),
            Dot(ml_center + LEFT * 0.8, color=BLUE)
        )
        ml_group_heatwave = VGroup(
            Dot(ml_center + UP * 0.8, color=BLUE),
            Dot(ml_center + DOWN * 0.5, color=BLUE),
            Dot(ml_center + LEFT * 0.6, color=BLUE),
            Dot(ml_center + RIGHT * 0.9, color=BLUE),
            Dot(ml_center, color=BLUE)
        )
        ml_group_late = VGroup(
            Dot(ml_center + DOWN * 0.7, color=BLUE),
            Dot(ml_center + RIGHT * 0.5, color=BLUE)
        )

        # Graph for Missed Requests (Recall Error)
        graph_origin = DOWN * 3.4 + LEFT * 5.5
        axis_h = Line(graph_origin, graph_origin + RIGHT * 11)
        axis_v = Line(graph_origin, graph_origin + UP * 1.2)
        graph_name = Text("Missed Requests (Recall Error)", font_size=15).next_to(axis_v, RIGHT, buff=0.2).shift(UP * 0.6)
        self.add(axis_h, axis_v, graph_name)

        # Static Error (High and Constant)
        error_static_line = Line(graph_origin + UP * 0.9, graph_origin + UP * 0.9 + RIGHT * 11, color=RED)
        self.add(error_static_line)

        # ML Error (Significantly lower and dynamic)
        # Using manual points to construct the path
        p_0 = graph_origin + UP * 0.5
        p_1 = graph_origin + RIGHT * 2.75 + UP * 0.2
        p_2 = graph_origin + RIGHT * 5.5 + UP * 0.4
        p_3 = graph_origin + RIGHT * 8.25 + UP * 0.15
        p_4 = graph_origin + RIGHT * 11 + UP * 0.1
        
        error_ml_vobject = VMobject(color=BLUE)
        error_ml_vobject.set_points_as_corners([p_0, p_1, p_2, p_3, p_4])

        # Marker for the timeline progression
        timeline_marker = Dot(time_start_pos, color=YELLOW)
        self.add(timeline_marker)

        # Sequence of Animations

            FadeIn(ml_group_early),
            run_time=2
        )
        # Part 2: June to July (Heatwave increases highlighted count)
        self.play(
            timeline_marker.animate.move_to(timeline_path.point_from_proportion(0.5)),
            ReplacementTransform(ml_group_early, ml_group_heatwave),
            run_time=2
        )
        # Part 3: July to August (Adaptive model decreases count as stress lowers)

            timeline_marker.animate.move_to(timeline_path.point_from_proportion(0.75)),
            ReplacementTransform(ml_group_heatwave, ml_group_late),
            run_time=2
        )
        # Part 4: August to September

            timeline_marker.animate.move_to(timeline_path.point_from_proportion(1.0)),
            FadeOut(ml_group_late),
            FadeIn(ml_group_early),
            run_time=2
        )

        self.wait(2)
