from manim import *
import numpy as np

class ElbowThreshold(Scene):
    def construct(self):
        # Create axes for Recall vs Fraction of Zones Flagged
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True, "color": WHITE},
            tips=True
        ).to_edge(LEFT, buff=0.5)
        
        labels = axes.get_axis_labels(
            x_label=Text("Fraction of Zones Flagged", font_size=18),
            y_label=Text("Recall", font_size=18)
        )
        
        # Concave curve representing validation results (y = x^0.4)
        def curve_func(x):
            if x < 0: return 0
            return x**0.4
            
        curve = axes.plot(curve_func, x_range=[0, 1], color=BLUE)
        
        # Formula display
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_corner(UL, buff=0.5)
        
        # Side Panel - Building Floorplan (5x5 grid)
        floor_plan = VGroup(*[
            Square(side_length=0.4, stroke_width=2).set_fill(GRAY_E, opacity=0.5)
            for _ in range(25)
        ]).arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=1)
        
        floor_title = Text("Building Zones", font_size=20).next_to(floor_plan, UP)
        
        # Interactive Point and Tracker
        vt = ValueTracker(0.1)
        
        # Highlighted point on the curve
        dot = Dot(color=YELLOW)
        def update_dot(mob):
            mob.move_to(axes.i2gp(vt.get_value(), curve))
        dot.add_updater(update_dot)
        
        # Updater for floor plan squares based on the fraction (ValueTracker)
        def update_floor_plan(group):
            val = vt.get_value()
            num_to_highlight = int(val * 25)
            for i, zone in enumerate(group):
                if i < num_to_highlight:
                    zone.set_fill(YELLOW, opacity=0.8)
                    zone.set_stroke(YELLOW, opacity=1)
                else:
                    zone.set_fill(GRAY_E, opacity=0.5)
                    zone.set_stroke(WHITE, opacity=1)
        
        floor_plan.add_updater(update_floor_plan)
        
        # Animation sequence
        self.play(Create(axes), Write(labels))
        self.play(Write(formula))
        self.play(Create(curve))
        self.play(Create(floor_plan), Write(floor_title))
        self.add(dot)
        self.wait(1)
        
        # Move point right: Higher recall, but inefficient (too many zones)
        self.play(vt.animate.set_value(0.9), run_time=4, rate_func=linear)
        self.wait(1)
        
        # Move point left to find the 'elbow'
        self.play(vt.animate.set_value(0.25), run_time=3)
        self.wait(0.5)
        
        # Highlight the 'elbow' with a glowing circle
        elbow_pos = axes.i2gp(0.25, curve)
        elbow_circle = Circle(radius=0.4, color=YELLOW).move_to(elbow_pos)
        elbow_circle.set_stroke(width=6)
        
        self.play(Create(elbow_circle))
        self.play(
            elbow_circle.animate.scale(1.2).set_stroke(opacity=0.3),
            Flash(elbow_pos, color=YELLOW, flash_radius=0.5),
            run_time=0.8
        )
        self.play(elbow_circle.animate.scale(1/1.2).set_stroke(opacity=1))
        
        elbow_label = Text("The 'Elbow' Sweet Spot", font_size=22, color=YELLOW).next_to(elbow_circle, UR)
        self.play(Write(elbow_label))
        
        self.wait(3)
