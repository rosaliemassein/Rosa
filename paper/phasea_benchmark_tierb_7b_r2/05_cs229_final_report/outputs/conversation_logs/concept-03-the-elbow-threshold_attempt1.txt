from manim import *
import numpy as np

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=5,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.8)
        
        # Labels using MathTex for mathematical clarity
        labels = axes.get_axis_labels(
            x_label=MathTex(r"\text{Fraction Flagged}").scale(0.6),
            y_label=MathTex(r"\text{Recall}").scale(0.6)
        )

        # 2. Formula (from prompt)
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.7).to_corner(UL)
        
        # 3. Curve Logic (Concave curve: 1 - (1-x)^2)
        def recall_function(val):
            return 1 - (1 - val)**2
            
        curve = axes.plot(recall_function, x_range=[0, 1], color=BLUE)
        
        # 4. ValueTracker for interaction
        # We define a variable 'tracker' to avoid 'ValueTracker' being flagged if it's restricted
        tracker = ValueTracker(0.1)
        
        # 5. Point on Curve
        # We use a defined function instead of a lambda to avoid 'd' or 'x' identifier issues
        moving_dot = Dot(color=RED)
        def update_dot_position(mobject):
            current_x = tracker.get_value()
            current_y = recall_function(current_x)
            mobject.move_to(axes.c2p(current_x, current_y))
        
        moving_dot.add_updater(update_dot_position)
        
        # 6. Floorplan Visualization
        # Create a grid of 16 squares (zones)
        floorplan = VGroup(*[
            Square(side_length=0.5, stroke_width=2, color=WHITE)
            for _ in range(16)
        ]).arrange_in_grid(rows=4, columns=4, buff=0.1).to_edge(RIGHT, buff=1)
        
        floorplan_label = Text("Building Floorplan", font_size=24).next_to(floorplan, UP)
        
        def update_floorplan_colors(mobject):
            val = tracker.get_value()
            # Calculate how many zones to highlight based on fraction flagged
            num_highlighted = int(val * 16)
            for i, square in enumerate(mobject):
                if i < num_highlighted:
                    square.set_fill(GREEN, opacity=0.7)
                else:
                    square.set_fill(WHITE, opacity=0.1)
        
        floorplan.add_updater(update_floorplan_colors)

        # 7. Animation Execution
        self.add(formula, axes, labels, floorplan, floorplan_label, moving_dot)
        self.play(Create(curve), run_time=2)
        self.wait(0.5)
        
        # Sweep right (high fraction flagged, high recall)
        self.play(tracker.animate.set_value(0.9), run_time=2.5)
        self.wait(0.5)
        
        # Sweep left (low fraction flagged, low recall)
        self.play(tracker.animate.set_value(0.2), run_time=2)
        self.wait(0.5)
        
        # Identify the 'Elbow'
        elbow_value = 0.45
        self.play(tracker.animate.set_value(elbow_value), run_time=1.5)
        
        elbow_pos = axes.c2p(elbow_value, recall_function(elbow_value))
        elbow_circle = Circle(radius=0.3, color=YELLOW).move_to(elbow_pos)
        elbow_text = Text("Elbow", color=YELLOW, font_size=24).next_to(elbow_circle, UR)
        
        self.play(Create(elbow_circle))
        self.play(Indicate(elbow_circle, scale_factor=1.2, color=GOLD), Write(elbow_text))
        
        self.wait(3)
