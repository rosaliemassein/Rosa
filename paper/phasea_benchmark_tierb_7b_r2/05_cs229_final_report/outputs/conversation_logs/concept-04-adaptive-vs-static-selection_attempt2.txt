import numpy as np
from manim import *

class AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. Setup Layout and Titles
        # We avoid FRAME_WIDTH and Point to satisfy the runner constraints
        l_col_x = -3.5
        r_col_x = 3.5
        
        static_title = Text("Static Benchmark", font_size=28).move_to([l_col_x, 3.2, 0])
        adaptive_title = Text("Logistic Model", font_size=28).move_to([r_col_x, 3.2, 0])
        self.add(static_title, adaptive_title)

        # 2. Floorplans (Simple grids)
        def make_grid(center_x):
            grid = VGroup()
            for i in range(3):
                for j in range(3):
                    r = Square(side_length=0.7, stroke_width=2, color=GRAY)
                    r.move_to([center_x - 0.8 + i*0.8, 1.0 + j*0.8, 0])
                    grid.add(r)
            return grid

        grid_l = make_grid(l_col_x)
        grid_r = make_grid(r_col_x)
        self.add(grid_l, grid_r)

        # 3. Static Highlights (Frozen state)
        # These represent the "rules of thumb" that never change
        static_dots = VGroup(*[
            Dot(grid_l[i].get_center(), color=RED) for i in [0, 2, 4, 6, 8]
        ])
        self.add(static_dots)

        # 4. Adaptive Highlights (Dynamic)
        # Create a pool of dots that will flicker based on an updater
        adaptive_dots = VGroup(*[
            Dot(grid_r[i].get_center(), color=GREEN) for i in range(9)
        ])
        
        # 5. Timer and Timeline
        # We use a hidden Mobject to track progress instead of ValueTracker
        timer_mob = Dot(radius=0).move_to([-4, -1.5, 0])
        
        timeline_line = Line([-4, -1.5, 0], [4, -1.5, 0], color=WHITE)
        months = ["May", "Jun", "Jul", "Aug", "Sep"]
        for i, month_name in enumerate(months):
            pos = timeline_line.point_from_proportion(i / 4.0)
            lbl = Text(month_name, font_size=18).next_to(pos, DOWN, buff=0.2)
            self.add(lbl)
        
        cursor = Triangle(fill_opacity=1, color=YELLOW).scale(0.12).rotate(PI)
        # Stick cursor to the timer_mob's x position
        cursor.add_updater(lambda mob: mob.move_to([timer_mob.get_center()[0], -1.3, 0]))
        self.add(timeline_line, cursor)

        # 6. Graph (Manual construction to avoid Axes/ValueTracker)
        graph_origin = [-2, -3.2, 0]
        g_x_axis = Line(graph_origin, [2, -3.2, 0], stroke_width=2)
        g_y_axis = Line(graph_origin, [-2, -2.0, 0], stroke_width=2)
        g_label = Text("Missed Requests (Recall Error)", font_size=16).move_to([0, -1.9, 0])
        self.add(g_x_axis, g_y_axis, g_label)

        # Pre-calculate line graph paths
        pts_static = []
        pts_ml = []
        for i in range(101):
            alpha = i / 100.0
            x_val = -2 + 4 * alpha
            # Static line: remains high (inefficient)
            y_s = -3.2 + 0.8 + 0.1 * np.sin(alpha * 15)
            pts_static.append([x_val, y_s, 0])
            # ML line: stays significantly lower
            y_m = -3.2 + 0.2 + 0.05 * np.cos(alpha * 10)
            pts_ml.append([x_val, y_m, 0])
        
        path_static = VMobject(color=BLUE, stroke_width=3).set_points_as_corners(pts_static)
        path_ml = VMobject(color=GREEN, stroke_width=3).set_points_as_corners(pts_ml)
        
        static_tag = Text("Static", color=BLUE, font_size=14).next_to(path_static, RIGHT, buff=0.1)
        ml_tag = Text("ML", color=GREEN, font_size=14).next_to(path_ml, RIGHT, buff=0.1)

        # 7. Adaptive Animation Updater
        def update_adaptive_dots(dots_group):
            # Calculate progress 0.0 to 1.0 based on timer_mob's x position (-4 to 4)
            progress = (timer_mob.get_center()[0] + 4) / 8.0
            for idx, dot in enumerate(dots_group):
                # Use sine waves to simulate flickering and count changes
                # Mid-summer (progress ~0.5) increases flagging count
                activation_val = np.sin(progress * 12 + idx) + np.cos(progress * 8 - idx * 1.5)
                # Dynamic threshold: lower during "heatwaves" (middle of timeline)
                threshold = 0.6 - 0.5 * np.sin(progress * PI)
                dot.set_opacity(1 if activation_val > threshold else 0)

        adaptive_dots.add_updater(update_adaptive_dots)
        self.add(adaptive_dots)

        # 8. Execution
        # Animate the timer moving, which drives updaters and reveals graphs
        self.play(
            timer_mob.animate.move_to([4, -1.5, 0]),
            Create(path_static),
            Create(path_ml),
            Write(static_tag),
            Write(ml_tag),
            run_time=10,
            rate_func=lambda t: t # linear rate function
        )
        self.wait(2)

        # Cleanup updaters
        adaptive_dots.clear_updaters()
        cursor.clear_updaters()

if __name__ == "__main__":
    scene = AdaptiveVsStaticSelection()
    scene.render()
