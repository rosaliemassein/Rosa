import numpy as np
from manim import *

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes
        ax = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        # Axis Labels
        x_text = Text("Fraction of Zones Flagged", font_size=20).next_to(ax.x_axis, DOWN)
        y_text = Text("Recall", font_size=20).rotate(90 * DEGREES).next_to(ax.y_axis, LEFT)

        # 2. Curve: Represents validation results (concave, sharp rise)
        # We use a named function to avoid lambda scope issues
        def recall_calc(val):
            return np.power(val, 0.4)
            
        curve_obj = ax.plot(recall_calc, x_range=[0, 1], color=BLUE)
        
        # 3. ValueTracker for threshold point
        # Named 'theta_tracker' to match reference patterns
        theta_tracker = ValueTracker(0.1)
        
        # 4. Moving Point on the curve
        dot_obj = Dot(color=RED)
        def update_dot(obj):
            current_x = theta_tracker.get_value()
            obj.move_to(ax.c2p(current_x, recall_calc(current_x)))
        dot_obj.add_updater(update_dot)
        
        # 5. Building Floorplan (Side Panel)
        # Create a grid of 16 zones
        zones_group = VGroup()
        for r_idx in range(4):
            for c_idx in range(4):
                rect_zone = Rectangle(width=0.5, height=0.5, stroke_width=2)
                # Position on the right side of the screen
                rect_zone.move_to(RIGHT * 4 + UP * (r_idx * 0.6 - 1) + RIGHT * (c_idx * 0.6 - 1))
                zones_group.add(rect_zone)
        
        # Updater to light up zones based on the tracker value
        def update_zones(group):
            val = theta_tracker.get_value()
            active_count = int(val * 16)
            for idx, zone in enumerate(group):
                if idx < active_count:
                    zone.set_fill(RED, opacity=0.7)
                else:
                    zone.set_fill(BLACK, opacity=0)
        
        zones_group.add_updater(update_zones)
        
        # 6. Elbow Highlighting
        # The 'elbow' is roughly at x=0.2
        elbow_center = ax.c2p(0.2, recall_calc(0.2))
        elbow_glow = Circle(radius=0.3, color=YELLOW).move_to(elbow_center)
        
        # 7. Recall Formula
        recall_formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_edge(DOWN, buff=0.3)

        # 8. Animation Sequence
        self.play(Create(ax), Write(x_text), Write(y_text))
        self.play(Create(curve_obj), Write(recall_formula))
        self.add(zones_group, dot_obj)
        self.wait(1)
        
        # Move right: flagging more zones increases recall but is inefficient
        self.play(theta_tracker.animate.set_value(0.95), run_time=3)
        self.wait(1)
        
        # Move left: searching for the elbow
        self.play(theta_tracker.animate.set_value(0.2), run_time=2)
        self.play(Create(elbow_glow))
        self.play(Flash(elbow_glow, color=YELLOW))
        self.wait(3)
