from manim import *
import numpy as np

class AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. Setup Layout
        title_static = Text("Static Benchmark (ASHRAE)", font_size=28).to_edge(UP).shift(LEFT * 3.5)
        title_logistic = Text("Logistic Model (Adaptive)", font_size=28).to_edge(UP).shift(RIGHT * 3.5)
        
        def create_floorplan():
            rooms = VGroup(*[Rectangle(width=0.8, height=0.6) for _ in range(6)])
            rooms.arrange_in_grid(rows=2, cols=3, buff=0.1)
            frame = SurroundingRectangle(rooms, color=GRAY, buff=0.2)
            return VGroup(frame, rooms)

        fp_static_group = create_floorplan().shift(LEFT * 3.5 + UP * 1.2)
        fp_logistic_group = create_floorplan().shift(RIGHT * 3.5 + UP * 1.2)
        
        # 2. Timeline at the bottom
        months = ["May", "Jun", "Jul", "Aug", "Sep"]
        timeline = NumberLine(
            x_range=[0, 4, 1],
            length=10,
            include_numbers=False,
            include_tip=True,
            color=GREY
        ).to_edge(BOTTOM, buff=0.8)
        
        month_labels = VGroup(*[
            Text(m, font_size=20).next_to(timeline.n2p(i), DOWN)
            for i, m in enumerate(months)
        ])
        
        time_tracker = ValueTracker(0)
        
        # 3. Highlighted Zones
        # Static: Always the same 2 zones (frozen)
        static_rooms = fp_static_group[1]
        static_highlights = VGroup(
            static_rooms[0].copy().set_fill(RED, opacity=0.6).set_stroke(RED, 2),
            static_rooms[4].copy().set_fill(RED, opacity=0.6).set_stroke(RED, 2)
        )
        
        # Logistic: Dynamic zones
        logistic_rooms = fp_logistic_group[1]
        logistic_highlights = VGroup(*[
            r.copy().set_fill(BLUE, opacity=0.0).set_stroke(BLUE, 2)
            for r in logistic_rooms
        ])
        
        def update_logistic_highlights(mob):
            t = time_tracker.get_value()
            for i, highlight in enumerate(mob):
                # Use sine waves with different phases to simulate "adaptive" changes
                val = np.sin(t * 2.5 + i * 1.2) + np.cos(t * 1.5 - i * 0.7)
                if val > 0.3:
                    highlight.set_fill(opacity=0.7)
                else:
                    highlight.set_fill(opacity=0)

        logistic_highlights.add_updater(update_logistic_highlights)

        # 4. Graphs (Missed Requests / Recall error)
        axes = Axes(
            x_range=[0, 4, 1], y_range=[0, 100, 25],
            x_length=10, y_length=1.5,
            axis_config={"include_tip": False, "color": GREY}
        ).next_to(timeline, UP, buff=0.5)
        
        graph_label = Text("Missed Requests (Recall Error)", font_size=20).next_to(axes, UP, buff=0.1)
        
        # Define line logic
        static_trace = TracedPath(
            lambda: axes.c2p(time_tracker.get_value(), 70 + 15 * np.sin(time_tracker.get_value() * 2)),
            stroke_color=RED, stroke_width=3
        )
        logistic_trace = TracedPath(
            lambda: axes.c2p(time_tracker.get_value(), 20 + 10 * np.cos(time_tracker.get_value() * 3)),
            stroke_color=BLUE, stroke_width=3
        )
        
        cursor = Triangle(fill_opacity=1, color=YELLOW).scale(0.1).rotate(PI)
        cursor.add_updater(lambda m: m.move_to(timeline.n2p(time_tracker.get_value()), UP))

        # 5. Animation
        self.add(title_static, title_logistic, fp_static_group, fp_logistic_group)
        self.add(timeline, month_labels, axes, graph_label)
        self.add(static_highlights)
        self.add(logistic_highlights, static_trace, logistic_trace, cursor)
        
        self.play(
            time_tracker.animate.set_value(4),
            run_time=10,
            rate_func=linear
        )
        
        self.wait(2)
