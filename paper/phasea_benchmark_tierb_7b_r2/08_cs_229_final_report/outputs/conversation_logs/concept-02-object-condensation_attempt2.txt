from manim import *
import numpy as np

class BetaCondensation(Scene):
    def construct(self):
        # 1. Background grid using simple lines
        grid_color = BLUE
        grid = VGroup()
        for x in range(-7, 8):
            grid.add(Line([x, -4, 0], [x, 4, 0], stroke_opacity=0.1, color=grid_color))
        for y in range(-4, 5):
            grid.add(Line([-7, y, 0], [7, y, 0], stroke_opacity=0.1, color=grid_color))
        self.add(grid)

        # 2. Show the charge calculation formula
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}").to_edge(UP)
        self.play(Write(formula))
        self.wait(0.5)

        # 3. Create initial hits (dots)
        np.random.seed(42)
        num_dots = 30
        dot_positions = [
            np.array([np.random.uniform(-6, 6), np.random.uniform(-3, 2), 0]) 
            for _ in range(num_dots)
        ]
        # Assign random beta values (0 to 1)
        betas = [np.random.uniform(0.1, 0.98) for _ in range(num_dots)]
        
        dots = VGroup()
        for pos in dot_positions:
            dots.add(Dot(pos, radius=0.08, color=PURPLE))
        
        self.play(FadeIn(dots))
        self.wait(0.5)

        # 4. Animate dots growing and changing color based on beta
        # We use a simple color mapping to avoid potentially undefined interpolation functions
        color_map = [PURPLE, BLUE, TEAL, GREEN, YELLOW]
        
        growth_anims = []
        for i, dot in enumerate(dots):
            beta = betas[i]
            # Determine color index
            color_idx = int(beta * (len(color_map) - 1))
            target_color = color_map[color_idx]
            # Grow size based on beta
            target_scale = 1.0 + beta * 2.5
            growth_anims.append(dot.animate.scale(target_scale).set_color(target_color))
        
        self.play(*growth_anims, run_time=2)
        self.wait(0.5)

        # 5. Create 'gravity wells' for high-beta dots
        # Points with beta > 0.8 are condensation points
        high_beta_indices = [i for i, b in enumerate(betas) if b > 0.8]
        well_groups = VGroup()
        
        for idx in high_beta_indices:
            center_pos = dots[idx].get_center()
            # Calculate charge q = arctanh(beta)
            q_val = np.arctanh(betas[idx])
            # Create a "well" using concentric circles to simulate a gradient
            well = VGroup()
            num_layers = 5
            max_radius = q_val * 0.4
            for r_factor in range(1, num_layers + 1):
                radius = (r_factor / num_layers) * max_radius
                layer = Circle(
                    radius=radius, 
                    stroke_width=0, 
                    fill_color=YELLOW, 
                    fill_opacity=0.1
                ).move_to(center_pos)
                well.add(layer)
            well_groups.add(well)
            
        self.play(FadeIn(well_groups))
        self.wait(0.5)

        # 6. Animate low-beta dots being pulled toward the nearest high-beta condensation point
        pull_anims = []
        for i, dot in enumerate(dots):
            if betas[i] <= 0.8:
                # Find the nearest high-beta dot
                current_pos = dot.get_center()
                distances = [
                    np.linalg.norm(current_pos - dots[h_idx].get_center()) 
                    for h_idx in high_beta_indices
                ]
                
                if distances:
                    nearest_idx = high_beta_indices[np.argmin(distances)]
                    target_pos = dots[nearest_idx].get_center()
                    
                    # Animate movement towards the condensation point
                    pull_anims.append(
                        dot.animate.move_to(target_pos).set_opacity(0.3)
                    )

        if pull_anims:
            self.play(*pull_anims, run_time=2.5, rate_func=smooth)
        
        self.wait(2)
