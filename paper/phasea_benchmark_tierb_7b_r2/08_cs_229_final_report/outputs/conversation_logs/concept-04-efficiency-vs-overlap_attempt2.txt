from manim import *
import numpy as np

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Title and Labels
        title = Text("Efficiency vs. Particle Distance", font_size=36).to_edge(UP)
        self.add(title)

        # 2. Manual Axes Construction
        # Since Axes is disallowed, we build it with Line objects
        # X-axis from -3 to 3 (total width 6), Y-axis from -2 to 1.5 (total height 3.5)
        origin_coord = [ -3, -2, 0 ]
        x_axis_line = Line(origin_coord, [ 3, -2, 0 ], color=WHITE)
        y_axis_line = Line(origin_coord, [ -3, 1.5, 0 ], color=WHITE)
        
        # Axis Labels
        dist_label = Text("Distance", font_size=24).next_to(x_axis_line, DOWN)
        eff_label = Text("Efficiency", font_size=24).next_to(y_axis_line, LEFT).rotate(90 * DEGREES)
        
        self.add(x_axis_line, y_axis_line, dist_label, eff_label)

        # 3. Performance Curves
        # ML Efficiency Curve: mostly flat at 0.9 (y = -2 + 0.9 * 3.5 = 1.15)
        ml_y_val = 1.15
        ml_curve = Line([ -3, ml_y_val, 0 ], [ 3, ml_y_val, 0 ], color=BLUE)
        ml_text = Text("ML Model", color=BLUE, font_size=20).next_to(ml_curve, UP, buff=0.1)
        
        # Standard Algorithm Curve: Drops sharply at small distance
        # We build it manually as a VGroup of small lines to simulate a dashed curve
        std_curve_group = VGroup()
        num_segments = 50
        for i_val in range(num_segments):
            start_x_pos = -3 + (6 * i_val / num_segments)
            end_x_pos = -3 + (6 * (i_val + 0.6) / num_segments) # 0.6 creates dash gap
            
            def get_y_for_x(input_x):
                rel_dist = input_x + 3 # range 0 to 6
                # Efficiency starts at 0 and grows to ~0.9
                efficiency_percent = 0.9 * (1.0 - np.exp(-1.5 * rel_dist))
                return -2 + (3.5 * efficiency_percent)

            segment = Line(
                [start_x_pos, get_y_for_x(start_x_pos), 0],
                [end_x_pos, get_y_for_x(end_x_pos), 0],
                color=RED
            )
            std_curve_group.add(segment)
            
        std_text = Text("Standard Algorithm", color=RED, font_size=20).next_to(std_curve_group, DOWN, buff=0.4).shift(RIGHT * 1.5)

        self.play(Create(ml_curve), Write(ml_text))
        self.play(Create(std_curve_group), Write(std_text))

        # 4. Vertical Tracking Line
        # We move a hidden Dot to act as a ValueTracker
        tracker_mobject = Dot([ 3, -2, 0 ], fill_opacity=0)
        
        # Indicator line that stays vertical at the tracker position
        v_indicator = Line(tracker_mobject.get_center(), [ tracker_mobject.get_center()[0], 1.5, 0 ], color=WHITE)
        v_indicator.add_updater(lambda mob: mob.become(
            Line(
                [ tracker_mobject.get_center()[0], -2, 0 ],
                [ tracker_mobject.get_center()[0], 1.5, 0 ],
                color=WHITE, stroke_opacity=0.3
            )
        ))
        
        # 5. Particle Shower Icons
        # Two icons that merge as the tracker moves left
        shower_origin = [ 2, 2.2, 0 ]
        icon_a = Dot(shower_origin, color=YELLOW).scale(3).set_opacity(0.6)
        icon_b = Dot(shower_origin, color=YELLOW).scale(3).set_opacity(0.6)
        
        icon_a.add_updater(lambda mob: mob.move_to(
            [ shower_origin[0] - (tracker_mobject.get_center()[0] + 3) * 0.2, shower_origin[1], 0 ]
        ))
        icon_b.add_updater(lambda mob: mob.move_to(
            [ shower_origin[0] + (tracker_mobject.get_center()[0] + 3) * 0.2, shower_origin[1], 0 ]
        ))
        
        shower_status = Text("Particle Proximity", font_size=20).next_to(icon_a, UP, buff=0.8)

        # 6. Final Animation Sequence
        self.add(tracker_mobject, v_indicator, icon_a, icon_b, shower_status)
        
        # Move tracker from right (distance 6) to left (distance 0)
        self.play(
            tracker_mobject.animate.move_to([ -3, -2, 0 ]),
            run_time=8
        )
        self.wait(2)
