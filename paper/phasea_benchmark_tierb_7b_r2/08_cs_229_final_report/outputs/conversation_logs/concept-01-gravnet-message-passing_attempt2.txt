from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Setup - The formula for distance-weighted connections
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").scale(0.8)
        formula.to_corner(UL)
        self.add(formula)

        # 2. Define geometry
        # Start with an irregular hexagonal-like grid around a central node
        center_pos = np.array([0, 0, 0])
        neighbor_pos_init = [
            np.array([1.8, 0.2, 0]),
            np.array([1.0, 1.6, 0]),
            np.array([-0.8, 1.8, 0]),
            np.array([-1.9, -0.1, 0]),
            np.array([-0.7, -1.7, 0]),
            np.array([1.1, -1.5, 0])
        ]

        # Targets: some move closer (learned relationship), some move farther
        neighbor_pos_final = [
            np.array([0.9, 0.1, 0]),    # closer
            np.array([1.2, 2.5, 0]),    # farther
            np.array([-0.4, 0.8, 0]),   # closer
            np.array([-2.5, -0.5, 0]),  # farther
            np.array([-0.3, -0.7, 0]),  # closer
            np.array([2.5, -2.5, 0])    # farther
        ]

        # 3. Mobject creation functions
        def create_connection(start_point, end_point):
            # Calculate distance
            dist_val = np.linalg.norm(start_point - end_point)
            # GravNet weight
            weight = np.exp(-(dist_val**2))
            
            # Create arrow where opacity and stroke width depend on weight
            # We add a floor to weight for visibility
            stroke = max(weight * 18, 0.5)
            opacity = max(weight, 0.1)
            
            return Arrow(
                start=start_point,
                end=end_point,
                buff=0.15,
                color=GREEN,
                stroke_width=stroke,
                max_tip_length_to_length_ratio=0.2
            ).set_opacity(opacity)

        # 4. Initialize Mobjects
        central_dot = Dot(point=center_pos, radius=0.12, color=RED)
        
        neighbor_dots = VGroup(*[
            Dot(point=p, radius=0.1, color=BLUE) 
            for p in neighbor_pos_init
        ])
        
        arrows_init = VGroup(*[
            create_connection(center_pos, p) 
            for p in neighbor_pos_init
        ])

        # Define targets for the transformation
        neighbor_dots_final = VGroup(*[
            Dot(point=p, radius=0.1, color=BLUE) 
            for p in neighbor_pos_final
        ])
        
        # Central dot moves slightly to show local frame change
        center_pos_final = np.array([0.2, -0.1, 0])
        central_dot_final = Dot(point=center_pos_final, radius=0.12, color=RED)

        arrows_final = VGroup(*[
            create_connection(center_pos_final, p) 
            for p in neighbor_pos_final
        ])

        # 5. Animation Sequence
        # Intro: Show the irregular geometry
        self.play(FadeIn(central_dot), FadeIn(neighbor_dots))
        self.wait(0.5)
        
        # Connection: Show message passing based on initial distance
        self.play(Create(arrows_init), run_time=1.5)
        self.wait(1)

        # Learning Update: Dots move in feature space, connections update dynamically
        self.play(
            Transform(central_dot, central_dot_final),
            Transform(neighbor_dots, neighbor_dots_final),
            Transform(arrows_init, arrows_final),
            run_time=4,
            rate_func=smooth
        )
        
        self.wait(2)
