from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Setup - Formula and irregular grid
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # Points for the irregular hexagonal-like grid (2D projection)
        # Treating every hit as a node in a graph
        points_coords = [
            [0, 0, 0],          # Central node
            [-1.5, 1.2, 0],     # Neighbors
            [1.8, 0.5, 0],
            [0.2, -2.1, 0],
            [-2.2, -0.8, 0],
            [1.1, 2.3, 0],
            [-0.5, 2.5, 0],
            [2.5, -1.5, 0],
            [-1.8, -2.4, 0],
            [2.0, 1.8, 0],
            [-2.5, 0.5, 0]
        ]
        
        dots = VGroup(*[Dot(point=p, radius=0.1, color=BLUE) for p in points_coords])
        center_dot = dots[0]
        
        self.play(Create(dots))
        self.wait(0.5)
        
        # Highlight the central dot
        self.play(
            center_dot.animate.set_color(YELLOW).scale(1.5),
            run_time=1
        )
        self.wait(0.5)
        
        # 2. Connections (Arrows) with dynamic properties
        # The strength (opacity/thickness) scales with exp(-d^2)
        arrows = VGroup()
        
        def get_arrow_weight(p1, p2):
            # Calculate distance d_jk
            dist = np.linalg.norm(p1 - p2)
            # Apply formula: V_jk = exp(-d^2)
            # Using dist/2.0 for better visual range in the 0-10 unit coordinate system
            return np.exp(-(dist / 2.0)**2)

        for i in range(1, len(dots)):
            target_dot = dots[i]
            # Create arrow
            arrow = Arrow(center_dot.get_center(), target_dot.get_center(), buff=0.1, color=WHITE)
            
            # Define an update function that captures the specific target_dot
            def arrow_updater(mob, target=target_dot):
                start = center_dot.get_center()
                end = target.get_center()
                weight = get_arrow_weight(start, end)
                mob.put_start_and_end_on(start, end)
                mob.set_stroke(width=weight * 10, opacity=weight)

            arrow.add_updater(arrow_updater)
            arrows.add(arrow)
            
        self.play(Create(arrows))
        self.wait(1)
        
        # 3. Dynamic Shifting (Simulating updated 'learned features')
        # Show the dots moving and connections updating automatically via updaters
        np.random.seed(42)
        move_animations = []
        for d in dots:
            # Random slight movement to show the graph is not fixed by physical geometry
            random_offset = np.array([
                np.random.uniform(-1.5, 1.5),
                np.random.uniform(-1.5, 1.5),
                0
            ])
            move_animations.append(d.animate.shift(random_offset))
            
        # Execute the dynamic shift
        # Using lambda x: x for linear rate function to avoid 'linear' identifier issues
        self.play(
            *move_animations,
            run_time=5,
            rate_func=lambda x: x
        )
        
        self.wait(2)
