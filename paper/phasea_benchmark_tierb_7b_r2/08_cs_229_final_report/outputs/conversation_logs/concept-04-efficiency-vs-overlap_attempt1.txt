from manim import *
import numpy as np

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # Create axes using parameters similar to reference examples
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 1.1, 0.2],
            x_length=9,
            y_length=5,
            axis_config={"include_tip": False}
        ).shift(DOWN * 0.5)

        # Plot labels using Text (safe and robust)
        x_label = Text("Distance between particles").scale(0.5).next_to(axes.x_axis, DOWN)
        y_label = Text("Efficiency").scale(0.5).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)

        # Plot curves using named functions to avoid linter issues with lambda arguments
        def ml_efficiency(x_val):
            return 0.9

        def standard_efficiency(x_val):
            # Curve that drops to zero as distance (x) closes
            if x_val < 0.05:
                return 0.0
            return 1.0 - np.exp(-0.6 * x_val)

        ml_model_curve = axes.plot(ml_efficiency, color=BLUE, x_range=[0, 10])
        # Standard curve as a solid line since DashedVMobject was flagged in some environments
        standard_algorithm_curve = axes.plot(standard_efficiency, color=RED, x_range=[0, 10])

        # Tags for the curves
        ml_tag = Text("ML Model", color=BLUE).scale(0.4).next_to(ml_model_curve.get_start(), UP)
        std_tag = Text("Standard", color=RED).scale(0.4).next_to(standard_algorithm_curve.get_point_from_proportion(0.5), DR)

        # Add initial static elements
        self.add(axes, x_label, y_label)
        self.play(Create(ml_model_curve), Create(standard_algorithm_curve), Write(ml_tag), Write(std_tag))
        self.wait(1)

        # ValueTracker for distance and particle movement
        # Moving from 10 (far) to 0.2 (near overlap)
        dist_tracker = ValueTracker(10.0)

        # Vertical indicator line using add_updater instead of always_redraw
        v_line = Line(axes.c2p(10, 0), axes.c2p(10, 1), color=YELLOW, stroke_width=2)
        
        def update_vline(mob):
            curr_x = dist_tracker.get_value()
            new_line = Line(axes.c2p(curr_x, 0), axes.c2p(curr_x, 1), color=YELLOW, stroke_width=2)
            mob.become(new_line)
        
        v_line.add_updater(update_vline)

        # Particle shower icons (simplified as dots)
        particle_a = Dot(radius=0.15, color=WHITE)
        particle_b = Dot(radius=0.15, color=WHITE)
        
        # Updaters for particles moving together
        def update_particle_a(mob):
            d = dist_tracker.get_value()
            mob.move_to(UP * 2.8 + LEFT * (d / 4.0))
            
        def update_particle_b(mob):
            d = dist_tracker.get_value()
            mob.move_to(UP * 2.8 + RIGHT * (d / 4.0))

        particle_a.add_updater(update_particle_a)
        particle_b.add_updater(update_particle_b)

        # Add dynamic elements to scene
        self.add(v_line, particle_a, particle_b)
        
        # Animate the distance closing
        # Using lambda t: t for rate_func to avoid potential "linear" undefined errors
        self.play(
            dist_tracker.animate.set_value(0.2),
            run_time=8,
            rate_func=lambda t: t
        )
        self.wait(2)

        # Final visual feedback of the efficiency gap
        gap_line = Line(axes.c2p(0.5, 0.1), axes.c2p(0.5, 0.85), color=YELLOW)
        gap_text = Text("Gap", color=YELLOW).scale(0.4).next_to(gap_line, RIGHT)
        self.play(Create(gap_line), Write(gap_text))
        self.wait(2)
