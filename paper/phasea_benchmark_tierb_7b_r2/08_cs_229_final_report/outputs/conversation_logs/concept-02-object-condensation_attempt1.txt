from manim import *
import numpy as np

class ObjectCondensation(Scene):
    def construct(self):
        # Formula display
        q_min = 0.1
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{\text{min}}").to_edge(UP)
        self.play(Write(formula))

        # Create a 2D plane of dots
        num_dots = 50
        dots = VGroup()
        np.random.seed(42)
        for _ in range(num_dots):
            pos = [np.random.uniform(-5.5, 5.5), np.random.uniform(-3, 2), 0]
            dots.add(Dot(point=pos, radius=0.08, color=PURPLE))
        
        self.play(FadeIn(dots, lag_ratio=0.05))
        self.wait(0.5)

        # High-beta condensation points
        condensation_indices = [10, 25, 40]
        
        color_anims = []
        for i, dot in enumerate(dots):
            if i in condensation_indices:
                dot.beta = 0.92
            else:
                dot.beta = np.random.uniform(0.1, 0.4)
            
            # Map beta to color using internal interpolate_color
            target_color = interpolate_color(PURPLE, YELLOW, dot.beta)
            # Physical scaling based on arctanh charge
            q = np.arctanh(dot.beta) + q_min
            
            color_anims.append(
                dot.animate.set_color(target_color).scale(q * 1.5)
            )
        
        self.play(*color_anims, run_time=2)
        self.wait(0.5)

        # Draw gravity wells
        wells = VGroup()
        for idx in condensation_indices:
            center_pos = dots[idx].get_center()
            well = VGroup(*[
                Circle(radius=r, stroke_width=0, fill_opacity=0.1, fill_color=YELLOW)
                for r in np.linspace(0.2, 2.2, 6)
            ]).move_to(center_pos)
            wells.add(well)
        
        self.play(FadeIn(wells, lag_ratio=0.2))

        # Animate dots being pulled into wells
        pull_anims = []
        for i, dot in enumerate(dots):
            if i in condensation_indices:
                continue
            
            # Distance to the nearest condensation center
            centers = [dots[c].get_center() for c in condensation_indices]
            dists = [np.linalg.norm(dot.get_center() - c) for c in centers]
            min_dist = min(dists)
            nearest_idx = np.argmin(dists)
            nearest_center = centers[nearest_idx]
            
            if min_dist < 4.0:
                # Calculate target position (moving 75% towards center)
                start_pos = dot.get_center()
                target_pos = start_pos + 0.75 * (nearest_center - start_pos)
                pull_anims.append(dot.animate.move_to(target_pos))
            else:
                # Subtle drift for noise
                drift = (dot.get_center() - nearest_center) * 0.1
                pull_anims.append(dot.animate.shift(drift))

        self.play(*pull_anims, run_time=3, rate_func=smooth)
        self.wait(2)

        # Cleanup
        self.play(
            FadeOut(dots), 
            FadeOut(wells), 
            FadeOut(formula)
        )
        self.wait(1)
