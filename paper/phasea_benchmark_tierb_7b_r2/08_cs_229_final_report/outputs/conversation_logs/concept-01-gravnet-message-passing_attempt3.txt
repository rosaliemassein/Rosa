from manim import *
import numpy as np

class Concept01GravnetMessagePassing(Scene):
    def construct(self):
        # 1. Setup formula
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").scale(1.2)
        formula.to_corner(UL)
        self.add(formula)

        # 2. Generate Irregular Grid (ECal Cells) in 2D
        # Using a jittered hexagonal-ish grid layout
        points_group = VGroup()
        rows, cols = 4, 4
        for x in range(rows):
            for y in range(cols):
                pos_x = (x - (rows-1)/2) * 1.5 + (y % 2) * 0.4
                pos_y = (y - (cols-1)/2) * 1.2
                
                # Add jitter for irregularity
                pos = np.array([
                    pos_x + np.random.uniform(-0.3, 0.3),
                    pos_y + np.random.uniform(-0.3, 0.3),
                    0
                ])
                dot = Dot(point=pos, radius=0.1, color=GRAY)
                points_group.add(dot)

        # 3. Identify and Highlight Central Dot
        center_dot = points_group[len(points_group) // 2]
        
        self.play(Create(points_group))
        self.wait(0.5)
        self.play(center_dot.animate.set_color(RED).scale(1.5))
        self.wait(0.5)

        # 4. Create Arrows with Updaters
        # The arrows will dynamically update their thickness and opacity based on distance
        arrows = VGroup()
        for dot in points_group:
            if dot is center_dot:
                continue
            
            # Function to calculate connection weight
            def get_weight(target_dot=dot):
                d = np.linalg.norm(center_dot.get_center() - target_dot.get_center())
                # Normalize distance for visual scaling (scale factor of 0.4)
                return np.exp(-(d * 0.4)**2)

            # Create an arrow that redraws every frame
            arrow = AlwaysRedraw(lambda target_dot=dot: Arrow(
                start=center_dot.get_center(),
                end=target_dot.get_center(),
                buff=0.15,
                color=BLUE,
                stroke_width=max(0.5, 12 * np.exp(-(np.linalg.norm(center_dot.get_center() - target_dot.get_center()) * 0.4)**2)),
                max_tip_length_to_length_ratio=0.15
            ).set_opacity(np.exp(-(np.linalg.norm(center_dot.get_center() - target_dot.get_center()) * 0.4)**2)))
            
            arrows.add(arrow)

        self.play(Create(arrows, lag_ratio=0.05, run_time=2))
        self.wait(1)

        # 5. Animate "Learned Features" Update
        # Moving dots slightly to show how connections change dynamically
        movements = []
        for dot in points_group:
            # Random target position nearby
            target_pos = dot.get_center() + np.array([
                np.random.uniform(-1.5, 1.5),
                np.random.uniform(-1.5, 1.5),
                0
            ])
            movements.append(dot.animate.move_to(target_pos))

        # Highlight formula while points move
        self.play(
            *movements,
            formula.animate.set_color(YELLOW),
            run_time=4,
            rate_func=smooth
        )
        
        self.wait(2)

        # Clean up
        self.play(FadeOut(arrows), FadeOut(points_group), FadeOut(formula))
        self.wait(1)
