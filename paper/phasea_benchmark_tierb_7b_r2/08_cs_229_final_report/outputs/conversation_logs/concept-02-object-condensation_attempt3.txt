from manim import *
import numpy as np

class ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup Grid of Dots
        # Using standard PURPLE color to ensure compatibility
        base_color = PURPLE
        high_color = YELLOW
        
        dots = VGroup()
        rows, cols = 7, 11
        for x in np.linspace(-5, 5, cols):
            for y in np.linspace(-3, 3, rows):
                dots.add(Dot(point=[x, y, 0], radius=0.08, color=base_color))
        
        self.play(FadeIn(dots, lag_ratio=0.05), run_time=1.5)
        self.wait(0.5)

        # 2. Pick Condensation Points (indices for the grid)
        cp1_idx = 38  # Center-ish left
        cp2_idx = 46  # Center-ish right
        
        # 3. Animate Beta Growth (Color and Size)
        # We simulate beta values based on proximity to condensation points
        beta_animations = []
        for dot in dots:
            p = dot.get_center()
            dist1 = np.linalg.norm(p - dots[cp1_idx].get_center())
            dist2 = np.linalg.norm(p - dots[cp2_idx].get_center())
            
            # Simple Gaussian-like beta value
            beta = np.exp(-dist1**2 / 2.0) + np.exp(-dist2**2 / 1.5)
            beta = min(max(beta, 0.05), 0.95)
            
            # Interpolate color from PURPLE to YELLOW based on beta
            target_color = interpolate_color(base_color, high_color, beta)
            beta_animations.append(
                dot.animate.set_color(target_color).scale(1 + beta * 1.5)
            )
        
        self.play(AnimationGroup(*beta_animations), run_time=2)
        self.wait(0.5)

        # 4. Draw Gravity Wells (Concentric Circles)
        wells = VGroup()
        for idx in [cp1_idx, cp2_idx]:
            center = dots[idx].get_center()
            # Create a "well" effect using multiple circles with varying opacity
            well = VGroup(*[
                Circle(radius=r, stroke_width=0, fill_opacity=0.2 - r/10, fill_color=YELLOW)
                for r in np.linspace(0.1, 1.5, 8)
            ]).move_to(center)
            wells.add(well)
        
        self.play(FadeIn(wells, scale=0.5), run_time=1)

        # 5. Display the Formula
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}", color=WHITE)
        formula.to_edge(UP)
        self.play(Write(formula))
        self.wait(0.5)

        # 6. Physical Attraction Animation
        # Pull neighboring dots toward the condensation points to simulate clustering
        pull_animations = []
        for i, dot in enumerate(dots):
            if i == cp1_idx or i == cp2_idx:
                continue
            
            p = dot.get_center()
            d1 = np.linalg.norm(p - dots[cp1_idx].get_center())
            d2 = np.linalg.norm(p - dots[cp2_idx].get_center())
            
            # If within range of a condensation point, pull it in
            if d1 < 2.0:
                target = dots[cp1_idx].get_center()
                pull_animations.append(dot.animate.move_to(p + (target - p) * 0.5))
            elif d2 < 2.0:
                target = dots[cp2_idx].get_center()
                pull_animations.append(dot.animate.move_to(p + (target - p) * 0.5))

        # Using 'smooth' rate function for stable animation
        self.play(AnimationGroup(*pull_animations), run_time=2, rate_func=smooth)
        
        # Highlight a charge indicator
        charge_label = MathTex(r"q \text{ acts as charge}", color=YELLOW, font_size=32)
        charge_label.next_to(dots[cp1_idx], DOWN, buff=0.5)
        self.play(Indicate(dots[cp1_idx]), Write(charge_label))
        
        self.wait(2)
