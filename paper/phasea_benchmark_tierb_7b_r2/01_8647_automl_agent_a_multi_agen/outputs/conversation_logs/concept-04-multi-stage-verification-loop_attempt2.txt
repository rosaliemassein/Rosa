from manim import *

class Concept04MultiStageVerificationLoop(Scene):
    def construct(self):
        # Formula at the top for reference
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=36
        ).to_edge(UP, buff=0.5)
        self.add(formula)

        # Coordinate positions
        planning_pos = LEFT * 5
        req_pos = LEFT * 2
        exec_pos = ORIGIN
        imp_pos = RIGHT * 2
        deploy_pos = RIGHT * 5

        # Planning Stage
        planning_box = Rectangle(width=1.8, height=1.8, color=GRAY).move_to(planning_pos)
        planning_label = Text("Planning Stage", font_size=20).next_to(planning_box, DOWN)
        self.add(planning_box, planning_label)

        # Three Gates (Vertical Lines)
        gate_height = 2.5
        req_gate = Line(UP, DOWN).scale(gate_height / 2).move_to(req_pos).set_color(BLUE)
        exec_gate = Line(UP, DOWN).scale(gate_height / 2).move_to(exec_pos).set_color(BLUE)
        imp_gate = Line(UP, DOWN).scale(gate_height / 2).move_to(imp_pos).set_color(BLUE)

        req_label = Text("ReqVer", font_size=20).next_to(req_gate, UP)
        exec_label = Text("ExecVer", font_size=20).next_to(exec_gate, UP)
        imp_label = Text("ImpVer", font_size=20).next_to(imp_gate, UP)

        self.add(req_gate, exec_gate, imp_gate, req_label, exec_label, imp_label)

        # Deployment Icon
        deploy_icon = Circle(radius=0.5, color=GREEN, fill_opacity=0.3).move_to(deploy_pos)
        deploy_label = Text("Deployment", font_size=20, color=GREEN).next_to(deploy_icon, DOWN)

        # The Plan object (a small square)
        plan = Square(side_length=0.4, fill_opacity=1, color=WHITE).move_to(planning_pos)
        self.add(plan)

        # --- ANIMATION SEQUENCE ---

        # 1. Plan moves to ReqVer and Passes
        self.play(plan.animate.move_to(req_pos), run_time=1.5)
        self.play(req_gate.animate.set_color(GREEN), plan.animate.set_color(GREEN), run_time=0.3)
        self.play(plan.animate.set_color(WHITE), run_time=0.2)

        # 2. Plan moves to ExecVer and Fails
        self.play(plan.animate.move_to(exec_pos), run_time=1.2)
        self.play(exec_gate.animate.set_color(RED), plan.animate.set_color(RED), run_time=0.3)

        # 3. Draw Feedback Loop (Curved Arrow replacement)
        # Using ArcBetweenPoints and add_tip to avoid undefined identifier issues
        feedback_path = ArcBetweenPoints(
            start=exec_pos + UP * 0.5, 
            end=planning_pos + UP * 0.5, 
            angle=PI / 2
        ).set_color(RED)
        feedback_path.add_tip()
        
        feedback_text = Text("Feedback / Revise", font_size=18, color=RED).next_to(feedback_path, UP)
        
        self.play(Create(feedback_path), FadeIn(feedback_text))
        self.play(plan.animate.move_to(planning_pos), run_time=1.5)
        self.play(FadeOut(feedback_path), FadeOut(feedback_text), plan.animate.set_color(WHITE))
        
        # Reset gate color for the next attempt
        self.play(exec_gate.animate.set_color(BLUE), run_time=0.2)

        # 4. Final Success Run through all gates
        # Move through ReqVer
        self.play(plan.animate.move_to(req_pos), run_time=0.8)
        # Move through ExecVer
        self.play(plan.animate.move_to(exec_pos), run_time=0.8)
        self.play(exec_gate.animate.set_color(GREEN), run_time=0.2)
        # Move to ImpVer
        self.play(plan.animate.move_to(imp_pos), run_time=0.8)
        self.play(imp_gate.animate.set_color(GREEN), plan.animate.set_color(GREEN), run_time=0.3)

        # 5. Reach Deployment
        self.play(plan.animate.move_to(deploy_pos), FadeIn(deploy_icon, deploy_label), run_time=1.2)
        self.play(plan.animate.scale(0), deploy_icon.animate.scale(1.2))

        self.wait(2)
