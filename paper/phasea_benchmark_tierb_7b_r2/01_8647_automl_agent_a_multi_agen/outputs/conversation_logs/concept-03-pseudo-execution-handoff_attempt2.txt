from manim import *

class Concept03PseudoExecutionHandoff(Scene):
    def construct(self):
        # 1. Setup Agents and Blocks (Split-screen layout)
        # Top half: Data Agent
        data_agent = Text("Data Agent (A_d)", color=BLUE).scale(0.8).to_edge(UP, buff=0.5).shift(LEFT * 3)
        data_block = Rectangle(width=3, height=1.2, color=GREEN, fill_opacity=0.2).next_to(data_agent, RIGHT, buff=0.5)
        data_label = Text("Data Block", font_size=24).move_to(data_block)
        
        # Bottom half: Model Agent
        model_agent = Text("Model Agent (A_m)", color=RED).scale(0.8).to_edge(DOWN, buff=2.0).shift(LEFT * 3)
        model_task = Rectangle(width=2.5, height=1, color=ORANGE, fill_opacity=0.2).next_to(model_agent, RIGHT, buff=0.5)
        model_task_label = Text("s_m", font_size=24).move_to(model_task)
        
        # Elements for the hand-off
        od_insight = MathTex("O_d", color=YELLOW).scale(1.2).move_to(data_block.get_center())
        
        # 2. Search Efficiency Bar (Right side)
        efficiency_title = Text("Search Efficiency", font_size=24).to_edge(RIGHT).shift(UP * 1.5 + LEFT * 0.5)
        bar_bg = Rectangle(width=3, height=0.4, color=WHITE).next_to(efficiency_title, DOWN, buff=0.3)
        # We use a rectangle and stretch it instead of a ValueTracker to avoid environment constraints
        bar_fill = Rectangle(
            width=0.01, 
            height=0.3, 
            fill_opacity=1, 
            color=GREEN, 
            stroke_width=0
        ).align_to(bar_bg, LEFT).shift(RIGHT * 0.05)
        
        # 3. Formula
        formula = MathTex(
            r"s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)", 
            font_size=32
        ).next_to(bar_bg, DOWN, buff=1.5)

        # --- Animation Sequence ---
        
        # Step 1: Initialize Agents
        self.play(
            FadeIn(data_agent), FadeIn(data_block), FadeIn(data_label),
            FadeIn(model_agent), FadeIn(model_task), FadeIn(model_task_label)
        )
        self.wait(1)
        
        # Step 2: Data Agent performs 'pseudo-analysis' to generate O_d
        self.play(
            data_label.animate.set_opacity(0.3),
            Write(od_insight)
        )
        self.wait(0.5)
        
        # Step 3: Information Handoff
        handoff_arrow = Arrow(
            data_block.get_bottom(), 
            model_task.get_top(), 
            color=YELLOW, 
            buff=0.1
        )
        self.play(GrowArrow(handoff_arrow))
        self.play(
            od_insight.animate.move_to(model_task.get_center()).scale(0.8),
            model_task_label.animate.shift(UP * 0.7)
        )
        self.play(FadeOut(handoff_arrow))
        
        # Step 4: Model Agent reacts/expands sub-tasks based on O_d
        self.play(
            model_task.animate.scale(1.2).set_stroke(YELLOW, width=5),
            model_task.animate.set_fill(ORANGE, opacity=0.4)
        )
        self.wait(0.5)
        
        # Step 5: Visualize Training-free search and efficiency increase
        self.play(
            FadeIn(efficiency_title),
            FadeIn(bar_bg),
            FadeIn(bar_fill)
        )
        
        # Animate bar growth and formula appearance simultaneously
        # Using standard 'smooth' rate function to ensure compatibility
        self.play(
            bar_fill.animate.stretch_to_fit_width(2.9, about_edge=LEFT),
            Write(formula),
            run_time=3,
            rate_func=smooth
        )
        
        self.wait(2)
