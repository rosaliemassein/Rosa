from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # Formula at the top
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}"
        ).to_edge(UP)
        self.add(formula)

        # Planning Stage Position
        planning_pos = LEFT * 5 + DOWN * 0.5
        planning_label = Text("Planning", font_size=24).next_to(planning_pos, DOWN)
        
        # Define the gates (ReqVer, ExecVer, ImpVer)
        gate_names = ["ReqVer", "ExecVer", "ImpVer"]
        gates = VGroup()
        labels = VGroup()
        for i, name in enumerate(gate_names):
            # Create vertical lines as gates
            gate = Line(UP, DOWN, color=WHITE).scale(1.5).shift(LEFT * 2 + i * 2.5 * RIGHT + DOWN * 0.5)
            label = Text(name, font_size=24).next_to(gate, UP)
            gates.add(gate)
            labels.add(label)

        # Deployment Icon (Placeholder for img-2.jpeg to ensure compilation)
        deployment_box = RoundedRectangle(height=1.2, width=2, color=GOLD).shift(RIGHT * 5.5 + DOWN * 0.5)
        deployment_text = Text("Deployment", font_size=20, color=GOLD).move_to(deployment_box.get_center())
        deployment = VGroup(deployment_box, deployment_text)

        self.play(Create(gates), Write(labels), Create(planning_label), FadeIn(deployment))

        # The 'Plan' object (a small square)
        plan = Square(side_length=0.4, fill_opacity=1, color=BLUE).move_to(planning_pos)
        self.play(FadeIn(plan))

        # --- ANIMATION SEQUENCE ---

        # 1. First Gate: FAIL scenario
        # Move to ReqVer
        target_gate_1 = gates[0].get_center()
        self.play(plan.animate.move_to(target_gate_1))
        
        # Turn red (fail)
        self.play(plan.animate.set_color(RED))
        
        # Curved Arrow looping back to the 'Planning' stage
        # Using ArcBetweenPoints and add_tip as a robust alternative to CurvedArrow
        loop_start = plan.get_top()
        loop_end = planning_pos + UP * 0.5
        loop_arrow = ArcBetweenPoints(start=loop_start, end=loop_end, angle=-TAU/3, color=RED).add_tip()
        feedback_text = Text("Feedback / Revise", font_size=18, color=RED).next_to(loop_arrow, UP)
        
        self.play(Create(loop_arrow), Write(feedback_text))
        
        # Move back and reset
        self.play(plan.animate.move_to(planning_pos).set_color(BLUE))
        self.play(FadeOut(loop_arrow), FadeOut(feedback_text))
        self.wait(0.5)

        # 2. SUCCESS scenario through all gates
        gate_positions = [g.get_center() for g in gates]
        
        # Move through ReqVer (Pass)
        self.play(plan.animate.move_to(gate_positions[0]))
        self.play(plan.animate.set_color(GREEN))
        
        # Move through ExecVer (Pass)
        self.play(plan.animate.move_to(gate_positions[1]))
        self.play(Flash(gate_positions[1], color=GREEN, flash_radius=0.5))
        
        # Move through ImpVer (Pass)
        self.play(plan.animate.move_to(gate_positions[2]))
        self.play(Flash(gate_positions[2], color=GREEN, flash_radius=0.5))

        # Reaches Deployment
        final_pos = deployment.get_left() + RIGHT * 0.5
        self.play(plan.animate.move_to(final_pos))
        
        # Final transition to indicate success
        self.play(
            plan.animate.scale(1.2).set_opacity(0),
            deployment.animate.set_fill(GOLD, opacity=0.3),
            run_time=1
        )

        self.wait(2)
