from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Custom Axes Construction (to avoid Axes/Table restrictions)
        origin = LEFT * 4 + DOWN * 2.5
        x_axis = Line(origin, origin + RIGHT * 6, color=WHITE)
        y_axis = Line(origin, origin + UP * 5.5, color=WHITE)
        
        x_label = Text("Grid Size (N)", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("Possible States (log)", font_size=20).rotate(90*DEGREES).next_to(y_axis, LEFT)
        
        # Ticks for X-axis (2 to 6)
        ticks = VGroup()
        for i in range(5):
            p = origin + RIGHT * (i + 1) * 1.2
            tick = Line(p + DOWN*0.1, p + UP*0.1)
            lbl = Text(str(i + 2), font_size=16).next_to(tick, DOWN)
            ticks.add(tick, lbl)

        # 2. Formula Reference
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP)
        
        # 3. Curve Construction (Manual VMobject)
        # We simulate 2^(N^2) growth visually
        curve = VMobject(color=BLUE)
        points = [
            origin + RIGHT * (t * 1.2) + UP * (0.1 * np.exp(t * 0.8))
            for t in np.linspace(0, 4.5, 30)
        ]
        curve.set_points_smoothly(points)
        
        # 4. Q-Table Visual (Manual)
        table_outline = Rectangle(height=4, width=3, color=WHITE).to_edge(RIGHT, buff=1)
        table_title = Text("Q-Table", font_size=24).next_to(table_outline, UP)
        
        # Rows representing states
        rows = VGroup(*[
            Line(table_outline.get_left() + RIGHT*0.2, table_outline.get_right() - RIGHT*0.2)
            for _ in range(15)
        ]).arrange(DOWN, buff=0.15).move_to(table_outline)
        
        headers = Text("State | Action | Q", font_size=14).next_to(rows, UP, buff=0.2)
        
        # 5. Animations
        self.play(Write(formula))
        self.play(Create(x_axis), Create(y_axis), FadeIn(x_label), FadeIn(y_label), Create(ticks))
        self.play(Create(table_outline), Write(table_title), FadeIn(headers), Create(rows))
        self.wait(1)
        
        # 6. Explosion Effect
        # Create a "blur" of rows to represent 2^25+ states
        dense_rows = VGroup(*[
            Line(table_outline.get_left() + RIGHT*0.1, table_outline.get_right() - RIGHT*0.1, stroke_width=0.5, stroke_opacity=0.3)
            for _ in range(100)
        ]).arrange(DOWN, buff=0.02).move_to(table_outline)
        
        blur_box = Rectangle(
            height=3.8, width=2.8, 
            fill_color=BLUE, fill_opacity=0.5, 
            stroke_width=0
        ).move_to(table_outline)
        
        # Tracker dot for the curve
        dot = Dot(color=YELLOW).move_to(points[0])
        
        # Animation sequence
        self.play(
            Create(curve, run_time=4, rate_func=smooth),
            UpdateFromAlpha(dot, lambda m, a: m.move_to(curve.point_from_proportion(a)), run_time=4, rate_func=smooth),
            ReplacementTransform(rows, dense_rows, run_time=2),
            FadeIn(blur_box, run_time=3),
        )
        
        # Final "Skyrocket" effect
        self.play(
            curve.animate.shift(UP * 15),
            dot.animate.shift(UP * 15),
            dense_rows.animate.shift(UP * 10),
            blur_box.animate.set_opacity(0.8),
            run_time=1.5,
            rate_func=smooth
        )
        
        self.wait(2)
