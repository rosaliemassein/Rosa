from manim import *

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Title and Formula
        # Using basic objects and avoiding keywords like 'x_range' or names like 'x'
        title_text = Text("State Space Explosion").to_edge(UP)
        formula_tex = MathTex(r"\text{Total States} = 2^{(N \cdot N)}")
        formula_tex.next_to(title_text, DOWN, buff=0.2)
        self.add(title_text, formula_tex)

        # 2. Manual Coordinate System
        # We avoid the forbidden Axes object entirely by using Line
        coord_origin = [ -3, -2.5, 0 ]
        horizontal_axis = Line(coord_origin, [ 3.5, -2.5, 0 ])
        vertical_axis = Line(coord_origin, [ -3, 2.5, 0 ])
        self.add(horizontal_axis, vertical_axis)
        
        label_size = Text("Matrix Size N").scale(0.5).next_to(horizontal_axis, DOWN)
        label_states = Text("Possible States (Log Scale)").scale(0.5).rotate(90*DEGREES).next_to(vertical_axis, LEFT)
        self.add(label_size, label_states)

        # 3. Small grid graphic for 2x2
        # Building the grid manually to show the initial state
        grid_visual = VGroup()
        for row_k in range(2):
            for col_k in range(2):
                cell_box = Square(side_length=0.4, fill_opacity=0.5, color=BLUE)
                grid_visual.add(cell_box)
        grid_visual.arrange_in_grid(2, 2, buff=0.1).to_corner(UL, buff=1.5)
        self.play(FadeIn(grid_visual))

        # 4. Table rows representation
        # Using white lines to represent entries in a Q-table
        rows_visual = VGroup()
        for row_k in range(6):
            row_line = Line(LEFT, RIGHT, color=WHITE).set_width(1.2)
            rows_visual.add(row_line)
        rows_visual.arrange(DOWN, buff=0.15).next_to(grid_visual, DOWN, buff=0.8)
        self.play(Create(rows_visual))

        # 5. Graph curve
        # We represent 2^(N^2) on a log scale by plotting a parabola
        # We manually calculate points to avoid 'axes.plot' and the forbidden 'x' identifier
        growth_curve = VMobject(color=YELLOW)
        curve_coordinates = []
        # Calculate 50 points along the path
        for step_k in range(51):
            # n_val varies from 2.0 to 6.2 (shooting off the top)
            n_val = 2.0 + (step_k / 50.0) * 4.2
            # Horizontal position based on n_val
            pos_h = coord_origin[0] + (n_val - 2.0) * 1.5
            # Vertical position based on square of n_val (exponential in log-space)
            pos_v = coord_origin[1] + (n_val * n_val * 0.18)
            curve_coordinates.append([pos_h, pos_v, 0])
        
        growth_curve.set_points_as_corners(curve_coordinates)
        
        # 6. Prepare 5x5 grid for transformation
        grid_large = VGroup()
        for row_k in range(5):
            for col_k in range(5):
                cell_box = Square(side_length=0.15, fill_opacity=0.5, color=RED)
                grid_large.add(cell_box)
        grid_large.arrange_in_grid(5, 5, buff=0.05).move_to(grid_visual)
        
        # 7. Animation sequence
        # We animate the curve being drawn while the grid expands
        self.play(
            Create(growth_curve),
            Transform(grid_visual, grid_large),
            run_time=4
        )
        
        # 8. Represent state space explosion with a blur
        # The table rows move fast and turn into a solid block of color
        explosion_blur = Rectangle(
            width=1.4, height=2.5, 
            fill_color=BLUE, fill_opacity=0.7, 
            stroke_width=0
        ).move_to(rows_visual).shift(DOWN * 0.5)
        
        self.play(
            FadeIn(explosion_blur),
            rows_visual.animate.shift(UP * 3).set_opacity(0),
            run_time=1
        )
        
        # 9. Concluding text for the 5x5 grid calculation
        summary_text = Text("2^25 > 33 Million States", color=YELLOW).scale(0.5)
        summary_text.next_to(explosion_blur, DOWN)
        self.play(Write(summary_text))
        
        self.wait(2)
