import numpy as np
from manim import *

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Manual Creation of Axes
        origin = LEFT * 4 + DOWN * 2.5
        x_axis = Line(origin, origin + RIGHT * 6, color=WHITE)
        y_axis = Line(origin, origin + UP * 5.5, color=WHITE)
        
        x_label = Text("Matrix Size", font_size=24).next_to(x_axis, DOWN)
        y_label = Text("Possible States (log scale)", font_size=24).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))

        # 2. Plotting the function manually (Log scale plot: y = x^2)
        def get_coords(x_val):
            # Scale x from [2, 6] to [0, 5] units
            # Scale y (x^2) from [4, 36] to [0, 5] units
            x_coord = (x_val - 2) * (5 / 4)
            y_coord = (x_val**2 - 4) * (5 / 32)
            return origin + RIGHT * x_coord + UP * y_coord

        graph_points = [get_coords(x) for x in np.linspace(2, 6, 40)]
        graph = VGroup()
        for i in range(len(graph_points) - 1):
            graph.add(Line(graph_points[i], graph_points[i+1], color=BLUE))
        
        # 3. Representing the State Space with a Grid
        grid_2x2 = VGroup(*[
            Square(side_length=0.6).move_to(RIGHT * 3 + UP * (1 - i*0.6) + RIGHT * (j*0.6))
            for i in range(2) for j in range(2)
        ])
        grid_label = Text("2x2 Grid (16 states)", font_size=20).next_to(grid_2x2, UP)
        
        self.play(Create(grid_2x2), Write(grid_label))
        self.play(Create(graph), run_time=3)

        # 4. Transition to high-dimensional state explosion (5x5 grid)
        grid_5x5 = VGroup(*[
            Square(side_length=0.25).move_to(RIGHT * 3 + UP * (1 - i*0.25) + RIGHT * (j*0.25))
            for i in range(5) for j in range(5)
        ])
        
        new_label = Text("5x5 Grid (2^25 states)", font_size=20).next_to(grid_5x5, UP)
        self.play(
            ReplacementTransform(grid_2x2, grid_5x5),
            ReplacementTransform(grid_label, new_label)
        )

        # 5. The "Explosion" - many scrolling rows
        rows = VGroup(*[
            Rectangle(height=0.05, width=2.5, fill_opacity=0.8, fill_color=BLUE, stroke_width=0.5)
            for _ in range(30)
        ]).arrange(DOWN, buff=0.05).move_to(grid_5x5.get_center())

        blur_box = Rectangle(
            height=4, width=3, 
            fill_opacity=0.9, 
            fill_color=BLUE,
            stroke_width=0
        ).move_to(grid_5x5.get_center())

        self.play(
            FadeOut(grid_5x5),
            FadeIn(rows),
            run_time=0.5
        )

        # 6. Animation: Scroll rows, show blur, graph shoots off
        # Using lambda for rate functions to avoid potential undefined issues
        self.play(
            rows.animate.shift(UP * 4),
            FadeIn(blur_box),
            graph.animate.shift(UP * 8),
            run_time=2,
            rate_func=lambda t: t**3 # Exponential-like acceleration
        )

        # 7. Final Formula
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP)
        self.play(Write(formula))
        
        self.wait(2)
