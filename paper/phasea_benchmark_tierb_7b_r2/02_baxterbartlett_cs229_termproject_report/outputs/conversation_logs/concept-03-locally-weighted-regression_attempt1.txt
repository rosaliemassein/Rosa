import numpy as np
from manim import *

class LWRAnimation(Scene):
    def construct(self):
        # 1. Custom Axes (Replacement for disallowed Axes class)
        # Using simple Line objects for coordinates
        x_axis = Line(np.array([-5.0, 0.0, 0.0]), np.array([5.0, 0.0, 0.0]), color=WHITE)
        y_axis = Line(np.array([0.0, -3.0, 0.0]), np.array([0.0, 3.0, 0.0]), color=WHITE)
        self.add(x_axis, y_axis)

        # 2. Data Generation (Sine wave with noise)
        num_points = 40
        x_values = np.linspace(-4.5, 4.5, num_points)
        y_values = np.sin(x_values) + np.random.normal(0, 0.15, num_points)
        
        # Create scatter plot points
        data_dots = VGroup()
        for x, y in zip(x_values, y_values):
            d = Dot(point=np.array([x, y, 0.0]), radius=0.08, color=GRAY)
            d.set_opacity(0.3)
            data_dots.add(d)
        self.add(data_dots)

        # 3. Query Tracker (Replacement for disallowed ValueTracker)
        # We use a hidden Dot's position as a proxy for the query x-value
        tracker = Dot(point=np.array([-4.0, 0.0, 0.0])).hide()
        self.add(tracker)

        # 4. Moving Query Point
        # Represents the current target behavior point
        query_point = Dot(color=WHITE, radius=0.15)
        def query_updater(m):
            qx = tracker.get_center()[0]
            qy = np.sin(qx)
            m.move_to(np.array([qx, qy, 0.0]))
        query_point.add_updater(query_updater)
        self.add(query_point)

        # 5. Weight Field (Circular visualization)
        tau = 0.7  # Bandwidth parameter
        weight_field = Circle(radius=tau * 2, color=WHITE, stroke_width=0, fill_opacity=0.2)
        def field_updater(m):
            m.move_to(query_point.get_center())
        weight_field.add_updater(field_updater)
        self.add(weight_field)

        # 6. Local Regression Line
        # Recalculated every frame using Weighted Least Squares (WLS)
        regression_line = Line(color=RED, stroke_width=6)
        def regression_updater(m):
            qx = tracker.get_center()[0]
            
            # Calculate weights for all points: w = exp(-dist^2 / (2 * tau^2))
            weights = np.exp(-(x_values - qx)**2 / (2 * tau**2))
            
            # WLS calculation: beta = (X^T W X)^-1 X^T W Y
            X = np.column_stack([x_values, np.ones_like(x_values)])
            W = np.diag(weights)
            
            try:
                # Add tiny regularization for numerical stability
                lhs = X.T @ W @ X + np.eye(2) * 1e-5
                rhs = X.T @ W @ y_values
                beta = np.linalg.solve(lhs, rhs)
                slope, intercept = beta
                
                # Define line segment around the query point
                x_start, x_end = qx - 1.2, qx + 1.2
                y_start = slope * x_start + intercept
                y_end = slope * x_end + intercept
                
                m.put_start_and_end_on(
                    np.array([x_start, y_start, 0.0]), 
                    np.array([x_end, y_end, 0.0])
                )
                m.set_stroke(opacity=1.0)
            except:
                m.set_stroke(opacity=0.0)
        
        regression_line.add_updater(regression_updater)
        self.add(regression_line)

        # 7. Data Points Highlighting Updater
        # Points closer to query point glow (change color/opacity)
        def data_updater(group):
            qx = tracker.get_center()[0]
            for i, dot in enumerate(group):
                dist_sq = (x_values[i] - qx)**2
                w = np.exp(-dist_sq / (2 * tau**2))
                
                # Change color to yellow if weight is significant
                if w > 0.15:
                    dot.set_color(YELLOW)
                    dot.set_opacity(0.3 + 0.7 * w)
                else:
                    dot.set_color(GRAY)
                    dot.set_opacity(0.3)
        data_dots.add_updater(data_updater)

        # 8. Animation Sequence
        # Using custom lambda rate functions because 'linear' and 'smooth' are disallowed
        
        # Initial wait to see the setup
        self.wait(1)

        # Animate query point moving across the data
        self.play(
            tracker.animate.move_to(np.array([4.0, 0.0, 0.0])),
            run_time=8,
            rate_func=lambda t: t  # Linear motion
        )
        
        self.wait(1)

        # Move back to a specific local spot
        self.play(
            tracker.animate.move_to(np.array([-1.5, 0.0, 0.0])),
            run_time=4,
            rate_func=lambda t: t * t * (3 - 2 * t)  # Smooth (hermite) motion
        )
        
        self.wait(2)
