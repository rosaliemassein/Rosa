from manim import *
import numpy as np

class InverseMapping(Scene):
    def construct(self):
        # 1. Formula Reference (Top)
        formula = MathTex(
            r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}",
            font_size=32
        ).to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Setup the Diagram (Split Screen)
        # Input Box: Desired Behavior
        input_rect = Rectangle(height=1.6, width=4.5, color=BLUE)
        input_rect.shift(LEFT * 3.5 + UP * 0.8)
        input_title = Text("Desired Behavior", font_size=22, color=BLUE).move_to(input_rect.get_center())
        input_group = VGroup(input_rect, input_title)

        # Output Box: PID Constants
        output_rect = Rectangle(height=1.6, width=4.5, color=GREEN)
        output_rect.shift(RIGHT * 3.5 + UP * 0.8)
        output_title = Text("PID Constants", font_size=22, color=GREEN).move_to(output_rect.get_center())
        output_group = VGroup(output_rect, output_title)

        # Large Arrow between them
        connector = Arrow(input_rect.get_right(), output_rect.get_left(), buff=0.2, stroke_width=5)
        
        diagram = VGroup(input_group, output_group, connector)

        # Initial Diagram Animation
        self.play(Create(input_group), Create(output_group))
        self.play(GrowArrow(connector))
        self.wait(1)

        # "Flips the Script": Rotate the entire diagram 180 degrees
        self.play(diagram.animate.rotate(PI))
        self.wait(1)

        # 3. Motor Velocity Plot (Manual Axes using Line)
        plot_origin = DOWN * 2.5 + LEFT * 4.5
        x_axis = Line(plot_origin, plot_origin + RIGHT * 9, color=WHITE)
        y_axis = Line(plot_origin, plot_origin + UP * 2.5, color=WHITE)
        
        x_label = Text("Time", font_size=18).next_to(x_axis, DOWN, buff=0.2)
        y_label = Text("RPM", font_size=18).next_to(y_axis, LEFT, buff=0.2)

        self.play(Create(x_axis), Create(y_axis))
        self.play(Write(x_label), Write(y_label))

        # Setpoint line (Manual dashes)
        setpoint_height = 1.5
        dashes = VGroup()
        for i in range(25):
            d_start = plot_origin + RIGHT * (i * 0.35) + UP * setpoint_height
            d_end = plot_origin + RIGHT * (i * 0.35 + 0.15) + UP * setpoint_height
            dashes.add(Line(d_start, d_end, color=WHITE, stroke_width=2))
        
        self.play(Create(dashes))

        # Motor Velocity Curve (Manual VMobject points)
        # Function: y = Setpoint * (1 - e^(-0.4x) * cos(3x))
        curve = VMobject(color=YELLOW)
        curve_points = []
        for j in range(91):
            px = j * 0.1
            py = setpoint_height * (1.0 - np.exp(-0.4 * px) * np.cos(3.0 * px))
            curve_points.append(plot_origin + RIGHT * px + UP * py)
        
        curve.set_points_as_corners(curve_points)
        self.play(Create(curve, run_time=3))

        # 4. Indicators (Braces/Labels)
        # Rise Time indicator (from 0 to first setpoint crossing ~0.55s)
        rt_x = 0.55
        rt_indicator = Line(plot_origin + DOWN * 0.3, plot_origin + RIGHT * rt_x + DOWN * 0.3, color=YELLOW)
        rt_tick1 = Line(plot_origin + DOWN * 0.4, plot_origin + DOWN * 0.2, color=YELLOW)
        rt_tick2 = Line(plot_origin + RIGHT * rt_x + DOWN * 0.4, plot_origin + RIGHT * rt_x + DOWN * 0.2, color=YELLOW)
        rt_label = Text("Rise Time", font_size=16, color=YELLOW).next_to(rt_indicator, DOWN, buff=0.1)

        # Steady State Error indicator (at the tail end)
        err_x = 8.5
        err_y = setpoint_height * (1.0 - np.exp(-0.4 * err_x) * np.cos(3.0 * err_x))
        err_indicator = Line(
            plot_origin + RIGHT * err_x + UP * err_y, 
            plot_origin + RIGHT * err_x + UP * setpoint_height, 
            color=RED, 
            stroke_width=4
        )
        err_label = Text("Steady State Error", font_size=14, color=RED).next_to(err_indicator, RIGHT, buff=0.1)

        self.play(Create(rt_indicator), Create(rt_tick1), Create(rt_tick2), Write(rt_label))
        self.play(Create(err_indicator), Write(err_label))

        self.wait(3)
