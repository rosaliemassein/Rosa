from manim import *
import numpy as np

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Manual Title (Avoid using Title class)
        header = Text("Parameter vs. Data Bottleneck", font_size=32)
        header.to_edge(UP, buff=0.5)
        self.add(header)

        # 2. Manual Plotting Framework (Avoid using Axes class)
        # Left Plot Area (Neural Network)
        origin_l = np.array([-5, 0, 0])
        x_axis_l = Line(origin_l, origin_l + RIGHT * 4, color=WHITE)
        y_axis_l = Line(origin_l, origin_l + UP * 2.5, color=WHITE)
        label_l = Text("Neural Network: High Variance", font_size=18, color=RED).next_to(y_axis_l, UP, buff=0.2)

        # Right Plot Area (LWR)
        origin_r = np.array([1, 0, 0])
        x_axis_r = Line(origin_r, origin_r + RIGHT * 4, color=WHITE)
        y_axis_r = Line(origin_r, origin_r + UP * 2.5, color=WHITE)
        label_r = Text("LWR: Robust Smoothness", font_size=18, color=GREEN).next_to(y_axis_r, UP, buff=0.2)

        # 3. Data Points (Representing 27 small data points)
        # We manually define a few key points to show the trend
        point_coords = [
            [0.5, 0.4], [1.0, 1.2], [1.5, 0.8], [2.0, 1.8], 
            [2.5, 1.1], [3.0, 2.2], [3.5, 1.5]
        ]
        
        dots_l = VGroup(*[Dot(origin_l + RIGHT*x + UP*y, radius=0.06, color=WHITE) for x, y in point_coords])
        dots_r = VGroup(*[Dot(origin_r + RIGHT*x + UP*y, radius=0.06, color=WHITE) for x, y in point_coords])

        # 4. Neural Network Curve (Overfitting/High Variance)
        # Create a wiggly path that hits points but oscillates wildly
        nn_curve_points = [
            origin_l + RIGHT*0.0 + UP*0.1,
            origin_l + RIGHT*0.5 + UP*0.4,
            origin_l + RIGHT*0.7 + UP*1.8, # Peak
            origin_l + RIGHT*1.0 + UP*1.2,
            origin_l + RIGHT*1.2 + UP*0.2, # Dip
            origin_l + RIGHT*1.5 + UP*0.8,
            origin_l + RIGHT*1.8 + UP*2.3, # Peak
            origin_l + RIGHT*2.0 + UP*1.8,
            origin_l + RIGHT*2.2 + UP*0.5, # Dip
            origin_l + RIGHT*2.5 + UP*1.1,
            origin_l + RIGHT*2.8 + UP*2.5, # Peak
            origin_l + RIGHT*3.0 + UP*2.2,
            origin_l + RIGHT*3.2 + UP*0.8, # Dip
            origin_l + RIGHT*3.5 + UP*1.5,
            origin_l + RIGHT*4.0 + UP*2.0
        ]
        nn_curve = VMobject(color=RED)
        nn_curve.set_points_as_corners(nn_curve_points)

        # 5. LWR Curve (Robust/Piecewise Linear approximation)
        # A smoother, simpler path through the points
        lwr_curve_points = [
            origin_r + RIGHT*0.0 + UP*0.3,
            origin_r + RIGHT*1.0 + UP*0.8,
            origin_r + RIGHT*2.0 + UP*1.4,
            origin_r + RIGHT*3.0 + UP*1.7,
            origin_r + RIGHT*4.0 + UP*2.1
        ]
        lwr_curve = VMobject(color=GREEN)
        lwr_curve.set_points_as_corners(lwr_curve_points)

        # 6. MSE Bars (Avoid BarChart class)
        # Formula
        mse_formula = MathTex(r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}", font_size=28)
        mse_formula.to_edge(DOWN, buff=1.2)

        # Bar Rectangles
        bar_nn = Rectangle(height=1.21, width=0.8, fill_opacity=0.8, color=RED)
        bar_nn.move_to([-1.5, -3.2, 0], aligned_edge=DOWN)
        val_nn = Text("1.21", font_size=20).next_to(bar_nn, UP, buff=0.1)
        tag_nn = Text("NN", font_size=18).next_to(bar_nn, DOWN, buff=0.1)

        bar_lwr = Rectangle(height=0.85, width=0.8, fill_opacity=0.8, color=GREEN)
        bar_lwr.move_to([1.5, -3.2, 0], aligned_edge=DOWN)
        val_lwr = Text("0.85", font_size=20).next_to(bar_lwr, UP, buff=0.1)
        tag_lwr = Text("LWR", font_size=18).next_to(bar_lwr, DOWN, buff=0.1)

        # Animations
        self.play(Write(header))
        self.play(
            Create(x_axis_l), Create(y_axis_l),
            Create(x_axis_r), Create(y_axis_r)
        )
        self.play(Write(label_l), Write(label_r))
        self.play(FadeIn(dots_l), FadeIn(dots_r))
        self.wait(1)

        self.play(Create(nn_curve), run_time=2.5)
        self.play(Create(lwr_curve), run_time=2.5)
        self.wait(1)

        self.play(Write(mse_formula))
        self.play(
            Create(bar_nn), Write(val_nn), Write(tag_nn),
            Create(bar_lwr), Write(val_lwr), Write(tag_lwr)
        )
        self.wait(3)
