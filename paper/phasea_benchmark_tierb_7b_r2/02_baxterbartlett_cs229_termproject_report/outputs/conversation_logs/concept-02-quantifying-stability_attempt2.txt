from manim import *
import numpy as np

class StabilityConcept(Scene):
    def construct(self):
        # 1. Coordinate System for Velocity (Manual construction using Line)
        # origin point for the plot
        v_origin = 3.5 * LEFT + 1 * DOWN
        v_x_axis = Line(v_origin, v_origin + 5 * RIGHT, color=WHITE)
        v_y_axis = Line(v_origin, v_origin + 3.5 * UP, color=WHITE)
        
        v_label_x = Tex("Time", font_size=24).next_to(v_x_axis, DOWN)
        v_label_y = Tex("RPM", font_size=24).rotate(90 * DEGREES).next_to(v_y_axis, LEFT)
        
        # 2. Velocity Plot (Fluctuating around 30 RPM)
        # We manually build the points for the curve to avoid Axes.plot or lambda scoping issues
        pts = []
        num_points = 60
        for i in range(num_points + 1):
            alpha = i / num_points
            # x goes from 0 to 5
            pos_x = alpha * 5
            # y fluctuates around 2.0 units up from origin (representing 30 RPM)
            fluctuation = 0.4 * np.sin(alpha * 12) + 0.2 * np.cos(alpha * 25)
            pos_y = 2.0 + fluctuation
            pts.append(v_origin + pos_x * RIGHT + pos_y * UP)
        
        velocity_curve = VMobject(color=BLUE)
        velocity_curve.set_points_as_corners(pts)
        
        # 3. Highlight Last 20%
        # 20% of 5 units is 1 unit.
        highlight_box = Rectangle(
            width=1.0, 
            height=3.5, 
            fill_color=BLUE, 
            fill_opacity=0.2, 
            stroke_width=0
        ).move_to(v_origin + 4.5 * RIGHT + 1.75 * UP)
        
        steady_state_label = Tex("Final 20\%", font_size=20, color=BLUE).next_to(highlight_box, UP)

        # 4. Histogram (Manual construction on the right side)
        h_origin = 2 * RIGHT + 1 * DOWN
        h_x_axis = Line(h_origin, h_origin + 4 * RIGHT, color=WHITE)
        h_y_axis = Line(h_origin, h_origin + 3 * UP, color=WHITE)
        h_title = Tex("Histogram", font_size=28).next_to(h_y_axis, UP)
        
        # Create bars: one tall bar at "30 RPM" (the mode)
        bar_heights = [0.5, 0.8, 2.5, 1.0, 0.4]
        bars = VGroup()
        for idx, h in enumerate(bar_heights):
            # The middle bar (idx=2) is the tall mode
            bar_color = YELLOW if idx == 2 else GREY
            b = Rectangle(
                width=0.5, 
                height=h, 
                fill_color=bar_color, 
                fill_opacity=0.8,
                stroke_color=WHITE,
                stroke_width=1
            )
            # Position bars along the h_x_axis
            b.move_to(h_origin + (idx * 0.7 + 0.5) * RIGHT, aligned_edge=DOWN)
            bars.add(b)
            
        mode_text = Tex("30 RPM (Mode)", font_size=18, color=YELLOW).next_to(bars[2], UP, buff=0.1)

        # 5. Stability Formula
        stability_formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=34
        ).to_edge(DOWN, buff=0.5)

        # 6. Animation sequence
        # Draw the plot
        self.play(Create(v_x_axis), Create(v_y_axis), Write(v_label_x), Write(v_label_y))
        self.play(Create(velocity_curve), run_time=2)
        self.wait(0.5)
        
        # Highlight last 20%
        self.play(FadeIn(highlight_box), Write(steady_state_label))
        self.wait(0.5)
        
        # Create Histogram
        self.play(Create(h_x_axis), Create(h_y_axis), Write(h_title))
        self.play(LaggedStart(*[FadeIn(b) for b in bars], lag_ratio=0.2))
        self.play(Write(mode_text))
        self.wait(0.5)
        
        # Show formula
        self.play(Write(stability_formula))
        self.play(Indicate(bars[2]), Indicate(stability_formula))
        
        self.wait(2)
