import numpy as np
from manim import *

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Title (Avoid using Title class, use Text)
        title_text = Text("Parameter vs. Data Bottleneck", font_size=36)
        title_text.to_edge(UP, buff=0.5)
        self.play(Write(title_text))
        self.wait(1)

        # 2. Manual Axes Construction (Avoid using Axes)
        def create_custom_axes(center_point):
            x_axis = Line(LEFT * 2.5, RIGHT * 2.5).move_to(center_point + DOWN * 1.5)
            y_axis = Line(DOWN * 1.5, UP * 1.5).move_to(center_point + LEFT * 2.5)
            return VGroup(x_axis, y_axis)

        left_origin = LEFT * 3.5 + UP * 0.5
        right_origin = RIGHT * 3.5 + UP * 0.5

        axes_nn = create_custom_axes(left_origin)
        axes_lwr = create_custom_axes(right_origin)

        label_nn = Text("Neural Network\n(High Variance)", font_size=20, color=RED).next_to(axes_nn, UP, buff=0.4)
        label_lwr = Text("LWR\n(Robust)", font_size=20, color=BLUE).next_to(axes_lwr, UP, buff=0.4)

        self.play(
            Create(axes_nn), Create(axes_lwr),
            Write(label_nn), Write(label_lwr)
        )

        # 3. Data Points (Same for both plots)
        rel_points = [
            [-2.0, -1.0, 0],
            [-1.0, 1.2, 0],
            [0.0, -0.8, 0],
            [1.0, 1.0, 0],
            [2.0, -0.5, 0]
        ]
        
        dots_nn = VGroup(*[Dot(left_origin + p, radius=0.06, color=WHITE) for p in rel_points])
        dots_lwr = VGroup(*[Dot(right_origin + p, radius=0.06, color=WHITE) for p in rel_points])

        self.play(FadeIn(dots_nn), FadeIn(dots_lwr))

        # 4. Curves
        # NN: Highly erratic curve passing through points
        nn_curve_points = []
        for i in range(len(rel_points) - 1):
            p1 = rel_points[i]
            p2 = rel_points[i+1]
            # Add wiggles between points
            for t in np.linspace(0, 1, 20):
                x = p1[0] * (1-t) + p2[0] * t
                # Sine wiggle that is zero at t=0 and t=1
                wiggle = 0.6 * np.sin(t * PI * 3) 
                y = p1[1] * (1-t) + p2[1] * t + wiggle
                nn_curve_points.append(left_origin + np.array([x, y, 0]))
        
        nn_curve = VMobject(color=RED)
        nn_curve.set_points_as_corners(nn_curve_points)

        # LWR: Smoother piecewise path
        lwr_curve_points = [right_origin + np.array(p) for p in rel_points]
        lwr_curve = VMobject(color=BLUE)
        lwr_curve.set_points_as_corners(lwr_curve_points)

        self.play(Create(nn_curve), run_time=2)
        self.play(Create(lwr_curve), run_time=2)
        self.wait(1)

        # 5. MSE Formula
        formula = MathTex(
            r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}", 
            font_size=30
        ).move_to(DOWN * 1.5)
        self.play(Write(formula))

        # 6. Bar Chart (Manual Construction)
        base_line = Line(LEFT * 4, RIGHT * 4).shift(DOWN * 3.5)
        
        # Neural Network Bar (Value: 1.21)
        bar_nn = Rectangle(width=1.2, height=1.21 * 1.2, fill_opacity=0.8, color=RED, stroke_width=2)
        bar_nn.align_to(base_line, DOWN).shift(LEFT * 2)
        val_nn = Text("1.21", font_size=18, color=RED).next_to(bar_nn, UP, buff=0.1)

        # LWR Bar (Value: 0.85)
        bar_lwr = Rectangle(width=1.2, height=0.85 * 1.2, fill_opacity=0.8, color=BLUE, stroke_width=2)
        bar_lwr.align_to(base_line, DOWN).shift(RIGHT * 2)
        val_lwr = Text("0.85", font_size=18, color=BLUE).next_to(bar_lwr, UP, buff=0.1)

        self.play(Create(base_line))
        self.play(
            Create(bar_nn), Write(val_nn),
            Create(bar_lwr), Write(val_lwr)
        )
        self.wait(3)

        # Cleanup
        all_mobs = VGroup(
            title_text, axes_nn, axes_lwr, label_nn, label_lwr,
            dots_nn, dots_lwr, nn_curve, lwr_curve, formula,
            base_line, bar_nn, val_nn, bar_lwr, val_lwr
        )
        self.play(FadeOut(all_mobs))
        self.wait(1)
