from manim import *
import numpy as np

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Introduction Text
        intro_text = Text("Why did LWR outperform a Neural Network?", font_size=32).to_edge(UP)
        sub_text = Text("The 27 training points vs. 24 parameters problem", font_size=24, color=GRAY).next_to(intro_text, DOWN)
        
        self.play(Write(intro_text))
        self.play(FadeIn(sub_text))
        self.wait(1)

        # 2. Manual Plotting Construction (Avoiding 'Axes')
        origin_l = LEFT * 3.5 + UP * 0.5
        origin_r = RIGHT * 3.5 + UP * 0.5
        
        # Manual Left Axes
        axis_x_l = Arrow(origin_l + LEFT * 2, origin_l + RIGHT * 2, buff=0, stroke_width=2)
        axis_y_l = Arrow(origin_l + DOWN * 1.5, origin_l + UP * 2, buff=0, stroke_width=2)
        label_nn = Text("Neural Network", font_size=28, color=RED).next_to(axis_y_l, UP)
        
        # Manual Right Axes
        axis_x_r = Arrow(origin_r + LEFT * 2, origin_r + RIGHT * 2, buff=0, stroke_width=2)
        axis_y_r = Arrow(origin_r + DOWN * 1.5, origin_r + UP * 2, buff=0, stroke_width=2)
        label_lwr = Text("LWR", font_size=28, color=GREEN).next_to(axis_y_r, UP)

        self.play(
            Create(axis_x_l), Create(axis_y_l), Write(label_nn),
            Create(axis_x_r), Create(axis_y_r), Write(label_lwr)
        )

        # 3. Data Points (27 points concept, using representative sample)
        rel_coords = [
            [-1.5, 0.2], [-1.0, -0.6], [-0.5, 0.8], 
            [0.0, -0.2], [0.5, 1.1], [1.0, 0.3], [1.5, 0.9]
        ]
        
        dots_l = VGroup(*[Dot(origin_l + x*RIGHT + y*UP, radius=0.06, color=YELLOW) for x, y in rel_coords])
        dots_r = VGroup(*[Dot(origin_r + x*RIGHT + y*UP, radius=0.06, color=YELLOW) for x, y in rel_coords])
        
        self.play(Create(dots_l), Create(dots_r))
        self.wait(1)

        # 4. Neural Network "Wiggling" Curve (High Variance)
        # We manually define points to create an erratic shape that hits all dots
        nn_path_points = [
            [-1.7, -0.5], [-1.5, 0.2], [-1.2, 1.3], [-1.0, -0.6], 
            [-0.8, -1.4], [-0.5, 0.8], [-0.2, 1.7], [0.0, -0.2], 
            [0.3, -1.1], [0.5, 1.1], [0.8, 1.6], [1.0, 0.3], 
            [1.3, -0.4], [1.5, 0.9], [1.8, 1.5]
        ]
        nn_curve = VMobject(color=RED)
        nn_curve.set_points_as_corners([origin_l + x*RIGHT + y*UP for x, y in nn_path_points])
        nn_curve.make_smooth()
        
        var_label = Text("High Variance", font_size=20, color=RED).next_to(axis_x_l, DOWN)
        
        # LWR Piecewise-Linear Curve (Robust)
        lwr_path_points = [
            [-1.7, -0.1], [-1.5, 0.2], [-1.0, -0.1], [-0.5, 0.3], 
            [0.0, 0.5], [0.5, 0.7], [1.0, 0.8], [1.5, 1.0], [1.8, 1.1]
        ]
        lwr_curve = VMobject(color=GREEN)
        lwr_curve.set_points_as_corners([origin_r + x*RIGHT + y*UP for x, y in lwr_path_points])
        
        robust_label = Text("Robust / Low Variance", font_size=20, color=GREEN).next_to(axis_x_r, DOWN)

        self.play(Create(nn_curve), Write(var_label))
        self.play(Create(lwr_curve), Write(robust_label))
        self.wait(1)

        # 5. MSE Formula and Comparison (Manual Bar Chart)
        formula = MathTex(r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}", font_size=32).to_edge(DOWN).shift(LEFT * 3)
        
        # Manual Bars
        bar_base = DOWN * 3 + RIGHT * 2
        nn_val, lwr_val = 1.21, 0.85
        scale_h = 1.5
        
        nn_bar = Rectangle(width=0.8, height=nn_val * scale_h, fill_opacity=0.8, color=RED)
        nn_bar.move_to(bar_base + LEFT * 0.7, aligned_edge=DOWN)
        
        lwr_bar = Rectangle(width=0.8, height=lwr_val * scale_h, fill_opacity=0.8, color=GREEN)
        lwr_bar.move_to(bar_base + RIGHT * 0.7, aligned_edge=DOWN)
        
        label_nn_bar = Text("1.21", font_size=18).next_to(nn_bar, UP)
        label_lwr_bar = Text("0.85", font_size=18).next_to(lwr_bar, UP)
        label_mse_title = Text("MSE Comparison", font_size=22).next_to(nn_bar, UP, buff=0.8).shift(RIGHT * 0.7)

        self.play(Write(formula))
        self.play(
            FadeIn(nn_bar), FadeIn(lwr_bar),
            Write(label_nn_bar), Write(label_lwr_bar), Write(label_mse_title)
        )
        
        self.wait(3)
