from manim import *
import numpy as np

class StabilityAnimation(Scene):
    def construct(self):
        # 1. Manual Coordinate System (Since Axes is disallowed)
        # We use simple lines to create the graph structure
        origin_point = LEFT * 4 + DOWN * 2
        x_axis = Line(origin_point, origin_point + RIGHT * 5, color=WHITE)
        y_axis = Line(origin_point, origin_point + UP * 4, color=WHITE)
        
        x_label = Text("Time", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("RPM", font_size=20).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        self.add(x_axis, y_axis, x_label, y_label)

        # 2. Velocity Plot
        # Create a fluctuating curve around a mean of 30 RPM (mapped to y=2.5)
        # We use a loop and VMobject to avoid 'Axes.plot' and potential lambda scope issues
        curve_points = []
        for i in range(101):
            t_ratio = i / 100.0  # 0 to 1
            # Fluctuating value: 2.5 is the baseline, sin/cos for variability
            fluctuation = 0.3 * np.sin(2 * t_ratio * 2 * np.pi) + 0.15 * np.cos(5 * t_ratio * 2 * np.pi)
            y_pos = 2.5 + fluctuation
            # Map t_ratio (0 to 1) to x-length (5 units)
            point = origin_point + RIGHT * (t_ratio * 5) + UP * y_pos
            curve_points.append(point)
        
        velocity_plot = VMobject(color=BLUE)
        velocity_plot.set_points_as_corners(curve_points)
        
        self.play(Create(velocity_plot), run_time=2)

        # 3. Highlight last 20% of the timeline
        # Timeline is 5 units long, last 20% is 1 unit (from x=4 to x=5)
        highlight_rect = Rectangle(
            width=1.0, 
            height=4.0, 
            fill_color=BLUE, 
            fill_opacity=0.3, 
            stroke_width=0
        ).move_to(origin_point + RIGHT * 4.5 + UP * 2)
        
        highlight_label = Text("Final 20%", font_size=16).next_to(highlight_rect, UP)
        self.play(FadeIn(highlight_rect), Write(highlight_label))
        self.wait(0.5)

        # 4. Histogram (Manual Rectangles)
        # Placed on the right side of the screen
        hist_center = RIGHT * 3 + DOWN * 0.5
        # bar_heights: low, high (mode), low
        bar_heights = [0.8, 2.5, 0.6]
        bars = VGroup()
        for i, h in enumerate(bar_heights):
            b = Rectangle(
                width=0.5, 
                height=h, 
                fill_color=BLUE if i != 1 else GREEN, 
                fill_opacity=0.8
            )
            # Align bottom of bars
            b.move_to(hist_center + RIGHT * (i * 0.7 - 0.7) + UP * (h / 2))
            bars.add(b)
        
        hist_title = Text("Histogram", font_size=24).next_to(bars, UP, buff=0.5)
        mode_text = Text("Mode (30 RPM)", font_size=14, color=GREEN).next_to(bars[1], DOWN)
        
        self.play(Create(bars), Write(hist_title), Write(mode_text))
        self.wait(1)

        # 5. Stability Formula
        # Using MathTex for the specific definition requested
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=32
        ).to_edge(UP, buff=0.5)
        
        self.play(Write(formula))
        
        # Animate visual relationship
        # Highlighting the numerator with the mode bar
        num_rect = SurroundingRectangle(formula[0][14:26], color=YELLOW)
        self.play(Create(num_rect), bars[1].animate.set_color(YELLOW))
        self.wait(1)
        
        # Highlighting the denominator with the other bars
        den_rect = SurroundingRectangle(formula[0][27:], color=BLUE)
        self.play(
            ReplacementTransform(num_rect, den_rect),
            bars[1].animate.set_color(GREEN),
            bars[0].animate.set_color(YELLOW),
            bars[2].animate.set_color(YELLOW)
        )
        self.wait(1)
        self.play(FadeOut(den_rect), bars[0].animate.set_color(BLUE), bars[2].animate.set_color(BLUE))

        self.wait(2)
