from manim import *

class StabilityConcept(Scene):
    def construct(self):
        # 1. Manually create Plot (using Lines instead of Axes)
        origin = [-5, -2, 0]
        x_axis = Line(origin, [-1, -2, 0], color=WHITE)
        y_axis = Line(origin, [-5, 1, 0], color=WHITE)
        
        y_label = Text("RPM", font_size=20).next_to(y_axis, LEFT)
        x_label = Text("Time", font_size=20).next_to(x_axis, DOWN)
        
        # Draw a fluctuating line around "30 RPM" (simulated at y offset 1)
        # Using segments to avoid VMobject/Axes dependencies
        points = []
        for x in range(21):
            px = -5 + (x * 4 / 20)
            py = -1 + (0.3 * np.sin(x) + np.random.uniform(-0.1, 0.1))
            points.append([px, py, 0])
        
        velocity_line = VGroup()
        for i in range(len(points) - 1):
            velocity_line.add(Line(points[i], points[i+1], color=BLUE))

        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(velocity_line))
        self.wait(1)

        # 2. Highlight last 20% of the x-axis
        # X range is -5 to -1 (width 4). Last 20% is 0.8 width.
        highlight_rect = Rectangle(
            width=0.8, height=3, 
            fill_color=BLUE, fill_opacity=0.3, stroke_width=0
        ).move_to([-1.4, -0.5, 0])
        
        sse_text = Text("Steady State Region", font_size=16).next_to(highlight_rect, UP)
        self.play(FadeIn(highlight_rect), Write(sse_text))
        self.wait(1)

        # 3. Histogram (Manual Rectangles)
        hist_origin = [1, -2, 0]
        hist_x = Line(hist_origin, [5, -2, 0])
        hist_title = Text("Histogram", font_size=24).move_to([3, 1.5, 0])
        
        # Bars: 10, 20, 30 (Mode), 40, 50
        bar_heights = [0.5, 0.8, 2.5, 0.7, 0.4]
        bars = VGroup()
        for i, h in enumerate(bar_heights):
            b = Rectangle(width=0.5, height=h, fill_color=GREEN, fill_opacity=0.7)
            b.move_to([1.5 + i*0.7, -2 + h/2, 0])
            bars.add(b)
            
        mode_tag = Text("Mode", font_size=18).next_to(bars[2], UP)

        self.play(Create(hist_x), Write(hist_title))
        self.play(Create(bars))
        self.play(Write(mode_tag))
        self.wait(1)

        # 4. Stability Formula
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=36
        ).to_edge(UP)
        
        self.play(Write(formula))
        
        # Highlight numerator and denominator logic
        num_box = SurroundingRectangle(formula[0][11:23], color=YELLOW)
        den_box = SurroundingRectangle(formula[0][24:], color=RED)
        
        self.play(Create(num_box))
        self.wait(0.5)
        self.play(ReplacementTransform(num_box, den_box))
        self.wait(1)
        self.play(FadeOut(den_box))
        
        self.wait(2)
