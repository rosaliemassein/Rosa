from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Setup Axes (Following Pattern #3)
        axes = Axes(
            x_range=[-4, 4, 1],
            y_range=[-1, 5, 1],
            x_length=10,
            y_length=6,
            axis_config={"color": BLUE}
        )
        axes_labels = axes.get_axis_labels(x_label="x", y_label="y")
        self.add(axes, axes_labels)

        # 2. Generate Data Points (Non-linear distribution)
        np.random.seed(42)
        x_vals = np.sort(np.random.uniform(-3.5, 3.5, 40))
        # Quadratic relationship with noise
        y_vals = 0.4 * x_vals**2 + np.random.normal(0, 0.3, size=40) + 0.5
        
        dots = VGroup(*[
            Dot(axes.c2p(x, y), radius=0.08, color=GRAY)
            for x, y in zip(x_vals, y_vals)
        ])
        self.add(dots)

        # 3. Query tracker and point (Following Pattern #2)
        # Using a tracker to control the current x-position of the "query"
        query_tracker = ValueTracker(-2.5)
        
        query_dot = Dot(color=WHITE).scale(1.2)
        # Position updater for query point
        query_dot.add_updater(lambda d: d.move_to(axes.c2p(query_tracker.get_value(), 0)))
        
        # 4. Weight field (Expanding circular indicator)
        weight_field = Circle(radius=1.5, color=WHITE, stroke_opacity=0.3)
        weight_field.add_updater(lambda c: c.move_to(query_dot.get_center()))
        
        # 5. Local regression line
        local_line = Line(color=YELLOW, stroke_width=6)
        
        def update_local_line(line):
            qx = query_tracker.get_value()
            tau = 0.8  # Bandwidth parameter
            
            # Weights formula: w = exp(-(x_i - x)^2 / (2 * tau^2))
            weights = np.exp(-(x_vals - qx)**2 / (2 * tau**2))
            
            # Weighted Least Squares calculation
            X = np.stack([np.ones_like(x_vals), x_vals], axis=1)
            W = np.diag(weights)
            
            try:
                # Normal Equation: theta = (X.T W X)^-1 X.T W y
                XTW = X.T @ W
                theta = np.linalg.inv(XTW @ X) @ XTW @ y_vals
                
                # Define a short segment around the query point
                lx = np.array([qx - 0.7, qx + 0.7])
                ly = theta[0] + theta[1] * lx
                
                line.put_start_and_end_on(axes.c2p(lx[0], ly[0]), axes.c2p(lx[1], ly[1]))
                line.set_stroke(opacity=1)
            except:
                line.set_stroke(opacity=0)

        local_line.add_updater(update_local_line)
        
        # 6. Dot highlighting updater
        def update_dots(group):
            qx = query_tracker.get_value()
            tau = 0.8
            for i, dot in enumerate(group):
                weight = np.exp(-(x_vals[i] - qx)**2 / (2 * tau**2))
                # interpolate_color is standard in Manim
                dot.set_color(interpolate_color(GRAY, GREEN, weight))
                dot.set_stroke(WHITE, width=weight * 2, opacity=weight)

        dots.add_updater(update_dots)

        # 7. Animation Execution
        self.play(FadeIn(dots), Write(axes_labels))
        self.play(FadeIn(query_dot), Create(weight_field))
        self.play(Create(local_line))
        self.wait(1)

        # Animate the query point moving across the curve
        # Default rate function is smooth
        self.play(query_tracker.animate.set_value(2.5), run_time=8)
        self.wait(1)
        
        # Move back to center
        self.play(query_tracker.animate.set_value(0), run_time=3)
        self.wait(2)
