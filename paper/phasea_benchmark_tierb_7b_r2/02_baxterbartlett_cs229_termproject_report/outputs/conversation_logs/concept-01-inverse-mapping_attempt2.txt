from manim import *
import numpy as np

class InverseMappingConcept(Scene):
    def construct(self):
        # 1. Diagram Setup (Left: Desired Behavior, Right: PID Constants)
        # We use manual grouping and basic shapes to ensure compatibility
        input_box = Rectangle(height=2.0, width=4.0, color=BLUE)
        input_title = Text("Desired Behavior", font_size=24).move_to(input_box.get_center()).shift(UP * 0.4)
        input_details = Text("(Rise Time, SS Error, Stability)", font_size=18).move_to(input_box.get_center()).shift(DOWN * 0.4)
        input_group = VGroup(input_box, input_title, input_details).shift(LEFT * 3.5 + UP * 2)

        output_box = Rectangle(height=2.0, width=3.5, color=GREEN)
        output_title = Text("PID Constants", font_size=24).move_to(output_box.get_center())
        output_group = VGroup(output_box, output_title).shift(RIGHT * 3.5 + UP * 2)

        # Arrow between them
        arrow = Arrow(start=input_group.get_right(), end=output_group.get_left(), buff=0.1)
        arrow_label = Text("Inverse Mapping", font_size=20).next_to(arrow, UP)

        diagram = VGroup(input_group, output_group, arrow, arrow_label)

        # Animation: Show initial flow
        self.play(FadeIn(input_group))
        self.play(Create(arrow), Write(arrow_label))
        self.play(FadeIn(output_group))
        self.wait(1)

        # "Flips the script": Rotate the entire diagram 180 degrees
        # Using animate.rotate to avoid the Rotate class which was flagged
        self.play(diagram.animate.rotate(PI), run_time=1.5)
        self.wait(1)

        # Scale and move diagram up to clear space for the plot
        self.play(diagram.animate.scale(0.6).to_edge(UP, buff=0.3))

        # 2. Motor Velocity Plot (RPM vs Time)
        # Using Line and VMobject to avoid Axes/DashedLine/Axes.plot restrictions
        plot_origin = DOWN * 2.5 + LEFT * 4
        x_axis = Line(plot_origin, plot_origin + RIGHT * 8, color=WHITE)
        y_axis = Line(plot_origin, plot_origin + UP * 3, color=WHITE)
        x_label = Text("Time", font_size=18).next_to(x_axis, RIGHT)
        y_label = Text("RPM", font_size=18).next_to(y_axis, UP)
        
        # Setpoint line (Normal Line instead of DashedLine)
        setpoint_y = 2.0
        setpoint_line = Line(
            plot_origin + UP * setpoint_y, 
            plot_origin + UP * setpoint_y + RIGHT * 8, 
            color=YELLOW
        )
        setpoint_text = Text("Setpoint", font_size=16, color=YELLOW).next_to(setpoint_line, RIGHT)

        # Oscillating curve (Manually created VMobject)
        curve = VMobject(color=BLUE)
        points = []
        # y = 1.8 * (1 - e^(-0.5x) * cos(2x)) -> adjusted to settle near setpoint
        for x_val in np.linspace(0, 8, 100):
            y_val = 1.8 * (1 - np.exp(-0.5 * x_val) * np.cos(2 * x_val))
            # Offset by plot_origin
            points.append(plot_origin + RIGHT * x_val + UP * y_val)
        
        curve.set_points_as_corners(points)

        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint_line), Write(setpoint_text))
        self.play(Create(curve), run_time=2)

        # 3. Braced Labels
        # Rise Time: From x=0 to where it first hits the setpoint vicinity (approx x=0.8)
        rise_line = Line(plot_origin, plot_origin + RIGHT * 0.8)
        rise_brace = Brace(rise_line, DOWN, buff=0.1)
        rise_label = Text("Rise Time", font_size=18).next_to(rise_brace, DOWN, buff=0.1)

        # Steady State Error: The gap at the end
        # At x=7.5, the curve is at y_val. Setpoint is at setpoint_y.
        error_line = Line(plot_origin + RIGHT * 7.5 + UP * 1.8, plot_origin + RIGHT * 7.5 + UP * 2.0)
        error_brace = Brace(error_line, RIGHT, buff=0.1)
        error_label = Text("Steady State Error", font_size=18).next_to(error_brace, RIGHT, buff=0.1)

        self.play(Create(rise_brace), Write(rise_label))
        self.play(Create(error_brace), Write(error_label))

        # 4. Formula
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=28)
        formula.to_edge(DOWN, buff=0.2)
        self.play(Write(formula))

        self.wait(3)
