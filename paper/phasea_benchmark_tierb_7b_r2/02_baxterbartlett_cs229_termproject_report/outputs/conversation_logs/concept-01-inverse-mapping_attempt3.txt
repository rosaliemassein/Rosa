from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # --- PART 1: DIAGRAM SETUP (MANUAL) ---
        # Input box
        input_box = Rectangle(width=4.0, height=2.0, color=BLUE)
        input_title = Text("Desired Behavior", font_size=20, color=YELLOW)
        input_details = Text("Rise Time, Error, Stability", font_size=14)
        input_content = VGroup(input_title, input_details).arrange(DOWN, buff=0.2)
        input_group = VGroup(input_box, input_content).shift(LEFT * 3.5)

        # Output box
        output_box = Rectangle(width=3.5, height=2.0, color=GREEN)
        output_title = Text("PID Constants", font_size=20, color=YELLOW)
        output_math = MathTex("K_p, K_i, K_d", font_size=24)
        output_content = VGroup(output_title, output_math).arrange(DOWN, buff=0.2)
        output_group = VGroup(output_box, output_content).shift(RIGHT * 3.5)

        # Arrow
        arrow = Arrow(start=input_group.get_right(), end=output_group.get_left(), buff=0.2)

        diagram = VGroup(input_group, output_group, arrow)
        self.play(FadeIn(diagram))
        self.wait(1)

        # "Flip the script" - Rotate 180 degrees
        # Using animate.rotate with a manual axis to avoid undefined 'Rotate' or 'OUT'
        self.play(diagram.animate.rotate(np.pi, axis=np.array([0, 0, 1])))
        self.wait(1)
        
        # Shrink and move diagram up to clear space
        self.play(diagram.animate.scale(0.6).to_edge(UP, buff=0.3))

        # --- PART 2: MOTOR VELOCITY PLOT (MANUAL) ---
        # Draw axes manually to avoid disallowed 'Axes'
        origin = DOWN * 2.5 + LEFT * 3.5
        x_axis = Line(origin, origin + RIGHT * 7)
        y_axis = Line(origin, origin + UP * 3)
        
        x_label = Text("Time", font_size=18).next_to(x_axis, DOWN, buff=0.1)
        y_label = Text("RPM", font_size=18).next_to(y_axis, LEFT, buff=0.1)
        
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))

        # Setpoint line (Manual Dashed Line to avoid disallowed 'DashedLine')
        setpoint_y = 2.0
        setpoint_group = VGroup()
        for i in range(24):
            start_p = origin + UP * setpoint_y + RIGHT * (i * 0.3)
            end_p = origin + UP * setpoint_y + RIGHT * (i * 0.3 + 0.15)
            setpoint_group.add(Line(start_p, end_p, color=GRAY, stroke_width=2))
        
        self.play(Create(setpoint_group))

        # Motor Curve: Damped Oscillation
        # y(t) = setpoint * (1 - exp(-0.7t) * cos(4t))
        curve_points = []
        for t_val in np.linspace(0, 7, 100):
            y_val = 2.0 * (1.0 - np.exp(-0.7 * t_val) * np.cos(4.0 * t_val))
            curve_points.append(origin + RIGHT * t_val + UP * y_val)
        
        curve = VMobject(color=BLUE)
        curve.set_points_as_corners(curve_points)
        self.play(Create(curve), run_time=2.5)

        # --- PART 3: ANNOTATIONS (MANUAL) ---
        # Rise Time Indicator (Manual Brace replacement)
        rise_time_val = 0.45
        rise_line = Line(origin + DOWN * 0.3, origin + RIGHT * rise_time_val + DOWN * 0.3, color=YELLOW)
        tick_1 = Line(origin + DOWN * 0.2, origin + DOWN * 0.4, color=YELLOW)
        tick_2 = Line(origin + RIGHT * rise_time_val + DOWN * 0.2, origin + RIGHT * rise_time_val + DOWN * 0.4, color=YELLOW)
        rise_text = Text("Rise Time", font_size=14).next_to(rise_line, DOWN, buff=0.1)
        
        self.play(Create(rise_line), Create(tick_1), Create(tick_2), Write(rise_text))

        # Steady State Error Indicator (Manual)
        err_x = 6.2
        curve_y_at_err = 2.0 * (1.0 - np.exp(-0.7 * err_x) * np.cos(4.0 * err_x))
        error_line = Line(origin + RIGHT * err_x + UP * curve_y_at_err, origin + RIGHT * err_x + UP * 2.0, color=RED)
        error_text = Text("Error", font_size=14).next_to(error_line, RIGHT, buff=0.1)
        
        self.play(Create(error_line), Write(error_text))

        # PID Formula
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=20)
        formula.to_corner(DR, buff=0.5)
        self.play(Write(formula))

        self.wait(3)
