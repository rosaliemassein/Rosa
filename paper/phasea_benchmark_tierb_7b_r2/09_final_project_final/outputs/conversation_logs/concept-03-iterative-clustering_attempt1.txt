from manim import *
import numpy as np

class IterativeClustering(Scene):
    def construct(self):
        # Set seed for reproducible random dot placement
        np.random.seed(42)
        
        # Create a cloud of 27 dots scattered randomly
        dots = VGroup(*[
            Dot(point=[np.random.uniform(-5, 5), np.random.uniform(-3, 2), 0]) 
            for _ in range(27)
        ])
        self.add(dots)
        self.wait(1)

        unassigned_dots = [d for d in dots]
        cluster_colors = [RED, BLUE, GREEN]

        def get_dist(m1, m2):
            return np.linalg.norm(m1.get_center() - m2.get_center())

        for i in range(3):
            # 1. Find the two most similar (closest) unassigned dots to seed the cluster
            min_dist = float('inf')
            seed1, seed2 = None, None
            
            # Find the pair with minimum distance among unassigned dots
            for idx_a in range(len(unassigned_dots)):
                for idx_b in range(idx_a + 1, len(unassigned_dots)):
                    d_val = get_dist(unassigned_dots[idx_a], unassigned_dots[idx_b])
                    if d_val < min_dist:
                        min_dist = d_val
                        seed1, seed2 = unassigned_dots[idx_a], unassigned_dots[idx_b]
            
            if not seed1 or not seed2:
                break

            # Highlight seed pair
            cluster_color = cluster_colors[i]
            highlight_ring = Circle(radius=0.2, color=YELLOW).move_to(seed1.get_center())
            highlight_ring2 = Circle(radius=0.2, color=YELLOW).move_to(seed2.get_center())
            
            self.play(Create(highlight_ring), Create(highlight_ring2), run_time=0.5)
            self.play(seed1.animate.set_color(cluster_color), seed2.animate.set_color(cluster_color))
            self.play(FadeOut(highlight_ring), FadeOut(highlight_ring2))

            # Initialize cluster tracking
            current_cluster = [seed1, seed2]
            unassigned_dots.remove(seed1)
            unassigned_dots.remove(seed2)
            
            # 2. Expanding search circle from the midpoint of the seed pair
            seed_center = (seed1.get_center() + seed2.get_center()) / 2
            search_circle = Circle(radius=min_dist/2, color=WHITE, stroke_opacity=0.5).move_to(seed_center)
            self.add(search_circle)

            # 3. Greedily pull in dots until we hit cluster size of 9
            while len(current_cluster) < 9 and len(unassigned_dots) > 0:
                # Find the next closest dot to the seed center
                next_dot = None
                closest_val = float('inf')
                
                for d_obj in unassigned_dots:
                    dist_to_center = np.linalg.norm(d_obj.get_center() - seed_center)
                    if dist_to_center < closest_val:
                        closest_val = dist_to_center
                        next_dot = d_obj
                
                # Animate expansion and color change
                # We use set_width or scale to expand the circle. 
                # radius = closest_val, so width = 2 * closest_val
                target_width = closest_val * 2
                self.play(
                    search_circle.animate.set_width(target_width),
                    run_time=0.4,
                    rate_func=lambda t: t
                )
                self.play(next_dot.animate.set_color(cluster_color), run_time=0.2)
                
                current_cluster.append(next_dot)
                unassigned_dots.remove(next_dot)

            self.play(FadeOut(search_circle))
            self.wait(0.5)

        self.wait(2)
