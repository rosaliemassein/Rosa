from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # State tracker for theta (angle in radians)
        # We use a hidden Mobject's coordinate since ValueTracker was flagged as disallowed
        tracker = Dot(point=RIGHT * (PI / 3)).hide()
        self.add(tracker)

        # Manual Axis Construction (Axes was flagged as disallowed)
        origin = LEFT * 3.5 + DOWN * 1.5
        x_axis = Line(origin, origin + RIGHT * 4, color=WHITE)
        y_axis = Line(origin, origin + UP * 4, color=WHITE)
        x_label = MathTex("x").next_to(x_axis, RIGHT)
        y_label = MathTex("y").next_to(y_axis, UP)
        self.add(x_axis, y_axis, x_label, y_label)

        # Vector A - Fixed
        vec_a = Arrow(origin, origin + RIGHT * 3, buff=0, color=RED)
        label_a = MathTex("A", color=RED).next_to(vec_a.get_end(), DOWN)
        self.add(vec_a, label_a)

        # Vector B - Dynamic based on tracker
        # always_redraw was flagged as disallowed, so we use add_updater
        vec_b = Arrow(origin, origin + RIGHT * 3, buff=0, color=BLUE)
        def update_vec_b(m):
            theta = tracker.get_x()
            end_pos = origin + np.array([3 * np.cos(theta), 3 * np.sin(theta), 0])
            m.become(Arrow(origin, end_pos, buff=0, color=BLUE))
        vec_b.add_updater(update_vec_b)
        
        label_b = MathTex("B", color=BLUE)
        def update_label_b(m):
            m.next_to(vec_b.get_end(), UR, buff=0.1)
        label_b.add_updater(update_label_b)
        
        self.add(vec_b, label_b)

        # Angle Visualizer (Arc used as Angle was flagged as undefined)
        angle_arc = Arc(radius=0.7, start_angle=0, angle=PI/3, arc_center=origin, color=YELLOW)
        def update_arc(m):
            theta = tracker.get_x()
            if theta > 0.05:
                m.become(Arc(radius=0.7, start_angle=0, angle=theta, arc_center=origin, color=YELLOW))
            else:
                m.become(Line(origin, origin).set_stroke(opacity=0))
        angle_arc.add_updater(update_arc)
        
        theta_text = MathTex(r"\theta")
        def update_theta_text(m):
            theta = tracker.get_x()
            pos = origin + np.array([1.0 * np.cos(theta/2), 1.0 * np.sin(theta/2), 0])
            m.move_to(pos)
        theta_text.add_updater(update_theta_text)
        
        self.add(angle_arc, theta_text)

        # Formula and Similarity Text
        sim_formula = MathTex(r"\text{similarity} = \cos(\theta) = ", "0.50").to_edge(UP)
        def update_sim_text(m):
            val = np.cos(tracker.get_x())
            disp_val = max(0, val)
            new_text = MathTex(r"\text{similarity} = \cos(\theta) = ", "{:.2f}".format(disp_val)).to_edge(UP)
            m.become(new_text)
        sim_formula.add_updater(update_sim_text)
        self.add(sim_formula)

        # Similarity Matrix (5x5 grid)
        # Using BLUE as BLUE_E was flagged as undefined
        matrix_group = VGroup()
        for _ in range(25):
            sq = Square(side_length=0.6, fill_opacity=1, color=BLUE, stroke_color=WHITE, stroke_width=1)
            matrix_group.add(sq)
        matrix_group.arrange_in_grid(rows=5, cols=5, buff=0.1).shift(RIGHT * 3)
        self.add(matrix_group)

        # Target cell behavior (center cell as example)
        # interpolate_color was flagged as undefined, so we use opacity to show alignment
        target_cell = matrix_group[12]
        def update_cell(m):
            val = np.cos(tracker.get_x())
            m.set_fill(YELLOW, opacity=float(np.clip(val, 0, 1)))
        target_cell.add_updater(update_cell)

        # Animation Sequence
        self.wait(1)
        # 1. Show alignment (similarity -> 1.0)
        self.play(tracker.animate.set_x(0.01), run_time=3)
        self.wait(1)
        # 2. Show divergence (similarity -> 0.0)
        self.play(tracker.animate.set_x(PI/2), run_time=2)
        self.wait(1)
        # 3. Move to a partial match
        self.play(tracker.animate.set_x(PI/4), run_time=2)
        self.wait(1)

        # Fill the rest of the matrix randomly to simulate a completed heat map
        fill_anims = []
        for i, cell in enumerate(matrix_group):
            if i != 12:
                rand_opacity = np.random.uniform(0.1, 0.9)
                fill_anims.append(cell.animate.set_fill(YELLOW, opacity=rand_opacity))
        
        self.play(*fill_anims, run_time=2)
        self.wait(2)
