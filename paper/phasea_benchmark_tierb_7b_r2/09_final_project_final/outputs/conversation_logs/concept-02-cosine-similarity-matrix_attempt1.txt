from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Setup Manual Axes (since Axes is disallowed)
        x_axis = Line(LEFT * 3, RIGHT * 3, color=WHITE).shift(LEFT * 3)
        y_axis = Line(DOWN * 3, UP * 3, color=WHITE).shift(LEFT * 3)
        origin = x_axis.get_center()
        
        # 2. Define Vectors
        # Vector A is fixed at approx 26.5 degrees (atan2(1, 2))
        vec_a_end = origin + RIGHT * 2 + UP * 1
        vector_a = Arrow(origin, vec_a_end, buff=0, color=BLUE)
        label_a = MathTex(r"\mathbf{A}", color=BLUE).next_to(vec_a_end, UP)
        
        # Value Tracking using a hidden Dot (since ValueTracker is disallowed)
        # We'll use the x-coordinate of this dot as our angle theta
        phi = np.arctan2(1, 2) # Angle of Vector A
        start_angle = phi + PI / 2
        tracker_dot = Dot(point=[start_angle, 0, 0]).set_opacity(0)
        
        # Vector B (dynamic)
        vector_b = Arrow(origin, origin + RIGHT * 2, buff=0, color=RED)
        
        def update_vector_b(mob):
            angle = tracker_dot.get_x()
            end_point = origin + np.array([2 * np.cos(angle), 2 * np.sin(angle), 0])
            mob.become(Arrow(origin, end_point, buff=0, color=RED))
            
        vector_b.add_updater(update_vector_b)
        
        label_b = MathTex(r"\mathbf{B}", color=RED)
        def update_label_b(mob):
            mob.next_to(vector_b.get_end(), RIGHT, buff=0.1)
        label_b.add_updater(update_label_b)

        # 3. Angle Arc and Text
        angle_arc = Arc(radius=0.6, start_angle=phi, angle=PI/2, arc_center=origin, color=YELLOW)
        def update_arc(mob):
            current_angle = tracker_dot.get_x()
            mob.become(Arc(
                radius=0.6, 
                start_angle=phi, 
                angle=current_angle - phi, 
                arc_center=origin, 
                color=YELLOW
            ))
        angle_arc.add_updater(update_arc)

        sim_label = MathTex(r"\cos(\theta) = ", color=WHITE).to_corner(UL).shift(RIGHT * 0.5)
        sim_value = MathTex("0.00", color=YELLOW).next_to(sim_label, RIGHT)
        
        def update_sim_value(mob):
            current_theta = tracker_dot.get_x() - phi
            val = np.cos(current_theta)
            new_tex = MathTex(f"{val:.2f}", color=YELLOW).next_to(sim_label, RIGHT)
            mob.become(new_tex)
        sim_value.add_updater(update_sim_value)

        self.add(x_axis, y_axis, vector_a, label_a, vector_b, label_b, angle_arc, tracker_dot, sim_label, sim_value)

        # 4. Animate alignment
        self.play(tracker_dot.animate.set_x(phi), run_time=3)
        self.wait(1)
        self.play(tracker_dot.animate.set_x(phi + PI/3), run_time=2)
        self.wait(1)

        # 5. Transition to Matrix
        geometry_group = VGroup(x_axis, y_axis, vector_a, label_a, vector_b, label_b, angle_arc)
        
        # Create 5x5 Matrix of Squares
        matrix_cells = VGroup()
        for i in range(5):
            for j in range(5):
                sq = Square(side_length=0.7).set_stroke(GRAY, 1)
                matrix_cells.add(sq)
        matrix_cells.arrange_in_grid(5, 5, buff=0.05).to_edge(RIGHT, buff=1)
        
        matrix_title = MathTex(r"\text{Similarity Matrix}", color=WHITE).next_to(matrix_cells, UP)
        formula = MathTex(
            r"\cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            color=WHITE
        ).scale(0.7).next_to(matrix_cells, DOWN)

        self.play(
            geometry_group.animate.scale(0.5).to_edge(LEFT, buff=0.5),
            sim_label.animate.scale(0.7).to_edge(LEFT).shift(UP*3),
            sim_value.animate.scale(0.7).next_to(sim_label, RIGHT),
            FadeIn(matrix_cells),
            FadeIn(matrix_title)
        )

        # 6. Matrix alignment effect
        # Highlight cells to represent heatmap
        highlight_anims = []
        for i, cell in enumerate(matrix_cells):
            # Create a simple pattern: center is bright
            row, col = i // 5, i % 5
            dist = np.sqrt((row-2)**2 + (col-2)**2)
            opacity = max(0.1, 1.0 - dist/3)
            highlight_anims.append(cell.animate.set_fill(YELLOW if dist < 1 else BLUE, opacity=opacity))

        self.play(tracker_dot.animate.set_x(phi), run_time=1)
        self.play(*highlight_anims, Write(formula))
        self.wait(2)
