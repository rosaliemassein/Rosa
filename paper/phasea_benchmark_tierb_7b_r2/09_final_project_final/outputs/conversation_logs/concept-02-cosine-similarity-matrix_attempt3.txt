from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Coordinate System (Manual creation to avoid restricted 'Axes' if necessary)
        # Creating simple arrows for axes as a robust alternative
        origin = 3 * LEFT
        x_axis = Arrow(origin, origin + 3 * RIGHT, color=WHITE, buff=0)
        y_axis = Arrow(origin, origin + 3 * UP, color=WHITE, buff=0)
        axes_labels = MathTex("x", "y").scale(0.8)
        axes_labels[0].next_to(x_axis, RIGHT)
        axes_labels[1].next_to(y_axis, UP)
        
        # 2. ValueTracker for rotation (As seen in Reference Example 2)
        theta_tracker = ValueTracker(60)
        
        # 3. Vectors A and B
        # Vector A is fixed
        vec_a = Arrow(origin, origin + 2 * RIGHT, color=GREEN, buff=0)
        label_a = MathTex(r"\mathbf{A}", color=GREEN).next_to(vec_a, RIGHT)
        
        # Vector B is dynamic
        # Reference line for rotation
        vec_b_ref = Arrow(origin, origin + 2 * RIGHT, color=RED, buff=0)
        vec_b = Arrow(origin, origin + 2 * RIGHT, color=RED, buff=0)
        label_b = MathTex(r"\mathbf{B}", color=RED)
        
        # 4. Angle Arc (Using Angle class as seen in Reference Example 2)
        angle_obj = Angle(vec_a, vec_b, radius=0.6, other_angle=False)
        theta_sym = MathTex(r"\theta").scale(0.8)
        
        # 5. Text Displays
        sim_title = MathTex(r"\text{similarity} = \cos(\theta) = ").to_edge(UP).shift(LEFT)
        sim_value = MathTex("0.50").next_to(sim_title, RIGHT)
        
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}"
        ).scale(0.7).to_edge(DOWN)

        # 6. Updaters (Pattern from Reference Example 2)
        def update_vec_b(v):
            new_v = Arrow(
                origin, 
                origin + 2 * RIGHT, 
                color=RED, 
                buff=0
            ).rotate(theta_tracker.get_value() * DEGREES, about_point=origin)
            v.become(new_v)

        def update_label_b(l):
            l.next_to(vec_b.get_end(), UP + RIGHT, buff=0.1)

        def update_angle(a):
            a.become(Angle(vec_a, vec_b, radius=0.6, other_angle=False))

        def update_theta_sym(t):
            p = Angle(vec_a, vec_b, radius=0.9, other_angle=False).point_from_proportion(0.5)
            t.move_to(p)

        def update_sim_value(s):
            val = np.cos(theta_tracker.get_value() * DEGREES)
            new_s = MathTex(f"{val:.2f}").next_to(sim_title, RIGHT)
            s.become(new_s)

        vec_b.add_updater(update_vec_b)
        label_b.add_updater(update_label_b)
        angle_obj.add_updater(update_angle)
        theta_sym.add_updater(update_theta_sym)
        sim_value.add_updater(update_sim_value)

        # 7. Similarity Matrix
        matrix = VGroup()
        for i in range(5):
            for j in range(5):
                sq = Square(side_length=0.5, fill_opacity=0.3, color=BLUE)
                matrix.add(sq)
        matrix.arrange_in_grid(5, 5, buff=0.1).to_edge(RIGHT, buff=1)
        matrix_title = MathTex(r"\text{Similarity Matrix}").scale(0.7).next_to(matrix, UP)
        
        # Highlight a specific cell (e.g., center cell) to react to the vectors
        target_cell = matrix[12]
        def update_cell(c):
            val = np.cos(theta_tracker.get_value() * DEGREES)
            if val > 0.9:
                c.set_fill(YELLOW, opacity=val)
            else:
                c.set_fill(BLUE, opacity=val)
        
        target_cell.add_updater(update_cell)

        # 8. Animation Sequence
        self.add(x_axis, y_axis, axes_labels, vec_a, label_a, vec_b, label_b)
        self.add(angle_obj, theta_sym, sim_title, sim_value, formula, matrix, matrix_title)
        
        # Animation: Align vectors
        self.play(theta_tracker.animate.set_value(0), run_time=3)
        self.wait(1)
        
        # Animation: Diverge vectors
        self.play(theta_tracker.animate.set_value(90), run_time=3)
        self.wait(1)
        
        # Static Matrix Coloring for visual completion
        final_anims = []
        for i in range(len(matrix)):
            if i != 12: # Don't conflict with updater
                final_anims.append(matrix[i].animate.set_fill(BLUE, opacity=np.random.uniform(0.1, 0.8)))
        
        self.play(*final_anims)
        self.wait(2)
