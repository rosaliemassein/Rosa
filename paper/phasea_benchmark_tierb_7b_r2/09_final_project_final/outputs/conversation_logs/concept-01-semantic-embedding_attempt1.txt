from manim import *

class Concept01SemanticEmbedding(Scene):
    def construct(self):
        # 1. Initialize coordinate system
        plane = NumberPlane()
        self.add(plane)

        # 2. Text labels for keywords
        dl_text = Text("Deep Learning", color=BLUE).scale(0.6)
        nn_text = Text("Neural Networks", color=GREEN).scale(0.6)
        rob_text = Text("Robotics", color=RED).scale(0.6)

        # Define vector positions as simple lists
        dl_pos = [2, 1, 0]
        nn_pos = [2.5, 0.5, 0]
        rob_pos = [-3, -1.5, 0]

        dl_text.move_to(dl_pos).shift(UP * 0.4)
        nn_text.move_to(nn_pos).shift(DOWN * 0.4)
        rob_text.move_to(rob_pos).shift(UP * 0.4)

        # 3. Create Dot objects
        dl_dot = Dot(dl_pos, color=BLUE)
        nn_dot = Dot(nn_pos, color=GREEN)
        rob_dot = Dot(rob_pos, color=RED)

        # Animation: Pop up labels
        self.play(Write(dl_text), Write(nn_text), Write(rob_text))
        self.wait(1)

        # Animation: Transform labels into Dot objects
        self.play(
            Transform(dl_text, dl_dot),
            Transform(nn_text, nn_dot),
            Transform(rob_text, rob_dot)
        )
        self.wait(0.5)

        # 4. Show distance between similar concepts
        # Note: Using Line with low opacity as DashedLine was flagged as undefined in the log
        dist_line = Line(dl_pos, nn_pos, color=WHITE).set_opacity(0.5)
        self.play(Create(dist_line))
        self.wait(1.5)

        # 5. Clear for formula and averaging logic
        self.play(
            FadeOut(dl_text), 
            FadeOut(nn_text), 
            FadeOut(rob_text), 
            FadeOut(dist_line), 
            FadeOut(plane)
        )

        # 6. Display Formula
        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}", color=WHITE).to_edge(UP)
        self.play(Write(formula))

        # 7. Word vectors for "Machine Learning"
        # Using specific vector endpoints
        v1_end = [-2, 2, 0]
        v2_end = [4, 0, 0]
        # Average vector: ([-2+4]/2, [2+0]/2) = (1, 1)
        v_avg_end = [1, 1, 0]

        word_m = Text("Machine", color=YELLOW).scale(0.7).move_to(v1_end).shift(UP * 0.5)
        word_l = Text("Learning", color=BLUE).scale(0.7).move_to(v2_end).shift(UP * 0.5)

        vec_m = Arrow([0, 0, 0], v1_end, buff=0, color=YELLOW)
        vec_l = Arrow([0, 0, 0], v2_end, buff=0, color=BLUE)
        
        avg_vec = Arrow([0, 0, 0], v_avg_end, buff=0, color=GREEN)
        avg_label = MathTex(r"v_{phrase}", color=GREEN).next_to(v_avg_end, RIGHT)

        # Animate word vectors
        self.play(Write(word_m), Write(word_l))
        self.play(GrowArrow(vec_m), GrowArrow(vec_l))
        self.wait(1)

        # 8. Merge vectors into their center of gravity (average)
        self.play(
            Transform(vec_m, avg_vec),
            Transform(vec_l, avg_vec),
            FadeOut(word_m),
            FadeOut(word_l),
            Write(avg_label)
        )
        self.wait(2)
