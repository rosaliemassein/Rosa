from manim import *

class GeneticSelection(Scene):
    def construct(self):
        # Formula reference at the bottom
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)").to_edge(DOWN)
        self.add(formula)

        # 1. Draw a central 'Parent' node
        parent = VGroup(
            Circle(radius=0.7, color=YELLOW, fill_opacity=0.3),
            Text("Parent", font_size=24)
        ).shift(UP * 2.5)

        # 2. Child nodes branching out
        child_positions = [LEFT * 4 + DOWN * 0.5, DOWN * 0.5, RIGHT * 4 + DOWN * 0.5]
        children = VGroup()
        lines = VGroup()
        for i, pos in enumerate(child_positions):
            child_node = VGroup(
                Circle(radius=0.6, color=WHITE, fill_opacity=0.1),
                Text(f"Child {i+1}", font_size=20)
            ).move_to(pos)
            children.add(child_node)
            lines.add(Line(parent.get_bottom(), child_node.get_top(), buff=0.1))

        # 3. Win Rate Bars
        win_rates = [0.45, 0.92, 0.65]
        bars = VGroup()
        bar_backgrounds = VGroup()
        bar_labels = VGroup()

        for i, (child, rate) in enumerate(zip(children, win_rates)):
            # Frame for the bar
            bg = Rectangle(width=0.6, height=2.0, stroke_width=1, color=GRAY_D)
            bg.next_to(child, DOWN, buff=0.4)
            # Actual bar
            bar = Rectangle(
                width=0.6, 
                height=rate * 2.0, 
                fill_opacity=1, 
                color=BLUE
            ).align_to(bg, DOWN)
            # Label
            label = Text(f"{int(rate*100)}%", font_size=16).next_to(bg, DOWN, buff=0.1)
            
            bars.add(bar)
            bar_backgrounds.add(bg)
            bar_labels.add(label)

        # Step 1: Parent appears
        self.play(Create(parent))
        self.wait(0.5)

        # Step 2: Children branch out
        self.play(
            LaggedStart(
                *[Create(line) for line in lines],
                *[Create(child) for child in children],
                lag_ratio=0.2
            )
        )
        self.wait(0.5)

        # Step 3: Win Rate bars grow (Using GrowFromPoint for stability)
        self.play(
            Create(bar_backgrounds),
            *[GrowFromPoint(bar, bar.get_bottom()) for bar in bars],
            Write(bar_labels)
        )
        self.wait(1)

        # Step 4: Highlight winner in gold (Child 2)
        winner_idx = 1
        self.play(
            bars[winner_idx].animate.set_color(GOLD),
            children[winner_idx][0].animate.set_color(GOLD).set_stroke(width=8),
            bar_labels[winner_idx].animate.set_color(GOLD).scale(1.2)
        )
        self.play(Indicate(children[winner_idx]))
        self.wait(1)

        # Step 5: Fade out parent and other children
        others = VGroup(
            parent, lines,
            children[0], children[2],
            bars[0], bars[2],
            bar_backgrounds[0], bar_backgrounds[2],
            bar_labels[0], bar_labels[2],
            bar_backgrounds[1]
        )
        
        self.play(
            FadeOut(others),
            FadeOut(bar_labels[winner_idx]),
            bars[winner_idx].animate.set_opacity(0)
        )

        # Step 6: Move winner to center to become the new Parent
        new_parent_center = parent.get_center()
        new_parent_text = Text("Parent", font_size=24).move_to(children[winner_idx][1])
        
        self.play(
            children[winner_idx].animate.move_to(new_parent_center).scale(1.2),
            Transform(children[winner_idx][1], new_parent_text),
            children[winner_idx][0].animate.set_color(YELLOW)
        )

        # Final visual hint
        self.play(Flash(children[winner_idx][0], color=YELLOW, line_length=0.4))
        self.wait(2)
