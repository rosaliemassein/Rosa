from manim import *

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Custom Axes Construction (Avoiding Axes and ValueTracker)
        origin_point = [ -5.5, -2.2, 0 ]
        x_axis = Line(origin_point, origin_point + [6, 0, 0], color=WHITE)
        y_axis = Line(origin_point, origin_point + [0, 5, 0], color=WHITE)
        
        # Labels
        x_lab = Text("Number of Players", font_size=20).move_to(origin_point + [3, -0.5, 0])
        y_lab = Text("Mean Squared Error", font_size=20).move_to(origin_point + [-0.8, 2.5, 0]).rotate(90*DEGREES)
        
        self.add(x_axis, y_axis, x_lab, y_lab)

        # 2. KNN Curve (Red, dipping then rising)
        knn_pts = []
        for i in range(21):
            val_x = i / 2.0 # range 0 to 10
            # MSE Parabola: y = 0.15 * (x - 3.5)^2 + 1.0
            val_y = 0.15 * (val_x - 3.5) * (val_x - 3.5) + 1.0
            px = origin_point[0] + (val_x * 0.6)
            py = origin_point[1] + (val_y * 0.8)
            knn_pts.append([px, py, 0])
        
        knn_curve = VMobject(color=RED).set_points_as_corners(knn_pts)
        knn_tag = Text("KNN", color=RED, font_size=18).next_to(knn_pts[-1], UP)

        # 3. Lasso Curve (Green, steady decline)
        lasso_pts = []
        for i in range(21):
            val_x = i / 2.0
            # Decline function: y = 4 / (1 + 0.4*x)
            val_y = 4.0 / (1.0 + 0.4 * val_x)
            px = origin_point[0] + (val_x * 0.6)
            py = origin_point[1] + (val_y * 0.8)
            lasso_pts.append([px, py, 0])
        
        lasso_curve = VMobject(color=GREEN).set_points_as_corners(lasso_pts)
        lasso_tag = Text("Lasso/Ridge", color=GREEN, font_size=18).next_to(lasso_pts[-1], RIGHT)

        self.play(Create(knn_curve), Write(knn_tag))
        self.play(Create(lasso_curve), Write(lasso_tag))

        # 4. Formula with lambda highlighting
        formula = MathTex(
            "J(\\theta) = \\|y - X\\theta\\|^2 + ", 
            "\\lambda", 
            "\\|\\theta\\|_1",
            font_size=32
        ).to_edge(UP).shift(RIGHT * 3)
        self.play(Write(formula))

        # 5. Coefficient Bar Chart
        bars_group = VGroup()
        initial_heights = []
        # Create 100 coefficient bars
        for i in range(100):
            h_val = ( (i * 17) % 100 ) / 55.0 # Deterministic "random" heights 0 to 1.8
            initial_heights.append(h_val)
            # Small line representing a coefficient
            bar_line = Line(ORIGIN, [0, h_val, 0], stroke_width=2, color=BLUE)
            bars_group.add(bar_line)
        
        bars_group.arrange(RIGHT, buff=0.02).move_to([3.5, -1.2, 0])
        chart_title = Text("Coefficients", font_size=20).next_to(bars_group, UP)
        
        self.play(Write(chart_title), Create(bars_group))

        # 6. Shrinkage Animation using a hidden tracker
        # Using a Dot's position to simulate a ValueTracker to avoid "disallowed" error
        penalty_tracker = Dot(radius=0).move_to([0, 0, 0])
        
        def update_bars_func(mobject):
            current_penalty = penalty_tracker.get_center()[0] # Use x-coord as penalty value
            for idx, single_bar in enumerate(mobject):
                h_start = initial_heights[idx]
                h_now = h_start - current_penalty
                if h_now < 0:
                    h_now = 0.001
                    single_bar.set_stroke(opacity=0)
                else:
                    single_bar.set_stroke(opacity=1)
                
                # Rescale the line height from the bottom
                single_bar.set_height(h_now, stretch=True, about_edge=DOWN)

        bars_group.add_updater(update_bars_func)
        
        # Animate lambda increasing and coefficients vanishing
        self.play(
            penalty_tracker.animate.shift([1.8, 0, 0]),
            formula[1].animate.set_color(YELLOW).scale(1.3),
            run_time=5,
            rate_func=lambda t: t
        )
        
        self.wait(2)
