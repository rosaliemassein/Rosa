from manim import *

class PlayerDepthImportance(Scene):
    def construct(self):
        # 1. Formula at the top
        formula = MathTex(
            r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})",
            font_size=36
        ).to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Setup Data
        # Coefficients representing the hierarchy (peaks at 1st, 2nd, 5th, 8th)
        coeffs_values = [3.5, 3.0, 1.5, 1.2, 2.5, 1.1, 1.0, 2.0, 0.8, 0.6]
        
        bars = VGroup()
        icons = VGroup()
        labels = VGroup()

        # Create a group for each player (Label + Icon + Bar)
        for i in range(10):
            # Simple Player Icon: Head (Circle) and Body (Square)
            head = Circle(radius=0.1, color=WHITE)
            body = Square(side_length=0.2, color=WHITE).next_to(head, DOWN, buff=0.05)
            icon = VGroup(head, body).scale(0.8)
            
            # Label
            label = Text(str(i+1), font_size=20)
            
            # Bar (The height corresponds to the coefficient)
            bar = Rectangle(
                width=0.4, 
                height=coeffs_values[i], 
                fill_opacity=0.7, 
                color=BLUE
            )
            
            # Vertical positioning: Label at bottom, then Icon, then Bar
            # We use next_to to stack them
            icon.next_to(label, UP, buff=0.2)
            bar.next_to(icon, UP, buff=0.1)
            
            # Move the bar's internal coordinate so stretching works from the bottom
            # Shift the bar so its bottom sits exactly on top of the icon
            # (By default next_to puts the center of the bar relative to the icon)
            
            unit = VGroup(bar, icon, label)
            # Distribute the units horizontally
            unit.shift(RIGHT * (i - 4.5) * 0.75)
            
            bars.add(bar)
            icons.add(icon)
            labels.add(label)

        # Center the whole collection on screen
        VGroup(bars, icons, labels).center().shift(DOWN * 0.5)

        # 3. Animation: Fade in labels and icons first
        self.play(
            FadeIn(icons),
            Write(labels)
        )
        self.wait(0.5)

        # 4. Animation: Bars grow to their respective heights
        # We simulate "growing" by stretching from the bottom edge
        for bar in bars:
            bar.save_state()
            bar.stretch_to_fit_height(0.01, about_edge=DOWN)
            bar.set_opacity(0)

        self.play(
            *[Restore(bar).set_opacity(0.7) for bar in bars],
            run_time=2,
            rate_func=smooth
        )
        self.wait(1)

        # 5. Highlight Stars (1st and 2nd bars) with Gold/Yellow
        stars = VGroup(bars[0], bars[1])
        self.play(
            stars.animate.set_color(YELLOW).set_fill(YELLOW, opacity=0.9),
            Indicate(stars, color=YELLOW, scale_factor=1.1)
        )
        self.wait(1)

        # 6. Highlight Depth Players (5th and 8th bars) with pulsing light
        # Indices 4 and 7 correspond to the 5th and 8th players
        depth_bars = VGroup(bars[4], bars[7])
        
        # Change color to distinguish them
        self.play(
            depth_bars.animate.set_color(WHITE).set_fill(WHITE, opacity=0.9)
        )
        
        # Pulse animation
        for _ in range(3):
            self.play(depth_bars.animate.scale(1.2), run_time=0.3)
            self.play(depth_bars.animate.scale(1/1.2), run_time=0.3)

        self.wait(2)
