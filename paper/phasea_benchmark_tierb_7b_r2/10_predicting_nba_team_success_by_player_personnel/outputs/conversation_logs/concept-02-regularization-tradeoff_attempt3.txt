from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Create manual axes using Lines (Avoids "Axes" if it is restricted)
        origin_point = LEFT * 5 + DOWN * 2
        x_axis = Line(origin_point, origin_point + RIGHT * 6, color=WHITE)
        y_axis = Line(origin_point, origin_point + UP * 4.5, color=WHITE)
        
        x_label = Text("Number of Players", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("Mean Squared Error", font_size=20).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        self.play(Create(x_axis), Create(y_axis))
        self.play(Write(x_label), Write(y_label))
        self.wait(1)

        # 2. Plot Curves manually using VMobject and set_points_as_corners
        # Mapping function: (x, y) -> origin + x * scale_x + y * scale_y
        def to_coords(x_val, y_val):
            return origin_point + (x_val / 10.0) * 6 * RIGHT + (y_val / 1.5) * 4 * UP

        # KNN Curve: Overfits (dips then rises)
        knn_points = []
        for x_step in np.linspace(1, 10, 50):
            # Parabolic shape: y = 0.03*(x-4)^2 + 0.3
            y_val = 0.03 * (x_step - 4)**2 + 0.3
            knn_points.append(to_coords(x_step, y_val))
        
        knn_curve = VMobject(color=RED)
        knn_curve.set_points_as_corners(knn_points)
        knn_text = Text("KNN", color=RED, font_size=18).next_to(knn_points[-1], UP)

        # Lasso/Ridge Curve: Steady decline
        reg_points = []
        for x_step in np.linspace(1, 10, 50):
            # Exponential decay: y = 0.8 * exp(-0.2 * (x-1)) + 0.1
            y_val = 0.8 * np.exp(-0.2 * (x_step - 1)) + 0.1
            reg_points.append(to_coords(x_step, y_val))
            
        reg_curve = VMobject(color=GREEN)
        reg_curve.set_points_as_corners(reg_points)
        reg_text = Text("Lasso/Ridge", color=GREEN, font_size=18).next_to(reg_points[-1], DOWN)

        self.play(Create(knn_curve), Write(knn_text))
        self.wait(0.5)
        self.play(Create(reg_curve), Write(reg_text))
        self.wait(1)

        # 3. Lasso cost function formula
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            color=WHITE
        ).scale(0.8).to_edge(UP, buff=0.5)
        
        box = SurroundingRectangle(formula, color=YELLOW, buff=0.1)
        self.play(Write(formula), Create(box))
        self.wait(1)

        # 4. Coefficient Bar Chart
        # Create 150 small bars
        bar_start_pos = RIGHT * 2 + DOWN * 1.5
        bars = VGroup()
        np.random.seed(42)
        heights = np.random.uniform(0.1, 1.2, 150)
        
        for i in range(150):
            bar = Line(ORIGIN, UP * heights[i], stroke_width=1, color=BLUE)
            bars.add(bar)
        
        bars.arrange(RIGHT, buff=0.01).move_to(bar_start_pos + RIGHT * 1.5)
        bar_title = Text("Coefficients", font_size=20).next_to(bars, UP)

        self.play(
            LaggedStart(*[Create(b) for b in bars], lag_ratio=0.005),
            Write(bar_title),
            run_time=2
        )
        self.wait(1)

        # 5. Animate regularized shrinking (Lambda increases)
        # Many coefficients hit zero in Lasso
        shrink_animations = []
        for i in range(len(bars)):
            # Randomly decide if a coefficient vanishes (feature selection) or just shrinks
            if i % 3 != 0: 
                # Vanish
                shrink_animations.append(bars[i].animate.set_height(0.001).set_opacity(0))
            else:
                # Shrink
                shrink_animations.append(bars[i].animate.scale(0.3, about_edge=DOWN))

        lambda_label = MathTex(r"\lambda \text{ increases}", color=YELLOW).scale(0.7).next_to(box, DOWN)
        
        self.play(
            Write(lambda_label),
            *shrink_animations,
            run_time=3
        )
        self.wait(2)

        # Cleanup
        self.play(
            *[FadeOut(m) for m in [x_axis, y_axis, x_label, y_label, knn_curve, knn_text, 
                                   reg_curve, reg_text, formula, box, bars, bar_title, lambda_label]]
        )
        self.wait(1)
