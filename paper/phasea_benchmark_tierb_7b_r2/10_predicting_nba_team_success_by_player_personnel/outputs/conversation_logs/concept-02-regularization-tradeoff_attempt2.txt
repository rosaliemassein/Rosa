from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Title and Header (using Text instead of Title)
        header = Text("Regularization Tradeoff", font_size=32).to_edge(UP)
        intro_text = Text(
            "KNN overfits with more features, while Lasso stays stable.",
            font_size=24
        ).next_to(header, DOWN)
        
        self.add(header)
        self.play(Write(intro_text))

        # 2. Manual Axes Construction (since Axes is disallowed)
        # We define a coordinate system manually
        origin_point = np.array([-4, -1, 0])
        x_axis = Line(origin_point, origin_point + RIGHT * 8, color=WHITE)
        y_axis = Line(origin_point, origin_point + UP * 4, color=WHITE)
        
        x_label = Text("Number of Players", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("MSE", font_size=20).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))

        # Coordinate mapping helper: x [1, 10] -> [0, 8], y [0, 5] -> [0, 4]
        def c2p(x, y):
            x_mapped = (x - 1) * (8 / 9)
            y_mapped = y * (4 / 5)
            return origin_point + np.array([x_mapped, y_mapped, 0])

        # 3. Plotting Curves
        # KNN Curve: Dips then rises sharply
        knn_points = [
            c2p(x, 0.2 * (x - 3.5)**2 + 1.2)
            for x in np.linspace(1, 10, 50)
        ]
        knn_line = VMobject(color=RED)
        knn_line.set_points_as_corners(knn_points)
        
        knn_tag = Text("KNN", color=RED, font_size=18).move_to(c2p(9, 4.5))

        # Lasso/Ridge Curve: Steady decline
        reg_points = [
            c2p(x, 3.5 * np.exp(-0.4 * (x - 1)) + 0.5)
            for x in np.linspace(1, 10, 50)
        ]
        reg_line = VMobject(color=GREEN)
        reg_line.set_points_as_corners(reg_points)
        
        reg_tag = Text("Lasso / Ridge", color=GREEN, font_size=18).move_to(c2p(9, 1.2))

        self.play(Create(knn_line), Write(knn_tag))
        self.wait(0.5)
        self.play(Create(reg_line), Write(reg_tag))

        # 4. Lasso Cost Function Formula
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            color=YELLOW
        ).scale(0.8).shift(DOWN * 1.5 + RIGHT * 2.5)
        
        self.play(Write(formula))

        # 5. Coefficient Bar Chart (using Lines)
        # Recreating a visual bar chart of 150 coefficients shrinking
        n_coeffs = 80 # Reduced count slightly for cleaner visual in limited space
        np.random.seed(42)
        initial_heights = np.random.uniform(0.1, 1.0, n_coeffs)
        
        bar_group = VGroup()
        bar_start_x = -5.5
        bar_y = -3.2
        
        for i in range(n_coeffs):
            h = initial_heights[i]
            # Create a line as a bar
            l = Line(
                np.array([bar_start_x + i * 0.08, bar_y, 0]),
                np.array([bar_start_x + i * 0.08, bar_y + h, 0]),
                stroke_width=2,
                color=BLUE
            )
            bar_group.add(l)

        chart_label = Text("Model Coefficients", font_size=16).next_to(bar_group, UP, buff=0.1)
        
        self.play(FadeIn(bar_group), Write(chart_label))
        self.wait(1)

        # 6. Animation: Shrinking and Vanishing (Regularization effect)
        vanishing_group = VGroup()
        for i, bar in enumerate(bar_group):
            start_p = bar.get_start()
            # Most coefficients go to zero (Lasso feature selection)
            if i % 10 == 0:
                end_h = bar.get_height() * 0.4
            else:
                end_h = 0.01
                
            new_bar = Line(
                start_p,
                start_p + UP * end_h,
                stroke_width=2,
                color=BLUE
            )
            if end_h < 0.05:
                new_bar.set_stroke(opacity=0)
            vanishing_group.add(new_bar)

        self.play(
            Transform(bar_group, vanishing_group),
            formula.animate.set_color(WHITE).scale(1.1),
            run_time=3
        )
        
        # Summary text
        summary = Text("Lasso zero-out noise stats.", font_size=24, color=YELLOW).to_edge(DOWN)
        self.play(Write(summary))
        self.wait(2)
