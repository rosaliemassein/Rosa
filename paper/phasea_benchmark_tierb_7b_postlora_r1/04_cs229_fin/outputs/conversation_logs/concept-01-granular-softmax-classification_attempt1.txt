from manim import *

class GranularSoftmaxClassification(Scene):
    def construct(self):
        # 1. Left: Plastic bottle image (Placeholder for image_ref: img-12.jpeg)
        try:
            plastic_bottle = ImageMobject("img-12.jpeg").scale(0.8)
        except:
            plastic_bottle = VGroup(
                Square(color=BLUE, fill_opacity=0.3),
                Text("Plastic Bottle", font_size=14)
            ).scale(0.8)
        
        plastic_bottle.to_edge(LEFT, buff=0.5).shift(UP * 0.5)

        # 2. CNN layers: series of vertical rectangles
        cnn_layers = VGroup(*[
            Rectangle(width=0.3, height=1.8, fill_opacity=0.2, color=BLUE_B)
            for _ in range(4)
        ]).arrange(RIGHT, buff=0.2).next_to(plastic_bottle, RIGHT, buff=0.8)

        # 3. Data flow arrows
        arrow1 = Arrow(plastic_bottle.get_right(), cnn_layers.get_left(), buff=0.1)
        
        self.play(FadeIn(plastic_bottle))
        self.play(Create(cnn_layers), GrowArrow(arrow1))

        # 4. Column vector of 20 raw values (MathTex labeled 'x_i')
        raw_vector = MathTex(
            r"x_i = \begin{bmatrix} x_1 \\ \vdots \\ x_{20} \end{bmatrix}",
            font_size=36
        ).next_to(cnn_layers, RIGHT, buff=0.8)
        
        arrow2 = Arrow(cnn_layers.get_right(), raw_vector.get_left(), buff=0.1)
        
        self.play(GrowArrow(arrow2), Write(raw_vector))
        self.wait(1)

        # 5. Softmax formula appearing at the top
        softmax_formula = MathTex(
            r"softmax(x_{i})=\frac{e^{x_{i}}}{\sum_{j=1}^{n}e^{x_{j}}}",
            font_size=40
        ).to_edge(UP, buff=0.5)
        
        self.play(Write(softmax_formula))
        self.wait(1)

        # 6. Transform raw values into a horizontal BarChart
        # We build the chart manually using Axes and Rectangles to ensure compatibility
        chart_axes = Axes(
            x_range=[0, 1.1, 0.5],
            y_range=[0, 21, 5],
            x_length=5,
            y_length=4,
            axis_config={"include_tip": False},
            tips=False
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)
        
        chart_label = Text("Probabilities", font_size=20).next_to(chart_axes.x_axis, DOWN)

        # Generate 20 probability values
        probs = [0.02] * 20
        probs[14] = 0.85 # The 15th category is "Plastic Bottle"
        probs[5] = 0.08
        
        bars = VGroup()
        for i, p in enumerate(probs):
            # Create a rectangle whose width is mapped to the x-axis scale
            bar_height = 0.12
            bar_width = chart_axes.x_axis.number_to_point(p)[0] - chart_axes.x_axis.number_to_point(0)[0]
            
            bar = Rectangle(
                width=bar_width, 
                height=bar_height,
                fill_opacity=0.8,
                color=BLUE_C,
                stroke_width=1
            )
            # Position the bar based on the y-axis
            bar_pos = chart_axes.c2p(0, i + 1)
            bar.move_to(bar_pos, aligned_edge=LEFT)
            bars.add(bar)

        # Transition to chart
        self.play(
            FadeOut(plastic_bottle),
            FadeOut(cnn_layers),
            FadeOut(arrow1),
            FadeOut(arrow2),
            raw_vector.animate.scale(0.6).to_edge(LEFT, buff=1),
            Create(chart_axes),
            Write(chart_label)
        )
        
        self.play(Create(bars, lag_ratio=0.05, run_time=2))
        self.wait(1)

        # 7. Indicate the highest bar and label it 'Plastic Bottle'
        highest_bar = bars[14]
        highest_bar.set_color(YELLOW)
        
        label_text = Text("Plastic Bottle", font_size=24, color=YELLOW).next_to(highest_bar, RIGHT, buff=0.2)
        
        self.play(
            Indicate(highest_bar),
            Write(label_text)
        )
        
        self.wait(3)
