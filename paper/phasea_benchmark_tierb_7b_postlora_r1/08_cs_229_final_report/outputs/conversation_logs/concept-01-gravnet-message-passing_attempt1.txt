import numpy as np
from manim import *

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Define Formula
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # 2. Setup irregular grid positions (2D only)
        def get_irregular_grid(noise_scale=0.2):
            points = []
            for q in range(-2, 3):
                for r in range(-2, 3):
                    if abs(q + r) <= 2:
                        # Hexagonal base coordinates
                        x = 1.4 * q
                        y = 1.2 * (r + q / 2) * np.sqrt(3)
                        # Add irregularity
                        pos = np.array([x, y, 0])
                        noise = np.array([
                            np.random.uniform(-noise_scale, noise_scale),
                            np.random.uniform(-noise_scale, noise_scale),
                            0
                        ])
                        points.append(pos + noise)
            return points

        initial_positions = get_irregular_grid(noise_scale=0.1)
        target_positions = get_irregular_grid(noise_scale=0.7)
        
        # Central point index (roughly middle of the list)
        center_idx = len(initial_positions) // 2

        # 3. Create Dots
        dots = VGroup()
        for i, pos in enumerate(initial_positions):
            dot_color = RED if i == center_idx else BLUE
            dot = Dot(point=pos, color=dot_color, radius=0.12)
            dots.add(dot)

        self.add(dots)

        # 4. Create Arrows with Updaters
        # We manually update arrows since always_redraw is disallowed
        arrows = VGroup()
        
        for i in range(len(dots)):
            if i == center_idx:
                continue
            
            # Initial arrow construction
            start_p = dots[center_idx].get_center()
            end_p = dots[i].get_center()
            
            # Create a line that will act as our arrow
            arrow_line = Line(start_p, end_p, buff=0.1, color=GREEN)
            
            # Logic to update arrow properties based on distance
            def update_connection(mob, index=i):
                p1 = dots[center_idx].get_center()
                p2 = dots[index].get_center()
                dist = np.linalg.norm(p1 - p2)
                
                # Weight formula: V = exp(-d^2). 
                # Scaling distance by 0.5 to keep weights visible in Manim units
                d_val = dist * 0.4
                weight = np.exp(-(d_val**2))
                
                # Re-draw the line and tip based on current positions
                new_line = Line(p1, p2, buff=0.1, color=GREEN)
                new_line.set_stroke(width=weight * 10, opacity=weight)
                # Add tip manually to avoid disallowed high-level arrow symbols
                new_line.add_tip(tip_length=weight * 0.2, tip_width=weight * 0.2)
                mob.become(new_line)

            arrow_line.add_updater(update_connection)
            arrows.add(arrow_line)

        self.add(arrows)

        # 5. Animation sequence
        # Move dots to new "learned feature" positions
        dot_animations = []
        for i in range(len(dots)):
            dot_animations.append(dots[i].animate.move_to(target_positions[i]))

        self.wait(1)
        # Play movements - updaters handle the arrow scaling and positioning
        self.play(
            *dot_animations,
            run_time=4,
            rate_func=lambda t: t * t * (3 - 2 * t) # Custom smooth
        )
        
        self.wait(1)

        # Return to initial positions to show dynamism
        back_animations = []
        for i in range(len(dots)):
            back_animations.append(dots[i].animate.move_to(initial_positions[i]))
            
        self.play(
            *back_animations,
            run_time=3,
            rate_func=lambda t: t * t * (3 - 2 * t)
        )
        
        self.wait(2)
