from manim import *
import numpy as np

class SpatialGridEncoding(Scene):
    def construct(self):
        # Configuration
        n = 5
        board_radius = 1
        
        # 1. Create hexagonal grid of circles
        hex_grid = VGroup()
        for r in range(n):
            for q in range(n):
                # Axial to pixel mapping (simplified)
                x = (q + r * 0.5) * 0.6
                y = (r * np.sqrt(3) / 2) * 0.6
                circle = Circle(radius=0.2, color=WHITE, stroke_width=2)
                circle.move_to([x, y, 0])
                hex_grid.add(circle)
        
        hex_grid.move_to(ORIGIN)

        # Show hex board
        self.play(Create(hex_grid), run_time=1.5)
        self.wait(0.5)

        # 2. Simulate tilting into perspective
        # Since 3DScene is disallowed, we simulate the effect with scaling and shearing
        self.play(
            hex_grid.animate.stretch(0.5, dim=1).shear(0.2, direction=RIGHT),
            run_time=1.5
        )
        self.wait(0.5)

        # 3. Morph into a square grid
        # Create base square grid
        layer_target = VGroup(*[
            Square(side_length=0.6, stroke_width=1).set_fill(GREEN, opacity=0.1) 
            for _ in range(n * n)
        ]).arrange_in_grid(n, n, buff=0.05)
        
        # We also need to "tilt" the square grid to match the perspective simulation
        layer_target.stretch(0.5, dim=1).shear(0.2, direction=RIGHT)
        layer_target.move_to(ORIGIN)

        self.play(ReplacementTransform(hex_grid, layer_target), run_time=1.5)
        self.wait(0.5)

        # 4. Create three separate layers
        layer_self = layer_target.copy().set_color(BLUE)
        layer_opp = layer_target.copy().set_color(RED)

        # Highlight pieces in each layer
        # Layer 1: Self
        for i in [2, 7, 12]:
            layer_self[i].set_fill(BLUE, opacity=0.8)
        # Layer 2: Target (already created, just highlight a few)
        for i in [22, 23, 24]:
            layer_target[i].set_fill(GREEN, opacity=0.8)
        # Layer 3: Opponent
        for i in [10, 11, 16]:
            layer_opp[i].set_fill(RED, opacity=0.8)

        # 5. Split layers vertically (simulated in 2D)
        self.play(
            layer_self.animate.shift(UP * 2.2),
            layer_opp.animate.shift(DOWN * 2.2),
            run_time=2
        )
        
        # Label the layers
        label_self = Text("Current Player pieces", color=BLUE).scale(0.5).next_to(layer_self, LEFT)
        label_target = Text("Target Destination", color=GREEN).scale(0.5).next_to(layer_target, LEFT)
        label_opp = Text("Opponent pieces", color=RED).scale(0.5).next_to(layer_opp, LEFT)

        self.play(Write(label_self), Write(label_target), Write(label_opp))
        self.wait(1)

        # 6. Show Dimensions and Formula
        dimensions = MathTex(r"n \times n \times 3", color=YELLOW).scale(1.2)
        dimensions.to_edge(RIGHT, buff=1)
        
        formula = MathTex(r"n = 4 \times \text{board radius} + 1", color=WHITE).scale(0.8)
        formula.to_edge(UP)

        self.play(FadeIn(dimensions), Write(formula))
        self.wait(3)

        # Final emphasis on the stack
        stack_bracket = Brace(VGroup(layer_self, layer_opp), direction=RIGHT, color=WHITE)
        stack_text = stack_bracket.get_text("3 Layers").scale(0.7)
        
        self.play(Create(stack_bracket), FadeIn(stack_text))
        self.wait(2)
