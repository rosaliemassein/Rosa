from manim import *
import numpy as np

class FeatureMatrixScene(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex("n = m \\times p").to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Create silhouettes of 15 players
        # Each player is a simple circle (head) and rounded rectangle (body)
        def create_player():
            head = Circle(radius=0.12, color=GRAY, fill_opacity=1)
            body = RoundedRectangle(corner_radius=0.1, height=0.4, width=0.3, color=GRAY, fill_opacity=1).next_to(head, DOWN, buff=0.05)
            return VGroup(head, body)

        players = VGroup(*[create_player() for _ in range(15)])
        players.arrange(DOWN, buff=0.1).to_edge(LEFT, buff=0.7)
        
        self.play(Create(players), run_time=1.5)
        self.wait(0.5)

        # 3. Animate 'minutes played' filter
        # Highlighting top 10 players
        filter_rect = Rectangle(width=0.6, height=0.6, color=YELLOW).move_to(players[0])
        self.play(Create(filter_rect))
        
        # Scan through the players
        for i in range(1, 10):
            self.play(filter_rect.animate.move_to(players[i]), run_time=0.1)
        
        top_10 = players[:10]
        others = players[10:]
        
        self.play(
            top_10.animate.set_color(WHITE),
            others.animate.set_opacity(0.2),
            filter_rect.animate.scale(1.1).set_stroke(width=5)
        )
        self.play(FadeOut(filter_rect))

        # 4. Create the 10x15 Feature Matrix
        # 10 rows (players), 15 columns (stats)
        rows = 10
        cols = 15
        grid = VGroup()
        for r in range(rows):
            row_group = VGroup()
            for c in range(cols):
                square = Square(side_length=0.35, stroke_width=1, stroke_color=WHITE)
                row_group.add(square)
            row_group.arrange(RIGHT, buff=0.05)
            grid.add(row_group)
        grid.arrange(DOWN, buff=0.05).shift(RIGHT * 2)

        # 5. Transition players into the rows of the grid
        self.play(
            ReplacementTransform(top_10.copy(), grid),
            run_time=2
        )

        # 6. Add labels for performance metrics
        stat_labels_list = ["PTS", "REB", "AST", "STL", "BLK", "TOV", "PF", "FG%", "3P%", "FT%", "+/-", "PIE", "MIN", "ORB", "DRB"]
        labels = VGroup(*[
            Text(stat, font_size=10).next_to(grid[0][i], UP, buff=0.2).rotate(45 * DEGREES)
            for i, stat in enumerate(stat_labels_list)
        ])
        self.play(Write(labels))

        # 7. Fill grid cells with data (varying blue intensities)
        fill_animations = []
        for row in grid:
            for cell in row:
                val = np.random.rand()
                fill_animations.append(cell.animate.set_fill(BLUE, opacity=val))
        
        self.play(*fill_animations, run_time=2)
        self.wait(1)

        # 8. Collapse 2D grid into a single horizontal vector (150 elements)
        vector = VGroup(*[
            Square(side_length=0.08, fill_color=BLUE, fill_opacity=np.random.rand(), stroke_width=0.1)
            for _ in range(150)
        ]).arrange(RIGHT, buff=0.01).to_edge(DOWN, buff=1.5)

        vector_label = Text("Input Vector (n=150)", font_size=20).next_to(vector, UP)

        self.play(
            FadeOut(labels),
            FadeOut(players),
            FadeOut(formula),
            ReplacementTransform(grid, vector),
            Write(vector_label),
            run_time=3
        )

        self.wait(2)
