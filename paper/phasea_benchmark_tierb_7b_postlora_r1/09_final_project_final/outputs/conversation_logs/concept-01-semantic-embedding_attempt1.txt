from manim import *

class Concept01SemanticEmbedding(Scene):
    def construct(self):
        # Use Axes instead of NumberPlane as it's more reliable in this environment
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6,
            axis_config={"color": GREY, "stroke_width": 1},
        ).set_opacity(0.2)
        self.add(axes)

        # 1. Pop up floating text labels
        dl_text = Text("Deep Learning", color=BLUE).scale(0.6).move_to([2, 1.5, 0])
        nn_text = Text("Neural Networks", color=GREEN).scale(0.6).move_to([2.8, 1.8, 0])
        rb_text = Text("Robotics", color=RED).scale(0.6).move_to([-3, -1.0, 0])

        self.play(Write(dl_text), Write(nn_text), Write(rb_text))
        self.wait(1)

        # 2. Animate the text labels transforming into Dot objects
        dl_dot = Dot([2, 1, 0], color=BLUE)
        nn_dot = Dot([2.5, 1.3, 0], color=GREEN)
        rb_dot = Dot([-3, -1.5, 0], color=RED)

        self.play(
            Transform(dl_text, dl_dot),
            Transform(nn_text, nn_dot),
            Transform(rb_text, rb_dot)
        )
        self.wait(1)

        # 3. Use lines to show distances (Using low opacity lines as DashedLine is disallowed)
        # Deep Learning and Neural Networks are close
        line_dl_nn = Line(dl_dot.get_center(), nn_dot.get_center(), color=WHITE).set_opacity(0.4)
        # Robotics is further away
        line_dl_rb = Line(dl_dot.get_center(), rb_dot.get_center(), color=WHITE).set_opacity(0.4)

        self.play(Create(line_dl_nn), Create(line_dl_rb))
        self.wait(1)

        # 4. Display the formula
        formula = MathTex(r"v_{phrase} = \frac{1}{n} \sum_{i=1}^{n} v_{i}", color=BLUE_B).scale(0.8)
        formula.to_edge(DOWN, buff=0.8)
        self.play(Write(formula))
        self.wait(1)

        # 5. Averaging vectors: Machine + Learning
        # Clear previous elements to focus on formula example
        self.play(
            FadeOut(line_dl_nn), FadeOut(line_dl_rb), 
            FadeOut(dl_text), FadeOut(nn_text), FadeOut(rb_text)
        )

        m_origin = ORIGIN
        m_vec_end = [-2, 1.5, 0]
        l_vec_end = [2, 0.5, 0]
        avg_vec_end = [0, 1, 0] # Average of [-2, 1.5] and [2, 0.5]

        # Individual word vectors
        m_vec = Line(m_origin, m_vec_end, color=BLUE, buff=0).add_tip(tip_length=0.2)
        l_vec = Line(m_origin, l_vec_end, color=GREEN, buff=0).add_tip(tip_length=0.2)
        
        m_label = Text("Machine", color=BLUE).scale(0.5).next_to(m_vec_end, UP)
        l_label = Text("Learning", color=GREEN).scale(0.5).next_to(l_vec_end, UP)

        self.play(
            Create(m_vec), Write(m_label),
            Create(l_vec), Write(l_label)
        )
        self.wait(1)

        # 6. Transform them into a single central vector (Average)
        avg_vec = Line(m_origin, avg_vec_end, color=YELLOW, buff=0).add_tip(tip_length=0.2)
        avg_label = Text("Machine Learning", color=YELLOW).scale(0.6).next_to(avg_vec_end, UP)
