from manim import *
import numpy as np

class InverseMapping(Scene):
    def construct(self):
        # 1. Diagram Setup (Left and Right Boxes)
        input_rect = Rectangle(height=2.5, width=4, color=BLUE).shift(LEFT * 3.5 + UP * 1.5)
        input_title = Text("Desired Behavior", font_size=24).move_to(input_rect.get_top() + DOWN * 0.4)
        input_content = VGroup(
            Text("Rise Time", font_size=18),
            Text("Steady State Error", font_size=18),
            Text("Stability", font_size=18)
        ).arrange(DOWN, aligned_edge=LEFT, buff=0.2).next_to(input_title, DOWN, buff=0.3)
        input_group = VGroup(input_rect, input_title, input_content)

        output_rect = Rectangle(height=2.5, width=4, color=GREEN).shift(RIGHT * 3.5 + UP * 1.5)
        output_title = Text("PID Constants", font_size=24).move_to(output_rect.get_top() + DOWN * 0.4)
        output_content = MathTex("K_p, K_i, K_d", font_size=36).move_to(output_rect.get_center())
        output_group = VGroup(output_rect, output_title, output_content)

        arrow = Line(input_group.get_right(), output_group.get_left(), buff=0.2).add_tip()
        arrow_label = Text("Flips the script", font_size=20).next_to(arrow, UP)

        diagram = VGroup(input_group, output_group, arrow, arrow_label)

        # 2. Intro Animations
        self.play(Create(input_group))
        self.wait(0.5)
        self.play(Create(arrow), Write(arrow_label))
        self.play(Create(output_group))
        self.wait(1)

        # 3. Rotate Diagram 180 Degrees
        # To keep text readable, we rotate the whole group, then rotate texts back
        self.play(diagram.animate.rotate(PI), run_time=2)
        self.play(
            input_title.animate.rotate(PI),
            input_content.animate.rotate(PI),
            output_title.animate.rotate(PI),
            output_content.animate.rotate(PI),
            arrow_label.animate.rotate(PI)
        )
        self.wait(1)

        # 4. Manual Plotting (Avoiding Axes and DashedLine)
        origin_point = DOWN * 2.5 + LEFT * 4
        x_axis = Line(origin_point, origin_point + RIGHT * 8).add_tip()
        y_axis = Line(origin_point, origin_point + UP * 3).add_tip()
        x_label = Text("Time", font_size=16).next_to(x_axis, RIGHT)
        y_label = Text("RPM", font_size=16).next_to(y_axis, UP)
        
        # Setpoint line (using regular Line with low opacity)
        setpoint_y = 1.5
        setpoint = Line(
            origin_point + UP * setpoint_y, 
            origin_point + UP * setpoint_y + RIGHT * 8, 
            color=YELLOW, 
            stroke_opacity=0.5
        )

        # Curve generation: 1 - exp(-0.5x) * cos(2.5x)
        # Scaled to fit our manual axes
        curve_points = []
        for val in np.linspace(0, 8, 100):
            # x_coord relative to origin
            # y_coord scaled: setpoint_y is the target (1.0 in function units)
            func_y = 1 - np.exp(-0.5 * val) * np.cos(2.5 * val)
            curve_points.append(origin_point + RIGHT * val + UP * (func_y * setpoint_y))
        
        curve = VMobject(color=BLUE)
        curve.set_points_as_corners(curve_points)

        # 5. Manual Braces (Avoiding Brace class)
        # Rise Time marker (approx start to first crossing)
        rise_line = Line(origin_point + DOWN * 0.2, origin_point + RIGHT * 0.7 + DOWN * 0.2, color=WHITE)
        rise_tick_1 = Line(origin_point + DOWN * 0.1, origin_point + DOWN * 0.3)
        rise_tick_2 = Line(origin_point + RIGHT * 0.7 + DOWN * 0.1, origin_point + RIGHT * 0.7 + DOWN * 0.3)
        rise_label = Text("Rise Time", font_size=14).next_to(rise_line, DOWN, buff=0.1)
        rise_measure = VGroup(rise_line, rise_tick_1, rise_tick_2, rise_label)

        # Steady State Error marker (at the end)
        error_x = 7.5
        error_top = origin_point + RIGHT * error_x + UP * setpoint_y
        error_bottom = curve_points[-10] # Point near the end
        error_line = Line(error_bottom, error_top, color=RED)
        error_tick_1 = Line(error_top + LEFT * 0.1, error_top + RIGHT * 0.1)
        error_tick_2 = Line(error_bottom + LEFT * 0.1, error_bottom + RIGHT * 0.1)
        error_label = Text("Steady State Error", font_size=14).next_to(error_line, RIGHT, buff=0.1)
        error_measure = VGroup(error_line, error_tick_1, error_tick_2, error_label)

        # 6. Final Animation
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint))
        self.play(Create(curve), run_time=3)
        self.play(Create(rise_measure))
        self.play(Create(error_measure))
        
        # Formula Overlay
        formula = MathTex("U = K_p e + K_i \\int e dt + K_d \\frac{de}{dt}", font_size=30)
        formula.to_corner(UP + RIGHT, buff=0.5)
        self.play(Write(formula))
        
        self.wait(3)
