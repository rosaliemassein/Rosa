import math
from manim import *

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Input/Output Boxes and Arrow
        input_rect = Rectangle(color=BLUE, fill_opacity=0.5, height=2.0, width=4.0)
        input_label = VGroup(
            Text("Desired Behavior", font_size=24),
            Text("(Rise Time, SSE, Stability)", font_size=16)
        ).arrange(DOWN, buff=0.1).move_to(input_rect.get_center())
        input_box = VGroup(input_rect, input_label)

        output_rect = Rectangle(color=GREEN, fill_opacity=0.5, height=2.0, width=4.0)
        output_label = Text("PID Constants", font_size=24).move_to(output_rect.get_center())
        output_box = VGroup(output_rect, output_label)

        # Arrangement
        boxes = VGroup(input_box, output_box).arrange(RIGHT, buff=1.5).to_edge(UP, buff=1.0)
        arrow = Arrow(input_box.get_right(), output_box.get_left(), color=WHITE)
        
        formula = MathTex(
            r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", 
            font_size=32, color=YELLOW
        ).next_to(boxes, DOWN, buff=0.4)
        
        diagram = VGroup(boxes, arrow, formula)

        # 2. Animation: Initial Show and Rotate
        self.play(FadeIn(input_box), FadeIn(output_box), GrowArrow(arrow))
        self.play(Write(formula))
        self.wait(1)
        
        # Flips the script: Rotate 180 degrees
        self.play(diagram.animate.rotate(PI))
        self.wait(1)

        # 3. Motor Velocity Plot (Manual implementation to avoid Axes/DashedLine/Rotate issues)
        origin_point = DOWN * 2.5 + LEFT * 3
        x_axis = Line(origin_point, origin_point + RIGHT * 6, color=WHITE)
        y_axis = Line(origin_point, origin_point + UP * 3, color=WHITE)
        
        x_label = Text("Time", font_size=18).next_to(x_axis, DOWN, buff=0.1)
        y_label = Text("RPM", font_size=18).next_to(y_axis, LEFT, buff=0.1)
        
        # Setpoint line at y=2.0 (relative to origin)
        setpoint = Line(
            origin_point + UP * 2.0, 
            origin_point + UP * 2.0 + RIGHT * 5.5, 
            color=YELLOW
        )
        setpoint_text = Text("Setpoint", font_size=14, color=YELLOW).next_to(setpoint, UP, buff=0.05)

        # Generate Curve Points manually
        # Curve: y = 2.0 * (1 - e^-0.8t * cos(3t))
        curve_points = []
        for i in range(101):
            t_val = i / 20.0 # 0 to 5 seconds
            # y math value
            y_math = 2.0 * (1 - math.exp(-0.8 * t_val) * math.cos(3 * t_val))
            # mapping to screen: 1 unit in math = 1 unit in manim (approx)
            # Scaling t_val slightly to fit 6 units width
            p = origin_point + RIGHT * t_val * 1.1 + UP * y_math * 1.0
            curve_points.append(p)
            
        curve = VMobject(color=RED)
        curve.set_points_as_corners(curve_points)

        # 4. Metrics: Rise Time and SSE
        # Rise time is approx where it first hits y=2.0 (t ~= 0.5)
        rise_line_start = origin_point
        rise_line_end = origin_point + RIGHT * 0.5 * 1.1
        rise_ref_line = Line(rise_line_start, rise_line_end, stroke_width=0)
        rise_brace = Brace(rise_ref_line, DOWN, buff=0.05)
        rise_text = rise_brace.get_text("Rise Time", font_size=16)

        # SSE at the end of the curve (t=5.0)
        end_t = 5.0
        end_y = 2.0 * (1 - math.exp(-0.8 * end_t) * math.cos(3 * end_t))
        sse_line_start = origin_point + RIGHT * end_t * 1.1 + UP * end_y
        sse_line_end = origin_point + RIGHT * end_t * 1.1 + UP * 2.0
        sse_ref_line = Line(sse_line_start, sse_line_end, stroke_width=0)
        sse_brace = Brace(sse_ref_line, RIGHT, buff=0.05)
        sse_text = sse_brace.get_text("SSE", font_size=16)

        # 5. Final Plot Animations
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint), Write(setpoint_text))
        self.play(Create(curve), run_time=3)
        self.play(
            FadeIn(rise_brace), Write(rise_text),
            FadeIn(sse_brace), Write(sse_text)
        )
        self.wait(2)
