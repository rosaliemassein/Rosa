from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Manual Axes (Avoiding 'Axes' class)
        origin_pt = LEFT * 3 + DOWN * 1
        x_axis = Line(origin_pt, origin_pt + RIGHT * 5, color=WHITE)
        y_axis = Line(origin_pt, origin_pt + UP * 4, color=WHITE)
        axes_group = VGroup(x_axis, y_axis)

        # 2. Angle Tracking (Avoiding 'ValueTracker' class)
        # Using a DecimalNumber as a value container
        theta_val = DecimalNumber(70).hide()
        self.add(theta_val)

        # 3. Vectors
        # Vector A (fixed)
        vec_a = Line(origin_pt, origin_pt + RIGHT * 4, color=RED).add_tip()
        label_a = MathTex(r"\mathbf{A}", color=RED).next_to(vec_a.get_end(), DOWN)

        # Vector B (dynamic) - Using add_updater instead of always_redraw
        vec_b = Line(origin_pt, origin_pt + RIGHT * 4, color=GREEN).add_tip()
        
        def update_vec_b(mob):
            angle_rad = theta_val.get_value() * DEGREES
            direction = np.array([np.cos(angle_rad), np.sin(angle_rad), 0])
            new_line = Line(origin_pt, origin_pt + direction * 4, color=GREEN).add_tip()
            mob.become(new_line)
        
        vec_b.add_updater(update_vec_b)

        # 4. Arc for Angle (Avoiding 'Angle' class)
        angle_arc = Arc(radius=0.8, start_angle=0, angle=70 * DEGREES, arc_center=origin_pt, color=YELLOW)
        
        def update_arc(mob):
            angle_rad = theta_val.get_value() * DEGREES
            new_arc = Arc(
                radius=0.8, 
                start_angle=0, 
                angle=angle_rad, 
                arc_center=origin_pt, 
                color=YELLOW
            )
            mob.become(new_arc)
        
        angle_arc.add_updater(update_arc)

        # 5. Similarity Label
        sim_label = MathTex(r"\cos(\theta) = ", "0.34").to_edge(UP).shift(LEFT * 2)
        
        def update_sim_label(mob):
            cos_val = np.cos(theta_val.get_value() * DEGREES)
            new_label = MathTex(
                r"\cos(\theta) = ", 
                f"{max(0, cos_val):.2f}"
            ).move_to(mob.get_center())
            mob.become(new_label)
            
        sim_label.add_updater(update_sim_label)

        # 6. Similarity Matrix
        matrix_cells = VGroup()
        for i in range(5):
            for j in range(5):
                sq = Square(side_length=0.6, fill_opacity=1, stroke_width=1, stroke_color=GRAY)
                sq.set_fill(BLUE, opacity=0.2)
                matrix_cells.add(sq)
        
        matrix_cells.arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=1)
        matrix_title = Text("Similarity Matrix", font_size=24).next_to(matrix_cells, UP)
        
        # Center cell reflects the vector alignment
        active_cell = matrix_cells[12] 
        
        def update_cell(mob):
            val = np.cos(theta_val.get_value() * DEGREES)
            # Use manual color interpolation for reliability
            col = interpolate_color(BLUE, YELLOW, max(0, val))
            mob.set_fill(col, opacity=1)
            
        active_cell.add_updater(update_cell)

        # 7. Animation
        self.play(Create(axes_group), Write(matrix_title))
        self.play(Create(matrix_cells))
        self.play(Create(vec_a), FadeIn(label_a))
        self.play(Create(vec_b), Create(angle_arc), Write(sim_label))
        
        self.wait(1)
        
        # Animate the "ValueTracker" (theta_val)
        # Move B towards A (0 degrees)
        self.play(
            theta_val.animate.set_value(0),
            run_time=3,
            rate_func=rate_functions.linear
        )
        self.wait(1)
        
        # Move B away from A (90 degrees)
        self.play(
            theta_val.animate.set_value(90),
            run_time=3,
            rate_func=rate_functions.linear
        )
        self.wait(1)

        # Show final formula
        formula = MathTex(
            r"\text{similarity} = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=32
        ).next_to(sim_label, DOWN, buff=0.5)
        self.play(Write(formula))
        
        self.wait(2)

        # Remove updaters
        vec_b.clear_updaters()
        angle_arc.clear_updaters()
        sim_label.clear_updaters()
        active_cell.clear_updaters()
