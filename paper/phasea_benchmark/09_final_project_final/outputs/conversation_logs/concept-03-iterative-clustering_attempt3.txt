from manim import *
import numpy as np

class IterativeClustering(Scene):
    def construct(self):
        # 1. Setup - Create 27 dots scattered randomly
        title = Text("Iterative Clustering Logic", font_size=32).to_edge(UP)
        voice_over_text = "Standard clustering methods often fail on sparse data,\nso this system uses an iterative approach."
        narration = Text(voice_over_text, font_size=18, line_spacing=1.2).to_edge(DOWN, buff=0.5)
        
        dots = VGroup(*[
            Dot(point=[np.random.uniform(-5, 5), np.random.uniform(-2, 2.5), 0], radius=0.08)
            for _ in range(27)
        ])
        
        self.add(title, narration, dots)
        self.wait(1)

        available_dots = list(dots)
        cluster_colors = [RED, BLUE, GREEN]
        
        # 2. Iterative Clustering Process
        for cluster_idx in range(3):
            current_color = cluster_colors[cluster_idx]
            current_cluster = []
            
            # Find initial seed pair (the two closest unassigned dots)
            min_dist = float('inf')
            seed_pair = (None, None)
            
            for i in range(len(available_dots)):
                for j in range(i + 1, len(available_dots)):
                    d1 = available_dots[i]
                    d2 = available_dots[j]
                    dist = np.linalg.norm(d1.get_center() - d2.get_center())
                    if dist < min_dist:
                        min_dist = dist
                        seed_pair = (d1, d2)
            
            d1, d2 = seed_pair
            seed_center = (d1.get_center() + d2.get_center()) / 2
            
            # Highlight seed pair with bright rings
            ring1 = Circle(radius=0.15, color=YELLOW, stroke_width=3).move_to(d1.get_center())
            ring2 = Circle(radius=0.15, color=YELLOW, stroke_width=3).move_to(d2.get_center())
            
            self.play(
                Create(ring1),
                Create(ring2),
                d1.animate.set_color(current_color),
                d2.animate.set_color(current_color),
                run_time=0.6
            )
            
            current_cluster.extend([d1, d2])
            available_dots.remove(d1)
            available_dots.remove(d2)
            
            # Greedily pull in the next most similar unassigned dots (7 more to reach 9)
            for _ in range(7):
                best_dot = None
                best_dist = float('inf')
                
                # Find the next closest unassigned dot to the existing cluster
                for d_avail in available_dots:
                    for d_clust in current_cluster:
                        dist = np.linalg.norm(d_avail.get_center() - d_clust.get_center())
                        if dist < best_dist:
                            best_dist = dist
                            best_dot = d_avail
                
                # Search circle expands from the cluster center
                cluster_center = VGroup(*current_cluster).get_center()
                radius_to_target = np.linalg.norm(best_dot.get_center() - cluster_center)
                
                search_circle = Circle(radius=0.01, color=current_color, stroke_opacity=0.4).move_to(cluster_center)
                
                # Animate search expansion
                self.play(
                    search_circle.animate.set_width(radius_to_target * 2),
                    run_time=0.25,
                    rate_func=lambda t: t # Linear expansion
                )
                
                # Dot changes color and gets absorbed
                self.play(
                    best_dot.animate.set_color(current_color),
                    FadeOut(search_circle, scale=1.1),
                    run_time=0.15
                )
                
                current_cluster.append(best_dot)
                available_dots.remove(best_dot)

            # Clean up highlight rings for this cluster
            self.play(FadeOut(ring1), FadeOut(ring2), run_time=0.3)

        # 3. Final Conclusion
        self.play(FadeOut(narration))
        final_text = "Each group of nine proposals now has\nthe highest possible thematic cohesion."
        narration = Text(final_text, font_size=20, color=YELLOW).to_edge(DOWN, buff=0.5)
        self.play(Write(narration))
        
        # Group and emphasize all clusters
        final_groups = VGroup(
            VGroup(*[d for d in dots if d.get_color() == RED]),
            VGroup(*[d for d in dots if d.get_color() == BLUE]),
            VGroup(*[d for d in dots if d.get_color() == GREEN])
        )
        
        self.play(final_groups.animate.scale(1.15), run_time=0.8)
        self.play(final_groups.animate.scale(1/1.15), run_time=0.8)
        self.wait(2)
