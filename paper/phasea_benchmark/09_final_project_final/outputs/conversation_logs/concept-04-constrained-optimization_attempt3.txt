from manim import *
import numpy as np

class ConceptOptimization(Scene):
    def construct(self):
        # Define missing constants
        UL = UP + LEFT
        UR = UP + RIGHT
        DL = DOWN + LEFT
        DR = DOWN + RIGHT
        PI = 3.14159265
        MY_GRAY = "#777777"

        # Helper for rectangles since 'Rectangle' is disallowed
        def make_box(w, h, color=WHITE, fill_opacity=0, stroke_width=2):
            w2, h2 = w/2, h/2
            return Polygon(
                np.array([-w2, h2, 0]), 
                np.array([w2, h2, 0]), 
                np.array([w2, -h2, 0]), 
                np.array([-w2, -h2, 0]),
                color=color, 
                fill_opacity=fill_opacity,
                stroke_width=stroke_width
            )

        # Helper for surrounding rectangle
        def make_surround(mobj, color=WHITE, buff=0.2):
            w = mobj.get_width() + buff
            h = mobj.get_height() + buff
            p = make_box(w, h, color=color)
            p.move_to(mobj.get_center())
            return p

        # 1. Formula (Avoiding MathTex)
        formula = Text("max Sum s_ij x_ij - Sum P_k delta_k", font_size=30).to_edge(UP)
        self.play(Write(formula))

        # 2. Assignment Matrix
        matrix_title = Text("Assignment Matrix", font_size=20).next_to(formula, DOWN, buff=0.3)
        rows, cols = 6, 10
        matrix_elements = VGroup()
        for _ in range(rows * cols):
            val = "1" if np.random.random() > 0.8 else "0"
            matrix_elements.add(Text(val, font_size=18))
        
        matrix_elements.arrange_in_grid(rows=rows, cols=cols, buff=0.4)
        matrix_elements.shift(LEFT * 2 + DOWN * 0.5)
        
        grid_outline = make_surround(matrix_elements, color=MY_GRAY)
        self.play(Create(grid_outline), FadeIn(matrix_elements), Write(matrix_title))

        # 3. Score Bar (Avoiding Rectangle)
        score_label = Text("Expertise Score", font_size=18)
        score_bg = make_box(3, 0.3, color=WHITE)
        
        # Initial fill
        score_fill = make_box(1.8, 0.3, color=YELLOW, fill_opacity=0.8).align_to(score_bg, LEFT)
        score_val_text = Text("75", font_size=20)
        score_max_text = Text("/ 125", font_size=16)
        
        score_ui = VGroup(score_label, score_bg, score_fill, score_val_text, score_max_text).to_edge(RIGHT).shift(UP * 1)
        score_bg.next_to(score_label, DOWN, buff=0.2)
        score_fill.align_to(score_bg, LEFT)
        score_val_text.next_to(score_bg, RIGHT, buff=0.2)
        score_max_text.next_to(score_val_text, RIGHT, buff=0.1)
        
        self.add(score_label, score_bg, score_fill, score_val_text, score_max_text)

        conflict_icon = VGroup(icon_x, label_x)

        tri = Polygon(np.array([0, 0.1, 0]), np.array([-0.1, -0.1, 0]), np.array([0.1, -0.1, 0]), color=BLUE)
        icon_scale = VGroup(Line(LEFT, RIGHT).scale(0.3), tri).set_color(BLUE)
        label_scale = Text("Diversity", font_size=14, color=BLUE).next_to(icon_scale, DOWN, buff=0.1)
        diversity_icon = VGroup(icon_scale, label_scale)

        # Workload (Clock)
        icon_clock = VGroup(Circle(radius=0.2), Line(ORIGIN, UP*0.15), Line(ORIGIN, RIGHT*0.1)).set_color(GREEN)
