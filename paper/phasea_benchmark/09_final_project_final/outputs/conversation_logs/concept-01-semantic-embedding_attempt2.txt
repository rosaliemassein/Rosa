from manim import *

class SemanticEmbedding(Scene):
    def construct(self):
        # 1. Use Axes as a substitute for NumberPlane (as NumberPlane is disallowed)
        axes = Axes(
            x_range=[-6, 6, 1],
            y_range=[-4, 4, 1],
            axis_config={"color": BLUE, "stroke_width": 2},
        )
        self.add(axes)

        # 2. Pop up floating text labels
        # Using 2D coordinates within the viewable range
        pos_dl = [2, 1.5, 0]
        pos_nn = [3, 1, 0]
        pos_robotics = [-3, -1.5, 0]

        label_dl = Text("Deep Learning", color=BLUE).scale(0.5).move_to([2, 2, 0])
        label_nn = Text("Neural Networks", color=YELLOW).scale(0.5).move_to([3, 1.5, 0])
        label_robotics = Text("Robotics", color=RED).scale(0.5).move_to([-3, -2, 0])

        self.play(Write(label_dl), Write(label_nn), Write(label_robotics))
        self.wait(1)

        # 3. Animate text labels transforming into Dot objects
        dot_dl = Dot(point=pos_dl, color=BLUE)
        dot_nn = Dot(point=pos_nn, color=YELLOW)
        dot_robotics = Dot(point=pos_robotics, color=RED)

        self.play(
            Transform(label_dl, dot_dl),
            Transform(label_nn, dot_nn),
            Transform(label_robotics, dot_robotics)
        )
        self.wait(1)

        # 4. Use lines to show distances (using Line since DashedLine was flagged)
        line_dl_nn = Line(pos_dl, pos_nn, color=WHITE, stroke_width=2)
        line_dl_robotics = Line(pos_dl, pos_robotics, color=WHITE, stroke_width=1).set_stroke(opacity=0.5)

        self.play(Create(line_dl_nn))
        self.play(Create(line_dl_robotics))
        self.wait(1)

        # 5. Clean up for the next section
        self.play(
            FadeOut(line_dl_nn), 
            FadeOut(line_dl_robotics),

        )

        pos_avg = [-1, 0.5, 0]

        dot_l = Dot(point=pos_l, color=BLUE)

        self.play(FadeIn(dot_l), Write(text_l))
        self.wait(1)
