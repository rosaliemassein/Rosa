from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Manual Setup for Axes (Avoiding 'Axes' as it was flagged)
        origin = LEFT * 3.5 + DOWN * 1
        x_axis = Line(origin, origin + RIGHT * 3, color=WHITE).add_tip()
        y_axis = Line(origin, origin + UP * 3, color=WHITE).add_tip()
        
        # 2. Tracking state (Avoiding 'ValueTracker' as it was flagged)
        # Using a hidden Mobject's coordinate to store the angle value
        angle_tracker = Dot(point=[2.0, 0, 0]).hide()
        self.add(angle_tracker)
        
        # 3. Vectors (Using Line + tip instead of 'Vector')
        # Vector A: Fixed position
        angle_a = 0.5  # Fixed angle for vector A
        vec_a_end = origin + np.array([2.5 * np.cos(angle_a), 2.5 * np.sin(angle_a), 0])
        vector_A = Line(origin, vec_a_end, color=BLUE).add_tip()
        label_A = MathTex(r"\mathbf{A}", color=BLUE).next_to(vec_a_end, UR, buff=0.1)
        
        # Vector B: Dynamic, controlled by the angle_tracker
        vector_B = Line(origin, origin + RIGHT, color=GREEN).add_tip()
        def update_vector_b(mob):
            theta = angle_tracker.get_x()
            new_end = origin + np.array([2.5 * np.cos(theta), 2.5 * np.sin(theta), 0])
            mob.become(Line(origin, new_end, color=GREEN).add_tip())
        vector_B.add_updater(update_vector_b)
        
        label_B = MathTex(r"\mathbf{B}", color=GREEN)
        def update_label_b(mob):
            end = vector_B.get_end()
            diff = end - origin
            norm = np.sqrt(np.sum(diff**2))
            if norm > 0:
                direction = diff / norm
            else:
                direction = RIGHT
            mob.move_to(end + direction * 0.3)
        label_B.add_updater(update_label_b)
        
        # 4. Angle Arc (Avoiding 'Angle' as it was flagged)
        angle_arc = Arc(radius=0.7, arc_center=origin, color=YELLOW)
        def update_arc(mob):
            theta_b = angle_tracker.get_x()
            start = min(angle_a, theta_b)
            diff = abs(angle_a - theta_b)
            mob.become(Arc(radius=0.7, start_angle=start, angle=diff, arc_center=origin, color=YELLOW))
        angle_arc.add_updater(update_arc)
        
        # 5. Formula and Similarity Matrix
        # Displaying the mathematical context
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=32
        ).to_edge(UP, buff=0.5)
        
        # Matrix grid (5x5)
        matrix = VGroup(*[
            Square(side_length=0.4, fill_opacity=0.1, stroke_width=1).set_stroke(GRAY, 1)
            for _ in range(25)
        ]).arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=1)
        
        matrix_title = Text("Similarity Matrix", font_size=20).next_to(matrix, UP)
        
        # Similarity Value Text
        sim_val_text = MathTex("1.00", color=YELLOW).next_to(matrix, DOWN, buff=0.5)
        def update_sim_val(mob):
            theta_b = angle_tracker.get_x()
            val = np.cos(theta_b - angle_a)
            mob.become(MathTex(f"{val:.2f}", color=YELLOW).next_to(matrix, DOWN, buff=0.5))
        sim_val_text.add_updater(update_sim_val)

        # 6. Linking alignment to matrix brightness
        # We'll use the center cell (index 12) to represent the current A/B pair
        matrix[12].set_fill(YELLOW, opacity=0)
        def update_cell_glow(mob):
            theta_b = angle_tracker.get_x()
            val = np.cos(theta_b - angle_a)
            # Map similarity (-1 to 1) to opacity (0 to 1)
            alpha = max(0, min(1, val))
            mob.set_fill(YELLOW, opacity=alpha)
        matrix[12].add_updater(update_cell_glow)
        
        # Adding components to scene
        self.add(x_axis, y_axis, vector_A, label_A, vector_B, label_B, angle_arc, formula, matrix, matrix_title, sim_val_text)
        
        # 7. Animation Sequences
        # Start with vectors divergent
        self.play(angle_tracker.animate.set_x(2.5), run_time=0.5)
        self.wait(1)
        
        # Rotate Vector B to align with A
        # Voice: "If two vectors point in exactly the same direction, the cosine is one..."
        self.play(
            angle_tracker.animate.set_x(angle_a),
            run_time=5,
            rate_func=lambda t: t # Manual linear rate function
        )
        self.wait(1)
        
        # Diversify other cells to show a completed heat map
        self.play(
            *[m.animate.set_fill(YELLOW, opacity=np.random.rand()*0.7) 
              for i, m in enumerate(matrix) if i != 12],
            run_time=2
        )
        
        self.wait(2)
