from manim import *
import numpy as np

class IterativeClusteringConcept(Scene):
    def construct(self):
        # Set seed for reproducible random layout
        np.random.seed(42)
        
        # 1. Generate a cloud of 27 Dot objects scattered randomly
        dot_positions = [
            np.array([np.random.uniform(-5.5, 5.5), np.random.uniform(-3, 2), 0])
            for _ in range(27)
        ]
        dots = VGroup(*[Dot(pos, color=GRAY_B) for pos in dot_positions])
        
        title = Text("Iterative Clustering Process", font_size=32).to_edge(UP)
        self.add(title)
        self.play(Create(dots))
        self.wait(1)

        available_indices = list(range(27))
        cluster_colors = [RED, GREEN, BLUE]
        
        # 2. Process: 3 clusters of 9 dots each
        for cluster_idx in range(3):
            cluster_color = cluster_colors[cluster_idx]
            
            # Step A: Find the two most similar (shortest distance) unassigned dots to seed
            min_dist = float('inf')
            seed_pair = (None, None)
            
            for i in range(len(available_indices)):
                for j in range(i + 1, len(available_indices)):
                    idx1 = available_indices[i]
                    idx2 = available_indices[j]
                    dist = np.linalg.norm(dots[idx1].get_center() - dots[idx2].get_center())
                    if dist < min_dist:
                        min_dist = dist
                        seed_pair = (idx1, idx2)
            
            s1, s2 = seed_pair
            current_cluster = [s1, s2]
            available_indices.remove(s1)
            available_indices.remove(s2)
            
            # Highlight the seed pair
            highlight1 = Circle(radius=0.2, color=YELLOW, stroke_width=2).move_to(dots[s1].get_center())
            highlight2 = Circle(radius=0.2, color=YELLOW, stroke_width=2).move_to(dots[s2].get_center())
            
            self.play(
                dots[s1].animate.set_color(cluster_color),
                dots[s2].animate.set_color(cluster_color),
                Create(highlight1), Create(highlight2),
                run_time=0.6
            )
            self.play(FadeOut(highlight1), FadeOut(highlight2), run_time=0.2)

            # Step B: Greedily pull in the next most similar unassigned proposals (7 more to reach 9)
            for _ in range(7):
                # We define cohesion by proximity to any member of the current cluster
                best_dist = float('inf')
                next_dot_idx = -1
                
                for candidate in available_indices:
                    for member in current_cluster:
                        d = np.linalg.norm(dots[candidate].get_center() - dots[member].get_center())
                        if d < best_dist:
                            best_dist = d
                            next_dot_idx = candidate
                
                # Animate search circle expanding from cluster centroid
                centroid = np.mean([dots[i].get_center() for i in current_cluster], axis=0)
                target_pos = dots[next_dot_idx].get_center()
                search_radius = np.linalg.norm(centroid - target_pos)
                
                search_circle = Circle(radius=0.1, color=WHITE, stroke_opacity=0.4).move_to(centroid)
                
                self.play(
                    search_circle.animate.set_width(search_radius * 2),
                    run_time=0.4,
                    rate_func=rate_functions.linear
                )
                self.play(
                    dots[next_dot_idx].animate.set_color(cluster_color),
                    FadeOut(search_circle),
                    run_time=0.3
                )
                
                current_cluster.append(next_dot_idx)
                available_indices.remove(next_dot_idx)
            
            self.wait(0.5)

        # Final layout check
        final_msg = Text("High Thematic Cohesion Achieved", font_size=24, color=YELLOW).to_edge(DOWN)
        self.play(Write(final_msg))
        self.wait(2)
