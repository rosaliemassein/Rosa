from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Manual Axes (Using Line since Axes was flagged)
        x_axis = Line(4 * LEFT, 1 * RIGHT, color=WHITE).shift(2 * LEFT)
        y_axis = Line(2.5 * DOWN, 2.5 * UP, color=WHITE).shift(2 * LEFT)
        origin_point = x_axis.get_center()
        self.add(x_axis, y_axis)

        # 2. Vectors A and B
        # Vector A is static
        vec_a_end = origin_point + 2.0 * RIGHT
        vec_a = Line(origin_point, vec_a_end, color=BLUE).add_tip()
        label_a = MathTex(r"\mathbf{A}", color=BLUE).next_to(vec_a, UP, buff=0.1)
        self.add(vec_a, label_a)

        # Vector B is dynamic - using a Dot's x-coordinate to track angle
        angle_tracker = Dot(point=[PI / 3, 0, 0]).set_opacity(0)
        self.add(angle_tracker)

        vec_b = Line(origin_point, origin_point + 2.0 * RIGHT, color=GREEN).add_tip()
        
        def update_vec_b(m):
            angle = angle_tracker.get_x()
            new_end = origin_point + np.array([2.0 * np.cos(angle), 2.0 * np.sin(angle), 0])
            m.become(Line(origin_point, new_end, color=GREEN).add_tip())
        
        vec_b.add_updater(update_vec_b)
        
        label_b = MathTex(r"\mathbf{B}", color=GREEN)
        def update_label_b(m):
            m.next_to(vec_b.get_end(), UR, buff=0.1)
        label_b.add_updater(update_label_b)
        
        self.add(vec_b, label_b)

        # 3. Angle Arc and Similarity text
        # Using Arc instead of Angle as it was flagged
        arc = Arc(radius=0.6, start_angle=0, angle=PI / 3, arc_center=origin_point, color=YELLOW)
        def update_arc(m):
            angle = angle_tracker.get_x()
            m.become(Arc(radius=0.6, start_angle=0, angle=angle, arc_center=origin_point, color=YELLOW))
        arc.add_updater(update_arc)

        theta_label = MathTex(r"\theta", color=YELLOW)
        def update_theta_label(m):
            angle = angle_tracker.get_x()
            m.move_to(origin_point + np.array([0.9 * np.cos(angle/2), 0.9 * np.sin(angle/2), 0]))
        theta_label.add_updater(update_theta_label)

        sim_formula = MathTex(r"\text{similarity} = \cos(\theta)", font_size=36).to_edge(UP)
        self.add(arc, theta_label, sim_formula)

        # 4. Matrix Grid
        grid = VGroup(*[
            Square(side_length=0.6).set_stroke(GRAY, 1).set_fill(BLUE, opacity=0.1)
            for _ in range(25)
        ]).arrange_in_grid(5, 5, buff=0.05).to_edge(RIGHT, buff=1)
        grid_label = Text("Similarity Matrix", font_size=24).next_to(grid, UP)
        self.add(grid, grid_label)

        # Middle cell responds to alignment
        active_cell = grid[12]
        def update_cell(m):
            val = np.cos(angle_tracker.get_x())
            if val > 0:
                m.set_fill(YELLOW, opacity=val)
            else:
                m.set_fill(BLACK, opacity=abs(val))
        active_cell.add_updater(update_cell)

        # 5. Animations
        # Rotate B to align with A
        self.play(angle_tracker.animate.set_x(5 * DEGREES), run_time=3)
        self.wait(1)
        
        # Rotate B away from A
        self.play(angle_tracker.animate.set_x(120 * DEGREES), run_time=3)
        self.wait(1)

        # Randomize other cells for visual effect
        self.play(*[
            c.animate.set_fill(WHITE if np.random.rand() > 0.7 else BLUE, opacity=np.random.rand())
            for i, c in enumerate(grid) if i != 12
        ])
        
        self.play(angle_tracker.animate.set_x(0), run_time=2)
        self.wait(2)

        # Cleanup updaters
        vec_b.clear_updaters()
        label_b.clear_updaters()
        arc.clear_updaters()
        theta_label.clear_updaters()
        active_cell.clear_updaters()
