from manim import *

class SemanticEmbedding(Scene):
    def construct(self):
        # Using Axes as a 2D alternative to the disallowed ThreeDAxes or NumberPlane
        axes = Axes(
            x_range=[-6, 6, 1],
            y_range=[-4, 4, 1],
            axis_config={"color": WHITE}
        )
        self.play(Create(axes))

        # Floating labels for conceptual keywords
        # Using Text because MathTex is disallowed in this environment
        dl_label = Text("Deep Learning").scale(0.5).move_to([3, 2, 0])
        nn_label = Text("Neural Networks").scale(0.5).move_to([3, -1, 0])
        rb_label = Text("Robotics").scale(0.5).move_to([-3, 0, 0])

        self.play(Write(dl_label), Write(nn_label), Write(rb_label))
        self.wait(1)

        # Vector points represented as Dot objects
        dl_dot = Dot([3, 2, 0], color=BLUE)
        nn_dot = Dot([3, -1, 0], color=BLUE)
        rb_dot = Dot([-3, 0, 0], color=RED)

        # Animate the transformation from text labels to coordinates
        self.play(
            Transform(dl_label, dl_dot),
            Transform(nn_label, nn_dot),
            Transform(rb_label, rb_dot)
        )
        self.wait(1)

        # Connection line to show proximity (Line used as DashedLine is restricted/undefined)
        connection = Line(dl_dot.get_center(), nn_dot.get_center(), color=YELLOW)
        self.play(Create(connection))
        self.wait(1)

        # Display the mathematical formula using standard Text
        formula = Text("v_phrase = sum(v_i) / n").scale(0.6)
        formula.to_edge([0, 1, 0]) # Equivalent to UP
        self.play(Write(formula))

        # Vectors for 'Machine' and 'Learning' to demonstrate averaging
        m_dot = Dot([-1, 2, 0], color=GREEN)
        l_dot = Dot([1, -1, 0], color=GREEN)
        m_text = Text("Machine").scale(0.4).next_to(m_dot, [0, 1, 0])
        l_text = Text("Learning").scale(0.4).next_to(l_dot, [0, 1, 0])

        self.play(Create(m_dot), Write(m_text))
        self.play(Create(l_dot), Write(l_text))
        self.wait(1)

        # Calculate the average position: ((-1+1)/2, (2-1)/2) = (0, 0.5)
        res_dot = Dot([0, 0.5, 0], color=WHITE)
        res_text = Text("Machine Learning").scale(0.5).next_to(res_dot, [0, 1, 0])

        # Merge the two vectors and their labels into the central average position
        self.play(
            ReplacementTransform(VGroup(m_dot, l_dot), res_dot),
            ReplacementTransform(VGroup(m_text, l_text), res_text),
            FadeOut(connection),
            FadeOut(dl_label),
            FadeOut(nn_label),
            FadeOut(rb_label)
        )
        self.wait(2)
