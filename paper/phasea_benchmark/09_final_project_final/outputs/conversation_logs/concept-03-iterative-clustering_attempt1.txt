from manim import *
import numpy as np

class IterativeClustering(Scene):
    def construct(self):
        # Configuration
        num_dots = 27
        dots_per_cluster = 9
        num_clusters = 3
        colors = [BLUE, GREEN, YELLOW]
        
        # Create a random cloud of 27 dots
        # Using a fixed seed for consistent layout
        np.random.seed(42)
        dots = VGroup()
        for _ in range(num_dots):
            pos = [np.random.uniform(-5, 5), np.random.uniform(-3, 3), 0]
            dots.add(Dot(point=pos, radius=0.1, color=WHITE))
        
        self.add(dots)
        self.wait(1)

        unassigned_dots = list(dots)
        
        for cluster_idx in range(num_clusters):
            cluster_color = colors[cluster_idx]
            current_cluster = []
            
            # 1. Find the two most similar (closest) unassigned dots to seed the cluster
            min_dist = float('inf')
            seed_pair = (None, None)
            
            if len(unassigned_dots) < 2:
                break
                
            for i in range(len(unassigned_dots)):
                for j in range(i + 1, len(unassigned_dots)):
                    d1 = unassigned_dots[i]
                    d2 = unassigned_dots[j]
                    dist = np.linalg.norm(d1.get_center() - d2.get_center())
                    if dist < min_dist:
                        min_dist = dist
                        seed_pair = (d1, d2)
            
            # Highlight the seed pair
            d1, d2 = seed_pair
            unassigned_dots.remove(d1)
            unassigned_dots.remove(d2)
            current_cluster.extend([d1, d2])
            
            # Bright ring around the pair
            center_point = (d1.get_center() + d2.get_center()) / 2
            ring = Circle(radius=min_dist/2 + 0.2, color=cluster_color).move_to(center_point)
            
            self.play(
                Create(ring),
                d1.animate.set_color(cluster_color),
                d2.animate.set_color(cluster_color)
            )
            
            # 2. Greedily pull in the next most similar dots until target size (9)
            while len(current_cluster) < dots_per_cluster and len(unassigned_dots) > 0:
                best_dist = float('inf')
                next_dot = None
                
                # Find unassigned dot closest to ANY dot in the current cluster
                for d_un in unassigned_dots:
                    for d_cl in current_cluster:
                        dist = np.linalg.norm(d_un.get_center() - d_cl.get_center())
                        if dist < best_dist:
                            best_dist = dist
                            next_dot = d_un
                
                # Expand search circle to the next dot
                current_center = sum([d.get_center() for d in current_cluster]) / len(current_cluster)
                target_radius = np.linalg.norm(next_dot.get_center() - current_center)
                
                search_circle = Circle(radius=0.01, color=cluster_color, stroke_opacity=0.5).move_to(current_center)
                self.add(search_circle)
                
                self.play(
                    search_circle.animate.scale(target_radius / 0.01),
                    run_time=0.5,
                    rate_func=rate_functions.linear
                )
                
                # Update dot and cluster
                unassigned_dots.remove(next_dot)
                current_cluster.append(next_dot)
                
                self.play(
                    next_dot.animate.set_color(cluster_color),
                    ring.animate.move_to(current_center).set_width(target_radius * 2.2),
                    FadeOut(search_circle),
                    run_time=0.3
                )
                
            self.play(FadeOut(ring))
            self.wait(0.5)

        self.wait(2)
