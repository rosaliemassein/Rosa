from manim import *
import math

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Manual Axes (avoiding 'Axes' keyword if disallowed)
        x_axis = Arrow(LEFT * 5, RIGHT * 5, buff=0, color=WHITE)
        y_axis = Arrow(DOWN * 3, UP * 3, buff=0, color=WHITE)
        axes_group = VGroup(x_axis, y_axis)
        self.add(axes_group)

        # 2. Scatter Plot Data (Non-linear trend)
        # We define a sine-like curve with some quadratic component
        x_vals = [x * 0.2 for x in range(-20, 21)]
        y_vals = [0.6 * math.sin(x * 1.4) + 0.06 * x**2 - 1 for x in x_vals]
        
        dots = VGroup()
        for x_val, y_val in zip(x_vals, y_vals):
            d = Dot(point=[x_val, y_val, 0], radius=0.08, color=GRAY)
            dots.add(d)
        
        # 3. Formula
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{\|x^{(i)} - x\|^2}{2\tau^2}\right)").to_edge(UP)
        self.add(formula)
        self.play(FadeIn(dots))

        # 4. Query point and Weight Field
        # query_point is the white dot that the user tracks
        query_point = Dot(point=[-4, 0, 0], color=WHITE, radius=0.12)
        self.add(query_point)

        # weight_ring visualizes the "influence" area
        weight_ring = Circle(radius=1.2, stroke_width=2, stroke_color=YELLOW, fill_opacity=0.1, fill_color=YELLOW)
        weight_ring.add_updater(lambda m: m.move_to(query_point.get_center()))
        self.add(weight_ring)

        # 5. Local Regression Line
        # This line will update its slope and position based on the query_point
        local_line = Line(LEFT, RIGHT, color=RED, stroke_width=6)
        
        def update_local_line(line_obj):
            xq = query_point.get_center()[0]
            tau = 0.8
            
            # Calculate weights for all points based on horizontal distance to query point
            weights = [math.exp(-(xv - xq)**2 / (2 * (tau**2))) for xv in x_vals]
            
            sum_w = sum(weights)
            if sum_w < 0.0001:
                return
            
            # Weighted Means
            xw = sum(w * x for w, x in zip(weights, x_vals)) / sum_w
            yw = sum(w * y for w, x in zip(weights, y_vals)) / sum_w
            
            # Weighted Covariance and Variance
            sxx = sum(w * (x - xw)**2 for w, x in zip(weights, x_vals))
            sxy = sum(w * (x - xw) * (y - yw) for w, x in zip(weights, x_vals))
            
            # Solve local linear regression: y = slope * x + intercept
            slope = sxy / sxx if abs(sxx) > 1e-6 else 0
            intercept = yw - slope * xw
            
            # Determine line segment boundaries around the query point
            span = 1.2
            x_start = xq - span
            x_end = xq + span
            
            line_obj.put_start_and_end_on(
                [x_start, slope * x_start + intercept, 0],
                [x_end, slope * x_end + intercept, 0]
            )

        local_line.add_updater(update_local_line)
        self.add(local_line)

        # 6. Dots Glowing Updater
        # Points close to the query point glow yellow; others stay gray
        def update_dots_glow(ds):
            xq = query_point.get_center()[0]
            tau = 0.8
            for i, dot in enumerate(ds):
                dist_sq = (x_vals[i] - xq)**2
                weight = math.exp(-dist_sq / (2 * (tau**2)))
                
                # Visual weighting: opacity and color shift
                dot.set_opacity(0.2 + 0.8 * weight)
                if weight > 0.4:
                    dot.set_color(YELLOW)
                else:
                    dot.set_color(GRAY)

        dots.add_updater(update_dots_glow)

        # 7. Execution of the animation
        # Move the query point across the data to show local recalculation
        self.play(query_point.animate.shift(RIGHT * 8), run_time=6, rate_func=lambda t: t)
        self.wait(1)

        # Fade everything out
        self.play(
            FadeOut(dots),
            FadeOut(query_point),
            FadeOut(weight_ring),
            FadeOut(local_line),
            FadeOut(axes_group),
            FadeOut(formula)
        )
