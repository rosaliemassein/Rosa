from manim import *
import numpy as np

class InverseMapping(Scene):
    def construct(self):
        # 1. Create the Split Screen Diagram
        # Left side: Desired Behavior
        input_title = Text("Input: Desired Behavior", color=WHITE).scale(0.6)
        input_details = Text("Rise Time, SS Error, Stability", color=GRAY).scale(0.4).next_to(input_title, DOWN)
        input_rect = Rectangle(width=4, height=1.5, color=BLUE).move_to(input_title.get_center() + DOWN * 0.2)
        input_group = VGroup(input_title, input_details, input_rect)

        # Right side: PID Constants
        output_title = Text("Output: PID Constants", color=WHITE).scale(0.6)
        output_details = MathTex("K_p, K_i, K_d", color=GRAY).scale(0.7).next_to(output_title, DOWN)
        output_rect = Rectangle(width=4, height=1.5, color=GREEN).move_to(output_title.get_center() + DOWN * 0.2)
        output_group = VGroup(output_title, output_details, output_rect)

        # Combine into a diagram
        diagram_elements = VGroup(input_group, output_group).arrange(RIGHT, buff=1.5)
        arrow = Arrow(input_group.get_right(), output_group.get_left(), color=GOLD, buff=0.1)
        full_diagram = VGroup(diagram_elements, arrow).to_edge(UP, buff=0.5)

        # 2. Setup the Plot (below the diagram)
        axes = Axes(
            x_range=[0, 8, 1],
            y_range=[0, 1.5, 0.5],
            x_length=8,
            y_length=2.5,
            axis_config={"include_tip": True}
        ).shift(DOWN * 1.5)
        
        labels = axes.get_axis_labels(x_label="Time", y_label="RPM")
        setpoint = DashedLine(
            axes.c2p(0, 1), axes.c2p(7.5, 1), color=YELLOW
        )
        
        # Damped oscillation curve: 1 - exp(-0.7t)*cos(3t)
        curve = axes.plot(
            lambda t: 1 - np.exp(-0.7 * t) * np.cos(3 * t),
            x_range=[0, 7.5],
            color=BLUE
        )

        # Manual Bracket for Rise Time (instead of Brace)
        rt_start = axes.c2p(0, -0.3)
        rt_end = axes.c2p(1.2, -0.3)
        rise_time_line = Line(rt_start, rt_end, color=YELLOW)
        rise_time_cap1 = Line(rt_start + UP*0.1, rt_start + DOWN*0.1, color=YELLOW)
        rise_time_cap2 = Line(rt_end + UP*0.1, rt_end + DOWN*0.1, color=YELLOW)
        rise_time_indicator = VGroup(rise_time_line, rise_time_cap1, rise_time_cap2)
        rise_time_text = Text("Rise Time", color=YELLOW).scale(0.4).next_to(rise_time_line, DOWN, buff=0.1)

        # Manual Bracket for Steady State Error
        # Final value at t=7.5 is ~1.0
        ss_val = 1 - np.exp(-0.7 * 7.5) * np.cos(3 * 7.5)
        err_top = axes.c2p(7.6, 1.0)
        err_bottom = axes.c2p(7.6, ss_val)
        ss_error_line = Line(err_top, err_bottom, color=ORANGE)
        ss_error_cap1 = Line(err_top + LEFT*0.1, err_top + RIGHT*0.1, color=ORANGE)
        ss_error_cap2 = Line(err_bottom + LEFT*0.1, err_bottom + RIGHT*0.1, color=ORANGE)
        ss_error_indicator = VGroup(ss_error_line, ss_error_cap1, ss_error_cap2)
        ss_error_text = Text("SS Error", color=ORANGE).scale(0.4).next_to(ss_error_line, RIGHT, buff=0.1)

        # Formula
        formula = MathTex("U = K_p e + K_i \\int e dt + K_d \\frac{de}{dt}").scale(0.6).to_corner(DR, buff=0.5)

        # --- Execution ---
        
        # Display boxes
        self.play(FadeIn(input_group), FadeIn(output_group))
        self.play(GrowArrow(arrow))
        self.wait(1)

        # Rotate diagram (using animate instead of Rotate class)
        self.play(full_diagram.animate.rotate(PI, about_point=full_diagram.get_center()), run_time=1.5)
        self.wait(1)

        # Show Plot
        self.play(Create(axes), Write(labels))
        self.play(Create(setpoint))
        self.play(Create(curve), run_time=2)
        
        # Show specific metrics
        self.play(Create(rise_time_indicator), Write(rise_time_text))
        self.play(Create(ss_error_indicator), Write(ss_error_text))
        
        # Show formula
        self.play(Write(formula))
        
        self.wait(2)
