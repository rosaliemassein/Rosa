from manim import *
import numpy as np

class Concept02QuantifyingStability(Scene):
    def construct(self):
        # 1. Manual Plot Setup (Avoiding disallowed Axes)
        # Coordinate mapping: Time 0-10 -> 6 units, Speed 20-40 RPM -> 4 units
        origin_point = LEFT * 4 + DOWN * 1.5
        x_axis = Line(origin_point, origin_point + RIGHT * 6)
        y_axis = Line(origin_point, origin_point + UP * 4)
        
        x_label = Text("t (s)", font_size=20).next_to(x_axis, RIGHT)
        y_label = Text("v (RPM)", font_size=20).next_to(y_axis, UP)
        
        # 30 RPM tick mark
        # (30 RPM - 20 RPM baseline) * (4 units / 20 RPM range) = 2 units up
        mid_y_point = origin_point + UP * 2
        tick_30 = Line(mid_y_point + LEFT * 0.1, mid_y_point + RIGHT * 0.1)
        label_30 = Text("30", font_size=15).next_to(tick_30, LEFT)

        # 2. Fluctuating Velocity Curve
        # Generate points for the motor fluctuating around 30 RPM
        curve_points = []
        for x_val in np.linspace(0, 10, 100):
            # Qualitative fluctuations around 30
            speed = 30 + 1.5 * np.sin(2 * x_val) + 0.7 * np.cos(5.3 * x_val) * np.exp(-x_val/10)
            # Map speed to scene coordinates
            px = origin_point[0] + (x_val * 0.6)
            py = origin_point[1] + (speed - 20) * 0.2
            curve_points.append([px, py, 0])
            
        curve = VMobject(color=YELLOW)
        curve.set_points_as_corners(curve_points)

        # 3. Highlight Last 20% (Time 8 to 10)
        # x position starts at origin + 8 * 0.6 = origin + 4.8
        highlight_rect = Rectangle(
            width=1.2, # 2 units * 0.6
            height=4, 
            fill_color=BLUE, 
            fill_opacity=0.2, 
            stroke_width=0
        )
        highlight_rect.move_to(origin_point + RIGHT * (9 * 0.6) + UP * 2)
        sse_label = Text("Steady State Region", font_size=14, color=BLUE).next_to(highlight_rect, UP, buff=0.1)

        # 4. Histogram on the Right
        hist_group = VGroup()
        # Bar heights: mode at 30 RPM (index 2) is tallest
        bar_heights = [0.4, 0.9, 3.2, 1.1, 0.5]
        colors = [BLUE_C, BLUE_C, GOLD, BLUE_C, BLUE_C]
        for i, h in enumerate(bar_heights):
            bar = Rectangle(
                width=0.4, 
                height=h, 
                fill_color=colors[i], 
                fill_opacity=0.7,
                stroke_width=1
            )
            # Position bars to the right of the plot
            bar.move_to(RIGHT * 3.5 + DOWN * 1.5 + RIGHT * i * 0.5, aligned_edge=DOWN)
            hist_group.add(bar)
            
        hist_title = Text("Speed Distribution", font_size=22).next_to(hist_group, UP, buff=0.4)
        mode_text = Text("Mode (30 RPM)", font_size=14, color=GOLD).next_to(hist_group[2], DOWN)

        # 5. Stability Formula
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=32
        ).to_edge(UP, buff=0.5)

        # 6. Narration (Voice text as caption)
        narration = Text(
            "Stability is the ratio of how often the motor stays at its most common speed\n"
            "versus deviations. Steady State Error uses the final 20% of the run.",
            font_size=18,
            line_spacing=1.5
        ).to_edge(DOWN, buff=0.3)

        # --- Execution ---
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(tick_30), Write(label_30))
        self.play(Create(curve), run_time=3)
        self.wait(0.5)
        
        self.play(FadeIn(highlight_rect), Write(sse_label))
        self.wait(1)
        
        self.play(Write(hist_title))
        self.play(LaggedStart(*[FadeIn(b, shift=UP) for b in hist_group], lag_ratio=0.1))
        self.play(Write(mode_text))
        self.wait(1)
        
        self.play(Write(formula))
        self.wait(1)
        
        self.play(Write(narration))
        self.wait(4)
