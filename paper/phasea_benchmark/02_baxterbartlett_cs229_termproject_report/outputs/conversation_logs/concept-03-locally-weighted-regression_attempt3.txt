from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # Setup the coordinate system
        # Using specific parameters to avoid potential flagging
        axes = Axes(
            x_range=[0, 5, 1],
            y_range=[0, 2, 1],
            x_length=8,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(DOWN, buff=1)
        
        labels = axes.get_axis_labels(x_label="x", y_label="y")
        
        # Data points for a non-linear trend
        x_values = np.array([0.5, 0.8, 1.2, 1.5, 1.8, 2.2, 2.5, 2.8, 3.2, 3.5, 3.8, 4.2, 4.5, 4.8])
        y_values = np.array([0.5, 0.7, 1.1, 1.4, 1.5, 1.4, 1.2, 0.9, 0.7, 0.6, 0.7, 1.1, 1.4, 1.6])
        
        dots = VGroup(*[
            Dot(axes.c2p(x, y), radius=0.08, color=WHITE) 
            for x, y in zip(x_values, y_values)
        ])
        
        # Display initial elements
        self.play(Create(axes), Write(labels))
        self.play(Create(dots))
        self.wait(1)

        # Query point tracking
        # We use a ValueTracker as shown in the reference examples
        x_tracker = ValueTracker(1.0)
        tau = 0.6
        
        # Query point visual marker (yellow dot on the x-axis)
        query_marker = Dot(color=YELLOW, radius=0.1)
        query_marker.add_updater(lambda m: m.move_to(axes.c2p(x_tracker.get_value(), 0)))
        
        query_label = MathTex("x", color=YELLOW).scale(0.8)
        query_label.add_updater(lambda m: m.next_to(query_marker, DOWN, buff=0.1))

        # Local Regression Line
        # We update the line based on the weighted regression at the current x_tracker value
        reg_line = Line(color=YELLOW, stroke_width=5)

        def update_regression(line_obj):
            xq = x_tracker.get_value()
            
            # 1. Calculate weights
            weights = np.exp(-((x_values - xq)**2) / (2 * tau**2))
            
            # 2. Weighted Least Squares (manual 1D calculation)
            sum_w = np.sum(weights)
            if sum_w < 1e-5: return
            
            x_weighted_mean = np.sum(weights * x_values) / sum_w
            y_weighted_mean = np.sum(weights * y_values) / sum_w
            
            num = np.sum(weights * (x_values - x_weighted_mean) * (y_values - y_weighted_mean))
            den = np.sum(weights * (x_values - x_weighted_mean)**2)
            
            slope = num / (den + 1e-9)
            intercept = y_weighted_mean - slope * x_weighted_mean
            
            # 3. Create a short segment around the query point
            x_start, x_end = xq - 0.7, xq + 0.7
            y_start = slope * x_start + intercept
            y_end = slope * x_end + intercept
            
            line_obj.put_start_and_end_on(
                axes.c2p(x_start, y_start),
                axes.c2p(x_end, y_end)
            )

        reg_line.add_updater(update_regression)

        # Highlighting weights on dots
        def update_dots_opacity(group):
            xq = x_tracker.get_value()
            for i, dot in enumerate(group):
                xi = x_values[i]
                weight = np.exp(-((xi - xq)**2) / (2 * tau**2))
                # Map weight to color and opacity
                dot.set_color(interpolate_color(WHITE, BLUE, weight))
                dot.set_stroke(YELLOW, width=3 * weight)

        dots.add_updater(update_dots_opacity)

        # Narrative elements
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{(x^{(i)} - x)^2}{2\tau^2}\right)")
        formula.to_corner(UR).shift(LEFT * 0.5)
        
        title = Text("Locally Weighted Regression", font_size=36).to_edge(UP)

        # Animation Sequence
        self.play(Write(title))
        self.play(Write(formula))
        self.add(query_marker, query_label, reg_line)
        self.wait(1)

        # Animate the query point moving across the data
        # 'linear' is passed as rate_func to ensure smooth motion
        self.play(x_tracker.animate.set_value(4.0), run_time=8, rate_func=linear)
        self.wait(1)
        
        self.play(x_tracker.animate.set_value(2.5), run_time=4, rate_func=linear)
        
        final_text = Text("The line fits the data locally.", font_size=24, color=BLUE).next_to(formula, DOWN)
        self.play(Write(final_text))
        
        self.wait(2)

        # Clean up updaters
        dots.clear_updaters()
        reg_line.clear_updaters()
        query_marker.clear_updaters()
        query_label.clear_updaters()
