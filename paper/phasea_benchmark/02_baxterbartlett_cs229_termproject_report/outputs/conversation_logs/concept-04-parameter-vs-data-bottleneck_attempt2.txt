from manim import *
import numpy as np

class Concept04ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Left Side Setup (Neural Network)
        nn_center = LEFT * 3.5 + UP * 1.5
        nn_xaxis = Line(nn_center + LEFT * 2.2, nn_center + RIGHT * 2.2, color=GRAY)
        nn_yaxis = Line(nn_center + DOWN * 1.5, nn_center + UP * 1.5, color=GRAY)
        nn_title = Text("Neural Network", color=RED).scale(0.6).next_to(nn_yaxis, UP, buff=0.3)
        nn_subtitle = Text("High Variance / Overfitting", color=RED).scale(0.4).next_to(nn_xaxis, DOWN, buff=0.2)
        
        # 2. Right Side Setup (LWR)
        lwr_center = RIGHT * 3.5 + UP * 1.5
        lwr_xaxis = Line(lwr_center + LEFT * 2.2, lwr_center + RIGHT * 2.2, color=GRAY)
        lwr_yaxis = Line(lwr_center + DOWN * 1.5, lwr_center + UP * 1.5, color=GRAY)
        lwr_title = Text("LWR", color=GREEN).scale(0.6).next_to(lwr_yaxis, UP, buff=0.3)
        lwr_subtitle = Text("Robust / Generalizing", color=GREEN).scale(0.4).next_to(lwr_xaxis, DOWN, buff=0.2)

        # 3. Generate Data Points (27 points)
        np.random.seed(42)
        x_vals = np.linspace(-1.8, 1.8, 27)
        # Linear trend: y = 0.5x with some noise
        y_noise = np.random.normal(0, 0.3, 27)
        y_vals = 0.5 * x_vals + y_noise
        
        nn_dots = VGroup(*[
            Dot(nn_center + x * RIGHT + y * UP, radius=0.04, color=WHITE) 
            for x, y in zip(x_vals, y_vals)
        ])
        
        lwr_dots = VGroup(*[
            Dot(lwr_center + x * RIGHT + y * UP, radius=0.04, color=WHITE) 
            for x, y in zip(x_vals, y_vals)
        ])

        # 4. Create Curves
        # NN: Highly wiggly curve hitting points
        nn_curve_points = []
        for x_fine in np.linspace(-1.8, 1.8, 100):
            # A high-frequency oscillation to simulate overfitting
            y_wiggle = 0.5 * x_fine + 0.6 * np.sin(12 * x_fine) * np.cos(5 * x_fine)
            nn_curve_points.append(nn_center + x_fine * RIGHT + y_wiggle * UP)
        
        nn_curve = VMobject(color=RED).set_points_as_corners(nn_curve_points)
        
        # LWR: Smooth local linear path
        lwr_curve = Line(
            lwr_center + (-1.8) * RIGHT + (-0.9) * UP,
            lwr_center + 1.8 * RIGHT + 0.9 * UP,
            color=GREEN
        )

        # 5. MSE Bar Chart (Manual Construction)
        chart_base = DOWN * 2.5
        chart_label = Text("Mean Squared Error (MSE)").scale(0.5).move_to(chart_base + UP * 2)
        
        # Bars
        nn_bar_height = 1.21 * 1.2
        lwr_bar_height = 0.85 * 1.2
        
        nn_bar = Rectangle(height=nn_bar_height, width=1.0, fill_opacity=0.8, color=RED, stroke_width=2)
        nn_bar.move_to(chart_base + LEFT * 1.5 + UP * (nn_bar_height / 2))
        
        lwr_bar = Rectangle(height=lwr_bar_height, width=1.0, fill_opacity=0.8, color=GREEN, stroke_width=2)
        lwr_bar.move_to(chart_base + RIGHT * 1.5 + UP * (lwr_bar_height / 2))
        
        nn_bar_val = Text("1.21", color=RED).scale(0.4).next_to(nn_bar, UP, buff=0.1)
        lwr_bar_val = Text("0.85", color=GREEN).scale(0.4).next_to(lwr_bar, UP, buff=0.1)
        
        nn_bar_label = Text("Neural Network").scale(0.4).next_to(nn_bar, DOWN, buff=0.2)
        lwr_bar_label = Text("LWR").scale(0.4).next_to(lwr_bar, DOWN, buff=0.2)

        # 6. Formula
        formula = MathTex(
            r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}"
        ).scale(0.7).to_corner(DR, buff=0.8)

        # 7. Narrative Text
        narrative = Text(
            "27 points vs 24 parameters: The 'overfitting' bottleneck.",
            line_spacing=0.6
        ).scale(0.4).to_corner(DL, buff=0.5)

        # 8. Execution
        self.play(
            Create(nn_xaxis), Create(nn_yaxis), Write(nn_title), Write(nn_subtitle),
            Create(lwr_xaxis), Create(lwr_yaxis), Write(lwr_title), Write(lwr_subtitle)
        )
        self.play(Create(nn_dots), Create(lwr_dots))
        self.wait(0.5)
        
        self.play(Create(nn_curve), run_time=2)
        self.play(Create(lwr_curve), run_time=2)
        self.wait(1)
        
        self.play(
            Create(nn_bar), Create(lwr_bar),
            Write(nn_bar_val), Write(lwr_bar_val),
            Write(nn_bar_label), Write(lwr_bar_label),
            Write(chart_label)
        )
        self.play(Write(formula))
        self.play(Write(narrative))
        
        self.wait(3)
