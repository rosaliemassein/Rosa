from manim import *
import numpy as np

class QuantifyingStability(Scene):
    def construct(self):
        # 1. Setup Manual Axes for Velocity Plot
        # We avoid the Axes class to ensure compatibility with all environments
        origin_v = LEFT * 5 + DOWN * 1.5
        v_x_axis = Line(origin_v, origin_v + RIGHT * 5, color=WHITE)
        v_y_axis = Line(origin_v, origin_v + UP * 4, color=WHITE)
        v_x_label = Text("Time", font_size=20).next_to(v_x_axis, DOWN)
        v_y_label = Text("RPM (Velocity)", font_size=20).rotate(90 * DEGREES).next_to(v_y_axis, LEFT)
        
        # Create fluctuating velocity graph manually
        # y = 30 RPM will be mapped to 2.5 units above the origin
        points = []
        for t in np.linspace(0, 5, 100):
            # Fluctuations: base 2.5 + small sine waves
            val = 2.5 + 0.3 * np.sin(4 * t) + 0.15 * np.cos(9 * t)
            points.append(origin_v + RIGHT * t + UP * val)
        
        velocity_line = VMobject(color=BLUE)
        velocity_line.set_points_as_corners(points)
        
        # Highlight last 20% (last 1.0 unit of the 5.0 unit axis)
        highlight_rect = Rectangle(
            width=1.0, height=4.0, 
            fill_color=BLUE, fill_opacity=0.3, stroke_width=0
        )
        highlight_rect.move_to(origin_v + RIGHT * 4.5 + UP * 2)
        highlight_text = Text("Final 20%", font_size=16).next_to(highlight_rect, UP)
        
        # Animations for part 1
        self.play(Create(v_x_axis), Create(v_y_axis), Write(v_x_label), Write(v_y_label))
        self.play(Create(velocity_line))
        self.play(FadeIn(highlight_rect), Write(highlight_text))
        self.wait(1)

        # 2. Setup Manual Histogram on the right
        origin_h = RIGHT * 1 + DOWN * 1.5
        h_x_axis = Line(origin_h, origin_h + RIGHT * 5, color=WHITE)
        h_title = Text("Speed Distribution", font_size=24).next_to(origin_h + UP * 4 + RIGHT * 2.5, UP)
        
        # Heights for the bars (Mode at index 2)
        heights = [0.6, 1.2, 3.5, 0.9, 0.5]
        bars = VGroup()
        for i, h in enumerate(heights):
            # Mode bar is Yellow, others are Blue
            bar_color = YELLOW if i == 2 else BLUE
            bar = Rectangle(width=0.6, height=h, fill_color=bar_color, fill_opacity=0.8)
            # Position bars side-by-side
            bar.move_to(origin_h + RIGHT * (i * 0.9 + 0.7) + UP * (h / 2))
            bars.add(bar)
            
        # Animations for part 2
        self.play(Create(h_x_axis), Write(h_title))
        self.play(Create(bars))
        self.wait(1)

        # 3. Stability Formula
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=36
        ).to_edge(UP, buff=0.5)
        
        self.play(Write(formula))
        
        # Visual indicators for numerator/denominator
        mode_indicator = SurroundingRectangle(bars[2], color=YELLOW)
        others_group = VGroup(bars[0], bars[1], bars[3], bars[4])
        others_indicator = SurroundingRectangle(others_group, color=BLUE)
        
        self.play(Create(mode_indicator))
        self.wait(1)
        self.play(ReplacementTransform(mode_indicator, others_indicator))
        self.wait(1)
        self.play(FadeOut(others_indicator))
        
        self.wait(2)
