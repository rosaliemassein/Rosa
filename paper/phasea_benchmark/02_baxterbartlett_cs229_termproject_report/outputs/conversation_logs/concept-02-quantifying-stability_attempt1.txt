from manim import *
import numpy as np

class QuantifyingStability(Scene):
    def construct(self):
        # 1. Main Velocity Plot
        # Using specific names to avoid potential identifier conflicts
        velocity_axes = Axes(
            x_range=[0, 10, 2],
            y_range=[20, 40, 5],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": False}
        ).to_edge(LEFT, buff=0.5)
        
        time_label = Text("Time", font_size=24).next_to(velocity_axes.x_axis, DOWN)
        rpm_label = Text("Velocity (RPM)", font_size=24).rotate(90 * DEGREES).next_to(velocity_axes.y_axis, LEFT)
        
        # Create a fluctuating velocity curve around 30 RPM
        t_values = np.linspace(0, 10, 100)
        # Generate data points manually to avoid lambda-based identifier issues
        v_values = [30 + 1.2 * np.sin(val * 2) + 0.6 * np.cos(val * 5) for val in t_values]
        graph_points = [velocity_axes.c2p(tv, vv) for tv, vv in zip(t_values, v_values)]
        
        velocity_curve = VMobject(color=WHITE)
        velocity_curve.set_points_as_corners(graph_points)
        
        # 2. Highlight the last 20% of the x-axis (from 8 to 10)
        # Calculate dimensions for the highlight box
        unit_width = velocity_axes.x_axis.get_unit_size()
        highlight_box = Rectangle(
            width=unit_width * 2,
            height=velocity_axes.y_axis.get_length(),
            fill_color=BLUE,
            fill_opacity=0.3,
            stroke_width=0
        )
        # Position it over the last 20% (center is at t=9)
        highlight_box.move_to(velocity_axes.c2p(9, 30))
        highlight_text = Text("Final 20%", font_size=20).next_to(highlight_box, UP)

        # 3. Histogram (constructed using Axes and Rectangles for compatibility)
        hist_axes = Axes(
            x_range=[0, 4, 1],
            y_range=[0, 50, 10],
            x_length=4,
            y_length=3,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5).shift(UP * 0.5)
        
        hist_title = Text("Histogram", font_size=24).next_to(hist_axes, UP)
        
        # Data for bins (29, 30, 31 RPM)
        # 30 RPM (middle bin) is significantly taller
        counts = [10, 40, 12]
        bars = VGroup()
        for i, count_val in enumerate(counts):
            bar_height = hist_axes.get_y_unit_size() * count_val
            bar = Rectangle(
                width=0.7,
                height=bar_height,
                fill_color=WHITE,
                fill_opacity=0.7,
                stroke_width=1
            )
            # Align bottom of bar to the axis
            bar.move_to(hist_axes.c2p(i + 1, 0), aligned_edge=DOWN)
            bars.add(bar)
            
        # Highlight the mode bin
        bars[1].set_color(YELLOW)
        
        bin_labels = VGroup(
            Text("29", font_size=18).next_to(bars[0], DOWN, buff=0.1),
            Text("30", font_size=18).next_to(bars[1], DOWN, buff=0.1),
            Text("31", font_size=18).next_to(bars[2], DOWN, buff=0.1)
        )

        # 4. Stability Formula
        stability_formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=32
        ).to_edge(DOWN, buff=0.8)

        # 5. Animation Sequence
        self.play(Create(velocity_axes), Write(time_label), Write(rpm_label))
        self.play(Create(velocity_curve), run_time=2)
        self.play(FadeIn(highlight_box), Write(highlight_text))
        self.wait(0.5)
        
        self.play(
            Create(hist_axes), 
            Create(bars), 
            Write(hist_title), 
            FadeIn(bin_labels)
        )
        self.wait(0.5)
        
        self.play(Write(stability_formula))
        self.play(Indicate(bars[1]), Indicate(stability_formula))
        
        self.wait(2)
