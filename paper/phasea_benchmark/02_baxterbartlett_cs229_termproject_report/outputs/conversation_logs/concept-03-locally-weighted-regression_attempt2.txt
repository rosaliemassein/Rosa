from manim import *
import numpy as np

class Concept03LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Create Axes
        ax = Axes(
            x_range=[-5, 5, 1],
            y_range=[-4, 4, 1],
            x_length=10,
            y_length=6,
            axis_config={"include_tip": True, "color": BLUE_E}
        )
        
        # 2. Generate scatter plot data (non-linear trend)
        x_data = np.array([-4.5, -4.0, -3.5, -3.2, -2.8, -2.5, -2.0, -1.5, -1.2, -0.8, -0.4, 0.0, 0.4, 0.8, 1.2, 1.5, 2.0, 2.5, 2.8, 3.2, 3.5, 4.0, 4.5])
        y_data = 0.05 * x_data**3 - 0.2 * x_data + np.random.normal(0, 0.3, size=len(x_data))
        
        dots = VGroup(*[Dot(ax.c2p(x, y), radius=0.08, color=GRAY) for x, y in zip(x_data, y_data)])
        
        # 3. Query point tracking
        q_x = ValueTracker(-4.0)
        query_dot = Dot(color=WHITE).scale(1.2)
        
        # 4. Weight visualizer (gradient circle effect)
        weight_viz = VGroup(*[
            Circle(radius=r, stroke_width=0, fill_opacity=0.05).set_fill(BLUE)
            for r in np.arange(0.1, 2.0, 0.2)
        ])
        
        # 5. Local regression line
        reg_line = Line(LEFT, RIGHT, color=YELLOW, stroke_width=6)
        
        # Hyperparameter tau
        tau = 0.8
        
        # 6. Updater for dynamic calculation
        def update_lwr_logic(mob):
            curr_x = q_x.get_value()
            
            # Calculate Gaussian weights based on distance to query x
            weights = np.exp(-(x_data - curr_x)**2 / (2 * tau**2))
            
            # Weighted Linear Regression Calculation
            w_sum = np.sum(weights)
            if w_sum < 1e-5:
                return
            
            x_w_mean = np.sum(weights * x_data) / w_sum
            y_w_mean = np.sum(weights * y_data) / w_sum
            
            # Weighted slope formula
            num = np.sum(weights * (x_data - x_w_mean) * (y_data - y_w_mean))
            den = np.sum(weights * (x_data - x_w_mean)**2)
            slope = num / den if den > 1e-5 else 0
            
            # Calculate y position at query point using the local model
            curr_y = y_w_mean + slope * (curr_x - x_w_mean)
            
            # Update Query Dot and weight field positions
            query_dot.move_to(ax.c2p(curr_x, curr_y))
            weight_viz.move_to(ax.c2p(curr_x, curr_y))
            
            # Update dots coloring and glow based on weights
            for i, dot in enumerate(dots):
                w = weights[i]
                # Manual linear interpolation of color components to ensure compatibility
                dot.set_color(interpolate_color(GRAY, YELLOW, w))
                dot.set_opacity(interpolate(0.2, 1.0, w))
            
            # Update the local regression line orientation
            # Line length of 2.0 in scene units
            p1 = ax.c2p(curr_x - 1, curr_y - slope * 1)
            p2 = ax.c2p(curr_x + 1, curr_y + slope * 1)
            reg_line.put_start_and_end_on(p1, p2)

        # Attach updater
        reg_line.add_updater(update_lwr_logic)

        # 7. Formulas and Labels
        formula = MathTex(
            r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)",
            color=WHITE
        ).scale(0.7).to_corner(UL)
        
        voice_text = Text(
            "Giving more 'weight' to training examples\nthat look similar to the target.",
            font_size=22,
            line_spacing=0.8
        ).to_corner(UR)

        # 8. Animation Sequence
        self.add(ax, dots, weight_viz, reg_line, query_dot, formula, voice_text)
        
        # Start at -4 and move across the x-axis
        self.play(q_x.animate.set_value(4), run_time=10, rate_func=linear)
        self.wait(1)
        
        # Final fade out
        self.play(FadeOut(VGroup(ax, dots, weight_viz, reg_line, query_dot, formula, voice_text)))
