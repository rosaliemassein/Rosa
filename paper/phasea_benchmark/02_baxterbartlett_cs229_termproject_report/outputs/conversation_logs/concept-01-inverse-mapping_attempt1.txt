from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. UI Elements (Top Half)
        # We manually build the boxes and labels to avoid high-level classes that might be disallowed
        
        # Left Box: Desired Behavior
        input_rect = Rectangle(width=4.5, height=1.5, color=WHITE)
        input_title = Text("Desired Behavior", font_size=22).move_to(input_rect.get_center() + UP * 0.2)
        input_sub = Text("(Rise Time, SS Error, Stability)", font_size=14, color=GRAY).move_to(input_rect.get_center() + DOWN * 0.3)
        input_group = VGroup(input_rect, input_title, input_sub).to_edge(LEFT, buff=0.8).shift(UP * 2)

        # Right Box: PID Constants
        output_rect = Rectangle(width=4, height=1.5, color=WHITE)
        output_title = Text("PID Constants", font_size=22).move_to(output_rect.get_center())
        output_group = VGroup(output_rect, output_title).to_edge(RIGHT, buff=0.8).shift(UP * 2)

        # Large Arrow between them
        arrow = Arrow(input_group.get_right(), output_group.get_left(), color=BLUE, buff=0.2)
        arrow_label = Text("Inverse Mapping", font_size=16).next_to(arrow, UP, buff=0.1)
        
        diagram = VGroup(input_group, output_group, arrow, arrow_label)

        # Presentation
        self.play(Create(input_group), Create(output_group))
        self.play(GrowArrow(arrow), Write(arrow_label))
        self.wait(1)

        # 2. "Flips the script" - Rotate diagram 180 degrees
        # Using .animate.rotate instead of the Rotate class to avoid disallowed identifiers
        self.play(diagram.animate.rotate(PI))
        self.wait(1)

        # 3. Motor Velocity Plot (Bottom Half)
        # Manually constructing a coordinate system using Lines
        origin = DOWN * 2.5 + LEFT * 4
        x_axis = Line(origin, origin + RIGHT * 8, color=WHITE)
        y_axis = Line(origin, origin + UP * 3, color=WHITE)
        x_label = Text("Time", font_size=16).next_to(x_axis, RIGHT, buff=0.2)
        y_label = Text("RPM", font_size=16).next_to(y_axis, UP, buff=0.2)
        
        # Setpoint line (Dashed effect created via manual line if DashedLine is disallowed)
        setpoint_height = 1.8
        setpoint_line = Line(origin + UP * setpoint_height, origin + UP * setpoint_height + RIGHT * 8, color=YELLOW)
        setpoint_tag = Text("Target", font_size=14, color=YELLOW).next_to(setpoint_line, UP, buff=0.05)

        # 4. Oscillating Curve (Damped Response)
        # Generating points manually to avoid using 'axes.plot' and lambdas
        curve_pts = []
        for i in range(101):
            # x goes from 0 to 7.5
            x_val = (i / 100.0) * 7.5
            # y = target * (1 - e^(-ax) * cos(bx))
            # Parameters: target=1.8, damping=0.5, frequency=2.5
            y_val = 1.8 * (1 - np.exp(-0.5 * x_val) * np.cos(2.5 * x_val))
            curve_pts.append(origin + RIGHT * x_val + UP * y_val)
        
        graph_curve = VMobject(color=RED)
        graph_curve.set_points_as_corners(curve_pts)

        # 5. Performance Metrics (Labels)
        # Rise Time indicator (approx x=0.6)
        rise_line = Line(origin + RIGHT * 0.6 + UP * 0.1, origin + RIGHT * 0.6 + DOWN * 0.1)
        rise_text = Text("Rise Time", font_size=14).next_to(rise_line, DOWN, buff=0.1)
        
        # Steady State Error indicator (at the end)
        # The curve slightly undershoots the target at the end of our range
        error_line = Line(origin + RIGHT * 7.2 + UP * 1.8, origin + RIGHT * 7.2 + UP * 1.6, color=BLUE)
        error_text = Text("SS Error", font_size=14).next_to(error_line, RIGHT, buff=0.1)

        # Final Animation Sequence
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint_line), Write(setpoint_tag))
        self.play(Create(graph_curve), run_time=3)
        self.play(Create(rise_line), Write(rise_text))
        self.play(Create(error_line), Write(error_text))

        # Formula at the bottom
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=24, color=BLUE)
        formula.to_corner(DR, buff=0.5)
        self.play(Write(formula))
        
        self.wait(2)
