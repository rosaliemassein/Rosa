from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Top Section: Input/Output Diagram
        # Desired Behavior Box (Input)
        behavior_text = Text("Desired Behavior", font_size=28, color=BLUE)
        behavior_sub = Text("(Rise Time, SS Error, Stability)", font_size=18, color=BLUE).next_to(behavior_text, DOWN)
        input_content = VGroup(behavior_text, behavior_sub)
        input_rect = SurroundingRectangle(input_content, color=BLUE, buff=0.3)
        input_group = VGroup(input_content, input_rect)

        # PID Constants Box (Output)
        pid_title = Text("PID Constants", font_size=28, color=GREEN)
        pid_params = MathTex(r"K_p, K_i, K_d", color=GREEN).next_to(pid_title, DOWN)
        output_content = VGroup(pid_title, pid_params)
        output_rect = SurroundingRectangle(output_content, color=GREEN, buff=0.3)
        output_group = VGroup(output_content, output_rect)

        # Arrow and positioning
        arrow = Arrow(LEFT, RIGHT, color=YELLOW, stroke_width=6).scale(1.2)
        diagram = VGroup(input_group, arrow, output_group).arrange(RIGHT, buff=0.8).to_edge(UP, buff=0.5)

        self.play(FadeIn(diagram))
        self.wait(1)

        # "Flip the script" - 180 degree rotation of the whole diagram
        # Using animate.rotate to ensure compatibility
        self.play(diagram.animate.rotate(PI), run_time=2)
        self.wait(1)

        # 2. Bottom Section: Motor Velocity Plot
        axes = Axes(
            x_range=[0, 7, 1],
            y_range=[0, 1.5, 0.5],
            x_length=9,
            y_length=3,
            axis_config={"include_tip": True}
        ).shift(DOWN * 1.8)

        # Labels for axes
        x_axis_label = Text("Time", font_size=18).next_to(axes.x_axis, RIGHT, buff=0.2)
        y_axis_label = Text("RPM", font_size=18).next_to(axes.y_axis, UP, buff=0.2)
        
        # Curve: Damped oscillation toward setpoint 1.0
        # Formula: 1 - exp(-t) * cos(2.5t)
        # Using 't' as the parameter to avoid linter conflicts
        curve = axes.plot(lambda t: 1 - np.exp(-t) * np.cos(2.5 * t), color=RED)
        setpoint = DashedLine(
            start=axes.c2p(0, 1),
            end=axes.c2p(7, 1),
            color=YELLOW
        )
        setpoint_label = Text("Setpoint", font_size=16, color=YELLOW).next_to(setpoint, RIGHT, buff=0.1)

        self.play(Create(axes), Write(x_axis_label), Write(y_axis_label))
        self.play(Create(setpoint), Write(setpoint_label))
        self.play(Create(curve), run_time=3)

        # 3. Annotations
        # Rise Time: using Brace on a helper line on the x-axis
        # Approximate first time it hits 1.0 is around t=0.6
        rt_line = Line(axes.c2p(0, 0), axes.c2p(0.6, 0))
        rt_brace = Brace(rt_line, DOWN, color=WHITE)
        rt_label = rt_brace.get_text("Rise Time", font_size=20)

        # Steady State Error: distance between curve tail and setpoint
        # Using a Line with tips as a safer version of DoubleArrow
        sse_line = Line(axes.c2p(6.0, 1), axes.c2p(6.0, 0.93), color=WHITE).add_tip(tip_length=0.1).add_tip(at_start=True, tip_length=0.1)
        sse_text = Text("Steady State Error", font_size=18).next_to(sse_line, UP, buff=0.2)

        self.play(Create(rt_brace), Write(rt_label))
        self.play(Create(sse_line), Write(sse_text))

        # 4. Reference Formula
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=24)
        formula.to_corner(DL, buff=0.5)
        self.play(Write(formula))

        self.wait(2)
