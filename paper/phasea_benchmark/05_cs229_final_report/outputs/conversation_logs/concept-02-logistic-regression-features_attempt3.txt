from manim import *
import numpy as np

class Concept02LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Coordinate System for Sigmoid Curve
        # The prompt asks for a NumberPlane on the right
        plane = NumberPlane(
            x_range=[-6, 6, 1],
            y_range=[-0.1, 1.1, 0.5],
            x_length=6,
            y_length=4,
            background_line_style={"stroke_opacity": 0.2}
        ).shift(RIGHT * 2.5)
        
        # Sigmoid logic using a named function to avoid lambda-related identifier issues
        def calculate_sigmoid(z_input):
            return 1 / (1 + np.exp(-z_input))
            
        sigmoid_curve = plane.plot(calculate_sigmoid, color=GREEN)
        curve_label = MathTex("P(y=1)", color=GREEN).next_to(plane, UP, buff=0.2)
        
        # 2. Trackers for Interactive Features
        val_1day = ValueTracker(0)
        val_30day = ValueTracker(0)
        
        # 3. Sliding Bars (UI on the left)
        bar_w = 0.5
        bar_h = 3
        
        # 1-Day T-CSP Bar
        frame_1day = Rectangle(width=bar_w, height=bar_h, stroke_color=WHITE).shift(LEFT * 5 + DOWN * 0.5)
        label_1day = Text("1-Day T-CSP", font_size=16).next_to(frame_1day, DOWN)
        
        def update_fill_1day():
            curr_val = val_1day.get_value()
            # Map range [-3, 3] to bar height [0.01, bar_h]
            mapped_h = np.clip((curr_val + 3) / 6 * bar_h, 0.01, bar_h)
            return Rectangle(
                width=bar_w, height=mapped_h, 
                fill_color=RED, fill_opacity=0.7, stroke_width=0
            ).align_to(frame_1day, DOWN)
            
        fill_1day = always_redraw(update_fill_1day)
        
        # 30-Day T-CSP Bar
        frame_30day = Rectangle(width=bar_w, height=bar_h, stroke_color=WHITE).shift(LEFT * 3.5 + DOWN * 0.5)
        label_30day = Text("30-Day T-CSP", font_size=16).next_to(frame_30day, DOWN)
        
        def update_fill_30day():
            curr_val = val_30day.get_value()
            mapped_h = np.clip((curr_val + 3) / 6 * bar_h, 0.01, bar_h)
            return Rectangle(
                width=bar_w, height=mapped_h, 
                fill_color=BLUE, fill_opacity=0.7, stroke_width=0
            ).align_to(frame_30day, DOWN)
            
        fill_30day = always_redraw(update_fill_30day)
        
        # 4. Formulas and Mapping
        # Feature Vector X
        feature_vector = MathTex(
            r"\mathbf{X} = [1, X_{1d}, X_{30d}]",
            font_size=32
        ).to_edge(UP, buff=0.5).shift(LEFT * 3.5)
        
        # Logistic Formula
        logistic_formula = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=30
        ).next_to(feature_vector, DOWN, buff=0.7)
        
        # Moving Dot on the Curve
        dot_on_curve = Dot(color=YELLOW, radius=0.1)
        
        def move_dot_on_curve(mobject_to_update):
            # Sum features to determine position along sigmoid
            total_z = val_1day.get_value() + val_30day.get_value()
            current_prob = calculate_sigmoid(total_z)
            mobject_to_update.move_to(plane.c2p(total_z, current_prob))
            
        dot_on_curve.add_updater(move_dot_on_curve)
        
        # 5. Animation Sequence
        self.add(plane, sigmoid_curve, curve_label)
        self.add(frame_1day, label_1day, fill_1day, frame_30day, label_30day, fill_30day)
        
        self.play(Write(feature_vector))
        self.wait(0.5)
        # Using ReplacementTransform as a robust alternative for formula mapping
        self.play(ReplacementTransform(feature_vector.copy(), logistic_formula))
        self.add(dot_on_curve)
        self.wait(1)
        
        # Animate the bars sliding and the dot moving along the curve
        self.play(
            val_1day.animate.set_value(2),
            val_30day.animate.set_value(1.5),
            run_time=3,
            rate_func=rate_functions.linear
        )
        self.wait(0.5)
        
        self.play(
            val_1day.animate.set_value(-2.5),
            val_30day.animate.set_value(-1.5),
            run_time=4,
            rate_func=rate_functions.smooth
        )
        self.wait(2)
