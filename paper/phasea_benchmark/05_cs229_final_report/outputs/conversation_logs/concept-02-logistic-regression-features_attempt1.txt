from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Setup Axes and Sigmoid Curve (on the right)
        # Using Axes as a NumberPlane representation for the sigmoid function
        axes = Axes(
            x_range=[-6, 6, 2],
            y_range=[0, 1.1, 0.5],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True}
        ).shift(RIGHT * 3)
        
        def sigmoid_func(val):
            return 1 / (1 + np.exp(-val))
            
        sigmoid_curve = axes.plot(
            sigmoid_func,
            color=BLUE
        )
        curve_label = MathTex(r"P(y=1)", color=BLUE).next_to(sigmoid_curve, UP)

        # 2. Formula (centered at the bottom)
        formula = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=36
        ).to_edge(DOWN, buff=1)

        # 3. Trackers for features
        x1_tracker = ValueTracker(0)
        x30_tracker = ValueTracker(0)
        
        # Weights for the calculation (fixed constants to drive the visual)
        beta_0 = -3
        beta_1 = 0.8
        beta_30 = 0.8

        def get_current_z():
            return beta_0 + (beta_1 * x1_tracker.get_value()) + (beta_30 * x30_tracker.get_value())

        def get_current_p():
            z_val = get_current_z()
            return 1 / (1 + np.exp(-z_val))

        # 4. Sliding Bars (on the left)
        bar_height_max = 3
        bar_width = 0.6
        
        # 1-Day Bar
        bar_bg_1d = Rectangle(height=bar_height_max, width=bar_width, color=GREY, stroke_opacity=0.5).shift(LEFT * 4 + UP * 1)
        
        def get_fill_1d():
            h = max(0.01, (x1_tracker.get_value() / 4) * bar_height_max)
            rect = Rectangle(
                height=h,
                width=bar_width,
                fill_opacity=0.8,
                fill_color=YELLOW,
                stroke_width=0
            )
            rect.align_to(bar_bg_1d, DOWN)
            return rect
            
        bar_fill_1d = always_redraw(get_fill_1d)
        label_1d = Text("1-Day T-CSP", font_size=20).next_to(bar_bg_1d, DOWN)

        # 30-Day Bar
        bar_bg_30d = Rectangle(height=bar_height_max, width=bar_width, color=GREY, stroke_opacity=0.5).shift(LEFT * 2.5 + UP * 1)
        
        def get_fill_30d():
            h = max(0.01, (x30_tracker.get_value() / 4) * bar_height_max)
            rect = Rectangle(
                height=h,
                width=bar_width,
                fill_opacity=0.8,
                fill_color=YELLOW,
                stroke_width=0
            )
            rect.align_to(bar_bg_30d, DOWN)
            return rect
            
        bar_fill_30d = always_redraw(get_fill_30d)
        label_30d = Text("30-Day T-CSP", font_size=20).next_to(bar_bg_30d, DOWN)

        # 5. Dot on Sigmoid Curve
        dot = Dot(color=RED)
        
        def update_dot_position(mob):
            z_coord = get_current_z()
            p_coord = get_current_p()
            mob.move_to(axes.c2p(z_coord, p_coord))
            
        dot.add_updater(update_dot_position)

        # Connection lines
        def get_h_line():
            return axes.get_horizontal_line(dot.get_center(), color=WHITE, stroke_width=2)
            
        def get_v_line():
            return axes.get_vertical_line(dot.get_center(), color=WHITE, stroke_width=2)
            
        h_line = always_redraw(get_h_line)
        v_line = always_redraw(get_v_line)

        # 6. Animation Execution
        self.play(
            Create(axes),
            Create(sigmoid_curve),
            Write(curve_label),
            Write(formula)
        )
        self.play(
            Create(bar_bg_1d), Create(bar_fill_1d), Write(label_1d),
            Create(bar_bg_30d), Create(bar_fill_30d), Write(label_30d)
        )
        self.play(FadeIn(dot), Create(h_line), Create(v_line))
        self.wait(1)

        # Animate features rising to show model prediction changes
        # Use rate_functions.linear to ensure it's defined
        self.play(
            x1_tracker.animate.set_value(4),
            run_time=2,
            rate_func=rate_functions.linear
        )
        self.wait(0.5)
        
        self.play(
            x30_tracker.animate.set_value(4),
            run_time=2,
            rate_func=rate_functions.linear
        )
        self.wait(1)

        # Highlight formula connection to features
        formula_box = SurroundingRectangle(formula, color=YELLOW)
        self.play(Create(formula_box))
        self.wait(1)
        
        # Reset bars to show lower probability
        self.play(
            x1_tracker.animate.set_value(1),
            x30_tracker.animate.set_value(0.5),
            FadeOut(formula_box),
            run_time=2
        )
        self.wait(2)
