from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # Define a concave curve function: f(x) = x^(1/3)
        def recall_curve(x):
            if x <= 0:
                return 0
            if x >= 1:
                return 1
            return x**(1/3)

        # Create axes for Fraction of Zones Flagged and Recall
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": False}
        ).to_edge(LEFT, buff=0.7)

        # Labels for the axes
        x_label = Text("Fraction of Zones Flagged", font_size=20).next_to(axes.x_axis, DOWN)
        y_label = Text("Recall", font_size=20).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)

        # Plot the curve
        curve = axes.plot(recall_curve, x_range=[0, 1], color=BLUE)

        # ValueTracker to manage the threshold point
        tracker = ValueTracker(0.1)

        # The moving point (dot) on the curve
        dot = Dot(color=GREEN)
        def update_dot(mob):
            val = tracker.get_value()
            mob.move_to(axes.c2p(val, recall_curve(val)))
        dot.add_updater(update_dot)

        # Side-panel for the building floorplan
        floorplan_title = Text("Building Floorplan", font_size=24).to_edge(RIGHT, buff=1).shift(UP * 2)
        
        # Create a grid of 9 'Zone' squares (3x3)
        zones = VGroup(*[
            Square(side_length=0.6, fill_opacity=0.2, fill_color=GRAY, color=WHITE)
            for _ in range(9)
        ]).arrange_in_grid(rows=3, cols=3, buff=0.1).next_to(floorplan_title, DOWN, buff=0.5)

        # Updater to highlight zones based on the tracker value
        def update_zones(mob):
            val = tracker.get_value()
            # Calculate how many zones to highlight (0 to 9)
            num_on = int(max(0, min(1, val)) * 9)
            for i, zone in enumerate(mob):
                if i < num_on:
                    zone.set_fill(YELLOW, opacity=0.8)
                else:
                    zone.set_fill(GRAY, opacity=0.2)
        zones.add_updater(update_zones)

        # Reference formula for Recall
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=30).to_corner(UL)

        # Animation sequence
        self.add(axes, x_label, y_label, formula, floorplan_title, zones)
        self.play(Create(curve))
        self.add(dot)
        self.wait(1)

        # Move the tracker to the 'elbow' (approx 0.2-0.3 for x^(1/3))
        self.play(tracker.animate.set_value(0.25), run_time=2)
        
        # Highlight the 'elbow' with a circle and a flash
        elbow_point = axes.c2p(0.25, recall_curve(0.25))
        elbow_circle = Circle(radius=0.4, color=YELLOW).move_to(elbow_point)
        self.play(Create(elbow_circle), Flash(elbow_point, color=YELLOW))
        self.play(Indicate(elbow_circle))
        self.wait(1)

        # Move the tracker towards the right (high sensitivity, low selectivity)
        self.play(tracker.animate.set_value(0.95), run_time=3)
        self.wait(2)

        # Return to the elbow point
        self.play(FadeOut(elbow_circle), tracker.animate.set_value(0.25), run_time=1.5)
        self.wait(2)
