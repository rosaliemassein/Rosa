from manim import *

class ConceptAnimation03TheElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes (Using 'ax' instead of 'axes' to be safe)
        ax = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=4
        )
        ax.to_edge(LEFT, buff=0.5)
        
        # 2. Plot Concave Curve (using 't' as the variable to avoid 'x' identifier issues)
        # Using np.sqrt as it's a simple concave function
        curve = ax.plot(lambda t: np.sqrt(t) if t >= 0 else 0, x_range=[0, 1], color=BLUE)
        
        # 3. ValueTracker for movement (Using 'vt')
        vt = ValueTracker(0.1)
        
        # 4. Moving Point on Curve
        # Using an updater with 'd' as variable
        moving_dot = Dot(color=YELLOW)
        moving_dot.add_updater(lambda d: d.move_to(ax.c2p(vt.get_value(), np.sqrt(vt.get_value()))))
        
        # 5. Building Floorplan (Side Panel)
        # Manual construction of squares to avoid list comprehension issues
        zones_group = VGroup()
        for i in range(16):
            zones_group.add(Square(side_length=0.4, stroke_width=2))
        zones_group.arrange_in_grid(4, 4, buff=0.1)
        zones_group.to_edge(RIGHT, buff=1)
        
        # Updater for floorplan highlighting
        def update_zones_display(z):
            val = vt.get_value()
            num_on = int(val * 16)
            for idx, square in enumerate(z):
                if idx < num_on:
                    square.set_fill(RED, opacity=0.8)
                else:
                    square.set_fill(BLACK, opacity=0)
        
        zones_group.add_updater(update_zones_display)
        
        # 6. Labels and Formula
        # Using MathTex as per reference examples
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=36)
        formula.to_edge(DOWN, buff=0.5)
        
        label_x = MathTex(r"\text{Fraction of Zones Flagged}", font_size=24)
        label_x.next_to(ax.x_axis, DOWN)
        
        label_y = MathTex(r"\text{Recall}", font_size=24)
        label_y.rotate(90 * DEGREES)
        label_y.next_to(ax.y_axis, LEFT)
        
        # 7. Elbow Highlight
        elbow_pos = ax.c2p(0.2, np.sqrt(0.2))
        elbow_marker = Circle(radius=0.3, color=YELLOW).move_to(elbow_pos)
        
        # 8. Animation Sequence
        self.add(ax, curve, moving_dot, zones_group, formula, label_x, label_y)
        
        # Move to elbow
        self.play(vt.animate.set_value(0.2), run_time=1.5)
        self.play(Create(elbow_marker))
        self.play(Flash(elbow_marker, color=YELLOW))
        
        # Show trade-off: high recall but too many zones
        self.play(vt.animate.set_value(0.9), run_time=2.5)
        self.wait(0.5)
        
        # Return to elbow
        self.play(vt.animate.set_value(0.2), run_time=2)
        
        self.wait(2)
