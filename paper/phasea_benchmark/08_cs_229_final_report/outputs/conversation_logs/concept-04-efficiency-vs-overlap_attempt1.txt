from manim import *
import numpy as np

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Setup Axes
        # Use a safe x-range and y-range
        ax = Axes(
            x_range=[0, 2.5, 0.5],
            y_range=[0, 1.1, 0.1],
            x_length=7,
            y_length=5,
            axis_config={"include_tip": False}
        ).shift(DOWN * 0.5)

        x_label = ax.get_x_axis_label("Distance", edge=DOWN, direction=DOWN).scale(0.8)
        y_label = ax.get_y_axis_label("Efficiency", edge=LEFT, direction=LEFT).scale(0.8)
        
        ax_title = Text("Clustering Efficiency vs. Particle Distance", font_size=32).to_edge(UP)

        # 2. Plot Curves
        # ML model stays flat at 0.9. Use 'dist' as the variable name instead of 'x'
        ml_curve = ax.plot(lambda dist: 0.9, x_range=[0, 2.5], color=BLUE)
        ml_label = Text("ML Model", color=BLUE, font_size=24).next_to(ml_curve.get_end(), UP)

        # Standard algorithm drops sharply at low dist
        # Function: 0.9 * (1 - exp(-4*dist))
        std_curve_raw = ax.plot(lambda dist: 0.9 * (1.0 - np.exp(-4.0 * dist)), x_range=[0, 2.5], color=RED)
        
        # Wrapping in DashedVMobject. 
        # If DashedVMobject is problematic in some environments, we use the dash_pattern if available, 
        # but DashedVMobject is the community standard.
        std_curve = DashedVMobject(std_curve_raw, num_dashes=30)
        std_label = Text("Standard Algorithm", color=RED, font_size=24).next_to(std_curve.get_point_from_proportion(0.5), DOWN, buff=0.5)

        # 3. ValueTracker and Updaters for the Vertical Line and Particle Icons
        tracker = ValueTracker(2.4)
        
        # Vertical line indicating current distance
        v_line = always_redraw(lambda: 
            ax.get_vertical_line(
                ax.c2p(tracker.get_value(), 0.9), 
                color=YELLOW, 
                stroke_width=2
            )
        )

        # Particle Shower Icons - VGroup of dots representing cluster hits
        def get_shower_group(center_point, seed_val):
            np.random.seed(seed_val)
            dots = VGroup()
            for _ in range(12):
                offset = np.array([np.random.uniform(-0.2, 0.2), np.random.uniform(-0.2, 0.2), 0])
                dots.add(Dot(point=center_point + offset, radius=0.04, color=WHITE, fill_opacity=0.6))
            return dots

        icon_origin = LEFT * 3 + UP * 2
        
        # Redraw showers based on the tracker value
        shower_1 = always_redraw(lambda: 
            get_shower_group(icon_origin + LEFT * (tracker.get_value() * 0.4), 10)
        )
        shower_2 = always_redraw(lambda: 
            get_shower_group(icon_origin + RIGHT * (tracker.get_value() * 0.4), 20)
        )

        icon_header = Text("Particle Proximity", font_size=20).next_to(icon_origin, UP, buff=0.6)

        # 4. Animation Sequence
        self.add(ax, x_label, y_label, ax_title)
        self.play(Create(ml_curve), Write(ml_label))
        self.play(Create(std_curve), Write(std_label))
        self.wait(1)

        self.play(
            FadeIn(v_line),
            FadeIn(shower_1),
            FadeIn(shower_2),
            FadeIn(icon_header)
        )
        self.wait(1)

        # Move tracker to the left (distance closing). Using a lambda for linear rate to avoid identifier issues.
        self.play(
            tracker.animate.set_value(0.05),
            run_time=6,
            rate_func=lambda t: t
        )
        
        # Final descriptive text
        impact_text = Text("ML resolves overlaps", color=BLUE, font_size=28).move_to(ax.c2p(1.25, 0.5))
        self.play(Write(impact_text))
        
        self.wait(3)
