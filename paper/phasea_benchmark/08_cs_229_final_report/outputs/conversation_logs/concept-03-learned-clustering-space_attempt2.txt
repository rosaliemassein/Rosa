from manim import *
import numpy as np

class LearnedClusteringAnimation(Scene):
    def construct(self):
        # 1. Setup Split-Screen View
        # Create two NumberPlanes for Physical and Learned spaces
        physical_plane = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-4, 4, 1],
            x_length=5.5,
            y_length=5.5,
            background_line_style={"stroke_opacity": 0.3}
        ).to_edge(LEFT, buff=0.5)

        learned_plane = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-4, 4, 1],
            x_length=5.5,
            y_length=5.5,
            background_line_style={"stroke_opacity": 0.3}
        ).to_edge(RIGHT, buff=0.5)

        p_title = Text("Physical Space", color=BLUE).scale(0.7).next_to(physical_plane, UP)
        l_title = Text("Learned Space", color=GREEN).scale(0.7).next_to(learned_plane, UP)

        self.add(physical_plane, learned_plane, p_title, l_title)

        # 2. Generate Clusters
        np.random.seed(42)
        num_dots = 40

        # Create points for Physical Space (High overlap)
        # Red cluster centered slightly right, Blue slightly left
        p_red_points = [
            physical_plane.c2p(np.random.normal(0.4, 0.8), np.random.normal(0.4, 0.8))
            for _ in range(num_dots)
        ]
        p_blue_points = [
            physical_plane.c2p(np.random.normal(-0.4, 0.8), np.random.normal(-0.4, 0.8))
            for _ in range(num_dots)
        ]

        # Create points for Learned Space (Distinct blobs)
        # Clusters are now far apart
        l_red_points = [
            learned_plane.c2p(np.random.normal(2.0, 0.4), np.random.normal(2.0, 0.4))
            for _ in range(num_dots)
        ]
        l_blue_points = [
            learned_plane.c2p(np.random.normal(-2.0, 0.4), np.random.normal(-2.0, 0.4))
            for _ in range(num_dots)
        ]

        dots_red = VGroup(*[Dot(p, color=RED, radius=0.06) for p in p_red_points])
        dots_blue = VGroup(*[Dot(p, color=BLUE, radius=0.06) for p in p_blue_points])

        # 3. Helper for Dashed Circles (Manual to avoid identifier errors)
        def get_dashed_circle(center, radius, color):
            dashes = VGroup()
            num_segments = 20
            for i in range(num_segments):
                dashes.add(Arc(
                    radius=radius,
                    start_angle=i * (TAU / num_segments),
                    angle=TAU / (num_segments * 2),
                    arc_center=center,
                    color=color,
                    stroke_width=2
                ))
            return dashes

        # 4. Animation Sequence
        # Show hits in physical space
        self.play(FadeIn(dots_red), FadeIn(dots_blue))
        self.wait(1)

        # Transform hits to learned space
        self.play(
            *[dots_red[i].animate.move_to(l_red_points[i]) for i in range(num_dots)],
            *[dots_blue[i].animate.move_to(l_blue_points[i]) for i in range(num_dots)],
            run_time=3,
            path_arc=0.3
        )

        # Draw classification circles
        circle_red = get_dashed_circle(learned_plane.c2p(2.0, 2.0), 1.0, RED)
        circle_blue = get_dashed_circle(learned_plane.c2p(-2.0, -2.0), 1.0, BLUE)
        
        self.play(Create(circle_red), Create(circle_blue))
        self.wait(0.5)

        # 5. Text and Formula
        voice_text = Text(
            "Mapping hits to a multi-dimensional 'learned coordinate' space.",
            slant="ITALIC"
        ).scale(0.45).to_edge(DOWN, buff=0.8)

        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            color=YELLOW
        ).scale(0.7).next_to(voice_text, DOWN, buff=0.2)

        self.play(Write(voice_text))
        self.play(FadeIn(formula))

        self.wait(3)
