from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup - 2D plane of dots
        num_dots = 40
        dots = VGroup()
        for _ in range(num_dots):
            dot = Dot(
                point=[np.random.uniform(-5, 5), np.random.uniform(-3, 2), 0],
                radius=0.08,
                color=PURPLE
            )
            dots.add(dot)
        
        # Define beta values
        betas = np.random.uniform(0.1, 0.5, num_dots)
        # Select 3 condensation points with high beta
        high_beta_indices = [5, 15, 25]
        for idx in high_beta_indices:
            betas[idx] = 0.95

        # 2. Formula Label
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula.to_edge(UP)
        
        self.add(dots)
        self.play(Write(formula))
        self.wait(1)

        # 3. Animate growth and color change based on beta
        # We'll use a simple color palette to simulate interpolation
        color_palette = [PURPLE, BLUE, TEAL, GREEN, YELLOW]
        
        grow_anims = []
        for i, dot in enumerate(dots):
            beta = betas[i]
            # Manual color selection based on beta
            color_idx = int(beta * (len(color_palette) - 1))
            target_color = color_palette[color_idx]
            # Manual radius interpolation
            target_radius = 0.05 * (1 - beta) + 0.25 * beta
            
            grow_anims.append(
                dot.animate.set_color(target_color).scale(target_radius / 0.08)
            )
            
        self.play(*grow_anims, run_time=2)
        self.wait(1)

        # 4. Create "Gravity Wells" (Concentric circles with gradient feel)
        wells = VGroup()
        condensation_points = []
        for idx in high_beta_indices:
            cp = dots[idx]
            condensation_points.append(cp)
            
            # Create a localized "mesh" or gradient well
            well = VGroup()
            for r in np.linspace(0.2, 1.8, 8):
                ring = Circle(radius=r, stroke_width=1, stroke_opacity=0.4 - r*0.2)
                ring.set_color(YELLOW)
                ring.move_to(cp.get_center())
                well.add(ring)
            wells.add(well)

        self.play(FadeIn(wells, lag_ratio=0.1))
        self.wait(1)

        # 5. Pulling effect - dots slide toward high-beta centers
        pull_anims = []
        for i, dot in enumerate(dots):
            if i in high_beta_indices:
                continue
            
            # Find the closest condensation point
            dot_pos = dot.get_center()
            distances = [np.linalg.norm(dot_pos - cp.get_center()) for cp in condensation_points]
            nearest_idx = np.argmin(distances)
            target_cp = condensation_points[nearest_idx]
            
            # Only pull if within range (to simulate gravity)
            if distances[nearest_idx] < 4.0:
                # Move 80% of the way to the center
                destination = dot_pos + 0.8 * (target_cp.get_center() - dot_pos)
                pull_anims.append(dot.animate.move_to(destination))

        # Using 'rush_into' as a standard Manim rate function for acceleration
        self.play(*pull_anims, run_time=2, rate_func=rush_into)
        self.wait(2)

        # Clean up
        self.play(
            FadeOut(dots),
            FadeOut(wells),
            FadeOut(formula)
        )
        self.wait(1)
