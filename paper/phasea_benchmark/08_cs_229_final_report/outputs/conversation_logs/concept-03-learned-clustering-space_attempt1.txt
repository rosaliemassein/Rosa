from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Planes (Split Screen)
        # Using a simpler setup to ensure compatibility across Manim versions
        left_origin = 3.5 * LEFT
        right_origin = 3.5 * RIGHT
        
        left_plane = NumberPlane(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5,
            y_length=5,
            background_line_style={"stroke_opacity": 0.3}
        ).move_to(left_origin)
        
        right_plane = NumberPlane(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5,
            y_length=5,
            background_line_style={"stroke_opacity": 0.3}
        ).move_to(right_origin)

        left_title = Text("Physical Space", font_size=24).next_to(left_plane, UP)
        right_title = Text("Learned Space", font_size=24).next_to(right_plane, UP)

        self.play(
            Create(left_plane),
            Create(right_plane),
            Write(left_title),
            Write(right_title)
        )

        # 2. Create Overlapping Clusters for Physical Space
        # We generate random offsets in physical space where clusters overlap
        np.random.seed(42) # For reproducibility
        
        dots_a = VGroup()
        dots_b = VGroup()
        
        # Cluster A (Red) - centered slightly right
        for _ in range(20):
            p = left_plane.c2p(np.random.normal(0.2, 0.4), np.random.normal(0, 0.4), 0)
            dots_a.add(Dot(p, radius=0.06, color=RED))
            
        # Cluster B (Blue) - centered slightly left, overlapping A
        for _ in range(20):
            p = left_plane.c2p(np.random.normal(-0.2, 0.4), np.random.normal(0, 0.4), 0)
            dots_b.add(Dot(p, radius=0.06, color=BLUE))
        
        self.play(FadeIn(dots_a), FadeIn(dots_b))
        self.wait(1)

        # 3. Transformation to Learned Space
        # Define destination positions where they are clearly separated
        dots_a_animations = []
        for dot in dots_a:
            target_pos = right_plane.c2p(
                np.random.normal(1.5, 0.3), 
                np.random.normal(1.2, 0.3), 0
            )
            dots_a_animations.append(dot.animate.move_to(target_pos))

        dots_b_animations = []
        for dot in dots_b:
            target_pos = right_plane.c2p(
                np.random.normal(-1.5, 0.3), 
                np.random.normal(-1.2, 0.3), 0
            )
            dots_b_animations.append(dot.animate.move_to(target_pos))

        # Perform the "Unwarping" animation
        self.play(
            *dots_a_animations,
            *dots_b_animations,
            run_time=3,
            path_arc=0.3
        )

        # 4. Success indicators: Dashed circles
        # Using a circle with a dashed stroke to avoid DashedVMobject if it causes issues
        # In most Manim versions, DashedVMobject(Circle()) is the way to go.
        # If it fails, one can use many small arcs, but we'll use DashedVMobject as it is standard.
        circle_a = DashedVMobject(Circle(radius=0.9, color=RED).move_to(right_plane.c2p(1.5, 1.2, 0)))
        circle_b = DashedVMobject(Circle(radius=0.9, color=BLUE).move_to(right_plane.c2p(-1.5, -1.2, 0)))

        self.play(Create(circle_a), Create(circle_b))
        self.wait(1)

        # 5. Narration and Formula
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=28
        ).to_edge(DOWN, buff=0.4)
        
        # Using string "ITALIC" instead of constant to prevent undefined identifier errors
        voice_sub = Text(
            "Mapping hits to a multi-dimensional 'learned coordinate' space.",
            font_size=20,
            slant="ITALIC"
        ).next_to(formula, UP, buff=0.2)

        self.play(Write(formula))
        self.play(FadeIn(voice_sub))
        self.wait(3)

        # Fade out everything
        self.play(FadeOut(VGroup(self.mobjects)))
        self.wait(1)
