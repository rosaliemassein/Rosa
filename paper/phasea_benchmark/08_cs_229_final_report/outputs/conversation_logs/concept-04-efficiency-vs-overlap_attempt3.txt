import numpy as np

class Concept04EfficiencyVsOverlap(Scene):

        label_dist = Text("Particle Distance").scale(0.4).next_to(horiz_axis, DOWN)
        label_eff = Text("Efficiency").scale(0.4).rotate(90 * DEGREES).next_to(vert_axis, LEFT)
        
        self.add(horiz_axis, vert_axis, label_dist, label_eff)

        # 2. Efficiency Curves
        # ML Model (BLUE) - Remains mostly flat at ~0.9 height
        ml_y_lvl = y_min_val + 3.6 
        ml_line = Line([x_min_val, ml_y_lvl, 0], [x_max_val, ml_y_lvl, 0], color=BLUE)
        ml_tag = Text("ML Approach", color=BLUE).scale(0.4).next_to(ml_line, UP, buff=0.1)

        # Standard Algorithm (RED) - Drops sharply at low distance
        # Constructing points manually to avoid plot/lambdas
        curve_points = []
        num_steps = 30
        for idx in range(num_steps + 1):
            ratio = idx / num_steps
            coord_x = x_min_val + (ratio * (x_max_val - x_min_val))
            # Sigmoid-like drop: starts at 0.9, drops to 0 at low ratio
            # Using basic math to avoid restricted functions if necessary
            sig_val = 1.0 / (1.0 + np.exp(-12.0 * (ratio - 0.25)))
            coord_y = y_min_val + (sig_val * 3.6)
            curve_points.append([coord_x, coord_y, 0])
        
        std_line = VMobject(color=RED)
        std_line.set_points_as_corners(curve_points)
        std_tag = Text("Standard Algorithm", color=RED).scale(0.4).next_to(std_line.get_end(), DOWN)

        self.add(ml_line, ml_tag, std_line, std_tag)

        # 3. Scanning Indicator
        # Moving vertical line
        scan_bar = Line(
            [x_max_val, y_min_val, 0], 
            [x_max_val, y_max_val + 0.5, 0], 
            color=YELLOW, 
            stroke_width=2
        )

        # 4. Particle Shower Visuals
        # Two dots that merge as the scan_bar moves left
        visual_box = Rectangle(width=3, height=1.5, color=GRAY_A).to_edge(UP, buff=0.3)
        p_shower_1 = Dot(color=YELLOW)
        p_shower_2 = Dot(color=YELLOW)
        
        # Updater function to handle the "merging" logic based on scan_bar position
        def merge_updater(mobject_ref):
            # Calculate progress based on scan_bar x coordinate
            # From x_max_val (far) to x_min_val (close)
            current_pos_x = scan_bar.get_center()[0]
            norm_progress = (current_pos_x - x_min_val) / (x_max_val - x_min_val)
            # Offset particles based on progress
            spacing = norm_progress * 1.0
            p_shower_1.move_to(visual_box.get_center() + LEFT * spacing)
            p_shower_2.move_to(visual_box.get_center() + RIGHT * spacing)

        scan_bar.add_updater(merge_updater)
        self.add(visual_box, p_shower_1, p_shower_2, scan_bar)

        # 5. Narrative Text

            "ML resolves overlapping clusters where standard methods fail.",
            color=WHITE
        ).scale(0.5).to_edge(DOWN, buff=0.2)

            run_time=8
        )
