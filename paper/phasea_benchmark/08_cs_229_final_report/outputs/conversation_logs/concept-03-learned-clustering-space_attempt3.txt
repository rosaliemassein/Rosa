from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Split Screen Planes
        left_plane = NumberPlane(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5,
            y_length=5,
            background_line_style={"stroke_opacity": 0.3}
        ).shift(3.5 * LEFT)

        right_plane = NumberPlane(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5,
            y_length=5,
            background_line_style={"stroke_opacity": 0.3}
        ).shift(3.5 * RIGHT)

        left_label = Text("Physical Space", color=WHITE, font_size=24).next_to(left_plane, UP)
        right_label = Text("Learned Space", color=WHITE, font_size=24).next_to(right_plane, UP)

        # 2. Formula (Reference from Prompt)
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            color=WHITE, font_size=32
        ).to_edge(DOWN, buff=0.4)

        # 3. Points Generation
        np.random.seed(42)
        n_points = 35
        
        # Left side: Overlapping clusters
        center_left = left_plane.get_center()
        red_group_phys = VGroup()
        blue_group_phys = VGroup()
        
        for _ in range(n_points):
            # Centered close together to overlap
            pos_red = center_left + np.array([np.random.normal(0.3, 0.6), np.random.normal(0, 0.6), 0])
            pos_blue = center_left + np.array([np.random.normal(-0.3, 0.6), np.random.normal(0, 0.6), 0])
            red_group_phys.add(Dot(pos_red, color=RED, radius=0.05))
            blue_group_phys.add(Dot(pos_blue, color=BLUE, radius=0.05))

        # Right side: Distinct blobs
        center_red_learned = right_plane.get_center() + 1.2 * UP + 0.7 * RIGHT
        center_blue_learned = right_plane.get_center() + 1.2 * DOWN + 0.7 * LEFT
        
        red_group_learned = VGroup()
        blue_group_learned = VGroup()
        
        for _ in range(n_points):
            pos_red = center_red_learned + np.array([np.random.normal(0, 0.4), np.random.normal(0, 0.4), 0])
            pos_blue = center_blue_learned + np.array([np.random.normal(0, 0.4), np.random.normal(0, 0.4), 0])
            red_group_learned.add(Dot(pos_red, color=RED, radius=0.05))
            blue_group_learned.add(Dot(pos_blue, color=BLUE, radius=0.05))

        # 4. Dashed Circles (Created manually using Arc segments for max compatibility)
        def get_dashed_circle(center, radius, color):
            dashes = VGroup()
            n_segments = 20
            for i in range(n_segments):
                dashes.add(Arc(
                    radius=radius,
                    start_angle=i * (TAU / n_segments),
                    angle=TAU / (2 * n_segments),
                    arc_center=center,
                    color=color,
                    stroke_width=2
                ))
            return dashes

        circle_red = get_dashed_circle(center_red_learned, 1.0, RED)
        circle_blue = get_dashed_circle(center_blue_learned, 1.0, BLUE)

        # 5. Narration Text
        narration = Text(
            "Mapping hits from physical detector space to learned coordinate space...",
            font_size=20
        ).next_to(formula, UP, buff=0.3)

        # 6. Execution
        self.add(left_plane, right_plane, left_label, right_label, formula)
        self.play(FadeIn(red_group_phys), FadeIn(blue_group_phys))
        self.wait(1)
        
        self.play(Write(narration))
        self.wait(0.5)

        # Transition to learned space
        self.play(
            ReplacementTransform(red_group_phys, red_group_learned),
            ReplacementTransform(blue_group_phys, blue_group_learned),
            run_time=3,
            path_arc=0.5
        )
        
        # Final classification indicator
        self.play(Create(circle_red), Create(circle_blue))
        
        final_narration = Text(
            "Complex overlapping energy deposits are now easily separable.",
            font_size=20
        ).move_to(narration)
        
        self.play(FadeOut(narration), FadeIn(final_narration))
        self.wait(2)
