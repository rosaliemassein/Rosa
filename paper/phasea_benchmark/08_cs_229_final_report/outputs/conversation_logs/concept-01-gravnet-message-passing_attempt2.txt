from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # 2. Create an irregular grid of dots
        # Using a 2D approach since 3D features were restricted
        dots = VGroup()
        rows, cols = 5, 5
        points = []
        for r in range(rows):
            for c in range(cols):
                # Hexagonal-like grid spacing
                x = (c + 0.5 * (r % 2)) * 1.4 - 3.0
                y = r * 1.0 - 2.0
                # Adding some jitter for "irregularity"
                x += np.random.uniform(-0.15, 0.15)
                y += np.random.uniform(-0.15, 0.15)
                points.append([x, y, 0])
        
        for p in points:
            dots.add(Dot(point=p, radius=0.08, color=GRAY_B))

        # Identify central dot
        central_idx = len(dots) // 2
        central_dot = dots[central_idx]
        
        self.play(FadeIn(dots))
        self.play(central_dot.animate.set_color(RED).scale(1.4))
        self.wait(0.5)

        # 3. Helper for distance-weighted styling
        def get_line_style(p1, p2):
            dist = np.linalg.norm(p1 - p2)
            # The formula exp(-d^2). We scale d slightly for better visuals in screen space.
            weight = np.exp(-(dist**2) / 2.0)
            return {
                "stroke_width": weight * 12,
                "stroke_opacity": weight
            }

        # 4. Create connections (lines) with updaters
        connections = VGroup()
        for i, dot in enumerate(dots):
            if i == central_idx:
                continue
            
            # Initial style
            style = get_line_style(central_dot.get_center(), dot.get_center())
            line = Line(
                central_dot.get_center(), 
                dot.get_center(), 
                color=YELLOW,
                buff=0.1
            ).set_stroke(width=style["stroke_width"], opacity=style["stroke_opacity"])
            
            # Dynamic updater to respond to moving dots
            def line_updater(obj, d_node=dot):
                p1 = central_dot.get_center()
                p2 = d_node.get_center()
                s = get_line_style(p1, p2)
                obj.put_start_and_end_on(p1, p2)
                obj.set_stroke(width=s["stroke_width"], opacity=s["stroke_opacity"])
            
            line.add_updater(line_updater)
            connections.add(line)

        self.play(Create(connections), run_time=2)
        self.wait(1)

        # 5. Animate "learned features" updating by moving dots
        # This will trigger the updaters on the connections
        move_animations = []
        for dot in dots:
            # Move each dot to a new nearby random position
            random_offset = np.array([
                np.random.uniform(-1.0, 1.0),
                np.random.uniform(-1.0, 1.0),
                0
            ])
            move_animations.append(dot.animate.shift(random_offset))

        self.play(
            *move_animations,
            run_time=4,
            rate_func=smooth
        )
        
        self.wait(1)
        
        # Highlight the final dynamic state
        self.play(Indicate(central_dot, color=RED))
        self.wait(2)
