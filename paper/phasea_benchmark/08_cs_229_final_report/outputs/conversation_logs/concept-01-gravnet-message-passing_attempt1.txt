from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Create an irregular grid of dots
        # We simulate a 3D-like perspective by projecting 3D coords to 2D
        dots = VGroup()
        points = []
        for row in range(-2, 3):
            for col in range(-2, 3):
                # Hexagonal grid base
                x = col * 1.2 + (0.6 if row % 2 else 0)
                y = row * 1.0
                # Add some irregularity
                pos = np.array([
                    x + np.random.uniform(-0.3, 0.3),
                    y + np.random.uniform(-0.3, 0.3),
                    0
                ])
                points.append(pos)
                dots.add(Dot(point=pos, color=RED, radius=0.08))
        
        # Highlight a central dot
        central_index = 12
        central_dot = dots[central_index]
        central_dot.set_color(GREEN)

        # 2. Formula and Text
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)")
        formula.to_corner(UL).scale(1.2)
        
        voiceover_text = Text(
            "GravNet: Distance-weighted graph connections", 
            font_size=24
        ).to_edge(DOWN)

        # 3. Create connections that update dynamically
        connections = VGroup()

        def get_weighted_connections(vg):
            vg.submobjects = []
            center_pos = central_dot.get_center()
            for dot in dots:
                if dot == central_dot:
                    continue
                
                target_pos = dot.get_center()
                dist = np.linalg.norm(target_pos - center_pos)
                
                # Use a threshold for neighborhood
                if dist < 3.5:
                    # Connection strength based on distance formula
                    weight = np.exp(-(dist**2) / 2.0)
                    
                    line = Line(
                        center_pos, target_pos,
                        stroke_width=weight * 8,
                        stroke_opacity=weight,
                        color=WHITE,
                        buff=0.1
                    )
                    # Add tip to make it an arrow
                    line.add_tip(tip_length=0.15 * weight, tip_width=0.15 * weight)
                    vg.add(line)

        connections.add_updater(get_weighted_connections)

        # 4. Animation Sequence
        self.add(dots, formula, connections, voiceover_text)
        self.play(Create(dots), Write(formula), FadeIn(voiceover_text))
        self.wait(1)

        # 5. Move dots to simulate "learned features" updating
        # This causes the connections to shift dynamically via the updater
        move_anims = []
        for dot in dots:
            # Shift dots randomly to represent the feature space evolution
            shift_vector = np.array([
                np.random.uniform(-1.0, 1.0),
                np.random.uniform(-1.0, 1.0),
                0
            ])
            move_anims.append(dot.animate.shift(shift_vector))

        update_desc = Text(
            "Learned features shift the graph structure...", 
            font_size=24
        ).to_edge(DOWN)

        self.play(
            Transform(voiceover_text, update_desc),
            *move_anims,
            run_time=4,
            rate_func=smooth
        )
        self.wait(1)

        # Final explanation text
        conclusion = Text(
            "Connection strength is dynamic, not fixed by geometry.", 
            font_size=24
        ).to_edge(DOWN)
        self.play(Transform(voiceover_text, conclusion))
        self.wait(2)

        # 6. Optional Image handling
        try:
            image = ImageMobject("img-0.jpeg")
            image.scale(0.5).to_corner(DR)
            self.play(FadeIn(image))
            self.wait(2)
        except:
            # Skip if file not found
            pass

    def get_image_ref(self):
        return "img-0.jpeg"
