from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # Configuration
        n_dots = 40
        
        # Create a set of dots in an irregular/jittered layout
        dots = VGroup()
        for i in range(n_dots):
            # Start in a randomish circular area to represent physical space
            angle = np.random.uniform(0, 2 * PI)
            radius = np.random.uniform(0, 3.5)
            pos = np.array([radius * np.cos(angle), radius * np.sin(angle), 0])
            dot = Dot(point=pos, color=BLUE, radius=0.1)
            dots.add(dot)

        # Select a central dot to focus on
        center_idx = 0
        center_dot = dots[center_idx]
        center_dot.set_color(RED)
        center_dot.set_z_index(10)

        # Formula and Title
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # Function to update line properties based on distance
        def update_connection(line, target_dot):
            p1 = center_dot.get_center()
            p2 = target_dot.get_center()
            dist = np.linalg.norm(p1 - p2)
            # Distance scaling for visualization: 
            # d=0 -> weight=1, d=3 -> weight ~0.1
            weight = np.exp(-(dist / 2.0)**2)
            
            line.put_start_and_end_on(p1, p2)
            line.set_stroke(width=weight * 12, opacity=weight)

        # Create connections (Lines)
        connections = VGroup()
        for i, dot in enumerate(dots):
            if i == center_idx:
                continue
            
            line = Line(center_dot.get_center(), dot.get_center(), color=YELLOW)
            # Use a closure to capture the specific dot for the updater
            line.add_updater(lambda l, d=dot: update_connection(l, d))
            connections.add(line)

        # Animation Sequence
        self.play(FadeIn(dots))
        self.wait(0.5)
        
        # Highlight the central dot and draw connections
        self.play(
            center_dot.animate.scale(1.5),
            Create(connections),
            run_time=1.5
        )
        self.wait(1)

        # Animation: Transition to "learned feature space"
        # We move dots to new positions where some previously far dots might become close
        move_animations = []
        for i, dot in enumerate(dots):
            # Generate a new target position
            if i == center_idx:
                new_pos = ORIGIN # Keep center at origin for clarity
            else:
                # Some dots move closer to center, some move further away
                # to show dynamic connectivity
                angle = np.random.uniform(0, 2 * PI)
                # Mixture of close and far to demonstrate weights
                dist = np.random.choice([0.8, 1.2, 4.0, 5.0], p=[0.2, 0.3, 0.3, 0.2])
                new_pos = np.array([dist * np.cos(angle), dist * np.sin(angle), 0])
            
            move_animations.append(dot.animate.move_to(new_pos))

        # Perform the transition
        # The updaters on 'connections' will automatically adjust stroke and opacity
        self.play(
            *move_animations,
            run_time=4,
            rate_func=bezier([0, 0, 1, 1])
        )
        
        self.wait(2)

        # Final jitter to show dynamic nature
        jitter_anims = []
        for dot in dots:
            jitter_pos = dot.get_center() + np.random.uniform(-0.4, 0.4, 3)
            jitter_anims.append(dot.animate.move_to(jitter_pos))
            
        self.play(*jitter_anims, run_time=2, rate_func=there_and_back)
        self.wait(1)
