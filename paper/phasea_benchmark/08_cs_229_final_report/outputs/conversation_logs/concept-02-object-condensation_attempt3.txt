from manim import *
import numpy as np

class Concept02ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup Data and Parameters
        q_min = 0.5
        num_dots = 40
        
        # 2. Display the Formula
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula.to_edge(UP)
        self.play(Write(formula))
        
        # 3. Create a 2D plane of dots
        # We manually manage dots to track their beta values
        dots_group = VGroup()
        dots_data = []
        
        # Define some indices to be high-beta "condensation points"
        cp_indices = [8, 22, 35]
        
        for i in range(num_dots):
            # Random position in the frame
            pos = [np.random.uniform(-5, 5), np.random.uniform(-3, 1), 0]
            
            # Assign beta values
            if i in cp_indices:
                beta = np.random.uniform(0.85, 0.95)
            else:
                beta = np.random.uniform(0.1, 0.4)
            
            # Start with small purple dots
            dot = Dot(point=pos, radius=0.08, color=PURPLE)
            dots_group.add(dot)
            dots_data.append({
                "mobject": dot,
                "beta": beta,
                "is_cp": i in cp_indices
            })
            
        self.play(Create(dots_group))
        self.wait(0.5)
        
        # 4. Animate hits to charges (Beta-charge mechanism)
        # Dots grow in size and change color based on beta
        growth_anims = []
        for item in dots_data:
            beta = item["beta"]
            # Calculate charge q using the formula (clipped for stability)
            q = np.arctanh(np.clip(beta, 0.01, 0.99)) + q_min
            
            # Determine color: interpolate from PURPLE to YELLOW based on beta
            # Since interpolate_color was reported as undefined, we use a simple scale
            if beta > 0.8:
                target_color = YELLOW
            elif beta > 0.5:
                target_color = ORANGE
            else:
                target_color = PURPLE
                
            # Scale factor based on charge q
            target_scale = q * 1.2
            
            growth_anims.append(
                item["mobject"].animate.set_color(target_color).scale(target_scale)
            )
            
        self.play(*growth_anims, run_time=2)
        self.wait(0.5)
        
        # 5. Create 'gravity wells' (potential functions)
        # Use concentric circles to represent the attractive potential
        wells = VGroup()
        cp_items = [d for d in dots_data if d["is_cp"]]
        for cp in cp_items:
            center = cp["mobject"].get_center()
            # Gradient circle effect using multiple concentric circles with low opacity
            well = VGroup(*[
                Circle(radius=r, stroke_width=1, color=YELLOW).set_opacity(0.4 - r/5)
                for r in [0.4, 0.8, 1.2, 1.6]
            ]).move_to(center)
            wells.add(well)
            
        self.play(FadeIn(wells, scale=0.8), run_time=1.5)
        self.wait(0.5)
        
        # 6. Clustering: Pull noise dots toward the condensation points
        # Neighboring low-beta dots slide toward the nearest gravity well
        pull_anims = []
        noise_items = [d for d in dots_data if not d["is_cp"]]
        for item in noise_items:
            start_pos = item["mobject"].get_center()
            # Find the center of the closest condensation point
            closest_cp_pos = min(
                [cp["mobject"].get_center() for cp in cp_items],
                key=lambda p: np.linalg.norm(p - start_pos)
            )
            
            # Calculate destination (move 70% toward the center)
            direction_vector = closest_cp_pos - start_pos
            destination = start_pos + direction_vector * 0.7
            
            # Use basic animate.move_to for maximum compatibility
            pull_anims.append(item["mobject"].animate.move_to(destination))
            
        self.play(*pull_anims, run_time=3, rate_func=bezier([0, 0, 1, 1]))
        
        # 7. Final Caption
        narration = Text("Condensation points act like gravity, pulling hits into particles.", font_size=24)
        narration.to_edge(DOWN)
        self.play(Write(narration))
        
        self.wait(2)
