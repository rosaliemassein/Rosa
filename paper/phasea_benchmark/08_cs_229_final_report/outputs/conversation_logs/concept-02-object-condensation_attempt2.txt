from manim import *
import numpy as np
import random

class ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup Plane and Formula
        plane = NumberPlane(
            x_range=[-7, 7, 1],
            y_range=[-4, 4, 1],
            background_line_style={"stroke_opacity": 0.2}
        )
        self.add(plane)

        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula.to_edge(UP)
        formula_bg = BackgroundRectangle(formula, color=BLACK, fill_opacity=0.8, buff=0.1)
        self.add(formula_bg, formula)

        # 2. Create Dots (Hits)
        num_dots = 40
        dots = VGroup()
        for _ in range(num_dots):
            pos = [random.uniform(-6, 6), random.uniform(-3, 2), 0]
            dot = Dot(point=pos, radius=0.08, color=PURPLE)
            # Assign a simulated beta value
            dot.beta_val = random.uniform(0.05, 0.2) 
            dots.add(dot)

        self.play(LaggedStart(*[FadeIn(d) for d in dots], lag_ratio=0.05))
        self.wait(0.5)

        # 3. Simulate Beta growth for "Condensation Points"
        num_clusters = 4
        # Choose random indices for condensation points
        cp_indices = random.sample(range(num_dots), num_clusters)
        
        grow_anims = []
        wells = VGroup()
        
        # Track CP objects
        condensation_points = []

        for idx in cp_indices:
            dot = dots[idx]
            dot.beta_val = random.uniform(0.8, 0.95)
            condensation_points.append(dot)
            
            # High beta dots turn yellow and grow
            grow_anims.append(dot.animate.set_color(YELLOW).scale(2.5))
            
            # Create "gravity well" effect using concentric circles
            well = VGroup(*[
                Circle(radius=r, color=YELLOW, stroke_width=1).set_opacity(0.6 - (r/1.5))
                for r in np.linspace(0.2, 1.2, 5)
            ])
            well.move_to(dot.get_center())
            wells.add(well)

        self.play(*grow_anims, run_time=1.5)
        self.play(Create(wells), run_time=1)
        self.wait(0.5)

        # 4. Pulling dots toward centers
        pull_anims = []
        for i, dot in enumerate(dots):
            if i in cp_indices:
                continue
            
            # Find the nearest condensation point manually to avoid lambda scope issues
            best_dist = float('inf')
            nearest_cp = None
            for cp_obj in condensation_points:
                d = np.linalg.norm(cp_obj.get_center() - dot.get_center())
                if d < best_dist:
                    best_dist = d
                    nearest_cp = cp_obj
            
            # Only pull dots within a certain range
            if best_dist < 3.5:
                # Move towards center (75% of the way)
                direction = nearest_cp.get_center() - dot.get_center()
                target_pos = dot.get_center() + direction * 0.75
                
                # Blend color from PURPLE to YELLOW
                new_color = interpolate_color(PURPLE, YELLOW, 0.6)
                pull_anims.append(
                    dot.animate.move_to(target_pos).set_color(new_color)
                )

        # 5. Run the clustering animation
        if pull_anims:
            self.play(*pull_anims, run_time=2, rate_func=rate_functions.ease_in_sine)
        
        # Final highlight
        self.play(Indicate(formula))
        self.wait(2)
