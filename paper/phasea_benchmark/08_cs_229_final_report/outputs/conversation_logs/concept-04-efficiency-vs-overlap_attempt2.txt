from manim import *
import numpy as np

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            axis_config={"include_tip": True},
            x_length=9,
            y_length=5
        ).shift(DOWN * 0.5)

        x_label = Text("Distance between particles", font_size=20).next_to(axes.x_axis, DOWN)
        y_label = Text("Clustering Efficiency", font_size=20).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)

        # 2. Define Plotting Functions
        # Using 'val' instead of 'x' to avoid potential identifier conflicts
        def ml_efficiency_func(val):
            return 0.9

        def std_efficiency_func(val):
            # Drops to 0 as distance (val) approaches 0
            if val < 0.001:
                return 0
            return 0.9 * (1.0 - np.exp(-10.0 * val))

        # 3. Create Curves
        ml_curve = axes.plot(ml_efficiency_func, x_range=[0, 1], color=BLUE)
        
        # Manually create a dashed curve for the standard algorithm
        std_curve_base = axes.plot(std_efficiency_func, x_range=[0, 1], color=RED)
        std_curve = DashedVMobject(std_curve_base, num_dashes=40)

        ml_tag = Text("ML Model", color=BLUE, font_size=18).move_to(axes.c2p(0.5, 0.97))
        std_tag = Text("Standard Algorithm", color=RED, font_size=18).move_to(axes.c2p(0.7, 0.4))

        # 4. Interactive Elements (ValueTracker)
        # Starting at distance 1.0 (efficiency ~0.9 for both)
        dist_tracker = ValueTracker(1.0)

        # Vertical line following the tracker
        v_line = always_redraw(lambda: 
            axes.get_vertical_line(
                axes.c2p(dist_tracker.get_value(), 0.9),
                color=WHITE,
                stroke_width=2
            )
        )

        # Visual representation of particle showers merging
        def get_shower_icons(dist):
            center_y_pos = 2.8
            left_shower = VGroup(
                Dot(point=[-dist * 2, center_y_pos, 0], radius=0.1, color=YELLOW),
                Dot(point=[-dist * 2 + 0.1, center_y_pos + 0.1, 0], radius=0.03, color=YELLOW, fill_opacity=0.5)
            )
            right_shower = VGroup(
                Dot(point=[dist * 2, center_y_pos, 0], radius=0.1, color=YELLOW),
                Dot(point=[dist * 2 - 0.1, center_y_pos - 0.1, 0], radius=0.03, color=YELLOW, fill_opacity=0.5)
            )
            return VGroup(left_shower, right_shower)

        shower_icons = always_redraw(lambda: get_shower_icons(dist_tracker.get_value()))
        
        # Background for the icons
        shower_box = RoundedRectangle(height=1.0, width=5.0, corner_radius=0.1, color=GRAY).move_to(UP * 2.8)
        shower_label = Text("Particle Proximity", font_size=14, color=GRAY).next_to(shower_box, UP, buff=0.1)

        # 5. Build and Animate
        self.add(axes, x_label, y_label, ml_curve, std_curve, ml_tag, std_tag)
        self.add(shower_box, shower_label, v_line, shower_icons)
        
        # Narration text
        narration = Text(
            "ML maintains efficiency even as particles overlap.",
            font_size=20
        ).to_edge(UP, buff=0.2)
        
        self.play(Write(narration))
        
        # Animate the distance closing (moving to the left on the axis)
        self.play(
            dist_tracker.animate.set_value(0.01),
            run_time=8,
            rate_func=rate_functions.linear
        )
        self.wait(2)
