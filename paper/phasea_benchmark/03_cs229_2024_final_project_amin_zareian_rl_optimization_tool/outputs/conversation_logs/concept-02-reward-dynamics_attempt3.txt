from manim import *

class RewardDynamics(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex(r"R_t = \text{Border}_{t} - \text{Border}_{t-1}").to_edge(UP)
        self.add(formula)

        # 2. Setup 3x3 Grid
        grid = VGroup()
        cells = []
        for i in range(3):
            row = []
            for j in range(3):
                sq = Square(side_length=1.0)
                sq.shift(RIGHT * (j - 1) + DOWN * (i - 1))
                sq.set_fill(RED, opacity=0.7)
                sq.set_stroke(WHITE, width=2)
                grid.add(sq)
                row.append(sq)
            cells.append(row)
        
        # State: Top neighbor is BLUE, others are RED
        cells[0][1].set_fill(BLUE, opacity=0.7)
        
        grid_group = grid.to_edge(LEFT, buff=1.5)
        self.add(grid_group)

        # 3. Focus on center pixel and neighbors
        center_sq = cells[1][1]
        highlight = SurroundingRectangle(center_sq, color=YELLOW, buff=0)
        self.play(Create(highlight))
        
        # Initial border line (White) between center (Red) and top neighbor (Blue)
        initial_border = Line(
            center_sq.get_top() + LEFT * 0.5, 
            center_sq.get_top() + RIGHT * 0.5, 
            color=WHITE, stroke_width=6
        )
        self.play(Create(initial_border))
        self.wait(0.5)

        # 4. Animate Flip (Red to Blue)
        self.play(center_sq.animate.set_fill(BLUE))

        # 5. Border Changes
        # Lost border at top (since Blue is now next to Blue)
        # Creating a manual Cross (Red 'X')
        cross_line1 = Line(UP + LEFT, DOWN + RIGHT, color=RED, stroke_width=6).scale(0.15)
        cross_line2 = Line(UP + RIGHT, DOWN + LEFT, color=RED, stroke_width=6).scale(0.15)
        cross_mark = VGroup(cross_line1, cross_line2).move_to(initial_border)
        
        # New borders at Left, Right, Bottom (Blue center next to Red neighbors)
        new_l = Line(center_sq.get_left() + UP * 0.5, center_sq.get_left() + DOWN * 0.5, color=GREEN, stroke_width=6)
        new_r = Line(center_sq.get_right() + UP * 0.5, center_sq.get_right() + DOWN * 0.5, color=GREEN, stroke_width=6)
        new_b = Line(center_sq.get_bottom() + LEFT * 0.5, center_sq.get_bottom() + RIGHT * 0.5, color=GREEN, stroke_width=6)

        self.play(Create(cross_mark))
        self.play(Create(new_l), Create(new_r), Create(new_b))
        
        # 6. Reward Floating Text
        reward_val = Text("+2", color=GREEN).next_to(center_sq, UR, buff=0.1)
        self.play(Write(reward_val), reward_val.animate.shift(UP * 0.4))

        # 7. Manual Cumulative Reward Bar Chart
        # Building it with Axes and Rectangles to ensure compatibility
        ax = Axes(
            x_range=[0, 4, 1], 
            y_range=[0, 6, 2],
            x_length=4, 
            y_length=3,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=1).shift(DOWN * 0.5)
        
        ax_label = Text("Cumulative Reward", font_size=24).next_to(ax, UP)
        
        # Rewards over time: t=1: 1, t=2: 2, t=3: 4 (current step)
        v = [1, 2, 4]
        bars = VGroup()
        for i, val in enumerate(v):
            # Calculate height relative to y-axis scaling
            h = val * (3.0 / 6.0) 
            bar = Rectangle(
                width=0.6, 
                height=h, 
                fill_opacity=0.8, 
                fill_color=GREEN if i == 2 else BLUE,
                stroke_color=WHITE
            )
            bar.move_to(ax.c2p(i + 1, 0), aligned_edge=DOWN)
            bars.add(bar)

        self.play(Create(ax), Write(ax_label))
        self.play(Create(bars), run_time=1.5)
        self.wait(2)

        # End of sequence
        self.play(FadeOut(VGroup(grid_group, highlight, initial_border, cross_mark, new_l, new_r, new_b, reward_val, ax, ax_label, bars, formula)))
