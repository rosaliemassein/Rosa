from manim import *
import numpy as np

class Concept01BorderOptimization(Scene):
    def construct(self):
        # 1. Create the 4x4 grid of squares
        # The prompt asks for a 4x4 VGroup of Square objects.
        squares = VGroup(*[Square(side_length=0.8, fill_opacity=1, stroke_width=2) for _ in range(16)])
        squares.arrange_in_grid(rows=4, cols=4, buff=0.1)
        
        # 2. Track color state internally to simplify calculation
        # Initially, color the left half red (cols 0,1) and the right half blue (cols 2,3)
        current_colors = []
        for i in range(16):
            col = i % 4
            if col < 2:
                current_colors.append(RED)
            else:
                current_colors.append(BLUE)
            squares[i].set_fill(current_colors[i], opacity=1)
            squares[i].set_stroke(WHITE, width=1)
        
        self.add(squares)

        # 3. Objective and Formula labels
        formula = MathTex(r"B = \sum_{i,j \in \text{Neighbors}} [P_i \neq P_j]", color=WHITE)
        formula.to_edge(UP, buff=0.5)
        
        counter_label = Text("Adjacencies: ", font_size=32)
        counter_num = Integer(0).set_color(GREEN)
        counter_group = VGroup(counter_label, counter_num).arrange(RIGHT)
        counter_group.next_to(formula, DOWN, buff=0.3)
        
        self.add(formula, counter_group)

        # 4. Border line generation function
        # Draws thick green lines between adjacent pixels of different colors.
        def get_border_mobjects(grid, colors):
            lines = VGroup()
            count = 0
            # Horizontal adjacency check (left-right neighbors)
            for r in range(4):
                for c in range(3):
                    idx1 = r * 4 + c
                    idx2 = r * 4 + (c + 1)
                    if colors[idx1] != colors[idx2]:
                        # Vertical line between squares
                        p1 = grid[idx1].get_corner(UR) + DOWN * 0.05
                        p2 = grid[idx1].get_corner(DR) + UP * 0.05
                        lines.add(Line(p1, p2, color=GREEN, stroke_width=8))
                        count += 1
            # Vertical adjacency check (top-bottom neighbors)
            for r in range(3):
                for c in range(4):
                    idx1 = r * 4 + c
                    idx2 = (r + 1) * 4 + c
                    if colors[idx1] != colors[idx2]:
                        # Horizontal line between squares
                        p1 = grid[idx1].get_corner(DL) + RIGHT * 0.05
                        p2 = grid[idx1].get_corner(DR) + LEFT * 0.05
                        lines.add(Line(p1, p2, color=GREEN, stroke_width=8))
                        count += 1
            return lines, count

        # Initial border display
        current_border_lines, count_val = get_border_mobjects(squares, current_colors)
        counter_num.set_value(count_val)
        self.add(current_border_lines)

        # 5. Animate the flipping to a checkerboard pattern
        # Checkerboard: (row + col) % 2 == 0 ? RED : BLUE
        for i in range(16):
            r, c = divmod(i, 4)
            target_color = RED if (r + c) % 2 == 0 else BLUE
            
            # If the current color is not the checkerboard color, flip it
            if current_colors[i] != target_color:
                current_colors[i] = target_color
                
                # Dynamic update of borders and counter
                new_border_lines, new_count = get_border_mobjects(squares, current_colors)
                
                self.play(
                    squares[i].animate.set_fill(target_color),
                    counter_num.animate.set_value(new_count),
                    FadeOut(current_border_lines, run_time=0.1),
                    FadeIn(new_border_lines, run_time=0.1),
                    run_time=0.4
                )
                current_border_lines = new_border_lines
        
        self.wait(2)
