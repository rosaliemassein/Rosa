from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Grid
        grid = VGroup(*[
            Square(side_length=1.0, fill_opacity=1, stroke_width=2) 
            for _ in range(9)
        ]).arrange_in_grid(rows=3, cols=3, buff=0.1)
        
        # Initial Colors: Center(4) is Red. 
        # Neighbors: Top(1) is Blue, others (3, 5, 7) are Red.
        for i in range(9):
            grid[i].set_fill(RED)
        grid[1].set_fill(BLUE) # Top neighbor is Blue to create a border with Red center
        
        center_pixel = grid[4]
        neighbors = VGroup(grid[1], grid[3], grid[5], grid[7])
        
        # 2. Setup Label and Formula
        formula = MathTex(r"R_t = \text{Border}_{t} - \text{Border}_{t-1}").to_edge(UP)
        
        # 3. Setup Manual Bar Chart using Axes and Rectangles
        axes = Axes(
            x_range=[0, 4, 1],
            y_range=[0, 10, 2],
            x_length=4,
            y_length=3,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5).shift(DOWN)
        
        chart_label = Text("Cumulative Reward", font_size=24).next_to(axes, UP)
        
        # Initial bar (empty)
        bars = VGroup(
            Rectangle(width=0.5, height=0.1, fill_opacity=1, color=BLUE).move_to(axes.c2p(1, 0), aligned_edge=DOWN),
            Rectangle(width=0.5, height=0.1, fill_opacity=1, color=BLUE).move_to(axes.c2p(2, 0), aligned_edge=DOWN),
            Rectangle(width=0.5, height=0.1, fill_opacity=1, color=BLUE).move_to(axes.c2p(3, 0), aligned_edge=DOWN),
        )

        self.add(grid, formula, axes, chart_label, bars)
        self.play(Indicate(center_pixel, color=YELLOW))
        self.play(Create(SurroundingRectangle(neighbors, color=YELLOW, buff=0.05)))
        self.wait(0.5)

        # 4. Animate Flip and Reward Dynamics
        # Border between 4 and 1 is lost (Blue-Blue).
        # Borders between 4 and 3, 5, 7 are created (Blue-Red).
        
        # Define border lines
        lost_border_pos = Line(grid[4].get_corner(UL), grid[4].get_corner(UR)).shift(UP*0.05)
        new_border_l = Line(grid[4].get_corner(UL), grid[4].get_corner(DL)).shift(LEFT*0.05)
        new_border_r = Line(grid[4].get_corner(UR), grid[4].get_corner(DR)).shift(RIGHT*0.05)
        new_border_b = Line(grid[4].get_corner(DL), grid[4].get_corner(DR)).shift(DOWN*0.05)
        
        # Manual Cross for lost border
        red_x = VGroup(
            Line(lost_border_pos.get_start(), lost_border_pos.get_end(), color=RED),
            Line(lost_border_pos.get_end(), lost_border_pos.get_start(), color=RED)
        ).scale(0.2).move_to(lost_border_pos.get_center())
        # Adding a cross-like shape
        cross_line1 = Line(UP+LEFT, DOWN+RIGHT, color=RED, stroke_width=4).scale(0.1).move_to(lost_border_pos.get_center())
        cross_line2 = Line(UP+RIGHT, DOWN+LEFT, color=RED, stroke_width=4).scale(0.1).move_to(lost_border_pos.get_center())
        red_cross = VGroup(cross_line1, cross_line2)
        
        # Green highlights for new borders
        green_borders = VGroup(
            new_border_l.set_color(GREEN).set_stroke(width=6),
            new_border_r.set_color(GREEN).set_stroke(width=6),
            new_border_b.set_color(GREEN).set_stroke(width=6)
        )

        # Animation sequence
        self.play(center_pixel.animate.set_fill(BLUE))
        
        # Show new and lost borders
        self.play(
            Create(green_borders),
            Create(red_cross),
            run_time=1
        )
        
        # Reward text
        reward_text = MathTex(r"+2", color=GREEN).scale(1.5).next_to(center_pixel, UR, buff=0.1)
        self.play(Write(reward_text))
        
        # Update Chart bars manually
        new_bars = VGroup(
            Rectangle(width=0.5, height=1.0, fill_opacity=1, color=BLUE).move_to(axes.c2p(1, 0), aligned_edge=DOWN),
            Rectangle(width=0.5, height=2.0, fill_opacity=1, color=BLUE).move_to(axes.c2p(2, 0), aligned_edge=DOWN),
            Rectangle(width=0.5, height=3.0, fill_opacity=1, color=BLUE).move_to(axes.c2p(3, 0), aligned_edge=DOWN),
        )
        
        self.play(Transform(bars, new_bars))
        self.wait(1)

        # 5. Narrated Sequence Final State
        final_text = Text("Value-Driven Decision", font_size=32, color=YELLOW).next_to(grid, DOWN)
        self.play(Write(final_text))
        self.wait(3)
