from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Setup Formula and Axes
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}", color=BLUE).to_edge(UP)
        
        # Axes to show exponential growth
        # x-axis: Matrix Size (2 to 6)
        # y-axis: Log base 10 of states
        axes = Axes(
            x_range=[2, 6.5, 1],
            y_range=[0, 12, 2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True},
            x_axis_config={"numbers_to_include": range(2, 7)},
            y_axis_config={"numbers_to_include": range(0, 13, 2)},
        ).shift(RIGHT * 3 + DOWN * 0.5)
        
        x_label = axes.get_x_axis_label("N", edge=RIGHT, direction=DOWN)
        y_label = axes.get_y_axis_label(r"10^k \text{ States}", edge=UP, direction=LEFT).scale(0.8)
        
        self.play(Write(formula), Create(axes), Write(x_label), Write(y_label))

        # 2. Trackers for synchronized animation
        n_tracker = ValueTracker(2)

        # 3. Dynamic Grid Visualization
        grid_pos = LEFT * 3.5 + UP * 1.5
        
        def get_grid():
            n_val = n_tracker.get_value()
            n = int(n_val)
            # Adjust size so it doesn't get too huge
            size = 2.0 / n
            grid = VGroup()
            for _ in range(n * n):
                color = BLUE if np.random.rand() > 0.5 else RED
                grid.add(Square(side_length=size).set_fill(color, opacity=0.8).set_stroke(width=1))
            grid.arrange_in_grid(rows=n, cols=n, buff=0.02)
            grid.move_to(grid_pos)
            return grid

        grid = always_redraw(get_grid)
        grid_label = always_redraw(lambda: 
            Text(f"{int(n_tracker.get_value())}x{int(n_tracker.get_value())} Grid", font_size=24)
            .next_to(grid, UP)
        )

        # 4. Table Representation
        table_rect = Rectangle(height=3, width=2, color=WHITE).next_to(grid_pos + DOWN * 2.5, ORIGIN)
        table_label = always_redraw(lambda:
            Text(f"{2**(int(n_tracker.get_value())**2):,} states", font_size=20)
            .next_to(table_rect, DOWN)
        )

        # Scrolling rows simulation
        rows = VGroup(*[
            Line(LEFT, RIGHT).set_width(1.8).set_stroke(GREY, opacity=0.4)
            for _ in range(15)
        ]).arrange(DOWN, buff=0.2).move_to(table_rect.get_center())

        def update_rows(m, dt):
            # Speed increases as N increases
            speed = (n_tracker.get_value() ** 2) * 0.4
            m.shift(UP * dt * speed)
            for sub_m in m:
                # Wrap around logic
                if sub_m.get_y() > table_rect.get_top()[1]:
                    sub_m.set_y(table_rect.get_bottom()[1])

        rows.add_updater(update_rows)

        # 5. Graph Animation
        # log10(2^(x^2)) = x^2 * log10(2)
        graph = always_redraw(lambda: axes.plot(
            lambda x_val: (x_val**2) * np.log10(2),
            x_range=[2, n_tracker.get_value()],
            color=YELLOW
        ))

        self.add(grid, grid_label, table_rect, rows, table_label, graph)
        self.wait()

        # 6. The Growth Phase
        self.play(
            n_tracker.animate.set_value(5),
            run_time=5,
            rate_func=rate_functions.linear
        )
        self.wait(0.5)

        # 7. The Explosion Phase
        blur = Rectangle(height=3, width=2).set_fill(color=[BLUE, RED, YELLOW], opacity=0.7).move_to(table_rect)
        explosion_text = Text("STATE SPACE EXPLOSION", color=RED).scale(0.7).to_edge(DOWN)
        
        self.play(
            n_tracker.animate.set_value(6.5),
            FadeIn(blur),
            Write(explosion_text),
            run_time=3,
            rate_func=rate_functions.ease_in_quad
        )

        # Final visual shake
        self.play(Wiggle(VGroup(axes, grid, table_rect, blur)))
        self.wait(2)
