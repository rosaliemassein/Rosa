from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Formula
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}", color=YELLOW).to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Setup Axes for State Space Explosion
        # x-axis: Matrix Size (2 to 6)
        # y-axis: Possible States (log scale base 2). y_range [0, 40] represents exponents of 2.
        axes = Axes(
            x_range=[2, 6, 1],
            y_range=[0, 40, 10],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True},
            x_axis_config={"numbers_to_include": [2, 3, 4, 5, 6]},
            y_axis_config={
                "scaling": LogScaling(base=2),
            }
        ).shift(RIGHT * 2 + DOWN * 0.5)

        x_label = axes.get_x_axis_label(r"\text{Matrix Size } N", edge=DOWN, direction=DOWN).scale(0.7)
        y_label = axes.get_y_axis_label(r"\text{States}", edge=LEFT, direction=LEFT).scale(0.7)
        
        # 3. Plot the function. 
        # On a LogScaling(base=2) axis, plotting 2**(x**2) will automatically map to y=x^2 on the screen.
        def state_function(n_val):
            return 2**(n_val**2)

        graph = axes.plot(state_function, x_range=[2, 5], color=BLUE)
        graph_ext = axes.plot(state_function, x_range=[5, 6], color=BLUE)

        # 4. Table Representation
        table_title = Text("Q-Table Rows", font_size=24).to_edge(LEFT, buff=1).shift(UP * 2)
        
        # A simple visual representing 16 rows for a 2x2 grid
        rows_16 = VGroup(*[
            Rectangle(height=2.0/16, width=2, stroke_width=1, fill_opacity=0.5, color=GREY)
            for _ in range(16)
        ]).arrange(DOWN, buff=0).next_to(table_title, DOWN, buff=0.5)
        
        # Blur/Solid color representing 2^25 states for 5x5
        blur_rect = Rectangle(height=3, width=2, stroke_width=0).next_to(table_title, DOWN, buff=0.5)
        blur_rect.set_fill(color=BLUE, opacity=0.8)
        
        # Lines to simulate high-speed scrolling/blur
        scrolling_lines = VGroup(*[
            Line(LEFT, RIGHT, stroke_width=0.5, color=WHITE).set_width(2)
            for _ in range(40)
        ]).arrange(DOWN, buff=0.06).move_to(blur_rect)

        # 5. Animation Sequence
        self.play(Create(axes), Write(x_label), Write(y_label))
        self.play(Create(rows_16), Write(table_title))
        self.wait(1)

        # Animate table growth and initial curve
        # "As the grid size increases to 5x5..."
        self.play(
            Create(graph),
            Transform(rows_16, blur_rect),
            table_title.animate.set_text("Q-Table (5x5): 33M+ states"),
            run_time=3
        )

        # Show scrolling blur lines
        self.add(scrolling_lines)
        
        # Graph shoots off the top as states explode
        self.play(
            Create(graph_ext),
            axes.animate.shift(UP * 15),
            graph.animate.shift(UP * 15),
            formula.animate.shift(UP * 10),
            scrolling_lines.animate.shift(UP * 4),
            rate_func=rate_functions.exponential_speed_up,
            run_time=2
        )

        # Final Summary Text
        narration_text = Text(
            "The state space grows exponentially,\nmaking training time skyrocket.",
            font_size=28,
            color=RED,
            line_spacing=1.5
        ).to_edge(DOWN, buff=1)
        
        self.play(FadeIn(narration_text))
        self.wait(2)
