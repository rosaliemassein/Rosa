from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Formula Header
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP)
        self.add(formula)

        # 2. Axes Setup
        # We use a linear axis to represent log-scale values (Log2)
        # y-axis represents N^2 (which is log2 of total states)
        ax = Axes(
            x_range=[2, 6, 1],
            y_range=[0, 40, 10],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(RIGHT, buff=0.5)
        
        ax_labels = VGroup(
            Text("Matrix Size (N)", font_size=16).next_to(ax.x_axis, DOWN),
            Text("Possible States (Log Scale)", font_size=16).rotate(90*DEGREES).next_to(ax.y_axis, LEFT)
        )

        # Plot log2(states) = val^2
        graph = ax.plot(
            lambda val: val**2,
            x_range=[2, 5],
            color=BLUE
        )

        # 3. Table Visualization
        table_rect = Rectangle(width=3, height=4, color=WHITE).to_edge(LEFT, buff=0.7)
        table_title = Text("Q-Table Rows", font_size=24).next_to(table_rect, UP)
        
        # Create scrolling rows using lines
        rows = VGroup(*[
            Line(LEFT, RIGHT, stroke_width=1, color=GRAY).set_width(2.6)
            for _ in range(30)
        ]).arrange(DOWN, buff=0.15).move_to(table_rect)
        
        # Masks to hide rows outside the table container
        top_mask = Rectangle(width=4, height=3, fill_color=BLACK, fill_opacity=1, stroke_width=0).next_to(table_rect, UP, buff=0)
        bottom_mask = Rectangle(width=4, height=3, fill_color=BLACK, fill_opacity=1, stroke_width=0).next_to(table_rect, DOWN, buff=0)

        # Trackers for the animation state
        n_tracker = ValueTracker(2)
        scroll_speed = ValueTracker(0.5)

        # Updater to make rows scroll continuously
        def update_rows(mob, dt):
            speed = scroll_speed.get_value()
            mob.shift(UP * speed * dt)
            for line in mob:
                if line.get_center()[1] > table_rect.get_top()[1]:
                    line.move_to(table_rect.get_bottom() + UP * 0.05)

        rows.add_updater(update_rows)
        
        # Counter label that updates with N
        counter = always_redraw(lambda: 
            VGroup(
                Text(f"N = {int(n_tracker.get_value())}", font_size=24),
                Text(f"States: 2^{int(n_tracker.get_value()**2)}", font_size=20, color=YELLOW)
            ).arrange(DOWN, buff=0.1).next_to(table_rect, DOWN)
        )

        self.add(ax, ax_labels, table_rect, table_title, rows, top_mask, bottom_mask, counter)
        
        # 4. Animation Sequence: Growing the matrix size
        self.play(Create(graph), run_time=2)
        
        # Matrix grows from 2x2 to 5x5, rows scroll faster
        self.play(
            n_tracker.animate.set_value(5),
            scroll_speed.animate.set_value(15),
            run_time=5,
            rate_func=lambda t: t # explicit linear rate
        )

        # 5. Explosion Transition: 5x5 to 6x6
        # Solid blur to represent extreme state density
        blur = Rectangle(
            width=2.8, height=3.8, 
            fill_color=BLUE, 
            fill_opacity=0.7, 
            stroke_width=0
        ).move_to(table_rect)
        
        # Graph shoots off the top of the chart
        shoot_off = ax.plot(
            lambda val: val**2,
            x_range=[5, 6.2],
            color=BLUE
        )

        self.play(
            FadeIn(blur),
            scroll_speed.animate.set_value(60), # Hyper-speed scroll
            Create(shoot_off),
            n_tracker.animate.set_value(6),
            run_time=2,
            rate_func=lambda t: t**2 # explicit ease-in-quad rate
        )
        
        self.wait(2)
