from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 11, 1],
            y_range=[0, 5, 1],
            x_length=7,
            y_length=4,
            axis_config={"include_tip": True},
            tips=False
        ).to_edge(LEFT, buff=0.5)
        
        labels = axes.get_axis_labels(
            x_label="Number of Players",
            y_label="MSE"
        )

        # 2. Data for curves
        x_vals = np.arange(1, 11)
        # KNN: dipping then rising sharply (overfitting)
        y_knn = [3.5, 2.0, 1.5, 1.8, 2.5, 3.5, 5.0, 7.0, 10.0, 14.0] 
        # Lasso: steady decline as it ignores noise
        y_lasso = [3.8, 3.2, 2.8, 2.5, 2.3, 2.1, 2.0, 1.9, 1.85, 1.8]

        # Manually constructing graphs to avoid potential version-specific plotting issues
        knn_points = [axes.c2p(x, min(y, 4.8)) for x, y in zip(x_vals, y_knn)]
        knn_graph = VMobject(color=RED, stroke_width=4).set_points_as_corners(knn_points)
        knn_label = Text("KNN", color=RED, font_size=24).next_to(knn_graph, UR, buff=-0.5)

        lasso_points = [axes.c2p(x, y) for x, y in zip(x_vals, y_lasso)]
        lasso_graph = VMobject(color=GREEN, stroke_width=4).set_points_as_corners(lasso_points)
        lasso_label = Text("Lasso/Ridge", color=GREEN, font_size=24).next_to(lasso_graph, RIGHT, buff=0.2)

        self.play(Create(axes), Write(labels))
        self.play(Create(knn_graph), Write(knn_label))
        self.wait(0.5)
        self.play(Create(lasso_graph), Write(lasso_label))
        self.wait(1)

        # 3. Formula and Feature Selection visualization
        # Shift the plot to make room for the penalty demo
        self.play(
            VGroup(axes, labels, knn_graph, knn_label, lasso_graph, lasso_label).animate.scale(0.7).to_edge(UL)
        )

        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            color=WHITE
        ).to_edge(UP).shift(RIGHT * 2)
        
        self.play(Write(formula))
        
        # 4. Coefficients Bar Chart to visualize Shrinkage
        num_coeffs = 40 
        np.random.seed(42)
        initial_weights = np.random.uniform(-1.5, 1.5, num_coeffs)
        
        lambda_tracker = ValueTracker(0)
        
        def get_bars():
            l_val = lambda_tracker.get_value()
            bars = VGroup()
            for i in range(num_coeffs):
                w = initial_weights[i]
                # Lasso penalty effect: soft thresholding
                # Coefficient shrinks towards zero and stays at zero
                new_h = np.sign(w) * max(0, np.abs(w) - l_val)
                bar = Line(
                    start=[i * 0.12, 0, 0],
                    end=[i * 0.12, new_h, 0],
                    stroke_width=3,
                    color=BLUE if abs(new_h) > 0.001 else GRAY
                )
                bars.add(bar)
            # Center the bar group and position it on the right
            return bars.center().to_edge(DOWN, buff=1.0).shift(RIGHT * 2.5)

        coeff_bars = always_redraw(get_bars)
        coeff_title = Text("Model Coefficients", font_size=20).next_to(coeff_bars, UP, buff=0.5)

        self.play(Create(coeff_bars), Write(coeff_title))
        self.wait(1)

        # 5. Animate Lambda Increase and show coefficients vanishing
        lambda_display_label = MathTex(r"\lambda = ", font_size=36).next_to(formula, DOWN, buff=0.5)
        lambda_val_text = always_redraw(lambda: 
            DecimalNumber(lambda_tracker.get_value(), num_decimal_places=2)
            .next_to(lambda_display_label, RIGHT)
        )

        self.play(Write(lambda_display_label), Write(lambda_val_text))
        
        # Use rate_functions.linear to ensure identifier exists in all Manim environments
        self.play(
            lambda_tracker.animate.set_value(1.6), 
            run_time=5, 
            rate_func=rate_functions.linear
        )
        self.wait(2)

        # Cleanup
        self.play(
            FadeOut(axes, labels, knn_graph, knn_label, lasso_graph, lasso_label),
            FadeOut(formula, lambda_display_label, lambda_val_text, coeff_bars, coeff_title)
        )
        self.wait(1)
