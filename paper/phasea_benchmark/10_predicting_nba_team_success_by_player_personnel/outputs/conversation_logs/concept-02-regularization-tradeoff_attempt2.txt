from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 10, 2],
            x_length=7,
            y_length=4,
            axis_config={"include_tip": True},
            tips=False
        ).to_edge(UP, buff=1.0).shift(LEFT * 2)

        x_label = axes.get_x_axis_label("Num Players", edge=DOWN, direction=DOWN, buff=0.2).scale(0.6)
        y_label = axes.get_y_axis_label("MSE", edge=LEFT, direction=LEFT, buff=0.2).scale(0.6).rotate(90 * DEGREES)
        
        # 2. Plot Curves using named functions to avoid static analysis issues
        def knn_path(t):
            return 0.3 * (t - 3)**2 + 1.5
            
        def lasso_path(t):
            return 7 / (np.sqrt(t) + 0.5)

        knn_curve = axes.plot(knn_path, x_range=[1, 8], color=RED)
        knn_label = Tex("KNN", color=RED).scale(0.6).next_to(knn_curve.get_end(), UR, buff=0.1)

        lasso_curve = axes.plot(lasso_path, x_range=[1, 9.5], color=GREEN)
        lasso_label = Tex("Lasso / Ridge", color=GREEN).scale(0.6).next_to(lasso_curve.get_end(), RIGHT, buff=0.1)

        self.play(Create(axes), Write(x_label), Write(y_label))
        self.play(Create(knn_curve), Write(knn_label))
        self.play(Create(lasso_curve), Write(lasso_label))
        self.wait(1)

        # 3. Lasso Formula
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            color=WHITE
        ).scale(0.9).to_edge(RIGHT, buff=1).shift(UP * 2)
        
        formula_box = SurroundingRectangle(formula, color=YELLOW, buff=0.2)
        
        self.play(Write(formula))
        self.play(Create(formula_box))
        self.wait(1)

        # 4. Coefficients Bar Chart (Simulating 150 weights)
        lambda_tracker = ValueTracker(0)
        
        np.random.seed(42)
        initial_weights = np.random.uniform(0.1, 2.5, 150)
        
        bars = VGroup(*[
            Rectangle(
                width=0.03, 
                height=w, 
                fill_color=BLUE, 
                fill_opacity=0.8, 
                stroke_width=0
            ) for w in initial_weights
        ]).arrange(RIGHT, buff=0.01).to_edge(DOWN, buff=0.5)

        bar_label = Tex("Coefficients", color=WHITE).scale(0.7).next_to(bars, UP, buff=0.2)
        
        lambda_display = MathTex(r"\lambda = ", color=YELLOW).scale(0.8).next_to(formula_box, DOWN, buff=0.5)
        lambda_value = DecimalNumber(0, color=YELLOW).scale(0.8).next_to(lambda_display, RIGHT)
        
        def update_decimal(mob):
            mob.set_value(lambda_tracker.get_value())
        lambda_value.add_updater(update_decimal)

        self.play(Create(bars), Write(bar_label))
        self.play(Write(lambda_display), Write(lambda_value))

        # 5. Animate the shrinking and vanishing coefficients (Lasso penalty)
        def update_bars(m):
            current_l = lambda_tracker.get_value()
            for i, bar in enumerate(m):
                # Lasso soft-thresholding logic
                val = initial_weights[i] - current_l
                new_h = val if val > 0.001 else 0.001
                bar.set_height(new_h, stretch=True, about_edge=DOWN)
                if new_h <= 0.01:
                    bar.set_opacity(0)
                else:
                    bar.set_opacity(0.8)

        bars.add_updater(update_bars)
        
        self.play(
            lambda_tracker.animate.set_value(2.0),
            run_time=5,
            rate_func=rate_functions.linear
        )
        self.wait(2)

        # Fade out everything
        self.play(FadeOut(Group(*self.mobjects)))
