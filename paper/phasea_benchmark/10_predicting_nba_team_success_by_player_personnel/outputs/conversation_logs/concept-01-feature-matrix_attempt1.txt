from manim import *
import random

class ConceptMatrix(Scene):
    def construct(self):
        # 1. Setup Player Silhouettes (15 players)
        players = VGroup(*[
            VGroup(
                Circle(radius=0.1, color=GRAY, fill_opacity=1),
                Rectangle(width=0.2, height=0.2, color=GRAY, fill_opacity=1)
            ).arrange(DOWN, buff=0.05)
            for _ in range(15)
        ]).arrange(DOWN, buff=0.15).to_edge(LEFT, buff=1)
        
        self.play(Create(players))
        self.wait(0.5)

        # 2. Minutes Played Filter
        filter_rect = Rectangle(width=1, height=0.4, color=YELLOW).move_to(players[0])
        self.play(FadeIn(filter_rect))
        
        # Animate scan down to the 10th player
        self.play(filter_rect.animate.move_to(players[9]), run_time=1.5)
        
        top_10 = players[:10]
        others = players[10:]
        
        self.play(
            top_10.animate.set_color(BLUE),
            others.animate.set_opacity(0.2),
            FadeOut(filter_rect)
        )
        self.wait(0.5)

        # 3. Create Feature Matrix (10 rows x 15 columns)
        matrix_cells = VGroup()
        for r in range(10):
            row = VGroup(*[
                Square(side_length=0.3, stroke_width=1, color=WHITE)
                for _ in range(15)
            ]).arrange(RIGHT, buff=0.05)
            matrix_cells.add(row)
        
        matrix_cells.arrange(DOWN, buff=0.05).shift(RIGHT * 2.5)
        
        # Stat Labels for Columns
        stat_names = ["PTS", "REB", "AST", "STL", "BLK", "TOV", "PF", "FG%", "FT%", "3P%", "+/-", "PER", "TS%", "USG%", "PIE"]
        stat_labels = VGroup(*[
            Text(stat, font_size=10).next_to(matrix_cells[0][i], UP, buff=0.2).rotate(45*DEGREES)
            for i, stat in enumerate(stat_names)
        ])

        # 4. Animate Players sliding into Matrix Rows
        self.play(
            ReplacementTransform(top_10.copy(), matrix_cells),
            Write(stat_labels),
            run_time=1.5
        )
        self.wait(0.5)

        # Fill grid cells with "data" intensities
        fill_animations = []
        for row in matrix_cells:
            for cell in row:
                fill_animations.append(cell.animate.set_fill(BLUE, opacity=random.uniform(0.1, 1.0)))
        
        self.play(*fill_animations, run_time=1.5)
        self.wait(1)

        # 5. Collapse 2D Grid into a single 1D Vector (150 elements)
        # Use MathTex for the formula reference
        formula = MathTex("n = m \\times p", font_size=36).to_corner(UL)
        formula_calc = MathTex("150 = 10 \\times 15", font_size=30).next_to(formula, DOWN, aligned_edge=LEFT)
        
        self.play(Write(formula))
        
        # Flatten all cells into one group
        all_squares = VGroup(*[cell for row in matrix_cells for cell in row])
        
        self.play(
            all_squares.animate.arrange(RIGHT, buff=0).scale_to_fit_width(FRAME_WIDTH - 1).move_to(ORIGIN),
            FadeOut(stat_labels),
            FadeOut(players),
            FadeOut(others),
            Write(formula_calc),
            run_time=2
        )
        
        vector_text = Text("150-Element Team Vector", font_size=24).next_to(all_squares, UP)
        self.play(FadeIn(vector_text))
        
        # Highlight the final vector
        self.play(all_squares.animate.set_color(YELLOW), run_time=0.5)
        self.play(all_squares.animate.set_color(BLUE), run_time=0.5)
        
        self.wait(2)
