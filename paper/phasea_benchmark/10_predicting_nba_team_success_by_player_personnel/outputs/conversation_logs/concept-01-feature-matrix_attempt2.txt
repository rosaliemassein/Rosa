from manim import *
import numpy as np

class FeatureMatrix(Scene):
    def construct(self):
        # 1. Create player silhouettes (15 dots/circles)
        players = VGroup(*[
            Circle(radius=0.15, fill_opacity=0.6, color=BLUE) 
            for _ in range(15)
        ]).arrange(DOWN, buff=0.2).to_edge(LEFT, buff=1)
        
        player_labels = VGroup(*[
            Text(f"P{i+1}", font_size=14).next_to(players[i], LEFT, buff=0.2)
            for i in range(15)
        ])
        
        self.play(Create(players), FadeIn(player_labels))
        self.wait(0.5)

        # 2. Minutes Played Filter
        filter_line = Line(LEFT, RIGHT, color=YELLOW).set_width(2)
        filter_line.next_to(players[0], UP, buff=0.1)
        filter_text = Text("Minutes Played Filter", font_size=20, color=YELLOW).next_to(filter_line, UP)
        
        self.play(FadeIn(filter_line), FadeIn(filter_text))
        # Animate filter passing over players down to the 10th player
        self.play(
            filter_line.animate.move_to(players[9].get_center() + DOWN*0.2),
            run_time=2
        )
        
        # Highlight top 10
        top_10 = players[:10]
        others = players[10:]
        others_labels = player_labels[10:]
        
        self.play(
            top_10.animate.set_color(GOLD).scale(1.2),
            FadeOut(others),
            FadeOut(others_labels),
            FadeOut(filter_line),
            FadeOut(filter_text)
        )
        self.wait(0.5)

        # 3. Create 10x15 Matrix
        matrix_group = VGroup()
        rows = 10
        cols = 15
        cell_size = 0.35
        
        # Colors for interpolation
        c1 = "#0000FF" # Pure Blue
        c2 = "#ADD8E6" # Light Blue
        
        for r in range(rows):
            for c in range(cols):
                val = np.random.random()
                color = interpolate_color(color_from=c1, color_to=c2, alpha=val)
                sq = Square(side_length=cell_size, fill_color=color, fill_opacity=0.8, stroke_width=1)
                matrix_group.add(sq)
        
        matrix_group.arrange_in_grid(rows=rows, cols=cols, buff=0.05).shift(RIGHT * 2)
        
        # 4. Slide top 10 players into rows
        # We transform the player dots into the first column of the matrix
        self.play(
            ReplacementTransform(top_10, matrix_group[::cols]),
            player_labels[:10].animate.next_to(matrix_group, LEFT, buff=0.2)
        )
        self.play(FadeIn(matrix_group[1:])) # Fade in the rest of the cells
        self.wait(0.5)

        # 5. Column Labels
        stat_names = ["PTS", "REB", "AST", "STL", "BLK", "TOV", "PF", "MIN", "FGM", "FGA", "3PM", "3PA", "FTM", "FTA", "PIE"]
        col_labels = VGroup(*[
            Text(stat_names[i], font_size=12).next_to(matrix_group[i], UP, buff=0.2)
            for i in range(cols)
        ])
        self.play(FadeIn(col_labels))
        self.wait(1)

        # 6. Formula
        formula = MathTex("n = m \\times p", font_size=36).to_edge(UP)
        self.play(Write(formula))
        self.wait(1)

        # 7. Collapse into 1D Vector (150 elements)
        # Create a single long horizontal vector
        vector_cells = VGroup(*[
            Square(side_length=0.08, fill_color=cell.get_fill_color(), fill_opacity=0.8, stroke_width=0.1)
            for cell in matrix_group
        ]).arrange(RIGHT, buff=0.01).move_to(DOWN * 2)
        
        # Ensure it fits the frame
        vector_cells.set_width(12)

        self.play(
            ReplacementTransform(matrix_group, vector_cells),
            FadeOut(col_labels),
            FadeOut(player_labels[:10]),
            formula.animate.scale(1.2).next_to(vector_cells, UP, buff=0.5)
        )
        
        label_150 = Text("150-Element Feature Vector", font_size=20).next_to(vector_cells, DOWN, buff=0.3)
        self.play(Write(label_150))
        self.wait(2)
