from manim import *

class PlayerDepthImportance(Scene):
    def construct(self):
        # 1. Setup Voiceover and Data
        voice_text = (
            "The models reveal a hidden hierarchy in team success. While the top player's "
            "Impact Estimate is the strongest predictor, the 'star' alone isn't enough. "
            "Interestingly, the statistics of the fifth and eighth players carry surprising weight. "
            "This suggests that a team's floor is set by the quality of its fifth starter and its "
            "final rotation player, proving that depth is just as measurable as stardom."
        )
        
        # Coefficients representing importance (1st, 2nd, 5th, and 8th are highlighted)
        coeffs = [0.85, 0.65, 0.22, 0.18, 0.52, 0.15, 0.12, 0.42, 0.08, 0.05]
        
        # 2. Create Player Icons (Circles with Rank Labels)
        players = VGroup()
        for i in range(1, 11):
            circle = Circle(radius=0.3, color=WHITE, fill_opacity=0.2)
            label = Text(str(i), font_size=20).move_to(circle.get_center())
            player_unit = VGroup(circle, label)
            players.add(player_unit)
        
        players.arrange(RIGHT, buff=0.4).to_edge(DOWN, buff=1.8)

        # 3. Create Bars (Vertical bars for Average Coefficient)
        bars = VGroup()
        for i, val in enumerate(coeffs):
            h = val * 4.5 + 0.1  # Scaling factor for visual height
            bar = Rectangle(
                width=0.4, 
                height=h, 
                fill_opacity=0.6, 
                stroke_width=1, 
                color=WHITE
            )
            # Align the bottom of the bar to the top of the player icon
            bar.next_to(players[i], UP, buff=0.1, aligned_edge=DOWN)
            bars.add(bar)

        # 4. Formula
        formula = MathTex(
            r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})",
            font_size=32
        ).to_edge(UP, buff=0.7)

        # 5. Narrative Text Setup
        narr_label = Text(voice_text, font_size=17, line_spacing=1.5)
        # Manually create a background rectangle for the text
        narr_bg = Rectangle(
            width=13.5, 
            height=1.8, 
            fill_color=BLACK, 
            fill_opacity=0.8, 
            stroke_width=0
        ).to_edge(DOWN, buff=0)
        narr_label.move_to(narr_bg.get_center())

        # 6. Animation Execution
        # Step 1: Show players
        self.play(Create(players), run_time=1.2)
        
        # Step 2: Bars grow and Formula appears
        # We use a simple Create animation which looks like growing for vertical rectangles
        self.play(
            *[Create(bar) for bar in bars],
            Write(formula),
            run_time=2,
            rate_func=smooth
        )
        self.wait(1)

        # Step 3: Highlight Stars (1st and 2nd)
        star_bars = VGroup(bars[0], bars[1])
        star_icons = VGroup(players[0][0], players[1][0]) # The circles
        self.play(
            star_bars.animate.set_color(GOLD).set_fill(GOLD, opacity=0.9),
            star_icons.animate.set_color(GOLD),
            run_time=1
        )
        # Pulse effect using scale
        self.play(star_bars.animate.scale(1.1, about_edge=DOWN), run_time=0.4)
        self.play(star_bars.animate.scale(1/1.1, about_edge=DOWN), run_time=0.4)
        self.wait(0.5)

        # Step 4: Highlight Depth (5th and 8th)
        depth_bars = VGroup(bars[4], bars[7])
        depth_icons = VGroup(players[4][0], players[7][0])
        self.play(
            depth_bars.animate.set_color(BLUE).set_fill(BLUE, opacity=0.9),
            depth_icons.animate.set_color(BLUE),
            run_time=1
        )
        
        # Pulsing blue light (2 pulses)
        for _ in range(2):
            self.play(depth_bars.animate.scale(1.15, about_edge=DOWN), run_time=0.35)
            self.play(depth_bars.animate.scale(1/1.15, about_edge=DOWN), run_time=0.35)

        # Step 5: Narration display
        self.play(FadeIn(narr_bg), Write(narr_label), run_time=6)
        
        # Emphasis on specific formula parts
        self.play(
            formula[0][6:16].animate.set_color(GOLD),  # beta_1(1st_PIE)
            formula[0][17:27].animate.set_color(BLUE), # beta_5(5th_PIE)
            formula[0][31:41].animate.set_color(BLUE), # beta_8(8th_PIE)
            run_time=2
        )
        
        self.wait(4)

if __name__ == "__main__":
    scene = PlayerDepthImportance()
    scene.render()
