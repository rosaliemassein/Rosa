from manim import *

class GeneticWinRateSelection(Scene):
    def construct(self):
        # 1. Formula and Title
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)")
        formula.to_edge(UP, buff=0.5)
        self.play(Write(formula))

        # 2. Central Parent Node
        parent = VGroup(
            Circle(radius=0.7, color=BLUE, fill_opacity=0.2),
            Text("Parent P_t", font_size=24)
        ).shift(UP * 1.0)
        
        self.play(Create(parent))
        self.wait(0.5)

        # 3. Three Child Nodes Branching Out
        child_targets = [LEFT * 4 + DOWN * 1, DOWN * 1, RIGHT * 4 + DOWN * 1]
        children = VGroup()
        arrows = VGroup()

        for i, pos in enumerate(child_targets):
            child = VGroup(
                Circle(radius=0.6, color=WHITE, fill_opacity=0.1),
                Text(f"Child {i+1}", font_size=20)
            ).move_to(pos)
            arrow = Arrow(parent.get_bottom(), child.get_top(), buff=0.1)
            children.add(child)
            arrows.add(arrow)

        self.play(
            LaggedStart(*[Create(a) for a in arrows], lag_ratio=0.2),
            LaggedStart(*[Create(c) for c in children], lag_ratio=0.2)
        )

        # 4. Bar Charts (Win Rates) represented by Rectangles
        # We define heights for the bars
        win_rate_values = [0.3, 0.9, 0.5] # Child 2 is the winner
        bars = VGroup()
        bar_labels = VGroup()
        
        for i, (child, val) in enumerate(zip(children, win_rate_values)):
            # Create a base for the bar
            base_line = Line(LEFT*0.5, RIGHT*0.5).next_to(child, DOWN, buff=0.3)
            # Create the bar itself
            bar = Rectangle(
                width=0.6, 
                height=0.1, # Start small
                fill_opacity=0.8, 
                color=BLUE_C
            ).move_to(base_line.get_center(), aligned_edge=DOWN)
            
            label = Text("Win Rate", font_size=16).next_to(base_line, DOWN, buff=0.1)
            
            bars.add(bar)
            bar_labels.add(VGroup(base_line, label))

        self.play(FadeIn(bar_labels))
        
        # 5. Animate bars growing to their specific heights
        growth_animations = []
        for bar, val in zip(bars, win_rate_values):
            growth_animations.append(bar.animate.stretch_to_fit_height(val * 2.0, about_edge=DOWN))
        
        self.play(*growth_animations, run_time=2)
        self.wait(1)

        # 6. Highlight the child with the highest bar in gold
        winner_idx = 1
        winner_child = children[winner_idx]
        winner_bar = bars[winner_idx]
        
        highlight_rect = SurroundingRectangle(VGroup(winner_child, winner_bar), color=GOLD, buff=0.2)
        
        self.play(
            Create(highlight_rect),
            winner_bar.animate.set_color(GOLD),
            winner_child[0].animate.set_color(GOLD)
        )
        self.play(Indicate(winner_child))
        self.wait(1)

        # 7. Fade out parent and other children
        losers = VGroup(
            parent,
            arrows,
            children[0], bars[0], bar_labels[0],
            children[2], bars[2], bar_labels[2],
            bar_labels[1], # the label for the winner bar
            highlight_rect
        )
        
        self.play(FadeOut(losers))

        # 8. Move the gold child to center to become the new parent
        new_parent_label = Text("New Parent P_{t+1}", font_size=30, color=GOLD)
        
        self.play(
            winner_child.animate.move_to(UP * 1.0),
            FadeOut(winner_bar)
        )
        
        # Replace child text with "New Parent"
        self.play(Transform(winner_child[1], new_parent_label.move_to(winner_child[1].get_center())))
        
        # 9. Final text to close out the "Evolution" loop
        evolution_text = Text("Evolutionary Gradient for Victory", font_size=32, color=GOLD).shift(DOWN * 2)
        self.play(Write(evolution_text))
        
        self.wait(2)
