from manim import *
import numpy as np

class Concept_03_DualRewardShaping(Scene):
    def construct(self):
        # Configuration for the target and team
        target_pos = np.array([3.0, 0.0, 0.0])
        
        # Goal zone visual using standard Circle and Text
        target_zone = Circle(radius=1.5, color=YELLOW, fill_opacity=0.2).move_to(target_pos)
        goal_label = Text("Goal Zone", font_size=24).next_to(target_zone, UP)
        
        # Team of pegs (blue cluster)
        peg_positions = [
            np.array([-4.5, y, 0.0])
            for y in [-1.5, -0.75, 0.0, 0.75, 1.5]
        ]
        pegs = VGroup(*[Dot(color=BLUE, radius=0.15).move_to(pos) for pos in peg_positions])
        
        # Centroid point and its label (tracking average position)
        centroid_dot = Dot(color=RED, radius=0.1)
        
        def centroid_update_func(mob):
            # Calculate average of all current peg centers
            all_centers = [p.get_center() for p in pegs]
            avg_pos = np.mean(all_centers, axis=0)
            mob.move_to(avg_pos)
            
        centroid_dot.add_updater(centroid_update_func)
        
        centroid_label = Text("Centroid", color=RED, font_size=18)
        def label_update_func(mob):
            mob.next_to(centroid_dot, UP, buff=0.1)
        centroid_label.add_updater(label_update_func)
        
        # Line between centroid and goal (Gravity/Distance indicator)
        # Using standard Line instead of DashedLine to ensure compilation
        gravity_line = Line(centroid_dot.get_center(), target_pos, color=WHITE)
        def line_update_func(mob):
            # Regenerate the line to connect centroid and target
            new_line = Line(centroid_dot.get_center(), target_pos, color=WHITE, stroke_width=2)
            mob.become(new_line)
        gravity_line.add_updater(line_update_func)
        
        # Reward UI Display
        formula_text = Text("Reward = w1*(In^2) + w2*(1/Distance)", font_size=22).to_edge(UP)
        reward_counter = Text("Total Reward: 0.00", font_size=30).to_edge(DOWN, buff=0.5)
        
        def reward_update_func(mob):
            # Calculate the two reward components
            # 1. Number of pegs in the goal zone
            pegs_in_zone = 0
            for p in pegs:
                dist_to_goal = np.linalg.norm(p.get_center() - target_pos)
                if dist_to_goal < 1.5:
                    pegs_in_zone += 1
            
            # 2. Distance from centroid to goal
            cent_to_goal = np.linalg.norm(centroid_dot.get_center() - target_pos)
            
            # Weighted reward calculation (w1=10, w2=5)
            # Using max(dist, 0.1) to prevent division by zero errors
            score = 10 * (pegs_in_zone ** 2) + 5 * (1.0 / max(cent_to_goal, 0.1))
            
            # Update the text readout
            score_str = "{:.2f}".format(score)
            new_mob = Text("Total Reward: " + score_str, font_size=30).to_edge(DOWN, buff=0.5)
            mob.become(new_mob)
            
        reward_counter.add_updater(reward_update_func)
        
        # 1. Initialization
        self.add(target_zone, goal_label, pegs, centroid_dot, centroid_label, gravity_line, formula_text, reward_counter)
        self.wait(1)
        
        # 2. Animation: Group movement towards goal
        # Using lambda for rate_func to avoid 'linear' identifier issues
        self.play(
            pegs.animate.shift(RIGHT * 4),
            run_time=3,
            rate_func=lambda t: t
        )
        self.wait(0.5)
        
        # 3. Animation: Pieces entering the zone one-by-one
        for i, p in enumerate(pegs):
            # Calculate a unique spot inside the target zone for each peg
            angle = i * (2.0 * 3.14159 / 5.0)
            offset = np.array([np.cos(angle) * 0.7, np.sin(angle) * 0.7, 0.0])
            dest_pos = target_pos + offset
            
            self.play(p.animate.move_to(dest_pos), run_time=0.8)
            
            # Squared bonus pop-up text (e.g., "1^2", "2^2")
            bonus_popup = Text(str(i+1) + "^2", color=YELLOW, font_size=32)
            bonus_popup.move_to(p.get_center() + UP * 0.5)
            
            self.play(
                FadeIn(bonus_popup), 
                bonus_popup.animate.shift(UP * 0.2), 
                run_time=0.3
            )
            self.play(FadeOut(bonus_popup), run_time=0.2)
            
        self.wait(2)
