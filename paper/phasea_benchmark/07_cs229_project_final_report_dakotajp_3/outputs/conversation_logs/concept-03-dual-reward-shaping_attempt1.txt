from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Create the game board and target zone
        board = Rectangle(width=10, height=6, color=WHITE)
        target_zone = Rectangle(width=2, height=4, color=YELLOW, fill_opacity=0.2).to_edge(RIGHT, buff=1)
        target_label = Text("Target Zone", color=YELLOW, font_size=24).next_to(target_zone, UP)
        self.add(board, target_zone, target_label)

        # 2. Add pegs to the board in a cluster
        pegs = VGroup(*[
            Dot(color=BLUE, radius=0.15)
            for _ in range(6)
        ])
        for i, peg in enumerate(pegs):
            peg.move_to([-4 + (i % 2) * 0.5, -1 + (i // 2) * 0.8, 0])
        self.add(pegs)

        # 3. Setup Centroid and Target Marker
        def get_centroid_pos():
            return np.mean([p.get_center() for p in pegs], axis=0)

        centroid = Dot(get_centroid_pos(), color=RED)
        centroid_text = Text("Centroid", color=RED, font_size=20)
        
        # Line from Centroid to Target Center
        target_center_point = target_zone.get_center()
        dist_line = Line(centroid.get_center(), target_center_point, color=WHITE)
        
        self.add(centroid, centroid_text, dist_line)

        # Updaters for movement
        centroid.add_updater(lambda m: m.move_to(get_centroid_pos()))
        centroid_text.add_updater(lambda m: m.next_to(centroid, UP, buff=0.1))
        dist_line.add_updater(lambda m: m.put_start_and_end_on(centroid.get_center(), target_center_point))

        # 4. Reward Display and Formula
        formula = MathTex(
            r"R = w_1(p_{in})^2 + w_2(dist)^{-1}",
            font_size=30
        ).to_edge(UP)
        
        reward_tex = MathTex("Reward: 0.00", color=GREEN).to_edge(DOWN, buff=0.5)
        self.add(formula, reward_tex)

        # 5. Animation: Pegs moving forward
        pegs_in_count = 0
        for i in range(len(pegs)):
            # Move peg into target zone
            target_pos = target_zone.get_center() + [
                np.random.uniform(-0.4, 0.4),
                np.random.uniform(-1.2, 1.2),
                0
            ]
            
            self.play(pegs[i].animate.move_to(target_pos), run_time=0.8)
            
            # Check if peg is in target (simple x-coordinate check)
            if pegs[i].get_center()[0] > target_zone.get_left()[0]:
                pegs_in_count += 1
                
                # Show squared bonus popup
                bonus_popup = MathTex(f"{pegs_in_count}^2", color=GREEN, font_size=36)
                bonus_popup.next_to(pegs[i], UP, buff=0.2)
                
                # Update Reward Text manually
                dist = np.linalg.norm(get_centroid_pos() - target_center_point)
                current_reward = (pegs_in_count**2) + (5.0 / (dist + 0.5))
                new_reward_tex = MathTex(f"Reward: {current_reward:.2f}", color=GREEN).to_edge(DOWN, buff=0.5)
                
                self.play(
                    FadeIn(bonus_popup, shift=UP*0.3),
                    reward_tex.become(new_reward_tex),
                    run_time=0.4
                )
                self.play(FadeOut(bonus_popup), run_time=0.2)

        # Finish
        self.wait(2)
        centroid.clear_updaters()
        centroid_text.clear_updaters()
        dist_line.clear_updaters()
