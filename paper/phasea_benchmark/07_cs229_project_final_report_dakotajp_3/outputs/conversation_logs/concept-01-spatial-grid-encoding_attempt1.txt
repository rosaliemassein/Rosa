from manim import *
import numpy as np

class ConceptGridEncoding(Scene):
    def construct(self):
        # 1. Hexagonal Board Setup (Radius 1)
        # Formula: n = 4 * radius + 1 = 5
        board_radius = 1
        n = 5
        hex_board = VGroup()
        # Axial coordinate system for hexagonal grid
        for q in range(-board_radius, board_radius + 1):
            r1 = max(-board_radius, -q - board_radius)
            r2 = min(board_radius, -q + board_radius)
            for r in range(r1, r2 + 1):
                # Hex to 2D Cartesian
                x = 0.7 * (3/2 * q)
                y = 0.7 * (np.sqrt(3)/2 * q + np.sqrt(3) * r)
                circle = Circle(radius=0.25, color=WHITE, stroke_width=3)
                circle.move_to([x, y, 0])
                hex_board.add(circle)
        
        self.play(Create(hex_board))
        self.wait(1)

        # 2. Simulate tilting into 3D Perspective (using 2D transformation)
        # Shear and scale to give perspective illusion
        tilted_matrix = [[1, 0.5], [0, 0.5]]
        self.play(
            hex_board.animate.apply_matrix(tilted_matrix).scale(0.8).shift(DOWN * 1.5),
            run_time=1.5
        )
        self.wait(0.5)

        # 3. Create three square grid layers (5x5)
        def create_layer(color, title_text, highlight_indices):
            layer_group = VGroup()
            grid = VGroup()
            for i in range(n * n):
                rect = Square(side_length=0.4, stroke_width=1, stroke_color=WHITE)
                if i in highlight_indices:
                    rect.set_fill(color, opacity=0.7)
                else:
                    rect.set_fill(color, opacity=0.1)
                grid.add(rect)
            grid.arrange_in_grid(rows=n, cols=n, buff=0.05)
            
            # Apply same "perspective" matrix to the square grid
            grid.apply_matrix(tilted_matrix)
            
            label = Text(title_text, font_size=18, color=color)
            label.next_to(grid, LEFT, buff=0.2)
            
            layer_group.add(grid, label)
            return layer_group

        # Sample indices for "pieces"
        layer1 = create_layer(BLUE, "Current Player", [6, 7, 12])
        layer2 = create_layer(GREEN, "Target Zone", [17, 18, 22, 23])
        layer3 = create_layer(RED, "Opponents", [2, 3, 8])

        # Position layers in a 2D stack to simulate vertical 3D floating
        layer1.move_to(ORIGIN + DOWN * 1.0)
        layer2.move_to(layer1.get_center() + UP * 1.2 + RIGHT * 0.4)
        layer3.move_to(layer2.get_center() + UP * 1.2 + RIGHT * 0.4)

        # 4. Transform hex board into the first square layer
        # Using first grid (layer1[0]) for the transform
        self.play(
            ReplacementTransform(hex_board, layer1[0]),
            FadeIn(layer1[1])
        )
        self.wait(0.5)

        # 5. Animate stacking of other layers
        self.play(
            FadeIn(layer2, shift=UP * 0.5 + RIGHT * 0.2),
            run_time=1
        )
        self.play(
            FadeIn(layer3, shift=UP * 0.5 + RIGHT * 0.2),
            run_time=1
        )
        self.wait(1)

        # 6. Add labels and formula
        formula = MathTex("n = 4 \\times \\text{board radius} + 1", font_size=32)
        formula.to_corner(UL)
        
        dimensions = MathTex("n \\times n \\times 3", font_size=44, color=YELLOW)
        dimensions.to_corner(UR, buff=1)
        
        explanation = Text(
            "Multi-layered Square Grid Encoding", 
            font_size=24, 
            slant=ITALIC
        ).to_edge(BOTTOM, buff=0.5)

        self.play(Write(formula))
        self.play(Write(dimensions))
        self.play(FadeIn(explanation))
        
        # Subtle "float" animation for the stack
        self.play(
            VGroup(layer1, layer2, layer3).animate.shift(UP * 0.2),
            rate_func=there_and_back,
            run_time=2
        )
        
        self.wait(2)
