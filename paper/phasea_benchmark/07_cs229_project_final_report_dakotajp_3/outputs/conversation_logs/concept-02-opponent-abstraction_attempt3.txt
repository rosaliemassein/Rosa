class OpponentAbstraction(Scene):
    def construct(self):
        # Narration strings
        voice_1 = (
            "The real secret to generalization lies in how we treat the competition. "
            "Instead of giving every opponent their own channel, we lump them all into "
            "a single 'opponent layer'."
        )
        voice_2 = (
            "This makes the 6-player game look structurally identical to the 2-player game. "
            "Strategy learned against one opponent naturally scales to five."
        )

        # Formula as Text (MathTex/Tex are disallowed)
        formula = Text("S in B^(n x n x 3)", font_size=24).to_edge(UP, buff=0.5)
        self.add(formula)

        # --- 2-PLAYER SETUP ---
        # 3 Layers: Me (Blue), Opponent (Red), Board (Grey)
        layer_2p_1 = Square(side_length=2, fill_opacity=0.8, color="#888888").shift(0.4 * DOWN + 0.4 * RIGHT)
        layer_2p_2 = Square(side_length=2, fill_opacity=0.8, color="#CF5044")
        layer_2p_3 = Square(side_length=2, fill_opacity=0.8, color="#236B8E").shift(0.4 * UP + 0.4 * LEFT)
        
        group_2p = VGroup(layer_2p_1, layer_2p_2, layer_2p_3).scale(0.8).shift(LEFT * 3)
        lbl_2p = Text("2-Player Model", font_size=20).next_to(group_2p, DOWN)
        
        # --- 6-PLAYER SETUP ---
        # Layers: 1 Me, 5 Opponents, 1 Board
        layers_6p = VGroup()
        # Board
        layers_6p.add(Square(side_length=2, fill_opacity=0.8, color="#888888").shift(0.8 * DOWN + 0.8 * RIGHT))
        # 5 Different Opponents (representing individual channels)
        colors_6p = ["#9B59B6", "#FFC0CB", "#FFA500", "#FFFF00", "#83C167"]
        for i, col in enumerate(colors_6p):
            layers_6p.add(Square(side_length=2, fill_opacity=0.8, color=col).shift((i - 2) * 0.25 * UP + (i - 2) * 0.25 * LEFT))
        # Me
        layers_6p.add(Square(side_length=2, fill_opacity=0.8, color="#236B8E").shift(1.0 * UP + 1.0 * LEFT))
        
        group_6p = layers_6p.scale(0.8).shift(RIGHT * 3)
        lbl_6p = Text("6-Player (Raw)", font_size=20).next_to(group_6p, DOWN)

        # Narration container
        narration_label = Text(voice_1, font_size=16).to_edge(DOWN, buff=0.5)

        # Animation sequence
        self.play(
            FadeIn(group_2p),
            Write(lbl_2p),
            Write(narration_label),
            run_time=2
        )
        self.wait(1)
        
        self.play(
            FadeIn(group_6p),
            Write(lbl_6p),
            run_time=2
        )
        self.wait(1)

        # --- ABSTRACTION ANIMATION ---

        opponents_to_cross = group_6p[1:6]
        cross_lines = VGroup()
        for mob in opponents_to_cross:
            l1 = Line(mob.get_corner(UL), mob.get_corner(DR), color="#FF0000", stroke_width=4)
            l2 = Line(mob.get_corner(UR), mob.get_corner(DL), color="#FF0000", stroke_width=4)
            cross_lines.add(l1, l2)

        self.play(Create(cross_lines), run_time=1.5)
        self.wait(0.5)

        p3 = unified_red.get_center() + RIGHT * 0.4 + UP * 0.4

            FadeIn(unified_red),

            lbl_6p.animate.become(Text("Abstracted Layer", font_size=20).next_to(group_6p, DOWN)),
            run_time=2
        )
