        def create_layer_grid(color, label_text):
            grid = VGroup(*[
                Square(side_length=0.3, stroke_width=2, color=color) 
                for _ in range(9)
            ]).arrange_in_grid(rows=3, cols=3, buff=0.05)
            lbl = Text(label_text, font_size=18, color=color).next_to(grid, DOWN)
            return VGroup(grid, lbl)

        opp_layer_2p = create_layer_grid(COLOR_RED, "Opponent Channel").shift(RIGHT * 2.5)

        self.play(Write(title), Write(formula))
        self.play(FadeIn(me_layer_2p), FadeIn(opp_layer_2p))
        self.wait(1)

        # 2. Transition to 6-Player (Many individual channels)
        title_6p = Text("6-Player: Individual Channels", font_size=36).to_edge(UP)
