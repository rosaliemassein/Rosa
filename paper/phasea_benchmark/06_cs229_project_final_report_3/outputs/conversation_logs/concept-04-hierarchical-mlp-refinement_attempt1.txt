from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Image Input
        # Use a placeholder since the image file may not exist
        image_placeholder = Rectangle(height=2, width=3, color=WHITE).set_fill(GREY, opacity=0.5)
        image_text = Text("Image Input", font_size=24).move_to(image_placeholder.get_center())
        image_group = VGroup(image_placeholder, image_text).to_edge(LEFT, buff=1)
        
        self.play(FadeIn(image_group))
        self.wait(0.5)

        # 2. CLIP Block and Country MLP Identification
        clip_block = VGroup(
            Rectangle(width=2, height=1, color=BLUE),
            Text("CLIP", font_size=30)
        ).next_to(image_group, RIGHT, buff=1)
        
        country_text = Text("Country MLP", color=BLUE, font_size=24).next_to(clip_block, DOWN, buff=0.5)
        arrow_to_clip = Arrow(image_group.get_right(), clip_block.get_left(), buff=0.1)
        
        self.play(Create(arrow_to_clip), FadeIn(clip_block))
        self.play(Write(country_text))
        self.wait(1)

        # 3. 3-Layer Neural Network (MLP)
        self.play(FadeOut(image_group, clip_block, arrow_to_clip, country_text))
        
        layer_sizes = [4, 6, 4]
        layers = VGroup()
        for size in layer_sizes:
            layer = VGroup(*[Dot(radius=0.1, color=YELLOW) for _ in range(size)])
            layer.arrange(DOWN, buff=0.3)
            layers.add(layer)
        layers.arrange(RIGHT, buff=1.5)
        layers.move_to(ORIGIN)

        connections = VGroup()
        for i in range(len(layers) - 1):
            for dot1 in layers[i]:
                for dot2 in layers[i+1]:
                    line = Line(dot1.get_center(), dot2.get_center(), stroke_width=1, stroke_opacity=0.3, color=WHITE)
                    connections.add(line)

        mlp_label = Text("Local Region MLP (ReLU)", font_size=30).to_edge(UP)
        
        self.play(Create(layers), Create(connections), Write(mlp_label))

        # 4. Animate signals flowing through the MLP
        for i in range(len(layers) - 1):
            signals = VGroup(*[
                Dot(point=dot.get_center(), radius=0.05, color=GOLD)
                for dot in layers[i]
            ])
            self.play(
                signals.animate.move_to(layers[i+1].get_center()),
                run_time=0.8,
                rate_func=linear
            )
            self.remove(signals)

        # 5. Formula transition
        formula = MathTex(r"c^{(i)}:=\arg\min_{j}\|x^{(i)}-\mu_{j}\|^{2}").to_edge(DOWN, buff=1)
        self.play(Write(formula))
        self.wait(1)

        # 6. Map and GPS Pin
        self.play(FadeOut(layers, connections, mlp_label, formula))
        
        country_map = Circle(radius=2, color=GREEN).set_fill(GREEN, opacity=0.2)
        map_label = Text("Specific Country Region", font_size=24).next_to(country_map, UP)
        
        centroids = VGroup(*[
            Dot(color=RED).move_to(
                country_map.get_center() + np.array([np.cos(a), np.sin(a), 0]) * 1.2
            ) for a in [0, 1.2, 2.5, 4, 5.5]
        ])
        target_centroid = centroids[2]
        
        self.play(Create(country_map), Write(map_label), Create(centroids))
        
        # Create a GPS Pin from scratch
        pin_head = Circle(radius=0.15, color=RED).set_fill(RED, opacity=1)
        pin_tip = Triangle(color=RED).scale(0.15).rotate(180*DEGREES).set_fill(RED, opacity=1)
        pin_tip.next_to(pin_head, DOWN, buff=-0.05)
        pin = VGroup(pin_head, pin_tip)
        pin.next_to(target_centroid, UP, buff=0)
        
        self.play(target_centroid.animate.scale(1.5).set_color(YELLOW))
        self.play(FadeIn(pin, shift=DOWN))
        self.wait(1)
