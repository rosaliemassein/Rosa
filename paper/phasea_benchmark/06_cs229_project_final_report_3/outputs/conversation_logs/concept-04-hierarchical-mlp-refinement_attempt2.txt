from manim import *

class Concept04HierarchicalMLPRefinement(Scene):
    def construct(self):
        # 1. Labels and Titles
        narration = Text("Refinement: Global to Fine-grained", font_size=28, color=BLUE).to_edge(UP)
        goal_text = Text("Two-stage Pipeline: Country -> Cluster", font_size=20, color=YELLOW).next_to(narration, DOWN)
        
        # 2. Image Representation
        image_vector = MathTex(r"\vec{x}_{\text{image}}", color=BLUE).shift(5 * LEFT + 1 * UP)
        image_label = Text("Image Vector", font_size=18).next_to(image_vector, DOWN)
        
        # 3. CLIP Block
        clip_rect = Rectangle(width=1.5, height=1, color=RED).shift(3 * LEFT + 1 * UP)
        clip_text = Text("CLIP", color=RED, font_size=24).move_to(clip_rect)
        clip_group = VGroup(clip_rect, clip_text)

        # 4. MLP Block (3-layer neural network)
        mlp_title = Text("Country-Specific MLP", color=GREEN, font_size=20).shift(1 * RIGHT + 3 * UP)
        
        # Create MLP layers with ReLU activations
        layers = VGroup()
        for i in range(3):
            layer_group = VGroup()
            for j in range(4): # 4 neurons per layer
                dot = Dot(radius=0.1, color=GREEN_A)
                layer_group.add(dot)
            layer_group.arrange(DOWN, buff=0.2)
            layers.add(layer_group)
        layers.arrange(RIGHT, buff=1.0).shift(1 * RIGHT + 1 * UP)
        
        # Drawing edges between layers
        edges = VGroup()
        for i in range(len(layers) - 1):
            for dot1 in layers[i]:
                for dot2 in layers[i+1]:
                    line = Line(dot1.get_center(), dot2.get_center(), stroke_width=0.5, stroke_opacity=0.5, color=WHITE)
                    edges.add(line)
        
        mlp_labels = VGroup(
            MathTex(r"\sigma(\text{ReLU})", font_size=18).next_to(layers[0], UP),
            MathTex(r"\sigma(\text{ReLU})", font_size=18).next_to(layers[1], UP),
            MathTex(r"\sigma(\text{ReLU})", font_size=18).next_to(layers[2], UP)
        )
        mlp_group = VGroup(mlp_title, layers, edges, mlp_labels)

        # 5. Map and Centroids
        map_bg = Rectangle(width=3, height=2, color=GRAY_B).shift(5 * RIGHT + 1 * UP)
        map_title = Text("Local Clusters", font_size=18).next_to(map_bg, UP)
        centroids = VGroup(*[Dot(radius=0.08, color=YELLOW) for _ in range(5)])
        centroids.arrange_in_grid(2, 3, buff=0.5).move_to(map_bg)
        target_centroid = centroids[2]
        
        # GPS Pin (Triangle)
        gps_pin = Triangle(color=RED).scale(0.15).set_fill(RED, opacity=1).rotate(180*DEGREES)
        gps_pin.next_to(target_centroid, UP, buff=0.05)

        # 6. Formula and Final Coordinate
        formula = MathTex(r"c^{(i)}:=\arg\min_{j}\|x^{(i)}-\mu_{j}\|^{2}", font_size=32).to_edge(LEFT, buff=1).shift(2 * DOWN)
        final_coord = MathTex(r"\text{Final Result: } (lat, lon)", color=RED).to_edge(RIGHT, buff=1).shift(2 * DOWN)

        # --- ANIMATION ---
        self.play(Write(narration))
        self.play(FadeIn(image_vector), Write(image_label))
        self.wait(0.5)

        # 1st Stage: CLIP
        self.play(Create(clip_group))
        self.play(Arrow(image_vector.get_right(), clip_rect.get_left(), buff=0.1))
        self.wait(0.5)

        # 2nd Stage: MLP
        self.play(FadeIn(mlp_title), Create(layers), Create(edges), FadeIn(mlp_labels))
        self.play(Arrow(clip_rect.get_right(), layers.get_left(), buff=0.1))

        # Signals flowing
        for i in range(len(layers)-1):
            flow_dots = VGroup(*[Dot(radius=0.05, color=YELLOW).move_to(d.get_center()) for d in layers[i]])
            self.play(
                *[flow_dots[j].animate.move_to(layers[i+1][j % 4].get_center()) for j in range(4)],
                run_time=0.5
            )
            self.remove(flow_dots)

        # 3rd Stage: Prediction on Map
        self.play(Create(map_bg), Write(map_title), Create(centroids))
        self.play(Arrow(layers.get_right(), target_centroid.get_center(), color=YELLOW))
        self.wait(0.5)
        
        # Drop Pin and Formula
        self.play(Write(formula))
        self.play(
            FadeIn(gps_pin, shift=DOWN),
            target_centroid.animate.set_color(RED).scale(2)
        )
        self.wait(1)

        # Final Transformation
        image_copy = image_vector.copy()
        self.play(ReplacementTransform(image_copy, final_coord))
        self.play(FadeIn(goal_text))
        
        self.wait(2)
