from manim import *
import numpy as np

class ZeroShotClassification(Scene):
    def construct(self):
        # 1. Formula reference at the top
        formula = MathTex(
            r"h(\mathbf{v};\mathbf{X})=\frac{\exp(\mathbf{X}\mathbf{v})}{\sum_{j=1}^{N}\exp(\mathbf{X}_{j}^{T}\mathbf{v})}",
            font_size=36
        ).to_edge(UP, buff=0.5)

        # 2. Image Representation (Vector v)
        image_box = Rectangle(height=2.2, width=2.2, color=BLUE_E, fill_opacity=0.3)
        image_text = Text("Image", font_size=18).move_to(image_box.get_center())
        image_group = VGroup(image_box, image_text).to_edge(LEFT, buff=1).shift(UP * 0.5)
        
        v_vector = Matrix([[0.8], [0.2], [0.5], ["\\vdots"]]).scale(0.5).next_to(image_group, RIGHT, buff=0.5)
        v_label = MathTex(r"\mathbf{v}").next_to(v_vector, UP, buff=0.1)
        
        # 3. Matrix X (Text embeddings for countries)
        countries = ["USA", "France", "Japan", "Brazil", "India", "Egypt"]
        matrix_rows = VGroup(*[
            VGroup(
                Text(c, font_size=16).set_width(0.8, stretch=True),
                Rectangle(height=0.25, width=1.5, color=BLUE_D, fill_opacity=0.4)
            ).arrange(RIGHT, buff=0.1)
            for c in countries
        ]).arrange(DOWN, buff=0.15).to_edge(RIGHT, buff=1).shift(UP * 0.5)
        
        x_label = MathTex(r"\mathbf{X}").next_to(matrix_rows, UP, buff=0.1)
        x_bracket = SurroundingRectangle(matrix_rows, color=WHITE, buff=0.1)
        matrix_X = VGroup(matrix_rows, x_label, x_bracket)

        # 4. Bar Chart for Softmax (Probability Distribution)
        axes = Axes(
            x_range=[0, 6, 1],
            y_range=[0, 1, 0.5],
            x_length=6,
            y_length=2,
            axis_config={"include_tip": False}
        ).to_edge(DOWN, buff=0.5)
        
        bar_labels = VGroup(*[
            Text(c, font_size=14).next_to(axes.c2p(i + 0.5, 0), DOWN, buff=0.1)
            for i, c in enumerate(countries)
        ])
        
        bars = VGroup(*[
            Rectangle(
                width=0.5, 
                height=0.01, 
                color=BLUE, 
                fill_opacity=0.7
            ).move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN)
            for i in range(len(countries))
        ])

        # 5. Initialization
        self.add(formula)
        self.play(FadeIn(image_group), Create(v_vector), Write(v_label))
        self.play(Create(matrix_X))
        self.play(Create(axes), Create(bar_labels))
        self.add(bars)

        # 6. Animation: Dot Product and Softmax Update
        highlight = Rectangle(
            height=0.35, width=2.6, color=YELLOW, stroke_width=4
        ).move_to(matrix_rows[0])
        
        # Probabilities: France (index 1) gets the highest score
        probs = [0.1, 0.8, 0.05, 0.02, 0.02, 0.01]
        
        for i in range(len(countries)):
            # Move highlight to current row in X
            self.play(highlight.animate.move_to(matrix_rows[i]), run_time=0.4)
            
            # Show connection line for dot product
            dot_line = Line(v_vector.get_right(), matrix_rows[i].get_left(), color=YELLOW, stroke_width=2)
            self.play(Create(dot_line), run_time=0.1)
            
            # Update bar chart height based on probability
            target_height = probs[i] * 1.8 # Scaled for visual clarity
            self.play(
                bars[i].animate.stretch_to_fit_height(max(0.05, target_height)).move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN),
                FadeOut(dot_line),
                run_time=0.3
            )
            
            if i == 1: # France highlight
                self.play(Indicate(matrix_rows[i], color=GOLD), bars[i].animate.set_color(GREEN))
                
        # Final mathematical summary
        prob_vec_label = MathTex(r"\text{Softmax}(\mathbf{X}\mathbf{v})", font_size=32).next_to(axes, UP, buff=0.2)
        self.play(Write(prob_vec_label))
        self.play(FadeOut(highlight))
        
        self.wait(2)
