from manim import *

class ContrastiveAlignment(Scene):
    def construct(self):
        # 1. Create Encoder Blocks
        img_encoder = VGroup(
            RoundedRectangle(corner_radius=0.1, height=1.5, width=3, color=BLUE),
            Text("Image Encoder", font_size=24)
        ).shift(LEFT * 3.5 + UP * 2.5)

        text_encoder = VGroup(
            RoundedRectangle(corner_radius=0.1, height=1.5, width=3, color=GREEN),
            Text("Text Encoder", font_size=24)
        ).shift(RIGHT * 3.5 + UP * 2.5)

        # 2. Show Inputs
        street_view_img = Square(side_length=1.2, fill_opacity=0.5, color=GRAY).next_to(img_encoder, LEFT, buff=0.5)
        img_placeholder_text = Text("Street\nView", font_size=16).move_to(street_view_img)
        
        france_text = Text("'A Street View photo in France'", font_size=20, color=YELLOW).next_to(text_encoder, RIGHT, buff=0.5)

        self.play(FadeIn(img_encoder, text_encoder, street_view_img, img_placeholder_text, france_text))
        self.wait(1)

        # 3. Coordinate System (Shared Mathematical Space)
        axes = Axes(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5,
            y_length=5,
            axis_config={"include_tip": True}
        ).shift(DOWN * 1)
        axes_labels = axes.get_axis_labels(x_label="x", y_label="y")
        space_title = Text("Shared Embedding Space", font_size=24).next_to(axes, UP)

        self.play(Create(axes), Write(axes_labels), Write(space_title))

        # 4. Vectors and Similarity Score
        # ValueTracker for similarity score
        sim_tracker = ValueTracker(0.42)
        
        # Similarity display
        sim_score_label = Text("Cosine Similarity:", font_size=24).to_edge(RIGHT).shift(UP * 0.5)
        sim_score_val = DecimalNumber(sim_tracker.get_value(), num_decimal_places=2, color=YELLOW).next_to(sim_score_label, DOWN)
        sim_score_val.add_updater(lambda d: d.set_value(sim_tracker.get_value()))

        # Define vectors
        # v1: Image Embedding (stays relatively fixed)
        # v2: Text Embedding (will move)
        origin = axes.c2p(0,0)
        img_vec = Arrow(origin, axes.c2p(2, 1, 0), buff=0, color=BLUE)
        img_vec_label = MathTex("z_{img}", color=BLUE, font_size=24).next_to(img_vec.get_end(), UR, buff=0.1)

        text_vec_france = Arrow(origin, axes.c2p(0.5, 2.2, 0), buff=0, color=GREEN)
        text_vec_label = MathTex("z_{text}", color=GREEN, font_size=24).next_to(text_vec_france.get_end(), UP, buff=0.1)

        self.play(GrowArrow(img_vec), Write(img_vec_label))
        self.play(GrowArrow(text_vec_france), Write(text_vec_label))
        self.play(Write(sim_score_label), FadeIn(sim_score_val))
        self.wait(1)

        # 5. Animate Alignment (Correct Label)
        target_text_vec = Arrow(origin, axes.c2p(1.9, 0.95, 0), buff=0, color=GREEN)
        
        self.play(
            Transform(text_vec_france, target_text_vec),
            text_vec_label.animate.next_to(target_text_vec.get_end(), DR, buff=0.1),
            sim_tracker.animate.set_value(0.98),
            run_time=2
        )
        self.wait(2)

        # 6. Show "Wrong" Label and Pushing Away
        japan_text = Text("'A Street View photo in Japan'", font_size=20, color=RED).move_to(france_text)
        
        self.play(Transform(france_text, japan_text))
        
        # Pushing Japan vector away
        wrong_text_vec = Arrow(origin, axes.c2p(-1.5, 1.8, 0), buff=0, color=RED)
        
        self.play(
            Transform(text_vec_france, wrong_text_vec),
            text_vec_label.animate.next_to(wrong_text_vec.get_end(), UL, buff=0.1),
            text_vec_france.animate.set_color(RED),
            text_vec_label.animate.set_color(RED),
            sim_tracker.animate.set_value(0.15),
            run_time=2
        )
        self.wait(2)

        # 7. Display Loss Formula
        formula = MathTex(
            r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})",
            font_size=36
        ).to_edge(LEFT).shift(DOWN * 0.5)
        
        self.play(Write(formula))
        self.play(Indicate(formula))
        self.wait(3)
