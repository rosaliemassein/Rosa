from manim import *

class ZeroShotClassification(Scene):
    def construct(self):
        # 1. Setup Image and Visual Vector v
        # Using a square as a placeholder if the image is missing
        try:
            image = ImageMobject("img-8.jpeg").scale(1.2).to_corner(UL, buff=0.5)
        except:
            image = Rectangle(height=2, width=3, color=BLUE).to_corner(UL, buff=0.5)
            image.add(Text("Image", font_size=20).move_to(image))

        # Vector v representation
        v_vector = MathTex(
            r"\mathbf{v} = \begin{bmatrix} 0.1 \\ 0.5 \\ \vdots \\ -0.2 \end{bmatrix}",
            color=YELLOW, font_size=32
        ).next_to(image, DOWN, buff=0.5)
        v_desc = Text("Image Representation", font_size=20).next_to(v_vector, DOWN, buff=0.2)

        # 2. Matrix X representation
        x_matrix = MathTex(
            r"\mathbf{X} = \begin{bmatrix} \text{France} \\ \text{Germany} \\ \vdots \\ \text{Japan} \end{bmatrix}",
            color=BLUE, font_size=32
        ).next_to(v_vector, RIGHT, buff=1.5)
        x_desc = Text("Country Embeddings", font_size=20).next_to(x_matrix, DOWN, buff=0.2)

        # 3. Probability Visualization (Manual Bar Chart for better compatibility)
        axes = Axes(
            x_range=[0, 4, 1],
            y_range=[0, 1, 0.5],
            x_length=4,
            y_length=2,
            axis_config={"include_tip": False},
            tips=False
        ).to_corner(UR, buff=1.0)
        
        labels = VGroup(
            Text("DE", font_size=16).next_to(axes.c2p(0.5, 0), DOWN),
            Text("UK", font_size=16).next_to(axes.c2p(1.5, 0), DOWN),
            Text("FR", font_size=16).next_to(axes.c2p(2.5, 0), DOWN),
            Text("JP", font_size=16).next_to(axes.c2p(3.5, 0), DOWN)
        )

        def get_bars(values):
            bars = VGroup()
            for i, val in enumerate(values):
                bar = Rectangle(
                    width=0.6, 
                    height=val * 2, 
                    fill_opacity=0.8, 
                    fill_color=GREEN if i == 2 else BLUE,
                    stroke_width=1
                )
                bar.move_to(axes.c2p(i + 0.5, val / 2))
                bars.add(bar)
            return bars

        initial_bars = get_bars([0.1, 0.1, 0.1, 0.1])
        final_bars = get_bars([0.05, 0.1, 0.8, 0.05])

        # 4. Mathematical Formula
        formula = MathTex(
            r"h(\mathbf{v};\mathbf{X}) = \frac{\exp(\mathbf{X}\mathbf{v})}{\sum_{j=1}^{N}\exp(\mathbf{X}_{j}^{T}\mathbf{v})}",
            font_size=36
        ).to_edge(DOWN, buff=0.8)

        # --- ANIMATION ---
        
        # Introduction
        intro_text = Text("Zero-Shot Classifier", font_size=32).to_edge(UP)
        self.play(FadeIn(image), Write(intro_text))
        self.wait(1)

        # Show Vector v and Matrix X
        self.play(Write(v_vector), Write(v_desc))
        self.play(Write(x_matrix), Write(x_desc))
        self.wait(1)

        # Pulse operation logic
        highlight_rect = SurroundingRectangle(x_matrix, color=YELLOW, buff=0.1)
        self.play(Create(highlight_rect))
        self.play(FadeOut(highlight_rect))

        # Show Equation
        self.play(Write(formula))
        self.wait(1)

        # Probabilities Updating
        self.play(Create(axes), Create(labels))
        self.play(Create(initial_bars))
        self.wait(0.5)
        self.play(Transform(initial_bars, final_bars), run_time=2)
        
        # Result Highlight
        prediction = Text("Prediction: France", font_size=28, color=GREEN).next_to(axes, DOWN, buff=0.8)
        self.play(Write(prediction))
        self.wait(2)
