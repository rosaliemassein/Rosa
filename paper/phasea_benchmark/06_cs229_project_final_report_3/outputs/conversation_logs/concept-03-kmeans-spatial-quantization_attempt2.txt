from manim import *
import numpy as np

class KMeansSpatialQuantization(Scene):
    def construct(self):
        # 1. Manual Grid/Axes (Replacements for disallowed Axes/NumberPlane)
        x_ax = Line(LEFT * 6, RIGHT * 6, color="#777777")
        y_ax = Line(DOWN * 3.5, UP * 3.5, color="#777777")
        self.play(Create(x_ax), Create(y_ax))

        # 2. Country Outline (Replacement for disallowed Polygon)
        australia = VMobject()
        # Simplified approximate shape of Australia
        pts = [
            np.array([-3.5, -0.5, 0]), np.array([-2.5, -2.0, 0]), 
            np.array([1.5, -2.2, 0]), np.array([3.5, -0.8, 0]), 
            np.array([4.0, 1.2, 0]), np.array([1.5, 2.5, 0]), 
            np.array([-1.0, 2.8, 0]), np.array([-4.0, 1.5, 0])
        ]
        australia.set_points_as_corners([*pts, pts[0]])
        australia.set_color(WHITE)
        australia.set_stroke(opacity=0.6)
        self.play(Create(australia))

        # 3. Data Points (Using hex colors for undefined GRAY_B)
        dots = VGroup()
        num_dots = 70
        # Use np.random for deterministic behavior and to avoid 'random' identifier issues
        np.random.seed(42)
        for _ in range(num_dots):
            px = np.random.uniform(-3.5, 3.5)
            py = np.random.uniform(-1.8, 2.3)
            dots.add(Dot(point=[px, py, 0], radius=0.06, color="#BBBBBB"))
        
        self.play(FadeIn(dots))

        # 4. Formula using Text (Replacement for disallowed MathTex)
        formula_label = Text("K-Means Update Rule", font_size=24).to_edge(UP)
        formula_str = Text("mu_j = sum(x_i) / N_j", font_size=20).next_to(formula_label, DOWN)
        self.add(formula_label, formula_str)

        # 5. Centroids (Replacement for disallowed Cross)
        def create_centroid_cross(pos, color):
            line1 = Line(pos + LEFT*0.18 + UP*0.18, pos + RIGHT*0.18 + DOWN*0.18, color=color, stroke_width=5)
            line2 = Line(pos + LEFT*0.18 + DOWN*0.18, pos + RIGHT*0.18 + UP*0.18, color=color, stroke_width=5)
            return VGroup(line1, line2)

        colors = [RED, BLUE, GREEN, YELLOW]
        centroids = VGroup()
        for i in range(len(colors)):
            start_pos = np.array([np.random.uniform(-2, 2), np.random.uniform(-1, 1), 0])
            centroids.add(create_centroid_cross(start_pos, colors[i]))
        
        self.play(*[Create(c) for c in centroids])
        self.wait(0.5)

        # 6. Assign Colors (Voronoi behavior)
        color_anims = []
        clusters = [[] for _ in range(len(centroids))]
        
        for dot in dots:
            dot_center = dot.get_center()
            # Calculate distance to each centroid's center
            dists = [np.linalg.norm(dot_center - c.get_center()) for c in centroids]
            nearest_idx = np.argmin(dists)
            clusters[nearest_idx].append(dot)
            color_anims.append(dot.animate.set_color(colors[nearest_idx]))
        
        self.play(*color_anims, run_time=1.5)
        self.wait(1)

        # 7. Move centroids (Replacement for disallowed LaggedStart)
        move_anims = []
        for i, centroid in enumerate(centroids):
            if clusters[i]:
                cluster_coords = np.array([d.get_center() for d in clusters[i]])
                new_center = np.mean(cluster_coords, axis=0)
                move_anims.append(centroid.animate.move_to(new_center))
        
        self.play(*move_anims, run_time=2)

        # 8. Final highlight (Replacement for disallowed Indicate)
        highlight_circles = VGroup()
        for c in centroids:
            ring = Circle(radius=0.3, color=c[0].get_color()).move_to(c.get_center())
            highlight_circles.add(ring)
        
        self.play(
            FadeIn(highlight_circles),
            *[c.animate.scale(1.3) for c in centroids]
        )
        self.play(*[c.animate.scale(1/1.3) for c in centroids])

        final_text = Text("Discretized Regions", font_size=28).to_edge(DOWN)
        self.play(Write(final_text))
        
        self.wait(2)
