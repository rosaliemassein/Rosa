from manim import *
import numpy as np

class Concept03KMeansSpatialQuantization(Scene):
    def construct(self):
        # 1. Manual Coordinate System (Replacing Axes/NumberPlane)
        x_axis = Line(LEFT * 6, RIGHT * 6, color=WHITE, stroke_opacity=0.3)
        y_axis = Line(UP * 3.5, DOWN * 3.5, color=WHITE, stroke_opacity=0.3)
        self.add(x_axis, y_axis)

        # 2. Simplified Country Outline using Lines (Replacing Polygon/VMobject)
        aus_pts = [
            [-3, 1, 0], [-1.5, 1.2, 0], [0, 0.5, 0], [1.5, 1.5, 0], 
            [3.5, 0, 0], [3, -1.5, 0], [1, -2.5, 0], [-2, -2.5, 0], 
            [-3.5, -1, 0], [-3, 1, 0]
        ]
        australia = VGroup()
        for i in range(len(aus_pts) - 1):
            australia.add(Line(aus_pts[i], aus_pts[i+1], color=BLUE))
        
        # 3. Formula (Replacing MathTex)
        formula_text = Text("Centroid Update Rule: mu_j = average(x_i in cluster j)", font_size=24)
        formula_text.to_edge(UP, buff=0.5)
        
        self.play(Create(australia), Write(formula_text))

        # 4. Generate Training Points (Replacing NumberPlane region methods)
        np.random.seed(42)
        dots = VGroup()
        for _ in range(60):
            # Random points roughly within the Australia box
            px = np.random.uniform(-3, 3)
            py = np.random.uniform(-2, 1)
            dots.add(Dot(point=[px, py, 0], radius=0.06, color=GRAY))
        
        self.play(FadeIn(dots))

        # 5. Centroid Crosses (Manual replacement for Indicate/AnimationGroup)
        # Colors: RED, GREEN, YELLOW, WHITE
        colors = [RED, GREEN, YELLOW, WHITE]
        centroid_positions = [[-2, 0, 0], [0.5, -1.5, 0], [2, 0.5, 0], [-1, -2, 0]]
        centroids = VGroup()
        
        for i in range(4):
            p = np.array(centroid_positions[i])
            # A cross is two lines
            l1 = Line(p + [-0.2, -0.2, 0], p + [0.2, 0.2, 0], color=colors[i], stroke_width=6)
            l2 = Line(p + [-0.2, 0.2, 0], p + [0.2, -0.2, 0], color=colors[i], stroke_width=6)
            centroids.add(VGroup(l1, l2))

        # Manual animation sequence (Replacing AnimationGroup)
        for c in centroids:
            self.play(Create(c), run_time=0.4)

        self.wait(1)

        # 6. Assign dots to nearest centroid
        assignments = []
        for d in dots:
            d_pos = d.get_center()
            dists = [np.linalg.norm(d_pos - c.get_center()) for c in centroids]
            assignments.append(np.argmin(dists))

        # Change dot colors
        color_anims = []
        for i, d in enumerate(dots):
            color_anims.append(d.animate.set_color(colors[assignments[i]]))
        
        self.play(*color_anims, run_time=1.5)
        self.wait(1)

        # 7. Move centroids to group means (The 'Convergence')
        move_anims = []
        for i in range(len(centroids)):
            cluster_points = [
                dots[j].get_center() 
                for j, cluster_idx in enumerate(assignments) if cluster_idx == i
            ]
            if cluster_points:
                new_pos = np.mean(cluster_points, axis=0)
                move_anims.append(centroids[i].animate.move_to(new_pos))
        
        self.play(*move_anims, run_time=2)

        # 8. Highlight final clusters (Replacing Indicate)
        highlight_anims = []
        for c in centroids:
            highlight_anims.append(c.animate.scale(1.5))
        self.play(*highlight_anims, run_time=0.5)
        
        highlight_back_anims = []
        for c in centroids:
            highlight_back_anims.append(c.animate.scale(1/1.5))
        self.play(*highlight_back_anims, run_time=0.5)

        self.wait(2)
