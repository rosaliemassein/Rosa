from manim import *

class SelectiveLayerActivation(Scene):
    def construct(self):
        # Create a horizontal chain of 24 Square objects representing layers 0 to 23
        squares = VGroup(*[
            Square(side_length=0.35, fill_opacity=0.6, stroke_width=2) 
            for _ in range(24)
        ])
        squares.arrange(RIGHT, buff=0.15)
        squares.center()

        # Define indices for sections
        frozen_indices = list(range(0, 7)) + list(range(11, 23))
        active_semantic_indices = range(7, 11)
        active_head_index = 23

        # Color the layers
        for i in frozen_indices:
            squares[i].set_color(GRAY)

        # Color layers 7-10 bright blue and simulate a Glow effect
        for i in active_semantic_indices:
            squares[i].set_color(BLUE)
            # Add a secondary glowing layer to these specific squares
            glow = squares[i].copy().set_stroke(BLUE, width=8, opacity=0.3).scale(1.2)
            squares[i].add(glow)

        # Color layer 23 bright orange
        squares[active_head_index].set_color(ORANGE)
        head_glow = squares[active_head_index].copy().set_stroke(ORANGE, width=8, opacity=0.3).scale(1.2)
        squares[active_head_index].add(head_glow)

        # Labels for the different sections
        frozen_label = Text("Frozen Backbone", font_size=18, color=GRAY).next_to(squares[0:7], UP, buff=0.5)
        active_sem_label = Text("Active Semantics", font_size=18, color=BLUE).next_to(squares[7:11], DOWN, buff=0.5)
        active_head_label = Text("Active Head", font_size=18, color=ORANGE).next_to(squares[23], UP, buff=0.5)

        # Add initial mobjects to the scene
        self.add(squares, frozen_label, active_sem_label, active_head_label)

        # Define the pulse sequence indices: 7, 8, 9, 10, then skip to 23
        pulse_indices = [7, 8, 9, 10, 23]
        
        # Animate pulses sequentially through active regions
        # Using rate_functions.linear to ensure the identifier is defined
        pulses = [Indicate(squares[i], color=WHITE, scale_factor=1.5) for i in pulse_indices]
        
        self.play(
            Succession(*pulses),
            run_time=4,
            rate_func=rate_functions.linear
        )
        
        self.wait(2)
