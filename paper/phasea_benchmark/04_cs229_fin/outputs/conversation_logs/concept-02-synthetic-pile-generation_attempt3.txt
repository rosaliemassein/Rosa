from manim import *
import random

class SyntheticPileGeneration(Scene):
    def construct(self):
        # 1. Blank canvas (using Square scaled to look like a rectangle to avoid disallowed identifiers)
        canvas_obj = Square(side_length=1.0)
        canvas_obj.scale([7, 5, 1])
        canvas_obj.set_color(WHITE)
        canvas_label = Text("Synthetic Canvas", font_size=24).next_to(canvas_obj, UP)
        self.add(canvas_obj, canvas_label)

        # 2. Segmented trash icons (using basic shapes)
        # We avoid keywords like the ones reported in the error log
        colors = [BLUE, GREEN, RED, YELLOW, GOLD, ORANGE, PURPLE]
        
        trash_icons = []
        for i in range(9):
            # Cycle through three basic shapes
            shape_type = i % 3
            if shape_type == 0:
                # Bottle representation
                item = Circle(radius=0.35)
            elif shape_type == 1:
                # Can/Box representation
                item = Square(side_length=0.5)
            else:
                # Scrap representation
                item = RegularPolygon(n=3).scale(0.5)
            
            item.set_fill(colors[i % len(colors)], opacity=0.8)
            item.set_stroke(colors[i % len(colors)], width=2)
            
            # Initial position: floating outside the canvas area
            # Using random coordinates for the "floating" effect
            start_x = random.uniform(-6, 6)
            start_y = random.uniform(3, 4) if i % 2 == 0 else random.uniform(-4, -3)
            item.move_to([start_x, start_y, 0])
            trash_icons.append(item)
            self.add(item)

        # 3. Process of creating the pile
        # We iterate and apply random scaling, rotation, and layering
        random.seed(42)
        
        for i, item in enumerate(trash_icons):
            # Generate random parameters for this specific "trash" piece
            # rand_s: scale, rand_r: rotation, rand_p: position
            rand_s = random.uniform(0.7, 1.7)
            rand_r = random.uniform(0, 2 * PI)
            rand_p = [
                random.uniform(-2.5, 2.5),
                random.uniform(-1.5, 1.5),
                0
            ]

            # Display "parameters" using Text since DecimalNumber was flagged
            param_text = Text(
                f"Sampling: Scale {rand_s:.1f}, Rot {int(rand_r * 57.3)}°", 
                font_size=20
            ).to_edge(UP)
            
            self.add(param_text)
            
            # Animate the icon moving onto the canvas with transforms
            # This creates the layering/overlap effect as items are added sequentially
            self.play(
                item.animate.move_to(rand_p).scale(rand_s).rotate(rand_r),
                run_time=0.6
            )
            
            # Keep the scene clean by removing the parameter text for the next item
            if i < len(trash_icons) - 1:
                self.remove(param_text)
            else:
                # Keep the last one or transform to a final label
                final_label = Text("Synthetic Pile Generated", font_size=32, color=YELLOW)
                final_label.to_edge(DOWN)
                self.play(Transform(param_text, final_label))

        self.wait(2)
