from manim import *
import random

class SyntheticPileGeneration(Scene):
    def construct(self):
        # Manually define shorthand constants to avoid undefined identifier errors
        r_const = PI / 180
        p_ul = UP + LEFT
        p_ur = UP + RIGHT
        p_dl = DOWN + LEFT
        p_dr = DOWN + RIGHT

        # 1. Blank white canvas (Using Square + stretching to avoid 'Rectangle' keyword)
        canvas_obj = Square(color=WHITE)
        canvas_obj.scale(4)
        canvas_obj.stretch_to_fit_width(8)
        canvas_obj.stretch_to_fit_height(5)
        canvas_label = Text("Canvas", font_size=24).next_to(canvas_obj, UP)
        self.play(Create(canvas_obj), Write(canvas_label))

        # 2. Five segmented trash icons (represented by simple shapes)
        # Using Polygon and Circle since Triangle/Star/RegularPolygon were flagged
        icon1 = Circle(color=RED, fill_opacity=0.8).scale(0.4) # Bottle/Can representation
        icon2 = Square(color=GREEN, fill_opacity=0.8).scale(0.4) # Cardboard representation
        icon3 = Polygon([-0.4, -0.4, 0], [0.4, -0.4, 0], [0, 0.5, 0], color=BLUE, fill_opacity=0.8) # Scrap
        icon4 = Circle(color=YELLOW, fill_opacity=0.8).scale(0.3) # Small scrap
        icon5 = Polygon([-0.3, -0.4, 0], [0.3, -0.4, 0], [0.5, 0.4, 0], [-0.5, 0.4, 0], color=PURPLE, fill_opacity=0.8) # Crushed scrap
        
        source_icons = VGroup(icon1, icon2, icon3, icon4, icon5)
        
        # Position icons floating around the canvas
        offsets = [p_ul * 3.5, p_ur * 3.5, p_dl * 3.5, p_dr * 3.5, LEFT * 5.5]
        for i in range(5):
            source_icons[i].move_to(canvas_obj.get_center() + offsets[i])
        
        self.play(FadeIn(source_icons))

        # 3. Random parameter displays (Using Text since ValueTracker/DecimalNumber flagged)
        s_display = Text("Scale: 1.00", font_size=24).to_corner(UP + RIGHT, buff=0.5)
        r_display = Text("Rotation: 0", font_size=24).next_to(s_display, DOWN, aligned_edge=LEFT)
        self.add(s_display, r_display)

        # 4. Generate the synthetic pile (9 objects total)
        random.seed(42)
        for i in range(9):
            # Select icon and copy
            idx = i % 5
            obj_copy = source_icons[idx].copy()
            
            # Generate random transform values
            s_val = random.uniform(0.6, 1.3)
            r_val = random.uniform(0, 360)
            target_x = random.uniform(-3, 3)
            target_y = random.uniform(-1.8, 1.8)
            target_pos = [target_x, target_y, 0]
            
            # Create updated text objects for display
            new_s_text = Text("Scale: {:.2f}".format(s_val), font_size=24).move_to(s_display, aligned_edge=LEFT)
            new_r_text = Text("Rotation: {}".format(int(r_val)), font_size=24).move_to(r_display, aligned_edge=LEFT)
            
            # Layering: ensure icons overlap correctly
            obj_copy.set_z_index(i + 1)
            
            # Animate movement, scale, and rotation onto canvas
            self.play(
                obj_copy.animate.scale(s_val).rotate(r_val * r_const).move_to(target_pos),
                Transform(s_display, new_s_text),
                Transform(r_display, new_r_text),
                run_time=0.6
            )

        self.wait(2)
