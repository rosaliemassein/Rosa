from manim import *

class Concept04SelectiveLayerActivation(Scene):
    def construct(self):
        # Create a horizontal chain of 24 Square objects representing layers 0 to 23
        layers = VGroup(*[Square(side_length=0.3) for _ in range(24)])
        layers.arrange(RIGHT, buff=0.1)
        layers.center()

        # Apply specific coloring based on the regions described
        # Layers 0-6: Frozen Backbone (Gray)
        # Layers 7-10: Active Semantics (Blue)
        # Layers 11-22: Frozen (Gray)
        # Layer 23: Active Head (Orange)
        for i in range(24):
            if 7 <= i <= 10:
                # Active Semantics
                layers[i].set_stroke(BLUE, width=4)
                layers[i].set_fill(BLUE, opacity=0.5)
            elif i == 23:
                # Active Head
                layers[i].set_stroke(ORANGE, width=4)
                layers[i].set_fill(ORANGE, opacity=0.5)
            else:
                # Frozen layers
                layers[i].set_stroke(GRAY, width=2)
                layers[i].set_fill(GRAY, opacity=0.2)

        # Labels for specific sections
        frozen_backbone_label = Text("Frozen Backbone", font_size=16).next_to(layers[0:7], DOWN, buff=0.5)
        active_semantics_label = Text("Active Semantics", font_size=16, color=BLUE).next_to(layers[7:11], UP, buff=0.5)
        active_head_label = Text("Active Head", font_size=16, color=ORANGE).next_to(layers[23], UP, buff=0.5)

        # Display the setup
        self.add(layers)
        self.play(
            FadeIn(frozen_backbone_label),
            FadeIn(active_semantics_label),
            FadeIn(active_head_label)
        )
        self.wait(0.5)

        # Active layers for the gradient pulse animation
        active_indices = [7, 8, 9, 10, 23]
        
        # Create a list of pulse animations
        # We use a loop to create the sequence of pulses
        pulses = [
            Indicate(layers[i], color=layers[i].get_stroke_color(), scale_factor=1.4)
            for i in active_indices
        ]

        # Animate the succession of pulses moving through the selected layers
        # Using a lambda function for rate_func to ensure linear progression without the 'linear' keyword
        self.play(
            Succession(*pulses),
            run_time=3,
            rate_func=lambda t: t
        )

        # Repeat once to represent continuous training gradients
        self.play(
            Succession(*pulses),
            run_time=3,
            rate_func=lambda t: t
        )

        self.wait(1)
