            grid.add(Line([i * step, -2.8, 0], [i * step, 2.8, 0], stroke_opacity=0.3))
        
        # Placeholder for image to ensure it runs even if file is missing

        gt_box.move_to([1.2, 0.4, 0])

        pred_box.move_to([0.8, 0.9, 0])

        pred_label = Text("Prediction", font_size=16, color=BLUE).next_to(pred_box, UP, buff=0.1)

        self.add(gt_box, pred_box, gt_label, pred_label)

        # 3. Arrows for Coordinate and Dimension differences
        dot_gt = Dot(gt_box.get_center(), color=RED, radius=0.06)
