from manim import *
import numpy as np

class YOLOv10LocalizationLoss(Scene):
    def construct(self):
        self.setup_image()
        self.visualize_cell_and_box()
        self.explain_loss_function()
        self.animate_training_process()

    def setup_image(self):
        # Create a placeholder or background image
        try:
            self.img = ImageMobject("img-11.jpeg")
            self.img.scale_to_fit_height(7)
        except:
            self.img = Rectangle(width=10, height=7, color=GRAY, fill_opacity=0.1)
        self.add(self.img)

    def visualize_cell_and_box(self):
        # Create a grid (7x7) manually
        grid = VGroup()
        grid_size = 7
        cell_size = 0.8
        start_point = -(grid_size * cell_size) / 2
        
        for i in range(grid_size + 1):
            grid.add(Line([start_point + i * cell_size, start_point, 0], 
                          [start_point + i * cell_size, -start_point, 0], 
                          stroke_width=1, stroke_opacity=0.5))
            grid.add(Line([start_point, start_point + i * cell_size, 0], 
                          [-start_point, start_point + i * cell_size, 0], 
                          stroke_width=1, stroke_opacity=0.5))
        
        grid.set_color(WHITE)
        self.play(Create(grid))
        self.grid = grid

        # Highlight cell
        target_cell_center = [start_point + 4.5 * cell_size, start_point + 2.5 * cell_size, 0]
        cell_highlight = Square(side_length=cell_size, color=YELLOW, stroke_width=4).move_to(target_cell_center)
        self.play(Create(cell_highlight))
        
        # Center point
        center_dot = Dot(target_cell_center + [0.1, -0.05, 0], color=RED, radius=0.05)
        self.play(Create(center_dot))
        
        # Ground Truth Box (using DashedLines manually for safety)
        gt_w, gt_h = 1.5, 2.2
        self.gt_rect = Rectangle(width=gt_w, height=gt_h, color=RED).move_to(center_dot.get_center())
        gt_dashed = VGroup(
            DashedLine(self.gt_rect.get_corner(UL), self.gt_rect.get_corner(UR), color=RED),
            DashedLine(self.gt_rect.get_corner(UR), self.gt_rect.get_corner(DR), color=RED),
            DashedLine(self.gt_rect.get_corner(DR), self.gt_rect.get_corner(DL), color=RED),
            DashedLine(self.gt_rect.get_corner(DL), self.gt_rect.get_corner(UL), color=RED)
        )
        
        # Predicted Box
        self.box_prediction = Rectangle(width=1.8, height=1.9, color=BLUE).move_to(center_dot.get_center() + [0.3, 0.4, 0])
        
        self.play(Create(gt_dashed))
        self.play(Create(self.box_prediction))
        
        # Arrows for coordinates
        gt_center = center_dot.get_center()
        pred_center = self.box_prediction.get_center()
        
        x_arrow = Arrow(start=[pred_center[0], pred_center[1], 0], 
                        end=[gt_center[0], pred_center[1], 0], 
                        color=GREEN, buff=0, stroke_width=2, max_tip_length_to_length_ratio=0.2)
        y_arrow = Arrow(start=[gt_center[0], pred_center[1], 0], 
                        end=[gt_center[0], gt_center[1], 0], 
                        color=GREEN, buff=0, stroke_width=2, max_tip_length_to_length_ratio=0.2)
        
        # Dimensions arrows
        w_arrow = Line(self.box_prediction.get_left(), self.box_prediction.get_right(), color=ORANGE).shift(DOWN * 0.1)
        w_arrow.add_tip(tip_length=0.15).add_tip(tip_length=0.15, at_start=True)
        
        h_arrow = Line(self.box_prediction.get_bottom(), self.box_prediction.get_top(), color=ORANGE).shift(LEFT * 0.1)
        h_arrow.add_tip(tip_length=0.15).add_tip(tip_length=0.15, at_start=True)

        self.play(Create(x_arrow), Create(y_arrow))
        self.play(Create(w_arrow), Create(h_arrow))
        
        self.visual_elements = VGroup(grid, cell_highlight, center_dot, gt_dashed, self.box_prediction, x_arrow, y_arrow, w_arrow, h_arrow)

    def explain_loss_function(self):
        bg_rect = Rectangle(width=14, height=2.5, fill_color=BLACK, fill_opacity=0.85, stroke_width=0).to_edge(UP, buff=0)
        self.add(bg_rect)

        self.loss_formula = MathTex(
            r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right)",
            r" + \sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)",
            font_size=30
        ).move_to(bg_rect.get_center())
        
        self.play(Write(self.loss_formula))

        coord_rect = SurroundingRectangle(self.loss_formula[0], color=GREEN)
        dim_rect = SurroundingRectangle(self.loss_formula[1], color=ORANGE)
        
        self.play(Create(coord_rect))
        self.wait(1)
        self.play(ReplacementTransform(coord_rect, dim_rect))
        self.dim_rect = dim_rect

    def animate_training_process(self):
        fade_out_group = VGroup(*[self.visual_elements[i] for i in [0, 1, 2, 5, 6, 7, 8]])
        self.play(FadeOut(fade_out_group), FadeOut(self.dim_rect))
        
        self.play(
            self.box_prediction.animate.move_to(self.gt_rect.get_center())
            .set_width(self.gt_rect.width)
            .set_height(self.gt_rect.height),
            self.loss_formula.animate.set_color(YELLOW),
            run_time=2
        )
        self.wait(2)
