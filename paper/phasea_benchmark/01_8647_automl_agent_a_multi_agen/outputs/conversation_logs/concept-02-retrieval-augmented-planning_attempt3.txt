class Concept02RetrievalAugmentedPlanning(Scene):

        # 1. Colors and Constants
        COLOR_MANAGER = BLUE
        COLOR_KB = "#888888" # GREY
        COLOR_PLAN_A = "#FFFF00" # YELLOW/GOLD
        COLOR_PLAN_B = "#00FF00" # GREEN
        COLOR_PLAN_C = "#FF0000" # RED

        # 2. Formula (Using Text instead of MathTex as it is disallowed)
        formula = Text("P = Amgr(RAP(R))", font_size=36).to_edge(UP)
        self.add(formula)

        # 3. Agent Manager (Center)
        mgr_rect = Rectangle(height=1.2, width=2.8, color=COLOR_MANAGER)
        mgr_label = Text("Agent Manager", font_size=20)
        agent_manager = VGroup(mgr_rect, mgr_label).move_to(ORIGIN)

        # 4. Knowledge Base (Left)
        # Constructing a cloud-like shape from circles
        c1 = Circle(radius=0.4, color=COLOR_KB, fill_opacity=0.6).shift(0.3 * (UP + LEFT))
        c2 = Circle(radius=0.4, color=COLOR_KB, fill_opacity=0.6).shift(0.3 * (UP + RIGHT))
        c3 = Circle(radius=0.5, color=COLOR_KB, fill_opacity=0.6).shift(0.1 * DOWN)
        kb_icon = VGroup(c1, c2, c3).shift(4 * LEFT)
        kb_text = Text("Knowledge Base", font_size=18).next_to(kb_icon, DOWN)
        kb_group = VGroup(kb_icon, kb_text)

        # 5. Animation Phase 1: Appearance
        self.play(FadeIn(agent_manager), FadeIn(kb_group))
        self.wait(0.5)

        # 6. Animation Phase 2: Knowledge flow (Arrow pulling into Manager)
        arrow = Arrow(kb_icon.get_right(), agent_manager.get_left(), color=WHITE)
        self.play(Create(arrow))
        self.wait(0.5)

        # 7. Animation Phase 3: Parallel Plans (Right)
        # Plan A - Triangle
        tri_pts = [[-0.3, -0.3, 0], [0.3, -0.3, 0], [0, 0.3, 0]]
        icon_a = Polygon(*tri_pts, color=COLOR_PLAN_A)
        label_a = Text("Plan A: Transformer", font_size=16).next_to(icon_a, RIGHT)
        plan_a = VGroup(icon_a, label_a)

        # Plan B - Square
        icon_b = Square(side_length=0.6, color=COLOR_PLAN_B)
        label_b = Text("Plan B: Standard CNN", font_size=16).next_to(icon_b, RIGHT)
        plan_b = VGroup(icon_b, label_b)

        # Plan C - Circle
        icon_c = Circle(radius=0.3, color=COLOR_PLAN_C)
        label_c = Text("Plan C: ML Strategy", font_size=16).next_to(icon_c, RIGHT)
        plan_c = VGroup(icon_c, label_c)

        plans_group = VGroup(plan_a, plan_b, plan_c).arrange(DOWN, buff=0.8).shift(4 * RIGHT)

        # Connecting lines from Manager to each Plan
        line_a = Line(agent_manager.get_right(), plan_a.get_left(), color=WHITE)
        line_b = Line(agent_manager.get_right(), plan_b.get_left(), color=WHITE)
        line_c = Line(agent_manager.get_right(), plan_c.get_left(), color=WHITE)

        # Animation of branching out
        self.play(
            Create(line_a), FadeIn(plan_a),
            Create(line_b), FadeIn(plan_b),
            Create(line_c), FadeIn(plan_c),
            run_time=2
        )

        # 8. Final Narration Text
        narration = Text(
            "Generating parallel diverse plans based on real-world evidence.",
            font_size=20
        ).to_edge(DOWN).shift(UP * 0.5)
        self.play(Write(narration))
        self.wait(2)
