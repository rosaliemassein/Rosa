from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup Title and Formula
        title = Text("Three-Stage Verification Loop", font_size=32, color=YELLOW).to_edge(UP)
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=36
        ).next_to(title, DOWN)
        self.add(title, formula)

        # 2. Define layout positions
        start_point = LEFT * 5 + DOWN * 1
        gate_x_coords = [-2, 0, 2]
        gate_labels = ["ReqVer", "ExecVer", "ImpVer"]
        deployment_point = RIGHT * 5 + DOWN * 1

        # 3. Create Gates and Labels
        gates = VGroup()
        labels = VGroup()
        for x, label_text in zip(gate_x_coords, gate_labels):
            # Create vertical lines as gates
            line = Line(UP * 0.5, DOWN * 2.5, color=WHITE).move_to([x, -1, 0])
            label = Text(label_text, font_size=20).next_to(line, UP)
            gates.add(line)
            labels.add(label)
        
        planning_label = Text("Planning", font_size=24, color=BLUE).move_to(start_point + UP * 1.5)
        deployment_label = Text("Deployment", font_size=24, color=GREEN).move_to(deployment_point + UP * 1.5)
        deployment_icon = Star(color=YELLOW, fill_opacity=1).scale(0.3).move_to(deployment_point)

        self.play(
            Create(gates), 
            Write(labels), 
            Write(planning_label), 
            Write(deployment_label), 
            FadeIn(deployment_icon)
        )

        # 4. Create the 'Plan' object (a small square)
        plan = Square(side_length=0.4, color=BLUE, fill_opacity=0.8).move_to(start_point)
        self.play(FadeIn(plan))

        # 5. Animation: Fail at the first gate and loop back
        # Move to first gate (ReqVer)
        self.play(plan.animate.move_to([gate_x_coords[0], -1, 0]), run_time=1)
        # Turn red to indicate failure
        self.play(plan.animate.set_color(RED))
        
        # Draw a feedback loop using ArcBetweenPoints (replaces CurvedArrow for better compatibility)
        loop_path = ArcBetweenPoints(
            plan.get_top(), 
            start_point + UP * 0.5, 
            angle=-TAU / 3,
            color=RED
        ).add_tip()
        
        self.play(Create(loop_path))
        self.play(
            plan.animate.move_to(start_point), 
            FadeOut(loop_path), 
            run_time=1.5
        )
        self.play(plan.animate.set_color(BLUE))

        # 6. Animation: Pass through all three verification stages
        for x in gate_x_coords:
            self.play(plan.animate.move_to([x, -1, 0]), run_time=0.8)
            # Indicate pass by turning green temporarily
            self.play(plan.animate.set_color(GREEN), run_time=0.2)
        
        # 7. Final Stage: Reach Deployment
        self.play(plan.animate.move_to(deployment_point), run_time=1)
        self.play(
            Indicate(deployment_icon, color=YELLOW, scale_factor=1.5),
            plan.animate.scale(0)
        )
        
        self.wait(2)
