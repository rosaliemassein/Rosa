from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup Formula and Labels
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=36
        ).to_edge(UP)
        self.add(formula)

        planning_pos = 4 * LEFT + 1 * UP
        planning_label = Text("Planning", font_size=24).move_to(planning_pos)
        self.add(planning_label)

        # 2. Setup Gates (Vertical Lines)
        gate_names = ["ReqVer", "ExecVer", "ImpVer"]
        gates = VGroup()
        for i, name in enumerate(gate_names):
            line = Line(UP, DOWN, color=WHITE).shift((2 * i - 1.5) * RIGHT)
            label = Text(name, font_size=20).next_to(line, UP)
            gates.add(VGroup(line, label))
        self.add(gates)

        # 3. Setup Deployment Icon
        deploy_pos = 4 * RIGHT + 0 * UP
        deploy_icon = VGroup(
            RegularPolygon(n=6, color=GOLD, fill_opacity=0.7).scale(0.4),
            Text("Deploy", font_size=20).next_to(ORIGIN, DOWN * 1.5)
        ).move_to(deploy_pos)
        self.add(deploy_icon)

        # 4. The Plan Object (Square)
        plan = Square(side_length=0.4, color=WHITE, fill_opacity=1).move_to(planning_pos)
        self.add(plan)

        # Narration setup
        narration = Text(
            "Reliability through a three-stage verification gate.",
            font_size=24,
            slant="ITALIC"
        ).to_edge(DOWN)
        self.play(Write(narration))

        # --- ANIMATION SEQUENCE ---

        # Step 1: Pass first gate
        self.play(plan.animate.move_to(gates[0][0].get_center()), run_time=1)
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.2)

        # Step 2: Fail second gate
        self.play(plan.animate.move_to(gates[1][0].get_center()), run_time=1)
        self.play(plan.animate.set_color(RED))
        self.wait(0.5)

        # Step 3: Feedback Loop
        # Using ArcBetweenPoints + add_tip as a robust alternative to CurvedArrow
        loop_path = ArcBetweenPoints(
            start=gates[1][0].get_center() + UP * 0.5,
            end=planning_pos + RIGHT * 0.5,
            angle=-PI/2,
            color=YELLOW
        ).add_tip()
        
        self.play(Create(loop_path))
        self.play(
            plan.animate.move_to(planning_pos).set_color(WHITE),
            run_time=1.5
        )
        self.play(FadeOut(loop_path))

        # Step 4: Full Successful Run
        for i in range(3):
            self.play(
                plan.animate.move_to(gates[i][0].get_center()).set_color(GREEN),
                run_time=0.7
            )
        
        # Step 5: Reach Deployment
        self.play(plan.animate.move_to(deploy_pos), run_time=1)
        self.play(Flash(deploy_icon, color=GOLD))
        self.play(plan.animate.scale(0).set_opacity(0))
        
        # Final Narration
        final_narration = Text(
            "Only deployment-ready models reach the finish line.",
            font_size=24,
            slant="ITALIC"
        ).to_edge(DOWN)
        self.play(Transform(narration, final_narration))
        self.wait(2)
