from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Technical Background
        plane = NumberPlane(
            background_line_style={
                "stroke_color": BLUE_D,
                "stroke_width": 1,
                "stroke_opacity": 0.2
            }
        )
        self.add(plane)

        # 2. Data Structures for the Pipeline
        titles = ["Data Retrieval", "Preprocessing", "Model Search", "Deployment"]
        circles = VGroup()
        labels = VGroup()
        nodes = VGroup()

        for title in titles:
            circle = Circle(radius=0.9, color=BLUE, fill_opacity=0.2)
            label = Text(title, font_size=20)
            label.move_to(circle.get_center())
            
            node = VGroup(circle, label)
            circles.add(circle)
            labels.add(label)
            nodes.add(node)

        # Arrange circles horizontally
        nodes.arrange(RIGHT, buff=1.0).shift(UP * 0.5)

        # 3. Connect them with thick arrows
        arrows = VGroup()
        for i in range(len(titles) - 1):
            arrow = Arrow(
                circles[i].get_right(),
                circles[i+1].get_left(),
                buff=0.1,
                stroke_width=10,
                color=BLUE,
                max_tip_length_to_stroke_width_ratio=2
            )
            arrows.add(arrow)

        # 4. Formula
        # Using MathTex with raw string for LaTeX compatibility
        formula = MathTex(
            r"\text{Pipeline} = \{ \text{Data} \to \text{Preprocessing} \to \text{Model} \to \text{HPO} \to \text{Deployment} \}",
            font_size=32
        ).to_edge(DOWN, buff=1.0)

        # --- ANIMATION ---

        # Entrance
        self.play(
            LaggedStart(
                Create(nodes),
                Create(arrows),
                Write(formula),
                lag_ratio=0.2
            ),
            run_time=3
        )
        self.wait(1)

        # "Ripples through" pulse animation
        # Start by pulsing the first node
        self.play(
            Indicate(circles[0], color=GOLD, scale_factor=1.3),
            labels[0].animate.set_color(GOLD),
            run_time=1
        )

        # Propagate the color change through the arrows to the end
        for i in range(len(arrows)):
            self.play(
                arrows[i].animate.set_color(GOLD),
                circles[i+1].animate.set_color(GOLD),
                labels[i+1].animate.set_color(GOLD),
                run_time=0.7
            )

        self.wait(2)
