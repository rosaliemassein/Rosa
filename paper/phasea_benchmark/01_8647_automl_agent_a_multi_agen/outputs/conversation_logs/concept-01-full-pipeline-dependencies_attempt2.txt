from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # Create a technical background
        number_plane = NumberPlane()
        number_plane.set_opacity(0.3)
        self.add(number_plane)

        # Pipeline step names from remarks
        step_names = ["Data Retrieval", "Preprocessing", "Model Search", "Deployment"]
        
        # Create circles
        circles = VGroup(*[
            Circle(radius=1.0, color=BLUE, fill_opacity=0.2) 
            for _ in range(len(step_names))
        ]).arrange(RIGHT, buff=1.2)
        
        # Labels for the circles
        labels = VGroup(*[
            Text(name, font_size=20).next_to(circles[i], UP)
            for i, name in enumerate(step_names)
        ])

        # Connect them with thick arrows
        arrows = VGroup(*[
            Arrow(
                circles[i].get_right(), 
                circles[i+1].get_left(), 
                buff=0.1, 
                stroke_width=10, 
                color=BLUE
            )
            for i in range(len(circles) - 1)
        ])

        # Formula at the bottom
        formula = MathTex(
            r"\text{Pipeline} = \{ \text{Data} \to \text{Preprocessing} \to \text{Model} \to \text{HPO} \to \text{Deployment} \}",
            font_size=30
        ).to_edge(DOWN, buff=1)

        # Initial display
        self.play(
            Create(circles),
            Write(labels),
            Create(arrows),
            run_time=2
        )
        self.play(Write(formula))
        self.wait(1)

        # Pulse animation on 'Data Retrieval' circle
        # This simulates the "ripples through" effect mentioned in voice
        data_circle = circles[0]
        pulse = data_circle.copy().set_color(YELLOW).set_stroke(width=12)
        
        self.play(
            pulse.animate.scale(1.5).set_stroke(opacity=0),
            data_circle.animate.set_color(YELLOW).set_fill(YELLOW, opacity=0.4),
            run_time=1
        )
        self.remove(pulse)

        # Propagate color change (ripple) through arrows and circles to Deployment
        for i in range(len(arrows)):
            self.play(
                arrows[i].animate.set_color(YELLOW),
                circles[i+1].animate.set_color(YELLOW).set_fill(YELLOW, opacity=0.4),
                run_time=0.7
            )

        # End-to-end highlight
        self.wait(2)
        self.play(
            VGroup(circles, labels, arrows).animate.scale(1.1),
            rate_func=there_and_back,
            run_time=2
        )
        self.wait(2)
