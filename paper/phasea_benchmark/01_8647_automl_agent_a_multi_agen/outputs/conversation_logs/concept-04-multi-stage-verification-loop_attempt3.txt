from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # Configuration and coordinates
        START_POS = LEFT * 4.5
        GATE_X_COORDS = [-1.5, 0.5, 2.5]
        END_POS = RIGHT * 5
        
        # 1. Background elements (Gates, Planning, Deployment)
        planning_box = RoundedRectangle(height=1.2, width=1.8, color=BLUE).move_to(START_POS)
        planning_label = Text("Planning Stage", font_size=24).next_to(planning_box, UP)
        
        gates = VGroup()
        gate_names = ["ReqVer", "ExecVer", "ImpVer"]
        for i, x in enumerate(GATE_X_COORDS):
            line = Line(UP * 0.8, DOWN * 0.8, color=WHITE)
            line.move_to(RIGHT * x)
            label = Text(gate_names[i], font_size=20).next_to(line, UP)
            gates.add(VGroup(line, label))
            
        deployment_icon = Star(color=YELLOW, fill_opacity=1).scale(0.4).move_to(END_POS)
        deployment_label = Text("Deployment", font_size=24).next_to(deployment_icon, UP)
        
        # Formula at the bottom
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=32
        ).to_edge(DOWN, buff=0.6)
        
        self.add(planning_box, planning_label, gates, deployment_icon, deployment_label, formula)

        # 2. The Plan Object
        plan = Square(side_length=0.4, fill_opacity=1, fill_color=BLUE, color=WHITE).move_to(START_POS)
        self.play(FadeIn(plan))

        # --- Iteration 1: Failure at the last gate ---
        
        # Move through first two gates
        for x in GATE_X_COORDS[:2]:
            self.play(plan.animate.move_to(RIGHT * x), run_time=0.8)
            self.play(plan.animate.set_fill(GREEN), run_time=0.3)
            
        # Move to last gate and fail
        self.play(plan.animate.move_to(RIGHT * GATE_X_COORDS[2]), run_time=0.8)
        self.play(plan.animate.set_fill(RED), run_time=0.3)
        
        # Feedback Loop: Draw a curved arrow back
        # Using ArcBetweenPoints and add_tip to avoid "CurvedArrow" undefined error
        loop_arc = ArcBetweenPoints(
            start=plan.get_center() + UP * 0.4,
            end=planning_box.get_center() + UP * 0.4,
            angle=-PI/2,
            color=RED
        )
        loop_arrow = loop_arc.add_tip()
        feedback_text = Text("Feedback to Manager", color=RED, font_size=18).next_to(loop_arc, UP, buff=0.1)
        
        self.play(Create(loop_arrow), Write(feedback_text))
        
        # Animate plan moving back along the curve
        self.play(
            plan.animate.move_to(START_POS),
            run_time=2,
            path_arc=-PI/2
        )
        self.play(FadeOut(loop_arrow), FadeOut(feedback_text), plan.animate.set_fill(BLUE))
        self.wait(0.5)

        # --- Iteration 2: Success through all gates ---
        for x in GATE_X_COORDS:
            self.play(plan.animate.move_to(RIGHT * x), run_time=0.6)
            self.play(plan.animate.set_fill(GREEN), run_time=0.2)
            
        # Final deployment
        self.play(plan.animate.move_to(END_POS), run_time=0.8)
        self.play(
            Transform(plan, deployment_icon),
            Flash(deployment_icon, color=YELLOW)
        )
        
        # Final caption using string for slant to avoid "ITALIC" undefined error
        final_caption = Text(
            "Reliability ensured through multi-stage verification.",
            font_size=24,
            slant="ITALIC"
        ).to_edge(UP, buff=0.5)
        
        self.play(Write(final_caption))
        self.wait(2)
