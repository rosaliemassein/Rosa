from manim import *
import numpy as np

class ConceptMatrixCosineSimilarity(Scene):
    def construct(self):
        # 1. Setup Axes and Vectors
        axes = Axes(
            x_range=[-0.5, 2.5, 1], 
            y_range=[-0.5, 2.5, 1], 
            x_length=4, 
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=1)
        
        # ValueTracker for the angle theta
        theta_tracker = ValueTracker(PI / 3)
        
        # Vector A (fixed) - Using Arrow to be safe
        vec_a = Arrow(axes.c2p(0, 0), axes.c2p(2, 0), buff=0, color=RED)
        vec_a_label = MathTex(r"\mathbf{A}", color=RED).next_to(vec_a.get_end(), DOWN)
        
        # Vector B (rotating)
        vec_b = always_redraw(lambda: Arrow(
            axes.c2p(0, 0),
            axes.c2p(2 * np.cos(theta_tracker.get_value()), 2 * np.sin(theta_tracker.get_value())),
            buff=0,
            color=BLUE
        ))
        
        vec_b_label = always_redraw(lambda: MathTex(r"\mathbf{B}", color=BLUE).next_to(vec_b.get_end(), UR, buff=0.1))

        # Angle Visualization using Arc
        angle_arc = always_redraw(lambda: Arc(
            radius=0.6,
            start_angle=0,
            angle=theta_tracker.get_value(),
            color=YELLOW,
            arc_center=axes.c2p(0, 0)
        ))
        
        theta_tex = always_redraw(lambda: MathTex(r"\theta").move_to(
            Arc(
                radius=0.9,
                start_angle=0,
                angle=theta_tracker.get_value(),
                arc_center=axes.c2p(0, 0)
            ).point_from_proportion(0.5)
        ))

        # 2. Labels and Formulas
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=32
        ).to_edge(UP, buff=0.5)
        
        sim_label = MathTex(r"\text{Similarity: }", font_size=36).to_edge(DOWN, buff=1.2).shift(LEFT * 1.5)
        sim_val = always_redraw(lambda: DecimalNumber(
            np.cos(theta_tracker.get_value()), 
            num_decimal_places=2,
            include_sign=True,
            font_size=36
        ).next_to(sim_label, RIGHT))

        # 3. Similarity Matrix
        matrix = VGroup(*[
            Square(side_length=0.5, fill_opacity=1, stroke_width=1, color=GRAY_E) 
            for _ in range(25)
        ]).arrange_in_grid(5, 5, buff=0.1).to_edge(RIGHT, buff=1)
        
        target_cell_index = 12 
        
        def update_matrix(m):
            val = np.cos(theta_tracker.get_value())
            # Simple color mapping logic to avoid potential undefined interpolate_color
            if val > 0.8:
                color = YELLOW
            elif val > 0.5:
                color = GOLD
            elif val > 0.2:
                color = ORANGE
            elif val > -0.2:
                color = BLUE_E
            else:
                color = BLACK
            
            m[target_cell_index].set_fill(color, opacity=1)
            for i, cell in enumerate(m):
                if i != target_cell_index:
                    cell.set_fill(BLUE_E, opacity=0.3)

        matrix.add_updater(update_matrix)

        # 4. Animations
        self.add(axes)
        self.play(Create(vec_a), Write(vec_a_label))
        self.play(Create(vec_b), Write(vec_b_label), Create(angle_arc), Write(theta_tex))
        self.play(Write(formula))
        self.play(Write(sim_label), Write(sim_val))
        self.wait(1)
        
        self.play(Create(matrix))
        self.wait(1)

        # Show angle changing and matrix cell reacting
        self.play(theta_tracker.animate.set_value(0.01), run_time=2)
        self.wait(1)
        
        self.play(theta_tracker.animate.set_value(PI / 2), run_time=2)
        self.wait(1)

        self.play(theta_tracker.animate.set_value(2 * PI / 3), run_time=2)
        self.wait(1)

        self.play(theta_tracker.animate.set_value(PI / 4), run_time=2)
        self.wait(2)

        matrix.remove_updater(update_matrix)
        self.play(FadeOut(VGroup(axes, vec_a, vec_a_label, vec_b, vec_b_label, angle_arc, theta_tex, formula, sim_label, sim_val, matrix)))
