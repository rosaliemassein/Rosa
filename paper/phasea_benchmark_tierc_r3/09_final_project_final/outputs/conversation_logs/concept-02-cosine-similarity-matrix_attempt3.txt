from manim import *
import numpy as np

class CosineSimilarityAnimation(Scene):
    def construct(self):
        # Setup colors and constants to avoid potential undefined identifier errors
        color_a = RED
        color_b = BLUE
        color_bg = GRAY
        color_highlight = YELLOW
        
        # 1. Setup Axes and Vectors
        axes = Axes(
            x_range=[-1.5, 1.5, 1], 
            y_range=[-1.5, 1.5, 1], 
            x_length=5, 
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        # ValueTracker for the angle theta
        theta_tracker = ValueTracker(60 * DEGREES)
        
        # Vector A - Static pointing along X-axis
        vec_a = Arrow(axes.get_center(), axes.c2p(1, 0, 0), buff=0, color=color_a)
        label_a = MathTex(r"\mathbf{A}", color=color_a).next_to(vec_a.get_end(), UP)
        
        # Vector B - Dynamic rotation
        vec_b = always_redraw(lambda: Arrow(
            axes.get_center(), 
            axes.c2p(np.cos(theta_tracker.get_value()), np.sin(theta_tracker.get_value()), 0), 
            buff=0, 
            color=color_b
        ))
        label_b = always_redraw(lambda: MathTex(r"\mathbf{B}", color=color_b).next_to(vec_b.get_end(), UR, buff=0.1))

        # Angle visualization using Arc (replaces Angle class for broader compatibility)
        angle_arc = always_redraw(lambda: Arc(
            radius=0.6,
            start_angle=0,
            angle=theta_tracker.get_value(),
            arc_center=axes.get_center(),
            color=color_highlight
        ))
        
        # Formula display
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=30
        ).to_edge(UP)
        
        # Similarity readout
        sim_text = Text("Similarity: ", font_size=24).to_corner(UL).shift(DOWN * 1.5)
        sim_val = always_redraw(lambda: DecimalNumber(
            np.cos(theta_tracker.get_value()), 
            num_decimal_places=2,
            color=color_highlight
        ).next_to(sim_text, RIGHT))

        # 2. Similarity Matrix (5x5 Grid)
        matrix = VGroup(*[
            Square(side_length=0.6, fill_opacity=1, stroke_width=1, color=color_bg)
            for _ in range(25)
        ]).arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=1)
        
        matrix_title = Text("Similarity Matrix", font_size=24).next_to(matrix, UP)
        target_cell = matrix[12] # Middle cell represents the current vectors

        # Update target cell color based on cosine similarity
        target_cell.add_updater(lambda m: m.set_fill(
            interpolate_color(BLACK, color_highlight, max(0, np.cos(theta_tracker.get_value()))), 
            opacity=1
        ))

        # 3. Animation Sequence
        self.play(Write(formula))
        self.play(Create(axes), Create(vec_a), Write(label_a))
        self.play(Create(vec_b), Write(label_b), Create(angle_arc))
        self.play(Create(matrix), Write(matrix_title), Write(sim_text), FadeIn(sim_val))
        self.wait(1)

        # Rotate B to align with A (Cosine Similarity approaches 1.0)
        # Using rate_functions.linear to ensure identifier is found
        self.play(
            theta_tracker.animate.set_value(0.001), 
            run_time=3,
            rate_func=rate_functions.linear
        )
        self.wait(1)

        # Move B away to 90 degrees (Cosine Similarity approaches 0.0)
        self.play(
            theta_tracker.animate.set_value(90 * DEGREES), 
            run_time=2,
            rate_func=rate_functions.smooth
        )
        self.wait(1)
        
        # Fill the rest of the matrix with random alignment colors for visual completeness
        other_cells = [c for i, c in enumerate(matrix) if i != 12]
        self.play(*[
            c.animate.set_fill(interpolate_color(BLUE, WHITE, np.random.rand()), opacity=0.7)
            for c in other_cells
        ], run_time=2)
        
        self.wait(2)
