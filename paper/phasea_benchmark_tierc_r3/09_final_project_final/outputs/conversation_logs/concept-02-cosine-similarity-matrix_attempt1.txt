from manim import *
import numpy as np

class Concept02CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Coordinate System
        axes = Axes(
            x_range=[-2, 2, 1], 
            y_range=[-2, 2, 1],
            x_length=4.5, 
            y_length=4.5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        origin = axes.get_origin()
        
        # Vector A (Reference)
        vec_a_coords = np.array([1.5, 0.5, 0])
        start_angle = np.arctan2(vec_a_coords[1], vec_a_coords[0])
        vec_a = Arrow(origin, axes.c2p(1.5, 0.5), buff=0, color=RED)
        label_a = MathTex(r"\mathbf{A}", color=RED).next_to(vec_a.get_end(), UR, buff=0.1)
        
        # Vector B (Moving)
        theta_tracker = ValueTracker(PI / 3) # Start at 60 degrees
        
        def get_b_end():
            angle = start_angle + theta_tracker.get_value()
            return axes.c2p(1.5 * np.cos(angle), 1.5 * np.sin(angle))
        
        vec_b = always_redraw(lambda: Arrow(
            origin, get_b_end(), buff=0, color=BLUE
        ))
        label_b = always_redraw(lambda: MathTex(r"\mathbf{B}", color=BLUE).next_to(vec_b.get_end(), UR, buff=0.1))
        
        # Angle Arc
        arc = always_redraw(lambda: Arc(
            radius=0.6,
            start_angle=start_angle,
            angle=theta_tracker.get_value(),
            arc_center=origin,
            color=YELLOW
        ))
        
        # 2. Labels and Formula
        formula = MathTex(
            r"\text{similarity} = \cos(\theta)",
            font_size=36
        ).to_edge(UP, buff=0.5)
        
        sim_value = always_redraw(lambda: MathTex(
            f"= {np.cos(theta_tracker.get_value()):.2f}",
            font_size=36,
            color=YELLOW
        ).next_to(formula, RIGHT))
        
        # 3. Similarity Matrix (5x5 Grid)
        matrix = VGroup(*[
            Square(side_length=0.5, fill_opacity=0.1, fill_color=BLUE, stroke_width=1)
            for _ in range(25)
        ]).arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=0.8)
        
        matrix_title = Text("Similarity Matrix", font_size=24).next_to(matrix, UP)
        
        # The specific cell (center) that represents our A-B pair
        active_cell = matrix[12]
        
        def update_active_cell(cell):
            cos_val = np.cos(theta_tracker.get_value())
            # Similarity of 1.0 (aligned) -> bright yellow
            # Similarity of 0.0 (orthogonal) -> dark blue/black
            if cos_val > 0:
                cell.set_fill(YELLOW, opacity=cos_val)
            else:
                cell.set_fill(BLUE, opacity=0.1)
                
        active_cell.add_updater(update_active_cell)
        
        # 4. Animation Sequence
        self.add(axes, vec_a, label_a, vec_b, label_b, arc, formula, sim_value, matrix, matrix_title)
        
        # Move B to align perfectly with A
        self.play(theta_tracker.animate.set_value(0), run_time=2)
        self.play(Indicate(active_cell, scale_factor=1.2))
        self.wait(1)
        
        # Move B to be perpendicular
        self.play(theta_tracker.animate.set_value(PI / 2), run_time=2)
        self.wait(1)
        
        # Move B to be opposite
        self.play(theta_tracker.animate.set_value(PI), run_time=2)
        self.wait(1)
        
        # Move B back to a strong overlap
        self.play(theta_tracker.animate.set_value(0.2), run_time=2)
        self.wait(2)
        
        active_cell.remove_updater(update_active_cell)
