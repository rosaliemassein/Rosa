from manim import *
import numpy as np

class SyntheticPileGeneration(Scene):
    def construct(self):
        # 1. Create a blank canvas representing the workspace
        canvas = Rectangle(width=8, height=6, color=WHITE, fill_opacity=0.1)
        canvas_label = Text("Canvas", font_size=24).next_to(canvas, UP)
        
        # 2. ValueTrackers for random scale and rotation values
        scale_tracker = ValueTracker(1.0)
        rot_tracker = ValueTracker(0)

        # 3. Parameter display labels
        scale_text = Text("Scale: ", font_size=24)
        scale_num = DecimalNumber(1.0).scale(0.8)
        rot_text = Text("Rotation: ", font_size=24)
        rot_num = DecimalNumber(0, unit="^\\circ").scale(0.8)

        # Group labels and numbers
        tracker_display = VGroup(
            VGroup(scale_text, scale_num).arrange(RIGHT),
            VGroup(rot_text, rot_num).arrange(RIGHT)
        ).arrange(DOWN, aligned_edge=LEFT).to_corner(UL)

        # Updaters for the trackers
        scale_num.add_updater(lambda m: m.set_value(scale_tracker.get_value()))
        rot_num.add_updater(lambda m: m.set_value(rot_tracker.get_value()))

        # 4. Define segmented trash icons (varied shapes/colors)
        base_icons = [
            Circle(radius=0.5, color=BLUE, fill_opacity=0.8), # Bottle/Can
            Square(side_length=0.7, color=GRAY, fill_opacity=0.8), # Cardboard
            Triangle(color=WHITE, fill_opacity=0.8).scale(0.6), # Paper
            Star(color=YELLOW, fill_opacity=0.8).scale(0.5), # Generic item
            RoundedRectangle(width=0.4, height=0.9, corner_radius=0.1, color=DARK_BROWN, fill_opacity=0.8) # Glass
        ]

        # Generate 9 objects to fulfill the "9 objects" remark
        moving_icons = VGroup()
        for i in range(9):
            icon = base_icons[i % 5].copy()
            # Position them floating around the canvas initially
            angle = i * (360/9) * DEGREES
            icon.move_to([5 * np.cos(angle), 3 * np.sin(angle), 0])
            moving_icons.add(icon)

        # 5. Animation sequence
        self.add(canvas, canvas_label, tracker_display)
        self.play(Create(canvas))

        # Data for randomized appearance
        scales_data = [1.2, 0.7, 1.4, 0.9, 1.1, 0.8, 1.3, 1.0, 0.75]
        rots_data = [45, 110, -30, 200, 25, 95, 310, 50, 175]
        # Target positions on the canvas
        pos_data = [
            [0, 0, 0], [1.8, 0.8, 0], [-1.2, 0.4, 0], 
            [0.6, -1.6, 0], [-2.1, -0.9, 0], [2.2, -0.6, 0],
            [-0.4, 1.4, 0], [1.3, -1.1, 0], [0.1, 1.1, 0]
        ]

        # Animate each icon onto the canvas with scaling and rotation
        for i in range(9):
            target_scale = scales_data[i]
            target_rot = rots_data[i]
            target_pos = pos_data[i]
            
            self.play(
                scale_tracker.animate.set_value(target_scale),
                rot_tracker.animate.set_value(target_rot),
                moving_icons[i].animate.move_to(target_pos).scale(target_scale).rotate(target_rot * DEGREES),
                run_time=0.6,
                rate_func=smooth
            )

        # Final pause to show the pile
        self.wait(2)
