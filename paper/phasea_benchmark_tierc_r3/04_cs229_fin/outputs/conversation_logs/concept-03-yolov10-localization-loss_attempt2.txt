from manim import *

class ConceptYolov10LocalizationLoss(Scene):
    def construct(self):
        # 1. Setup Background and 7x7 Grid
        # Using a background rectangle since external images might not be available
        image_bg = Rectangle(width=14, height=8, fill_color=DARK_GRAY, fill_opacity=0.4)
        image_text = Text("Image: Multiple Trash Items", font_size=24, color=WHITE, opacity=0.5).to_edge(UP)
        
        # 7x7 Grid (NumberPlane)
        # 7 units wide/high, centered at -3 on the x-axis to leave room for formula
        grid = NumberPlane(
            x_range=[-3.5, 3.5, 1],
            y_range=[-3.5, 3.5, 1],
            x_length=6,
            y_length=6,
            background_line_style={"stroke_color": BLUE, "stroke_width": 2, "stroke_opacity": 0.5},
            axis_config={"include_numbers": False}
        ).shift(LEFT * 3)
        
        self.add(image_bg, image_text, grid)

        # 2. Highlight a specific cell (soda can center)
        # Let's pick cell (1, 1) in coordinate space
        cell_highlight = Rectangle(width=6/7, height=6/7, color=YELLOW, stroke_width=4)
        cell_highlight.move_to(grid.c2p(1, 1))
        
        obj_center = Dot(grid.c2p(1.2, 0.8), color=RED, radius=0.08)
        obj_label = Text("Object Center", font_size=14, color=RED).next_to(obj_center, UR, buff=0.05)

        self.play(Create(cell_highlight), FadeIn(obj_center), Write(obj_label))
        self.wait(0.5)

        # 3. Ground Truth Box (Dashed) and Predicted Box (Solid)
        # GT Box
        gt_w, gt_h = 1.5, 2.0
        gt_rect_base = Rectangle(width=gt_w, height=gt_h, color=WHITE).move_to(grid.c2p(1.2, 0.8))
        # Create dashed effect using DashedLine segments to avoid "DashedVMobject" errors
        gt_corners = [gt_rect_base.get_corner(UL), gt_rect_base.get_corner(UR), gt_rect_base.get_corner(DR), gt_rect_base.get_corner(DL)]
        gt_dashed = VGroup(*[
            DashedLine(gt_corners[i], gt_corners[(i+1)%4], color=WHITE) for i in range(4)
        ])
        gt_label = Text("Ground Truth", font_size=14, color=WHITE).next_to(gt_dashed, UP, buff=0.1)

        # Predicted Box
        pred_w, pred_h = 2.0, 2.6
        pred_box = Rectangle(width=pred_w, height=pred_h, color=YELLOW, stroke_width=3).move_to(grid.c2p(1.7, 0.4))
        pred_label = Text("Prediction", font_size=14, color=YELLOW).next_to(pred_box, DOWN, buff=0.1)

        self.play(Create(gt_dashed), Write(gt_label))
        self.play(Create(pred_box), Write(pred_label))
        self.wait(0.5)

        # 4. Arrows for x, y distance and w, h difference
        arrow_x = Arrow(start=pred_box.get_center(), end=[gt_rect_base.get_x(), pred_box.get_y(), 0], buff=0, color=BLUE, stroke_width=3)
        arrow_y = Arrow(start=[gt_rect_base.get_x(), pred_box.get_y(), 0], end=gt_rect_base.get_center(), buff=0, color=GREEN, stroke_width=3)
        
        # Dimension difference indicators (width/height)
        arrow_w = DoubleArrow(
            start=[gt_rect_base.get_right()[0], gt_rect_base.get_y(), 0],
            end=[pred_box.get_right()[0], gt_rect_base.get_y(), 0],
            color=PINK, buff=0, tip_length=0.1
        )
        
        self.play(GrowArrow(arrow_x), GrowArrow(arrow_y))
        self.play(Create(arrow_w))
        self.wait(1)

        # 5. LaTeX Formula

            font_size=20
        ).to_edge(RIGHT, buff=0.3).shift(UP * 0.5)
        
        self.play(Write(formula))
        self.wait(1)

            run_time=1.5
        )

            run_time=1.5
