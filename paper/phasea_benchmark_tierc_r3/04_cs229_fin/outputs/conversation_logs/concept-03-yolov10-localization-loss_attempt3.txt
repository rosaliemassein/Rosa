from manim import *

class YOLOv10LocalizationLoss(Scene):
    def construct(self):
        # 1. Image and Grid Background
        try:
            image = ImageMobject("img-11.jpeg")
            image.height = 6
        except:
            image = Rectangle(width=8, height=6, color=GRAY, fill_opacity=0.2)
            label = Text("img-11.jpeg", font_size=24).move_to(image)
            image = Group(image, label)
        
        self.add(image)

        # 7x7 grid (NumberPlane)
        grid = NumberPlane(
            x_range=[-3.5, 3.5, 1],
            y_range=[-3.5, 3.5, 1],
            x_length=6,
            y_length=6,
            background_line_style={
                "stroke_color": BLUE,
                "stroke_width": 2,
                "stroke_opacity": 0.5
            },
            axis_config={"display_numbers": False}
        )
        self.add(grid)

        # 2. Highlight Specific Cell (center of item)
        cell_unit = grid.x_length / 7
        target_cell_pos = grid.c2p(1, 1) # Cell at (1,1)
        
        selected_cell = Rectangle(
            width=cell_unit,
            height=cell_unit,
            color=YELLOW,
            stroke_width=5,
            fill_opacity=0.3
        ).move_to(target_cell_pos)
        
        # 3. Define Boxes (Ground Truth and Predicted)
        # Ground Truth - using DashedVMobject (standard in Manim CE)
        gt_rect = Rectangle(
            width=cell_unit * 2.5,
            height=cell_unit * 1.5,
            color=RED,
            stroke_width=4
        ).move_to(grid.c2p(1.3, 1.2))
        ground_truth = DashedVMobject(gt_rect)

        # Predicted Box
        predicted_box = Rectangle(
            width=cell_unit * 1.8,
            height=cell_unit * 2.2,
            color=BLUE,
            stroke_width=4
        ).move_to(grid.c2p(0.7, 0.8))

        # 4. Animation Sequence
        self.play(Create(selected_cell))
        self.wait(0.5)
        self.play(Create(ground_truth), Create(predicted_box))
        
        # 5. Arrows for (x, y) and (w, h) differences
        p_center = predicted_box.get_center()
        g_center = ground_truth.get_center()
        
        # X and Y distance arrows
        x_arrow = Arrow(start=[p_center[0], p_center[1], 0], end=[g_center[0], p_center[1], 0], 
                        color=GREEN, buff=0, stroke_width=3, max_tip_length_to_length_ratio=0.2)
        y_arrow = Arrow(start=[g_center[0], p_center[1], 0], end=[g_center[0], g_center[1], 0], 
                        color=ORANGE, buff=0, stroke_width=3, max_tip_length_to_length_ratio=0.2)
        
        # Width/Height difference indicator (visualized as expansion arrows)
        w_arrow = DoubleArrow(
            start=predicted_box.get_right(),
            end=predicted_box.get_right() + RIGHT * (ground_truth.width - predicted_box.width)/2,
            color=PINK, buff=0, stroke_width=3
        )

        self.play(GrowFromCenter(x_arrow), GrowFromCenter(y_arrow))
        self.wait(0.5)

        # 6. Formula Presentation
        formula = MathTex(
            r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}\mathbb{1}_{ij}^{obj}\left[(x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right]",
            r"+ \sum_{i=0}^{S^{2}}\sum_{j=0}^{B}\mathbb{1}_{ij}^{obj}\left[(\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right]",
            font_size=30
        ).to_edge(DOWN, buff=0.3).set_background_stroke(color=BLACK, width=5)
        
        self.play(Write(formula))
        self.wait(1)

        # 7. Mock Training: Align Boxes
        # Highlight (x,y) part of formula
        self.play(formula[0].animate.set_color(YELLOW))
        self.play(
            predicted_box.animate.move_to(g_center),
            FadeOut(x_arrow),
            FadeOut(y_arrow),
            run_time=1.5
        )
        
        # Highlight (w,h) part of formula
        self.play(
            formula[0].animate.set_color(WHITE),
            formula[1].animate.set_color(YELLOW),
            GrowArrow(w_arrow)
        )
        
        self.play(
            predicted_box.animate.stretch_to_fit_width(gt_rect.width).stretch_to_fit_height(gt_rect.height),
            FadeOut(w_arrow),
            run_time=1.5
        )
        
        self.play(formula[1].animate.set_color(WHITE))
