from manim import *

class GranularSoftmaxClassification(Scene):
    def construct(self):
        # 1. Load and display the image of a plastic bottle on the left
        # In a real environment, 'img-12.jpeg' must be present.
        try:
            img = ImageMobject("img-12.jpeg").scale(1.2).to_edge(LEFT, buff=0.5)
        except:
            img = Rectangle(height=3, width=2.5, color=WHITE).to_edge(LEFT, buff=0.5)
            img.add(Text("Plastic Bottle", font_size=20).move_to(img))
        
        self.play(FadeIn(img))

        # 2. Draw CNN layers as vertical rectangles
        cnn_layers = VGroup(*[
            Rectangle(height=2.5, width=0.4, fill_opacity=0.6, color=BLUE_B, stroke_color=BLUE_A)
            for _ in range(4)
        ]).arrange(RIGHT, buff=0.5).next_to(img, RIGHT, buff=0.8)

        # 3. Draw arrows to show data flow
        arrow_to_cnn = Arrow(img.get_right(), cnn_layers.get_left(), buff=0.1)
        internal_arrows = VGroup(*[
            Arrow(cnn_layers[i].get_right(), cnn_layers[i+1].get_left(), buff=0.1)
            for i in range(len(cnn_layers) - 1)
        ])
        
        self.play(Create(cnn_layers), Create(arrow_to_cnn), Create(internal_arrows))

        # 4. Display raw scores as a column vector (MathTex labeled 'x_i')
        x_i_vector = MathTex(
            r"x_i = \begin{bmatrix} x_1 \\ x_2 \\ \vdots \\ x_{20} \end{bmatrix}"
        ).scale(0.8).next_to(cnn_layers, RIGHT, buff=1)
        
        arrow_to_vec = Arrow(cnn_layers.get_right(), x_i_vector.get_left(), buff=0.1)
        self.play(Create(arrow_to_vec), Write(x_i_vector))
        self.wait(1)

        # 5. Animate the Softmax formula appearing at the top
        softmax_formula = MathTex(
            r"softmax(x_{i})=\frac{e^{x_{i}}}{\sum_{j=1}^{n}e^{x_{j}}}"
        ).scale(0.85).to_edge(UP, buff=0.6)
        self.play(Write(softmax_formula))
        self.wait(1)

        # 6. Transform the raw values into a BarChart
        # We use Axes and Rectangles to manually build the bar chart for maximum compatibility
        axes = Axes(
            x_range=[0, 20, 1],
            y_range=[0, 1, 0.5],
            x_length=6,
            y_length=3,
            axis_config={"include_tip": False, "stroke_width": 2}
        ).scale(0.8).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)

        # Mock probability distribution (20 categories, category 16 is the "Plastic Bottle")
        probabilities = [0.03] * 20
        probabilities[15] = 0.82
        
        bars = VGroup()
        for i, prob in enumerate(probabilities):
            bar = Rectangle(
                width=axes.x_axis.get_unit_size() * 0.7,
                height=prob * axes.y_axis.get_unit_size(),
                fill_opacity=0.8,
                fill_color=BLUE_C,
                stroke_width=1
            )
            bar.move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN)
            bars.add(bar)

        # Transition: x_i vector transforms/fades into the probability bars
        self.play(
            FadeOut(x_i_vector),
            FadeOut(arrow_to_vec),
            Create(axes),
            Create(bars),
            run_time=2
        )

        # 7. Indicate the highest bar and label it 'Plastic Bottle'
        highest_bar = bars[15]
        self.play(Indicate(highest_bar, color=YELLOW, scale_factor=1.2))
        
        bottle_label = Text("Plastic Bottle", font_size=24, color=YELLOW).next_to(highest_bar, UP, buff=0.2)
        self.play(Write(bottle_label))

        self.wait(3)
