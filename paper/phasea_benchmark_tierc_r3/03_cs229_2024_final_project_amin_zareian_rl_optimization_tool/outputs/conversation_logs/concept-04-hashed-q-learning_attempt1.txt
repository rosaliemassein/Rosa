from manim import *

class HashedQLearningConcept(Scene):
    def construct(self):
        # Define Data
        voice_text = "To solve the memory problem, we use a Hashed Q-table. Instead of reserving a unique slot for every possible grid, we pass the grid state through a hash function to map it into a smaller, fixed-size table. This allows us to handle larger designs, though it can lead to collisions where different patterns share the same table entry, resulting in sub-optimal 'almost-checkered' designs."
        goal_text_str = "Understand the trade-off between memory efficiency and optimization accuracy in hashed RL."
        formula_str = r"\text{Index} = \text{Hash}(\text{MatrixState}) \pmod L"

        # 1. Goal and Formula
        goal_display = Text(goal_text_str, font_size=20, color=GREEN).to_edge(UP)
        formula_display = MathTex(formula_str, font_size=32).next_to(goal_display, DOWN, buff=0.3)
        self.add(goal_display, formula_display)

        # 2. Helper to create 4x4 grids
        def create_4x4_grid(pattern_type=0):
            grid = VGroup()
            for i in range(4):
                for j in range(4):
                    if pattern_type == 0: # Checkered
                        fill = WHITE if (i + j) % 2 == 0 else BLACK
                    elif pattern_type == 1: # Random-ish
                        fill = WHITE if (i * j) % 3 == 0 else BLACK
                    else: # Inverse Checkered
                        fill = BLACK if (i + j) % 2 == 0 else WHITE
                    
                    sq = Square(side_length=0.2, fill_opacity=1, fill_color=fill, stroke_width=1, stroke_color=GRAY)
                    sq.move_to([i * 0.2, j * 0.2, 0])
                    grid.add(sq)
            return grid

        # 3. Create Left Side Grids
        grid1 = create_4x4_grid(0)
        grid2 = create_4x4_grid(1)
        grid3 = create_4x4_grid(2)
        grids = VGroup(grid1, grid2, grid3).arrange(DOWN, buff=0.6).to_edge(LEFT, buff=1.5)
        
        # 4. Create Hash Box
        hash_box = Rectangle(width=2.5, height=1.5, color=BLUE)
        hash_label = Text("Hash Function", font_size=24).move_to(hash_box.get_center())
        hash_group = VGroup(hash_box, hash_label).move_to(ORIGIN)

        # 5. Create Q-Table
        table_rows = VGroup(*[
            VGroup(
                Rectangle(width=2, height=0.4, stroke_width=2),
                Text(f"Index {i}", font_size=14).move_to(ORIGIN)
            ) for i in range(6)
        ]).arrange(DOWN, buff=0).to_edge(RIGHT, buff=1.5)
        table_label = Text("Q-Table", font_size=20).next_to(table_rows, UP, buff=0.2)

        # 6. Arrows
        arrow_in1 = Arrow(grid1.get_right(), hash_box.get_left(), buff=0.1)
        arrow_in2 = Arrow(grid2.get_right(), hash_box.get_left(), buff=0.1)
        arrow_in3 = Arrow(grid3.get_right(), hash_box.get_left(), buff=0.1)

        arrow_out1 = Arrow(hash_box.get_right(), table_rows[1].get_left(), buff=0.1, color=YELLOW)
        arrow_out2 = Arrow(hash_box.get_right(), table_rows[4].get_left(), buff=0.1, color=YELLOW)
        arrow_out3 = Arrow(hash_box.get_right(), table_rows[1].get_left(), buff=0.1, color=RED)

        # 7. Animation Sequence
        self.play(FadeIn(grids))
        self.play(Create(hash_group))
        self.play(Create(table_rows), Write(table_label))
        self.wait(1)

        # Grid 1 -> Index 1
        self.play(Create(arrow_in1))
        self.play(Create(arrow_out1))
        self.play(table_rows[1].animate.set_fill(YELLOW, opacity=0.3))
        self.wait(0.5)

        # Grid 2 -> Index 4
        self.play(Create(arrow_in2))
        self.play(Create(arrow_out2))
        self.play(table_rows[4].animate.set_fill(YELLOW, opacity=0.3))
        self.wait(0.5)

        # Grid 3 -> Index 1 (Collision)
        self.play(Create(arrow_in3))
        self.play(Create(arrow_out3))
        
        collision_highlight = Rectangle(width=2.1, height=0.5, color=RED, stroke_width=4).move_to(table_rows[1].get_center())
        collision_text = Text("COLLISION", color=RED, font_size=18, weight="BOLD").next_to(collision_highlight, RIGHT, buff=0.2)
        
        self.play(
            Create(collision_highlight),
            FadeIn(collision_text),
            table_rows[1].animate.set_fill(RED, opacity=0.5)
        )
        
        # Flashing red effect
        self.play(Flash(table_rows[1], color=RED, flash_radius=0.5))
        # Custom there_and_back using sin for scaling pulse
        self.play(collision_highlight.animate(rate_func=lambda t: np.sin(PI * t)).scale(1.1), run_time=0.4)
        
        # Voiceover Text Rendering
        voice_display = Text(voice_text, font_size=14, line_spacing=1.2).to_edge(DOWN, buff=0.5)
        self.play(Write(voice_display), run_time=6)
        
        self.wait(2)
