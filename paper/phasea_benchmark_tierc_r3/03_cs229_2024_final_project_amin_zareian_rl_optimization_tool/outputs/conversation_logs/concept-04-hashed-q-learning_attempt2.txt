from manim import *

class Concept04HashedQLearning(Scene):
    def construct(self):
        # 1. Create the Grid Patterns (Left)
        def create_mini_grid(color_indices):
            grid = VGroup(*[Square(side_length=0.2) for _ in range(16)]).arrange_in_grid(4, 4, buff=0.05)
            for i in color_indices:
                grid[i].set_fill(BLUE, opacity=0.8)
            return grid

        grid1 = create_mini_grid([0, 2, 5, 7, 8, 10, 13, 15]).shift(LEFT * 5 + UP * 2)
        grid2 = create_mini_grid([1, 3, 4, 6, 9, 11, 12, 14]).shift(LEFT * 5)
        grid3 = create_mini_grid([0, 1, 2, 3]).shift(LEFT * 5 + DOWN * 2)
        grids = VGroup(grid1, grid2, grid3)

        # 2. Create the Hash Function Box (Center)
        hash_rect = Rectangle(width=3, height=2, color=BLUE)
        hash_text = Text("Hash Function", font_size=24)
        hash_box = VGroup(hash_rect, hash_text).move_to(ORIGIN)

        # 3. Create the Q-Table (Right)
        table_rows = VGroup(*[
            Rectangle(width=2, height=0.5) for _ in range(6)
        ]).arrange(DOWN, buff=0).shift(RIGHT * 5)
        table_label = Text("Q-Table", font_size=24).next_to(table_rows, UP)
        q_table = VGroup(table_rows, table_label)

        # 4. Create Arrows
        arrows_to_hash = VGroup(*[
            Arrow(g.get_right(), hash_box.get_left(), buff=0.1) for g in grids
        ])

        # From Hash Box to Table Rows
        arrow_out1 = Arrow(hash_box.get_right(), table_rows[1].get_left(), buff=0.1, color=WHITE)
        arrow_out2 = Arrow(hash_box.get_right(), table_rows[4].get_left(), buff=0.1, color=WHITE)

        # 5. Animations
        self.play(FadeIn(grids), FadeIn(hash_box), FadeIn(q_table))
        self.wait(1)

        # Draw Arrows to Hash
        self.play(Create(arrows_to_hash))
        self.wait(1)

        # Show Mapping 1: Grid 1 -> Row 1
        self.play(grid1.animate.set_color(YELLOW), Create(arrow_out1))
        self.play(table_rows[1].animate.set_fill(YELLOW, opacity=0.5))
        self.wait(0.5)

        # Show Mapping 2: Grid 2 -> Row 4
        self.play(grid2.animate.set_color(GREEN), Create(arrow_out2))
        self.play(table_rows[4].animate.set_fill(GREEN, opacity=0.5))
        self.wait(0.5)

        # Show Collision: Grid 3 -> Row 1
        collision_arrow = Arrow(hash_box.get_right(), table_rows[1].get_left(), buff=0.1, color=RED)
        self.play(grid3.animate.set_color(RED))
        self.play(Create(collision_arrow))
        
        # Flash for collision
        flash_box = Rectangle(width=2, height=0.5, color=RED, stroke_width=6).move_to(table_rows[1])
        self.play(FadeIn(flash_box))
        self.play(flash_box.animate.set_opacity(0), run_time=0.3, rate_func=there_and_back)
        self.play(flash_box.animate.set_opacity(1), run_time=0.3, rate_func=there_and_back)
        self.play(FadeOut(flash_box))

        # 6. Formula
        formula = MathTex(
            r"\text{Index} = \text{Hash}(\text{MatrixState}) \pmod L"

            width=formula.width + 0.4, 
            height=formula.height + 0.4, 
            fill_color=BLACK, 
            fill_opacity=0.8, 
            stroke_width=0
        ).move_to(formula)
        
        self.play(FadeIn(formula_bg), Write(formula))
        self.wait(3)

            *[FadeOut(mobj) for mobj in self.mobjects]
