from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Planes (Split Screen Layout)
        # Physical Space on the left
        plane_phys = NumberPlane(
            x_range=[-4, 4], y_range=[-4, 4],
            background_line_style={"stroke_opacity": 0.3}
        ).scale(0.5).to_edge(LEFT, buff=0.5)
        
        # Learned Space on the right
        plane_learn = NumberPlane(
            x_range=[-4, 4], y_range=[-4, 4],
            background_line_style={"stroke_opacity": 0.3}
        ).scale(0.5).to_edge(RIGHT, buff=0.5)
        
        title_phys = Text("Physical Space").scale(0.5).next_to(plane_phys, UP)
        title_learn = Text("Learned Space").scale(0.5).next_to(plane_learn, UP)

        # 2. Formula at bottom
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)"
        ).scale(0.7).to_edge(DOWN, buff=0.5)

        # 3. Create Hits (Dots)
        # Random seed for reproducibility
        np.random.seed(42)
        n_hits = 25
        
        # Physical positions: clusters are overlapping near center
        hits_a_phys = VGroup(*[
            Dot(plane_phys.c2p(np.random.normal(0.3, 0.6), np.random.normal(0.3, 0.6)), color=BLUE, radius=0.05)
            for _ in range(n_hits)
        ])
        hits_b_phys = VGroup(*[
            Dot(plane_phys.c2p(np.random.normal(-0.3, 0.6), np.random.normal(-0.3, 0.6)), color=RED, radius=0.05)
            for _ in range(n_hits)
        ])
        
        # Learned positions: clusters are pulled apart
        hits_a_learn = VGroup(*[
            Dot(plane_learn.c2p(np.random.normal(2, 0.4), np.random.normal(1.5, 0.4)), color=BLUE, radius=0.05)
            for _ in range(n_hits)
        ])
        hits_b_learn = VGroup(*[
            Dot(plane_learn.c2p(np.random.normal(-2, 0.4), np.random.normal(-1.5, 0.4)), color=RED, radius=0.05)
            for _ in range(n_hits)
        ])

        # 4. Classification circles in learned space
        # Using DashedVMobject which is standard in Manim
        circle_a = DashedVMobject(Circle(radius=0.7, color=BLUE)).move_to(plane_learn.c2p(2, 1.5))
        circle_b = DashedVMobject(Circle(radius=0.7, color=RED)).move_to(plane_learn.c2p(-2, -1.5))

        # 5. Animation Sequence
        # Initial display
        self.add(plane_phys, title_phys)
        self.play(FadeIn(hits_a_phys), FadeIn(hits_b_phys))
        self.wait(1)

        # Show Learned Plane and Formula
        self.play(Write(formula))
        self.play(Create(plane_learn), Write(title_learn))
        self.wait(1)

        # Transform the hits from Physical to Learned
        # rate_func=smooth is the standard version of "slow into"
        self.play(
            Transform(hits_a_phys, hits_a_learn),
            Transform(hits_b_phys, hits_b_learn),
            run_time=3,
            rate_func=smooth
        )
        self.wait(0.5)

        # Draw classification markers
        self.play(Create(circle_a), Create(circle_b))
        
        self.wait(2)
