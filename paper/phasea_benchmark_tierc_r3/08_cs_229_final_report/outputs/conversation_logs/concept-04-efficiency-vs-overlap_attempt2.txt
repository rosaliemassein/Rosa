from manim import *
import numpy as np

class Concept04EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Setup Axes
        # Using explicit names to avoid identifier conflicts
        axes_obj = Axes(
            x_range=[0, 4.5, 1],
            y_range=[0, 1.2, 0.2],
            x_length=9,
            y_length=4.5,
            axis_config={"include_tip": True}
        ).shift(DOWN * 0.8)
        
        # Axis labels
        labels_obj = axes_obj.get_axis_labels(
            x_label="Distance between particles", 
            y_label="Efficiency"
        ).scale(0.6)

        # 2. Plotting functions defined with 'def' to avoid lambda parameter issues
        def get_ml_performance(val_x):
            return 0.9

        def get_std_performance(val_x):
            # Logistic-like drop as distance closes
            if val_x > 0.4:
                return 0.9 * (1.0 - np.exp(-2.5 * (val_x - 0.4)))
            return 0.0

        ml_curve_obj = axes_obj.plot(get_ml_performance, color=BLUE, x_range=[0, 4.2])
        std_curve_obj = axes_obj.plot(get_std_performance, color=RED, x_range=[0, 4.2]).set_stroke(dash_pattern=[5, 5])
        
        # Legend labels
        ml_text = Text("ML Model", color=BLUE).scale(0.4).move_to(axes_obj.c2p(3.8, 1.0))
        std_text = Text("Standard Algorithm", color=RED).scale(0.4).move_to(axes_obj.c2p(3.8, 0.4))

        self.add(axes_obj, labels_obj, ml_curve_obj, std_curve_obj, ml_text, std_text)

        # 3. Dynamic Elements using ValueTracker
        dist_tracker = ValueTracker(4.0)

        # Vertical line indicating current position
        def get_v_line_dynamic():
            curr_pos = dist_tracker.get_value()
            return axes_obj.get_vertical_line(axes_obj.c2p(curr_pos, 0.9), color=YELLOW, stroke_opacity=0.6)

        v_line_obj = always_redraw(get_v_line_dynamic)

        # 4. Shower Merging Animation Icons
        # Pre-calculated offsets for particle hits
        np.random.seed(42)
        shower_offsets = []
        for _ in range(20):
            offset_val = np.array([np.random.uniform(-0.2, 0.2), np.random.uniform(-0.2, 0.2), 0])
            shower_offsets.append(offset_val)

        def get_shower_group():
            # Calculate distance based on tracker
            current_gap = dist_tracker.get_value() * 0.3
            y_center = 2.8
            
            # Left shower dots
            left_dots_list = []
            for i_idx in range(10):
                d_pos = UP * y_center + LEFT * current_gap + shower_offsets[i_idx]
                left_dots_list.append(Dot(radius=0.03, color=BLUE).move_to(d_pos))
            
            # Right shower dots
            right_dots_list = []
            for i_idx in range(10, 20):
                d_pos = UP * y_center + RIGHT * current_gap + shower_offsets[i_idx]
                right_dots_list.append(Dot(radius=0.03, color=WHITE).move_to(d_pos))
                
            return VGroup(VGroup(*left_dots_list), VGroup(*right_dots_list))

        showers_obj = always_redraw(get_shower_group)
        
        # Overlay Box and title
        box_obj = SurroundingRectangle(showers_obj, color=WHITE, buff=0.3, stroke_width=1)
        def update_box_pos(mob_to_move):
            mob_to_move.move_to(showers_obj.get_center())
        box_obj.add_updater(update_box_pos)
        
        title_obj = Text("Particle Detector View", color=WHITE).scale(0.35)
        def update_title_pos(mob_to_move):
            mob_to_move.next_to(box_obj, UP, buff=0.1)
        title_obj.add_updater(update_title_pos)

        self.add(v_line_obj, showers_obj, box_obj, title_obj)

        # 5. Narration Label
        narration_lbl = Text(
            "ML model resolves overlapping clusters that merge in standard methods.",
            font_size=22,
            color=GRAY_A
        ).to_edge(UP, buff=0.2)
        self.add(narration_lbl)

        # 6. Execution of Animation
        # Move tracker from high distance to low distance
        self.play(
            dist_tracker.animate.set_value(0.1),
            run_time=10,
            rate_func=rate_functions.linear
        )
        self.wait(2)
