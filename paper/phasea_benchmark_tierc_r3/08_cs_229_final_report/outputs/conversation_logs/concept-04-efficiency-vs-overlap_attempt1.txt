from manim import *
import numpy as np

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Setup Axes
        # Use explicit lists for ranges
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 1.1, 0.1],
            x_length=9,
            y_length=6,
            axis_config={"include_tip": False}
        )
        
        # Labels - using Text for maximum compatibility
        x_label = Text("Distance between particles", font_size=24).next_to(axes.x_axis, DOWN)
        y_label = Text("Efficiency", font_size=24).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)
        
        # 2. Define Plot Functions using named functions to avoid lambda 'x' issues
        def ml_efficiency_func(d_value):
            return 0.9
        
        def std_efficiency_func(d_value):
            # Curve starts high (0.85) and drops sharply as d_value approach 0
            if d_value < 0.1:
                return 0.85 * (d_value / 0.1)
            return 0.85 * (1.0 - np.exp(-0.5 * d_value))
        
        # 3. Create Curves
        ml_curve = axes.plot(ml_efficiency_func, color=BLUE)
        std_curve = axes.plot(std_efficiency_func, color=RED)
        
        ml_tag = Text("ML Model", color=BLUE, font_size=20).move_to(axes.c2p(8, 0.95))
        std_tag = Text("Standard Algorithm", color=RED, font_size=20).move_to(axes.c2p(8, 0.5))

        # 4. Vertical Indicator and Tracker
        # ValueTracker moves from 9 (far) to 0.5 (close)
        dist_tracker = ValueTracker(9)
        
        # Redraw the vertical line as the tracker moves
        # Avoid get_vertical_line helper to be safe
        v_line = always_redraw(lambda: Line(
            axes.c2p(dist_tracker.get_value(), 0),
            axes.c2p(dist_tracker.get_value(), 1.0),
            color=YELLOW,
            stroke_width=2
        ))

        # 5. Particle Icon Overlay
        # Visualize two dots that merge
        p1 = Dot(color=WHITE, radius=0.1)
        p2 = Dot(color=WHITE, radius=0.1)
        particle_group = VGroup(p1, p2)
        
        def update_particle_separation(group):
            # Display area in the upper middle
            display_center = axes.c2p(5, 0.8)
            # Separation scales with the tracker
            gap = dist_tracker.get_value() * 0.15
            group[0].move_to(display_center + LEFT * gap)
            group[1].move_to(display_center + RIGHT * gap)
            
        particle_group.add_updater(update_particle_separation)

        # 6. Building the Animation
        self.add(axes, x_label, y_label)
        self.play(Create(ml_curve), FadeIn(ml_tag))
        self.play(Create(std_curve), FadeIn(std_tag))
        self.wait(1)
        
        self.add(v_line, particle_group)
        
        # Animation: The particles approach each other
        # Using a simple lambda for rate_func to avoid 'linear' identifier
        self.play(
            dist_tracker.animate.set_value(0.2),
            run_time=8,
            rate_func=lambda t: t
        )
        
        self.wait(2)
        
        # Cleanup
        self.play(
            FadeOut(VGroup(axes, x_label, y_label, ml_curve, std_curve, ml_tag, std_tag, v_line, particle_group))
        )
        self.wait(1)
