from manim import *
import numpy as np

class Concept02ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup Title and Formula
        title = Text("Object Condensation", color=BLUE).scale(1.0).to_edge(UP)
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}").next_to(title, DOWN)
        
        self.play(Write(title))
        self.play(Write(formula))
        self.wait(1)

        # 2. Create Hits (Dots)
        num_dots = 40
        dots = VGroup()
        beta_values = np.random.uniform(0.1, 0.9, num_dots)
        
        # Determine 3 points that will be "Condensation Points" (highest beta)
        condensation_indices = np.argsort(beta_values)[-3:]
        
        # Use standard manim functions explicitly to avoid linter issues
        color_interpolator = utils.color.interpolate_color
        smooth_rate = rate_functions.smooth
        
        for i in range(num_dots):
            # Start as small purple dots scattered in a 2D plane
            pos = [np.random.uniform(-5, 5), np.random.uniform(-3, 1), 0]
            dot = Dot(pos, radius=0.06, color=PURPLE)
            dot.beta = beta_values[i]
            dots.add(dot)
        
        self.add(dots)
        self.play(Create(dots))
        self.wait(1)

        # 3. Animate Growth and Color Change (Beta Visualization)
        # High beta = Yellow and Large, Low beta = Purple and Small
        animations = []
        for i, dot in enumerate(dots):
            target_color = color_interpolator(PURPLE, YELLOW, dot.beta)
            # q calculation: q = arctanh(beta)
            q = np.arctanh(dot.beta)
            target_scale = 1.0 + q * 0.8
            animations.append(dot.animate.scale(target_scale).set_color(target_color))
        
        self.play(*animations, run_time=2)
        self.wait(1)

        # 4. Create "Gravity Wells" (Gradient circles) for high-beta dots
        wells = VGroup()
        for idx in condensation_indices:
            center_dot = dots[idx]
            # Create a concentric radial gradient effect using circles with decreasing opacity
            well = VGroup(*[
                Circle(
                    radius=r, 
                    stroke_width=0, 
                    fill_opacity=0.2 * (1 - r/1.5), 
                    fill_color=YELLOW
                ).move_to(center_dot.get_center())
                for r in np.linspace(0.1, 1.5, 10)
            ])
            wells.add(well)
        
        self.play(FadeIn(wells))
        self.wait(1)

        # 5. Pull neighboring dots toward condensation points
        # Each non-condensation dot moves toward the nearest condensation point
        pull_animations = []
        condensation_centers = [dots[idx].get_center() for idx in condensation_indices]
        
        for i, dot in enumerate(dots):
            if i in condensation_indices:
                continue
            
            # Find nearest center to simulate attraction
            distances = [np.linalg.norm(dot.get_center() - c) for c in condensation_centers]
            nearest_center = condensation_centers[np.argmin(distances)]
            
            # Calculate target: move dot towards the center proportional to its own beta (charge)
            pull_factor = 0.4 + (dot.beta * 0.4)
            target_pos = dot.get_center() + (nearest_center - dot.get_center()) * pull_factor
            pull_animations.append(dot.animate.move_to(target_pos))

        # Use explicitly referenced rate function
        self.play(*pull_animations, run_time=3, rate_func=smooth_rate)
        self.wait(2)

        # Final cleanup
        self.play(
            FadeOut(dots), 
            FadeOut(wells), 
            FadeOut(title), 
            FadeOut(formula)
        )
        self.wait(1)
