from manim import *
import numpy as np

class Concept02ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup: 2D plane of dots
        dots = VGroup(*[Dot(radius=0.08) for _ in range(24)])
        dots.arrange_in_grid(rows=4, cols=6, buff=1.0)
        self.add(dots)

        # 2. Formula label
        # Formula: q_i = arctanh(beta_i) + q_min
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}", color=WHITE)
        formula.to_edge(UP)
        self.play(Write(formula))
        self.wait(0.5)

        # 3. Define Condensation Points (High Beta)
        # We'll pick two dots to be the centers
        center_indices = [7, 16]
        centers = VGroup(dots[7], dots[16])
        
        # Color dots: Purple (low beta) to Yellow (high beta)
        animations = []
        for i, dot in enumerate(dots):
            if i in center_indices:
                beta = 0.9
                target_color = YELLOW
                scale_factor = 2.5
            else:
                beta = np.random.uniform(0.1, 0.4)
                # interpolate_color works with two colors and a weight
                target_color = interpolate_color(PURPLE, WHITE, beta)
                scale_factor = 1.0 + beta
            
            animations.append(dot.animate.set_color(target_color).scale(scale_factor))
        
        self.play(*animations, run_time=2)
        self.wait(0.5)

        # 4. Create "Gravity Wells" (Gradient Circles)
        # Using concentric circles to simulate a gradient potential well
        wells = VGroup()
        for center_dot in centers:
            well = VGroup(*[
                Circle(radius=r, stroke_width=0, fill_opacity=0.15 - (r/20))
                .set_fill(YELLOW, opacity=0.15 - (r/5))
                .move_to(center_dot.get_center())
                for r in np.arange(0.1, 2.5, 0.4)
            ])
            wells.add(well)
        
        self.play(FadeIn(wells, shift=UP * 0.2), run_time=1.5)

        # 5. Pull neighboring dots toward the centers
        pull_animations = []
        for i, dot in enumerate(dots):
            if i in center_indices:
                continue
            
            # Find closest center
            pos = dot.get_center()
            c1_pos = dots[7].get_center()
            c2_pos = dots[16].get_center()
            
            dist_to_c1 = np.linalg.norm(pos - c1_pos)
            dist_to_c2 = np.linalg.norm(pos - c2_pos)
            
            target_center_pos = c1_pos if dist_to_c1 < dist_to_c2 else c2_pos
            
            # Calculate a vector pointing toward the center
            # Move dots partially toward the condensation point
            direction = target_center_pos - pos
            pull_animations.append(
                dot.animate.shift(direction * 0.6)
            )
            
        self.play(*pull_animations, run_time=3, rate_func=smooth)
        self.wait(2)
