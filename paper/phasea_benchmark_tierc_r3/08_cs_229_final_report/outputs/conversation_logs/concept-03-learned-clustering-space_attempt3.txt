from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Split Screen Planes
        # Using BLUE and GREEN as safer alternatives to BLUE_E/GREEN_E
        left_plane = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-4, 4, 1],
            x_length=5,
            y_length=5,
            background_line_style={"stroke_color": BLUE, "stroke_opacity": 0.2}
        ).to_edge(LEFT, buff=0.7)

        right_plane = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-4, 4, 1],
            x_length=5,
            y_length=5,
            background_line_style={"stroke_color": GREEN, "stroke_opacity": 0.2}
        ).to_edge(RIGHT, buff=0.7)

        left_title = Text("Physical Space", font_size=24).next_to(left_plane, UP)
        right_title = Text("Learned Space", font_size=24).next_to(right_plane, UP)

        self.add(left_plane, right_plane, left_title, right_title)

        # 2. Create Overlapping Clusters in Physical Space
        np.random.seed(42)
        num_dots = 15
        
        # Cluster 1 (Red)
        dots_red = VGroup(*[
            Dot(left_plane.c2p(np.random.uniform(-0.6, 0.6), np.random.uniform(-0.6, 0.6), 0), 
                color=RED, radius=0.07)
            for _ in range(num_dots)
        ])
        
        # Cluster 2 (Yellow) - Heavily overlapping
        dots_yellow = VGroup(*[
            Dot(left_plane.c2p(np.random.uniform(-0.7, 0.5), np.random.uniform(-0.5, 0.7), 0), 
                color=YELLOW, radius=0.07)
            for _ in range(num_dots)
        ])

        self.add(dots_red, dots_yellow)
        self.wait(1)

        # 3. Transform Animation to Learned Space
        # Define separated target positions in the right plane
        red_targets = [
            right_plane.c2p(2 + np.random.normal(0, 0.3), 1.5 + np.random.normal(0, 0.3), 0)
            for _ in range(num_dots)
        ]
        yellow_targets = [
            right_plane.c2p(-2 + np.random.normal(0, 0.3), -1.5 + np.random.normal(0, 0.3), 0)
            for _ in range(num_dots)
        ]

        self.play(
            *[dots_red[i].animate.move_to(red_targets[i]) for i in range(num_dots)],
            *[dots_yellow[i].animate.move_to(yellow_targets[i]) for i in range(num_dots)],
            run_time=2.5,
            path_arc=0.5
        )

        # 4. Draw Dashed Circles (Manual implementation to avoid DashedVMobject issues)
        def get_custom_dashed_circle(center, color):
            dashed_circle = VGroup()
            n_segments = 20
            radius = 0.9
            for i in range(n_segments):
                start_angle = i * (2 * PI / n_segments)
                arc_angle = PI / n_segments # Half segment to create dash
                segment = Arc(radius=radius, start_angle=start_angle, angle=arc_angle, color=color, stroke_width=3)
                dashed_circle.add(segment)
            return dashed_circle.move_to(center)

        circle_red = get_custom_dashed_circle(right_plane.c2p(2, 1.5, 0), RED)
        circle_yellow = get_custom_dashed_circle(right_plane.c2p(-2, -1.5, 0), YELLOW)

        self.play(Create(circle_red), Create(circle_yellow))
        self.wait(1)

        # 5. Formula and Text
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=30
        ).to_edge(DOWN, buff=0.5)

        narration_sub = Text(
            "Mapping overlapping hits into distinct, separable blobs.",
            font_size=18,
            color=GRAY
        ).next_to(formula, UP, buff=0.2)

        self.play(Write(formula), FadeIn(narration_sub))
        self.wait(3)
