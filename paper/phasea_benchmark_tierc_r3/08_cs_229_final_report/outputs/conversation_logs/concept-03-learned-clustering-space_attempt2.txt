from manim import *
import numpy as np

class Concept03LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Planes (Split Screen)
        physical_plane = NumberPlane(
            x_range=[-4, 4, 1], y_range=[-4, 4, 1], 
            x_length=5, y_length=5,
            background_line_style={"stroke_opacity": 0.3}
        ).shift(LEFT * 3.5)
        
        learned_plane = NumberPlane(
            x_range=[-4, 4, 1], y_range=[-4, 4, 1],
            x_length=5, y_length=5,
            background_line_style={"stroke_opacity": 0.3}
        ).shift(RIGHT * 3.5)
        
        physical_title = Text("Physical Space", font_size=24).next_to(physical_plane, UP)
        learned_title = Text("Learned Space", font_size=24).next_to(learned_plane, UP)

        # 2. Formula at the top
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=32
        ).to_edge(UP, buff=0.3)

        self.add(physical_plane, learned_plane, physical_title, learned_title)
        self.play(Write(formula))
        self.wait(1)

        # 3. Create Overlapping Clusters in Physical Space
        n_hits = 20
        # Centers are very close to show overlap
        center_a_phys = physical_plane.c2p(0.3, 0.3)
        center_b_phys = physical_plane.c2p(-0.3, -0.3)
        
        np.random.seed(42) # For consistent look
        
        hits_a = VGroup(*[
            Dot(
                point=center_a_phys + np.array([np.random.normal(0, 0.6), np.random.normal(0, 0.6), 0]),
                color=BLUE, radius=0.07, fill_opacity=0.8
            ) for _ in range(n_hits)
        ])
        
        hits_b = VGroup(*[
            Dot(
                point=center_b_phys + np.array([np.random.normal(0, 0.6), np.random.normal(0, 0.6), 0]),
                color=RED, radius=0.07, fill_opacity=0.8
            ) for _ in range(n_hits)
        ])

        self.play(FadeIn(hits_a), FadeIn(hits_b))
        self.wait(1)

        # 4. Define Learned Space Positions (Clearly Separated)
        center_a_learn = learned_plane.c2p(-1.5, 1.5)
        center_b_learn = learned_plane.c2p(1.5, -1.5)

        hits_a_target = VGroup(*[
            Dot(
                point=center_a_learn + np.array([np.random.normal(0, 0.4), np.random.normal(0, 0.4), 0]),
                color=BLUE, radius=0.07, fill_opacity=0.8
            ) for _ in range(n_hits)
        ])

        hits_b_target = VGroup(*[
            Dot(
                point=center_b_learn + np.array([np.random.normal(0, 0.4), np.random.normal(0, 0.4), 0]),
                color=RED, radius=0.07, fill_opacity=0.8
            ) for _ in range(n_hits)
        ])

        # 5. Transform Animation
        self.play(
            Transform(hits_a, hits_a_target),
            Transform(hits_b, hits_b_target),
            run_time=3
        )
        self.wait(1)

        # 6. Helper function for dashed circles using Arc segments (avoids DashedVMobject issues)
        def get_dashed_circle(center, radius, color):
            dashes = VGroup()
            num_segments = 12
            for i in range(num_segments):
                start_angle = i * (TAU / num_segments)
                dash = Arc(
                    radius=radius, 
                    start_angle=start_angle, 
                    angle=TAU / (2 * num_segments), 
                    color=color,
                    stroke_width=2
                )
                dashes.add(dash)
            dashes.move_to(center)
            return dashes

        circle_a = get_dashed_circle(center_a_learn, 1.0, BLUE)
        circle_b = get_dashed_circle(center_b_learn, 1.0, RED)
        
        classification_label = Text("Hits Separated", font_size=24, color=GREEN).next_to(learned_plane, DOWN)

        self.play(
            Create(circle_a),
            Create(circle_b),
            Write(classification_label)
        )
        self.wait(2)

        # Cleanup
        self.play(
            FadeOut(formula),
            FadeOut(classification_label),
            FadeOut(hits_a),
            FadeOut(hits_b),
            FadeOut(circle_a),
            FadeOut(circle_b)
        )
        self.wait(1)
