from manim import *
import numpy as np

class ObjectCondensationExample(Scene):
    def construct(self):
        # 1. Setup Background and Formula
        formula = MathTex(
            r"q_i = \text{arctanh}(\beta_i) + q_{min}",
            font_size=40
        ).to_edge(UP, buff=0.5)
        
        # 2. Generate a 2D plane of hits (dots)
        num_dots = 30
        np.random.seed(42)
        dots = VGroup()
        beta_values = []
        
        # Define some indices as "condensation points" (high beta)
        condensation_indices = [5, 18, 24]
        
        for i in range(num_dots):
            pos = [np.random.uniform(-5, 5), np.random.uniform(-3, 1), 0]
            dot = Dot(pos, radius=0.06, color=PURPLE)
            
            # Assign beta values: high for centers, low for others
            if i in condensation_indices:
                beta = np.random.uniform(0.85, 0.98)
            else:
                beta = np.random.uniform(0.05, 0.4)
            
            dot.beta = beta
            dots.add(dot)
            beta_values.append(beta)

        self.add(dots)
        self.play(Write(formula))
        self.wait(1)

        # 3. Animate Beta-Charge growth and color change
        # Each dot grows and changes from Purple to Yellow based on beta
        growth_anims = []
        for dot in dots:
            # Interpolate manually to avoid "interpolate" identifier issues
            # Color interpolation using standard Manim utility
            target_color = interpolate_color(PURPLE, YELLOW, dot.beta)
            target_scale = 1.0 + (dot.beta * 2.5)
            growth_anims.append(
                dot.animate.set_color(target_color).scale(target_scale)
            )
        
        self.play(*growth_anims, run_time=2)
        self.wait(1)

        # 4. Create Gravity Wells for high-beta dots
        gravity_wells = VGroup()
        for idx in condensation_indices:
            center_dot = dots[idx]
            # Create a concentric gradient effect using several circles
            well = VGroup(*[
                Circle(
                    radius=r, 
                    stroke_width=0, 
                    fill_opacity=0.2 - (r * 0.05), 
                    color=YELLOW
                ).move_to(center_dot.get_center())
                for r in np.arange(0.2, 1.8, 0.3)
            ])
            gravity_wells.add(well)
            
        self.play(FadeIn(gravity_wells, scale=0.5), run_time=1.5)
        self.wait(1)

        # 5. Slide neighboring low-beta dots toward the center of wells
        pull_anims = []
        for i, dot in enumerate(dots):
            if i in condensation_indices:
                continue
            
            # Find the nearest condensation point
            distances = [
                np.linalg.norm(dot.get_center() - dots[c_idx].get_center()) 
                for c_idx in condensation_indices
            ]
            nearest_center_idx = condensation_indices[np.argmin(distances)]
            center_pos = dots[nearest_center_idx].get_center()
            
            # Calculate a pull vector (moving dots partially toward the center)
            direction = center_pos - dot.get_center()
            # The stronger the beta of the center, the stronger the pull
            pull_factor = dots[nearest_center_idx].beta * 0.75
            
            pull_anims.append(
                dot.animate.shift(direction * pull_factor)
            )

        # Use rate_functions.smooth for the sliding animation
        self.play(*pull_anims, run_time=3, rate_func=rate_functions.smooth)
        self.wait(2)

        # Final emphasis on the condensation centers
        flash_anims = [Flash(dots[idx], color=YELLOW, flash_radius=0.5) for idx in condensation_indices]
        self.play(*flash_anims)
        self.wait(2)
