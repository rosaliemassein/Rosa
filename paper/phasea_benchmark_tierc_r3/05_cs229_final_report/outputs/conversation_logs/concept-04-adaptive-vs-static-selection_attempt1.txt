import numpy as np
from manim import *

class AdaptiveVSStaticSelection(Scene):
    def construct(self):
        # 1. Background and Split Screen Setup
        self.camera.background_color = BLACK
        
        static_title = Text("Static Benchmark (ASHRAE)", font_size=28, color=BLUE).to_edge(UP).shift(LEFT * 3.5)
        adaptive_title = Text("Logistic Model (ML)", font_size=28, color=GREEN).to_edge(UP).shift(RIGHT * 3.5)
        divider = Line(UP * 4, DOWN * 1, color=WHITE)
        self.add(static_title, adaptive_title, divider)

        # 2. Floorplan Creation
        def create_floorplan(pos):
            floor = VGroup(*[Square(side_length=0.7, fill_opacity=0.2, color=GREY) for _ in range(9)])
            floor.arrange_in_grid(rows=3, cols=3, buff=0.15)
            floor.move_to(pos)
            return floor

        static_floor = create_floorplan(LEFT * 3.5 + UP * 1)
        adaptive_floor = create_floorplan(RIGHT * 3.5 + UP * 1)
        
        # Static side: Set specific zones to be permanently "flagged" (red)
        static_indices = [0, 4, 8]
        for i in static_indices:
            static_floor[i].set_fill(RED, opacity=0.8)
            static_floor[i].set_color(RED)
            
        self.add(static_floor, adaptive_floor)

        # 3. Timeline and Navigation
        timeline_line = Line(LEFT * 5, RIGHT * 5, color=GREY).shift(DOWN * 1.5)
        months = ["May", "Jun", "Jul", "Aug", "Sep"]
        month_labels = VGroup(*[Text(m, font_size=24) for m in months])
        month_labels.arrange(RIGHT, buff=1.5).next_to(timeline_line, DOWN)
        cursor = Dot(color=YELLOW).move_to(timeline_line.get_start())
        self.add(timeline_line, month_labels, cursor)

        # 4. Graphing (Performance Comparison)
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 100, 20],
            x_length=10,
            y_length=1.5,
            axis_config={"include_tip": False}
        ).to_edge(DOWN, buff=0.3)
        
        graph_label = Text("Missed Requests (Recall Error)", font_size=20).next_to(axes, UP, buff=0.1)
        
        # Tracking variable for synchronization
        time_tracker = ValueTracker(0)

        # Static Line: Consistently high error
        static_error_line = axes.plot(lambda x: 75, color=BLUE)
        static_tag = Text("Static", color=BLUE, font_size=18).next_to(static_error_line, RIGHT, buff=0.1)

        # ML Line: Dynamic and lower error
        # Note: 'x' is defined locally within the lambda passed to axes.plot
        ml_error_line = always_redraw(lambda: axes.plot(
            lambda x_val: 20 + 25 * np.exp(-((x_val - 5)**2) / 2), 
            x_range=[0, max(0.01, time_tracker.get_value())],
            color=GREEN
        ))
        
        self.add(axes, graph_label, static_error_line, static_tag, ml_error_line)

        # 5. Adaptive Logic (Updaters)
        def update_adaptive_floor(mob):
            t = time_tracker.get_value()
            # Heat intensity peaks in mid-summer (t=5)
            heat_intensity = np.exp(-((t - 5)**2) / 4)
            
            # Logic: More zones turn red as heat increases
            for i, square in enumerate(mob):
                # Use a pseudo-random but time-dependent threshold for flickering effect
                threshold = 0.2 + 0.6 * heat_intensity
                noise = (np.sin(i * 3 + t * 2) + 1) / 2
                if noise < threshold:
                    square.set_fill(RED, opacity=0.8)
                    square.set_color(RED)
                else:
                    square.set_fill(GREY, opacity=0.2)
                    square.set_color(GREY)

        adaptive_floor.add_updater(update_adaptive_floor)

        # 6. Animation Execution
        # Ensure 'linear' is accessed from the rate_functions module
        self.play(
            time_tracker.animate.set_value(10),
            cursor.animate.move_to(timeline_line.get_end()),
            run_time=12,
            rate_func=rate_functions.linear
        )
        
        self.wait(2)
