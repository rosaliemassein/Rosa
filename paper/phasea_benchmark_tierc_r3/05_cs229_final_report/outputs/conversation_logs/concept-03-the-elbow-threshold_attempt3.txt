from manim import *
import numpy as np

class TheElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes and Labels
        ax = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        labels = ax.get_axis_labels(
            x_label="Fraction\\ of\\ Zones", 
            y_label="Recall"
        )
        labels[0].scale(0.6).shift(DOWN * 0.3)
        labels[1].scale(0.6).shift(LEFT * 0.3)

        # 2. Formula
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.7).to_corner(UL)

        # 3. Curve (Concave, steep then levels off)
        # Using f(x) = (1 - e^(-5x)) / (1 - e^-5) to normalize [0,1]
        def elbow_func(x):
            return (1 - np.exp(-5 * x)) / (1 - np.exp(-5))
        
        plot = ax.plot(elbow_func, x_range=[0, 1], color=BLUE)
        
        # 4. ValueTracker and Moving Point
        tracker = ValueTracker(0.1)
        
        dot = Dot(color=YELLOW)
        dot.add_updater(lambda mobj: mobj.move_to(ax.c2p(tracker.get_value(), elbow_func(tracker.get_value()))))
        
        # 5. Side Panel: Floorplan
        # Creating a 5x5 grid of squares to represent zones
        floorplan_title = Text("Building Floorplan", font_size=24).to_edge(RIGHT, buff=1.5).shift(UP * 2.5)
        zones = VGroup(*[
            Square(side_length=0.4, stroke_width=2, color=WHITE) 
            for _ in range(25)
        ]).arrange_in_grid(rows=5, cols=5, buff=0.1).next_to(floorplan_title, DOWN, buff=0.5)

        def update_zones(group):
            # The number of zones highlighted corresponds to the tracker value (fraction of zones)
            num_to_highlight = int(tracker.get_value() * 25)
            for i, zone in enumerate(group):
                if i < num_to_highlight:
                    zone.set_fill(RED, opacity=0.8)
                else:
                    zone.set_fill(BLACK, opacity=0)

        zones.add_updater(update_zones)

        # 6. Elbow Highlight (Static circle at the "sweet spot")
        elbow_point_x = 0.35 # Approx elbow for -5x decay
        elbow_point = ax.c2p(elbow_point_x, elbow_func(elbow_point_x))
        elbow_circle = Circle(radius=0.3, color=YELLOW).move_to(elbow_point)
        elbow_glow = elbow_circle.copy().set_stroke(width=10, opacity=0.3)
        elbow_label = Text("The 'Elbow'", font_size=20, color=YELLOW).next_to(elbow_circle, UR, buff=0.1)
        elbow_group = VGroup(elbow_circle, elbow_glow, elbow_label)

        # 7. Animation sequence
        self.add(ax, labels, formula, floorplan_title, zones)
        self.play(Create(plot), run_time=2)
        self.add(dot)
        
        # Rate function is linear by default in some contexts, but we specify it for clarity
        self.play(tracker.animate.set_value(0.9), run_time=4, rate_func=linear)
        self.wait(0.5)
        
        self.play(tracker.animate.set_value(0.1), run_time=3, rate_func=linear)
        self.wait(0.5)
        
        self.play(tracker.animate.set_value(elbow_point_x), run_time=2)
        
        self.play(FadeIn(elbow_group))
        self.play(Indicate(elbow_group))
        
        # Final note
        note = Text("Optimizing Catch Rate vs. Efficiency", font_size=24, color=GREEN).next_to(zones, DOWN, buff=0.5)
        self.play(Write(note))
        
        self.wait(3)
