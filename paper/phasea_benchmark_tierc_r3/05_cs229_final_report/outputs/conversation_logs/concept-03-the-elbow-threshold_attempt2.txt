from manim import *
import numpy as np

class Concept03TheElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        x_label = axes.get_x_axis_label("Fraction\\ of\\ Zones\\ Flagged").scale(0.5)
        y_label = axes.get_y_axis_label("Recall").scale(0.5)
        
        # 2. Mathematical function for the "Elbow" curve (concave)
        def elbow_function(val):
            # Using val^0.25 to create a curve that rises sharply then levels off
            return val**0.25

        curve = axes.plot(elbow_function, x_range=[0.01, 1.0], color=BLUE)
        
        # 3. ValueTracker to control the "Fraction of Zones Flagged"
        tracker = ValueTracker(0.05)
        
        # 4. Moving Point on the curve
        moving_dot = Dot(color=YELLOW)
        moving_dot.add_updater(
            lambda mobj: mobj.move_to(
                axes.c2p(tracker.get_value(), elbow_function(tracker.get_value()))
            )
        )
        
        # 5. Building Floorplan (Side Panel)
        floorplan_title = Text("Building Floorplan", font_size=24).to_edge(RIGHT, buff=1.0).shift(UP * 3)
        
        # Create a grid of 25 zones
        zones = VGroup(*[
            Square(side_length=0.4, fill_opacity=0.1, stroke_width=1, color=WHITE)
            for _ in range(25)
        ]).arrange_in_grid(rows=5, cols=5, buff=0.1).next_to(floorplan_title, DOWN, buff=0.5)
        
        # Updater to light up squares based on the fraction tracker
        def update_zones(group):
            fraction = tracker.get_value()
            # How many of the 25 zones should be 'flagged' (yellow)
            num_flagged = int(fraction * 25)
            for idx, zone in enumerate(group):
                if idx < num_flagged:
                    zone.set_fill(YELLOW, opacity=0.8)
                    zone.set_stroke(YELLOW, width=2)
                else:
                    zone.set_fill(BLACK, opacity=0.1)
                    zone.set_stroke(WHITE, width=1)
        
        zones.add_updater(update_zones)
        
        # 6. Formula and Title
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_corner(UL)
        
        # --- ANIMATION ---
        
        # Initial display
        self.play(
            Create(axes),
            Write(x_label),
            Write(y_label),
            Write(formula),
            run_time=1.5
        )
        self.play(Create(curve), run_time=2)
        self.play(
            FadeIn(floorplan_title),
            Create(zones),
            FadeIn(moving_dot),
            run_time=1.5
        )
        self.wait(1)
        
        # Move right: flagging more zones (high recall, but low selectivity)
        # Using linear rate for steady movement
        self.play(tracker.animate.set_value(0.9), run_time=4, rate_func=linear)
        self.wait(1)
        
        # Move left: flagging fewer zones (high selectivity, but missing recall)
        self.play(tracker.animate.set_value(0.02), run_time=3, rate_func=linear)
        self.wait(1)
        
        # Move to the "Elbow" point
        elbow_fraction = 0.15
        self.play(tracker.animate.set_value(elbow_fraction), run_time=2)
        
        # Highlight the elbow with a glowing effect
        elbow_coord = axes.c2p(elbow_fraction, elbow_function(elbow_fraction))
        glow_circles = VGroup(*[
            Circle(radius=r, color=YELLOW, stroke_opacity=1 - (r/0.4), stroke_width=2)
            for r in [0.1, 0.2, 0.3, 0.4]
        ]).move_to(elbow_coord)
        
        elbow_text = Text("The 'Elbow'", color=YELLOW, font_size=24).next_to(glow_circles, UR, buff=0.2)
        
        self.play(
            FadeIn(glow_circles),
            Write(elbow_text),
            Flash(elbow_coord, color=YELLOW)
        )
        
        self.wait(3)
        
        # Cleanup
        self.play(
            FadeOut(VGroup(axes, x_label, y_label, curve, moving_dot, formula, zones, floorplan_title, glow_circles, elbow_text)),
            run_time=1.5
        )
