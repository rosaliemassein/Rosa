from manim import *
import numpy as np

class Concept03TheElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True},
            x_axis_config={"numbers_to_include": [0, 0.5, 1]},
            y_axis_config={"numbers_to_include": [0, 0.5, 1]},
        ).to_edge(LEFT, buff=0.5)
        
        labels = axes.get_axis_labels(
            x_label=Text("Fraction of Zones Flagged", font_size=18),
            y_label=Text("Recall", font_size=18)
        )

        # 2. Define the Curve (Concave, rising sharply then leveling off)
        # Using 1 - exp(-k*x) normalized to [0,1]
        def curve_func(x_val):
            k = 5
            return (1 - np.exp(-k * x_val)) / (1 - np.exp(-k))
        
        curve = axes.plot(curve_func, x_range=[0, 1], color=BLUE)
        
        # 3. Floorplan Visualization (Side Panel)
        # Create a grid of squares representing building zones
        floorplan = VGroup(*[
            Square(side_length=0.4, stroke_width=2, color=WHITE) 
            for _ in range(20)
        ]).arrange_in_grid(rows=4, cols=5, buff=0.1)
        floorplan.to_corner(UP + RIGHT, buff=1)
        
        floorplan_label = Text("Zone Status", font_size=24).next_to(floorplan, UP)
        
        # 4. Interactive Elements (ValueTracker and Dot)
        tracker = ValueTracker(0.1)
        
        dot = Dot(color=YELLOW)
        # Fixed the lambda variable to 'd' to avoid potential identifier issues
        dot.add_updater(lambda d: d.move_to(axes.c2p(tracker.get_value(), curve_func(tracker.get_value()))))
        
        # Updater for floorplan squares lighting up based on fraction flagged
        def update_squares(group):
            val = tracker.get_value()
            num_to_light = int(val * 20)
            for i, sq in enumerate(group):
                if i < num_to_light:
                    sq.set_fill(GOLD, opacity=0.8)
                else:
                    sq.set_fill(BLACK, opacity=0)
                    
        floorplan.add_updater(update_squares)

        # 5. Narration Text (Side Panel)
        narration_text = Text(
            "We have to decide on a \nprobability threshold.",
            font_size=22,
            line_spacing=1.2
        ).next_to(floorplan, DOWN, buff=1)

        # 6. The Animation Sequence
        self.add(axes, labels, curve, dot, floorplan, floorplan_label, narration_text)
        
        # Move tracker to the right (Flagging many zones)
        # Replaced 'linear' identifier with a lambda to ensure compatibility
        self.play(tracker.animate.set_value(0.9), run_time=3, rate_func=lambda t: t)
        
        mid_text = Text(
            "Too low: many zones flagged.\nToo high: we miss requests.",
            font_size=22,
            line_spacing=1.2,
            t2c={"low": RED, "high": RED}
        ).move_to(narration_text)
        
        self.play(Transform(narration_text, mid_text))
        self.play(tracker.animate.set_value(0.05), run_time=2, rate_func=lambda t: t)
        self.wait(0.5)
        
        # Move to the Elbow point
        elbow_val = 0.35
        final_text = Text(
            "The 'elbow' is the sweet spot.",
            font_size=22,
            color=GOLD
        ).move_to(narration_text)
        
        self.play(Transform(narration_text, final_text))
        self.play(tracker.animate.set_value(elbow_val), run_time=1.5)
        
        # Highlight the Elbow with a glowing circle
        elbow_coord = axes.c2p(elbow_val, curve_func(elbow_val))
        elbow_circle = Circle(radius=0.4, color=GOLD).move_to(elbow_coord)
        elbow_circle.set_stroke(width=6)
        
        self.play(Create(elbow_circle))
        self.play(Flash(elbow_circle, color=GOLD, flash_radius=0.5))
        
        # Show formula reference
        formula = MathTex(r"\text{Recall} = \frac{TP}{TP + FN}", font_size=30)
        formula.to_corner(DL, buff=0.5)
        self.play(Write(formula))
        
        self.wait(3)
