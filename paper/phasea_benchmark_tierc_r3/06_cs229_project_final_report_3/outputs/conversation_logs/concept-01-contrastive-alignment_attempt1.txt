from manim import *

class ContrastiveAlignment(Scene):
    def construct(self):
        # 1. Setup Encoders and UI
        img_enc_rect = Rectangle(height=1.2, width=2.5, color=BLUE)
        img_enc_text = Text("Image Encoder", font_size=20).move_to(img_enc_rect.get_center())
        img_enc = VGroup(img_enc_rect, img_enc_text)

        txt_enc_rect = Rectangle(height=1.2, width=2.5, color=GREEN)
        txt_enc_text = Text("Text Encoder", font_size=20).move_to(txt_enc_rect.get_center())
        txt_enc = VGroup(txt_enc_rect, txt_enc_text)
        
        encoders = VGroup(img_enc, txt_enc).arrange(DOWN, buff=1.5).to_edge(LEFT, buff=1)
        
        input_img = Square(side_length=1, fill_opacity=0.8, fill_color=BLUE).next_to(img_enc, LEFT, buff=1.5)
        input_img_label = Text("Street View", font_size=16).next_to(input_img, DOWN)
        
        input_txt_fr = Text("'...in France'", font_size=18).next_to(txt_enc, LEFT, buff=1.5)
        
        # 2. Embedding Space
        axes = Axes(
            x_range=[-1.5, 1.5, 1], 
            y_range=[-1.5, 1.5, 1], 
            x_length=4, 
            y_length=4, 
            axis_config={"include_tip": True}
        ).to_edge(RIGHT, buff=1)
        space_label = Text("Shared Embedding Space", font_size=22).next_to(axes, UP)
        origin_point = axes.coords_to_point(0, 0)
        
        # 3. Vectors and Labels
        v_img = Arrow(origin_point, axes.coords_to_point(1, 0.5), buff=0, color=YELLOW)
        v_txt = Arrow(origin_point, axes.coords_to_point(-0.5, 1), buff=0, color=GREEN)
        
        v_img_lab = Text("Image", color=YELLOW, font_size=18)
        v_img_lab.add_updater(lambda mob: mob.next_to(v_img.get_end(), UR, buff=0.1))
        
        v_txt_lab = Text("Text (France)", color=GREEN, font_size=18)
        v_txt_lab.add_updater(lambda mob: mob.next_to(v_txt.get_end(), UL, buff=0.1))

        # 4. Similarity Score and Formula
        sim_tracker = ValueTracker(0.12)
        sim_label = Text("Cosine Similarity:", font_size=24).to_edge(DOWN, buff=1.2).shift(LEFT * 1)
        sim_val = DecimalNumber(sim_tracker.get_value(), num_decimal_places=2, color=WHITE).next_to(sim_label, RIGHT)
        sim_val.add_updater(lambda mob: mob.set_value(sim_tracker.get_value()))

        formula = MathTex(r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})", font_size=28).to_edge(DOWN, buff=0.4)

        # --- ANIMATION ---
        self.add(encoders, space_label, axes, formula)
        self.play(FadeIn(input_img), FadeIn(input_img_label), FadeIn(input_txt_fr))
        self.wait(0.5)
        
        # Process Inputs
        self.play(
            input_img.animate.move_to(img_enc.get_center()).scale(0.2).set_opacity(0),
            input_txt_fr.animate.move_to(txt_enc.get_center()).scale(0.2).set_opacity(0),
            run_time=1.5
        )
        
        # Show Vectors
        self.play(GrowArrow(v_img), GrowArrow(v_txt), FadeIn(v_img_lab), FadeIn(v_txt_lab))
        self.play(FadeIn(sim_label), FadeIn(sim_val))
        self.wait(1)

        # Align Vectors (Positive Pairing)
        target_v_img = Arrow(origin_point, axes.coords_to_point(1, 1), buff=0, color=YELLOW)
        target_v_txt = Arrow(origin_point, axes.coords_to_point(1, 1), buff=0, color=GREEN)
        
        self.play(
            Transform(v_img, target_v_img),
            Transform(v_txt, target_v_txt),
            sim_tracker.animate.set_value(1.00),
            run_time=2.5
        )
        self.wait(1)

        # Contrastive Step (Negative Pairing)
        input_txt_jp = Text("'...in Japan'", font_size=18).move_to(txt_enc.get_left()).shift(LEFT * 1.5)
        self.play(FadeIn(input_txt_jp))
        
        self.play(
            input_txt_jp.animate.move_to(txt_enc.get_center()).scale(0.2).set_opacity(0),
            run_time=1
        )
        
        # Switch text vector label and push away
        v_txt_lab_new = Text("Text (Japan)", color=RED, font_size=18)
        # We manually manage the label transformation to avoid conflicts with updaters
        v_txt_lab.clear_updaters()
        v_txt_lab_new.add_updater(lambda mob: mob.next_to(v_txt.get_end(), DL, buff=0.1))
        
        pushed_v_txt = Arrow(origin_point, axes.coords_to_point(-1, -0.5), buff=0, color=RED)
        
        self.play(
            v_txt.animate.set_color(RED),
            Transform(v_txt_lab, v_txt_lab_new),
            Transform(v_txt, pushed_v_txt),
            sim_tracker.animate.set_value(-0.45),
            run_time=2.5
        )
        self.wait(2)
        
        # Final cleanup
        v_img_lab.clear_updaters()
        v_txt_lab_new.clear_updaters()
        sim_val.clear_updaters()
        self.play(FadeOut(VGroup(v_img, v_txt, v_img_lab, v_txt_lab, sim_label, sim_val)))
        self.wait(1)
