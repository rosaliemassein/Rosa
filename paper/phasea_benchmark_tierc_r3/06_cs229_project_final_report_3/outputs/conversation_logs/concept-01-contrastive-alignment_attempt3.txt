from manim import *
import numpy as np

class ContrastiveAlignment(Scene):
    def construct(self):
        # 1. Setup Neural Network Encoders
        image_encoder = VGroup(
            Rectangle(width=3, height=1.5, color=BLUE),
            Text("Image Encoder", font_size=24)
        ).shift(UP * 2.5 + LEFT * 3.5)

        text_encoder = VGroup(
            Rectangle(width=3, height=1.5, color=GREEN),
            Text("Text Encoder", font_size=24)
        ).shift(UP * 2.5 + RIGHT * 3.5)

        # Formula at the top
        formula = MathTex(
            r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})",
            font_size=36
        ).to_edge(UP, buff=0.2)

        # 2. Setup Inputs
        # Placeholder if image file is missing
        try:
            street_view_img = ImageMobject("img-7.jpeg").scale(0.5).move_to(LEFT * 5 + DOWN * 2)
        except:
            street_view_img = VGroup(
                Rectangle(width=2, height=1.5, fill_opacity=0.3, color=GRAY),
                Text("Street View", font_size=20)
            ).move_to(LEFT * 5 + DOWN * 2)

        france_text = Text("A Street View photo in France", font_size=22, color=YELLOW).move_to(RIGHT * 5 + DOWN * 2)
        japan_text = Text("A Street View photo in Japan", font_size=22, color=RED).move_to(RIGHT * 5 + DOWN * 2)

        # 3. Coordinate System (Shared Embedding Space)
        axes = Axes(
            x_range=[-1.5, 1.5], y_range=[-1.5, 1.5],
            x_length=4, y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(DOWN, buff=0.5).shift(LEFT * 1)
        space_label = Text("Shared Embedding Space", font_size=24).next_to(axes, UP, buff=0.2)

        # 4. Vectors (using Arrow instead of Vector for compatibility)
        origin_pt = axes.get_origin()
        
        # Image vector is fixed for this demonstration
        img_arrow = Arrow(origin_pt, axes.c2p(1, 1, 0), buff=0, color=BLUE)
        img_label = Text("Image", font_size=18, color=BLUE).next_to(img_arrow.get_end(), UR, buff=0.1)

        # Text vector uses a tracker for rotation
        angle_tracker = ValueTracker(0.1) # Starting angle in radians
        
        # Update function for the text arrow
        def get_dynamic_text_arrow():
            angle = angle_tracker.get_value()
            target = axes.c2p(np.cos(angle) * 1.3, np.sin(angle) * 1.3, 0)
            return Arrow(origin_pt, target, buff=0, color=GREEN)

        text_arrow = always_redraw(get_dynamic_text_arrow)
        text_label = Text("Text", font_size=18, color=GREEN)
        text_label.add_updater(lambda m: m.next_to(text_arrow.get_end(), UP + RIGHT, buff=0.1))

        # 5. Similarity Score
        sim_tracker = ValueTracker(0.1)
        sim_group = VGroup(
            Text("Similarity Score: ", font_size=28),
            DecimalNumber(0, num_decimal_places=2, color=YELLOW).scale(1.2)
        ).arrange(RIGHT).to_edge(RIGHT).shift(UP * 0.5)

        def update_similarity_text(obj):
            obj.set_value(sim_tracker.get_value())
        
        sim_group[1].add_updater(update_similarity_text)

        # --- ANIMATION SEQUENCE ---

        # Show structure
        self.play(FadeIn(image_encoder), FadeIn(text_encoder), Write(formula))
        self.wait(0.5)

        # Inputs enter encoders
        self.play(FadeIn(street_view_img), FadeIn(france_text))
        self.play(
            street_view_img.animate.move_to(image_encoder.get_center()).scale(0.1).set_opacity(0),
            france_text.animate.move_to(text_encoder.get_center()).scale(0.1).set_opacity(0),
            run_time=1.5
        )

        # Show Embedding Space
        self.play(Create(axes), Write(space_label))
        self.play(GrowArrow(img_arrow), Write(img_label))
        self.play(FadeIn(text_arrow), Write(text_label), FadeIn(sim_group))
        self.wait(1)

        # Align (Correct Label: France)
        # Cosine similarity increases as they point in the same direction
        self.play(
            angle_tracker.animate.set_value(PI/4),
            sim_tracker.animate.set_value(1.0),
            run_time=2,
            rate_func=smooth
        )
        self.wait(1)

        # Wrong Label (Japan)
        self.play(
            ReplacementTransform(Text("France", font_size=1).move_to(text_encoder), japan_text.scale(0.1).move_to(text_encoder)),
            sim_tracker.animate.set_value(0.9)
        )
        
        # Push Away (Incorrect Label)
        self.play(
            angle_tracker.animate.set_value(PI + PI/6),
            sim_tracker.animate.set_value(-0.85),
            run_time=2.5,
            rate_func=smooth
        )
        self.wait(2)
