from manim import *

class ContrastiveAlignment(Scene):
    def construct(self):
        # 1. Setup Encoders
        # Using ORIGIN instead of CENTER to avoid potential undefined identifier issues
        img_enc = VGroup(
            Rectangle(height=1.5, width=2.5, color=BLUE),
            Text("Image Encoder", font_size=20)
        ).arrange(DOWN, buff=0.1).to_edge(UP).shift(LEFT * 3)
        
        txt_enc = VGroup(
            Rectangle(height=1.5, width=2.5, color=GREEN),
            Text("Text Encoder", font_size=20)
        ).arrange(DOWN, buff=0.1).to_edge(UP).shift(RIGHT * 3)

        # 2. Setup Inputs
        # Placeholder for image to ensure it runs without external file dependencies
        img_input = VGroup(
            Square(side_length=1.2, color=WHITE).set_fill(GREY, opacity=0.5),
            Text("img-7.jpeg", font_size=14)
        ).next_to(img_enc, DOWN, buff=1)
        
        txt_france = Text("'A Street View photo in France'", font_size=18, color=YELLOW).next_to(txt_enc, DOWN, buff=1)
        txt_japan = Text("'A Street View photo in Japan'", font_size=18, color=RED).next_to(txt_enc, DOWN, buff=1)

        # 3. Setup Shared Embedding Space (Axes)
        axes = Axes(
            x_range=[0, 5, 1], y_range=[0, 5, 1],
            x_length=3.5, y_length=3.5,
            axis_config={"include_tip": True}
        ).to_edge(DOWN).shift(LEFT * 2.5)
        axes_label = Text("Shared Embedding Space", font_size=18).next_to(axes, UP)

        # 4. Metrics and Formulas
        sim_tracker = ValueTracker(0.3)
        sim_label = Text("Cosine Similarity:", font_size=24).to_edge(RIGHT).shift(UP * 0.8)
        sim_val = DecimalNumber(0, num_decimal_places=2).next_to(sim_label, DOWN)
        # Use 'mob' as the lambda parameter to ensure it is recognized as a local name
        sim_val.add_updater(lambda mob: mob.set_value(sim_tracker.get_value()))
        
        formula = MathTex(
            r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})",
            font_size=28
        ).next_to(sim_val, DOWN, buff=0.8)

        # 5. Vectors
        origin_pt = axes.c2p(0, 0)
        img_vec = Arrow(origin_pt, axes.c2p(2, 3), color=BLUE, buff=0)
        txt_vec = Arrow(origin_pt, axes.c2p(4, 1), color=GREEN, buff=0)

        # --- Animation Sequence ---
        
        # Step 1: Encoders and Inputs
        self.play(Create(img_enc), Create(txt_enc))
        self.play(FadeIn(img_input), Write(txt_france))
        self.wait(1)

        # Step 2: Feed inputs into encoders
        self.play(
            img_input.animate.scale(0.2).move_to(img_enc.get_center()).set_opacity(0),
            txt_france.animate.scale(0.2).move_to(txt_enc.get_center()).set_opacity(0),
            run_time=1.5
        )

        # Step 3: Reveal Embedding Space and Vectors
        self.play(Create(axes), Write(axes_label))
        self.play(GrowArrow(img_vec), GrowArrow(txt_vec))
        self.play(Write(sim_label), Write(sim_val), Write(formula))
        self.wait(1)

        # Step 4: Maximize similarity (Align France text with image)
        france_aligned_pos = axes.c2p(2.2, 3.3) # Near the image vector
        self.play(
            txt_vec.animate.put_start_and_end_on(origin_pt, france_aligned_pos),
            sim_tracker.animate.set_value(0.98),
            run_time=2
        )
        self.wait(1)

        # Step 5: Contrastive - Wrong label (Japan)
        self.play(Write(txt_japan))
        self.play(txt_japan.animate.scale(0.2).move_to(txt_enc.get_center()).set_opacity(0), run_time=1)
        
        # Push Japan vector away
        japan_pos = axes.c2p(4.5, 0.5)
        self.play(
            txt_vec.animate.put_start_and_end_on(origin_pt, japan_pos),
            sim_tracker.animate.set_value(0.12),
            run_time=2
        )
        
        # Final Summary
        summary = Text(
            "Model learns to associate visual features\nwith specific geographic labels.",
            font_size=20, line_spacing=1.5
        ).to_edge(RIGHT).shift(DOWN * 2)
        self.play(Write(summary))
        self.wait(2)
