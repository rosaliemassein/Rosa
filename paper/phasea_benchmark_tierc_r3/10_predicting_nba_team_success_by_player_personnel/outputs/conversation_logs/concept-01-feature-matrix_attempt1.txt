from manim import *
import numpy as np

class TeamFeatureMatrix(Scene):
    def construct(self):
        # 1. Represent each team as 15 players (silhouettes)
        players = VGroup(*[
            VGroup(
                Circle(radius=0.12, color=WHITE, fill_opacity=1),
                Line(DOWN * 0.12, DOWN * 0.35, color=WHITE),
                Line(DOWN * 0.15, DOWN * 0.25 + LEFT * 0.1, color=WHITE),
                Line(DOWN * 0.15, DOWN * 0.25 + RIGHT * 0.1, color=WHITE)
            )
            for _ in range(15)
        ]).arrange(DOWN, buff=0.15).to_edge(LEFT, buff=0.7)

        player_labels = VGroup(*[
            Text(f"Player {i+1}", font_size=14).next_to(players[i], LEFT, buff=0.1)
            for i in range(15)
        ])

        self.play(Create(players), Write(player_labels))
        self.wait(0.5)

        # 2. Minutes played filter highlighting top 10
        filter_rect = SurroundingRectangle(players[:10], color=YELLOW, buff=0.1)
        filter_text = Text("Top 10 (Minutes Played)", font_size=20, color=YELLOW).next_to(filter_rect, UP)

        self.play(Create(filter_rect), Write(filter_text))
        self.play(
            players[10:].animate.set_opacity(0.2),
            player_labels[10:].animate.set_opacity(0.2),
            run_time=1
        )
        self.wait(0.5)

        # 3. Create 10x15 grid representing the feature matrix
        rows, cols = 10, 15
        grid = VGroup(*[
            Square(side_length=0.32, stroke_width=1, color=GRAY_A)
            for _ in range(rows * cols)
        ]).arrange_in_grid(rows=rows, cols=cols, buff=0.05).shift(RIGHT * 1.8)

        # 4. Slide top 10 players into the rows of the grid
        self.play(
            FadeOut(filter_rect),
            FadeOut(filter_text),
            FadeOut(players[10:]),
            FadeOut(player_labels[10:]),
            *[players[i].animate.scale(0.6).move_to(grid[i * cols].get_left() + LEFT * 0.4) for i in range(10)],
            *[player_labels[i].animate.scale(0.7).next_to(grid[i * cols], LEFT, buff=0.6) for i in range(10)],
            Create(grid),
            run_time=1.5
        )

        # 5. Label metrics for columns
        metrics = ["PTS", "REB", "AST", "STL", "BLK", "TOV", "FG%", "FT%", "3P%", "ORB", "DRB", "PF", "+/-", "USG", "PIE"]
        metric_labels = VGroup(*[
            Text(metrics[i], font_size=10).next_to(grid[i], UP, buff=0.15).rotate(45 * DEGREES)
            for i in range(cols)
        ])
        self.play(Write(metric_labels))

        # 6. Fill grid cells with intensity (normalized data)
        # Using loop to simulate data population
        for cell in grid:
            cell.set_fill(BLUE_D, opacity=np.random.uniform(0.1, 0.95))
        
        self.play(
            *[cell.animate.set_style(fill_opacity=cell.fill_opacity) for cell in grid],
            run_time=1.5
        )
        self.wait(1)

        # 7. Collapse 2D grid into a single long horizontal vector
        # Reference Formula: n = m * p
        formula = MathTex("n = 10 \\times 15 = 150", font_size=32).to_edge(UP, buff=0.5)
        
        vector_group = grid.copy()
        
        self.play(
            FadeOut(players[:10]),
            FadeOut(player_labels[:10]),
            FadeOut(metric_labels),
            Write(formula)
        )

        self.play(
            grid.animate.set_opacity(0.1).scale(0.8).to_edge(UP, buff=1.5),
            vector_group.animate.arrange(RIGHT, buff=0).scale_to_fit_width(12).move_to(DOWN * 1.5),
            run_time=2
        )

        # 8. Final vector labeling (without using disallowed Brace)
        bracket_line = Line(vector_group.get_left() + DOWN * 0.3, vector_group.get_right() + DOWN * 0.3, color=WHITE)
        tick_l = Line(bracket_line.get_left() + UP * 0.1, bracket_line.get_left() + DOWN * 0.1, color=WHITE)
        tick_r = Line(bracket_line.get_right() + UP * 0.1, bracket_line.get_right() + DOWN * 0.1, color=WHITE)
        
        label_vector = Text("Flattened Feature Vector (1x150)", font_size=24).next_to(bracket_line, DOWN, buff=0.3)
        
        self.play(
            Create(bracket_line),
            Create(tick_l),
            Create(tick_r),
            Write(label_vector)
        )
        
        self.wait(2)
