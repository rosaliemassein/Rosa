from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # Using 't' for time throughout to avoid potential 'x' identifier conflicts
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 50, 10],
            x_length=6,
            y_length=4,
            axis_config={"color": WHITE},
            tips=False
        ).to_edge(LEFT, buff=0.5)

        # Labels for the axes
        labels = axes.get_axis_labels(Tex("t"), Tex("RPM"))

        # Define the function to plot fluctuating velocity
        def motor_speed_func(t):
            return 30 + 3 * np.sin(2 * t) + 2 * np.cos(5 * t)

        # Create the plot
        velocity_plot = axes.plot(motor_speed_func, color=BLUE)

        # 1. Animation: Display Plot
        self.play(Create(axes), Write(labels))
        self.play(Create(velocity_plot))
        self.wait(1)

        # 2. Highlight the last 20% of the x-axis (t=8 to t=10)
        # We calculate dimensions based on axis unit sizes
        rect_width = axes.x_axis.get_unit_size() * 2
        rect_height = axes.y_axis.get_unit_size() * 50
        highlight_area = Rectangle(
            width=rect_width,
            height=rect_height,
            fill_color=BLUE_B,  # Standard Manim color
            fill_opacity=0.3,
            stroke_width=0
        ).move_to(axes.c2p(9, 25)) # Center of the last 20% block

        steady_state_text = Text("Steady State Error (Last 20%)", font_size=18).next_to(highlight_area, UP)

        self.play(FadeIn(highlight_area), Write(steady_state_text))
        self.wait(2)

        # 3. Show a 'Histogram' appearing on the side
        hist_title = Text("Histogram", font_size=24).to_edge(RIGHT, buff=1.5).shift(UP * 2)
        
        # Represent counts with mode at center being tallest
        bar_height_values = [0.4, 0.8, 2.6, 1.2, 0.5]
        histogram_bars = VGroup(*[
            Rectangle(
                width=0.4, 
                height=h, 
                fill_opacity=0.8, 
                color=YELLOW if idx == 2 else GREY_B
            ) for idx, h in enumerate(bar_height_values)
        ]).arrange(RIGHT, aligned_edge=DOWN, buff=0.1)
        histogram_bars.next_to(hist_title, DOWN, buff=0.5)
        
        mode_indicator = Text("Mode", font_size=16, color=YELLOW).next_to(histogram_bars[2], UP, buff=0.1)

        self.play(Write(hist_title), Create(histogram_bars), Write(mode_indicator))
        self.wait(1)

        # 4. Animate the Stability Formula
        # Using \mathrm for text formatting within MathTex for robustness
        stability_formula = MathTex(
            r"\mathrm{Stability} = \frac{\mathrm{counts\ at\ mode}}{\mathrm{counts\ not\ at\ mode}}",
            font_size=30
        ).to_edge(DOWN, buff=0.7)

        # Visual link between histogram and formula
        box_mode = SurroundingRectangle(histogram_bars[2], color=YELLOW)
        other_bars = VGroup(histogram_bars[0], histogram_bars[1], histogram_bars[3], histogram_bars[4])
        box_others = SurroundingRectangle(other_bars, color=RED)

        self.play(Write(stability_formula))
        self.play(Create(box_mode))
        self.wait(0.5)
        self.play(ReplacementTransform(box_mode, box_others))
        self.wait(1)
        self.play(FadeOut(box_others))

        self.wait(3)
