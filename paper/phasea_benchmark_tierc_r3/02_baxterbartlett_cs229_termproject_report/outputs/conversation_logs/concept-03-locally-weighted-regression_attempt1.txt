from manim import *
import numpy as np

class Concept03LocallyWeightedRegression(Scene):
    def construct(self):
        # Setup the coordinate system
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6,
            axis_config={"include_tip": False}
        ).to_edge(DOWN)

        # Generate non-linear data points (sine-wave with noise)
        np.random.seed(42)
        x_vals = np.linspace(-4, 4, 30)
        y_vals = 1.5 * np.sin(x_vals * 0.8) + np.random.normal(0, 0.2, 30)
        
        dots = VGroup()
        for x, y in zip(x_vals, y_vals):
            d = Dot(axes.c2p(x, y), radius=0.08, color=BLUE)
            dots.add(d)
        
        self.add(axes, dots)

        # ValueTracker for the moving query point
        query_x_tracker = ValueTracker(-3.5)
        tau = 0.7  # The 'bandwidth' parameter for weighting

        # The Query Point and the weight field (visualized as concentric circles)
        query_dot = Dot(color=WHITE, z_index=10)
        
        weight_field = VGroup(*[
            Circle(radius=r, stroke_width=0, fill_color=YELLOW, fill_opacity=0.1)
            for r in np.linspace(0.2, 2.0, 6)
        ])
        
        # Updater to keep the query group positioned correctly
        def update_query_group(mob):
            qx = query_x_tracker.get_value()
            # Follow the sine curve trend
            qy = 1.5 * np.sin(qx * 0.8)
            mob.move_to(axes.c2p(qx, qy))

        query_group = VGroup(weight_field, query_dot)
        query_group.add_updater(update_query_group)
        self.add(query_group)

        # The local regression line that fits the weighted neighborhood
        reg_line = Line(LEFT, RIGHT, color=RED, stroke_width=6).set_z_index(5)

        def update_line(line_mob):
            qx = query_x_tracker.get_value()
            # Weights follow the formula w = exp(-d^2 / 2tau^2)
            weights = np.exp(-(x_vals - qx)**2 / (2 * tau**2))
            
            # Perform Weighted Least Squares: solve (X.T * W * X) * beta = X.T * W * y
            X = np.column_stack([np.ones_like(x_vals), x_vals])
            W = np.diag(weights)
            try:
                XTW = X.T @ W
                beta = np.linalg.solve(XTW @ X, XTW @ y_vals)
                intercept, slope = beta
                
                # Define a short segment around the query point for the visualization
                x_start, x_end = qx - 0.7, qx + 0.7
                p1 = axes.c2p(x_start, intercept + slope * x_start)
                p2 = axes.c2p(x_end, intercept + slope * x_end)
                line_mob.put_start_and_end_on(p1, p2)
            except:
                pass

        reg_line.add_updater(update_line)
        
        # Updater to change dot color and opacity based on proximity to query point
        def update_dots_behavior(dots_mob):
            qx = query_x_tracker.get_value()
            weights = np.exp(-(x_vals - qx)**2 / (2 * tau**2))
            for i, d in enumerate(dots_mob):
                w = weights[i]
                d.set_opacity(0.3 + 0.7 * w)
                if w > 0.4:
                    d.set_color(YELLOW)
                    d.set_scale(1.2)
                else:
                    d.set_color(BLUE)
                    d.set_scale(1.0)

        dots.add_updater(update_dots_behavior)

        self.add(reg_line)

        # Labels and Narration Text
        label = Text("Locally Weighted Regression", font_size=32).to_edge(UP)
        formula_tex = MathTex(
            r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)",
            font_size=36
        ).to_edge(UP)
        
        self.play(Write(label))
        self.wait(0.5)
        
        # Animate the query moving along the dataset
        # Using lambda t: t as a replacement for linear rate function
        self.play(query_x_tracker.animate.set_value(3.5), run_time=8, rate_func=lambda t: t)
        self.wait(1)
        
        self.play(Transform(label, formula_tex))
        # Animate back to illustrate dynamic recalculation
        self.play(query_x_tracker.animate.set_value(-3.5), run_time=6, rate_func=lambda t: t)
        self.wait(2)
