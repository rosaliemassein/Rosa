from manim import *
import numpy as np

class ConcepTweenInverseMapping(Scene):
    def construct(self):
        # 1. PID Formula reference
        formula = MathTex(
            r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}",
            font_size=36
        ).to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Create Input and Output boxes for the Diagram
        # Desired Behavior Box (Left)
        input_rect = Rectangle(width=4.5, height=2.5, color=YELLOW)
        input_title = Text("Desired Behavior", font_size=28, color=YELLOW).move_to(input_rect.get_top() + DOWN * 0.5)
        input_items = Text("• Rise Time\n• Steady State Error\n• Stability", font_size=20, line_spacing=1.2).next_to(input_title, DOWN, buff=0.3)
        input_group = VGroup(input_rect, input_title, input_items)

        # PID Constants Box (Right)
        output_rect = Rectangle(width=4.5, height=2.5, color=BLUE)
        output_title = Text("PID Constants", font_size=28, color=BLUE).move_to(output_rect.get_center())
        output_group = VGroup(output_rect, output_title)

        # Arrow between them
        arrow = Arrow(LEFT, RIGHT, stroke_width=4, buff=0.2)
        
        # Group diagram together
        diagram = VGroup(input_group, arrow, output_group).arrange(RIGHT, buff=0.5).shift(UP * 0.2)

        # 3. Animate Diagram
        self.play(Create(diagram))
        self.wait(1)

        # 4. "Flip the script" - Rotate 180 degrees
        # Using explicit axis and rate function to avoid linter issues
        self.play(Rotate(diagram, angle=PI, axis=np.array([0, 0, 1])), run_time=1.5)
        self.wait(1)

        # 5. Motor Velocity Plot
        # Scale down diagram to fit plot below
        self.play(
            diagram.animate.scale(0.6).to_edge(UP, buff=1.0),
            formula.animate.scale(0.7).to_edge(UP, buff=0.1)
        )

        axes = Axes(
            x_range=[0, 8, 1],
            y_range=[0, 1.5, 0.5],
            x_length=9,
            y_length=3,
            axis_config={"include_tip": True, "color": WHITE},
            tips=True
        ).to_edge(DOWN, buff=0.8)

        y_label = axes.get_y_axis_label("RPM", edge=LEFT, direction=LEFT, buff=0.3)
        x_label = axes.get_x_axis_label("Time", edge=DOWN, direction=DOWN, buff=0.3)

        # Dashed horizontal line representing the setpoint
        setpoint_line = DashedLine(
            start=axes.c2p(0, 1),
            end=axes.c2p(8, 1),
            color=GREY_A
        )
        setpoint_label = Text("Setpoint", font_size=16, color=GREY_A).next_to(setpoint_line, UP, buff=0.1).shift(LEFT * 3)

        # Oscillating curve towards setpoint: y = 1 - exp(-0.7t) * cos(3t)
        # Using 't' as variable for lambda to avoid linter confusion with 'x'
        motor_curve = axes.plot(
            lambda t: 1 - np.exp(-0.7 * t) * np.cos(3 * t),
            x_range=[0, 8],
            color=RED
        )

        # 6. Animate Plotting
        self.play(Create(axes), Write(x_label), Write(y_label))
        self.play(Create(setpoint_line), Write(setpoint_label))
        self.play(Create(motor_curve, run_time=3, rate_func=rate_functions.linear))

        # 7. Labels using standard Brace and Line
        # Rise Time: Approximately the first time it reaches setpoint
        rise_line = Line(axes.c2p(0, -0.1), axes.c2p(0.8, -0.1))
        rise_time_brace = Brace(rise_line, DOWN, color=YELLOW)
        rise_time_text = rise_time_brace.get_text("Rise Time", font_size=20)

        # Steady State Error: Visual gap at the end
        error_line = Line(axes.c2p(7.8, 1.0), axes.c2p(7.8, 0.97))
        ss_error_brace = Brace(error_line, RIGHT, color=BLUE, buff=0.1)
        ss_error_text = ss_error_brace.get_text("Steady State\nError", font_size=18).next_to(ss_error_brace, RIGHT)

        self.play(
            Create(rise_time_brace), Write(rise_time_text),
            Create(ss_error_brace), Write(ss_error_text)
        )
        self.wait(2)
