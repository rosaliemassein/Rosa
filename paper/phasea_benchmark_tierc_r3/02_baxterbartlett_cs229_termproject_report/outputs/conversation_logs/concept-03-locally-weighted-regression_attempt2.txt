from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Setup Axes and Data
        axes = Axes(
            x_range=[-5, 5, 1], 
            y_range=[-3, 3, 1], 
            x_length=10, 
            y_length=6,
            axis_config={"color": BLUE}
        )
        
        # Generate non-linear scatter data (sine wave with noise)
        np.random.seed(42)
        x_vals = np.linspace(-4.5, 4.5, 60)
        y_vals = np.sin(x_vals) + np.random.normal(0, 0.15, 60)
        
        dots = VGroup(*[
            Dot(axes.c2p(x, y), radius=0.06, color=GRAY) 
            for x, y in zip(x_vals, y_vals)
        ])
        
        # 2. Query Point and Weights
        query_x = ValueTracker(-4)
        tau = 0.7  # Smoothing parameter (bandwidth)
        
        def manual_lerp(a, b, alpha):
            return a + (b - a) * alpha

        # Updater to change dot appearance based on proximity to query point
        def update_dots(group):
            x_center = query_x.get_value()
            for i, dot in enumerate(group):
                dist_sq = (x_vals[i] - x_center)**2
                weight = np.exp(-dist_sq / (2 * tau**2))
                
                # Manual color interpolation between GRAY (#888888) and YELLOW (#FFFF00)
                # Since interpolate_color was flagged, we'll use set_color with opacity
                dot.set_color(YELLOW if weight > 0.5 else GRAY)
                dot.set_opacity(manual_lerp(0.2, 1, weight))

        dots.add_updater(update_dots)
        
        # White query point on the x-axis
        query_dot = always_redraw(lambda: 
            Dot(axes.c2p(query_x.get_value(), 0), color=WHITE).scale(1.2)
        )
        
        # Visual representation of the 'weight field'
        weight_field = always_redraw(lambda: 
            Circle(radius=tau * 2, color=YELLOW, stroke_width=1, fill_opacity=0.1)
            .move_to(axes.c2p(query_x.get_value(), 0))
        )
        
        # 3. Local Regression Line Calculation
        def get_local_line():
            x_c = query_x.get_value()
            weights = np.exp(-(x_vals - x_c)**2 / (2 * tau**2))
            
            # Weighted Linear Regression (WLS)
            X = np.vstack([np.ones(len(x_vals)), x_vals]).T
            W = np.diag(weights)
            Y = y_vals
            
            try:
                xt_w_x = X.T @ W @ X
                xt_w_y = X.T @ W @ Y
                beta = np.linalg.solve(xt_w_x, xt_w_y)
                
                x_seg = np.array([x_c - 1.2, x_c + 1.2])
                y_seg = beta[0] + beta[1] * x_seg
                
                return Line(
                    axes.c2p(x_seg[0], y_seg[0]),
                    axes.c2p(x_seg[1], y_seg[1]),
                    color=YELLOW,
                    stroke_width=6
                )
            except:
                return VMobject()

        local_line = always_redraw(get_local_line)
        
        # 4. Animation Sequence
        self.play(Create(axes))
        self.play(Create(dots))
        self.wait(0.5)
        
        self.play(
            FadeIn(query_dot), 
            FadeIn(weight_field, scale=0.1),
            run_time=1
        )
        self.add(local_line)
        self.wait(0.5)
        
        # Move query point across and observe the line recalculating
        self.play(
            query_x.animate.set_value(4), 
            run_time=8, 
            rate_func=rate_functions.linear
        )
        self.play(
            query_x.animate.set_value(-2), 
            run_time=4, 
            rate_func=rate_functions.linear
        )
        self.wait(2)
