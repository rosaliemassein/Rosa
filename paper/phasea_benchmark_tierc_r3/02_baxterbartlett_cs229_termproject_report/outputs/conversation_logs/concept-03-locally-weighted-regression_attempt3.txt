from manim import *
import numpy as np

class Concept03LocallyWeightedRegression(Scene):
    def construct(self):
        # Create Axes
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6,
            axis_config={"include_tip": False}
        )
        labels = axes.get_axis_labels(x_label="x", y_label="y")
        self.add(axes, labels)

        # Generate non-linear data points with noise
        np.random.seed(42)
        x_vals = np.linspace(-4.5, 4.5, 40)
        y_vals = np.sin(x_vals) + np.random.normal(0, 0.2, size=len(x_vals))
        
        points_data = [axes.c2p(x, y) for x, y in zip(x_vals, y_vals)]
        dots = VGroup(*[Dot(p, radius=0.08, color=GRAY) for p in points_data])
        
        self.play(Create(dots))
        self.wait(0.5)

        # Query point starting position
        query_x = ValueTracker(-3)
        
        query_point = Dot(color=WHITE, radius=0.12)
        def update_query_point(mob):
            xq = query_x.get_value()
            yq = np.sin(xq)
            mob.move_to(axes.c2p(xq, yq))
        query_point.add_updater(update_query_point)
        
        # Radial weight field visual
        weight_field = Circle(radius=1.5, color=BLUE, stroke_opacity=0.5, fill_opacity=0.1)
        def update_weight_field(mob):
            mob.move_to(query_point.get_center())
        weight_field.add_updater(update_weight_field)
        
        self.play(FadeIn(query_point), FadeIn(weight_field))

        # Formula label
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)", font_size=34)
        formula.to_corner(UL)
        self.add(formula)

        # Local regression line parameters
        tau = 0.8
        
        def get_regression_line():
            xq = query_x.get_value()
            # Calculate weights based on formula provided
            weights = np.exp(- (x_vals - xq)**2 / (2 * tau**2))
            
            # Weighted Least Squares calculation
            # X matrix with intercept column
            X = np.stack([np.ones(len(x_vals)), x_vals], axis=1)
            # Weight matrix
            W = np.diag(weights)
            
            try:
                # Normal Equation for Weighted Least Squares: theta = (X^T W X)^-1 X^T W y
                theta = np.linalg.inv(X.T @ W @ X) @ X.T @ W @ y_vals
                
                # Create a local line segment around the query point
                x_start, x_end = xq - 1.2, xq + 1.2
                y_start = theta[0] + theta[1] * x_start
                y_end = theta[0] + theta[1] * x_end
                
                return Line(
                    axes.c2p(x_start, y_start),
                    axes.c2p(x_end, y_end),
                    color=YELLOW,
                    stroke_width=6
                )
            except:
                return Line(axes.c2p(-1, 0), axes.c2p(1, 0), stroke_opacity=0)

        reg_line = always_redraw(get_regression_line)

        # Updater for points glowing based on distance from query point
        def update_dots_glow(mob_group):
            xq = query_x.get_value()
            weights = np.exp(- (x_vals - xq)**2 / (2 * tau**2))
            for i, dot in enumerate(mob_group):
                w = weights[i]
                # Manim community interpolation
                c = interpolate_color(GRAY, YELLOW, w)
                dot.set_color(c)
                dot.set_stroke(color=YELLOW, width=w*4, opacity=w)

        dots.add_updater(update_dots_glow)

        self.play(Create(reg_line))
        self.wait(1)

        # Animate the query point moving across the space to show "local" adaptation
        # Explicitly referencing rate_functions to avoid undefined identifier issues
        self.play(query_x.animate.set_value(3), run_time=6, rate_func=rate_functions.linear)
        self.wait(1)
        self.play(query_x.animate.set_value(-1), run_time=4, rate_func=rate_functions.smooth)
        self.wait(2)

        # Cleanup updaters before finishing
        dots.remove_updater(update_dots_glow)
        query_point.remove_updater(update_query_point)
        weight_field.remove_updater(update_weight_field)
        
        self.play(
            FadeOut(weight_field), 
            FadeOut(reg_line), 
            FadeOut(query_point), 
            FadeOut(dots), 
            FadeOut(formula)
        )
        self.wait(1)
