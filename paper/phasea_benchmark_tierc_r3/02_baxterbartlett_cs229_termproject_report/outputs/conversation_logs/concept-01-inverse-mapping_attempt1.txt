from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Diagram Setup (Top Half)
        input_title = Text("Desired Behavior", font_size=24)
        input_metrics = Text("Rise Time, Steady State Error, Stability", font_size=16, color=GRAY)
        input_group = VGroup(input_title, input_metrics).arrange(DOWN, buff=0.2)
        input_rect = SurroundingRectangle(input_group, color=BLUE, buff=0.3)
        input_box = VGroup(input_rect, input_group).shift(LEFT * 3 + UP * 2)

        output_title = Text("PID Constants", font_size=24)
        output_rect = SurroundingRectangle(output_title, color=GREEN, buff=0.3)
        output_box = VGroup(output_rect, output_title).shift(RIGHT * 3 + UP * 2)

        # Main connecting arrow
        arrow_obj = Arrow(input_box.get_right(), output_box.get_left(), color=YELLOW, buff=0.2)
        
        # PID Formula reference
        formula = MathTex(
            r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", 
            font_size=20
        ).next_to(output_box, DOWN, buff=0.4)

        diagram = VGroup(input_box, output_box, arrow_obj)

        # 2. Motor Performance Plot (Bottom Half)
        axes = Axes(
            x_range=[0, 10, 2],
            y_range=[0, 1.5, 0.5],
            x_length=8,
            y_length=3,
            axis_config={"include_tip": True}
        ).shift(DOWN * 2)
        
        axes_labels = axes.get_axis_labels(
            x_label=Text("Time", font_size=18), 
            y_label=Text("RPM", font_size=18)
        )

        setpoint = DashedLine(
            axes.c2p(0, 1), axes.c2p(10, 1), color=RED
        )
        setpoint_label = Text("Setpoint", font_size=12, color=RED).next_to(setpoint, RIGHT, buff=0.1)

        # Motor response curve
        # x is used as the variable to avoid naming conflicts
        curve = axes.plot(
            lambda x: 1.1 - np.exp(-0.5 * x) * np.cos(2 * x),
            x_range=[0, 10],
            color=BLUE
        )

        # Annotations (Rise Time and Steady State Error)
        # Using Line with two tips as a manual DoubleArrow implementation
        rise_line = Line(
            axes.c2p(0, -0.3), axes.c2p(1.5, -0.3), color=YELLOW
        ).add_tip(tip_length=0.1).add_tip(tip_length=0.1, at_start=True)
        rise_label = Text("Rise Time", font_size=14, color=YELLOW).next_to(rise_line, DOWN, buff=0.1)

        error_line = Line(
            axes.c2p(9.5, 1), axes.c2p(9.5, 1.1), color=ORANGE
        ).add_tip(tip_length=0.1).add_tip(tip_length=0.1, at_start=True)
        error_label = Text("Steady State Error", font_size=14, color=ORANGE).next_to(error_line, RIGHT, buff=0.1)

        # 3. Animation Sequence
        # Voice: "In traditional robotics, we choose PID constants and observe..."
        self.play(Create(input_box), Create(output_box))
        self.play(Create(arrow_obj))
        self.play(Write(formula))
        self.wait(1)

        # Voice: "...This paper flips the script."
        # Rotate the entire diagram 180 degrees using .animate to avoid Rotate identifier issues
        self.play(
            diagram.animate.rotate(PI),
            formula.animate.alpha_blend(0).scale(0), # Fade out formula or move it
            run_time=2
        )
        self.wait(1)

        # Voice: "It treats the desired behavior... as the input..."
        self.play(Create(axes), Write(axes_labels))
        self.play(Create(setpoint), Write(setpoint_label))
        self.play(Create(curve), run_time=2)
        
        # Show specific metrics on the plot
        self.play(Create(rise_line), Write(rise_label))
        self.play(Create(error_line), Write(error_label))
        
        self.wait(2)
