from manim import *
import numpy as np

class InverseMappingConcept(Scene):
    def construct(self):
        # 1. Input and Output Boxes
        input_title = Text("Desired Behavior", font_size=24)
        bullet1 = Text("• Rise Time", font_size=20)
        bullet2 = Text("• Steady State Error", font_size=20)
        bullet3 = Text("• Stability", font_size=20)
        bullets = VGroup(bullet1, bullet2, bullet3).arrange(DOWN, aligned_edge=LEFT, buff=0.1)
        input_group = VGroup(input_title, bullets).arrange(DOWN, buff=0.2)
        
        input_box_rect = RoundedRectangle(height=2.5, width=4, color=BLUE)
        input_box = VGroup(input_box_rect, input_group)

        output_text = Text("PID Constants", font_size=30)
        output_box_rect = RoundedRectangle(height=2.5, width=4, color=GREEN)
        output_box = VGroup(output_box_rect, output_text)

        diagram = VGroup(input_box, output_box).arrange(RIGHT, buff=2)
        diagram.to_edge(UP, buff=0.5)

        arrow = Arrow(input_box.get_right(), output_box.get_left(), color=WHITE)
        label_arrow = Text("Traditional Mapping", font_size=20).next_to(arrow, UP)
        
        full_diagram = VGroup(diagram, arrow, label_arrow)
        
        self.play(Create(input_box), Create(output_box))
        self.play(GrowArrow(arrow), Write(label_arrow))
        self.wait(1)

        # 2. Flip the script - Rotate 180 degrees
        self.play(full_diagram.animate.rotate(PI), run_time=1.5)
        self.wait(1)

        # 3. Motor Velocity Plot (RPM vs Time)
        axes = Axes(
            x_range=[0, 8, 1],
            y_range=[0, 1.5, 0.5],
            x_length=7,
            y_length=3,
            axis_config={"include_tip": True, "stroke_width": 2}
        ).to_edge(DOWN, buff=0.8)
        
        axis_labels = axes.get_axis_labels(x_label="Time", y_label="RPM")
        
        # Define a function for a typical PID step response
        def response_func(t):
            return 1 - np.exp(-0.8 * t) * (np.cos(2 * t) + 0.4 * np.sin(2 * t))

        curve = axes.plot(response_func, color=YELLOW, x_range=[0, 8])
        
        setpoint = DashedLine(
            start=axes.c2p(0, 1),
            end=axes.c2p(8, 1),
            color=RED,
            stroke_width=2
        )
        
        self.play(Create(axes), Write(axis_labels))
        self.play(Create(setpoint))
        self.play(Create(curve), run_time=3)

        # 4. Braced labels for Rise Time and Steady State Error
        # Invisible line for the Rise Time brace (approx start to where it hits y=1)
        rt_line = Line(axes.c2p(0, -0.2), axes.c2p(1.3, -0.2), stroke_opacity=0)
        rise_time_brace = Brace(rt_line, DOWN)
        rise_time_text = Text("Rise Time", font_size=18).next_to(rise_time_brace, DOWN)
        
        # Vertical gap at the end for Steady State Error
        sse_point_curve = axes.c2p(7.5, response_func(7.5))
        sse_point_target = axes.c2p(7.5, 1)
        sse_line = Line(sse_point_curve, sse_point_target, stroke_opacity=0)
        sse_brace = Brace(sse_line, RIGHT, buff=0.1)
        sse_text = Text("Steady State Error", font_size=16).next_to(sse_brace, RIGHT)

        self.play(Create(rise_time_brace), Write(rise_time_text))
        self.play(Create(sse_brace), Write(sse_text))
        
        # 5. PID Formula Reference
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}").scale(0.75)
        formula.to_corner(UL, buff=0.5)
        self.play(Write(formula))
        
        self.wait(2)
