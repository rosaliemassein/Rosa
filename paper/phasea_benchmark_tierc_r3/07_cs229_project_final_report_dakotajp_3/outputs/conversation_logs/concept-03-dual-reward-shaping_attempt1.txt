from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # Use a safe rate function for linear motion
        linear_rate = lambda t: t
        
        # Configuration
        w1 = 10
        w2 = 5
        target_pos = RIGHT * 4
        
        # 1. Create the board and pegs
        # Arrange pegs in a cluster on the left using a safe loop
        pegs_list = []
        for i in range(8):
            random_pos = np.array([
                -4 + np.random.uniform(-0.6, 0.6),
                np.random.uniform(-1, 1),
                0
            ])
            pegs_list.append(Dot(point=random_pos, color=BLUE))
        pegs = VGroup(*pegs_list)
            
        target_zone = Circle(radius=1.2, color=RED).move_to(target_pos)
        target_label = Text("Target Zone", font_size=24, color=RED).next_to(target_zone, UP)
        
        # 2. Centroid and Label with named updater functions
        centroid = Dot(color=GREEN)
        
        def update_centroid_pos(mob_to_update):
            all_centers = [p.get_center() for p in pegs]
            avg_pos = np.mean(all_centers, axis=0)
            mob_to_update.move_to(avg_pos)
            
        centroid.add_updater(update_centroid_pos)
        
        centroid_label = Text("Centroid", font_size=20, color=GREEN)
        
        def update_centroid_label_pos(mob_to_update):
            mob_to_update.next_to(centroid, UP, buff=0.1)
            
        centroid_label.add_updater(update_centroid_label_pos)
        
        # 3. Dashed line from Centroid to Target using always_redraw
        # A lambda with no arguments is typically safer for static analysis
        dashed_line = always_redraw(lambda: DashedLine(
            centroid.get_center(), 
            target_zone.get_center(), 
            color=GREEN,
            buff=0.1
        ))
        
        # 4. Formula and Reward Counter
        formula = MathTex(
            R"R = w_1(\text{pegs}_{in})^2 + w_2(\text{dist}(\text{centroid, target}))^{-1}",
            font_size=30
        ).to_edge(UP)
        
        reward_val = DecimalNumber(0, num_decimal_places=2, font_size=32)
        reward_label = Text("Reward: ", font_size=32)
        reward_display = VGroup(reward_label, reward_val).arrange(RIGHT).to_edge(DOWN, buff=1)
        
        # Update function for the reward counter to show real-time progress
        def update_reward_text(reward_group):
            current_centroid = centroid.get_center()
            current_target = target_zone.get_center()
            dist = np.linalg.norm(current_centroid - current_target)
            
            # Count pegs inside the radius
            pegs_in = 0
            for p in pegs:
                dist_to_target = np.linalg.norm(p.get_center() - current_target)
                if dist_to_target < target_zone.radius:
                    pegs_in += 1
            
            # Formula: R = w1 * (count^2) + w2 / dist
            # max(dist, 0.1) prevents division by zero
            new_val = w1 * (pegs_in**2) + w2 / (max(dist, 0.1))
            reward_group[1].set_value(new_val)

        reward_display.add_updater(update_reward_text)

        # Initial Setup
        self.add(target_zone, target_label, pegs, centroid, centroid_label, dashed_line, formula, reward_display)
        self.wait(1)

        # 5. Animation Part 1: Moving the entire team forward
        # This demonstrates the 'centroid gravity' part of the reward
        self.play(
            pegs.animate.shift(RIGHT * 3.5),
            run_time=3,
            rate_func=linear_rate
        )
        self.wait(0.5)

        # 6. Animation Part 2: Moving pieces into target zone individually
        # This demonstrates the 'squared bonus' for completion
        for i in range(4):
            peg_to_move = pegs[i]
            
            # Calculate a unique spot inside the circle for each peg
            angle = i * (PI / 2)
            dist_offset = 0.5
            target_spot = target_pos + np.array([
                np.cos(angle) * dist_offset, 
                np.sin(angle) * dist_offset, 
                0
            ])
            
            # Visual indicator for the squared bonus
            bonus_amount = (i + 1)**2
            bonus_text = Text(f"+{bonus_amount} Bonus", font_size=30, color=YELLOW).next_to(peg_to_move, UP)
            
            self.play(
                peg_to_move.animate.move_to(target_spot),
                FadeIn(bonus_text, shift=UP*0.3),
                run_time=1
            )
            self.play(FadeOut(bonus_text), run_time=0.4)

        self.wait(2)
        
        # Remove updaters to prevent errors during cleanup
        centroid.clear_updaters()
        centroid_label.clear_updaters()
        reward_display.clear_updaters()
