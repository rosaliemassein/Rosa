from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Setup Board and Target Zone
        board = Rectangle(width=10, height=6, color=BLUE)
        board.set_stroke(opacity=0.5)
        
        target_zone = Rectangle(width=4, height=2, color=GREEN)
        target_zone.set_fill(GREEN, opacity=0.2)
        target_zone.to_edge(UP, buff=0.5)
        
        target_label = Text("Target Zone", font_size=20, color=GREEN).next_to(target_zone, UP, buff=0.1)
        self.add(board, target_zone, target_label)

        # 2. Create Pegs
        pegs = VGroup(*[
            Dot(color=BLUE, radius=0.15).move_to([
                np.random.uniform(-4, 4), 
                np.random.uniform(-2.5, -1), 
                0
            ])
            for _ in range(5)
        ])
        self.add(pegs)

        # 3. Centroid Logic
        centroid_dot = Dot(color=YELLOW, radius=0.12)
        
        def get_centroid_pos():
            return np.mean([p.get_center() for p in pegs], axis=0)
        
        centroid_dot.add_updater(lambda mob: mob.move_to(get_centroid_pos()))
        
        centroid_label = Text("Centroid", font_size=18, color=YELLOW)
        centroid_label.add_updater(lambda mob: mob.next_to(centroid_dot, DOWN, buff=0.1))
        
        # Dashed line from Centroid to Target
        dist_line = always_redraw(lambda: DashedLine(
            centroid_dot.get_center(), 
            target_zone.get_center(), 
            color=WHITE,
            stroke_width=2,
            dash_length=0.15
        ))
        
        self.add(centroid_dot, centroid_label, dist_line)

        # 4. Reward Counter Logic
        # R = w1 * (pegs_in)^2 + w2 * (1/dist)
        reward_label_text = Text("Total Reward:", font_size=28).to_corner(UL).shift(RIGHT * 0.5)
        reward_value = DecimalNumber(0, num_decimal_places=2, color=YELLOW).next_to(reward_label_text, RIGHT)
        
        def calculate_current_reward():
            pegs_in = sum(1 for p in pegs if p.get_center()[1] > target_zone.get_bottom()[1])
            dist = np.linalg.norm(centroid_dot.get_center() - target_zone.get_center())
            dist = max(dist, 0.1) # Prevent division by zero
            return 10 * (pegs_in**2) + 20 * (1.0 / dist)

        reward_value.add_updater(lambda mob: mob.set_value(calculate_current_reward()))
        self.add(reward_label_text, reward_value)

        # 5. Narrative Text
        narrative = Text(
            "Winning is sparse. We reward piece count and centroid distance.",
            font_size=22
        ).to_edge(DOWN, buff=0.4)
        self.add(narrative)

        # 6. Animation sequence
        for i in range(len(pegs)):
            # Define target destination inside the zone
            dest = target_zone.get_center() + np.array([
                np.random.uniform(-1.5, 1.5),
                np.random.uniform(-0.6, 0.6),
                0
            ])
            
            # Move the peg
            self.play(
                pegs[i].animate.move_to(dest),
                run_time=1.5,
                rate_func=lambda t: t # explicitly define linear
            )
            
            # Show a pop-up for the squared reward
            pegs_in_count = i + 1
            bonus_text = MathTex(f"{pegs_in_count}^2", color=GREEN).scale(1.2)
            bonus_text.move_to(pegs[i].get_center() + UP * 0.6)
            
            self.play(
                FadeIn(bonus_text, shift=UP * 0.3),
                run_time=0.4
            )
            self.play(
                FadeOut(bonus_text),
                run_time=0.3
            )
            
            self.wait(0.2)

        self.wait(1)
