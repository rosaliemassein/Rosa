from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Environment Setup
        board = Rectangle(width=10, height=6, color=WHITE, stroke_opacity=0.5)
        target_zone = Rectangle(width=3, height=6, fill_color=YELLOW, fill_opacity=0.2, stroke_color=YELLOW)
        target_zone.to_edge(RIGHT, buff=0)
        target_label = Text("Target Zone", font_size=24, color=YELLOW).next_to(target_zone, UP)
        
        goal_point = target_zone.get_center()
        
        # 2. Agents (Pegs)
        pegs = VGroup(*[
            Dot(LEFT * 4 + UP * np.random.uniform(-2, 2) + RIGHT * np.random.uniform(-1, 1), color=BLUE) 
            for _ in range(6)
        ])
        
        # 3. Centroid Tracking
        centroid_dot = Dot(color=WHITE)
        def update_centroid_pos(obj):
            positions = [p.get_center() for p in pegs]
            obj.move_to(np.mean(positions, axis=0))
        
        centroid_dot.add_updater(update_centroid_pos)
        
        centroid_label = Text("Centroid", font_size=20)
        def update_centroid_label(obj):
            obj.next_to(centroid_dot, UP, buff=0.1)
        
        centroid_label.add_updater(update_centroid_label)
        
        line = always_redraw(lambda: DashedLine(
            centroid_dot.get_center(), 
            goal_point, 
            stroke_width=2, 
            color=GRAY_A
        ))

        # 4. Reward UI
        reward_tracker = ValueTracker(0.0)
        reward_label = Text("Total Reward:", font_size=28).to_corner(UL)
        reward_counter = DecimalNumber(0, num_decimal_places=1, font_size=28).next_to(reward_label, RIGHT)
        
        def update_reward_counter(obj):
            obj.set_value(reward_tracker.get_value())
        reward_counter.add_updater(update_reward_counter)
        
        formula = MathTex(
            r"R = w_1(\text{pegs}_{in})^2 + w_2(\text{dist}(\text{centroid, target}))^{-1}",
            font_size=24
        ).next_to(reward_label, DOWN, aligned_edge=LEFT)

        self.add(board, target_zone, target_label, pegs, centroid_dot, centroid_label, line, reward_label, reward_counter, formula)

        # Logic to calculate reward based on current peg positions
        def calculate_current_reward():
            pegs_in_zone = sum(1 for p in pegs if p.get_center()[0] > target_zone.get_left()[0])
            dist_to_goal = np.linalg.norm(centroid_dot.get_center() - goal_point)
            # Constant w1=5, w2=10
            return (pegs_in_zone**2) * 5 + (10 / (dist_to_goal + 0.1))

        # Scene updater to keep reward_tracker fresh
        def track_reward_value(dt):
            reward_tracker.set_value(calculate_current_reward())
        self.add_updater(track_reward_value)

        # 5. Animations
        # Initial team movement (The 'gravity' pull)
        self.play(
            pegs.animate.shift(RIGHT * 2.5),
            run_time=2,
            rate_func=linear
        )
        
        self.wait(0.5)

        # Moving pieces into the goal one by one
        for i in range(len(pegs)):
            peg = pegs[i]
            target_pos = target_zone.get_center() + UP * np.random.uniform(-2, 2) + RIGHT * np.random.uniform(-0.5, 0.5)
            
            self.play(
                peg.animate.move_to(target_pos),
                run_time=0.8
            )
            
            # Check if peg entered and show visual feedback
            if peg.get_center()[0] > target_zone.get_left()[0]:
                current_count = sum(1 for p in pegs if p.get_center()[0] > target_zone.get_left()[0])
                bonus_popup = Text(f"{current_count}^2 Bonus!", font_size=30, color=YELLOW).next_to(peg, UP)
                self.play(
                    Write(bonus_popup),
                    peg.animate.set_color(YELLOW),
                    run_time=0.4
                )
                self.play(FadeOut(bonus_popup), run_time=0.2)

        self.wait(2)
