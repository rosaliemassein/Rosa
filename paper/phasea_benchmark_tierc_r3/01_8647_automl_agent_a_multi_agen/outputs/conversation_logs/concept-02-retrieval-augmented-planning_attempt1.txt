from manim import *

class Concept02RetrievalAugmentedPlanning(Scene):
    def construct(self):
        # 1. Setup Formula
        formula = MathTex(r"\mathbf{P} = \mathcal{A}_{mgr}(\mathrm{RAP}(R))").to_edge(UP, buff=0.5)
        self.play(Write(formula))

        # 2. Agent Manager (Central Icon)
        manager_box = RoundedRectangle(corner_radius=0.1, height=1.2, width=2.2, color=BLUE, fill_opacity=0.2)
        manager_label = Text("Agent Manager", font_size=20)
        manager = VGroup(manager_box, manager_label).center()

        # 3. Knowledge Base (Cloud-like Icon)
        c1 = Circle(radius=0.4, color=WHITE, fill_opacity=0.3).shift(LEFT * 0.2)
        c2 = Circle(radius=0.4, color=WHITE, fill_opacity=0.3).shift(RIGHT * 0.2)
        c3 = Circle(radius=0.4, color=WHITE, fill_opacity=0.3).shift(UP * 0.2)
        kb_icon = VGroup(c1, c2, c3)
        kb_label = Text("Knowledge Base", font_size=18).next_to(kb_icon, DOWN, buff=0.1)
        knowledge_base = VGroup(kb_icon, kb_label).scale(0.8).to_edge(LEFT, buff=1.2)

        # 4. Display Initial Components
        self.play(FadeIn(knowledge_base), FadeIn(manager))

        # 5. Animate Knowledge Flow (Arrows pulling into Manager)
        arrow_path = Arrow(knowledge_base.get_right(), manager.get_left(), color=YELLOW, buff=0.1)
        self.play(GrowArrow(arrow_path))
        
        # Manual particle animation instead of MoveAlongPath
        for _ in range(3):
            dot = Dot(color=YELLOW, radius=0.06).move_to(arrow_path.get_start())
            self.play(dot.animate.move_to(arrow_path.get_end()), run_time=0.5, rate_func=linear)
            self.remove(dot)

        # 6. Branching into Parallel Plans
        # Plan A
        plan_a_box = RoundedRectangle(height=0.8, width=2, color=GREEN)
        plan_a_text = Text("Plan A", font_size=18).shift(UP * 0.1)
        plan_a_strat = Text("Transformer", font_size=12, color=GRAY).shift(DOWN * 0.2)
        plan_a = VGroup(plan_a_box, plan_a_text, plan_a_strat).move_to(RIGHT * 4 + UP * 1.5)

        # Plan B
        plan_b_box = RoundedRectangle(height=0.8, width=2, color=ORANGE)
        plan_b_text = Text("Plan B", font_size=18).shift(UP * 0.1)
        plan_b_strat = Text("CNN", font_size=12, color=GRAY).shift(DOWN * 0.2)
        plan_b = VGroup(plan_b_box, plan_b_text, plan_b_strat).move_to(RIGHT * 4)

        # Plan C
        plan_c_box = RoundedRectangle(height=0.8, width=2, color=PURPLE)
        plan_c_text = Text("Plan C", font_size=18).shift(UP * 0.1)
        plan_c_strat = Text("Standard", font_size=12, color=GRAY).shift(DOWN * 0.2)
        plan_c = VGroup(plan_c_box, plan_c_text, plan_c_strat).move_to(RIGHT * 4 + DOWN * 1.5)

        plans = [plan_a, plan_b, plan_c]
        arrows_to_plans = VGroup(*[
            Arrow(manager.get_right(), p.get_left(), buff=0.1, color=WHITE) 
            for p in plans
        ])

        # ReplacementTransform is a safer alternative to TransformMatchingShapes
        self.play(
            LaggedStart(
                *[Create(arrow) for arrow in arrows_to_plans],
                *[ReplacementTransform(manager.copy(), plan) for plan in plans],
                lag_ratio=0.2
            )
        )

        # 7. Add specific strategy icons
        t_icon = MathTex(r"\mathcal{T}", color=GREEN).scale(0.8).next_to(plan_a, RIGHT)
        c_icon = MathTex(r"\mathcal{C}", color=ORANGE).scale(0.8).next_to(plan_b, RIGHT)
        s_icon = MathTex(r"\mathcal{S}", color=PURPLE).scale(0.8).next_to(plan_c, RIGHT)

        self.play(Write(t_icon), Write(c_icon), Write(s_icon))

        self.wait(2)
