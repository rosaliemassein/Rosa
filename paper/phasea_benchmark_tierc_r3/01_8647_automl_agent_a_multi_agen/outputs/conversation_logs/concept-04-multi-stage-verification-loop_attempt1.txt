from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup positions
        planning_pos = LEFT * 4.5
        req_pos = LEFT * 1.5
        exec_pos = ORIGIN
        imp_pos = RIGHT * 1.5
        deploy_pos = RIGHT * 4.5

        # 2. UI Elements (Texts and Formula)
        voice_text = Text(
            "Reliability through a three-stage verification gate.",
            font_size=24, color=GREEN
        ).to_edge(UP, buff=0.5)
        
        goal_text = Text(
            "Goal: Iterative feedback loop and verification stages.",
            font_size=20, color=BLUE
        ).next_to(voice_text, DOWN, buff=0.2)
        
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=32
        ).to_edge(DOWN, buff=0.5)

        self.add(voice_text, goal_text, formula)

        # 3. Gates and Labels
        gate_req = DashedLine(UP * 1.5, DOWN * 1.5).move_to(req_pos)
        gate_exec = DashedLine(UP * 1.5, DOWN * 1.5).move_to(exec_pos)
        gate_imp = DashedLine(UP * 1.5, DOWN * 1.5).move_to(imp_pos)
        
        label_req = Text("ReqVer", font_size=18).next_to(gate_req, UP)
        label_exec = Text("ExecVer", font_size=18).next_to(gate_exec, UP)
        label_imp = Text("ImpVer", font_size=18).next_to(gate_imp, UP)

        planning_label = Text("Planning", font_size=20).next_to(planning_pos, UP)
        deploy_icon = VGroup(
            RoundedRectangle(width=1.6, height=0.8, corner_radius=0.1, color=YELLOW),
            Text("DEPLOY", font_size=16, color=YELLOW)
        ).move_to(deploy_pos)

        gates_group = VGroup(
            gate_req, gate_exec, gate_imp, 
            label_req, label_exec, label_imp, 
            planning_label, deploy_icon
        )
        self.add(gates_group)

        # 4. The 'Plan' object (Square)
        plan = Square(side_length=0.4, color=BLUE, fill_opacity=0.6).move_to(planning_pos)
        self.add(plan)

        # 5. Animation Segment 1: Failure at ExecVer
        # Move to and pass ReqVer
        self.play(plan.animate.move_to(req_pos), run_time=1)
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.2)
        
        # Move to and fail ExecVer
        self.play(plan.animate.move_to(exec_pos), run_time=1)
        self.play(plan.animate.set_color(RED))
        
        # Feedback Arc (Manual construction to avoid CurvedArrow issues)
        feedback_path = ArcBetweenPoints(
            exec_pos + DOWN * 0.5, 
            planning_pos + DOWN * 0.5, 
            angle=PI / 2, 
            color=RED
        )
        feedback_path.add_tip()
        feedback_label = Text("Feedback / Revise", font_size=14, color=RED).next_to(feedback_path, DOWN)
        
        self.play(Create(feedback_path), Write(feedback_label))
        self.play(
            plan.animate.move_to(planning_pos).set_color(BLUE),
            FadeOut(feedback_path),
            FadeOut(feedback_label),
            run_time=1.5
        )

        # 6. Animation Segment 2: Full Success
        # Pass ReqVer
        self.play(plan.animate.move_to(req_pos).set_color(GREEN), run_time=0.7)
        # Pass ExecVer
        self.play(plan.animate.move_to(exec_pos).set_color(GREEN), run_time=0.7)
        # Pass ImpVer
        self.play(plan.animate.move_to(imp_pos).set_color(GREEN), run_time=0.7)
        # Final Deployment
        self.play(plan.animate.move_to(deploy_pos), run_time=1)
        self.play(Indicate(deploy_icon, color=YELLOW))
        
        self.wait(2)
