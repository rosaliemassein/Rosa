from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup Labels and Formula
        title = Text("Multi-Stage Verification Loop", font_size=32).to_edge(UP, buff=0.5)
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=28
        ).next_to(title, DOWN, buff=0.2)
        
        self.add(title, formula)

        # 2. Layout Elements
        planning_pos = LEFT * 5
        deployment_pos = RIGHT * 5
        
        planning_box = VGroup(
            RoundedRectangle(width=2, height=1.5, color=BLUE_A),
            Text("Planning", font_size=20)
        ).move_to(planning_pos)

        deployment_icon = VGroup(
            Star(n_points=5, color=YELLOW).scale(0.5),
            Text("Deployment", font_size=20).next_to(ORIGIN, DOWN * 1.5)
        ).move_to(deployment_pos)

        # Gates
        gate_names = ["ReqVer", "ExecVer", "ImpVer"]
        gate_x_coords = [-2, 0, 2]
        gates = VGroup()
        for name, x in zip(gate_names, gate_x_coords):
            line = Line(UP * 1.5, DOWN * 1.5, color=GRAY_A)
            line.shift(RIGHT * x)
            label = Text(name, font_size=18).next_to(line, UP)
            gates.add(VGroup(line, label))

        self.add(planning_box, deployment_icon, gates)

        # 3. The Plan Object
        plan = Square(side_length=0.4, fill_opacity=1, color=BLUE).move_to(planning_pos)
        self.add(plan)

        # 4. Animation Sequence: Failure at ExecVer
        # Move to Gate 1 (ReqVer)
        self.play(plan.animate.move_to(gate_x_coords[0] * RIGHT), run_time=1.5)
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.2)
        
        # Move to Gate 2 (ExecVer) and FAIL
        self.play(plan.animate.move_to(gate_x_coords[1] * RIGHT), run_time=1)
        self.play(plan.animate.set_color(RED))
        
        # Loop back arrow using ArcBetweenPoints
        loop_back_arrow = ArcBetweenPoints(
            start=gate_x_coords[1] * RIGHT + UP * 0.5,
            end=planning_pos + UP * 0.5,
            angle=-TAU/4,
            color=RED
        ).add_tip()
        
        self.play(Create(loop_back_arrow))
        self.play(
            plan.animate.move_to(planning_pos).set_color(BLUE),
            FadeOut(loop_back_arrow),
            run_time=1.5
        )
        self.wait(0.5)

        # 5. Animation Sequence: Full Success
        # Pass Gate 1
        self.play(plan.animate.move_to(gate_x_coords[0] * RIGHT), run_time=1)
        self.play(plan.animate.set_color(GREEN))
        
        # Pass Gate 2
        self.play(plan.animate.move_to(gate_x_coords[1] * RIGHT), run_time=1)
        self.play(plan.animate.set_color(GREEN))
        
        # Pass Gate 3 (ImpVer)
        self.play(plan.animate.move_to(gate_x_coords[2] * RIGHT), run_time=1)
        self.play(plan.animate.set_color(GREEN))
        
        # Reach Deployment
        self.play(plan.animate.move_to(deployment_pos), run_time=1)
        self.play(
            Flash(deployment_icon[0], color=YELLOW, flash_radius=0.5),
            plan.animate.scale(0.5).set_opacity(0)
        )

        # 6. Final Narration/Caption
        narration = Text(
            "Iterative loop ensures only verified models reach deployment.",
            font_size=20,
            color=YELLOW_A
        ).to_edge(DOWN, buff=0.8)
        self.play(Write(narration))
        self.wait(2)
