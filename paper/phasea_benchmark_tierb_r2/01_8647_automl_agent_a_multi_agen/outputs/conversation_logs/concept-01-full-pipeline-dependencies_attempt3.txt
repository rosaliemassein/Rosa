class FullPipelineDependencies(Scene):
    def construct(self):
        # Technical background: Manual grid to avoid NumberPlane restrictions
        grid = VGroup()
        for x in range(-7, 8):
            grid.add(Line([x, -4, 0], [x, 4, 0], stroke_opacity=0.15, stroke_width=1))
        for y in range(-4, 5):
            grid.add(Line([-7, y, 0], [7, y, 0], stroke_opacity=0.15, stroke_width=1))
        self.add(grid)

        # Stage definitions
        stage_names = ["Data\nRetrieval", "Prepro-\ncessing", "Model\nSearch", "Deploy-\nment"]
        
        circles = VGroup()
        labels = VGroup()
        
        for name in stage_names:
            c = Circle(radius=0.9, color=BLUE, fill_opacity=0.2)
            l = Text(name, font_size=20)
            circles.add(c)
            labels.add(l)

        # Assemble and Position Nodes
        nodes = VGroup()
        for i in range(len(circles)):
            labels[i].move_to(circles[i].get_center())
            nodes.add(VGroup(circles[i], labels[i]))
        
        nodes.arrange(RIGHT, buff=0.8).move_to(ORIGIN)

        # Connecting Arrows
        arrows = VGroup()
        for i in range(len(nodes) - 1):
            arr = Arrow(
                circles[i].get_right(),
                circles[i+1].get_left(),
                buff=0.1,
                color=BLUE,
                stroke_width=8,
                max_tip_length_to_length_ratio=0.2
            )
            arrows.add(arr)

        # Pipeline formula at the top
        formula = MathTex(
            r"Pipeline = \{Data \to Preprocessing \to Model \to HPO \to Deployment\}",
            font_size=32
        ).to_edge(UP, buff=0.6)

        # ANIMATION SEQUENCE
