from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup Titles and Reference Formula
        title = Text("Multi-Stage Verification Loop", font_size=36).to_edge(UP)
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=32
        ).to_edge(DOWN)

        # 2. Stage positions
        planning_pos = LEFT * 5 + DOWN * 0.5
        planning_label = Text("Planning", font_size=24).next_to(planning_pos, UP)
        
        deployment_pos = RIGHT * 5 + DOWN * 0.5
        deployment_label = Text("Deployment", font_size=24).next_to(deployment_pos, UP)
        deployment_icon = Star(color=YELLOW, fill_opacity=1).scale(0.4).move_to(deployment_pos)

        # 3. Draw Verification Gates (Vertical Lines)
        gate_list = ["ReqVer", "ExecVer", "ImpVer"]
        gates = VGroup()
        for i, label_text in enumerate(gate_list):
            line = Line(1.2 * UP, 1.2 * DOWN).shift(LEFT * 2 + i * 2.2 * RIGHT)
            label = Text(label_text, font_size=20).next_to(line, UP)
            gates.add(VGroup(line, label))

        self.add(title, formula, planning_label, deployment_label, deployment_icon, gates)

        # 4. Plan Object (Square)
        plan = Square(side_length=0.4, color=BLUE, fill_opacity=0.6).move_to(planning_pos)
        self.play(FadeIn(plan))

        # --- ANIMATION: Initial Run (Pass Stage 1, Fail Stage 2) ---
        
        # Stage 1: ReqVer Pass
        self.play(plan.animate.move_to(gates[0][0].get_center()))
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.2)

        # Stage 2: ExecVer Fail
        self.play(plan.animate.move_to(gates[1][0].get_center()))
        self.play(plan.animate.set_color(RED))
        
        # Draw Loop-back Arrow (Feedback Loop)
        # Using ArcBetweenPoints and add_tip to avoid undefined CurvedArrow identifier
        loop_start = plan.get_bottom()
        loop_end = planning_pos + DOWN * 0.5
        loop_path = ArcBetweenPoints(
            start=loop_start, 
            end=loop_end, 
            angle=PI / 2
        )
        loop_arrow = loop_path.add_tip().set_color(RED)
        
        feedback_text = Text("Feedback / Revise", font_size=18, color=RED).next_to(loop_path, DOWN, buff=0.1)

        self.play(Create(loop_arrow), Write(feedback_text))
        # Move plan back to beginning
        self.play(plan.animate.move_to(planning_pos).set_color(BLUE))
        self.play(FadeOut(loop_arrow), FadeOut(feedback_text))

        # --- ANIMATION: Final Success Run ---
        
        # Pass all gates
        self.play(plan.animate.move_to(gates[0][0].get_center()).set_color(GREEN))
        self.play(plan.animate.move_to(gates[1][0].get_center()).set_color(GREEN))
        self.play(plan.animate.move_to(gates[2][0].get_center()).set_color(GREEN))
        
        # Move to Deployment
        self.play(plan.animate.move_to(deployment_icon.get_center()))
        self.play(Indicate(deployment_icon, color=GREEN))
        self.play(FadeOut(plan))

        # Final goal text
        goal_text = Text(
            "Iterative feedback ensures deployment-ready models.",
            font_size=24, color=BLUE
        ).next_to(formula, UP, buff=0.7)
        self.play(Write(goal_text))
        self.wait(2)
