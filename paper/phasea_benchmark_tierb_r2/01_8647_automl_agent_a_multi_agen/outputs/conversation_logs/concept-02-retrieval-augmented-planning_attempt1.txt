from manim import *

class Concept02RetrievalAugmentedPlanning(Scene):
    def construct(self):
        # 1. Central 'Agent Manager' icon
        manager_box = Rectangle(width=3.2, height=1.2, color=BLUE)
        manager_text = Text("Agent Manager", font_size=30)
        agent_manager = VGroup(manager_box, manager_text)
        agent_manager.move_to(ORIGIN)

        # 2. 'Knowledge Base' icon (Cloud shape)
        cloud_parts = VGroup(
            Circle(radius=0.4).shift(0.25 * UP),
            Circle(radius=0.35).shift(0.25 * DOWN + 0.35 * LEFT),
            Circle(radius=0.35).shift(0.25 * DOWN + 0.35 * RIGHT),
            Circle(radius=0.3).shift(0.45 * LEFT),
            Circle(radius=0.3).shift(0.45 * RIGHT)
        ).set_fill(WHITE, opacity=0.3).set_stroke(WHITE, width=2)
        kb_label = Text("Knowledge Base", font_size=20).next_to(cloud_parts, DOWN)
        knowledge_base = VGroup(cloud_parts, kb_label).to_edge(LEFT, buff=1).shift(UP)

        # 3. Animate arrows and knowledge pulling
        kb_to_manager_arrow = Arrow(knowledge_base.get_right(), agent_manager.get_left(), buff=0.1, color=YELLOW)
        
        # Knowledge particles
        particles = VGroup(*[Dot(radius=0.06, color=YELLOW) for _ in range(3)])
        particles.move_to(knowledge_base.get_right())

        self.play(FadeIn(knowledge_base), FadeIn(agent_manager))
        self.play(Create(kb_to_manager_arrow))
        
        # Movement of knowledge into manager
        self.play(
            particles[0].animate.move_to(agent_manager.get_center()),
            particles[1].animate.move_to(agent_manager.get_center()).set_run_time(1.2),
            particles[2].animate.move_to(agent_manager.get_center()).set_run_time(1.5),
            rate_func=lambda t: t
        )
        self.play(FadeOut(particles))

        # 4. Branching into three distinct plans
        plan_a_icon = Triangle(color=YELLOW).scale(0.3)
        plan_a_label = Text("Plan A: Transformer", font_size=18, color=YELLOW)
        plan_a = VGroup(plan_a_icon, plan_a_label).arrange(RIGHT, buff=0.2)

        plan_b_icon = Square(color=GREEN).scale(0.3)
        plan_b_label = Text("Plan B: CNN", font_size=18, color=GREEN)
        plan_b = VGroup(plan_b_icon, plan_b_label).arrange(RIGHT, buff=0.2)

        plan_c_icon = Circle(color=RED).scale(0.3)
        plan_c_label = Text("Plan C: Hybrid", font_size=18, color=RED)
        plan_c = VGroup(plan_c_icon, plan_c_label).arrange(RIGHT, buff=0.2)

        plans = VGroup(plan_a, plan_b, plan_c).arrange(DOWN, buff=0.9).to_edge(RIGHT, buff=1)

        # Create branching connections manually using Line + add_tip + path_arc
        # This replaces CurvedArrow to avoid potential undefined identifier issues
        conn_a = Line(agent_manager.get_right(), plan_a.get_left(), path_arc=-0.4).add_tip()
        conn_b = Line(agent_manager.get_right(), plan_b.get_left()).add_tip()
        conn_c = Line(agent_manager.get_right(), plan_c.get_left(), path_arc=0.4).add_tip()
        
        branching_connectors = VGroup(conn_a, conn_b, conn_c)

        # 5. Transform and branching animation
        # Using ReplacementTransform for the branching effect as requested
        self.play(
            ReplacementTransform(agent_manager.copy(), plans),
            Create(branching_connectors),
            run_time=2
        )
        self.wait(1)

        # 6. Formula display
        formula = MathTex(r"\mathbf{P} = \mathcal{A}_{mgr}(\mathrm{RAP}(R))", color=BLUE).scale(1.1)
        formula.to_edge(DOWN, buff=0.7)
        
        self.play(Write(formula))
        self.play(Indicate(formula, color=YELLOW))
        self.wait(2)
