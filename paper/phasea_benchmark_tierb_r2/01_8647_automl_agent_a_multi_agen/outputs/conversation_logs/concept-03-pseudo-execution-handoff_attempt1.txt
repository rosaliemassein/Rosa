from manim import *

class PseudoExecutionHandoff(Scene):
    def construct(self):
        # 1. Setup Agents and Layout
        # Using standard colors (BLUE, RED, GREEN, YELLOW, GRAY) to avoid undefined color variant errors
        data_agent = VGroup(
            Circle(radius=0.8, color=BLUE, fill_opacity=0.2),
            Text("Data Agent (A_d)", font_size=24)
        ).shift(UP * 2 + LEFT * 3.5)
        
        model_agent = VGroup(
            Circle(radius=0.8, color=RED, fill_opacity=0.2),
            Text("Model Agent (A_m)", font_size=24)
        ).shift(DOWN * 2 + LEFT * 3.5)
        
        # 2. Data Block and Insights
        data_block = Rectangle(width=2.5, height=1.2, color=BLUE, fill_opacity=0.1).next_to(data_agent, RIGHT, buff=1)
        data_text = Text("Data Block", font_size=22).move_to(data_block.get_center())
        
        # 3. Efficiency Bar (Bottom)
        eff_label = Text("Search Efficiency", font_size=24).to_edge(DOWN).shift(LEFT * 3)
        eff_bar_bg = Rectangle(width=6, height=0.4, color=GRAY).next_to(eff_label, RIGHT)
        # Start with a very small width to animate growth
        eff_bar_fill = Rectangle(width=0.01, height=0.4, color=GREEN, fill_opacity=1).align_to(eff_bar_bg, LEFT)
        
        # 4. Formula
        formula = MathTex(r"s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)").scale(0.9).to_corner(UR)

        # --- Execution ---
        self.add(data_agent, model_agent, eff_label, eff_bar_bg, eff_bar_fill)
        self.play(Create(data_block), Write(data_text))
        self.wait(0.5)
        
        # Pseudo-analysis: Data Block -> Data Insights (O_d)
        insights_label = MathTex("O_i^d").scale(1.2).move_to(data_block.get_center())
        insights_text = Text("Data Insights", font_size=18).next_to(insights_label, DOWN, buff=0.1)
        insights_group = VGroup(insights_label, insights_text)
        
        self.play(
            Transform(data_text, insights_group),
            data_block.animate.set_fill(opacity=0.3)
        )
        self.wait(1)
        
        # 5. Handoff to Model Agent
        handoff_arrow = Arrow(data_block.get_bottom(), DOWN * 1.2 + data_block.get_x() * RIGHT, color=YELLOW)
        self.play(Create(handoff_arrow))
        
        # Moving O_d down to Model Agent area
        s_m_block = Rectangle(width=3, height=1.5, color=RED, fill_opacity=0.1).next_to(model_agent, RIGHT, buff=1)
        s_m_label = MathTex("s_m").scale(1.5).move_to(s_m_block.get_center())
        
        # Transition: Data insights move down and transform into the model task s_m
        self.play(
            data_text.animate.move_to(s_m_block.get_center()).scale(0.8),
            FadeOut(data_block),
            FadeOut(handoff_arrow)
        )
        
        # 6. Model Agent task expansion and Efficiency Increase
        self.play(
            Create(s_m_block),
            Transform(data_text, s_m_label),
            Write(formula)
        )
        
        # Animate task expansion and efficiency bar simultaneously
        # Replaced 'linear' identifier with lambda for maximum compatibility
        self.play(
            s_m_block.animate.scale(1.3).set_fill(opacity=0.4),
            eff_bar_fill.animate.stretch_to_fit_width(6).align_to(eff_bar_bg, LEFT),
            run_time=4,
            rate_func=lambda t: t
        )
        
        self.wait(2)
