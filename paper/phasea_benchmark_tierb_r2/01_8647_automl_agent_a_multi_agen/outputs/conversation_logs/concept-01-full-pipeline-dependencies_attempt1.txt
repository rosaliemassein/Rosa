from manim import *

class ConceptPipelineDependencies(Scene):
    def construct(self):
        # Manual background grid for technical feel
        grid = VGroup()
        for i in range(-7, 8):
            grid.add(Line([i, -4, 0], [i, 4, 0], stroke_width=1, stroke_opacity=0.15, color=BLUE))
        for j in range(-4, 5):
            grid.add(Line([-7, j, 0], [7, j, 0], stroke_width=1, stroke_opacity=0.15, color=BLUE))
        self.add(grid)

        # Step definitions
        step_names = ["Data Retrieval", "Preprocessing", "Model Search", "Deployment"]
        nodes = VGroup()
        for name in step_names:
            c = Circle(radius=0.9, color=BLUE, fill_opacity=0.1)
            t = Text(name, font_size=18).next_to(c, DOWN)
            node = VGroup(c, t)
            nodes.add(node)
        
        # Position nodes
        nodes.arrange(RIGHT, buff=0.7).shift(UP * 0.5)

        # Create connection arrows
        arrows = VGroup()
        for i in range(len(nodes) - 1):
            arr = Arrow(
                nodes[i][0].get_right(), 
                nodes[i+1][0].get_left(), 
                stroke_width=6, 
                buff=0.1, 
                color=BLUE_C
            )
            arrows.add(arr)

        # Formula at the bottom
        formula = MathTex(
            r"\text{Pipeline} = \{ \text{Data} \to \text{Preprocessing} \to \text{Model} \to \text{HPO} \to \text{Deployment} \}",
            font_size=28
        ).to_edge(DOWN, buff=0.6)

        # Initial animations
        self.play(Create(grid), run_time=1)
        self.play(
            LaggedStart(
                *[FadeIn(node) for node in nodes],
                lag_ratio=0.2
            )
        )
        self.play(Create(arrows))
        self.play(Write(formula))
        self.wait(1)

        # Ripple through animation
        # 1. Pulse 'Data Retrieval' (first circle)

        # 2. Sequential propagation through arrows and nodes

            self.play(
                arrows[i].animate.set_color(GOLD),
                nodes[i+1][0].animate.set_color(GOLD).set_fill(GOLD, opacity=0.3),
                run_time=0.6
            )
        
        # Final emphasis
        final_text = Text("Pipeline Inter-dependency", color=GOLD, font_size=32).to_edge(UP)
        self.play(Write(final_text))
        self.wait(2)
