from manim import *

class Concept01BorderOptimization(Scene):
    def construct(self):
        # 1. Formula reference
        formula = MathTex(r"B = \sum_{i,j \in \text{Neighbors}} [P_i \neq P_j]").to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Create 4x4 grid of Square objects
        # We'll use a specific layout to match the prompt requirements
        grid_buff = 0.2
        side = 0.8
        squares = VGroup(*[
            Square(side_length=side, fill_opacity=0.8, stroke_color=WHITE, stroke_width=1)
            for _ in range(16)
        ]).arrange_in_grid(rows=4, cols=4, buff=grid_buff).shift(LEFT * 2)

        # 3. Initially color the left half red and the right half blue
        # Index i: row = i // 4, col = i % 4
        for i in range(16):
            col = i % 4
            if col < 2:
                squares[i].set_fill(RED)
            else:
                squares[i].set_fill(BLUE)

        self.add(squares)

        # 4. Create a counter for the border count
        count_title = Text("Border Adjacencies:", font_size=28).to_edge(RIGHT, buff=1)
        count_num = DecimalNumber(0, num_decimal_places=0, color=GREEN).scale(1.5).next_to(count_title, DOWN)
        self.add(count_title, count_num)

        # 5. Define dynamic border drawing logic
        # This uses always_redraw to update the green lines every frame
        def get_borders():
            borders = VGroup()
            for i in range(16):
                r, c = divmod(i, 4)
                # Check neighbor to the right
                if c < 3:
                    if squares[i].get_fill_color() != squares[i+1].get_fill_color():
                        # Vertical line between squares
                        mid_x = (squares[i].get_right()[0] + squares[i+1].get_left()[0]) / 2
                        y_center = squares[i].get_center()[1]
                        line = Line(
                            [mid_x, y_center - side/2, 0],
                            [mid_x, y_center + side/2, 0],
                            color=GREEN, stroke_width=8
                        )
                        borders.add(line)
                # Check neighbor below
                if r < 3:
                    if squares[i].get_fill_color() != squares[i+4].get_fill_color():
                        # Horizontal line between squares
                        mid_y = (squares[i].get_bottom()[1] + squares[i+4].get_top()[1]) / 2
                        x_center = squares[i].get_center()[0]
                        line = Line(
                            [x_center - side/2, mid_y, 0],
                            [x_center + side/2, mid_y, 0],
                            color=GREEN, stroke_width=8
                        )
                        borders.add(line)
            return borders

        # Helper to update the adjacency count
        def get_adj_count():
            count = 0
            for i in range(16):
                r, c = divmod(i, 4)
                if c < 3 and squares[i].get_fill_color() != squares[i+1].get_fill_color():
                    count += 1
                if r < 3 and squares[i].get_fill_color() != squares[i+4].get_fill_color():
                    count += 1
            return count

        # Add the borders as an updating object
        dynamic_borders = always_redraw(get_borders)
        self.add(dynamic_borders)
        
        # Link counter value to the current state
        count_num.add_updater(lambda d: d.set_value(get_adj_count()))

        self.wait(1)

        # 6. Animate squares flipping to checkerboard pattern
        # Target state: (row + col) % 2 == 0 ? RED : BLUE
        to_flip = []
        for i in range(16):
            r, c = divmod(i, 4)
            target_color = RED if (r + c) % 2 == 0 else BLUE
            if squares[i].get_fill_color() != target_color:
                to_flip.append(i)

        # Perform the sequential flips
        for idx in to_flip:
            target_color = RED if ( (idx // 4) + (idx % 4) ) % 2 == 0 else BLUE
            self.play(
                squares[idx].animate.set_fill(target_color),
                run_time=0.3
            )
            # The updater for dynamic_borders and count_num handles the visual changes automatically

        self.wait(2)
