from manim import *
import numpy as np

class Concept03StateSpaceExplosion(Scene):
    def construct(self):
        # Define local rate function and color constants to avoid undefined identifier issues
        def ease_in_q(t):
            return t * t

        C_BLUE = "#58C4DD"
        C_RED = "#FC6255"
        C_GRAY = "#888888"

        # 1. Formula Display
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Create Coordinate System using NumberPlane (as a fallback for Axes)
        # x-axis: Matrix Size (2 to 6)
        # y-axis: 2^n^2 in log scale (showing the power)
        ax = NumberPlane(
            x_range=[2, 7, 1],
            y_range=[0, 40, 10],
            x_length=6,
            y_length=5,
            background_line_style={"stroke_opacity": 0.2},
            axis_config={"include_numbers": True}
        ).shift(LEFT * 2.5 + DOWN * 0.5)
        
        y_label = MathTex(r"\log_2(\text{States})").rotate(90 * DEGREES).next_to(ax, LEFT)
        x_label = Text("Matrix Size (N)", font_size=24).next_to(ax, DOWN)
        
        self.play(Create(ax), FadeIn(y_label), FadeIn(x_label))

        # 3. Initial 2x2 Grid and Q-Table representation
        grid = VGroup(*[
            Square(side_length=0.4).set_fill(C_BLUE, opacity=0.7).set_stroke(WHITE, 1) 
            for _ in range(4)
        ]).arrange_in_grid(rows=2, cols=2, buff=0.1).to_edge(RIGHT, buff=1.5).shift(UP * 1.5)
        
        grid_title = Text("2x2 Matrix", font_size=24).next_to(grid, UP)

        # Mock Q-Table Frame
        q_table_frame = Rectangle(height=3, width=2, color=WHITE).next_to(grid, DOWN, buff=1)
        q_table_header = Text("Q-Table", font_size=24).next_to(q_table_frame, UP, buff=0.1)
        
        # Initial rows
        rows = VGroup(*[
            Line(LEFT, RIGHT, color=C_GRAY).scale(0.8) 
            for _ in range(12)
        ]).arrange(DOWN, buff=0.15).move_to(q_table_frame)

        self.play(
            Create(grid), 
            Write(grid_title), 
            Create(q_table_frame), 
            Create(rows), 
            Write(q_table_header)
        )
        self.wait(1)

        # 4. The Growth Animation
        # Define the plotting function explicitly to avoid lambda 'x' issues
        def plot_func(val):
            return val**2

        graph = ax.plot(plot_func, x_range=[2, 5.5], color=C_RED)
        
        # Transition grid to 5x5
        grid_5x5 = VGroup(*[
            Square(side_length=0.2).set_fill(C_BLUE if i % 2 == 0 else C_RED, opacity=0.7).set_stroke(WHITE, 0.5) 
            for i in range(25)
        ]).arrange_in_grid(rows=5, cols=5, buff=0.05).move_to(grid)
        
        grid_5x5_title = Text("5x5 Matrix", font_size=24).next_to(grid_5x5, UP)

        # Simulate massive row multiplication by scrolling many lines rapidly
        scrolling_rows = VGroup(*[
            Line(LEFT, RIGHT, color=C_GRAY, stroke_width=1).scale(0.8) 
            for _ in range(100)
        ]).arrange(DOWN, buff=0.04).move_to(q_table_frame.get_bottom(), aligned_edge=DOWN)

        # Animation sequence
        self.play(
            Create(graph),
            Transform(grid, grid_5x5),
            Transform(grid_title, grid_5x5_title),
            scrolling_rows.animate.shift(UP * 10),
            run_time=4,
            rate_func=ease_in_q
        )

        # 5. Representing State Space Explosion
        blur = Rectangle(
            height=3, width=2, 
            fill_opacity=0.9, 
            stroke_width=0
        ).set_fill(color=[C_BLUE, C_RED]).move_to(q_table_frame)
        
        explosion_text = Text("33M+ States", font_size=24, color=WHITE).move_to(blur)
        
        # Curve shoots off the screen
        self.play(
            FadeIn(blur),
            Write(explosion_text),
            graph.animate.shift(UP * 15),
            run_time=1
        )
        
        self.wait(2)
