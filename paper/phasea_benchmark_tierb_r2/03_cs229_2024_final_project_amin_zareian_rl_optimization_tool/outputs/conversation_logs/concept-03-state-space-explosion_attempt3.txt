from manim import *

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Create Axes (Pattern from Example 3)
        # x-axis: Matrix Size (2 to 6)
        # y-axis: Possible States (log scale)
        # Total States = 2^(N*N) => log2(States) = N^2. We plot n^2 on a linear y-axis.
        ax = Axes(
            x_range=[2, 7, 1],
            y_range=[0, 45, 10],
            x_length=7,
            y_length=4,
            axis_config={"include_numbers": True}
        ).to_edge(LEFT, buff=0.5)

        xlabel = ax.get_x_axis_label("N (Matrix Size)").scale(0.8)
        ylabel = ax.get_y_axis_label(r"\log_2(\text{States})").scale(0.8)

        # 2. Plotting the function (Pattern from Example 3)
        # Using 'n' instead of 'x' to avoid potential identifier issues
        graph = ax.plot(lambda n: n**2, x_range=[2, 6.2], color=YELLOW)
        
        # 3. Formula reference
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP)

        # 4. Initial 2x2 Grid setup
        # Represents 2^4 = 16 states
        grid = VGroup(*[
            Square(side_length=0.4, fill_opacity=0.5, fill_color=BLUE)
            for _ in range(4)
        ]).arrange_in_grid(2, 2, buff=0.1)
        grid.to_edge(RIGHT, buff=1.5).shift(UP * 1.5)
        grid_title = Text("2x2 Grid", font_size=24).next_to(grid, UP)

        # 5. Q-Table Representation
        # A box containing lines representing Q-table rows
        table_bg = Rectangle(width=2.2, height=3.2, color=WHITE).next_to(grid, DOWN, buff=0.5)
        table_rows = VGroup(*[
            Line(LEFT, RIGHT, color=BLUE).set_width(2)
            for _ in range(12)
        ]).arrange(DOWN, buff=0.15).move_to(table_bg)
        table_title = Text("Q-Table Rows", font_size=24).next_to(table_bg, UP, buff=0.1)

        # 6. Intro Animations
        self.add(ax, xlabel, ylabel, formula, grid, grid_title, table_bg, table_rows, table_title)
        self.play(Create(graph), run_time=2)
        self.wait(0.5)

        # 7. State Space Explosion Animation
        # Target: 5x5 Grid (2^25 states)
        grid_5x5 = VGroup(*[
            Square(side_length=0.15, fill_opacity=0.5, fill_color=RED)
            for _ in range(25)
        ]).arrange_in_grid(5, 5, buff=0.05).move_to(grid)
        
        # High-speed scrolling simulation using many dense lines
        fast_rows = VGroup(*[
            Line(LEFT, RIGHT, color=RED, stroke_width=1).set_width(2)
            for _ in range(50)
        ]).arrange(DOWN, buff=0.03).move_to(table_bg)
        
        # Blur represention of millions of states
        blur_rect = Rectangle(
            width=2.2, height=3.2, 
            fill_opacity=0.9, 
            stroke_width=0
        ).move_to(table_bg)
        blur_rect.set_fill(color=[BLUE, RED, PURPLE])

        # Execute transformation sequence
        self.play(
            Transform(grid, grid_5x5),
            grid_title.animate.become(Text("5x5 Grid", font_size=24).next_to(grid, UP)),
            table_rows.animate.shift(UP * 4).set_opacity(0),
            FadeIn(fast_rows),
            run_time=3
        )
        
        # Final state: Graph shoots off top, table turns into blur
        self.play(
            FadeIn(blur_rect),
            graph.animate.shift(UP * 10),
            ax.animate.shift(DOWN * 3),
            run_time=2
        )
        
        # Conclusion Narration
        narration = Text(
            "State space grows exponentially with matrix size.",
            font_size=28,
            color=BLUE
        ).to_edge(DOWN)
        
        self.play(Write(narration))
        self.wait(2)
