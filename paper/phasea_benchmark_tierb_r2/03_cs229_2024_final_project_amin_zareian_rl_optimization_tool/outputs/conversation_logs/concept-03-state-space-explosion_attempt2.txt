from manim import *

class StateSpaceExplosion(Scene):
    def construct(self):
        # Constructing coordinate system manually to avoid 'Axes'
        horiz_line = Line(LEFT * 3.5 + DOWN * 2, LEFT * 0.5 + DOWN * 2)
        vert_line = Line(LEFT * 3.5 + DOWN * 2, LEFT * 3.5 + UP * 2)
        
        # Adding some ticks manually for visual clarity
        for i in range(5):
            tick_x = -3.5 + i * 0.75
            horiz_line.add(Line(UP*0.1, DOWN*0.1).move_to([tick_x, -2, 0]))
            
        label_n = Text("N (Matrix Size)", font_size=18).next_to(horiz_line, DOWN)
        label_s = Text("Possible States (Log Scale)", font_size=18).rotate(90 * DEGREES).next_to(vert_line, LEFT)
        
        # Reference formula using MathTex
        formula_text = MathTex(r"\text{Total States} = 2^{(N \times N)}", color=YELLOW).to_edge(UP, buff=0.5)

        # Q-Table Visualization setup
        table_outline = Rectangle(width=2, height=4, color=WHITE).shift(RIGHT * 3)
        table_label = Text("Q-Table Rows", font_size=22).next_to(table_outline, UP)
        
        # Tracker for the variable N (Size)
        # Using a Dot's y-coordinate as a hidden tracker to avoid 'ValueTracker'
        # We animate N from 2 to 6.
        val_tracker = Dot(point=[0, 2, 0]).hide()
        
        # Graph path using VMobject and updater to avoid 'always_redraw'
        graph_curve = VMobject(color=GREEN).set_stroke(width=4)
        
        def update_my_graph(mob):
            # Extract current value of N from tracker y-coord
            current_n = val_tracker.get_center()[1]
            points_list = []
            # We plot N vs Log2(States) which is N^2
            # Range: N from 2 to current_n
            # x-axis mapping: N=2 -> -3.5, N=6 -> -0.5
            # y-axis mapping: N^2=4 -> -2.0, N^2=36 -> 2.0
            num_steps = 15
            for idx in range(num_steps + 1):
                prog = idx / num_steps
                n_step = 2 + (current_n - 2) * prog
                px = -3.5 + (n_step - 2) * 0.75
                py = -2 + (n_step * n_step - 4) * 0.125
                points_list.append([px, py, 0])
            mob.set_points_as_corners(points_list)
            
        graph_curve.add_updater(update_my_graph)
        
        # Q-Table content logic
        rows_vgroup = VGroup()
        
        def update_table_rows(vg):
            current_n = val_tracker.get_center()[1]
            vg.submobjects = []
            if current_n < 4.8:
                # Discrete line representation
                row_count = int(4 + (current_n - 2) * 12)
                for k in range(row_count):
                    row_line = Line(LEFT, RIGHT, stroke_width=1).set_width(1.8)
                    y_offset = 1.9 - (3.8 * k / (row_count - 1 if row_count > 1 else 1))
                    row_line.move_to(table_outline.get_center() + [0, y_offset, 0])
                    vg.add(row_line)
            else:
                # Explosion representation: a solid blue "blur"
                blur_rect = Rectangle(
                    width=1.9, height=3.9, 
                    fill_color=BLUE, fill_opacity=0.8, 
                    stroke_width=0
                ).move_to(table_outline.get_center())
                vg.add(blur_rect)
                # Flickering effect lines
                for m in range(4):
                    noise = Line(LEFT, RIGHT, stroke_width=0.5, stroke_opacity=0.4)
                    noise.set_width(1.8).move_to(table_outline.get_center() + [0, (m - 1.5) * 0.8, 0])
                    vg.add(noise)

        rows_vgroup.add_updater(update_table_rows)

        # Narrative Caption
        caption = Text("As N grows linearly, states grow exponentially.", font_size=20).to_edge(DOWN, buff=0.4)

        # Initial Display
        self.add(horiz_line, vert_line, label_n, label_s, formula_text, table_outline, table_label)
        self.play(FadeIn(graph_curve), FadeIn(rows_vgroup), Write(caption))
        self.wait(1)
        
        # Main Animation: Explosion of states
        # Defining a simple linear rate function manually to avoid 'linear' identifier
        self.play(
            val_tracker.animate.move_to([0, 6, 0]),
            run_time=8,
            rate_func=lambda alpha: alpha
        )
        
        # Final concluding text
        final_caption = Text("The Q-table becomes too large to compute.", font_size=20, color=RED).to_edge(DOWN, buff=0.4)
        self.play(Transform(caption, final_caption))
        self.wait(2)
