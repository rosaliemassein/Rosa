from manim import *

class Concept02RewardDynamics(Scene):
    def construct(self):
        # 1. Setup Grid (3x3 pixels)
        # Using Square to represent pixels
        grid = VGroup(*[
            Square(side_length=1).set_stroke(GRAY, 0.5) 
            for _ in range(9)
        ]).arrange_in_grid(3, 3, buff=0.1)
        
        # Initial colors: Center is Red, 3 neighbors Red, 1 neighbor Blue
        # This setup ensures flipping center Red -> Blue creates 2 net new borders
        # (Loses 1 border with the top Blue, gains 3 with the other Red neighbors)
        # Total change: +2
        colors = [
            WHITE, BLUE,  WHITE,
            RED,   RED,   RED,
            WHITE, RED,   WHITE
        ]
        # Only set color for the 3x3 core
        for i, color in enumerate(colors):
            grid[i].set_fill(color, opacity=0.8)

        self.add(grid)

        # 2. Narration Label & Formula
        formula = MathTex("R_t = Border_{t} - Border_{t-1}", font_size=36).to_edge(UP)
        self.add(formula)

        # 3. Focus on Center Pixel and Neighbors
        center_idx = 4
        neighbor_indices = [1, 3, 5, 7] # Up, Left, Right, Down
        
        center_pixel = grid[center_idx]
        neighbors = VGroup(*[grid[i] for i in neighbor_indices])
        
        self.play(Indicate(center_pixel, color=WHITE))
        self.play(neighbors.animate.set_stroke(YELLOW, 4))
        self.wait(0.5)

        # 4. Flip Animation and Reward logic
        # Positions for border highlights
        cp = center_pixel.get_center()
        # Top border (Existing boundary between Red and Blue)
        b_top = Line(cp + [-0.5, 0.5, 0], cp + [0.5, 0.5, 0], color=RED, stroke_width=8)
        # Lost border marker
        red_x = Cross(b_top, stroke_color=RED, stroke_width=6).scale(0.2)
        
        # New borders (will appear after flip)
        b_left = Line(cp + [-0.5, -0.5, 0], cp + [-0.5, 0.5, 0], color=GREEN, stroke_width=8)
        b_right = Line(cp + [0.5, -0.5, 0], cp + [0.5, 0.5, 0], color=GREEN, stroke_width=8)
        b_down = Line(cp + [-0.5, -0.5, 0], cp + [0.5, -0.5, 0], color=GREEN, stroke_width=8)

        # Execution
        self.play(center_pixel.animate.set_fill(BLUE))
        
        # Show new borders created and the one lost
        self.play(
            Create(red_x),
            Create(b_left), 
            Create(b_right), 
            Create(b_down),
            run_time=1
        )

        # 5. Floating Reward Text
        reward_text = Text("+2", color=GREEN, font_size=48).next_to(center_pixel, RIGHT, buff=0.6)
        self.play(Write(reward_text), reward_text.animate.shift(UP * 0.5))

        # 6. Manual Bar Chart (Avoiding 'Axes' as it was flagged as disallowed)
        chart_origin = RIGHT * 4.5 + DOWN * 1.5
        x_axis = Line(chart_origin, chart_origin + RIGHT * 2.5)
        y_axis = Line(chart_origin, chart_origin + UP * 3)
        y_label = Text("Reward", font_size=18).next_to(y_axis, UP)
        
        # Bar 1 (Previous step reward)
        bar_t_minus_1 = Rectangle(width=0.4, height=0.8, fill_opacity=0.8, color=BLUE).move_to(chart_origin + RIGHT * 0.6 + UP * 0.4)
        # Bar 2 (Current step reward)
        bar_t = Rectangle(width=0.4, height=2.2, fill_opacity=0.8, color=GREEN).move_to(chart_origin + RIGHT * 1.4 + UP * 1.1)
        
        chart_group = VGroup(x_axis, y_axis, y_label)
        
        self.play(Create(chart_group))
        self.play(FadeIn(bar_t_minus_1))
        self.play(FadeIn(bar_t, shift=UP))

        # Final transition of reward value to chart
        self.play(reward_text.animate.move_to(bar_t.get_top() + UP * 0.3).scale(0.6))

        self.wait(2)
