from manim import *

class Concept04HashedQLearning(Scene):
    def construct(self):
        # 1. Formula Header
        formula = MathTex(r"\text{Index} = \text{Hash}(\text{MatrixState}) \pmod L")
        formula.to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Create three different 4x4 grid patterns using Matrix
        # Pattern 1: Checkerboard
        p1 = [[(i + j) % 2 for i in range(4)] for j in range(4)]
        # Pattern 2: Diagonal blocks
        p2 = [[1 if i < 2 else 0 for i in range(4)] for j in range(4)]
        # Pattern 3: Border pattern
        p3 = [[1 if i == 0 or i == 3 or j == 0 or j == 3 else 0 for i in range(4)] for j in range(4)]

        grid1 = Matrix(p1, h_buff=0.5, v_buff=0.5).scale(0.3)
        grid2 = Matrix(p2, h_buff=0.5, v_buff=0.5).scale(0.3)
        grid3 = Matrix(p3, h_buff=0.5, v_buff=0.5).scale(0.3)

        # Arrange grids on the left
        grid_group = VGroup(grid1, grid2, grid3).arrange(DOWN, buff=0.5).to_edge(LEFT, buff=1)

        # 3. Hash Function Box (Central)
        hash_label = Text("Hash Function", font_size=24)
        hash_box_rect = Rectangle(width=3, height=1.5, color=WHITE)
        hash_function = VGroup(hash_box_rect, hash_label).move_to(ORIGIN)

        # 4. Compact Q-Table (Right)
        # Fixed size table with 8 rows
        table_rows = VGroup(*[
            Rectangle(width=2, height=0.4, stroke_width=2) for _ in range(8)
        ]).arrange(DOWN, buff=0.05).to_edge(RIGHT, buff=1)
        q_table_label = Text("Q-Table", font_size=24).next_to(table_rows, UP)
        q_table = VGroup(table_rows, q_table_label)

        # 5. Define Arrows
        # From Grids to Hash Box
        arrow_g1_to_hash = Arrow(grid1.get_right(), hash_function.get_left(), buff=0.1)
        arrow_g2_to_hash = Arrow(grid2.get_right(), hash_function.get_left(), buff=0.1)
        arrow_g3_to_hash = Arrow(grid3.get_right(), hash_function.get_left(), buff=0.1)

        # From Hash Box to Table Rows
        # Grid 1 and 2 map to the same row (Collision)
        collision_row = table_rows[2]
        normal_row = table_rows[5]

        arrow_hash_to_coll = Arrow(hash_function.get_right(), collision_row.get_left(), buff=0.1, color=RED)
        arrow_hash_to_norm = Arrow(hash_function.get_right(), normal_row.get_left(), buff=0.1)

        # 6. Animation Sequence
        self.play(
            Create(grid_group),
            Create(hash_function),
            Create(q_table),
            rate_func=smooth
        )
        self.wait(0.5)

        # Draw arrows from grids to hash function
        self.play(
            GrowArrow(arrow_g1_to_hash),
            GrowArrow(arrow_g2_to_hash),
            GrowArrow(arrow_g3_to_hash)
        )
        self.wait(0.5)

        # Draw arrows from hash function to table
        # First a normal mapping
        self.play(GrowArrow(arrow_hash_to_norm))
        self.wait(0.5)

        # Then the collision mapping
        self.play(
            GrowArrow(arrow_hash_to_coll),
            collision_row.animate.set_fill(RED, opacity=0.3).set_stroke(RED, width=4)
        )

        # Highlight collision using Flash and text
        collision_text = Text("Collision!", color=RED, font_size=20).next_to(collision_row, LEFT, buff=0.2)
        self.play(
            Flash(collision_row, color=RED, flash_radius=0.6),
            Write(collision_text)
        )
        
        self.wait(2)
