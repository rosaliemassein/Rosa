from manim import *
import numpy as np

class ConceptKMeansSpatialQuantization(Scene):
    def construct(self):
        # 1. Coordinate System (Using Axes instead of NumberPlane to avoid restrictions)
        axes = Axes(
            x_range=[-7, 7, 1],
            y_range=[-4, 4, 1],
            axis_config={"color": BLUE_D, "stroke_width": 2},
            tips=False
        ).fade(0.7)
        self.add(axes)

        # 2. Draw Australia Outline
        aus_points = [
            [-3.5, -1, 0], [-3, -2.5, 0], [-1, -2.5, 0], [0.5, -3.5, 0],
            [3, -3, 0], [4, -1, 0], [3.5, 1, 0], [1, 3, 0],
            [-1, 2, 0], [-2, 2.5, 0], [-3.5, 1.5, 0]
        ]
        australia = Polygon(*aus_points, color=WHITE, stroke_width=2)
        australia.set_fill(GRAY_E, opacity=0.3)
        self.play(Create(australia))

        # 3. Generate Data Points (Gray dots)
        np.random.seed(42)
        points_data = []
        while len(points_data) < 80:
            test_p = np.array([np.random.uniform(-4, 4.5), np.random.uniform(-4, 3.5), 0])
            if self.is_inside(test_p, aus_points):
                points_data.append(test_p)
        
        dots = VGroup(*[Dot(p, radius=0.05, color=GRAY) for p in points_data])
        self.play(FadeIn(dots, lag_ratio=0.05), run_time=1.5)

        # 4. Centroids (mu_j) appearing as crosses
        colors = [RED, YELLOW, GREEN]
        start_pos = [LEFT * 2 + UP, RIGHT * 2 + UP, ORIGIN + DOWN]
        
        centroids = VGroup()
        for i in range(3):
            cross = VGroup(
                Line([-0.15, -0.15, 0], [0.15, 0.15, 0]),
                Line([-0.15, 0.15, 0], [0.15, -0.15, 0])
            ).set_stroke(width=5).set_color(colors[i]).move_to(start_pos[i])
            centroids.add(cross)

        mu_labels = VGroup(*[
            MathTex(fr"\mu_{i+1}", color=colors[i], font_size=30).next_to(centroids[i], UR, buff=0.05)
            for i in range(3)
        ])

        self.play(
            LaggedStart(*[FadeIn(VGroup(centroids[i], mu_labels[i])) for i in range(3)], lag_ratio=0.4)
        )
        self.wait(0.5)

        # 5. Color points based on nearest centroid
        def update_dot_colors():
            anims = []
            for dot in dots:
                dists = [np.linalg.norm(dot.get_center() - c.get_center()) for c in centroids]
                nearest_idx = np.argmin(dists)
                anims.append(dot.animate.set_color(colors[nearest_idx]))
            return anims

        self.play(*update_dot_colors())
        self.wait(0.5)

        # 6. Show Convergence Formula
        formula = MathTex(
            r"\mu_{j}:=\frac{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}x^{(i)}}{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}}",
            font_size=32
        ).to_edge(UP, buff=0.5).add_background_rectangle()
        self.play(Write(formula))

        # 7. Move centroids to the centers of their groups
        move_anims = []
        for i in range(3):
            assigned_points = [
                d.get_center() for d in dots 
                if np.argmin([np.linalg.norm(d.get_center() - c.get_center()) for c in centroids]) == i
            ]
            if assigned_points:
                target_pos = np.mean(assigned_points, axis=0)
                move_anims.append(centroids[i].animate.move_to(target_pos))
                move_anims.append(mu_labels[i].animate.next_to(target_pos, UR, buff=0.05))

        self.play(*move_anims, run_time=2)
        
        # 8. Re-color dots once more for final groups
        self.play(*update_dot_colors())

        # 9. Conclusion text
        goal_box = Text(
            "Regression turns into a manageable classification task.",
            font_size=24, color=WHITE
        ).to_edge(DOWN, buff=0.5).add_background_rectangle()
        
        self.play(FadeIn(goal_box))
        self.play(*[Indicate(c) for c in centroids])
        self.wait(3)

    def is_inside(self, point, poly):
        """Ray casting algorithm to determine if point is inside polygon."""
        x, y = point[0], point[1]
        n = len(poly)
        inside = False
        p1x, p1y = poly[0][0], poly[0][1]
        for i in range(n + 1):
            p2x, p2y = poly[i % n][0], poly[i % n][1]
            if y > min(p1y, p2y):
                if y <= max(p1y, p2y):
                    if x <= max(p1x, p2x):
                        if p1y != p2y:
                            xints = (y - p1y) * (p2x - p1x) / (p2y - p1y) + p1x
                        if p1x == p2x or x <= xints:
                            inside = not inside
            p1x, p1y = p2x, p2y
        return inside
