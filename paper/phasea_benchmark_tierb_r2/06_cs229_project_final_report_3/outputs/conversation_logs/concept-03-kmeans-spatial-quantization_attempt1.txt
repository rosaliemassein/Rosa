from manim import *
import numpy as np

class KMeansSpatialQuantization(Scene):
    def construct(self):
        # 1. Setup Axes and Country Outline (Australia-like shape)
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6,
            axis_config={"include_tip": False}
        ).fade(0.7)
        
        country_points = [
            [-3, 1.5, 0], [-1, 1.8, 0], [1.2, 1.4, 0], [3.2, 0.5, 0], 
            [2.8, -1.5, 0], [1, -2.2, 0], [-1.5, -2, 0], [-3.2, -1, 0]
        ]
        country = Polygon(*country_points, color=BLUE, fill_opacity=0.1)
        self.add(axes)
        self.play(Create(country))

        # 2. Populate with data points
        n_points = 80
        points_list = []
        for _ in range(n_points):
            # Sample points roughly inside the country bounds
            p = [np.random.uniform(-2.8, 2.8), np.random.uniform(-1.8, 1.5), 0]
            dot = Dot(p, radius=0.06, color=GRAY_B)
            points_list.append(dot)
        
        data_points = VGroup(*points_list)
        self.play(FadeIn(data_points, lag_ratio=0.02), run_time=1.5)

        # 3. Create Centroid Crosses manually (since 'Cross' was undefined)
        def create_custom_cross(color):
            l1 = Line(0.15 * UL, 0.15 * DR, color=color, stroke_width=5)
            l2 = Line(0.15 * UR, 0.15 * DL, color=color, stroke_width=5)
            return VGroup(l1, l2)

        num_clusters = 4
        cluster_colors = [RED, GREEN, YELLOW, MAGENTA]
        centroid_positions = [
            [1.8, 1.2, 0], [-1.8, 1.2, 0], [1.8, -1.2, 0], [-1.8, -1.2, 0]
        ]
        
        centroids = VGroup()
        for i in range(num_clusters):
            cross = create_custom_cross(cluster_colors[i]).move_to(centroid_positions[i])
            centroids.add(cross)
        
        self.play(Create(centroids))
        
        # 4. Show the K-means clustering formula
        formula = MathTex(
            r"\mu_{j}:=\frac{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}x^{(i)}}{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}}",
            font_size=34
        ).to_edge(UP, buff=0.3).add_background_rectangle()
        self.play(Write(formula))

        # 5. Assignment phase: Change point colors based on nearest centroid
        assignments = []
        color_anims = []
        for dot in data_points:
            dists = [np.linalg.norm(dot.get_center() - c.get_center()) for c in centroids]
            nearest_idx = np.argmin(dists)
            assignments.append(nearest_idx)
            color_anims.append(dot.animate.set_color(cluster_colors[nearest_idx]))
            
        self.play(*color_anims, run_time=1.5)
        self.wait(0.5)

        # 6. Update phase: Move centroids to the mean of assigned points
        move_anims = []
        for i in range(num_clusters):
            assigned_points = [
                data_points[j].get_center() 
                for j, cluster_idx in enumerate(assignments) 
                if cluster_idx == i
            ]
            if assigned_points:
                new_mean = np.mean(assigned_points, axis=0)
                move_anims.append(centroids[i].animate.move_to(new_mean))
        
        self.play(*move_anims, run_time=2)

        # 7. Highlight final cluster centers
        highlights = VGroup(*[
            Circle(radius=0.35, color=c[0].get_color()).move_to(c.get_center())
            for c in centroids
        ])
        self.play(Create(highlights))
        self.play(highlights.animate.scale(1.3).set_opacity(0), run_time=1)
        
        self.wait(2)
