from manim import *
import numpy as np

class ConceptZeroShotClassification(Scene):
    def construct(self):
        # 1. Narration and Formula
        formula = MathTex(
            r"h(\mathbf{v};\mathbf{X})=\frac{\exp(\mathbf{X}\mathbf{v})}{\sum_{j=1}^{N}\exp(\mathbf{X}_{j}^{T}\mathbf{v})}"
        ).scale(0.8).to_edge(UP, buff=0.5)
        
        # 2. Representation of Image Vector 'v'
        # Using Rectangle instead of Vector to avoid potential identifier issues
        v_rect = Rectangle(width=0.7, height=3.5, color=BLUE, fill_opacity=0.6)
        v_label = MathTex(r"\mathbf{v}").next_to(v_rect, DOWN)
        v_group = VGroup(v_rect, v_label).to_edge(LEFT, buff=1.5)
        
        # 3. Representation of Text Matrix 'X' (Country Embeddings)
        x_rect = Rectangle(width=4.0, height=3.5, color=WHITE)
        # Create individual rows
        num_rows = 11
        row_y_values = [1.5, 1.2, 0.9, 0.6, 0.3, 0, -0.3, -0.6, -0.9, -1.2, -1.5]
        rows = VGroup(*[
            Line(x_rect.get_left() + 0.1 * RIGHT, x_rect.get_right() - 0.1 * RIGHT).shift(UP * y)
            for y in row_y_values
        ]).set_stroke(width=1, opacity=0.4)
        
        # Adding some country labels inside the matrix rows
        country_labels = VGroup(
            Text("France", font_size=12).move_to(rows[2]),
            Text("Japan", font_size=12).move_to(rows[5]),
            Text("Brazil", font_size=12).move_to(rows[8])
        )
        
        x_label = MathTex(r"\mathbf{X}").next_to(x_rect, DOWN)
        x_group = VGroup(x_rect, rows, country_labels, x_label).next_to(v_group, RIGHT, buff=1.2)
        
        # Animation: Intro
        self.play(Write(formula))
        self.play(Create(v_group), Create(x_group))
        self.wait(1)

        # 4. Pulsing highlight for Dot Product Operation
        highlight_row = Rectangle(width=4.0, height=0.3, color=YELLOW, fill_opacity=0.5, stroke_width=0)
        dot_expr = MathTex(r"\mathbf{X}_i^T \mathbf{v}", color=YELLOW).scale(0.8)
        
        for i in [2, 5, 8]:
            highlight_row.move_to(rows[i])
            dot_expr.next_to(highlight_row, RIGHT, buff=0.2)
            self.play(
                FadeIn(highlight_row),
                Write(dot_expr),
                v_rect.animate.set_stroke(YELLOW, width=6),
                run_time=0.5
            )
            self.play(
                FadeOut(highlight_row),
                FadeOut(dot_expr),
                v_rect.animate.set_stroke(BLUE, width=2),
                run_time=0.3
            )

        # 5. Transform into Dot Product expression then to results
        dot_product_result = MathTex(r"\mathbf{X}\mathbf{v}").move_to(x_group.get_center())
        self.play(
            ReplacementTransform(x_group, dot_product_result),
            FadeOut(v_group),
            run_time=1
        )
        self.wait(0.5)
        self.play(Indicate(formula))
        self.play(FadeOut(dot_product_result))

        # 6. Bar Chart for Softmax (Probability Distribution)
        country_names = ["USA", "Japan", "France", "Brazil", "India"]
        bars = VGroup(*[
            Rectangle(width=0.8, height=0.1, color=WHITE, fill_opacity=0.7)
            for _ in country_names
        ]).arrange(RIGHT, buff=0.4, aligned_edge=DOWN).shift(DOWN * 1)
        
        labels = VGroup(*[
            Text(name, font_size=20).next_to(bar, DOWN)
            for name, bar in zip(country_names, bars)
        ])
        
        self.play(Create(bars), Write(labels))
        
        # Animate Softmax Result: France (index 2) shoots up
        target_heights = [0.4, 0.7, 4.2, 0.5, 0.6]
        
        bar_animations = []
        for i, (bar, h) in enumerate(zip(bars, target_heights)):
            if i == 2: # France
                bar_animations.append(bar.animate.stretch_to_fit_height(h, about_edge=DOWN).set_color(GREEN))
            else:
                bar_animations.append(bar.animate.stretch_to_fit_height(h, about_edge=DOWN))
        
        # Replaced sine_out with smooth to fix the undefined identifier error
        self.play(*bar_animations, run_time=2.5, rate_func=smooth)
        
        # 7. Final Prediction Highlighting
        prediction_text = Text("First-stage Prediction: France", font_size=30, color=GREEN).next_to(bars[2], UP, buff=0.5)
        self.play(Write(prediction_text))
        self.play(Indicate(bars[2]), Flash(bars[2], color=GREEN))
        
        self.wait(2)
