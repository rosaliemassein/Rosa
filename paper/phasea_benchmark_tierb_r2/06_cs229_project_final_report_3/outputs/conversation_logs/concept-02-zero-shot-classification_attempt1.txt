from manim import *

class ZeroShotClassification(Scene):
    def construct(self):
        # 1. Image Placeholder
        # Using a square/rectangle as a safe alternative to external image files
        image_rect = Rectangle(height=3, width=4, color=BLUE_E, fill_opacity=0.2)
        image_label = Text("Input Image", font_size=24).move_to(image_rect.get_center())
        image_group = VGroup(image_rect, image_label).to_corner(UL, buff=0.5)

        # 2. Vector v (Image Embedding)
        v_vector = MathTex(
            r"\mathbf{v} = \begin{bmatrix} 0.12 \\ 0.85 \\ \vdots \\ 0.43 \end{bmatrix}",
            font_size=36
        ).next_to(image_group, DOWN, buff=0.5)

        # 3. Matrix X (Text Embeddings for Countries)
        matrix_title = Text("Country Embeddings (X)", font_size=24).shift(RIGHT * 3 + UP * 3)
        countries = ["USA", "Canada", "France", "UK", "Japan", "Brazil"]
        
        # Create visual rows for the matrix
        rows = VGroup()
        for i, name in enumerate(countries):
            row_text = Text(name, font_size=20)
            row_vec = MathTex(r"[ \dots ]", font_size=24)
            row_group = VGroup(row_text, row_vec).arrange(RIGHT, buff=0.5)
            rows.add(row_group)
        
        rows.arrange(DOWN, buff=0.3).next_to(matrix_title, DOWN, buff=0.4)
        matrix_box = SurroundingRectangle(rows, color=WHITE, buff=0.3)
        x_group = VGroup(matrix_title, rows, matrix_box)

        # 4. Softmax Formula
        formula = MathTex(
            r"h(\mathbf{v};\mathbf{X})=\frac{\exp(\mathbf{X}\mathbf{v})}{\sum_{j=1}^{N}\exp(\mathbf{X}_{j}^{T}\mathbf{v})}",
            font_size=32
        ).to_edge(DOWN, buff=0.4)

        # 5. Probability Bar Chart (Built using Primitives for safety)
        axes = Axes(
            x_range=[0, 6, 1],
            y_range=[0, 1, 0.5],
            x_length=5,
            y_length=2.5,
            axis_config={"include_tip": False, "font_size": 20},
            tips=False
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)
        
        # Adding labels to X-axis
        x_labels = VGroup()
        for i, name in enumerate(countries):
            lab = Text(name, font_size=14).next_to(axes.c2p(i + 0.5, 0), DOWN)
            x_labels.add(lab)

        # Initial zero bars
        bars = VGroup(*[
            Rectangle(
                width=0.4, 
                height=0.01, 
                fill_opacity=0.8, 
                color=BLUE, 
                stroke_width=1
            ).move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN)
            for i in range(len(countries))
        ])

        # --- ANIMATION SEQUENCE ---

        # Phase 1: Image to Vector
        self.play(Create(image_group))
        self.wait(0.5)
        self.play(Write(v_vector))
        self.wait(1)

        # Phase 2: Matrix X
        self.play(Create(x_group))
        self.wait(1)

        # Phase 3: Dot Product Pulse
        # We highlight the row for France (index 2)
        dot_highlight = SurroundingRectangle(rows[2], color=YELLOW, buff=0.1)
        self.play(Create(dot_highlight))
        self.play(Indicate(rows[2], scale_factor=1.2, color=YELLOW))
        self.play(Indicate(v_vector, color=YELLOW))
        self.wait(1)

        # Phase 4: Softmax and Probabilities
        self.play(Write(formula))
        self.play(Create(axes), Create(x_labels))
        self.play(Create(bars))
        self.wait(0.5)

        # Update bars to reflect softmax output
        # Probabilities: USA: 0.05, Canada: 0.1, France: 0.7, UK: 0.05, Japan: 0.05, Brazil: 0.05
        prob_values = [0.1, 0.15, 0.75, 0.05, 0.1, 0.05]
        
        new_bars = VGroup()
        for i, val in enumerate(prob_values):
            new_height = val * axes.y_length # scale value by axis length
            nb = Rectangle(
                width=0.4, 
                height=new_height, 
                fill_opacity=0.8, 
                color=GREEN if i == 2 else BLUE, 
                stroke_width=1
            ).move_to(axes.c2p(i + 0.5, 0), aligned_edge=DOWN)
            new_bars.add(nb)

        self.play(
            Transform(bars, new_bars),
            FadeOut(dot_highlight),
            run_time=2
        )
        
        # Highlight prediction
        prediction_text = Text("Prediction: France", color=GREEN, font_size=30).next_to(axes, UP)
        self.play(Write(prediction_text))
        self.play(Indicate(bars[2], color=GOLD))
        
        self.wait(3)
