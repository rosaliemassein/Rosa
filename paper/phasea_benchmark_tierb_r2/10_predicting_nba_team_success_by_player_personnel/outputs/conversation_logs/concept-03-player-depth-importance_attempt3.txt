from manim import *

class ConceptScene(Scene):
    def construct(self):
        # 1. Setup Title and Description
        title = Text("Player Depth Importance", font_size=36).to_edge(UP, buff=0.5)
        description = Text(
            "Relationship between player rank and team success coefficients.", 
            font_size=22, color=GRAY
        ).next_to(title, DOWN, buff=0.2)
        
        self.add(title, description)

        # 2. Create Player Icons (10 basketball player simplified icons)
        players = VGroup()
        for i in range(10):
            head = Dot(radius=0.1)
            body = Line(ORIGIN, DOWN * 0.25).next_to(head, DOWN, buff=0.02)
            player = VGroup(head, body).set_color(WHITE)
            players.add(player)
        
        players.arrange(RIGHT, buff=0.5).shift(DOWN * 1.5)
        
        labels = VGroup(*[
            Text(str(i+1), font_size=20).next_to(players[i], DOWN, buff=0.2)
            for i in range(10)
        ])

        # 3. Create Vertical Bars (Heights representing 'Average Coefficient')
        # Using values that emphasize 1st, 2nd, 5th, and 8th positions
        height_values = [2.8, 2.2, 1.2, 1.1, 1.9, 1.0, 0.8, 1.5, 0.6, 0.5]
        bars = VGroup()
        for i in range(10):
            bar = Rectangle(
                width=0.4, 
                height=height_values[i], 
                fill_opacity=0.8, 
                stroke_width=1,
                color=WHITE
            )
            bar.next_to(players[i], UP, buff=0.1)
            bars.add(bar)

        # 4. Reference Formula
        formula = MathTex(
            r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})",
            font_size=30
        ).to_edge(DOWN, buff=0.5)

        # 5. Animation Sequence
        self.play(Create(players), Write(labels))
        self.wait(0.5)

        # Animate bars growing to their heights
        self.play(
            LaggedStart(*[
                Create(bar) for bar in bars
            ], lag_ratio=0.1),
            run_time=2
        )
        self.wait(0.5)

        # Highlight 1st and 2nd bars with Gold Glow (The Stars)
        star_highlights = VGroup(bars[0], bars[1])
        gold_glow1 = SurroundingRectangle(bars[0], color=GOLD, buff=0.08).set_stroke(width=6)
        gold_glow2 = SurroundingRectangle(bars[1], color=GOLD, buff=0.08).set_stroke(width=6)
        
        self.play(
            star_highlights.animate.set_color(GOLD),
            Create(gold_glow1),
            Create(gold_glow2),
            run_time=1
        )
        self.wait(0.5)

        # Highlight 5th and 8th bars with Pulsing Blue light
        depth_highlights = VGroup(bars[4], bars[7])
        blue_glow1 = SurroundingRectangle(bars[4], color=BLUE, buff=0.08).set_stroke(width=6)
        blue_glow2 = SurroundingRectangle(bars[7], color=BLUE, buff=0.08).set_stroke(width=6)
        
        self.play(
            depth_highlights.animate.set_color(BLUE),
            Create(blue_glow1),
            Create(blue_glow2),
            run_time=1
        )

        # Pulse the blue highlights to emphasize finding
        self.play(
            blue_glow1.animate.scale(1.2).set_opacity(0.3),
            blue_glow2.animate.scale(1.2).set_opacity(0.3),
            rate_func=there_and_back,
            run_time=0.8,
            iteration_count=2
        )

        # Show the formula
        self.play(Write(formula))
        self.wait(3)
