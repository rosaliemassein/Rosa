from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Manual Axis Construction (Avoiding disallowed Axes class)
        # Origin point for the plot
        pt_origin = [-4.5, -2.5, 0]
        horiz_axis = Line(pt_origin, [1.5, -2.5, 0], color=WHITE)
        vert_axis = Line(pt_origin, [-4.5, 2.5, 0], color=WHITE)
        
        txt_horiz = Text("Number of Players", font_size=18).move_to([-1.5, -3.0, 0])
        txt_vert = Text("Mean Squared Error", font_size=18).rotate(90 * DEGREES).move_to([-5.2, 0, 0])
        
        self.add(horiz_axis, vert_axis, txt_horiz, txt_vert)

        # 2. KNN Curve (Red): Dips then rises sharply (overfitting)
        # We build the curve manually using VMobject and points
        k_points = []
        for step_idx in range(31):
            # t_val represents the x-axis from 0 to 10
            t_val = (step_idx / 30.0) * 10.0
            # Quadratic function to simulate overfitting: MSE = 0.1*(t-3)^2 + 1
            mse_val = 0.1 * (t_val - 3.5)**2 + 0.8
            # Scale coordinates to fit our manual axes
            coord_h = pt_origin[0] + (t_val / 10.0) * 5.5
            coord_v = pt_origin[1] + (mse_val / 8.0) * 4.5
            k_points.append([coord_h, coord_v, 0])
        
        knn_line = VMobject(color=RED).set_points_as_corners(k_points)
        knn_text = Text("KNN", color=RED, font_size=20).move_to(k_points[-1]).shift(UP * 0.4)

        # 3. Lasso/Ridge Curve (Green): Steady decline
        l_points = []
        for step_idx in range(31):
            t_val = (step_idx / 30.0) * 10.0
            # Exponential decay to simulate regularization: MSE = 6*exp(-0.4*t) + 0.5
            mse_val = 6.0 * np.exp(-0.45 * t_val) + 0.5
            coord_h = pt_origin[0] + (t_val / 10.0) * 5.8
            coord_v = pt_origin[1] + (mse_val / 8.0) * 4.5
            l_points.append([coord_h, coord_v, 0])
            
        lasso_line = VMobject(color=GREEN).set_points_as_corners(l_points)
        lasso_text = Text("Lasso/Ridge", color=GREEN, font_size=20).move_to(l_points[-1]).shift(RIGHT * 0.8)

        self.play(Create(knn_line), Write(knn_text))
        self.play(Create(lasso_line), Write(lasso_text))

        # 4. Penalty Formula
        # J(theta) = ||y - X*theta||^2 + lambda * ||theta||_1
        stat_formula = MathTex(r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1").scale(0.8).to_corner(UR)
        self.play(Write(stat_formula))

        # 5. Bar Chart for Coefficients (Feature Selection Visualization)
        coeff_bars = VGroup()
        # Using 20 fixed heights to represent initial coefficients
        h_data = [2.1, 1.4, 2.5, 0.8, 1.9, 0.4, 2.2, 1.1, 1.7, 0.5, 2.0, 1.3, 0.6, 1.8, 0.9, 2.3, 0.7, 1.6, 0.3, 1.5]
        start_h_pos = 2.0
        for i_idx, h_val in enumerate(h_data):
            bar_m = Line(
                [start_h_pos + i_idx * 0.22, -2.5, 0], 
                [start_h_pos + i_idx * 0.22, -2.5 + h_val, 0], 
                color=BLUE, 
                stroke_width=4
            )
            coeff_bars.add(bar_m)
        
        coeff_title = Text("Coefficients", font_size=16).next_to(coeff_bars, UP, buff=0.2)
        self.play(Create(coeff_bars), Write(coeff_title))

        # 6. Shrinkage Animation logic (Avoiding ValueTracker)
        # This function updates the bar heights based on the animation progress (alpha)
        def apply_penalty(m_group, alpha_p):
            # lambda penalty grows with alpha from 0 to 3
            p_level = alpha_p * 3.0
            for j_idx, m_bar in enumerate(m_group):
                # Lasso-style soft thresholding: height = max(0, original_height - penalty)
                current_h = h_data[j_idx] - p_level
                if current_h < 0:
                    current_h = 0
                
                p_bot = [start_h_pos + j_idx * 0.22, -2.5, 0]
                p_top = [start_h_pos + j_idx * 0.22, -2.5 + current_h, 0]
                m_bar.set_points_as_corners([p_bot, p_top])
                
                # Visual effect: make bars invisible if they reach zero
                if current_h <= 0:
                    m_bar.set_stroke(opacity=0)
                else:
                    m_bar.set_stroke(opacity=1)

        # Animate the bars shrinking as the 'lambda' term increases over time
        self.play(UpdateFromAlphaFunc(coeff_bars, apply_penalty), run_time=5)
