from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Coordinate System
        # Using NumberPlane as a fallback for Axes if it's restricted
        plane = NumberPlane(
            x_range=[0, 10, 2],
            y_range=[0, 100, 20],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True},
            background_line_style={"stroke_opacity": 0.2}
        ).to_edge(LEFT, buff=0.5).shift(UP * 0.5)

        x_label = Tex("Number of Players", font_size=24).next_to(plane.x_axis, DOWN)
        y_label = Tex("MSE", font_size=24).rotate(90 * DEGREES).next_to(plane.y_axis, LEFT)

        # 2. Define Plot Functions
        def knn_func(t):
            if t < 3:
                return 10 * (3 - t) + 25
            else:
                return 2 * (t - 3)**2 + 25

        def reg_func(t):
            return 70 * np.exp(-0.3 * t) + 15

        knn_curve = plane.plot(knn_func, x_range=[1, 9], color=RED)
        reg_curve = plane.plot(reg_func, x_range=[1, 10], color=GREEN)

        knn_tag = Text("KNN", color=RED, font_size=20).next_to(knn_curve.get_end(), UR, buff=0.1)
        reg_tag = Text("Lasso/Ridge", color=GREEN, font_size=20).next_to(reg_curve.get_end(), RIGHT, buff=0.1)

        # 3. Formula
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            font_size=32
        ).to_edge(UP, buff=0.5).shift(RIGHT * 3)

        # 4. Manual Bar Chart for Coefficients
        # Using Rectangles to avoid BarChart dependency issues
        num_bars = 20
        bar_width = 0.15
        bar_spacing = 0.05
        
        bars = VGroup()
        np.random.seed(42)
        initial_heights = np.random.uniform(0.5, 2.0, size=num_bars)
        
        for i in range(num_bars):
            bar = Rectangle(
                width=bar_width, 
                height=initial_heights[i],
                fill_opacity=0.8,
                fill_color=BLUE,
                stroke_width=1
            )
            bars.add(bar)
        
        bars.arrange(RIGHT, buff=bar_spacing, aligned_edge=DOWN)
        bars.to_edge(DOWN, buff=1).shift(RIGHT * 3)
        
        baseline = Line(bars.get_left(), bars.get_right()).next_to(bars, DOWN, buff=0)
        coef_label = Text("Coefficients", font_size=20).next_to(bars, UP)
        lambda_label = MathTex(r"\lambda \uparrow", font_size=30).next_to(bars, LEFT)

        # 5. Animation Execution
        self.play(Create(plane), Write(x_label), Write(y_label))
        self.wait(0.5)
        
        self.play(Create(knn_curve), Write(knn_tag), run_time=2)
        self.play(Create(reg_curve), Write(reg_tag), run_time=2)
        self.wait(1)
        
        self.play(Write(formula))
        self.play(Create(baseline), Create(bars), Write(coef_label))
        self.wait(1)
        
        # Shrinking coefficients (Lasso effect)
        shrink_animations = []
        for i, bar in enumerate(bars):
            # Most bars go to zero height, some stay small
            target_height = 0.01 if (i % 3 != 0) else bar.height * 0.1
            shrink_animations.append(
                bar.animate.stretch_to_fit_height(target_height, about_edge=DOWN)
            )
        
        self.play(Write(lambda_label))
        self.play(*shrink_animations, run_time=3, rate_func=smooth)
        self.wait(2)
