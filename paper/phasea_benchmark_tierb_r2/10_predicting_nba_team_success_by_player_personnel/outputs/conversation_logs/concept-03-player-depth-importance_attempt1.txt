from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # Data and Strings from JSON
        voice_text = "The models reveal a hidden hierarchy in team success. While the top player's Impact Estimate is the strongest predictor, the 'star' alone isn't enough. Interestingly, the statistics of the fifth and eighth players carry surprising weight. This suggests that a team's floor is set by the quality of its fifth starter and its final rotation player, proving that depth is just as measurable as stardom."
        formula_str = r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})"
        
        # Coefficients representing the "Average Coefficient" 
        # (Relative heights based on the idea that 1-2 are high, 5 and 8 are local peaks)
        coefficients = [0.9, 0.75, 0.35, 0.3, 0.55, 0.25, 0.2, 0.45, 0.15, 0.1]
        
        # 1. Create Formula
        formula = MathTex(formula_str, font_size=36).to_edge(UP, buff=0.5)
        
        # 2. Create Player Icons (Circles with numbers)
        player_icons = VGroup(*[
            VGroup(
                Circle(radius=0.3, color=WHITE, fill_opacity=0.2),
                Text(str(i+1), font_size=20)
            ) for i in range(10)
        ]).arrange(RIGHT, buff=0.5).shift(DOWN * 2)
        
        labels = VGroup(*[
            Text(f"{i+1}th" if i+1 not in [1,2,3] else ["1st","2nd","3rd"][i], font_size=16)
            .next_to(player_icons[i], DOWN, buff=0.2)
            for i in range(10)
        ])

        # 3. Create Bars
        bars = VGroup()
        for i, coeff in enumerate(coefficients):
            bar = Rectangle(
                width=0.4,
                height=coeff * 4,
                color=WHITE,
                fill_opacity=0.7
            )
            bar.next_to(player_icons[i], UP, buff=0.1)
            # Align the bottom of the bar with the top of the icon area
            bar.align_to(player_icons[i].get_top() + UP*0.1, DOWN)
            bars.add(bar)

        # 4. Narration Text (Subtitles)
        narration = Text(
            "Depth (5th & 8th) is as measurable as stardom", 
            font_size=24, 
            color=WHITE
        ).next_to(formula, DOWN, buff=0.3)

        # ANIMATION SEQUENCE
        
        # Display Formula and Players
        self.play(Write(formula))
        self.play(
            Create(player_icons),
            Write(labels),
            run_time=1.5
        )
        self.wait(0.5)

        # Bars growing to respective heights
        self.play(
            LaggedStart(
                *[GrowFromEdge(bar, DOWN) for bar in bars],
                lag_ratio=0.1,
                run_time=2
            )
        )
        self.wait(1)

        # Highlight 1st and 2nd with Gold Glow
        stars_highlight = VGroup(bars[0], bars[1])
        glow_box = SurroundingRectangle(stars_highlight, color=GOLD, buff=0.1)
        
        self.play(
            stars_highlight.animate.set_color(GOLD).set_fill(GOLD, opacity=0.9),
            Create(glow_box),
            run_time=1
        )
        self.wait(1)

        # Highlight 5th and 8th with Pulsing Blue Light
        # Indices 4 and 7 (0-indexed)
        depth_bars = VGroup(bars[4], bars[7])
        
        self.play(Write(narration))
        self.play(
            depth_bars.animate.set_color(BLUE).set_fill(BLUE, opacity=0.9),
            glow_box.animate.set_stroke(opacity=0.2)
        )
        
        # Pulsing Animation via Indication
        self.play(
            Indicate(bars[4], color=BLUE, scale_factor=1.2),
            Indicate(bars[7], color=BLUE, scale_factor=1.2),
            Flash(bars[4], color=BLUE, line_length=0.3),
            Flash(bars[7], color=BLUE, line_length=0.3),
            run_time=2
        )
        
        self.wait(1)
        
        # Show specific voice text part
        full_voice = Text("Stardom is not enough; depth defines the floor.", font_size=20).to_edge(DOWN)
        self.play(FadeIn(full_voice))
        self.wait(3)

        # Fade out everything
        self.play(*[FadeOut(mob) for mob in self.mobjects])
