from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Manually create Axes using Line objects (avoiding 'Axes')
        # Plot 1: MSE curves (Left side)
        mse_origin = LEFT * 5 + DOWN * 1.5
        mse_x_axis = Line(mse_origin, mse_origin + RIGHT * 4)
        mse_y_axis = Line(mse_origin, mse_origin + UP * 3.5)
        
        mse_x_label = Text("Number of Players", font_size=14).next_to(mse_x_axis, DOWN)
        mse_y_label = Text("Mean Squared Error", font_size=14).next_to(mse_y_axis, LEFT).rotate(90 * DEGREES)
        
        # 2. Curves for Plot 1 (Constructed from small Line segments as VMobject might be restricted)
        # Red curve (KNN) - Overfits: dips then rises sharply
        knn_group = VGroup()
        n_segments = 30
        for i in range(n_segments):
            t1 = i / n_segments
            t2 = (i + 1) / n_segments
            # Quadratic shape: y = 8 * (t-0.3)^2 + 0.4
            y1 = 8 * (t1 - 0.3)**2 + 0.4
            y2 = 8 * (t2 - 0.3)**2 + 0.4
            knn_group.add(Line(mse_origin + t1*3.5*RIGHT + y1*UP, mse_origin + t2*3.5*RIGHT + y2*UP, color=RED))
        
        # Green curve (Lasso/Ridge) - Steady decline due to regularization
        reg_group = VGroup()
        for i in range(n_segments):
            t1 = i / n_segments
            t2 = (i + 1) / n_segments
            # Exponential decay: y = 3 * exp(-2.5*t) + 0.2
            y1 = 3 * np.exp(-2.5 * t1) + 0.2
            y2 = 3 * np.exp(-2.5 * t2) + 0.2
            reg_group.add(Line(mse_origin + t1*3.5*RIGHT + y1*UP, mse_origin + t2*3.5*RIGHT + y2*UP, color=GREEN))
        
        knn_label = Text("KNN", color=RED, font_size=14).move_to(mse_origin + 3.5*RIGHT + 3*UP)
        reg_label = Text("Lasso/Ridge", color=GREEN, font_size=14).move_to(mse_origin + 3.5*RIGHT + 0.5*UP)

        self.add(mse_x_axis, mse_y_axis, mse_x_label, mse_y_label)
        self.play(Create(knn_group), Write(knn_label))
        self.play(Create(reg_group), Write(reg_label))
        
        # 3. Regularization Formula
        formula = MathTex(r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1")
        formula.scale(0.8).to_edge(UP, buff=0.5)
        self.play(Write(formula))
        
        # 4. Feature Selection / Coefficient Shrinkage (Right side)
        bar_origin = RIGHT * 1 + DOWN * 1.5
        bar_x_axis = Line(bar_origin, bar_origin + RIGHT * 5)
        bar_y_axis = Line(bar_origin, bar_origin + UP * 3)
        bar_title = Text("Feature Selection (Lasso)", font_size=14).next_to(bar_y_axis, UP)
        
        # Tracker for lambda (using a hidden Dot's position instead of ValueTracker)
        lambda_ptr = Dot(radius=0).move_to(ORIGIN)
        
        num_features = 40
        # Data: list of tuples (initial_height, decay_rate)
        np.random.seed(123)
        coeff_data = [(np.random.uniform(0.2, 2.8), np.random.uniform(0.5, 5.0)) for _ in range(num_features)]
        
        bars = VGroup()
        for i in range(num_features):
            h = coeff_data[i][0]
            bar_line = Line(
                bar_origin + (i * 0.12 + 0.1) * RIGHT,
                bar_origin + (i * 0.12 + 0.1) * RIGHT + h * UP,
                color=BLUE,
                stroke_width=2
            )
            bars.add(bar_line)
            
        # Manually updating the height of lines to simulate shrinkage
        def update_bars(m):
            # Get current lambda value from the tracker dot's x-coordinate
            l_val = lambda_ptr.get_center()[0]
            for idx, b in enumerate(m):
                init_h, decay = coeff_data[idx]
                # Lasso-style shrinkage: height = max(0, initial - lambda * decay)
                current_h = init_h - (l_val * decay)
                if current_h < 0: 
                    current_h = 0
                
                # Update line geometry
                start_point = bar_origin + (idx * 0.12 + 0.1) * RIGHT
                end_point = start_point + current_h * UP
                b.put_start_and_end_on(start_point, end_point)
                
                # Fade out if coefficient hits zero
                if current_h <= 0:
                    b.set_stroke(opacity=0)
                else:
                    b.set_stroke(opacity=1)
        
        bars.add_updater(update_bars)
        self.add(bar_x_axis, bar_y_axis, bar_title, bars)
        
        # 5. Final Animation: Increasing Lambda
        lambda_symbol = MathTex(r"\lambda").next_to(formula, DOWN, buff=0.2).set_color(YELLOW)
        self.play(
            lambda_ptr.animate.shift(RIGHT * 1.5),
            Write(lambda_symbol),
            run_time=6,
            rate_func=lambda t: t # Equivalent to linear
        )
        self.wait(2)
