from manim import *
import numpy as np

class CosineSimilarityConcept(Scene):
    def construct(self):
        # Tracker replacement using a hidden Mobject's coordinate
        # This avoids using the disallowed 'ValueTracker' class.
        tkr_dot = Dot(point=[PI/3, 0, 0]).set_opacity(0)

        # Manual Coordinate Lines as 'Axes' is disallowed
        x_ln = Line(LEFT * 2.5, RIGHT * 2.5, color=GREY).shift(LEFT * 4)
        y_ln = Line(DOWN * 2.5, UP * 2.5, color=GREY).shift(LEFT * 4)
        org = x_ln.get_center()
        
        # Static Vector A
        v_a = Arrow(org, org + RIGHT * 2, buff=0, color=YELLOW)
        l_a = MathTex(r"\mathbf{A}", color=YELLOW).next_to(v_a.get_end(), DOWN)

        # Dynamic Vector B
        v_b = Arrow(org, org + RIGHT * 2, buff=0, color=RED)
        def up_v_b(m):
            ang = tkr_dot.get_x()
            target = org + np.array([np.cos(ang), np.sin(ang), 0]) * 2
            m.become(Arrow(org, target, buff=0, color=RED))
        v_b.add_updater(up_v_b)

        l_b = MathTex(r"\mathbf{B}", color=RED)
        def up_l_b(m):
            ang = tkr_dot.get_x()
            pos_offset = np.array([np.cos(ang), np.sin(ang), 0]) * 0.3
            m.move_to(v_b.get_end() + pos_offset)
        l_b.add_updater(up_l_b)

        # Angle Arc (Manually updated as 'Angle' is undefined)
        arc_m = Arc(radius=0.6, start_angle=0, angle=PI/3, arc_center=org, color=WHITE)
        def up_arc(m):
            ang = tkr_dot.get_x()
            # Ensure angle is valid for Arc creation
            safe_ang = max(ang, 0.001)
            m.become(Arc(radius=0.6, start_angle=0, angle=safe_ang, arc_center=org, color=WHITE))
        arc_m.add_updater(up_arc)

        # Formula at the top
        fml = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=34
        ).to_edge(UP)

        # Cosine value display at the bottom
        val_txt = MathTex(r"\cos(\theta) = 0.50").to_edge(DOWN, buff=1)
        def up_val(m):
            s = np.cos(tkr_dot.get_x())
            m.become(MathTex(r"\cos(\theta) = {:.2f}".format(s)).move_to(val_txt.get_center()))
        val_txt.add_updater(up_val)

        # Similarity Matrix construction
        sqs = VGroup(*[
            Square(side_length=0.5, fill_opacity=0.1, color=BLUE) for _ in range(25)
        ])
        
        # Arrange squares manually into a grid
        rows = VGroup()
        for i in range(5):
            row = sqs[i*5 : (i+1)*5].arrange(RIGHT, buff=0.1)
            rows.add(row)
        rows.arrange(DOWN, buff=0.1).to_edge(RIGHT, buff=1)

        # Matrix updater to simulate heat map glow
        def up_mat(group):
            sim = np.cos(tkr_dot.get_x())
            # Normalize sim [-1, 1] to a brightness/opacity range [0, 1]
            alpha = (sim + 1) / 2
            # Target the center cell (index 12) for visual feedback
            sqs[12].set_fill(YELLOW, opacity=alpha)
            sqs[12].set_stroke(WHITE, width=alpha * 3)
            # Other squares stay dim
            for i in range(25):
                if i != 12:
                    sqs[i].set_fill(BLUE, opacity=0.1)
        
        sqs.add_updater(up_mat)

        # Assemble scene
        self.add(x_ln, y_ln, v_a, l_a, v_b, l_b, arc_m, fml, val_txt, rows)

        # Animation sequence
        # Perfect match (theta = 0, similarity = 1)
        self.play(tkr_dot.animate.set_x(0), run_time=2)
        self.wait(1)
        
        # Orthogonal (theta = PI/2, similarity = 0)
        self.play(tkr_dot.animate.set_x(PI / 2), run_time=2)
        self.wait(1)
        
        # Opposite (theta = PI, similarity = -1)
        self.play(tkr_dot.animate.set_x(PI), run_time=2)
        self.wait(1)
        
        # High overlap (theta = PI/6, similarity ~ 0.86)
        self.play(tkr_dot.animate.set_x(PI / 6), run_time=2)
        self.wait(2)

        # Cleanup
        self.play(FadeOut(VGroup(x_ln, y_ln, v_a, l_a, v_b, l_b, arc_m, fml, val_txt, rows)))
        self.wait(1)
