from manim import *
import numpy as np

class ConceptIterativeClustering(Scene):
    def construct(self):
        # Configuration
        n_dots = 27
        dots_per_cluster = 9
        # Basic colors to ensure compatibility
        cluster_colors = [YELLOW, RED, GREEN]
        
        # Create a cloud of 27 Dot objects scattered randomly
        np.random.seed(42)
        dot_positions = []
        for _ in range(n_dots):
            pos = [
                np.random.uniform(-5, 5),
                np.random.uniform(-2.5, 1.5),
                0
            ]
            dot_positions.append(pos)
            
        dots = VGroup(*[Dot(pos, color=BLUE) for pos in dot_positions])
        self.add(dots)

        # Narration setup
        voiceover_text = Text(
            "Iterative clustering for thematic cohesion", 
            font_size=24
        ).to_edge(UP)
        self.add(voiceover_text)

        available_indices = list(range(n_dots))
        
        for c_idx in range(3):
            # 1. Find the two closest among available
            best_pair = (0, 0)
            min_dist = float('inf')
            
            for i in range(len(available_indices)):
                for j in range(i + 1, len(available_indices)):
                    idx1 = available_indices[i]
                    idx2 = available_indices[j]
                    d = np.linalg.norm(dots[idx1].get_center() - dots[idx2].get_center())
                    if d < min_dist:
                        min_dist = d
                        best_pair = (idx1, idx2)
            
            cluster_indices = [best_pair[0], best_pair[1]]
            available_indices.remove(best_pair[0])
            available_indices.remove(best_pair[1])
            
            # Highlight seed pair
            seed_color = cluster_colors[c_idx]
            highlight_rings = VGroup(
                Circle(radius=0.2, color=seed_color).move_to(dots[best_pair[0]]),
                Circle(radius=0.2, color=seed_color).move_to(dots[best_pair[1]])
            )
            
            self.play(
                dots[best_pair[0]].animate.set_color(seed_color),
                dots[best_pair[1]].animate.set_color(seed_color),
                Create(highlight_rings),
                run_time=0.6
            )

            # 2. Greedily pull in the next closest until target size 9
            while len(cluster_indices) < dots_per_cluster:
                target_idx = -1
                min_d_to_cluster = float('inf')
                
                # Search for closest unassigned dot to ANY dot in current cluster
                for candidate_idx in available_indices:
                    for member_idx in cluster_indices:
                        d = np.linalg.norm(dots[candidate_idx].get_center() - dots[member_idx].get_center())
                        if d < min_d_to_cluster:
                            min_d_to_cluster = d
                            target_idx = candidate_idx
                
                # Animate search circle expanding from the cluster centroid
                cluster_points = [dots[i].get_center() for i in cluster_indices]
                cluster_center = np.mean(cluster_points, axis=0)
                
                search_circle = Circle(radius=0.01, color=seed_color, stroke_opacity=0.4).move_to(cluster_center)
                self.add(search_circle)
                
                # Expand until it reaches the target
                dist_to_target = np.linalg.norm(dots[target_idx].get_center() - cluster_center)
                
                self.play(
                    search_circle.animate.set_width(dist_to_target * 2 + 0.1),
                    run_time=0.3,
                    rate_func=rate_functions.linear
                )
                
                self.play(
                    dots[target_idx].animate.set_color(seed_color),
                    FadeOut(search_circle),
                    run_time=0.15
                )
                
                cluster_indices.append(target_idx)
                available_indices.remove(target_idx)
            
            # Cleanup highlight for next cluster
            self.play(FadeOut(highlight_rings), run_time=0.4)

        # Final caption
        final_text = Text(
            "Groups of nine with highest possible thematic cohesion.", 
            font_size=24
        ).to_edge(UP)
        self.play(Transform(voiceover_text, final_text))
        self.wait(2)
