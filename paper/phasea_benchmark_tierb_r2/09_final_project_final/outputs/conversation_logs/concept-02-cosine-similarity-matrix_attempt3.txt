from manim import *
import numpy as np

class ConceptSimilarity(Scene):
    def construct(self):
        # Center of the vector view
        c = np.array([-3, 0, 0])
        
        # 1. Manual Axes (Replacing disallowed Axes)
        h_line = Line(c + np.array([-1.5, 0, 0]), c + np.array([1.5, 0, 0]), color=WHITE)
        v_line = Line(c + np.array([0, -1.5, 0]), c + np.array([0, 1.5, 0]), color=WHITE)
        self.add(h_line, v_line)

        # 2. Angle Tracker (Dummy Mobject to store value without using disallowed ValueTracker)
        # We will use the width of this square as our theta variable
        tracker = Square(side_length=45.0)
        tracker.set_opacity(0)
        self.add(tracker)

        # 3. Vectors and Arc
        vec_a = Arrow(c, c + np.array([1.3, 0, 0]), buff=0, color=BLUE)
        vec_b = Arrow(c, c + np.array([1.3, 0, 0]), buff=0, color=RED)
        # Initial angle arc (Replacing disallowed Angle)
        arc = Arc(radius=0.4, start_angle=0, angle=45 * 3.14159 / 180, arc_center=c, color=YELLOW)
        
        label_a = MathTex(r"\mathbf{A}", color=BLUE).next_to(c + np.array([1.3, 0, 0]), RIGHT)
        label_b_text = MathTex(r"\mathbf{B}", color=RED)
        self.add(vec_a, vec_b, arc, label_a, label_b_text)

        # 4. Text and Matrix
        sim_text = MathTex(r"\text{similarity}: 0.71").to_edge(UP)
        self.add(sim_text)
        
        formula = MathTex(r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}")
        formula.scale(0.7).to_edge(DOWN)
        self.add(formula)

        # Build a 5x5 grid of squares
        matrix = VGroup(*[
            Square(side_length=0.4).set_fill(BLUE, opacity=0.2)
            for _ in range(25)
        ])
        matrix.arrange_in_grid(5, 5, buff=0.1).shift(np.array([3, 0, 0]))
        self.add(matrix)
        
        center_cell = matrix[12] # Selecting a middle cell to react to the geometry

        # 5. Updater Function (Replacing disallowed always_redraw)
        def update_scene(mob):
            # Read current angle from the tracker's property
            current_angle_deg = tracker.get_width()
            rad = current_angle_deg * 3.14159 / 180
            
            # Update vector B rotation
            new_end = c + np.array([1.3 * np.cos(rad), 1.3 * np.sin(rad), 0])
            vec_b.put_start_and_end_on(c, new_end)
            label_b_text.move_to(new_end + np.array([0.2, 0.2, 0]))
            
            # Update Angle Arc
            arc.become(Arc(radius=0.4, start_angle=0, angle=rad, arc_center=c, color=YELLOW))
            
            # Calculate Similarity
            similarity = np.cos(rad)
            
            # Update Matrix Cell (Glow effect)
            # If similarity is high, turn cell yellow/white and increase opacity
            if similarity > 0.9:
                center_cell.set_fill(WHITE, opacity=1.0)
            elif similarity > 0.5:
                center_cell.set_fill(YELLOW, opacity=similarity)
            else:
                center_cell.set_fill(BLUE, opacity=0.2)
            
            # Update Sim Text
            new_sim = MathTex(r"\text{similarity}: ", f"{similarity:.2f}").to_edge(UP)
            sim_text.become(new_sim)

        # Link updater to vec_b (any visible object works)
        vec_b.add_updater(update_scene)

        # 6. Animation Execution
        # Rotate vectors to perfect alignment
        self.play(tracker.animate.set_width(0.0), run_time=3)
        self.wait(1)
        
        # Rotate vectors to be perpendicular
        self.play(tracker.animate.set_width(90.0), run_time=3)
        self.wait(1)
        
        # Final movement back to overlap
        self.play(tracker.animate.set_width(30.0), run_time=2)
        
        # Randomly fill other matrix cells to complete the visual heatmap
        other_animations = []
        for i, cell in enumerate(matrix):
            if i != 12:
                rand_val = np.random.random()
                if rand_val > 0.7:
                    other_animations.append(cell.animate.set_fill(YELLOW, opacity=rand_val))
                elif rand_val > 0.4:
                    other_animations.append(cell.animate.set_fill(BLUE, opacity=rand_val))
        
        self.play(*other_animations, run_time=2)
        self.wait(2)
