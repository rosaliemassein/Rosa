from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # Manual Axes Construction (avoiding disallowed Axes)
        origin_pt = LEFT * 4 + DOWN * 1.5
        xaxis = Line(origin_pt, origin_pt + RIGHT * 4, color=WHITE)
        yaxis = Line(origin_pt, origin_pt + UP * 4, color=WHITE)
        self.add(xaxis, yaxis)

        # Vector A (Stationary)
        angle_a = 25 * (PI / 180)
        # Using Arrow instead of Vector (as Vector was undefined in log)
        vec_a = Arrow(origin_pt, origin_pt + [2.8 * np.cos(angle_a), 2.8 * np.sin(angle_a), 0], buff=0, color=BLUE)
        lbl_a = MathTex("A", color=BLUE).next_to(vec_a.get_end(), RIGHT, buff=0.1)
        self.add(vec_a, lbl_a)

        # Tracker (Using a Dot's x-coordinate to simulate ValueTracker which was disallowed)
        # Starting angle for Vector B
        tracker = Dot(point=[80 * (PI / 180), 0, 0], radius=0)
        self.add(tracker)

        # Vector B (Dynamic)
        vec_b = Arrow(origin_pt, origin_pt + RIGHT * 3, buff=0, color=RED)
        def update_b(m):
            current_angle = tracker.get_center()[0]
            m.put_start_and_end_on(origin_pt, origin_pt + [2.8 * np.cos(current_angle), 2.8 * np.sin(current_angle), 0])
        vec_b.add_updater(update_b)

        lbl_b = MathTex("B", color=RED)
        def update_lbl_b(m):
            m.next_to(vec_b.get_end(), UR, buff=0.1)
        lbl_b.add_updater(update_lbl_b)
        self.add(vec_b, lbl_b)

        # Angle Theta (Using Arc instead of Angle as Angle was undefined in log)
        theta_arc = Arc(radius=0.6, arc_center=origin_pt, color=YELLOW)
        def update_arc(m):
            angle_b = tracker.get_center()[0]
            start_angle = min(angle_a, angle_b)
            diff = abs(angle_a - angle_b)
            # Replacing become(always_redraw) logic manually
            m.become(Arc(radius=0.6, start_angle=start_angle, angle=diff, arc_center=origin_pt, color=YELLOW))
        theta_arc.add_updater(update_arc)
        self.add(theta_arc)

        # Similarity Text Display (Using Text/MathTex instead of DecimalNumber)
        formula_txt = MathTex(r"\cos(\theta) = ", font_size=32).to_edge(DOWN).shift(LEFT * 2)
        val_txt = Text("0.00", color=YELLOW, font_size=24).next_to(formula_txt, RIGHT)
        def update_val(m):
            cos_val = np.cos(tracker.get_center()[0] - angle_a)
            m.become(Text(f"{cos_val:.2f}", color=YELLOW, font_size=24).next_to(formula_txt, RIGHT))
        val_txt.add_updater(update_val)
        self.add(formula_txt, val_txt)

        # Similarity Matrix Grid
        matrix = VGroup()
        for _ in range(25):
            sq = Square(side_length=0.6, fill_opacity=0.1, stroke_width=2, color=WHITE)
            matrix.add(sq)
        matrix.arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=1)
        self.add(matrix)

        # Update specific cell color based on alignment
        center_cell = matrix[12]
        def update_cell(m):
            cos_val = np.cos(tracker.get_center()[0] - angle_a)
            # Interpolate manually by changing opacity and color based on value
            intensity = max(0.1, cos_val)
            if cos_val > 0.7:
                m.set_fill(YELLOW, opacity=intensity)
            else:
                m.set_fill(BLUE, opacity=intensity)
        center_cell.add_updater(update_cell)

        # Animation Sequence
        # 1. Align B with A (Similarity increases to 1.0)
        self.play(tracker.animate.move_to([angle_a, 0, 0]), run_time=3)
        self.wait(1)
        
        # 2. Diverge B from A (Similarity decreases)
        self.play(tracker.animate.move_to([angle_a + 85 * (PI / 180), 0, 0]), run_time=3)
        self.wait(1)
        
        # 3. Fill the rest of the matrix with random values
        self.play(*[
            sq.animate.set_fill(BLUE, opacity=np.random.uniform(0.1, 0.5))
            for i, sq in enumerate(matrix) if i != 12
        ])
        
        self.wait(2)

        # Remove updaters for clean exit
        vec_b.clear_updaters()
        lbl_b.clear_updaters()
        theta_arc.clear_updaters()
        val_txt.clear_updaters()
        center_cell.clear_updaters()
