from manim import *
import numpy as np

class StabilityAnimation(Scene):
    def construct(self):
        # Setup coordinates manually
        origin = LEFT * 5 + DOWN * 2
        x_axis = Line(origin, origin + RIGHT * 7, color=WHITE)
        y_axis = Line(origin, origin + UP * 5, color=WHITE)
        
        x_label = Text("Time", font_size=24).next_to(x_axis, DOWN)
        y_label = Text("RPM", font_size=24).next_to(y_axis, LEFT).rotate(90 * DEGREES)
        
        # 1. Fluctuating motor speed around 30 RPM
        # Scale: 1 unit x = 0.6 units scene, 10 units RPM = 1.0 units scene
        # Base RPM is 30, which is 3.0 units up.
        plot_points = []
        for i in range(101):
            t_val = i / 10.0  # 0 to 10
            # fluctuation around 3.0 units (30 RPM)
            y_val = 3.0 + 0.2 * np.sin(4 * t_val) + 0.1 * np.cos(13 * t_val)
            point = origin + (t_val * 0.6 * RIGHT) + (y_val * 1.0 * UP)
            plot_points.append(point)
            
        velocity_plot = VMobject(color=YELLOW)
        velocity_plot.set_points_as_corners(plot_points)
        
        # 2. Highlight the last 20% (from t=8 to t=10)
        highlight_origin = origin + (8 * 0.6 * RIGHT)
        highlight_rect = Rectangle(
            width=2 * 0.6,
            height=5,
            fill_color=BLUE,
            fill_opacity=0.3,
            stroke_width=0
        ).move_to(highlight_origin + 0.6 * RIGHT + 2.5 * UP)
        
        steady_state_text = Text("Final 20%", font_size=20).next_to(highlight_rect, UP)

        # 3. Histogram on the side
        hist_center = RIGHT * 4 + DOWN * 1
        heights = [0.5, 1.0, 3.5, 1.2, 0.4]
        bars = VGroup()
        for i in range(len(heights)):
            h = heights[i]
            bar_color = BLUE
            if i == 2:
                bar_color = RED
            bar = Rectangle(
                width=0.5,
                height=h,
                fill_color=bar_color,
                fill_opacity=0.8,
                stroke_color=WHITE
            )
            bar.move_to(hist_center + i * 0.6 * RIGHT, aligned_edge=DOWN)
            bars.add(bar)
            
        hist_label = Text("Histogram", font_size=24).next_to(bars, UP, buff=0.5)
        mode_label = Text("Mode (30 RPM)", font_size=18, color=RED).next_to(bars[2], UP, buff=0.1)

        # 4. Stability Formula
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=36
        ).to_edge(UP, buff=0.5)

        # 5. Animation Sequence
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(velocity_plot), run_time=2)
        self.wait(1)
        
        self.play(FadeIn(highlight_rect), Write(steady_state_text))
        self.wait(1)
        
        self.play(Create(bars), Write(hist_label))
        self.play(Indicate(bars[2], color=RED))
        self.play(Write(mode_label))
        self.wait(1)
        
        self.play(Write(formula))
        self.play(formula.animate.set_color(YELLOW))
        
        self.wait(3)
