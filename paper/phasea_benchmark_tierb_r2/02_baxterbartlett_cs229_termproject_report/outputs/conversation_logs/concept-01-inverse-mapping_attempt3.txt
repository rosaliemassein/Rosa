from manim import *
import numpy as np

class InverseMappingConcept(Scene):
    def construct(self):
        # 1. Input and Output boxes
        input_rect = Rectangle(height=2, width=4.5, color=BLUE)
        input_text = VGroup(
            Text("Desired Behavior", font_size=28),
            Text("(Rise Time, SS Error)", font_size=20)
        ).arrange(DOWN, buff=0.2)
        input_box = VGroup(input_rect, input_text).shift(3 * LEFT + 2 * UP)

        output_rect = Rectangle(height=2, width=4.5, color=GREEN)
        output_text = VGroup(
            Text("PID Constants", font_size=28),
            MathTex(r"K_p, K_i, K_d", font_size=32)
        ).arrange(DOWN, buff=0.2)
        output_box = VGroup(output_rect, output_text).shift(3 * RIGHT + 2 * UP)

        # Draw a large arrow between the boxes
        arrow = Arrow(input_box.get_right(), output_box.get_left(), buff=0.2)
        
        diagram = VGroup(input_box, output_box, arrow)

        # Formula at the top
        formula = MathTex(
            r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", 
            font_size=36
        ).to_edge(UP, buff=0.2)

        self.add(diagram, formula)
        self.wait(1)

        # 2. "Flips the script" - Rotate the diagram
        # Rotate the group 180 degrees. 
        # To avoid inverted text, we'll rotate the positions of the boxes.
        self.play(
            input_box.animate.move_to(3 * RIGHT + 2 * UP),
            output_box.animate.move_to(3 * LEFT + 2 * UP),
            arrow.animate.rotate(PI),
            run_time=2
        )
        self.wait(1)

        # 3. Motor velocity plot (Manual construction to avoid restricted Axes/Brace)
        origin = 3.5 * LEFT + 2 * DOWN
        x_axis = Line(origin, origin + 7 * RIGHT, color=WHITE)
        y_axis = Line(origin, origin + 3 * UP, color=WHITE)
        x_label = Text("Time", font_size=20).next_to(x_axis, RIGHT)
        y_label = Text("RPM", font_size=20).next_to(y_axis, UP)
        
        # Setpoint line
        setpoint_y = origin + 2 * UP
        setpoint = Line(setpoint_y, setpoint_y + 6 * RIGHT, color=YELLOW, stroke_opacity=0.5)
        
        # Plot curve: 1 - exp(-t)*cos(t)
        # We manually generate points for the curve
        points = []
        for t in np.linspace(0, 6, 100):
            val = 1 - np.exp(-0.7 * t) * np.cos(4 * t)
            # Scale and shift to the plot origin
            coord = origin + t * RIGHT + (val * 2) * UP
            points.append(coord)
            
        curve = VMobject(color=BLUE)
        curve.set_points_as_corners(points)

        # 4. Labels for Rise Time and Steady State Error
        # Rise Time marker (approx at first crossing)
        rise_x_coord = origin + 0.55 * RIGHT
        rise_line = Line(rise_x_coord, rise_x_coord + 0.3 * DOWN, color=RED)
        rise_label = Text("Rise Time", font_size=18, color=RED).next_to(rise_line, DOWN)
        
        # Steady state error (end of curve)
        end_point = points[-1]
        error_line = Line(end_point, [end_point[0], setpoint_y[1], 0], color=ORANGE)
        error_label = Text("SS Error", font_size=18, color=ORANGE).next_to(error_line, RIGHT, buff=0.1)

        # Animate plot
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(setpoint))
        self.play(Create(curve), run_time=3)
        self.play(
            Create(rise_line), Write(rise_label),
            Create(error_line), Write(error_label)
        )

        self.wait(2)
