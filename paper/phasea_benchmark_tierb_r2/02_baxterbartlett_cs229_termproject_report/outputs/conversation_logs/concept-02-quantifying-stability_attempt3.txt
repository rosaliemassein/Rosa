from manim import *
import numpy as np

class QuantifyingStability(Scene):
    def construct(self):
        # 1. Custom Axes for Velocity Plot (Manual construction to avoid restricted Axes class)
        # Time axis from x=-5 to x=-1 (4 units long)
        # RPM axis from y=-1.5 to y=1.5 (3 units high)
        origin = np.array([-5, -1.5, 0])
        x_axis = Line(origin, origin + [4, 0, 0], color=WHITE)
        y_axis = Line(origin, origin + [0, 3, 0], color=WHITE)
        
        x_label = Text("Time", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("RPM", font_size=20).next_to(y_axis, LEFT)
        
        # 2. Velocity Plot Data
        # Mapping: Time (0-10) -> x (-5 to -1), RPM (20-40) -> y (-1.5 to 1.5)
        # 30 RPM is the baseline (centered at y=0)
        def get_velocity_point(t):
            # Fluctuating behavior around 30 RPM
            rpm = 30 + 2.5 * np.sin(t * 1.5) + 1.2 * np.cos(t * 4)
            x_coord = -5 + (t / 10) * 4
            y_coord = -1.5 + ((rpm - 20) / 20) * 3
            return [x_coord, y_coord, 0]

        velocity_curve = VMobject(color=BLUE)
        velocity_curve.set_points_as_corners([get_velocity_point(t) for t in np.linspace(0, 10, 80)])
        
        # 3. Highlight the last 20% of the x-axis
        # x start: -5 + (0.8 * 4) = -1.8. x end: -1.0.
        # Rectangle centered at -1.4, width 0.8
        highlight_box = Rectangle(
            width=0.8, 
            height=3.0, 
            fill_color=BLUE, 
            fill_opacity=0.3, 
            stroke_width=0
        ).move_to([-1.4, 0, 0])
        
        highlight_text = Text("Steady State (Last 20%)", font_size=16).next_to(highlight_box, UP)

        # 4. Histogram Construction (Right Side)
        hist_base = Line([1, -1.5, 0], [5, -1.5, 0], color=WHITE)
        hist_title = Text("Frequency Distribution", font_size=20).move_to([3, 2, 0])
        
        # Histogram bars (mode at index 2)
        heights = [0.5, 1.1, 2.8, 0.9, 0.4]
        bars = VGroup()
        for i, h in enumerate(heights):
            bar = Rectangle(
                width=0.5, 
                height=h, 
                fill_color=GREEN if i == 2 else GRAY, 
                fill_opacity=0.8,
                stroke_width=1
            )
            # Space out bars: 1.5, 2.2, 2.9, 3.6, 4.3
            bar.move_to([1.5 + i * 0.7, -1.5 + h/2, 0])
            bars.add(bar)

        # 5. Stability Formula
        stability_formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=34
        ).to_edge(DOWN, buff=0.8)

        # 6. Animation Playback
        # Part 1: Plotting the velocity
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(velocity_curve), run_time=2.5)
        self.wait(0.5)
        
        # Part 2: Highlighting the steady state
        self.play(FadeIn(highlight_box), Write(highlight_text))
        self.wait(1)
        
        # Part 3: Show Histogram
        self.play(Create(hist_base), Write(hist_title))
        self.play(Create(bars))
        self.wait(0.5)
        
        # Part 4: Define Stability
        self.play(Write(stability_formula))
        
        # Highlight correlation between mode bin and numerator
        mode_rect = SurroundingRectangle(bars[2], color=YELLOW)
        self.play(Create(mode_rect))
        self.play(stability_formula[0][11:23].animate.set_color(YELLOW))
        self.wait(1)
        
        # Highlight correlation between other bins and denominator
        other_bars = VGroup(*[bars[i] for i in [0, 1, 3, 4]])
        other_rect = SurroundingRectangle(other_bars, color=RED)
        self.play(ReplacementTransform(mode_rect, other_rect))
        self.play(stability_formula[0][24:].animate.set_color(RED))
        
        self.wait(2)
