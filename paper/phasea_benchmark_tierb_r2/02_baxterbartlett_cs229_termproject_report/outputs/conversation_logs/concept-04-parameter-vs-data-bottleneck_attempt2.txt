from manim import *
import numpy as np

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Narration Text (Title)
        narration = Text(
            "Why did the simple LWR outperform a powerful Neural Network?",
            font_size=26,
            color=BLUE
        ).to_edge(UP, buff=0.5)
        self.play(Write(narration))
        self.wait(1)

        # 2. Layout for Side-by-Side Plots
        # Left side: Neural Network (Overfitting)
        left_axes = Axes(
            x_range=[0, 10, 2],
            y_range=[0, 10, 2],
            x_length=5,
            y_length=3.5,
            axis_config={"include_tip": False}
        ).shift(LEFT * 3.5 + UP * 0.5)

        # Right side: LWR (Local Linear/Smooth)
        right_axes = Axes(
            x_range=[0, 10, 2],
            y_range=[0, 10, 2],
            x_length=5,
            y_length=3.5,
            axis_config={"include_tip": False}
        ).shift(RIGHT * 3.5 + UP * 0.5)

        nn_title = Text("Neural Network", font_size=24).next_to(left_axes, UP)
        lwr_title = Text("LWR", font_size=24).next_to(right_axes, UP)

        self.play(
            Create(left_axes), Create(right_axes),
            Write(nn_title), Write(lwr_title)
        )

        # 3. Data Points (The same for both)
        np.random.seed(10)
        x_data = np.linspace(1, 9, 10)
        y_data = 0.8 * x_data + 1 + np.random.normal(0, 0.8, 10)
        
        dots_left = VGroup(*[Dot(left_axes.c2p(x, y), color=YELLOW, radius=0.06) for x, y in zip(x_data, y_data)])
        dots_right = VGroup(*[Dot(right_axes.c2p(x, y), color=YELLOW, radius=0.06) for x, y in zip(x_data, y_data)])

        self.play(Create(dots_left), Create(dots_right))

        # 4. Curves
        # NN: Highly wiggling curve hitting data points
        nn_curve = left_axes.plot(
            lambda x: 0.8 * x + 1 + 1.2 * np.sin(3 * x), 
            color=RED, 
            x_range=[1, 9]
        )
        nn_tag = Text("High Variance", font_size=20, color=RED).next_to(left_axes, DOWN, buff=0.2)

        # LWR: Smoother path (piecewise-linear/local approximation)
        lwr_curve = right_axes.plot(
            lambda x: 0.8 * x + 1, 
            color=GREEN, 
            x_range=[1, 9]
        )
        lwr_tag = Text("Robust / Smooth", font_size=20, color=GREEN).next_to(right_axes, DOWN, buff=0.2)

        self.play(Create(nn_curve), Write(nn_tag))
        self.play(Create(lwr_curve), Write(lwr_tag))
        self.wait(1)

        # 5. MSE Bar Chart and Formula
        # Shrink the top section slightly to make room
        top_elements = VGroup(left_axes, right_axes, nn_title, lwr_title, dots_left, dots_right, nn_curve, lwr_curve, nn_tag, lwr_tag)
        self.play(top_elements.animate.scale(0.75).shift(UP * 0.5))

        chart = BarChart(
            values=[1.21, 0.85],
            bar_names=["Neural Network", "LWR"],
            y_range=[0, 1.5, 0.5],
            y_length=2,
            x_length=6,
            bar_colors=[RED, GREEN],
            axis_config={"font_size": 20}
        ).to_edge(DOWN, buff=0.5).shift(LEFT * 1.5)

        mse_label = Text("MSE", font_size=22).next_to(chart, UP, buff=0.2)
        
        formula = MathTex(
            r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}",
            font_size=28
        ).next_to(chart, RIGHT, buff=0.8)

        self.play(Create(chart), Write(mse_label))
        self.play(Write(formula))

        # Value labels on bars
        val_1 = MathTex("1.21", font_size=22).next_to(chart.bars[0], UP, buff=0.1)
        val_2 = MathTex("0.85", font_size=22).next_to(chart.bars[1], UP, buff=0.1)
        self.play(Write(val_1), Write(val_2))

        self.wait(2)
