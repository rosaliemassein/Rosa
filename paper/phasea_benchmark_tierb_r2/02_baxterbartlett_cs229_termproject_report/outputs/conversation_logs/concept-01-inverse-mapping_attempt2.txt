from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Create Input Box (Desired Behavior)
        input_rect = Rectangle(width=4.5, height=2.5, color=RED)
        input_title = Text("Desired Behavior", color=RED).scale(0.6).shift(UP * 0.7)
        input_list = VGroup(
            Text("- Rise Time", color=WHITE).scale(0.4),
            Text("- Steady State Error", color=WHITE).scale(0.4),
            Text("- Stability", color=WHITE).scale(0.4)
        ).arrange(DOWN, aligned_edge=LEFT).next_to(input_title, DOWN, buff=0.2)
        input_group = VGroup(input_rect, input_title, input_list)

        # 2. Create Output Box (PID Constants)
        output_rect = Rectangle(width=3, height=2.5, color=BLUE)
        output_title = Text("PID Constants", color=BLUE).scale(0.6)
        output_group = VGroup(output_rect, output_title)

        # 3. Layout and Arrow
        diagram_layout = VGroup(input_group, output_group).arrange(RIGHT, buff=1.5).to_edge(UP, buff=0.5)
        connecting_arrow = Arrow(input_group.get_right(), output_group.get_left(), color=WHITE)
        
        # Whole group for rotation
        flip_group = VGroup(input_group, output_group, connecting_arrow)

        # 4. Initial Animation
        self.play(Create(input_group), Create(output_group))
        self.play(Create(connecting_arrow))
        self.wait(1)

        # 5. Flip the Script (Rotate 180 degrees)
        self.play(flip_group.animate.rotate(PI))
        self.wait(1)

        # 6. Manual Plot Construction (Avoiding 'Axes' and 'DashedLine' as per constraints)
        origin_point = DOWN * 2.5 + LEFT * 3.5
        x_axis_line = Line(origin_point, origin_point + RIGHT * 7)
        y_axis_line = Line(origin_point, origin_point + UP * 2.5)
        x_axis_label = Text("Time", color=WHITE).scale(0.4).next_to(x_axis_line, RIGHT)
        y_axis_label = Text("RPM", color=WHITE).scale(0.4).next_to(y_axis_line, UP)
        
        # Setpoint Reference (thin line instead of DashedLine)
        setpoint_ref = Line(origin_point + UP * 1.5, origin_point + RIGHT * 7 + UP * 1.5, color=YELLOW)
        setpoint_ref.set_stroke(width=2, opacity=0.5)

        # Curve Points Generation
        curve_points = []
        for i in range(101):
            coord_x = (i / 100) * 6
            # Damped oscillation formula: 1 - exp(-0.7x) * cos(5x)
            coord_y = 1 - np.exp(-0.7 * coord_x) * np.cos(5 * coord_x)
            curve_points.append(origin_point + RIGHT * coord_x + UP * coord_y * 1.5)
        
        oscillation_curve = VMobject(color=GREEN)
        oscillation_curve.set_points_as_corners(curve_points)

        self.play(Create(x_axis_line), Create(y_axis_line), Write(x_axis_label), Write(y_axis_label))
        self.play(Create(setpoint_ref))
        self.play(Create(oscillation_curve), run_time=3)

        # 7. Labels using Brace
        # Rise Time segment (x = 0 to roughly x = 0.5)
        rt_segment = Line(origin_point, origin_point + RIGHT * 0.5)
        rt_brace = Brace(rt_segment, DOWN)
        rt_text = rt_brace.get_text("Rise Time").scale(0.4)

        # Steady State Error segment (at the end of the curve)
        sse_segment = Line(origin_point + RIGHT * 6 + UP * 1.5, origin_point + RIGHT * 6 + UP * 1.7)
        sse_brace = Brace(sse_segment, RIGHT)
        sse_text = sse_brace.get_text("Steady State Error").scale(0.4)

        self.play(Create(rt_brace), Write(rt_text))
        self.play(Create(sse_brace), Write(sse_text))

        # 8. PID Formula and Goal Text

        self.play(Write(formula_tex))
        self.play(FadeIn(goal_note))
