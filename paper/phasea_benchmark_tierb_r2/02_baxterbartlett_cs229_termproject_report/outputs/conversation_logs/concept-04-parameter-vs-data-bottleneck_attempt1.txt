from manim import *

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Title and Voice Narration (Text blocks)
        title = Text("Parameter vs Data Bottleneck", font_size=32).to_edge(UP)
        self.play(Write(title))
        
        # 2. Manual Plotting Construction (Avoiding Axes class)
        def create_custom_plot(origin_point, title_text, color):
            x_axis = Line(origin_point, origin_point + 4 * RIGHT, color=WHITE)
            y_axis = Line(origin_point, origin_point + 3 * UP, color=WHITE)
            plot_title = Text(title_text, font_size=20, color=color).next_to(y_axis, UP)
            
            # Data points (shared conceptual locations)
            pts = [[0.5, 0.8], [1.2, 2.2], [2.0, 1.3], [2.8, 2.7], [3.5, 1.8]]
            dots = VGroup(*[Dot(origin_point + x*RIGHT + y*UP, radius=0.05) for x, y in pts])
            
            return VGroup(x_axis, y_axis, plot_title, dots), pts

        left_origin = 4.5 * LEFT + 0.5 * DOWN
        right_origin = 0.5 * RIGHT + 0.5 * DOWN
        
        plot_nn, pts_nn = create_custom_plot(left_origin, "Neural Network\n(High Variance)", RED)
        plot_lwr, pts_lwr = create_custom_plot(right_origin, "LWR\n(Robust)", GREEN)
        
        self.play(Create(plot_nn), Create(plot_lwr))
        self.wait(1)

        # 3. Create Curves
        # NN: Highly erratic wiggly curve
        nn_curve = VMobject(color=RED)
        nn_points = []
        for i in range(len(pts_nn)-1):
            p1 = left_origin + pts_nn[i][0]*RIGHT + pts_nn[i][1]*UP
            p2 = left_origin + pts_nn[i+1][0]*RIGHT + pts_nn[i+1][1]*UP
            mid = (p1 + p2) / 2
            # Add some "overfitting" wiggles
            nn_points.extend([p1, mid + 0.5*UP, p2])
        nn_curve.set_points_as_corners(nn_points)
        
        # LWR: Smooth piecewise-linear or linear path
        lwr_curve = Line(
            right_origin + 0.3*RIGHT + 1.0*UP, 
            right_origin + 3.7*RIGHT + 2.2*UP, 
            color=GREEN
        )

        self.play(Create(nn_curve), run_time=2)
        self.play(Create(lwr_curve), run_time=2)
        self.wait(1)

        # 4. Bar Chart Construction (Avoiding BarChart class)
        # Scale: 1 unit of MSE = 1 unit of height
        bar_nn = Rectangle(width=0.6, height=1.21 * 1.5, fill_opacity=0.8, color=RED)
        bar_lwr = Rectangle(width=0.6, height=0.85 * 1.5, fill_opacity=0.8, color=GREEN)
        
        bar_nn.move_to(3 * DOWN + 2 * LEFT, EDGE_BELOW)
        bar_lwr.move_to(3 * DOWN + 1 * LEFT, EDGE_BELOW)
        
        label_nn = Text("NN: 1.21", font_size=16).next_to(bar_nn, DOWN)
        label_lwr = Text("LWR: 0.85", font_size=16).next_to(bar_lwr, DOWN)
        mse_title = Text("MSE Comparison", font_size=18).next_to(bar_nn, UP, buff=0.5).shift(RIGHT * 0.5)

        # 5. Formula
        formula = MathTex(
            r"MSE = \frac{\|\mathbf{Y}_{test} - \mathbf{Y}_{pred}\|_F^2}{3n}",
            font_size=32
        ).to_edge(RIGHT).shift(DOWN * 2)

        self.play(
            Create(bar_nn), Create(bar_lwr),
            Write(label_nn), Write(label_lwr),
            Write(mse_title)
        )
        self.play(Write(formula))
        self.wait(3)

        # 6. Conclusion Text
        conclusion = Text(
            "LWR is more robust to small sample sizes", 
            font_size=24, 
            color=BLUE
        ).to_edge(DOWN, buff=0.2)
        
        self.play(Write(conclusion))
        self.wait(3)
