from manim import *
import numpy as np

class QuantifyingStability(Scene):
    def construct(self):
        # 1. Manual Axes Construction (Avoiding the 'Axes' class)
        origin = LEFT * 5 + DOWN * 2
        x_axis = Arrow(origin, origin + RIGHT * 6, buff=0, color=WHITE)
        y_axis = Arrow(origin, origin + UP * 5, buff=0, color=WHITE)
        
        y_label = Text("Velocity (RPM)", font_size=20).rotate(90*DEGREES).next_to(y_axis, LEFT, buff=0.2)
        x_label = Text("Time", font_size=20).next_to(x_axis, DOWN, buff=0.2)
        
        # 2. Create the Velocity Plot
        # We simulate a motor fluctuating around a mean "30 RPM"
        # x scale: 0 to 5 units, y scale: 0 to 4 units (where 2 units = 30 RPM)
        def get_motor_point(t):
            # t goes from 0 to 5
            fluctuation = 0.3 * np.sin(4 * t) + 0.15 * np.cos(9 * t) + 0.1 * np.sin(15 * t)
            y_val = 2.5 + fluctuation # Centered around 30 RPM (y=2.5)
            return origin + RIGHT * t + UP * y_val

        points = [get_motor_point(t) for t in np.linspace(0, 5, 100)]
        graph = VMobject(color=WHITE)
        graph.set_points_as_corners(points)

        # 3. Highlight last 20% of the x-axis
        # Last 20% of 5 units is the range from t=4 to t=5
        highlight_box = Rectangle(
            width=1.0, 
            height=4.5, 
            fill_color=BLUE, 
            fill_opacity=0.3, 
            stroke_width=0
        )
        highlight_box.move_to(origin + RIGHT * 4.5 + UP * 2.25)
        
        sse_text = Text("Final 20%\nSteady State", font_size=16, color=BLUE).next_to(highlight_box, UP)

        # Animation Sequence 1
        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(graph), run_time=3)
        self.play(FadeIn(highlight_box), Write(sse_text))
        self.wait(1)

        # 4. Histogram on the side
        hist_origin = RIGHT * 3 + DOWN * 2
        heights = [0.5, 0.8, 3.5, 1.2, 0.6] # Index 2 is the "Mode" at 30 RPM
        bins = VGroup()
        for i, h in enumerate(heights):
            color = YELLOW if i == 2 else GRAY
            r = Rectangle(width=0.4, height=h, fill_color=color, fill_opacity=0.8, stroke_color=WHITE)
            r.move_to(hist_origin + RIGHT * (i * 0.5), aligned_edge=DOWN)
            bins.add(r)
        
        hist_label = Text("Speed Distribution", font_size=20).next_to(bins, UP)
        mode_tag = Text("30 RPM (Mode)", font_size=16, color=YELLOW).next_to(bins[2], DOWN)

        self.play(
            LaggedStart(*[FadeIn(b, shift=UP) for b in bins], lag_ratio=0.1),
            Write(hist_label),
            Write(mode_tag)
        )
        self.wait(1)

        # 5. Stability Formula
        # Formula: Stability = counts at mode / counts not at mode
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=36
        ).to_edge(UP, buff=1)

        # Visualizing the parts of the fraction
        numerator_highlight = SurroundingRectangle(bins[2], color=YELLOW, buff=0.1)
        denominator_group = VGroup(bins[0], bins[1], bins[3], bins[4])
        denominator_highlight = SurroundingRectangle(denominator_group, color=WHITE, buff=0.1)

        self.play(Write(formula))
        self.play(Create(numerator_highlight))
        self.wait(0.5)
        self.play(ReplacementTransform(numerator_highlight, denominator_highlight))
        self.wait(2)

        # Final cleanup or static hold
        self.play(FadeOut(numerator_highlight), FadeOut(denominator_highlight))
        self.wait(2)
