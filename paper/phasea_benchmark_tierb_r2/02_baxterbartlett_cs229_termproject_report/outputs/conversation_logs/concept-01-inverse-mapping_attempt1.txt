from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Formula Reference (Top)
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=36)
        formula.to_edge(UP, buff=0.3)
        self.add(formula)

        # 2. Relationship Diagram (Split Screen)
        # We define a group for easy rotation later
        input_rect = Rectangle(height=2, width=4, color=BLUE)
        input_title = Text("INPUT", font_size=20, color=BLUE).next_to(input_rect.get_top(), DOWN, buff=0.2)
        input_label = Text("Desired Behavior", font_size=24).next_to(input_title, DOWN, buff=0.3)
        input_box = VGroup(input_rect, input_title, input_label).shift(LEFT * 3.5 + UP * 0.5)

        output_rect = Rectangle(height=2, width=4, color=GREEN)
        output_title = Text("OUTPUT", font_size=20, color=GREEN).next_to(output_rect.get_top(), DOWN, buff=0.2)
        output_label = Text("PID Constants", font_size=24).next_to(output_title, DOWN, buff=0.3)
        output_box = VGroup(output_rect, output_title, output_label).shift(RIGHT * 3.5 + UP * 0.5)

        arrow = Arrow(input_box.get_right(), output_box.get_left(), buff=0.2)
        diagram = VGroup(input_box, output_box, arrow)

        self.play(Create(input_box), Create(output_box))
        self.play(GrowArrow(arrow))
        self.wait(1)

        # "Flip the script": Rotate the diagram 180 degrees
        self.play(diagram.animate.rotate(PI), run_time=1.5)
        self.wait(1)

        # 3. Motor Velocity Plot (Bottom)
        # Using manual construction to avoid potential Axes/DashedLine issues in restricted environments
        origin = DOWN * 2.8 + LEFT * 3.5
        x_axis = Arrow(origin, origin + RIGHT * 7.5, buff=0, color=WHITE)
        y_axis = Arrow(origin, origin + UP * 2.5, buff=0, color=WHITE)
        x_label = Text("Time", font_size=16).next_to(x_axis, RIGHT, buff=0.1)
        y_label = Text("RPM", font_size=16).next_to(y_axis, UP, buff=0.1)

        # Manual creation of a dashed setpoint line at y = 1.5 (relative to origin)
        setpoint_height = 1.5
        segments = []
        for i in range(25):
            start_x = i * 0.3
            if start_x > 7: break
            seg = Line(
                origin + RIGHT * start_x + UP * setpoint_height,
                origin + RIGHT * (start_x + 0.15) + UP * setpoint_height,
                stroke_width=2,
                color=GREY
            )
            segments.append(seg)
        dashed_setpoint = VGroup(*segments)

        # Calculate points for oscillating curve: y = setpoint * (1 - e^-t * cos(3t))
        curve_points = []
        for t_val in np.linspace(0, 7, 100):
            # Map t to x-coordinate, and calculated y to y-coordinate
            x_coord = origin[0] + t_val
            y_calc = setpoint_height * (1 - np.exp(-0.6 * t_val) * np.cos(3 * t_val))
            y_coord = origin[1] + y_calc
            curve_points.append([x_coord, y_coord, 0])

        curve = VMobject(color=RED)
        curve.set_points_as_corners(curve_points)

        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(dashed_setpoint))
        self.play(Create(curve), run_time=3)

        # 4. Labels for Plot
        # Rise Time indication (approx first cross of setpoint)
        rt_line = Line(origin + RIGHT * 0.6, origin + RIGHT * 0.6 + UP * 0.2, color=YELLOW)
        rt_text = Text("Rise Time", font_size=16, color=YELLOW).next_to(rt_line, DOWN, buff=0.1)
        
        # Steady State Error (difference at end of plot)
        sse_arrow = Arrow(origin + RIGHT * 6.5 + UP * 1.4, origin + RIGHT * 6.5 + UP * 1.6, buff=0, color=YELLOW, tip_length=0.1)
        sse_text = Text("Steady State Error", font_size=16, color=YELLOW).next_to(sse_arrow, RIGHT, buff=0.1)

        self.play(Create(rt_line), Write(rt_text))
        self.play(Create(sse_arrow), Write(sse_text))

        self.wait(2)
