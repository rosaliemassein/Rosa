from manim import *

class GeneticWinRateSelection(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)")
        formula.to_edge(UP)
        self.add(formula)

        # 2. Initial Parent Node
        parent_node = self.get_node("Parent", WHITE)
        parent_node.move_to(UP * 1.5)
        self.play(Create(parent_node))

        # Repeat the cycle to show evolution
        for cycle in range(2):
            # 3. Branching Children
            child_labels = ["C1", "C2", "C3"]
            child_nodes = VGroup(*[self.get_node(lbl, BLUE) for lbl in child_labels])
            child_nodes.arrange(RIGHT, buff=1.5).next_to(parent_node, DOWN, buff=1.5)
            
            connections = VGroup(*[
                Line(parent_node.get_bottom(), child.get_top(), buff=0.1) 
                for child in child_nodes
            ])
            
            self.play(Create(connections), Create(child_nodes))

            # 4. Win Rate Bars
            # Set heights for the bars
            win_rates = [0.4, 0.9, 0.5] if cycle == 0 else [0.85, 0.3, 0.6]
            bars = VGroup()
            for i, rate in enumerate(win_rates):
                bar = Rectangle(
                    width=0.6, 
                    height=rate * 2.0, 
                    fill_opacity=0.8, 
                    color=BLUE,
                    fill_color=BLUE
                )
                # Position bar so its top is near the bottom of the child node
                bar.next_to(child_nodes[i], DOWN, buff=0.2)
                bars.add(bar)

            # Animate bars growing downward from the node
            self.play(*[GrowFromPoint(bar, bar.get_edge_center(UP)) for bar in bars])
            self.wait(0.5)

            # 5. Highlight the winner
            winner_idx = win_rates.index(max(win_rates))
            winner_node = child_nodes[winner_idx]
            winner_bar = bars[winner_idx]

            self.play(
                winner_node[0].animate.set_color(GOLD),
                winner_bar.animate.set_color(GOLD),
                run_time=0.5
            )
            self.play(Indicate(winner_node), Flash(winner_node, color=GOLD))

            # 6. Fade out others
            losers = VGroup(
                parent_node,
                connections,
                *[child_nodes[i] for i in range(3) if i != winner_idx],
                *[bars[i] for i in range(3) if i != winner_idx]
            )
            self.play(FadeOut(losers), FadeOut(winner_bar))

            # 7. Move Winner to Parent Position
            target_parent = self.get_node("Parent", WHITE).move_to(UP * 1.5)
            
            self.play(
                winner_node.animate.move_to(target_parent.get_center()),
                run_time=1
            )
            
            # Change label and color to finalize the transformation into the new Parent
            self.play(Transform(winner_node, target_parent))
            parent_node = winner_node
            self.wait(1)

    def get_node(self, label_text, color):
        """Helper to create a node with a circle and text label."""
        circle = Circle(radius=0.5, color=color, fill_opacity=0.2)
        label = Text(label_text, font_size=24).move_to(circle.get_center())
        return VGroup(circle, label)
