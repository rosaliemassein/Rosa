from manim import *

class GeneticWinRateSelection(Scene):
    def construct(self):
        # Display the evolution formula at the top
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)").to_edge(UP)
        self.add(formula)

        # Starting Parent node
        parent_node = Dot(color=BLUE).shift(UP * 1.5)
        parent_label = Text("Parent", font_size=24).next_to(parent_node, UP)
        current_parent = VGroup(parent_node, parent_label)
        self.play(Create(current_parent))

        # Perform two generations of selection
        for gen in range(2):
            # Define target positions for children
            child_positions = [LEFT * 3.5 + DOWN * 0.5, DOWN * 0.5, RIGHT * 3.5 + DOWN * 0.5]
            win_rates = [0.35, 0.6, 0.85] if gen == 0 else [0.9, 0.45, 0.7]
            
            children = VGroup()
            lines = VGroup()
            
            for pos in child_positions:
                c_dot = Dot(color=BLUE).move_to(pos)
                c_label = Text("Child", font_size=20).next_to(c_dot, UP)
                children.add(VGroup(c_dot, c_label))
                lines.add(Line(parent_node.get_center(), c_dot.get_center(), stroke_width=2, stroke_opacity=0.5))

            self.play(Create(lines), FadeIn(children))

            # Create bars for win rates
            bars = VGroup()
            bar_labels = VGroup()
            for i, rate in enumerate(win_rates):
                # Create the bar at full size but we will animate its growth
                bar = Rectangle(
                    height=rate * 2.0, 
                    width=0.6, 
                    fill_opacity=0.8, 
                    fill_color=YELLOW, 
                    stroke_color=WHITE, 
                    stroke_width=1
                ).next_to(children[i], DOWN, buff=0.5)
                # Align the bottom of the bar
                bar.align_to(children[i].get_bottom() + DOWN * 2.5, DOWN)
                
                label = Text(f"{int(rate*100)}%", font_size=16).next_to(bar, DOWN, buff=0.1)
                bars.add(bar)
                bar_labels.add(label)
            
            # Animate the bars growing from their base
            self.play(
                AnimationGroup(
                    *[GrowFromEdge(bar, DOWN) for bar in bars],
                    Write(bar_labels),
                    lag_ratio=0.2
                )
            )
            self.wait(0.5)

            # Select the winner
            winner_idx = win_rates.index(max(win_rates))
            winner_rect = SurroundingRectangle(children[winner_idx], color=GOLD, buff=0.2)
            
            self.play(
                Create(winner_rect),
                bars[winner_idx].animate.set_color(GOLD)
            )
            self.wait(1)

            # Preparation for next generation: Winner becomes parent
            new_parent_pos = UP * 1.5
            
            # Fade out everything except the winning child and the formula
            other_mobjects = [
                lines, 
                bar_labels, 
                bars, 
                winner_rect, 
                current_parent
            ]
            for i in range(len(children)):
                if i != winner_idx:
                    other_mobjects.append(children[i])

            self.play(
                *[FadeOut(m) for m in other_mobjects],
                children[winner_idx].animate.move_to(new_parent_pos)
            )
            
            # Rename the winning child to "Parent" for the next loop
            new_parent_label = Text("Parent", font_size=24).next_to(children[winner_idx][0], UP)
            self.play(ReplacementTransform(children[winner_idx][1], new_parent_label))
            
            # Update references for the next iteration
            parent_node = children[winner_idx][0]
            current_parent = VGroup(parent_node, new_parent_label)
            self.wait(0.5)

        self.wait(2)
