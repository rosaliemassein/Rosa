from manim import *

class Concept01DominantZoneMechanism(Scene):
    def construct(self):
        # 1. Colors and Constants
        COLOR_AHU_START = BLUE
        COLOR_AHU_END = DARK_BLUE
        COLOR_ZONE_NORMAL = GREEN
        COLOR_ZONE_HOT = ORANGE
        COLOR_ZONE_OVERCOOLED = "#E0FFFF" # Chilled white-blue
        
        # 2. Create the central AHU rectangle
        ahu = Rectangle(width=3, height=2, color=COLOR_AHU_START, fill_opacity=1)
        ahu_label = Text("AHU", color=WHITE, font_size=32).move_to(ahu.get_center())
        ahu_group = VGroup(ahu, ahu_label)
        
        # 3. Create five smaller rectangles representing 'Zones'
        # Position them around the central AHU
        zones = VGroup(*[
            Rectangle(width=1.8, height=1.2, color=COLOR_ZONE_NORMAL, fill_opacity=1)
            for _ in range(5)
        ])
        
        # Layout: 2 on the right (will be dominant), 3 on the left (will be overcooled)
        zones[0].move_to(4.5 * RIGHT + 1.5 * UP)    # Dominant Zone 1
        zones[1].move_to(4.5 * RIGHT + 1.5 * DOWN)  # Dominant Zone 2
        zones[2].move_to(4.5 * LEFT + 2.2 * UP)    # Normal Zone 3
        zones[3].move_to(4.5 * LEFT + 0 * UP)       # Normal Zone 4
        zones[4].move_to(4.5 * LEFT + 2.2 * DOWN)  # Normal Zone 5
        
        zone_labels = VGroup(*[
            Text(f"Zone {i+1}", font_size=24).next_to(zones[i], UP, buff=0.1)
            for i in range(5)
        ])

        self.add(ahu_group, zones, zone_labels)
        self.wait(1)

        # 4. Animate a heat gradient (orange) growing in two zones
        self.play(
            zones[0].animate.set_fill(COLOR_ZONE_HOT),
            zones[1].animate.set_fill(COLOR_ZONE_HOT),
            run_time=2
        )
        self.wait(0.5)

        # 5. Show dashed arrows from orange zones to the AHU
        # Using DashedLine with a tip added because DashedArrow is not a standard class
        req_arrow1 = DashedLine(zones[0].get_left(), ahu.get_right(), color=RED).add_tip()
        req_arrow2 = DashedLine(zones[1].get_left(), ahu.get_right(), color=RED).add_tip()
        req_text = Text("Cooling Request", color=RED, font_size=28).next_to(ahu, UP, buff=0.5)
        
        self.play(
            Create(req_arrow1),
            Create(req_arrow2),
            Write(req_text),
            run_time=1.5
        )
        self.wait(0.5)

        # 6. Change AHU color to indicate response
        self.play(
            ahu.animate.set_fill(COLOR_AHU_END),
            FadeOut(req_text),
            run_time=1
        )
        self.wait(0.5)

        # 7. Show 'Supply Air' arrows flowing back to ALL zones
        supply_arrows = VGroup(*[
            DashedLine(ahu.get_center(), zones[i].get_center(), color=COLOR_AHU_END).add_tip()
            for i in range(5)
        ])
        supply_text = Text("Supply Air", color=COLOR_AHU_END, font_size=28).next_to(ahu, DOWN, buff=0.5)
        
        self.play(
            FadeOut(req_arrow1),
            FadeOut(req_arrow2),
            Create(supply_arrows),
            Write(supply_text),
            run_time=2
        )
        self.wait(0.5)

        # 8. The green zones turn a 'chilled' white-blue color to signify overcooling
        self.play(
            zones[2].animate.set_fill(COLOR_ZONE_OVERCOOLED),
            zones[3].animate.set_fill(COLOR_ZONE_OVERCOOLED),
            zones[4].animate.set_fill(COLOR_ZONE_OVERCOOLED),
            run_time=2
        )
        self.wait(2)
