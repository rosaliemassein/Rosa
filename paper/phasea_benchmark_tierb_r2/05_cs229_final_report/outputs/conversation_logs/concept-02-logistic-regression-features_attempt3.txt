from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # Weights for the logistic model
        weight_0 = -2.0
        weight_1 = 0.5
        weight_30 = 0.5

        # Manual Coordinate System setup (to avoid restricted Axes/NumberPlane)
        origin_ref = RIGHT * 3 + DOWN * 1.5
        horiz_line = Line(origin_ref + LEFT * 3, origin_ref + RIGHT * 3, color=WHITE)
        vert_line = Line(origin_ref, origin_ref + UP * 3.5, color=WHITE)
        horiz_label = Text("Logit Value (z)", font_size=16).next_to(horiz_line, DOWN)
        vert_label = Text("Probability P(y=1)", font_size=16).next_to(vert_line, LEFT).rotate(90 * DEGREES)
        self.add(horiz_line, vert_line, horiz_label, vert_label)

        # Helper function to map math values to scene positions
        def get_sigmoid_pos(z_in):
            # Standard sigmoid formula
            p_out = 1.0 / (1.0 + np.exp(-z_in))
            # x-position: maps z from [-4, 4] to 6 units of screen width
            # y-position: maps prob from [0, 1] to 3 units of screen height
            return origin_ref + RIGHT * (z_in * 0.75) + UP * (p_out * 3)

        # Draw the curve manually using a series of points
        curve_steps = np.linspace(-4, 4, 50)
        curve_points = [get_sigmoid_pos(val) for val in curve_steps]
        logistic_curve = VMobject(color=RED)
        logistic_curve.set_points_as_corners(curve_points)
        self.add(logistic_curve)

        # Manual Sliders on the left side
        # Slider for 1-Day T-CSP
        path_1d = Line(LEFT * 6 + UP * 1, LEFT * 3 + UP * 1)
        knob_1d = Dot(path_1d.get_start(), color=GREEN)
        label_1d = Text("1-Day T-CSP", font_size=20).next_to(path_1d, UP)
        
        # Slider for 30-Day T-CSP
        path_30d = Line(LEFT * 6 + DOWN * 1, LEFT * 3 + DOWN * 1)
        knob_30d = Dot(path_30d.get_start(), color=BLUE)
        label_30d = Text("30-Day T-CSP", font_size=20).next_to(path_30d, UP)
        
        self.add(path_1d, knob_1d, label_1d, path_30d, knob_30d, label_30d)

        # Prediction Dot that follows the sigmoid curve
        res_dot = Dot(color=YELLOW)
        
        def refresh_res_dot(mob):
            # Convert knob horizontal position to a feature value (0 to 10 scale)
            # Path width is 3.0 units
            feat_1 = (knob_1d.get_center()[0] - path_1d.get_start()[0]) / 3.0 * 10
            feat_30 = (knob_30d.get_center()[0] - path_30d.get_start()[0]) / 3.0 * 10
            
            # Weighted sum for the exponent
            logit_z = weight_0 + weight_1 * feat_1 + weight_30 * feat_30

            clamped_z = max(min(logit_z, 4), -4)

        self.add(res_dot)
