from manim import *

class Concept01DominantZoneMechanism(Scene):
    def construct(self):
        # Colors (using hex to avoid undefined identifier errors)
        C_LIGHT_BLUE = "#ADD8E6"
        C_DEEP_BLUE = "#00008B"
        C_GREEN = "#008000"
        C_ORANGE = "#FFA500"
        C_RED = "#FF0000"
        C_CHILLED = "#E0FFFF"
        C_SUPPLY_BLUE = "#0000FF"

        # 1. Initialize the AHU (Central Air Handler) at the bottom
        ahu = Rectangle(width=3.0, height=1.5, color=C_LIGHT_BLUE, fill_opacity=0.5)
        ahu.to_edge(DOWN, buff=1.0)
        ahu_label = Text("AHU", font_size=24).move_to(ahu.get_center())
        ahu_group = VGroup(ahu, ahu_label)

        # 2. Initialize five zones at the top
        zones = VGroup(*[
            Rectangle(width=1.5, height=1.2, color=C_GREEN, fill_opacity=0.8) 
            for _ in range(5)
        ])
        zones.arrange(RIGHT, buff=0.4)
        zones.to_edge(UP, buff=1.0)
        
        zone_labels = VGroup(*[
            Text(f"Zone {i+1}", font_size=20).move_to(zones[i].get_center())
            for i in range(5)
        ])

        self.add(ahu_group, zones, zone_labels)
        self.wait(1)

        # 3. Animate heat gradient growing in two "dominant" zones (Zone 2 and Zone 4)
        dominant_indices = [1, 3]
        self.play(
            *[zones[i].animate.set_fill(C_ORANGE) for i in dominant_indices],
            run_time=2
        )
        self.wait(1)

        # 4. Cooling requests (dashed lines with tips) from orange zones to AHU
        requests = VGroup()
        for i in dominant_indices:
            req = DashedLine(zones[i].get_bottom(), ahu.get_top(), color=C_RED, dash_length=0.1)
            req.add_tip(tip_length=0.2)
            requests.add(req)

        request_label = Text("Cooling Request", color=C_RED, font_size=20).next_to(requests, RIGHT)

        self.play(Create(requests), Write(request_label))
        self.wait(1)

        # 5. AHU changes to deep blue as it responds
        self.play(
            ahu.animate.set_fill(C_DEEP_BLUE, opacity=0.9).set_stroke(C_DEEP_BLUE),
            FadeOut(requests),
            FadeOut(request_label),
            run_time=1
        )

        # 6. Supply Air arrows flowing back to ALL zones
        supply_arrows = VGroup()
        for i in range(5):
            arr = Arrow(ahu.get_top(), zones[i].get_bottom(), color=C_SUPPLY_BLUE, buff=0.1)
            supply_arrows.add(arr)
            
        supply_label = Text("Supply Air (Colder)", color=C_SUPPLY_BLUE, font_size=20).next_to(supply_arrows, DOWN, buff=0.5).shift(UP * 2)

        self.play(Create(supply_arrows), Write(supply_label))
        self.wait(1)

        # 7. Green zones turn "chilled" white-blue (overcooling)
        overcooled_indices = [0, 2, 4]
        
        self.play(
            *[zones[i].animate.set_fill(C_CHILLED) for i in overcooled_indices],
            *[zones[i].animate.set_fill(C_GREEN) for i in dominant_indices], # Dominant zones satisfied
            run_time=2
        )
        
        overcool_text = Text("Overcooled", color=C_LIGHT_BLUE, font_size=24).move_to(zones[0].get_top() + UP*0.4)
        self.play(Write(overcool_text))
        
        self.wait(2)
