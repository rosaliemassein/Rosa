from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Sigmoid Math Function
        def sigmoid_calc(val_z):
            return 1.0 / (1.0 + np.exp(-val_z))

        # 2. Coordinate System
        # Using NumberPlane as requested in remarks
        plane = NumberPlane(
            x_range=[-6, 6, 2],
            y_range=[0, 1.2, 0.5],
            x_length=6,
            y_length=4,
            background_line_style={"stroke_opacity": 0.3}
        ).to_edge(RIGHT, buff=0.5)
        
        # Sigmoid Curve
        sigmoid_curve = plane.plot(sigmoid_calc, color=BLUE)
        
        # 3. Trackers for Features
        # ValueTracker is used in Reference Example 2
        tracker_1d = ValueTracker(0.0)
        tracker_30d = ValueTracker(0.0)
        
        # 4. Sliders (Left Side)
        line_1d = Line(DOWN, UP).scale(1.5).shift(LEFT * 4.5 + UP * 1.2)
        line_30d = Line(DOWN, UP).scale(1.5).shift(LEFT * 4.5 + DOWN * 1.2)
        
        dot_1d = Dot(color=YELLOW)
        dot_30d = Dot(color=YELLOW)
        
        # Using named functions for updaters to avoid potential lambda issues
        def update_slider_1d(mob):
            val = tracker_1d.get_value()
            # Map range [-3, 3] to [0, 1] proportion
            prop = (val + 3.0) / 6.0
            if prop < 0: prop = 0
            if prop > 1: prop = 1
            mob.move_to(line_1d.point_from_proportion(prop))
            
        def update_slider_30d(mob):
            val = tracker_30d.get_value()
            prop = (val + 3.0) / 6.0
            if prop < 0: prop = 0
            if prop > 1: prop = 1
            mob.move_to(line_30d.point_from_proportion(prop))
            
        dot_1d.add_updater(update_slider_1d)
        dot_30d.add_updater(update_slider_30d)
        
        label_1d = Text("1-Day T-CSP", font_size=20).next_to(line_1d, UP)
        label_30d = Text("30-Day T-CSP", font_size=20).next_to(line_30d, UP)
        
        # 5. Prediction Dot on Curve
        prediction_dot = Dot(color=RED)
        
        def update_prediction_dot(mob):
            z_score = tracker_1d.get_value() + tracker_30d.get_value()
            # Clamp z_score to plane range
            if z_score < -6: z_score = -6
            if z_score > 6: z_score = 6
            y_prob = sigmoid_calc(z_score)
            mob.move_to(plane.c2p(z_score, y_prob))
        
        prediction_dot.add_updater(update_prediction_dot)
        
        # 6. Formulas
        formula_initial = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=32
        ).to_edge(DOWN, buff=0.6)
        
        # 7. Rendering and Animation
        self.add(plane, sigmoid_curve, line_1d, line_30d, dot_1d, dot_30d, label_1d, label_30d, prediction_dot, formula_initial)
        
        # Move sliders and animate dot
        self.play(tracker_1d.animate.set_value(2.5))
        self.wait(0.5)
        self.play(tracker_30d.animate.set_value(-1.5))
        self.wait(0.5)
        
        # Show final scaling
        formula_scaled = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(0 + 1(2.0) + 1(1.0))}}",
            font_size=32
        ).move_to(formula_initial)
        
        self.play(
            tracker_1d.animate.set_value(2.0),
            tracker_30d.animate.set_value(1.0)
        )
        self.play(ReplacementTransform(formula_initial, formula_scaled))
        self.wait(2)
