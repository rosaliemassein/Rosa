from manim import *

class Concept03TheElbowThreshold(Scene):
    def construct(self):
        # 1. Coordinate System
        # Using explicit function definitions to avoid 'undefined' errors in lambdas
        def curve_func(x_val):
            # A concave curve that rises sharply (1 - (1-x)^3)
            return 1 - (1 - x_val)**3

        axes = Axes(
            x_range=[0, 1.05, 0.2],
            y_range=[0, 1.05, 0.2],
            x_length=5,
            y_length=5,
            axis_config={"include_tip": True, "color": WHITE}
        ).to_edge(LEFT, buff=0.7)

        x_label = axes.get_x_axis_label(
            Tex("Fraction of Zones Flagged", font_size=24),
            edge=DOWN,
            direction=DOWN,
            buff=0.3
        )
        y_label = axes.get_y_axis_label(
            Tex("Recall", font_size=24),
            edge=LEFT,
            direction=LEFT,
            buff=0.3
        ).shift(UP * 0.5)

        # 2. Plot the curve
        curve = axes.plot(curve_func, x_range=[0, 1], color=BLUE)

        # 3. Tracker and Moving Point
        # ValueTracker is standard in Manim
        tracker = ValueTracker(0.0)

        dot = Dot(color=RED)

        def update_dot(mob):
            val = tracker.get_value()
            mob.move_to(axes.c2p(val, curve_func(val)))
        
        dot.add_updater(update_dot)

        # 4. Side-panel Building Floorplan
        # A 5x4 grid representing zones
        zone_squares = VGroup(*[
            Square(side_length=0.4, stroke_width=2, color=WHITE)
            for _ in range(20)
        ]).arrange_in_grid(rows=5, cols=4, buff=0.1).to_edge(RIGHT, buff=1)
        
        zone_title = Text("Building Zones", font_size=24).next_to(zone_squares, UP)

        def update_zones(group):
            val = tracker.get_value()
            threshold = int(val * 20)
            for i in range(len(group)):
                if i < threshold:
                    group[i].set_fill(YELLOW, opacity=0.8)
                else:
                    group[i].set_fill(BLACK, opacity=0)
        
        zone_squares.add_updater(update_zones)

        # 5. Text and Formula
        formula = MathTex(r"\text{Recall} = \frac{TP}{TP + FN}", font_size=36).to_edge(UP)
        
        # 6. Animation Sequence
        self.play(Create(axes), Write(x_label), Write(y_label))
        self.play(Create(curve), Write(formula))
        self.play(Create(zone_squares), Write(zone_title))
        self.add(dot)

        # Move dot across the curve - low to high
        self.play(tracker.animate.set_value(0.9), run_time=5, rate_func=linear)
        self.wait(1)
        
        # Move back to the 'elbow' (approx 0.35)
        self.play(tracker.animate.set_value(0.35), run_time=3)
        
        # 7. Highlight the Elbow
        elbow_pos = axes.c2p(0.35, curve_func(0.35))
        elbow_circle = Circle(radius=0.4, color=GOLD).move_to(elbow_pos)
        elbow_text = Text("The 'Elbow'", font_size=28, color=GOLD).next_to(elbow_circle, UR)
        
        self.play(Create(elbow_circle))
        self.play(Write(elbow_text))
        self.play(Flash(elbow_pos, color=GOLD))
        self.wait(2)

        # Summary text
        summary = Text(
            "Maximize catch rate (Recall)\nMinimize targeted zones",
            font_size=22,
            line_spacing=1.5,
            t2c={"Maximize": GREEN, "Minimize": RED}
        ).next_to(axes, DOWN, buff=1)
        
        self.play(FadeIn(summary))
        self.wait(4)

        # Remove updaters
        dot.remove_updater(update_dot)
        zone_squares.remove_updater(update_zones)
