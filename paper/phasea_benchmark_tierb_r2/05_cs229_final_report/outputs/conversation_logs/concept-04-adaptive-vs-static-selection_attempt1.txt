from manim import *
import numpy as np

class Concept04AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. Setup Titles and Layout
        title_static = Text("Static Benchmark", color=YELLOW).scale(0.7)
        title_ml = Text("Logistic Model (ML)", color=BLUE).scale(0.7)
        
        titles = VGroup(title_static, title_ml).arrange(RIGHT, buff=4).to_edge(UP)
        self.add(titles)

        # 2. Floorplans (Simple Grids)
        def create_floorplan():
            grid = VGroup(*[
                Square(side_length=0.6).set_stroke(WHITE, 2)
                for _ in range(9)
            ]).arrange_in_grid(3, 3, buff=0.1)
            return grid

        floor_left = create_floorplan().next_to(title_static, DOWN, buff=0.5)
        floor_right = create_floorplan().next_to(title_ml, DOWN, buff=0.5)

        # Static Highlights (Frozen)
        static_highlights = [0, 4, 8]
        for idx in static_highlights:
            floor_left[idx].set_fill(RED, opacity=0.6)

        self.add(floor_left, floor_right)

        # 3. Timeline (Bottom)
        months = ["May", "Jun", "Jul", "Aug", "Sep"]
        timeline_line = Line(LEFT * 5, RIGHT * 5).shift(DOWN * 1.5)
        month_labels = VGroup(*[
            Text(m).scale(0.5).move_to(timeline_line.point_from_proportion(i/4) + DOWN * 0.3)
            for i, m in enumerate(months)
        ])
        
        timeline_cursor = Dot(color=YELLOW).move_to(timeline_line.get_start())
        self.add(timeline_line, month_labels, timeline_cursor)

        # 4. Graph (Missed Requests / Recall Error)
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 10, 1],
            x_length=10,
            y_length=1.5,
            axis_config={"include_tip": False}
        ).to_edge(DOWN, buff=0.2)
        
        graph_label = Text("Missed Requests (Recall Error)", color=WHITE).scale(0.4).next_to(axes, UP, buff=0.1)
        
        # Curves
        static_curve = axes.plot(lambda x: 8, color=RED, x_range=[0, 10])
        ml_curve = axes.plot(lambda x: 3 + 1.5 * np.sin(x), color=BLUE, x_range=[0, 10])
        
        static_label = Text("Static", color=RED).scale(0.3).next_to(static_curve, RIGHT, buff=0.1)
        ml_label = Text("ML", color=BLUE).scale(0.3).next_to(ml_curve, RIGHT, buff=0.1)

        # 5. Dynamic Logic using Updaters
        time_tracker = ValueTracker(0)

        # Timeline cursor updater
        timeline_cursor.add_updater(
            lambda d: d.move_to(timeline_line.point_from_proportion(time_tracker.get_value() / 10))
        )

        # Adaptive Floorplan Updater (Right side changing)
        def update_adaptive_floor(mob):
            t = time_tracker.get_value()
            for i, sq in enumerate(mob):
                # Deterministic pattern based on tracker value
                intensity = np.sin(t * 1.5 + i * 0.5)
                if intensity > 0.3:
                    sq.set_fill(BLUE, opacity=0.7)
                else:
                    sq.set_fill(BLACK, opacity=0)

        floor_right.add_updater(update_adaptive_floor)

        # 6. Animation Sequence
        self.play(
            Create(axes),
            Write(graph_label),
            run_time=1
        )
        
        # Animate curves and timeline simultaneously
        self.play(
            Create(static_curve),
            Create(ml_curve),
            Write(static_label),
            Write(ml_label),
            time_tracker.animate.set_value(10),
            run_time=8,
            rate_func=linear
        )
        
        self.wait(1)
        
        # Final Summary Text
        summary = Text(
            "Adaptive ML recognizes dynamic patterns while static benchmarks stay frozen.",
            color=YELLOW
        ).scale(0.5).to_edge(UP, buff=0.1).set_background_stroke(BLACK, width=4, opacity=1)
        
        self.play(FadeIn(summary))
        self.wait(2)
