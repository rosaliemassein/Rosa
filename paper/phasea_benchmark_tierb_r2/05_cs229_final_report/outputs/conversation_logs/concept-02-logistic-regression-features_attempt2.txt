from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Setup Plane and Sigmoid Curve (Avoiding 'Axes' as requested)
        plane = NumberPlane(
            x_range=[-6, 6, 2],
            y_range=[0, 1.1, 0.5],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(RIGHT, buff=0.5)
        
        # Sigmoid function: 1 / (1 + exp(-z))
        sigmoid_curve = plane.plot(
            lambda x: 1 / (1 + np.exp(-x)),
            color=YELLOW
        )
        
        curve_label = MathTex("P(y=1) = \\sigma(z)").next_to(plane, UP, buff=0.2)

        # 2. Mathematical Formulas (Avoiding 'TransformMatchingTex')
        formula_initial = MathTex(
            "P(y=1) = \\frac{1}{1 + e^{-z}}"
        ).scale(0.8).to_edge(UP).shift(LEFT * 2.5)
        
        formula_expanded = MathTex(
            "P(y=1) = \\frac{1}{1 + e^{-(\\beta_0 + \\beta_1 X_{1d} + \\beta_{30} X_{30d})}}"
        ).scale(0.8).to_edge(UP).shift(LEFT * 2.5)

        # 3. Vertical Sliders (Mimicking input features X_1d and X_30d)
        # Using Dot positions to track values instead of 'ValueTracker'
        slider1_line = Line(DOWN, UP).scale(1.5).shift(LEFT * 5.5)
        slider1_dot = Dot(color=BLUE).move_to(slider1_line.get_center())
        slider1_label = Text("1-Day T-CSP", font_size=18).next_to(slider1_line, DOWN)

        slider30_line = Line(DOWN, UP).scale(1.5).shift(LEFT * 4.0)
        slider30_dot = Dot(color=BLUE).move_to(slider30_line.get_center())
        slider30_label = Text("30-Day T-CSP", font_size=18).next_to(slider30_line, DOWN)

        # 4. Prediction Dot on the Sigmoid Curve
        # Logistic weights constants
        B0, B1, B30 = -0.5, 1.2, 0.8
        
        prediction_dot = Dot(color=RED, radius=0.12)
        
        def update_prediction(m):
            # Calculate feature values from slider positions (range approx -2 to 2)
            x1 = (slider1_dot.get_center()[1] - slider1_line.get_center()[1]) / 1.5 * 3
            x30 = (slider30_dot.get_center()[1] - slider30_line.get_center()[1]) / 1.5 * 3
            
            # Calculate logit z
            z = B0 + B1 * x1 + B30 * x30
            # Map z to the plane and sigmoid value
            y = 1 / (1 + np.exp(-z))
            m.move_to(plane.c2p(z, y))

        prediction_dot.add_updater(update_prediction)

        # 5. Animation Sequence
        self.play(Create(plane), Create(sigmoid_curve), Write(curve_label))
        self.play(Write(formula_initial))
        self.wait(1)
        
        # Transform formula to show parameters
        self.play(ReplacementTransform(formula_initial, formula_expanded))
        self.wait(1)

        # Show Sliders
        self.play(
            Create(slider1_line), Create(slider1_dot), Write(slider1_label),
            Create(slider30_line), Create(slider30_dot), Write(slider30_label)
        )
        
        # Add the interactive dot
        self.add(prediction_dot)
        self.wait(1)

        # Animate Sliders to show how features affect probability
        # Moving 1-Day gap up (Higher cooling load memory)
        self.play(
            slider1_dot.animate.move_to(slider1_line.get_top()),
            run_time=2
        )
        self.wait(0.5)

        # Moving 30-Day gap up
        self.play(
            slider30_dot.animate.move_to(slider30_line.get_top()),
            run_time=2
        )
        self.wait(1)

        # Moving both down (Low cooling load memory)
        self.play(
            slider1_dot.animate.move_to(slider1_line.get_bottom()),
            slider30_dot.animate.move_to(slider30_line.get_bottom()),
            run_time=3
        )
        self.wait(2)

        # Final return to center
        self.play(
            slider1_dot.animate.move_to(slider1_line.get_center()),
            slider30_dot.animate.move_to(slider30_line.get_center()),
            run_time=1.5
        )
        self.wait(2)
