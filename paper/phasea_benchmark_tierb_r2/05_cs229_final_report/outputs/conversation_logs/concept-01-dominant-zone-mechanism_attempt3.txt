from manim import *

class DominantZoneMechanism(Scene):
    def construct(self):
        # Define specific hex colors to avoid undefined identifier errors
        color_ahu_light = "#87CEEB" # Light Blue
        color_ahu_deep = "#00008B"  # Deep Blue
        color_zone_green = "#228B22" # Forest Green
        color_zone_heat = "#FF8C00"  # Dark Orange
        color_zone_chilled = "#E0FFFF" # Light Cyan / White-Blue
        color_white = "#FFFFFF"

        # 1. Setup AHU and Zones
        # AHU at the top
        ahu_rect = Rectangle(width=3, height=1.5, color=color_ahu_light, fill_opacity=0.8)
        ahu_label = Text("AHU", font_size=24, color=color_white).move_to(ahu_rect.get_center())
        ahu = VGroup(ahu_rect, ahu_label).to_edge(UP, buff=0.8)

        # 5 Zones at the bottom
        zones = VGroup(*[
            VGroup(
                Rectangle(width=2.0, height=1.2, color=color_zone_green, fill_opacity=0.6),
                Text(f"Zone {i+1}", font_size=18, color=color_white)
            ) for i in range(5)
        ])
        zones.arrange(RIGHT, buff=0.4).to_edge(DOWN, buff=1.2)

        self.add(ahu, zones)
        self.wait(1)

        # 2. Heat gradient growing in two dominant zones (Zone 1 and Zone 2)
        dominant_indices = [0, 1]
        self.play(
            *[zones[i][0].animate.set_fill(color_zone_heat, opacity=0.9).set_stroke(color_zone_heat) 
              for i in dominant_indices],
            run_time=2
        )
        self.wait(0.5)

        # 3. Cooling request arrows (Dashed) moving from orange zones to AHU
        req_arrows = VGroup()
        for i in dominant_indices:
            # Create a dashed line and add a tip to make it an arrow
            line = DashedLine(
                start=zones[i].get_top(), 
                end=ahu_rect.get_bottom(), 
                color=color_zone_heat,
                dash_length=0.15
            )
            line.add_tip(tip_length=0.2)
            req_arrows.add(line)
        
        req_label = Text("Cooling Request", color=color_zone_heat, font_size=20).next_to(req_arrows, RIGHT, buff=0.5)

        self.play(Create(req_arrows))
        self.play(Write(req_label))
        self.wait(1)

        # 4. AHU changes color to deep blue
        self.play(
            ahu_rect.animate.set_color(color_ahu_deep).set_fill(color_ahu_deep),
            FadeOut(req_arrows),
            FadeOut(req_label),
            run_time=1
        )
        self.wait(0.5)

        # 5. Deep blue 'Supply Air' arrows flowing back to ALL zones
        supply_arrows = VGroup()
        for i in range(5):
            arr = Arrow(
                start=ahu_rect.get_bottom(), 
                end=zones[i].get_top(), 
                color=color_ahu_deep, 
                buff=0.1,
                stroke_width=5
            )
            supply_arrows.add(arr)
        
        supply_label = Text("Supply Air (Chilled)", color=color_ahu_deep, font_size=20).next_to(supply_arrows, RIGHT, buff=0.1)

        self.play(Create(supply_arrows))
        self.play(Write(supply_label))
        self.wait(1)

        # 6. Green zones (non-dominant) turn white-blue to signify overcooling
        other_indices = [2, 3, 4]
        self.play(
            *[zones[i][0].animate.set_fill(color_zone_chilled, opacity=0.8).set_color(color_zone_chilled) 
              for i in other_indices],
            FadeOut(supply_arrows),
            FadeOut(supply_label),
            run_time=2
        )

        # Final explanatory text
        explanation = Text(
            "Majority of building overcooled to satisfy dominant zones.",
            font_size=24,
            color=color_white
        ).to_edge(DOWN, buff=0.3)
        
        self.play(Write(explanation))
        self.wait(3)
