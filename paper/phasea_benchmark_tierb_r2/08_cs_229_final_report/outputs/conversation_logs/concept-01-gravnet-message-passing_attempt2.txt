from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Setup Formula and Reference Text
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # 2. Create an irregular grid of dots (representing ECal cells)
        dots = VGroup()
        rows, cols = 4, 4
        spacing = 1.5
        
        # Generate a grid with hexagonal layout and random jitter
        for r in range(rows):
            for c in range(cols):
                # Standard hexagonal grid math
                x = (c - cols / 2) * spacing + (0.5 * spacing if r % 2 else 0)
                y = (r - rows / 2) * spacing * 0.866
                # Add jitter for "irregularity"
                pos = np.array([x, y, 0]) + np.random.normal(0, 0.2, 3)
                pos[2] = 0  # Ensure it stays in 2D plane
                dots.add(Dot(point=pos, color=GREEN, radius=0.12))

        # Identify a central dot to focus on
        center_idx = len(dots) // 2 + 1
        central_dot = dots[center_idx]
        
        self.add(dots)
        self.play(
            central_dot.animate.set_color(YELLOW).scale(1.5),
            Write(formula),
            run_time=1
        )

        # 3. Create arrows with updaters to handle dynamic distance scaling
        # The connection strength is determined by the formula V = exp(-d^2)
        arrows = VGroup()
        
        def get_weighted_arrow(target_dot):
            p1 = central_dot.get_center()
            p2 = target_dot.get_center()
            dist = np.linalg.norm(p1 - p2)
            # Apply the formula exp(-d^2). 
            # We use a divisor (2.0) to make the visual transition smoother for screen units.
            weight = np.exp(-(dist**2) / 2.0)
            
            return Arrow(
                p1, p2, 
                buff=0.1, 
                stroke_width=weight * 12, # Thickness scales with connection strength
                max_tip_length_to_length_ratio=0.15,
                color=WHITE
            ).set_opacity(max(0.1, weight)) # Opacity scales with connection strength

        for dot in dots:
            if dot == central_dot:
                continue
            
            # Initial state
            arr = get_weighted_arrow(dot)
            # Add updater: lambda receives the arrow object (m) and we pass the target dot (d)
            arr.add_updater(lambda m, d=dot: m.become(get_weighted_arrow(d)))
            arrows.add(arr)

        self.play(FadeIn(arrows), run_time=1.5)
        self.wait(1)

        # 4. Animate "Learned Feature Space" movement
        # Dots shift based on learned relationships, and connections update automatically
        move_animations = []
        for dot in dots:
            # Shift dots by a random vector to simulate the dynamic nature of GravNet
            random_shift = np.array([
                np.random.uniform(-1.2, 1.2),
                np.random.uniform(-1.2, 1.2),
                0
            ])
            move_animations.append(dot.animate.shift(random_shift))
        
        # Move the central dot too
        move_animations.append(central_dot.animate.shift(RIGHT * 0.5 + UP * 0.2))

        self.play(*move_animations, run_time=5, rate_func=smooth)
        self.wait(2)
