from manim import *
import numpy as np

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # Instead of Axes, use manual Lines for the graph
        # Center of the plot area
        origin = DOWN * 2 + LEFT * 4
        x_axis = Line(origin, origin + RIGHT * 8, color=WHITE)
        y_axis = Line(origin, origin + UP * 5, color=WHITE)
        
        # Labels
        x_label = Tex("Distance between particles").scale(0.8).next_to(x_axis, DOWN)
        y_label = Tex("Efficiency").scale(0.8).rotate(90 * DEGREES).next_to(y_axis, LEFT)
        
        self.add(x_axis, y_axis, x_label, y_label)

        # Plot ML Curve (mostly flat at 0.9 height)
        # 0.9 efficiency = origin + UP * (5 * 0.9)
        ml_start = origin + UP * 4.5
        ml_end = origin + RIGHT * 8 + UP * 4.5
        ml_line = Line(ml_start, ml_end, color=BLUE)
        ml_tag = Tex("ML Model", color=BLUE).scale(0.7).next_to(ml_end, UP)
        
        # Plot Standard Algorithm Curve (drops sharply)
        # Create points manually to avoid 'lambda x' issues
        std_points = []
        for val in np.linspace(0, 8, 50):
            # Efficiency formula: 0.9 * (1 - e^-x)
            eff = 4.5 * (1 - np.exp(-val))
            std_points.append(origin + RIGHT * val + UP * eff)
            
        std_curve = VMobject(color=RED)
        std_curve.set_points_as_corners(std_points)
        # Manually create a dashed version to be safe
        std_dashed = DashedLine(origin, origin + RIGHT * 8).match_points(std_curve)
        std_dashed.set_color(RED)
        
        std_tag = Tex("Standard Algorithm", color=RED).scale(0.7).next_to(origin + RIGHT * 6 + UP * 3.5, DOWN)

        self.add(ml_line, std_dashed, ml_tag, std_tag)

        # Tracker: Using a hidden Mobject's position as a value
        # Move from x=8 to x=0.2
        tracker_obj = Dot(origin + RIGHT * 8).set_opacity(0)
        
        # Vertical line indicating current distance
        # Use add_updater instead of always_redraw
        v_line = Line(origin, origin + UP * 4.5, color=YELLOW, stroke_width=2)
        def update_vline(mob):
            curr_x = tracker_obj.get_center()[0]
            mob.set_x(curr_x)
            mob.set_y(origin[1] + 2.25) # center vertically
        v_line.add_updater(update_vline)

        # Particle shower icons (circles merging)
        shower_center = UP * 2 + RIGHT * 3
        dot_l = Dot(radius=0.3, color=WHITE, fill_opacity=0.5)
        dot_r = Dot(radius=0.3, color=WHITE, fill_opacity=0.5)
        
        def update_dots(mob):
            # distance factor based on tracker position
            dist_factor = (tracker_obj.get_center()[0] - origin[0]) / 8.0
            mob[0].move_to(shower_center + LEFT * dist_factor * 1.5)
            mob[1].move_to(shower_center + RIGHT * dist_factor * 1.5)
            
        shower_group = VGroup(dot_l, dot_r)
        shower_group.add_updater(update_dots)
        
        shower_label = Tex("Detector Hits").scale(0.7).next_to(shower_center, UP, buff=1)

        self.add(v_line, shower_group, shower_label, tracker_obj)

        # Animation
        self.play(
            tracker_obj.animate.move_to(origin + RIGHT * 0.2),
            run_time=6,
            rate_func=linear
        )
        
        self.wait(2)
