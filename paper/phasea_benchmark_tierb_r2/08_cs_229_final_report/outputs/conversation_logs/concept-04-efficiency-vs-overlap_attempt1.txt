from manim import *
import numpy as np

class Concept04EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Manual Axis Construction
        # We avoid Axes and ValueTracker as they are disallowed in this constrained subset.
        origin_pt = LEFT * 3.5 + DOWN * 2
        x_ax = Line(origin_pt, origin_pt + RIGHT * 7)
        y_ax = Line(origin_pt, origin_pt + UP * 5)
        
        dist_label = Text("Particle Distance").scale(0.5).next_to(x_ax, DOWN)
        eff_label = Text("Efficiency").scale(0.5).rotate(90 * DEGREES).next_to(y_ax, LEFT)
        
        # 2. ML Curve (Efficiency stays mostly flat at 0.9)
        # Height mapping: 1.0 efficiency = 4.5 units high
        ml_height = 4.05 
        ml_curve = Line(origin_pt + UP * ml_height, origin_pt + UP * ml_height + RIGHT * 6, color=BLUE)
        ml_text = Text("ML Model", color=BLUE).scale(0.4).next_to(ml_curve, UP, buff=0.1)
        
        # 3. Standard Curve (Starts high, drops sharply at small distances)
        # We build the curve point-by-point to avoid the 'plot' and 'lambda x' issues.
        std_pts = []
        for i in range(41):
            t = i / 40.0  # Normalized x-coordinate (0 to 1)
            dist_x = t * 6
            # Sigmoid-like drop-off: starts at 0.9, collapses as distance (t) decreases
            # Formula: 0.9 / (1 + exp(-12 * (t - 0.2)))
            efficiency = 0.9 / (1.0 + np.exp(-12.0 * (t - 0.2)))
            std_pts.append(origin_pt + RIGHT * dist_x + UP * (efficiency * 4.5))
        
        std_curve = VMobject(color=RED)
        std_curve.set_points_as_corners(std_pts)
        std_text = Text("Standard Algorithm", color=RED).scale(0.4).next_to(std_pts[-1], DOWN, buff=0.2)

        # 4. Moving Indicator Tracker
        # Instead of ValueTracker, we use a hidden Dot and track its position.
        tracker_dot = Dot(radius=0).move_to(origin_pt + RIGHT * 6)
        
        # Vertical scanning line
        v_line = Line(DOWN * 2, UP * 2.5, color=YELLOW, stroke_width=2)
        v_line.add_updater(lambda m: m.move_to([tracker_dot.get_center()[0], origin_pt[1] + 2.25, 0]))
        
        # 5. Particle Shower Simulation (Overlay at the top-right)
        shower_base = UP * 2.5 + RIGHT * 2.5
        p1 = Dot(color=RED, radius=0.15)
        p2 = Dot(color=GREEN, radius=0.15)
        
        # Updaters for the "merging" effect as distance decreases
        def update_p1(m):
            # Calculate distance fraction based on tracker dot (0.0 to 1.0)
            dist_frac = (tracker_dot.get_center()[0] - origin_pt[0]) / 6.0
            m.move_to(shower_base + LEFT * (dist_frac * 1.2))
            
        def update_p2(m):
            dist_frac = (tracker_dot.get_center()[0] - origin_pt[0]) / 6.0
            m.move_to(shower_base + RIGHT * (dist_frac * 1.2))
            
        p1.add_updater(update_p1)
        p2.add_updater(update_p2)
        
        shower_label = Text("Detector Hit Resolution", color=WHITE).scale(0.4).next_to(shower_base, UP, buff=0.5)

        # 6. Assembly and Animation
        self.add(x_ax, y_ax, dist_label, eff_label)
        self.play(Create(ml_curve), Write(ml_text))
        self.play(Create(std_curve), Write(std_text))
        self.wait(1)
        
        # Add tracking elements
        self.add(v_line, p1, p2, shower_label)
        
        # Animate the tracker moving left, illustrating the efficiency drop vs ML stability
        self.play(
            tracker_dot.animate.move_to(origin_pt + RIGHT * 0.1),
            run_time=8,
            rate_func=lambda t: t
        )
        
        self.wait(2)
        
        # Final Highlight: Standard efficiency collapses while ML stays high
        highlight_text = Text("ML resolves overlaps", color=BLUE).scale(0.5).to_edge(UP)
        self.play(Write(highlight_text))
        self.wait(2)
