from manim import *
import numpy as np

class Concept04EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Axes setup
        # Use standard Axes as shown in reference examples
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=8,
            y_length=5,
            axis_config={"include_tip": False}
        ).shift(DOWN * 0.5)
        
        # Adding simple text labels for axes
        x_label = Text("Distance between particles", font_size=24).next_to(axes.x_axis, DOWN)
        y_label = Text("Efficiency", font_size=24).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)

        # 2. Curve Functions
        # ML model stays flat at 0.9 efficiency
        def ml_efficiency(x):
            return 0.9
        
        # Standard algorithm efficiency drops sharply at low distance
        def standard_efficiency(x):
            # Sigmoid-like curve: high at large x, drops to 0 at x=0
            if x < 0.001: return 0
            return 0.85 * (1.0 - np.exp(-12.0 * x))

        ml_curve = axes.plot(ml_efficiency, x_range=[0, 1.0], color=BLUE)
        std_curve = axes.plot(standard_efficiency, x_range=[0, 1.0], color=RED)
        
        # Labels for the curves
        ml_tag = Text("ML Model", color=BLUE, font_size=20).move_to(axes.c2p(0.8, 0.95))
        std_tag = Text("Standard Algorithm", color=RED, font_size=20).move_to(axes.c2p(0.8, 0.4))

        # 3. ValueTracker for animation
        # distance_tracker will move from right to left (1.0 to 0.05)
        distance_tracker = ValueTracker(0.9)

        # 4. Moving Elements (Vertical line and particle icons)
        # Vertical line indicating current distance on the x-axis
        v_line = Line(color=YELLOW, stroke_width=2)
        v_line.add_updater(lambda m: m.become(
            Line(
                axes.c2p(distance_tracker.get_value(), 0),
                axes.c2p(distance_tracker.get_value(), 1),
                color=YELLOW
            )
        ))

        # Particle icons (represented by two dots) merging at the top
        dot_left = Dot(color=WHITE)
        dot_right = Dot(color=WHITE)
        showers = VGroup(dot_left, dot_right)
        
        def update_showers(group):
            val = distance_tracker.get_value()
            # Move dots toward center as value decreases
            group[0].move_to(UP * 2.5 + LEFT * (val * 1.5))
            group[1].move_to(UP * 2.5 + RIGHT * (val * 1.5))
        
        showers.add_updater(update_showers)

        # 5. Scene Construction
        self.add(axes, x_label, y_label, ml_curve, std_curve, ml_tag, std_tag, v_line, showers)

        # 6. Animation Sequence
        # Animate the distance closing over 6 seconds
        self.play(
            distance_tracker.animate.set_value(0.05),
            run_time=6,
            rate_func=lambda t: t  # Linear movement
        )
        
        # Final pause to observe the efficiency gap
        self.wait(2)
