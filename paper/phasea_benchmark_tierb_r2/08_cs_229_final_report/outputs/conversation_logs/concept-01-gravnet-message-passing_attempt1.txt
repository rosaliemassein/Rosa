from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Create an irregular grid of dots (simulating nodes/hits)
        np.random.seed(42)
        dots = VGroup()
        for x in range(-3, 4):
            for y in range(-2, 3):
                # Add randomness to simulate irregular geometry
                pos = [
                    x * 1.5 + np.random.uniform(-0.5, 0.5),
                    y * 1.2 + np.random.uniform(-0.5, 0.5),
                    0
                ]
                dots.add(Dot(point=pos, radius=0.08, color=BLUE))

        # Select a central dot and highlight it
        central_dot = dots[len(dots) // 2]
        highlight_ring = Circle(radius=0.2, color=RED).move_to(central_dot)
        
        # 2. Define the formula
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL).scale(0.8)

        # 3. Create dynamic arrows with distance-weighted properties
        # We use a custom function for arrow update logic
        arrows = VGroup()
        for dot in dots:
            if dot == central_dot:
                continue

            # Create the arrow using always_redraw to dynamically update its scale and opacity
            arrow = always_redraw(lambda d=dot: self.get_weighted_arrow(central_dot, d))
            arrows.add(arrow)

        # 4. Narrative Text
        narr_1 = Text("Hits in the irregular LDMX calorimeter are nodes in a graph.", font_size=20).to_edge(DOWN)
        narr_2 = Text("GravNet weights connections by distance in a learned feature space.", font_size=20).to_edge(DOWN)
        narr_3 = Text("The graph structure updates dynamically as the model 'learns'.", font_size=20).to_edge(DOWN)

        # 5. Animation sequence
        self.play(Create(dots), run_time=1.5)
        self.play(central_dot.animate.set_color(RED), Create(highlight_ring))
        self.play(Write(narr_1))
        self.wait(1.5)

        self.play(FadeIn(arrows, lag_ratio=0.05), Write(formula))
        self.play(ReplacementTransform(narr_1, narr_2))
        self.wait(2)

        self.play(ReplacementTransform(narr_2, narr_3))
        
        # 6. Simulate the "learned features" update by moving dots
        # This triggers the always_redraw arrows to change their appearance
        movements = []
        for dot in dots:
            if dot == central_dot:
                # Central dot moves slightly
                target = dot.get_center() + np.array([np.random.uniform(-0.5, 0.5), np.random.uniform(-0.5, 0.5), 0])
            else:
                # Other dots move to change clustering
                target = dot.get_center() + np.array([np.random.uniform(-1.5, 1.5), np.random.uniform(-1.5, 1.5), 0])
            movements.append(dot.animate.move_to(target))
        
        # Also move the highlight ring with the central dot
        movements.append(highlight_ring.animate.move_to(movements[len(dots)//2].target))

        self.play(*movements, run_time=5, rate_func=smooth)
        self.wait(2)

    def get_weighted_arrow(self, start_obj, end_obj):
        start = start_obj.get_center()
        end = end_obj.get_center()
        dist = np.linalg.norm(end - start)
        
        # Formula implementation: V = exp(-d^2)
        # We scale distance for visual clarity
        scaled_dist = dist * 0.5
        weight = np.exp(-(scaled_dist**2))
        
        # Visual thresholds to avoid clutter
        if weight < 0.05:
            return Line(start, start, stroke_width=0).set_opacity(0)
            
        return Arrow(
            start, end, 
            buff=0.1, 
            stroke_width=weight * 8, 
            max_tip_length_to_length_ratio=0.15,
            color=WHITE
        ).set_opacity(weight)
