from manim import *
import numpy as np

class ConceptMatrix(Scene):
    def construct(self):
        # Configuration
        rows, cols = 10, 15
        cell_size = 0.3
        grid_buff = 0.05
        
        # 1. Create silhouette of 15 players
        players = VGroup(*[
            Dot(radius=0.1, color=GRAY) for _ in range(15)
        ]).arrange(DOWN, buff=0.2).to_edge(LEFT, buff=1)
        player_label = Text("All Players", font_size=20).next_to(players, UP)
        
        self.play(Create(players), Write(player_label))
        self.wait(0.5)

        # 2. 'Minutes Played' filter
        filter_line = Line(players.get_left(), players.get_right(), color=YELLOW).scale(1.5)
        filter_line.move_to(players[0].get_top() + UP * 0.1)
        filter_text = Text("Minutes Played", color=YELLOW, font_size=18).next_to(filter_line, RIGHT)
        
        self.play(Create(filter_line), FadeIn(filter_text))
        
        # Highlight top 10
        self.play(
            filter_line.animate.move_to(players[9].get_bottom() + DOWN * 0.1),
            *[players[i].animate.set_color(BLUE) for i in range(10)],
            *[players[i].animate.set_color(DARK_GRAY) for i in range(10, 15)],
            run_time=2
        )
        self.play(FadeOut(filter_line), FadeOut(filter_text), FadeOut(player_label))
        
        # 3. Create 10x15 grid (Matrix)
        grid = VGroup(*[
            Square(side_length=cell_size, stroke_width=1, color=WHITE) 
            for _ in range(rows * cols)
        ]).arrange_in_grid(rows=rows, cols=cols, buff=grid_buff).shift(RIGHT * 1.5)
        
        # 4. Players slide into rows
        row_starts = VGroup(*[grid[i * cols] for i in range(rows)])
        
        self.play(
            *[ReplacementTransform(players[i], grid[i * cols]) for i in range(10)],
            *[FadeOut(players[i]) for i in range(10, 15)],
            Create(VGroup(*[c for i, c in enumerate(grid) if i % cols != 0]))
        )
        self.wait(0.5)

        # 5. Column Labels
        stat_names = ["PTS", "REB", "AST", "STL", "BLK", "TOV", "FG%", "3P%", "FT%", "ORB", "DRB", "PF", "MIN", "PM", "PIE"]
        labels = VGroup()
        for i in range(cols):
            lab = Text(stat_names[i], font_size=12).next_to(grid[i], UP, buff=0.1).rotate(45 * DEGREES)
            labels.add(lab)
        
        self.play(Write(labels))
        self.wait(0.5)

        # 6. Fill grid with varying intensities
        for cell in grid:
            val = np.random.random()
            cell.set_fill(BLUE, opacity=val)
            cell.set_stroke(WHITE, width=0.5)

        self.play(
            *[cell.animate.set_fill(BLUE, opacity=np.random.random()) for cell in grid],
            run_time=1.5
        )
        self.wait(0.5)

        # 7. Collapse 2D grid into a single horizontal vector
        formula = MathTex("n = 10 \\times 15 = 150").to_edge(UP)
        self.play(Write(formula))
        
        # Rearrange cells into one long row
        self.play(
            FadeOut(labels),
            grid.animate.arrange(RIGHT, buff=0.01).scale(0.4).center().to_edge(DOWN, buff=2),
            run_time=2
        )
        
        # Substitute for Brace (Disallowed)
        vec_line = Line(grid.get_left() + UP*0.2, grid.get_right() + UP*0.2, color=WHITE)
        vec_text = Text("150-element Feature Vector", font_size=24).next_to(vec_line, UP, buff=0.2)
        
        self.play(Create(vec_line), Write(vec_text))
        self.wait(1)

        # Final Formula
        final_formula = MathTex("n = m \\times p").next_to(vec_text, UP, buff=0.5)
        self.play(Write(final_formula))
        self.wait(2)

        self.play(FadeOut(grid), FadeOut(vec_line), FadeOut(vec_text), FadeOut(final_formula), FadeOut(formula))
