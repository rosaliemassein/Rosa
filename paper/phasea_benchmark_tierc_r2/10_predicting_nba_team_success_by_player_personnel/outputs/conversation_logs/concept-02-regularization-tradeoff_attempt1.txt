from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup the Axes
        # MSE is typically positive. We'll plot KNN overfitting and Lasso/Ridge improving.
        axes = Axes(
            x_range=[1, 10, 1],
            y_range=[0, 10, 2],
            x_length=7,
            y_length=4,
            axis_config={"include_tip": True}
        ).shift(DOWN * 0.5)

        # Labels for the axes
        labels = axes.get_axis_labels(
            Tex("Number of Players").scale(0.6), 
            Tex("Mean Squared Error").scale(0.6).rotate(90 * DEGREES)
        )
        
        # Define functions for plotting to avoid lambda-related scope issues in some environments
        def knn_func(t):
            return 0.3 * (t - 3.5)**2 + 1.5

        def lasso_func(t):
            return 7 * np.exp(-0.25 * t) + 1

        knn_curve = axes.plot(knn_func, x_range=[1, 10], color=RED)
        knn_tag = Tex("KNN (Overfitting)", color=RED).scale(0.5).next_to(knn_curve.point_from_proportion(0.9), UP + RIGHT, buff=0.1)
        
        lasso_curve = axes.plot(lasso_func, x_range=[1, 10], color=GREEN)
        lasso_tag = Tex("Lasso/Ridge", color=GREEN).scale(0.5).next_to(lasso_curve.point_from_proportion(0.9), DOWN + RIGHT, buff=0.1)

        # 2. Formula: Lasso Cost Function
        formula = MathTex(r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1").to_edge(UP, buff=0.5)

        # Initial Plot Animation
        self.play(Create(axes), Write(labels))
        self.play(Create(knn_curve), Write(knn_tag))
        self.play(Create(lasso_curve), Write(lasso_tag))
        self.play(Write(formula))
        self.wait(2)

        # 3. Transition to Penalty Visualization
        plot_group = VGroup(axes, labels, knn_curve, knn_tag, lasso_curve, lasso_tag)
        self.play(
            plot_group.animate.scale(0.6).to_edge(LEFT, buff=0.5),
            formula.animate.scale(0.8).to_edge(UP, buff=0.3)
        )

        # 4. Coefficients Bar Chart (Representing 150 features)
        n_coeffs = 150
        bar_width = 0.02
        
        # Generate random initial values for coefficients
        np.random.seed(42) # For consistency
        initial_heights = [np.random.uniform(0.2, 2.5) for _ in range(n_coeffs)]
        bars = VGroup(*[
            Rectangle(
                width=bar_width, 
                height=h, 
                fill_color=BLUE, 
                fill_opacity=0.8, 
                stroke_width=0
            ) for h in initial_heights
        ]).arrange(RIGHT, buff=0.005, aligned_edge=DOWN).next_to(plot_group, RIGHT, buff=0.8).shift(DOWN * 0.5)

        coeff_title = Tex("Model Coefficients").scale(0.7).next_to(bars, UP, buff=0.5)
        
        # Lambda tracker for dynamic text display
        lambda_tracker = ValueTracker(0.0)
        # Using a function and redraw to keep lambda text updated
        lambda_text = always_redraw(lambda: 
            MathTex(r"\lambda = " + f"{lambda_tracker.get_value():.2f}")
            .scale(0.8)
            .next_to(coeff_title, UP, buff=0.2)
        )

        self.play(
            LaggedStart(*[FadeIn(b, shift=UP * 0.2) for b in bars], lag_ratio=0.005),
            Write(coeff_title),
            Write(lambda_text)
        )
        self.wait(1)

        # 5. Shrink/Vanishing Animation (Lasso Feature Selection)
        shrink_animations = []
        for i, b in enumerate(bars):
            # Roughly 70% of coefficients hit zero (vanishing)
            if np.random.rand() < 0.7:
                new_h = 0.01 
                new_op = 0
            else:
                # The remaining shrink significantly but stay visible
                new_h = b.height * np.random.uniform(0.1, 0.4)
                new_op = 0.8
            
            shrink_animations.append(
                b.animate.set_height(new_h, stretch=True, about_edge=DOWN).set_fill(opacity=new_op)
            )

        # Play the penalty effect: lambda increases, coefficients shrink
        self.play(
            lambda_tracker.animate.set_value(5.0),
            *shrink_animations,
            run_time=4,
            rate_func=rate_functions.smooth
        )
        self.wait(3)
