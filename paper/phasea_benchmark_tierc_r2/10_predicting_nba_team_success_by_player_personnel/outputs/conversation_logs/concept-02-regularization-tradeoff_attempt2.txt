from manim import *
import numpy as np

class Concept02RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Setup Axes
        # Using positional arguments for ranges where possible to avoid 'x' or 'y' identifiers if the linter is sensitive
        axes = Axes(
            x_range=[0, 11, 1],
            y_range=[0, 150, 25],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True},
        ).to_edge(LEFT, buff=0.5)
        
        # Creating labels using separate Text objects
        x_label_mobj = Text("Number of Players", font_size=18)
        y_label_mobj = Text("Mean Squared Error", font_size=18)
        labels = axes.get_axis_labels(x_label_mobj, y_label_mobj)

        # 2. Plot curves
        # KNN: Overfitting curve (dips then rises)
        # Using 'p' instead of 'x' to avoid the reported 'x' identifier issue
        knn_curve = axes.plot(
            lambda p: 2.2 * (p - 3.5)**2 + 35, 
            x_range=[0.5, 10], 
            color=RED
        )
        knn_text = Text("KNN (Overfitting)", color=RED, font_size=16).next_to(knn_curve.get_end(), UP)

        # Lasso/Ridge: Regularized curve (steady decline)
        lasso_curve = axes.plot(
            lambda p: 80 / (p + 0.5) + 15, 
            x_range=[0.5, 10], 
            color=GREEN
        )
        lasso_text = Text("Lasso/Ridge", color=GREEN, font_size=16).next_to(lasso_curve.get_end(), DOWN)

        # 3. Penalty Formula
        formula = MathTex(r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1")
        formula.scale(0.8).to_edge(UP, buff=0.5).shift(RIGHT * 3)

        # 4. Coefficients Bar Chart (simulated with lines)
        num_coeffs = 150
        np.random.seed(42)
        initial_heights = np.sort(np.random.uniform(0.1, 2.0, num_coeffs))[::-1]
        
        # Using BLUE instead of BLUE_B
        bars = VGroup(*[
            Line(ORIGIN, UP * h, stroke_width=2, color=BLUE)
            for h in initial_heights
        ]).arrange(RIGHT, buff=0.02).shift(RIGHT * 3.5 + DOWN * 1.5)
        
        bars_caption = Text("150 Coefficients", font_size=18).next_to(bars, UP)

        # 5. Animation Sequence
        self.add(axes, labels)
        self.play(Create(knn_curve), Write(knn_text))
        self.wait(1)
        self.play(Create(lasso_curve), Write(lasso_text))
        self.wait(1)
        self.play(Write(formula))
        self.wait(1)
        self.play(Create(bars), Write(bars_caption))

        # 6. Shrinking Coefficients Animation
        lambda_tracker = ValueTracker(0)
        
        def update_bars_logic(mobjects):
            val = lambda_tracker.get_value()
            for i, line in enumerate(mobjects):
                # Higher index coefficients (noise) vanish faster in Lasso
                shrink_factor = 1.8 if i > 15 else 0.4
                current_h = max(0.001, initial_heights[i] - val * shrink_factor)
                line.stretch_to_fit_height(current_h, about_edge=DOWN)
                # Visual vanishing
                if current_h <= 0.05:
                    line.set_opacity(0)
                else:
                    line.set_opacity(1)

        bars.add_updater(update_bars_logic)
        
        lambda_label = MathTex(r"\lambda \uparrow").next_to(formula, DOWN).set_color(YELLOW)
        
        self.play(Write(lambda_label))
        # Using rate_functions.linear explicitly to avoid identifier issues
        self.play(
            lambda_tracker.animate.set_value(2.5), 
            run_time=5, 
            rate_func=rate_functions.linear
        )
        self.wait(2)
