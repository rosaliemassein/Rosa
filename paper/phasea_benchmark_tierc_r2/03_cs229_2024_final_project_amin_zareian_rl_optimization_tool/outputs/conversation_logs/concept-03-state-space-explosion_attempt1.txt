from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Define Colors and Rate Functions explicitly
        # This avoids issues with specific environment identifiers
        COLOR_BLUE_D = "#2b5c8a"
        COLOR_BLUE_E = "#1c3d5a"
        lin_rate = rate_functions.linear

        # 2. Formula Setup
        # Reference the formula provided in the prompt
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP, buff=0.5)
        
        # 3. Axes Setup
        # y-axis is Possible States (log scale). Visually, 2^(N^2) on log scale is a parabola N^2.
        # We use a linear axis scale (0-40) to represent the exponents (2^0 to 2^40).
        axes = Axes(
            x_range=[2, 7, 1],
            y_range=[0, 45, 10],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).shift(RIGHT * 2 + DOWN * 0.5)

        x_label_text = Text("Matrix Size (N)", font_size=20).next_to(axes.x_axis, DOWN)
        y_label_text = Text("States (log2 scale)", font_size=20).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)
        
        # Define the plotting function locally to avoid potential lambda issues
        def state_growth(val):
            return val**2

        graph = axes.plot(state_growth, x_range=[2, 4], color=RED)
        graph_label = MathTex(r"2^{N^2}", color=RED).next_to(graph, UR, buff=0.1)

        self.add(formula)
        self.play(Create(axes), Write(x_label_text), Write(y_label_text))
        self.play(Create(graph), Write(graph_label))
        self.wait(0.5)

        # 4. Table Graphic for 2x2 grid (16 rows)
        table_title = Text("Q-Table (2x2)", font_size=24).to_edge(LEFT, buff=0.5).shift(UP * 2.5)
        
        # Binary strings to represent state configurations
        rows = VGroup()
        for i in range(16):
            binary_val = bin(i)[2:].zfill(4)
            row_text = Text(binary_val, font_size=14, font="Courier")
            rows.add(row_text)
            
        rows.arrange(DOWN, buff=0.1).next_to(table_title, DOWN)
        table_box = SurroundingRectangle(rows, color=BLUE, buff=0.2)
        table_group = VGroup(table_title, rows, table_box)
        
        self.play(FadeIn(table_group))
        self.wait(1)

        # 5. Animate the growth to 5x5
        # The number of states becomes 2^25 = 33,554,432
        # The graph shoots up, and the table turns into a blur
        
        extended_graph = axes.plot(state_growth, x_range=[2, 6.3], color=RED)
        
        # Create "blur" lines to simulate high-speed scrolling rows
        blur_lines = VGroup()
        for _ in range(50):
            line_obj = Line(LEFT, RIGHT, color=COLOR_BLUE_E, stroke_width=1).set_width(1.0)
            blur_lines.add(line_obj)
        blur_lines.arrange(DOWN, buff=0.04).move_to(rows)

        # Define an updater for the scrolling effect
        def scroll_updater(m, dt):
            m.shift(UP * 10 * dt)
            if m.get_top()[1] > table_box.get_top()[1]:
                m.shift(DOWN * 1.8)

        new_table_title = Text("Q-Table (5x5)", font_size=24).move_to(table_title)
        
        self.play(
            Transform(table_title, new_table_title),
            FadeOut(rows),
            FadeIn(blur_lines),
            Transform(graph, extended_graph),
            graph_label.animate.move_to(axes.c2p(6.0, 40)),
            run_time=4,
            rate_func=lin_rate
        )
        
        blur_lines.add_updater(scroll_updater)
        
        # A solid blur represents the over 33 million configurations
        solid_blur_rect = Rectangle(
            width=table_box.get_width() * 0.95,
            height=table_box.get_height() * 0.95,
            fill_color=COLOR_BLUE_D,
            fill_opacity=0.7,
            stroke_width=0
        ).move_to(table_box)
        
        self.play(FadeIn(solid_blur_rect), run_time=2)
        self.wait(2)
        
        blur_lines.remove_updater(scroll_updater)
        
        # 6. Conclusion
        conclusion_text = Text("Exponential State Space Explosion!", color=YELLOW, font_size=32).to_edge(DOWN, buff=0.3)
        self.play(Write(conclusion_text))
        self.wait(2)
