from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Create a 4x4 VGroup of Square objects
        squares = VGroup(*[
            Square(side_length=1.0, fill_opacity=1).set_stroke(color=WHITE, width=2)
            for _ in range(16)
        ]).arrange_in_grid(4, 4, buff=0)

        # 2. Initially color the left half red and the right half blue
        # Columns 0, 1 -> RED | Columns 2, 3 -> BLUE
        for i in range(16):
            col = i % 4
            if col < 2:
                squares[i].set_fill(RED)
            else:
                squares[i].set_fill(BLUE)

        # Helper to find adjacent indices (Right and Bottom neighbors)
        def get_adjacencies():
            adj = []
            for r in range(4):
                for c in range(4):
                    idx = r * 4 + c
                    # Horizontal adjacency (check right neighbor)
                    if c < 3:
                        adj.append((idx, idx + 1, "h"))
                    # Vertical adjacency (check bottom neighbor)
                    if r < 3:
                        adj.append((idx, idx + 4, "v"))
            return adj

        adj_list = get_adjacencies()

        # 3. Dynamic green border lines
        def get_border_lines():
            lines = VGroup()
            for i1, i2, orient in adj_list:
                # Compare colors using numpy
                color1 = squares[i1].get_fill_color()
                color2 = squares[i2].get_fill_color()
                if not np.allclose(color1, color2):
                    if orient == "h":
                        # Line between column neighbors
                        line = Line(
                            squares[i1].get_corner(UR),
                            squares[i1].get_corner(DR),
                            color=GREEN,
                            stroke_width=8
                        )
                    else:
                        # Line between row neighbors
                        line = Line(
                            squares[i1].get_corner(DL),
                            squares[i1].get_corner(DR),
                            color=GREEN,
                            stroke_width=8
                        )
                    lines.add(line)
            return lines

        borders = always_redraw(get_border_lines)

        # 4. Adjacency Counter
        # Using DecimalNumber with 0 decimals as a replacement for Integer to ensure compatibility
        counter_num = DecimalNumber(0, num_decimal_places=0).scale(1.2)
        counter_num.to_edge(UP, buff=0.5).shift(RIGHT * 2)
        counter_label = Text("Border Length: ", font_size=32).next_to(counter_num, LEFT)
        
        def update_counter(obj):
            # The count is simply the number of lines in our borders VGroup
            obj.set_value(len(borders))
        
        counter_num.add_updater(update_counter)

        # 5. Formula display
        formula = MathTex(r"B = \sum_{i,j \in \text{Neighbors}} [P_i \neq P_j]").to_edge(DOWN, buff=0.5)

        # Add initial state to scene
        self.add(squares, borders, counter_label, counter_num, formula)
        self.wait(1)

        # 6. Animate flipping to checkerboard
        # Logic: In a checkerboard, Square(r, c) is RED if (r+c) is even
        for i in range(16):
            r, c = i // 4, i % 4
            target_is_red = ((r + c) % 2 == 0)
            
            # Determine current state to see if it needs a flip
            # Initial state: Red if c < 2
            # We compare with target to decide if we play animation
            target_color = RED if target_is_red else BLUE
            
            # We check the fill color of the square mobject
            # RED in Manim is usually [0.992, 0.203, 0.231]
            if not np.allclose(squares[i].get_fill_color(), color_to_rgb(target_color)):
                self.play(
                    squares[i].animate.set_fill(target_color),
                    run_time=0.25
                )

        self.wait(2)
