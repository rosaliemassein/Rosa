from manim import *

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Formula reference from the prompt
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_edge(UP)
        self.add(formula)

        # 2. Create Axes
        # x-axis is Matrix Size (N from 2 to 6).
        # y-axis is log-scale States. We represent this by plotting y = N^2 
        # and labeling the y-axis with powers of 2 (2^10, 2^20, etc.).
        axes = Axes(
            x_range=[2, 7, 1],
            y_range=[0, 45, 10],
            x_length=7,
            y_length=5,
            axis_config={"include_numbers": True},
        ).to_edge(RIGHT, buff=0.5).shift(DOWN * 0.5)

        # Manually create y-axis labels to indicate log scale (base 2)
        y_labels = VGroup()
        for exponent in [10, 20, 30, 40]:
            label = MathTex(f"2^{{{exponent}}}", font_size=24).next_to(axes.c2p(2, exponent), LEFT)
            y_labels.add(label)
        
        x_label = axes.get_x_axis_label("N").scale(0.8)
        y_label = axes.get_y_axis_label(r"\text{States}").scale(0.8)
        self.add(axes, x_label, y_label, y_labels)

        # 3. Plot the function: since y-axis is log2, we plot f(t) = t^2 
        # to represent States = 2^(N^2)
        graph = axes.plot(
            lambda t: t**2,
            x_range=[2, 5.0],
            color=YELLOW
        )

        # 4. Table Graphic for a small grid (2x2 grid = 16 rows)
        table_title = Text("Q-Table (2x2)", font_size=20)
        # Represent rows as blue lines
        rows = VGroup(*[
            Line(LEFT, RIGHT, color=BLUE, stroke_width=2).set_width(1.5)
            for _ in range(16)
        ]).arrange(DOWN, buff=0.1)
        
        table_box = SurroundingRectangle(VGroup(table_title, rows), color=WHITE, buff=0.2)
        table_group = VGroup(table_box, table_title, rows).to_edge(LEFT, buff=0.5)

        self.add(table_group)
        self.play(Create(graph), run_time=3, rate_func=rate_functions.linear)

        # 5. Animate transition as grid size increases to 5x5 (2^25 states)
        new_title = Text("States: 2^25", font_size=20).move_to(table_title)
        
        # Dense lines to represent many rows
        dense_rows = VGroup(*[
            Line(LEFT, RIGHT, color=BLUE, stroke_width=1).set_width(1.5)
            for _ in range(60)
        ]).arrange(DOWN, buff=0.03).move_to(rows)
        
        # Blur effect for the table (solid blue rectangle)
        blur_fill = Rectangle(
            width=table_box.get_width() - 0.1,
            height=table_box.get_height() - 0.1,
            fill_color=BLUE,
            fill_opacity=0.8,
            stroke_width=0
        ).move_to(table_box)

        # Graph continues shooting off the top
        extra_graph = axes.plot(
            lambda t: t**2,
            x_range=[5.0, 6.5],
            color=YELLOW
        )

        # Execute animations
        self.play(
            ReplacementTransform(rows, dense_rows),
            Transform(table_title, new_title),
            run_time=1
        )
        
        self.play(
            dense_rows.animate.shift(UP * 10),
            FadeIn(blur_fill),
            Create(extra_graph),
            run_time=2,
            rate_func=rate_functions.ease_in_expo
        )

        # 6. Final warning/explosion indicator
        warning_mark = Text("!", font_size=120, color=RED).move_to(table_group)
        self.play(
            Write(warning_mark),
            table_group.animate.set_color(RED),
            axes.animate.set_color(RED)
        )
        
        self.wait(2)
