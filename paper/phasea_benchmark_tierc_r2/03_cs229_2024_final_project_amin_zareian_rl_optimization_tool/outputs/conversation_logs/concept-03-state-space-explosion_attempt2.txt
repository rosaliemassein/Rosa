from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Setup Axes with Log Scale (Base 10)
        # Total states = 2^(N*N). Log10(States) = (N*N) * log10(2)
        axes = Axes(
            x_range=[2, 6.5, 1],
            y_range=[0, 10, 2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True},
            x_axis_config={"numbers_to_include": [2, 3, 4, 5, 6]},
        ).to_edge(LEFT, buff=0.7)
        
        x_label = axes.get_x_axis_label("Matrix Size (N)", edge=DOWN, direction=DOWN).scale(0.7)
        y_label = Text("Possible States (log scale)", font_size=24).rotate(90*DEGREES).next_to(axes, LEFT)
        
        # Initial graph from N=2 to N=3
        graph = axes.plot(
            lambda t: (t**2) * np.log10(2), 
            x_range=[2, 3], 
            color=YELLOW
        )
        
        formula = MathTex(r"\text{Total States} = 2^{(N \times N)}").to_corner(UL)

        # 2. Initial Graphics: 2x2 Grid and Q-Table
        grid_2x2 = VGroup(*[
            Square(side_length=0.4, fill_opacity=0.5, color=BLUE, stroke_width=2) 
            for _ in range(4)
        ]).arrange_in_grid(2, 2, buff=0.1)
        grid_2x2.to_corner(UR, buff=1.5)
        grid_label = MathTex(r"2 \times 2").next_to(grid_2x2, UP)

        table_box = Rectangle(height=2.5, width=1.8, color=WHITE).next_to(grid_2x2, DOWN, buff=1)
        table_rows = VGroup(*[
            Line(LEFT*0.7, RIGHT*0.7, stroke_width=2, color=GRAY) 
            for _ in range(10)
        ]).arrange(DOWN, buff=0.18).move_to(table_box)
        table_title = Text("Q-Table", font_size=20).next_to(table_box, UP, buff=0.1)
        row_count_label = Text("16 Rows", font_size=18).next_to(table_box, DOWN)

        # Initial Scene Setup
        self.add(axes, x_label, y_label, formula)
        self.play(
            Create(graph),
            Create(grid_2x2),
            Write(grid_label),
            Create(table_box),
            Create(table_rows),
            Write(table_title),
            Write(row_count_label)
        )
        self.wait(1)

        # 3. Expansion Animation (N=2 to N=6)
        # N=5 results in 2^25 ~ 33M states. Log10(2^25) ~ 7.5
        # N=6 results in 2^36 ~ 68B states. Log10(2^36) ~ 10.8 (Shoots off screen)
        
        grid_5x5 = VGroup(*[
            Square(side_length=0.12, fill_opacity=0.5, color=BLUE, stroke_width=1) 
            for _ in range(25)
        ]).arrange_in_grid(5, 5, buff=0.04).move_to(grid_2x2)
        
        grid_label_5x5 = MathTex(r"5 \times 5").move_to(grid_label)
        
        # Scrolling effect for the table
        blur_lines = VGroup(*[
            Line(LEFT*0.7, RIGHT*0.7, stroke_width=0.5, stroke_opacity=0.4, color=WHITE) 
            for _ in range(60)
        ]).arrange(DOWN, buff=0.04).move_to(table_box)
        
        solid_blur = Rectangle(
            height=2.5, width=1.8, 
            fill_color=BLUE, fill_opacity=0.4, 
            stroke_width=0
        ).move_to(table_box)

        # Animation sequence
        self.play(
            ReplacementTransform(grid_2x2, grid_5x5),
            ReplacementTransform(grid_label, grid_label_5x5),
            ReplacementTransform(table_rows, blur_lines),
            FadeIn(solid_blur),
            graph.animate.become(axes.plot(lambda t: (t**2) * np.log10(2), x_range=[2, 6], color=YELLOW)),
            row_count_label.animate.set_text("33,554,432 Rows"),
            run_time=4,
            rate_func=rate_functions.linear
        )
        
        # Continuous scroll animation
        def scroll_updater(mobj, dt):
            mobj.shift(UP * 2 * dt)
            if mobj.get_bottom()[1] > table_box.get_top()[1]:
                mobj.set_y(table_box.get_bottom()[1])

        blur_lines.add_updater(scroll_updater)
        
        # 4. Final Narration/Conclusion
        narration_text = Text(
            "Training time skyrockets exponentially.",
            font_size=28,
            color=RED
        ).to_edge(DOWN, buff=0.5)
        
        self.play(Write(narration_text))
        self.wait(3)
        
        # Cleanup
        blur_lines.remove_updater(scroll_updater)
        self.wait()
