from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Setup Axes and Vectors
        axes = Axes(
            x_range=[-0.5, 2.5, 1],
            y_range=[-0.5, 2.5, 1],
            x_length=4,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=1)

        # Vector A (Static) at 20 degrees
        angle_a = 20 * DEGREES
        vec_a_end = axes.c2p(2 * np.cos(angle_a), 2 * np.sin(angle_a))
        vec_a = Line(axes.get_origin(), vec_a_end, color=RED, stroke_width=6)
        label_a = MathTex(r"\mathbf{A}", color=RED).next_to(vec_a_end, UR, buff=0.1)

        # ValueTracker for Vector B's angle
        theta_tracker = ValueTracker(90 * DEGREES)

        # Vector B (Dynamic)
        vec_b = always_redraw(lambda: Line(
            axes.get_origin(),
            axes.c2p(2 * np.cos(theta_tracker.get_value()), 2 * np.sin(theta_tracker.get_value())),
            color=BLUE,
            stroke_width=6
        ))
        label_b = always_redraw(lambda: MathTex(r"\mathbf{B}", color=BLUE).next_to(vec_b.get_end(), UR, buff=0.1))

        # 2. Angle and Similarity Labels
        # Using Angle class which is part of Manim
        angle_arc = always_redraw(lambda: Angle(
            vec_a, vec_b, radius=0.6, color=YELLOW, other_angle=False
        ))
        
        theta_label = always_redraw(lambda: MathTex(r"\theta", color=YELLOW).move_to(
            Angle(vec_a, vec_b, radius=0.9, other_angle=False).point_from_proportion(0.5)
        ))

        formula = MathTex(r"\text{similarity} = \cos(\theta) = ", font_size=32).to_edge(DOWN, buff=1).shift(LEFT * 2)
        sim_value = always_redraw(lambda: DecimalNumber(
            np.cos(theta_tracker.get_value() - angle_a),
            num_decimal_places=3,
            color=YELLOW
        ).scale(0.8).next_to(formula, RIGHT))

        # 3. Similarity Matrix (5x5 Grid)
        matrix_group = VGroup()
        for i in range(25):
            sq = Square(side_length=0.5, stroke_width=1, stroke_color=WHITE)
            sq.set_fill(BLUE, opacity=0.2)
            matrix_group.add(sq)
        
        matrix_group.arrange_in_grid(rows=5, cols=5, buff=0.05)
        matrix_group.to_edge(RIGHT, buff=1.2)
        matrix_label = MathTex(r"\text{Similarity Matrix}").scale(0.7).next_to(matrix_group, UP)

        # Link a specific cell to the interactive vectors
        target_cell = matrix_group[12] # Center cell (2,2)
        
        def update_cell_color(m):
            # Calculate similarity
            val = np.cos(theta_tracker.get_value() - angle_a)
            # Clamp value between 0 and 1 for color interpolation
            alpha = np.clip(val, 0, 1)
            # Interpolate between dark blue and bright yellow
            new_color = interpolate_color(BLUE_D, YELLOW, alpha)
            m.set_fill(new_color, opacity=0.3 + 0.7 * alpha)
            m.set_stroke(new_color, width=1 + 2 * alpha)

        target_cell.add_updater(update_cell_color)

        # 4. Animation sequence
        self.add(axes, vec_a, label_a)
        self.play(Create(vec_b), Write(label_b))
        self.play(Create(angle_arc), Write(theta_label))
        self.play(Write(formula), Write(sim_value))
        self.play(Create(matrix_group), Write(matrix_label))
        self.wait(1)

        # Rotate Vector B to align with A
        self.play(
            theta_tracker.animate.set_value(angle_a),
            run_time=4,
            rate_func=rate_functions.smooth
        )
        self.wait(1)

        # Rotate Vector B away to 110 degrees
        self.play(
            theta_tracker.animate.set_value(angle_a + 90 * DEGREES),
            run_time=3,
            rate_func=rate_functions.smooth
        )
        self.wait(2)

        # Cleanup
        target_cell.remove_updater(update_cell_color)
