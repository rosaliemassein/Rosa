from manim import *
import numpy as np

class ConceptCosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Setup Data and ValueTrackers
        theta_tracker = ValueTracker(60)  # angle in degrees
        
        # 2. Left Side: Geometric Intuition (Axes & Vectors)
        axes = Axes(
            x_range=[-0.2, 1.2, 0.5],
            y_range=[-0.2, 1.2, 0.5],
            x_length=4,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=1)
        
        labels = axes.get_axis_labels(x_label="Keywords", y_label="Reviewers")
        origin = axes.get_origin()
        
        # Vector A (Fixed)
        vec_a_end = axes.c2p(1, 0, 0)
        vec_a = Arrow(origin, vec_a_end, buff=0, color=BLUE)
        label_a = MathTex(r"\mathbf{A}", color=BLUE).next_to(vec_a.get_end(), RIGHT)
        
        # Vector B (Rotating)
        vec_b = always_redraw(lambda: Arrow(
            origin,
            axes.c2p(
                np.cos(theta_tracker.get_value() * DEGREES),
                np.sin(theta_tracker.get_value() * DEGREES),
                0
            ),
            buff=0,
            color=RED
        ))
        
        label_b = always_redraw(lambda: MathTex(r"\mathbf{B}", color=RED).next_to(vec_b.get_end(), UP))
        
        # Angle Arc and Theta Label
        def get_angle_arc():
            # Use lines to define the angle
            l1 = Line(origin, axes.c2p(1, 0, 0))
            l2 = Line(origin, vec_b.get_end())
            return Angle(l1, l2, radius=0.7, other_angle=False, color=WHITE)
        
        angle_arc = always_redraw(get_angle_arc)
        theta_tex = always_redraw(lambda: MathTex(r"\theta").move_to(
            get_angle_arc().point_from_proportion(0.5) + 0.3 * (get_angle_arc().point_from_proportion(0.5) - origin)
        ))

        # Similarity Value Text
        sim_val_label = MathTex(r"\text{Similarity} = \cos(\theta) = ", font_size=36).to_edge(UP, buff=0.5).shift(RIGHT * 2)
        sim_decimal = always_redraw(lambda: DecimalNumber(
            np.cos(theta_tracker.get_value() * DEGREES),
            num_decimal_places=2,
            color=YELLOW
        ).next_to(sim_val_label, RIGHT))

        # 3. Right Side: Similarity Matrix
        matrix_squares = VGroup()
        rows, cols = 5, 5
        for i in range(rows):
            for j in range(cols):
                sq = Square(side_length=0.7, stroke_width=1, color=WHITE)
                sq.set_fill(BLUE, opacity=0.1)
                matrix_squares.add(sq)
        
        matrix_squares.arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=1)
        
        # Target cell that tracks the angle
        target_cell_index = 12 # center
        
        def update_cells(m):
            current_cos = np.cos(theta_tracker.get_value() * DEGREES)
            for idx, cell in enumerate(m):
                if idx == target_cell_index:
                    val = current_cos
                else:
                    # Deterministic pseudo-randomness for other cells
                    val = (np.sin(idx * 1.5) + 1) / 2 * 0.6
                
                # Manual color interpolation since interpolate_color might be tricky in some versions
                # Map 0 -> Dark Blue, 1 -> Yellow/White
                color_val = max(0, min(1, val))
                cell.set_fill(interpolate_color(BLUE_E, YELLOW, color_val), opacity=max(0.2, color_val))

        matrix_squares.add_updater(update_cells)

        # 4. Animation Sequence
        self.add(axes, labels)
        self.play(Create(vec_a), Write(label_a))
        self.play(Create(vec_b), Write(label_b))
        self.play(Create(angle_arc), Write(theta_tex))
        self.wait(1)

        self.play(Write(sim_val_label), Write(sim_decimal))
        self.play(theta_tracker.animate.set_value(0), run_time=2)
        self.wait(0.5)
        self.play(theta_tracker.animate.set_value(90), run_time=2)
        self.wait(0.5)

        self.play(FadeIn(matrix_squares))
        
        # Formula display
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=32
        ).next_to(matrix_squares, DOWN, buff=0.5)
        self.play(Write(formula))

        # Show alignment and matrix response
        self.play(theta_tracker.animate.set_value(15), run_time=3)
        self.wait(1)
        self.play(theta_tracker.animate.set_value(80), run_time=2)
        self.wait(2)
