from manim import *
import numpy as np

class CosineSimilarityConcept(Scene):
    def construct(self):
        # 1. Setup - split screen intuition
        # Use a ValueTracker for the angle theta in radians
        theta_tracker = ValueTracker(1.0) 
        
        # --- Left Side: Geometric Intuition ---
        # Create axes for vector visualization
        axes = Axes(
            x_range=[0, 2, 1], 
            y_range=[0, 2, 1], 
            x_length=3.5, 
            y_length=3.5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        # Vector A (Fixed on the x-axis)
        a_end_point = axes.c2p(1.5, 0)
        vector_a = Arrow(axes.c2p(0, 0), a_end_point, buff=0, color=BLUE)
        label_a = MathTex(r"\mathbf{A}", color=BLUE).next_to(a_end_point, DOWN)
        
        # Vector B (Moving based on ValueTracker)
        def get_b_endpoint():
            current_angle = theta_tracker.get_value()
            return axes.c2p(1.5 * np.cos(current_angle), 1.5 * np.sin(current_angle))
            
        def vector_b_updater(mob_to_update):
            new_arrow = Arrow(axes.c2p(0, 0), get_b_endpoint(), buff=0, color=RED)
            mob_to_update.become(new_arrow)
            
        vector_b = Arrow(axes.c2p(0, 0), get_b_endpoint(), buff=0, color=RED)
        vector_b.add_updater(vector_b_updater)
        
        def label_b_updater(mob_to_update):
            mob_to_update.move_to(get_b_endpoint() + 0.3 * UR)
            
        label_b = MathTex(r"\mathbf{B}", color=RED)
        label_b.add_updater(label_b_updater)
        
        # Angle Arc representing theta
        def arc_updater(mob_to_update):
            current_ang = theta_tracker.get_value()
            # Ensure angle is valid for Arc creation
            safe_ang = max(0.001, current_ang)
            new_arc = Arc(radius=0.6, start_angle=0, angle=safe_ang, color=WHITE)
            new_arc.move_arc_center_to(axes.c2p(0,0))
            mob_to_update.become(new_arc)
            
        arc_visual = Arc(radius=0.6, start_angle=0, angle=1.0, color=WHITE)
        arc_visual.move_arc_center_to(axes.c2p(0,0))
        arc_visual.add_updater(arc_updater)

        # --- Top: Similarity Value Display ---
        similarity_decimal = DecimalNumber(0, num_decimal_places=2)
        def decimal_updater(mob_to_update):
            mob_to_update.set_value(np.cos(theta_tracker.get_value()))
        similarity_decimal.add_updater(decimal_updater)
        
        similarity_label = MathTex(r"\text{Similarity} = \cos(\theta) = ")
        similarity_vgroup = VGroup(similarity_label, similarity_decimal).arrange(RIGHT).to_edge(UP)

        # --- Right Side: Similarity Matrix Grid ---
        matrix_grid = VGroup()
        for _ in range(25):
            square_cell = Square(side_length=0.6, stroke_color=GRAY, stroke_width=1)
            square_cell.set_fill(BLACK, opacity=1.0)
            matrix_grid.add(square_cell)
        matrix_grid.arrange_in_grid(rows=5, cols=5, buff=0.1).to_edge(RIGHT, buff=0.5)
        
        # Pick one cell (e.g., center) to glow based on vector alignment
        target_cell_index = 12
        def cell_glow_updater(mob_to_update):
            current_cos = np.cos(theta_tracker.get_value())
            # Map cosine result (0 to 1) to opacity of yellow fill
            mob_to_update.set_fill(YELLOW, opacity=max(0, current_cos))
            
        matrix_grid[target_cell_index].add_updater(cell_glow_updater)

        # --- Animation Sequence ---
        # Initial display
        self.add(axes, vector_a, label_a, vector_b, label_b, arc_visual, similarity_vgroup, matrix_grid)
        self.wait(1)
        
        # 1. Rotate B to align with A (Angle -> 0, Similarity -> 1.0)
        self.play(theta_tracker.animate.set_value(0.0), run_time=3)
        self.wait(1)
        
        # 2. Rotate B away from A (Angle -> PI/2, Similarity -> 0.0)
        self.play(theta_tracker.animate.set_value(PI/2), run_time=3)
        self.wait(1)
        
        # 3. Show final formal formula
        cosine_formula = MathTex(
            r"\text{similarity} = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}"
        ).scale(0.8).next_to(matrix_grid, DOWN, buff=0.5)
        self.play(Write(cosine_formula))
        self.wait(2)
