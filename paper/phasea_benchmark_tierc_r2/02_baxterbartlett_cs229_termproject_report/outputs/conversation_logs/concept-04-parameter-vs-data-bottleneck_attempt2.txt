from manim import *
import numpy as np

class ParameterVsDataBottleneck(Scene):
    def construct(self):
        # Narration
        narration_text = Text(
            "With only 27 points, the NN (24 params) overfits.",
            font_size=24, color=YELLOW
        ).to_edge(UP)
        
        # Data Setup
        np.random.seed(42)
        x_data = np.linspace(-3, 3, 27)
        y_true = 0.5 * x_data
        y_noise = np.random.normal(0, 0.4, size=27)
        y_data = y_true + y_noise
        
        # Axes Setup
        left_axes = Axes(
            x_range=[-4, 4, 1], y_range=[-3, 3, 1], 
            x_length=5, y_length=4,
            axis_config={"include_tip": False}
        ).to_edge(LEFT, buff=0.5).shift(UP * 0.2)
        
        right_axes = Axes(
            x_range=[-4, 4, 1], y_range=[-3, 3, 1], 
            x_length=5, y_length=4,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5).shift(UP * 0.2)
        
        # Labels
        nn_title = Text("Neural Network", color=RED).scale(0.6).next_to(left_axes, UP)
        lwr_title = Text("LWR", color=GREEN).scale(0.6).next_to(right_axes, UP)
        high_variance = Text("High Variance", color=RED).scale(0.5).next_to(left_axes, DOWN, buff=0.1)
        robust_label = Text("Robust / Smooth", color=GREEN).scale(0.5).next_to(right_axes, DOWN, buff=0.1)

        # Data Points
        dots_left = VGroup(*[Dot(left_axes.c2p(x, y), radius=0.04, color=WHITE) for x, y in zip(x_data, y_data)])
        dots_right = VGroup(*[Dot(right_axes.c2p(x, y), radius=0.04, color=WHITE) for x, y in zip(x_data, y_data)])
        
        # NN: Highly wiggly curve passing through every point
        nn_curve = left_axes.plot_line_graph(
            x_values=x_data, 
            y_values=y_data, 
            add_vertex_dots=False, 
            line_config={"stroke_width": 2, "color": RED}
        )
        
        # LWR: Smoother, linear path
        lwr_curve = right_axes.plot(lambda x: 0.5 * x, color=GREEN, stroke_width=3)
        
        # Formula
        formula = MathTex(r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}").scale(0.6).to_edge(DOWN, buff=1.8)

        # Bar Chart for MSE
        mse_nn_val = 1.21
        mse_lwr_val = 0.85
        scale_h = 1.2
        
        bar_nn = Rectangle(width=0.8, height=mse_nn_val * scale_h, fill_opacity=0.8, fill_color=RED, stroke_color=WHITE)
        bar_lwr = Rectangle(width=0.8, height=mse_lwr_val * scale_h, fill_opacity=0.8, fill_color=GREEN, stroke_color=WHITE)
        
        bars = VGroup(bar_nn, bar_lwr).arrange(RIGHT, buff=1.5).to_edge(DOWN, buff=0.4)
        bar_nn.align_to(bars, DOWN)
        bar_lwr.align_to(bars, DOWN)
        
        label_nn = MathTex("1.21").scale(0.6).next_to(bar_nn, UP, buff=0.1)
        label_lwr = MathTex("0.85").scale(0.6).next_to(bar_lwr, UP, buff=0.1)
        text_nn = Text("NN MSE", font_size=16).next_to(bar_nn, DOWN, buff=0.1)
        text_lwr = Text("LWR MSE", font_size=16).next_to(bar_lwr, DOWN, buff=0.1)

        # Animation
        self.play(Write(narration_label))
        self.play(Create(left_axes), Create(right_axes))
        self.play(Create(dots_left), Create(dots_right))
        self.wait(0.5)
        
        self.play(
            Create(nn_curve), 
            Create(lwr_curve),
            Write(nn_title),
            Write(lwr_title),
            run_time=2
        )
        self.play(Write(high_variance), Write(robust_label))
        self.wait(1)
        
        self.play(Write(formula))
        self.play(
            Create(bar_nn),
            Create(bar_lwr),
            Write(label_nn),
            Write(label_lwr),
            Write(text_nn),
            Write(text_lwr)
        )
        
        self.wait(3)
