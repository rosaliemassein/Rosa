from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Formula Header
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=36)
        formula.to_edge(UP, buff=0.5)
        self.add(formula)

        # 2. Top Diagram: Inverse Mapping Concept
        # Input Box
        input_rect = RoundedRectangle(height=2, width=5, color=BLUE)
        input_title = Text("Input", font_size=20, color=BLUE).next_to(input_rect, UP, buff=-0.3)
        input_content = Text("Desired Behavior\n(Rise Time, SSE, Stability)", font_size=22).move_to(input_rect.get_center())
        input_box = VGroup(input_rect, input_title, input_content)

        # Output Box
        output_rect = RoundedRectangle(height=2, width=4, color=GREEN)
        output_title = Text("Output", font_size=20, color=GREEN).next_to(output_rect, UP, buff=-0.3)
        output_content = Text("PID Constants", font_size=22).move_to(output_rect.get_center())
        output_box = VGroup(output_rect, output_title, output_content)

        # Positioning
        input_box.move_to(LEFT * 3.5 + UP * 1.5)
        output_box.move_to(RIGHT * 3.5 + UP * 1.5)

        # Arrow
        arrow = Arrow(input_box.get_right(), output_box.get_left(), buff=0.2)
        
        diagram = VGroup(input_box, output_box, arrow)
        
        self.play(FadeIn(input_box), FadeIn(output_box), GrowArrow(arrow))
        self.wait(1)

        # "Flips the script" - Rotate 180 degrees
        # Using [0,0,1] for axis to avoid 'OUT' identifier issues
        self.play(Rotate(diagram, angle=PI, axis=np.array([0, 0, 1])), run_time=1.5)
        self.wait(1)

        # 3. Bottom Diagram: Motor Velocity Plot
        axes = Axes(
            x_range=[0, 8, 1],
            y_range=[0, 1.5, 0.5],
            x_length=8,
            y_length=3,
            axis_config={"include_tip": True}
        ).shift(DOWN * 2)
        
        y_axis_label = axes.get_y_axis_label(Text("RPM", font_size=18), edge=LEFT, direction=LEFT, buff=0.3)
        x_axis_label = axes.get_x_axis_label(Text("Time", font_size=18), edge=DOWN, direction=DOWN, buff=0.3)
        
        # Damped oscillation curve: 1 - e^(-0.5t) * cos(3t)
        def motor_curve(t):
            return 1 - np.exp(-0.5 * t) * np.cos(3 * t)

        curve = axes.plot(motor_curve, x_range=[0, 8], color=YELLOW)
        
        # Setpoint line at y=1
        setpoint = DashedLine(axes.c2p(0, 1), axes.c2p(8, 1), color=WHITE)
        setpoint_label = Text("Setpoint", font_size=16).next_to(setpoint, RIGHT, buff=0.1)

        self.play(Create(axes), Write(y_axis_label), Write(x_axis_label))
        self.play(Create(setpoint), Write(setpoint_label))
        self.play(Create(curve), run_time=2.5)

        # 4. Braces and Performance Metrics
        # Rise Time: Approximately where the curve first reaches the setpoint (t ~ 0.52)
        rt_line = Line(axes.c2p(0, 0), axes.c2p(0.52, 0)).set_visible(False)
        rise_time_brace = Brace(rt_line, direction=DOWN, buff=0.1)
        rise_time_label = rise_time_brace.get_text("Rise Time", font_size=18).set_color(RED)

        # Steady State Error: The gap at the end
        # At t=7.5, the curve is very close to 1, we exaggerate for visibility if needed
        # We create a small vertical line between setpoint and curve
        sse_line = Line(axes.c2p(7.5, 1), axes.c2p(7.5, motor_curve(7.5)), color=ORANGE)
        # Use a brace on the vertical line
        sse_brace = Brace(sse_line, direction=RIGHT, buff=0.1)
        sse_label = sse_brace.get_text("Steady State Error", font_size=18).set_color(ORANGE)

        self.play(Create(rise_time_brace), Write(rise_time_label))
        self.wait(0.5)
        self.play(Create(sse_brace), Write(sse_label))

        self.wait(3)
