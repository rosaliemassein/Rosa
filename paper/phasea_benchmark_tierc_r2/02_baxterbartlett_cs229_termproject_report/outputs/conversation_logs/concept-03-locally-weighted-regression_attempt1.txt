from manim import *
import numpy as np

class LWRConcept(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6,
            axis_config={"include_tip": True}
        )
        axes_labels = axes.get_axis_labels(x_label="x", y_label="y")
        
        # 2. Generate Data (Sine wave with noise)
        np.random.seed(42)
        x_data = np.linspace(-4, 4, 50)
        y_data = np.sin(x_data) + np.random.normal(0, 0.15, size=len(x_data))
        
        dots = VGroup(*[
            Dot(axes.c2p(x, y), radius=0.06, color=GRAY)
            for x, y in zip(x_data, y_data)
        ])
        
        # 3. Query Point and Tracker
        query_x_tracker = ValueTracker(-3.5)
        tau = 0.6  # Bandwidth parameter
        
        query_dot = Dot(color=WHITE, z_index=10).scale(1.2)
        query_dot.add_updater(lambda m: m.move_to(axes.c2p(query_x_tracker.get_value(), np.sin(query_x_tracker.get_value()))))
        
        query_label = Text("Query Point", font_size=20).add_updater(lambda m: m.next_to(query_dot, UP, buff=0.2))

        # 4. Weight Field (Radial Gradient Visual)
        weight_field = Circle(radius=1.5, stroke_width=0, fill_opacity=0.15)
        weight_field.set_color(YELLOW)
        weight_field.add_updater(lambda m: m.move_to(query_dot.get_center()))

        # 5. Local Line Mobject
        local_line = Line(color=RED, stroke_width=6)

        def get_lwr_params(x_q):
            # Weights based on distance to query point
            weights = np.exp(-((x_data - x_q)**2) / (2 * tau**2))
            
            # Weighted Linear Regression: y = theta0 + theta1 * x
            # X matrix [1, x]
            X = np.stack([np.ones(len(x_data)), x_data], axis=1)
            W = np.diag(weights)
            
            # Normal Equation: theta = (X^T W X)^-1 X^T W y
            try:
                XTW = X.T @ W
                theta = np.linalg.inv(XTW @ X) @ XTW @ y_data
                return theta, weights
            except np.linalg.LinAlgError:
                return np.array([0, 0]), weights

        def update_local_elements(line):
            xq = query_x_tracker.get_value()
            theta, weights = get_lwr_params(xq)
            
            # Update dot appearances based on weights
            for i, dot in enumerate(dots):
                w = weights[i]
                # Map weight to color and opacity
                dot.set_color(interpolate_color(GRAY, YELLOW, w))
                dot.set_opacity(max(0.2, w))
                if w > 0.5:
                    dot.set_stroke(WHITE, width=w*2)
                else:
                    dot.set_stroke(width=0)

            # Update line geometry
            # Line centered at xq, y = theta0 + theta1 * xq
            yq = theta[0] + theta[1] * xq
            start_x = xq - 0.7
            end_x = xq + 0.7
            start_y = theta[0] + theta[1] * start_x
            end_y = theta[0] + theta[1] * end_x
            
            line.set_points_as_corners([
                axes.c2p(start_x, start_y),
                axes.c2p(end_x, end_y)
            ])

        local_line.add_updater(update_local_elements)

        # 6. Formula
        formula = MathTex(
            r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)",
            font_size=32
        ).to_corner(UL).shift(RIGHT * 0.5)

        # Build Scene
        self.add(axes, axes_labels)
        self.play(Create(dots), run_time=2)
        self.wait(0.5)
        
        self.play(FadeIn(query_dot), FadeIn(query_label))
        self.play(Write(formula))
        self.play(FadeIn(weight_field))
        self.play(Create(local_line))
        self.wait(1)

        # Animation: Sweep across the data
        self.play(
            query_x_tracker.animate.set_value(3.5),
            run_time=8,
            rate_func=linear
        )
        self.wait(1)
        
        # Return to center
        self.play(
            query_x_tracker.animate.set_value(0),
            run_time=3
        )
        self.wait(2)
