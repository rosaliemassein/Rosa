from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Diagram Setup
        # Create 'Input' box labeled 'Desired Behavior'
        input_rect = Rectangle(height=2.5, width=4.5, color=BLUE)
        input_title = Text("Input: Desired Behavior", font_size=24, color=BLUE).shift(UP * 0.5)
        input_metrics = Text("Rise Time, SSE, Stability", font_size=18).next_to(input_title, DOWN, buff=0.3)
        input_box = VGroup(input_rect, input_title, input_metrics)

        # Create 'Output' box labeled 'PID Constants'
        output_rect = Rectangle(height=2.5, width=4.5, color=RED)
        output_title = Text("Output: PID Constants", font_size=24, color=RED)
        output_box = VGroup(output_rect, output_title)

        # Positioning
        diagram_group = VGroup(input_box, output_box).arrange(RIGHT, buff=2).to_edge(UP, buff=0.5)
        arrow = Arrow(input_box.get_right(), output_box.get_left(), buff=0.2, color=WHITE)
        
        full_diagram = VGroup(diagram_group, arrow)

        # Formula
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}").scale(0.8)
        formula.next_to(full_diagram, DOWN, buff=0.4)

        # Initial Display
        self.play(FadeIn(full_diagram), Write(formula))
        self.wait(1)

        # 2. "Flips the script" - Rotate the entire diagram 180 degrees
        # Using self.play(Rotate(...)) which is the standard Animation class
        self.play(Rotate(full_diagram, angle=PI), run_time=1.5)
        self.wait(1)

        # 3. Motor Velocity Plot
        axes = Axes(
            x_range=[0, 10, 2],
            y_range=[0, 6, 1],
            x_length=9,
            y_length=3,
            axis_config={"include_tip": True}
        ).to_edge(DOWN, buff=0.7)
        
        y_label = axes.get_y_axis_label("RPM", edge=LEFT, direction=LEFT, buff=0.3)
        x_label = axes.get_x_axis_label("Time", edge=DOWN, direction=DOWN, buff=0.3)
        
        setpoint_val = 4
        setpoint_line = DashedLine(
            axes.c2p(0, setpoint_val), 
            axes.c2p(10, setpoint_val), 
            color=GREY
        )

        # Damped oscillation function for the motor
        def motor_function(t):
            # y = setpoint * (1 - e^(-0.6t) * cos(2.5t)) - small steady state error offset
            return setpoint_val * (1 - np.exp(-0.6 * t) * np.cos(2.5 * t)) - 0.4 * (1 - np.exp(-0.1 * t))

        motor_curve = axes.plot(motor_function, x_range=[0, 10], color=YELLOW)

        self.play(Create(axes), FadeIn(y_label), FadeIn(x_label))
        self.play(Create(setpoint_line))
        self.play(Create(motor_curve), run_time=2.5)

        # 4. Annotations with Braces
        # Rise Time brace (from t=0 to roughly the first crossing of setpoint)
        rise_time_x = 0.65
        rise_brace_line = Line(axes.c2p(0, 0), axes.c2p(rise_time_x, 0))
        rise_brace = Brace(rise_brace_line, DOWN, color=BLUE_B)
        rise_label = rise_brace.get_text("Rise Time", font_size=18).set_color(BLUE_B)

        # Steady State Error brace (at the end of the plot)
        final_x = 10
        final_y = motor_function(final_x)
        sse_brace_line = Line(axes.c2p(final_x, final_y), axes.c2p(final_x, setpoint_val))
        sse_brace = Brace(sse_brace_line, RIGHT, color=RED_B)
        sse_label = sse_brace.get_text("Steady State Error", font_size=18).set_color(RED_B)

        self.play(
            FadeIn(rise_brace), FadeIn(rise_label),
            FadeIn(sse_brace), FadeIn(sse_label)
        )
        
        self.wait(2)
