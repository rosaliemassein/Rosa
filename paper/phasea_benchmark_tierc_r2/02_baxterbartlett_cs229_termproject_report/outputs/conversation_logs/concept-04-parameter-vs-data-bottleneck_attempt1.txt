from manim import *
import numpy as np

class Concept04ParameterVsDataBottleneck(Scene):
    def construct(self):
        # 1. Configuration and Data Setup
        np.random.seed(42)
        points_x = np.linspace(-2.5, 2.5, 27)
        # Create a underlying pattern with noise to represent "learning the noise"
        base_pattern = 0.5 * np.sin(points_x * 1.5)
        noise = np.random.normal(0, 0.15, 27)
        points_y = base_pattern + noise

        # 2. Axes and Layout
        ax_left = Axes(
            x_range=[-3, 3, 1], y_range=[-1.5, 1.5, 1],
            x_length=5.5, y_length=3.5,
            axis_config={"include_tip": False}
        ).to_edge(LEFT, buff=0.5).shift(UP * 1.0)

        ax_right = Axes(
            x_range=[-3, 3, 1], y_range=[-1.5, 1.5, 1],
            x_length=5.5, y_length=3.5,
            axis_config={"include_tip": False}
        ).to_edge(RIGHT, buff=0.5).shift(UP * 1.0)

        label_nn = Text("Neural Network", color=BLUE).scale(0.7).next_to(ax_left, UP)
        label_hv = Text("High Variance", color=RED).scale(0.5).next_to(label_nn, DOWN, buff=0.1)
        label_lwr = Text("LWR", color=GREEN).scale(0.7).next_to(ax_right, UP)

        # 3. Data Points
        dots_left = VGroup(*[Dot(ax_left.c2p(x, y), radius=0.05, color=WHITE) for x, y in zip(points_x, points_y)])
        dots_right = VGroup(*[Dot(ax_right.c2p(x, y), radius=0.05, color=WHITE) for x, y in zip(points_x, points_y)])

        # 4. Curves
        # Neural Network: Complex, wiggling curve hitting every point
        nn_curve = ax_left.plot(
            lambda x: np.interp(x, points_x, points_y) + 0.15 * np.sin(40 * x),
            x_range=[-2.5, 2.5],
            color=RED,
            use_smoothing=True
        )
        
        # LWR: Piecewise-linear / smoother path
        lwr_curve = ax_right.plot(
            lambda x: np.interp(x, points_x, points_y),
            x_range=[-2.5, 2.5],
            color=GREEN,
            use_smoothing=False
        )

        # 5. Formula and Bar Chart (Bottom Section)
        formula = MathTex(r"MSE = \frac{||Y_{test} - Y_{pred}||_F^2}{3n}").scale(0.7)
        formula.to_edge(DOWN, buff=1.2).shift(LEFT * 3.5)

        chart = BarChart(
            values=[1.21, 0.85],
            bar_names=["NN", "LWR"],
            y_range=[0, 1.5, 0.5],
            y_length=2.2,
            x_length=4.5,
            bar_colors=[BLUE, GREEN],
            y_axis_config={"font_size": 18},
            x_axis_config={"font_size": 24}
        ).scale(0.8).to_edge(DOWN, buff=0.3).shift(RIGHT * 3.2)
        
        chart_labels = chart.get_bar_labels(font_size=24)

        # 6. Animation Sequence
        self.play(Create(ax_left), Create(ax_right))
        self.play(Write(label_nn), Write(label_lwr))
        self.play(Create(dots_left), Create(dots_right))
        self.wait(1)

        self.play(Create(nn_curve), Write(label_hv))
        self.play(Create(lwr_curve))
        self.wait(1)

        self.play(Write(formula))
        self.play(Create(chart), Write(chart_labels))
        
        # Adding concluding Narration text
        narration = Text(
            "With 27 points, the NN overfits noise (24 params),\nwhile LWR's local linear approach stays robust.",
            line_spacing=0.8
        ).scale(0.4).to_edge(UP, buff=0.1)
        
        self.play(FadeIn(narration, shift=DOWN))
        self.wait(4)

        # Final cleanup for nice ending
        self.play(FadeOut(narration), FadeOut(ax_left), FadeOut(ax_right), 
                  FadeOut(dots_left), FadeOut(dots_right), FadeOut(nn_curve), 
                  FadeOut(lwr_curve), FadeOut(label_nn), FadeOut(label_hv), 
                  FadeOut(label_lwr), FadeOut(formula), FadeOut(chart), FadeOut(chart_labels))
        self.wait(1)
