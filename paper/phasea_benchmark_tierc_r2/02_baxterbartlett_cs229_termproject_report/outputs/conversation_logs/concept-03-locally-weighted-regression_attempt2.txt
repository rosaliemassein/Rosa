from manim import *
import numpy as np
import random

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Setup Axes
        ax = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            x_length=10,
            y_length=6,
            axis_config={"include_tip": True}
        ).to_edge(DOWN, buff=0.5)
        
        labels = ax.get_axis_labels(x_label="x", y_label="y")
        self.add(ax, labels)

        # 2. Generate Data Points (sine wave with noise)
        random.seed(42)
        x_values = np.linspace(-4, 4, 50)
        y_values = np.sin(x_values) + np.random.normal(0, 0.15, 50)
        
        dots = VGroup()
        data_coords = []
        for x, y in zip(x_values, y_values):
            coord = ax.c2p(x, y)
            dot = Dot(coord, radius=0.08, color=BLUE, fill_opacity=0.4)
            dots.add(dot)
            data_coords.append((x, y))

        self.play(Create(dots))
        self.wait(0.5)

        # 3. Query Point and weight visual
        query_x_tracker = ValueTracker(-3)
        
        query_point = Dot(color=WHITE, z_index=10).scale(1.2)
        query_point.add_updater(lambda mob: mob.move_to(
            ax.c2p(query_x_tracker.get_value(), np.sin(query_x_tracker.get_value()))
        ))
        
        weight_field = Dot(radius=1.5, color=YELLOW, fill_opacity=0.15, stroke_width=0)
        weight_field.add_updater(lambda mob: mob.move_to(query_point.get_center()))
        
        self.play(FadeIn(query_point), FadeIn(weight_field))

        # 4. Locally Weighted Regression Logic
        tau = 0.8

        def get_wls_params(q_x):
            weights = np.array([np.exp(-((x - q_x)**2) / (2 * tau**2)) for x in x_values])
            sum_w = np.sum(weights)
            if sum_w == 0: return 0, 0, 0
            
            w_x_mean = np.sum(weights * x_values) / sum_w
            w_y_mean = np.sum(weights * y_values) / sum_w
            
            num = np.sum(weights * (x_values - w_x_mean) * (y_values - w_y_mean))
            den = np.sum(weights * (x_values - w_x_mean)**2)
            slope = num / den if den != 0 else 0
            return w_x_mean, w_y_mean, slope

        # 5. Updaters
        def update_dots(dots_group):
            q_x = query_x_tracker.get_value()
            for i, dot in enumerate(dots_group):
                x_i = data_coords[i][0]
                weight = np.exp(-((x_i - q_x)**2) / (2 * tau**2))
                # Manual interpolation for opacity: 0.2 to 1.0
                new_opacity = 0.2 + weight * 0.8
                dot.set_fill(opacity=new_opacity)
                if weight > 0.5:
                    dot.set_color(YELLOW)
                else:
                    dot.set_color(BLUE)

        dots.add_updater(update_dots)

        local_line = Line(LEFT, RIGHT, color=RED, stroke_width=6)
        
        def update_line(line_mob):
            q_x = query_x_tracker.get_value()
            mx, my, slope = get_wls_params(q_x)
            x_start, x_end = mx - 1.0, mx + 1.0
            y_start = my + slope * (x_start - mx)
            y_end = my + slope * (x_end - mx)
            line_mob.put_start_and_end_on(ax.c2p(x_start, y_start), ax.c2p(x_end, y_end))

        local_line.add_updater(update_line)
        
        # Display Formula
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)")
        formula.to_corner(UR, buff=0.5).scale(0.8)
        self.play(Write(formula))
        
        self.add(local_line)
        self.wait(1)

        # 6. Animation: Moving the query point
        self.play(
            query_x_tracker.animate.set_value(3), 
            run_time=6, 
            rate_func=rate_functions.linear
        )
        self.wait(1)
        self.play(
            query_x_tracker.animate.set_value(-2), 
            run_time=4, 
            rate_func=rate_functions.smooth
        )
        self.wait(2)

        # Cleanup
        dots.remove_updater(update_dots)
        local_line.remove_updater(update_line)
        query_point.clear_updaters()
        weight_field.clear_updaters()
        self.play(FadeOut(VGroup(ax, labels, dots, query_point, weight_field, local_line, formula)))
