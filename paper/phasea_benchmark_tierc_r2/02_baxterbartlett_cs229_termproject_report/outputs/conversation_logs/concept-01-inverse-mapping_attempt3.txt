from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Diagram Setup: "Input" vs "PID Constants"
        behavior_text = Text("Desired Behavior", font_size=24, color=BLUE)
        item1 = Text("• Rise Time", font_size=18)
        item2 = Text("• Steady State Error", font_size=18)
        item3 = Text("• Stability", font_size=18)
        behavior_items = VGroup(item1, item2, item3).arrange(DOWN, aligned_edge=LEFT, buff=0.2)
        behavior_group = VGroup(behavior_text, behavior_items).arrange(DOWN, buff=0.4)
        left_rect = SurroundingRectangle(behavior_group, color=WHITE, buff=0.3)
        left_side = VGroup(left_rect, behavior_group)

        pid_text = Text("PID Constants", font_size=24, color=RED)
        pid_formula_simple = MathTex("K_p, K_i, K_d", font_size=28)
        pid_group = VGroup(pid_text, pid_formula_simple).arrange(DOWN, buff=0.4)
        right_rect = SurroundingRectangle(pid_group, color=WHITE, buff=0.3)
        right_side = VGroup(right_rect, pid_group)

        diagram = VGroup(left_side, right_side).arrange(RIGHT, buff=2).to_edge(UP, buff=0.5)
        arrow = Arrow(left_side.get_right(), right_side.get_left(), buff=0.1)
        
        full_diagram = VGroup(left_side, right_side, arrow)

        # Reference Formula
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=24)
        formula.next_to(diagram, DOWN, buff=0.5)

        self.play(Create(left_side), Create(right_side), GrowArrow(arrow))
        self.play(Write(formula))
        self.wait(1)

        # 2. "Flip the script": Rotate the diagram 180 degrees
        # We rotate everything and then rotate the text back to keep it upright
        self.play(
            full_diagram.animate.rotate(PI),
            behavior_text.animate.rotate(-PI),
            item1.animate.rotate(-PI),
            item2.animate.rotate(-PI),
            item3.animate.rotate(-PI),
            pid_text.animate.rotate(-PI),
            pid_formula_simple.animate.rotate(-PI),
            run_time=2
        )
        self.wait(1)

        # 3. Motor Velocity Plot (RPM vs Time)
        axes = Axes(
            x_range=[0, 10, 2],
            y_range=[0, 1.5, 0.5],
            x_length=7,
            y_length=2.5,
            axis_config={"include_tip": True}
        ).to_edge(DOWN, buff=0.8)
        
        x_lab = axes.get_x_axis_label("Time", edge=DOWN, direction=DOWN, buff=0.1).scale(0.6)
        y_lab = axes.get_y_axis_label("RPM", edge=LEFT, direction=LEFT, buff=0.1).scale(0.6)

        setpoint = DashedLine(
            start=axes.c2p(0, 1),
            end=axes.c2p(10, 1),
            color=GRAY
        )

        def response_func(x_val):
            return 1 - np.exp(-0.5 * x_val) * np.cos(3 * x_val)

        curve = axes.plot(
            response_func,
            x_range=[0, 10],
            color=YELLOW
        )

        # 4. Indicators (Replacing Brace with Lines/Arrows)
        # Rise time (roughly from t=0 to first cross of 1.0 at t=0.52)
        rise_time_line = Line(axes.c2p(0, -0.15), axes.c2p(0.52, -0.15), color=WHITE)
        rt_cap1 = Line(axes.c2p(0, -0.2), axes.c2p(0, -0.1), color=WHITE)
        rt_cap2 = Line(axes.c2p(0.52, -0.2), axes.c2p(0.52, -0.1), color=WHITE)
        rise_time_label = Text("Rise Time", font_size=14).next_to(rise_time_line, DOWN, buff=0.1)

        # Steady State Error at the tail end
        val_at_end = response_func(9.5)
        ss_line = DoubleArrow(
            axes.c2p(9.5, val_at_end), 
            axes.c2p(9.5, 1.0), 
            buff=0, 
            tip_length=0.1, 
            color=WHITE
        )
        ss_label = Text("Steady State Error", font_size=14).next_to(ss_line, RIGHT, buff=0.1)

        # Layout cleanup
        self.play(
            full_diagram.animate.scale(0.7).to_edge(UP, buff=0.2),
            formula.animate.scale(0.8).next_to(axes, UP, buff=0.1),
            Create(axes),
            Write(x_lab), Write(y_lab),
            Create(setpoint)
        )
        self.play(Create(curve), run_time=2)
        self.play(
            Create(rise_time_line), Create(rt_cap1), Create(rt_cap2), Write(rise_time_label),
            Create(ss_line), Write(ss_label)
        )
        
        self.wait(2)
