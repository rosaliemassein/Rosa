from manim import *

class Concept03PseudoExecutionHandoff(Scene):
    def construct(self):
        # 1. Background / Split Screen Setup
        line = Line(LEFT * 7, RIGHT * 7, color=GRAY)
        top_label = Text("Data Agent (A_d)", font_size=24, color=BLUE).to_edge(UP).shift(LEFT * 5)
        bottom_label = Text("Model Agent (A_m)", font_size=24, color=GREEN).to_edge(DOWN).shift(LEFT * 5)
        self.add(line, top_label, bottom_label)

        # 2. Narrator Text (Voice)
        narrator_text = Text(
            "The real magic happens during plan execution.",
            font_size=28
        ).to_edge(UP, buff=0.5)
        
        # 3. Data Agent and Data Block
        data_block = Rectangle(height=1.5, width=2, fill_opacity=0.5, color=BLUE).shift(UP * 1.5 + LEFT * 3)
        data_text = Text("Data Block", font_size=20).move_to(data_block)
        data_agent_icon = Circle(radius=0.5, color=BLUE, fill_opacity=0.8).next_to(data_block, LEFT, buff=1)
        data_agent_label = Text("A_d", font_size=24).move_to(data_agent_icon)
        
        self.play(Write(narrator_text))
        self.play(Create(data_agent_icon), Write(data_agent_label), Create(data_block), Write(data_text))
        self.wait(1)

        # 4. Pseudo-analysis and Output O_d
        o_d = MathTex("O_d", color=YELLOW).scale(1.2).move_to(data_block.get_center())
        insight_box = SurroundingRectangle(o_d, color=YELLOW, buff=0.2)
        data_insights = VGroup(insight_box, o_d)

        self.play(
            narrator_text.animate.become(Text("Data Agent performs 'pseudo-analysis' to identify key insights.", font_size=28).to_edge(UP, buff=0.5)),
            data_block.animate.set_fill(opacity=0.2),
            FadeIn(data_insights)
        )
        self.wait(1)

        # 5. Model Agent and Task s_m
        model_agent_icon = Circle(radius=0.5, color=GREEN, fill_opacity=0.8).shift(DOWN * 1.5 + LEFT * 4)
        model_agent_label = Text("A_m", font_size=24).move_to(model_agent_icon)
        
        # Formula: s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)
        s_m_formula = MathTex("s_i^m = \\mathrm{PD}(R, \\mathcal{A}_m, \\mathbf{p}_i, O_i^d)", font_size=30).shift(DOWN * 1.5 + RIGHT * 1)
        s_m_box = Rectangle(height=1, width=5.5, color=GREEN).move_to(s_m_formula)
        model_task = VGroup(s_m_box, s_m_formula)

        # 6. Information Handoff Animation
        handoff_arrow = Arrow(data_insights.get_bottom(), s_m_box.get_top(), color=YELLOW)
        
        self.play(
            narrator_text.animate.become(Text("It then hands off these insights to the Model Agent.", font_size=28).to_edge(UP, buff=0.5)),
            Create(model_agent_icon), Write(model_agent_label),
            GrowArrow(handoff_arrow)
        )
        
        o_d_copy = data_insights.copy()
        self.play(
            o_d_copy.animate.scale(0.5).move_to(s_m_formula.get_right() + RIGHT * 0.5),
            FadeIn(model_task, shift=UP),
            run_time=1.5
        )
        self.wait(1)

        # 7. Search Efficiency ValueTracker
        efficiency_tracker = ValueTracker(0.01)
        
        # Efficiency Bar
        bar_bg = Rectangle(width=4, height=0.4, color=WHITE).to_edge(DOWN, buff=1.2).shift(RIGHT * 3)
        bar_fill = always_redraw(lambda: Rectangle(
            width=efficiency_tracker.get_value() * 4,
            height=0.4,
            fill_opacity=0.8,
            fill_color=GOLD,
            stroke_width=0
        ).align_to(bar_bg, LEFT))
        
        efficiency_label = Text("Search Efficiency", font_size=24).next_to(bar_bg, LEFT)
        efficiency_percent = always_redraw(lambda: DecimalNumber(
            efficiency_tracker.get_value() * 100,
            num_decimal_places=0,
            unit="%"
        ).scale(0.8).next_to(bar_bg, RIGHT))

        self.play(
            narrator_text.animate.become(Text("Efficiency increases via training-free search.", font_size=28).to_edge(UP, buff=0.5)),
            Create(bar_bg),
            Create(bar_fill),
            Write(efficiency_label),
            Write(efficiency_percent)
        )

        # 8. Final Animation of Efficiency Increasing
        self.play(
            efficiency_tracker.animate.set_value(0.95),
            model_task.animate.scale(1.1).set_color(GOLD),
            run_time=4,
            rate_func=smooth
        )
        
        final_text = Text("No GPU Training Required", font_size=24, color=GOLD).next_to(bar_bg, UP)
        self.play(Write(final_text))
        self.wait(2)

        self.play(FadeOut(VGroup(*self.mobjects)))
