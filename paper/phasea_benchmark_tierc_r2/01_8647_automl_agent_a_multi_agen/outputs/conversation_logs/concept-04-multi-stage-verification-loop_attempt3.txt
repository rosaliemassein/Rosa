from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup stages and layout
        planning_label = Text("Planning", color=BLUE).scale(0.6).shift(LEFT * 5 + UP * 0.5)
        planning_box = SurroundingRectangle(planning_label, color=BLUE, buff=0.2)
        planning_group = VGroup(planning_label, planning_box)

        # Draw three vertical lines acting as 'Gates'
        gate_labels = ["ReqVer", "ExecVer", "ImpVer"]
        gates = VGroup()
        for i, label in enumerate(gate_labels):
            line = Line(UP * 1.5, DOWN * 1.5, color=WHITE)
            text = Text(label).scale(0.5).next_to(line, UP)
            gate = VGroup(line, text).shift(LEFT * (2 - i * 2.5))
            gates.add(gate)

        # Deployment icon
        deploy_icon = VGroup(
            RegularPolygon(n=6, color=GOLD),
            Text("Deployment", color=GOLD).scale(0.4).next_to(ORIGIN, DOWN)
        ).scale(0.7).shift(RIGHT * 5.5)

        # Formula at the bottom
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}"
        ).scale(0.8).to_edge(DOWN, buff=0.5)

        self.add(planning_group, gates, deploy_icon, formula)

        # 2. Initialize the Plan object (a small square)
        plan = Square(side_length=0.4, color=YELLOW, fill_opacity=1).move_to(planning_group.get_center())
        
        # 3. Animate movement and logic
        self.play(FadeIn(plan))
        
        # Stage 1: Request Verification (Passes)
        self.play(plan.animate.move_to(gates[0][0].get_center()))
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.2)

        # Stage 2: Execution Verification (Simulate Failure)
        self.play(plan.animate.move_to(gates[1][0].get_center()))
        self.play(plan.animate.set_color(RED))
        
        # Feedback loop back to planning
        # Using ArcBetweenPoints + add_tip as a robust alternative to CurvedArrow
        start_pt = gates[1][0].get_center() + UP * 0.5
        end_pt = planning_group.get_center() + UP * 0.8
        feedback_arrow = ArcBetweenPoints(start_pt, end_pt, angle=-TAU / 3, color=RED).add_tip()
        feedback_text = Text("Feedback", color=RED).scale(0.4).next_to(feedback_arrow, UP, buff=0.1)
        
        self.play(Create(feedback_arrow), Write(feedback_text))
        self.play(plan.animate.move_to(planning_group.get_center()).set_color(YELLOW))
        self.play(FadeOut(feedback_arrow), FadeOut(feedback_text))

        # 4. Final successful pass through all gates
        for i in range(3):
            self.play(plan.animate.move_to(gates[i][0].get_center()), run_time=0.6)
            self.play(plan.animate.set_color(GREEN), run_time=0.2)
        
        # 5. Reach Deployment
        self.play(plan.animate.move_to(deploy_icon.get_center()))
        self.play(Flash(deploy_icon, color=GOLD, line_length=0.3))
        
        # Final narrative label
        narrative = Text(
            "Deployment-ready models reach the finish line.",
            slant="ITALIC"
        ).scale(0.5).to_edge(UP)
        self.play(Write(narrative))
        
        self.wait(2)
