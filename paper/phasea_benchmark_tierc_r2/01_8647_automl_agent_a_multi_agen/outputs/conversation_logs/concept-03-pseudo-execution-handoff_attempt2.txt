from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Agents (Top and Bottom for split-screen feel)
        data_agent = VGroup(
            Dot(color=BLUE, radius=0.2),
            Text("Data Agent ($A_d$)", font_size=24)
        ).arrange(RIGHT, buff=0.3).to_edge(UP, buff=1)

        model_agent = VGroup(
            Dot(color=GREEN, radius=0.2),
            Text("Model Agent ($A_m$)", font_size=24)
        ).arrange(RIGHT, buff=0.3).shift(DOWN * 0.5)

        # 2. Data Block and Output
        data_block = Rectangle(width=1.5, height=1, color=BLUE, fill_opacity=0.2)
        data_block_label = Text("Raw Data", font_size=20).move_to(data_block)
        data_group = VGroup(data_block, data_block_label).next_to(data_agent, DOWN, buff=0.5)

        o_d_box = Rectangle(width=2, height=0.8, color=WHITE, fill_opacity=0.1)
        o_d_label = MathTex("O_d", font_size=30).move_to(o_d_box)
        o_d_text = Text("Data Insights", font_size=16).next_to(o_d_box, DOWN, buff=0.1)
        o_d_group = VGroup(o_d_box, o_d_label, o_d_text)

        # 3. Model Agent Task Block
        task_block = Rectangle(width=2, height=0.8, color=GREEN, fill_opacity=0.3)
        task_label = MathTex("s_m", font_size=30).move_to(task_block)
        task_group = VGroup(task_block, task_label).next_to(model_agent, DOWN, buff=0.5)

        # 4. Search Efficiency Bar
        efficiency_tracker = ValueTracker(0.1)
        bar_bg = Rectangle(width=4, height=0.4, color=GRAY, stroke_width=2)
        bar_fill = always_redraw(lambda: 
            Rectangle(
                width=max(0.01, efficiency_tracker.get_value() * 4), 
                height=0.4, 
                color=YELLOW, 
                fill_opacity=0.8,
                stroke_width=0
            ).align_to(bar_bg, LEFT)
        )
        bar_label = Text("Search Efficiency", font_size=20).next_to(bar_bg, UP, buff=0.2)
        efficiency_bar_group = VGroup(bar_bg, bar_fill, bar_label).to_edge(RIGHT, buff=0.5).shift(UP * 0.5)

        # 5. Formula and Narration
        formula = MathTex(r"s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)").scale(0.8)
        formula.to_edge(DOWN, buff=0.5)

        # --- ANIMATION SEQUENCE ---
        
        # Introduction
        self.play(Create(data_agent), Create(model_agent))
        self.play(Create(data_group))
        self.wait(1)

        # Data Agent Processes Data -> O_d
        self.play(data_group.animate.set_color(YELLOW), run_time=0.5)
        o_d_group.move_to(data_group.get_center())
        self.play(FadeIn(o_d_group), FadeOut(data_group))
        
        # Handoff O_d to Model Agent
        arrow = Arrow(o_d_group.get_bottom(), model_agent.get_top(), color=WHITE)
        self.play(Create(arrow))
        
        # Move insight to model agent
        self.play(
            o_d_group.animate.scale(0.6).next_to(model_agent, LEFT, buff=0.5),
            FadeOut(arrow)
        )
        
        # Model Agent generates task s_m (Expanding)
        task_group.save_state()
        task_group.scale(0.1).move_to(model_agent.get_bottom())
        self.play(
            task_group.animate.restore(),
            FadeIn(efficiency_bar_group)
        )

        # Show training-free search efficiency increase
        self.play(
            efficiency_tracker.animate.set_value(0.9),
            Write(formula),
            run_time=3,
            rate_func=smooth
        )
        
        self.wait(2)

        # Cleanup
        self.play(FadeOut(data_agent), FadeOut(model_agent), FadeOut(o_d_group), FadeOut(task_group), FadeOut(efficiency_bar_group), FadeOut(formula))
        self.wait(1)
