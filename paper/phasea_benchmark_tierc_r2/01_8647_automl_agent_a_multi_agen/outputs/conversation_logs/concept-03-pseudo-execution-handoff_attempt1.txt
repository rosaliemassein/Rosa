from manim import *

class ConceptAnimation03PseudoExecutionHandoff(Scene):
    def construct(self):
        # 1. Setup Agents and Elements
        data_agent_label = Text("Data Agent (A_d)", font_size=28, color=BLUE)
        data_agent_box = Square(side_length=1.5, color=BLUE)
        data_agent = VGroup(data_agent_label, data_agent_box).arrange(DOWN).shift(UP * 2 + LEFT * 3)

        model_agent_label = Text("Model Agent (A_m)", font_size=28, color=GREEN)
        model_agent_box = Square(side_length=1.5, color=GREEN)
        model_agent = VGroup(model_agent_label, model_agent_box).arrange(DOWN).shift(DOWN * 2 + LEFT * 3)

        # Blocks
        data_block = Rectangle(width=1, height=0.6, fill_opacity=0.5, color=YELLOW).next_to(data_agent_box, LEFT)
        insight_label = MathTex("O_d", color=YELLOW).scale(0.8)
        insight_box = VGroup(
            Rectangle(width=1, height=0.6, fill_opacity=0.8, color=YELLOW),
            insight_label
        ).next_to(data_agent_box, RIGHT)

        task_block = Rectangle(width=1.2, height=0.7, fill_opacity=0.5, color=GREEN).move_to(model_agent_box.get_center())
        
        # Search Efficiency Bar
        efficiency_tracker = ValueTracker(0.1)
        bar_label = Text("Search Efficiency", font_size=24).shift(RIGHT * 3.5 + UP * 1.5)
        bar_outline = Rectangle(width=3, height=0.4, color=WHITE).next_to(bar_label, DOWN)
        
        bar_fill = always_redraw(lambda: Rectangle(
            width=max(0.01, efficiency_tracker.get_value() * 3),
            height=0.4,
            fill_color=RED,
            fill_opacity=0.8,
            stroke_width=0
        ).align_to(bar_outline, LEFT))
        
        efficiency_percent = always_redraw(lambda: DecimalNumber(
            efficiency_tracker.get_value() * 100, 
            unit="%", 
            num_decimal_places=0,
            font_size=24
        ).next_to(bar_outline, RIGHT))

        # Formula
        formula = MathTex(r"s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)", font_size=32).to_edge(DOWN)

        # 2. Animation Sequence
        # Initial state
        self.play(FadeIn(data_agent), FadeIn(model_agent))
        self.play(FadeIn(data_block))
        self.wait(1)

        # Pseudo-analysis animation
        self.play(
            data_block.animate.move_to(data_agent_box.get_center()),
            run_time=1
        )
        self.play(FadeIn(insight_box))
        self.play(Indicate(insight_box))
        self.wait(1)

        # Information Handoff
        self.play(
            insight_box.animate.move_to(model_agent_box.get_center() + RIGHT * 1.8),
            Create(Arrow(insight_box.get_center(), model_agent_box.get_center(), buff=0.2)),
            run_time=1.5
        )

        # Model Agent receiving input and formula appearing
        self.play(FadeIn(task_block))
        self.play(
            insight_box.animate.scale(0.5).move_to(task_block.get_right() + LEFT * 0.2),
            task_block.animate.scale(1.5),
            Write(formula),
            run_time=1.5
        )

        # Search Efficiency Bar animation
        self.play(
            Create(bar_label), 
            Create(bar_outline), 
            Create(bar_fill), 
            Create(efficiency_percent)
        )
        
        # Training-free search (efficiency increases)
        self.play(
            efficiency_tracker.animate.set_value(0.98),
            run_time=4,
            rate_func=rate_functions.linear
        )
        
        self.wait(2)
