from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup Labels and UI elements
        voice_text = "Finally, we ensure reliability through a three-stage verification gate."
        subtitle = Text(voice_text, font_size=24).to_edge(DOWN, buff=0.5)
        
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}",
            font_size=30
        ).to_edge(UP, buff=0.5)

        planning_label = Text("Planning", font_size=24).move_to(4 * LEFT + 1.5 * UP)
        planning_rect = Rectangle(width=2, height=1.5, color=BLUE).move_to(4 * LEFT)
        
        gate_labels = ["ReqVer", "ExecVer", "ImpVer"]
        gate_positions = [-1.5, 0.5, 2.5]
        gates = VGroup()
        
        for label_text, pos_x in zip(gate_labels, gate_positions):
            line = Line(2 * UP, 2 * DOWN, color=WHITE)
            line.move_to(pos_x * RIGHT)
            label = Text(label_text, font_size=20).next_to(line, UP)
            gates.add(VGroup(line, label))

        deployment_icon = VGroup(
            RegularPolygon(n=6, color=YELLOW),
            Text("Deploy", font_size=20).shift(0.8 * DOWN)
        ).scale(0.6).move_to(4.5 * RIGHT)

        self.add(subtitle, formula, planning_rect, planning_label, gates, deployment_icon)

        # 2. Define the Plan object
        plan = Square(side_length=0.4, fill_opacity=1, color=WHITE).move_to(planning_rect.get_center())
        
        # 3. Animation Sequence
        self.play(FadeIn(plan))

        # --- FIRST ATTEMPT: Fail at ReqVer ---
        # Move to first gate
        self.play(plan.animate.move_to(gate_positions[0] * RIGHT))
        self.play(plan.animate.set_color(RED))
        
        # Feedback Loop logic using ArcBetweenPoints
        fail_arrow = ArcBetweenPoints(
            start=gate_positions[0] * RIGHT + 0.5 * UP,
            end=planning_rect.get_center() + 0.5 * UP,
            angle=-PI/2
        ).add_tip().set_color(RED)
        
        feedback_text = Text("Feedback", font_size=18, color=RED).next_to(fail_arrow, UP, buff=0.1)
        
        self.play(Create(fail_arrow), Write(feedback_text))
        self.play(plan.animate.move_to(planning_rect.get_center()), run_time=0.7)
        self.play(FadeOut(fail_arrow), FadeOut(feedback_text), plan.animate.set_color(WHITE))
        self.wait(0.2)

        # --- SUCCESSFUL RUN ---
        # Stage 1: ReqVer
        self.play(plan.animate.move_to(gate_positions[0] * RIGHT))
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.1)
        
        # Stage 2: ExecVer
        self.play(plan.animate.move_to(gate_positions[1] * RIGHT))
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.1)
        
        # Stage 3: ImpVer
        self.play(plan.animate.move_to(gate_positions[2] * RIGHT))
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.1)
        
        # Final: Deployment
        self.play(plan.animate.move_to(deployment_icon.get_center()))
        self.play(Flash(deployment_icon, color=YELLOW, line_length=0.3, num_lines=12))
        
        self.wait(2)
