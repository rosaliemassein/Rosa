from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # Text and Formula from prompt
        goal_text_str = "Understand the iterative feedback loop and verification stages"
        voice_text_str = "Ensuring only deployment-ready models reach the finish line."
        formula_str = r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}"

        # 1. UI Setup
        goal_label = Text(goal_text_str, font_size=24, color=BLUE).to_edge(UP, buff=0.3)
        formula = MathTex(formula_str, font_size=32).next_to(goal_label, DOWN, buff=0.2)
        self.add(goal_label, formula)

        # 2. Layout Definitions
        start_point = LEFT * 5
        gate1_x = -2.5
        gate2_x = 0
        gate3_x = 2.5
        end_point = RIGHT * 5

        # 3. Create Gates (Vertical Lines)
        gate_height = 2.5
        req_gate = Line(UP * gate_height/2, DOWN * gate_height/2, color=BLUE).move_to([gate1_x, 0, 0])
        exec_gate = Line(UP * gate_height/2, DOWN * gate_height/2, color=YELLOW).move_to([gate2_x, 0, 0])
        imp_gate = Line(UP * gate_height/2, DOWN * gate_height/2, color=RED).move_to([gate3_x, 0, 0])

        req_label = Text("ReqVer", font_size=18).next_to(req_gate, UP)
        exec_label = Text("ExecVer", font_size=18).next_to(exec_gate, UP)
        imp_label = Text("ImpVer", font_size=18).next_to(imp_gate, UP)

        planning_label = Text("Planning", font_size=22).move_to(start_point + UP * 1.5)
        deployment_label = Text("Deployment", font_size=22, color=GREEN).move_to(end_point + UP * 1.5)

        self.add(req_gate, exec_gate, imp_gate, req_label, exec_label, imp_label, planning_label, deployment_label)

        # 4. Plan Object (Square)
        plan = Square(side_length=0.4, fill_opacity=1, color=WHITE).move_to(start_point)
        self.add(plan)

        # 5. Animation - Stage 1: Move to ReqVer and Pass
        self.play(plan.animate.move_to([gate1_x, 0, 0]), run_time=1)
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.2)

        # 6. Animation - Stage 2: Move to ExecVer and Fail
        self.play(plan.animate.move_to([gate2_x, 0, 0]), run_time=1)
        self.play(plan.animate.set_color(RED))
        
        # Feedback loop: Curved arrow back to Planning
        # Using ArcBetweenPoints + add_tip to avoid CurvedArrow identifier issues
        feedback_path = ArcBetweenPoints(
            start=np.array([gate2_x, 0.5, 0]), 
            end=np.array([start_point[0], 0.5, 0]), 
            angle=-PI/2, 
            color=RED
        ).add_tip()
        
        self.play(Create(feedback_path))
        self.play(
            plan.animate.move_to(start_point).set_color(WHITE),
            FadeOut(feedback_path),
            run_time=1.5
        )

        # 7. Animation - Stage 3: Full Success Run
        # Pass ReqVer
        self.play(plan.animate.move_to([gate1_x, 0, 0]), run_time=0.7)
        self.play(plan.animate.set_color(GREEN))
        
        # Pass ExecVer
        self.play(plan.animate.move_to([gate2_x, 0, 0]), run_time=0.7)
        self.play(plan.animate.set_color(GREEN))
        
        # Pass ImpVer
        self.play(plan.animate.move_to([gate3_x, 0, 0]), run_time=0.7)
        self.play(plan.animate.set_color(GREEN))
        
        # Reach Deployment
        deployment_box = Square(side_length=0.6, color=GREEN).move_to(end_point)
        self.play(
            plan.animate.move_to(end_point).scale(1.2),
            Create(deployment_box),
            run_time=1
        )

        # 8. Voiceover Caption
        voice_label = Text(voice_text_str, font_size=20, slant="ITALIC").to_edge(DOWN, buff=0.5)
        self.play(Write(voice_label))
        
        self.wait(2)
