from manim import *
import numpy as np

class GeneticWinRateSelection(Scene):
    def construct(self):
        # 1. Title and Formula
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)")
        formula.to_edge(UP)
        self.play(Write(formula))

        # 2. Initial Parent Node
        parent_node = self.create_node("Parent", YELLOW)
        parent_node.move_to(UP * 1.5)
        self.play(Create(parent_node))
        self.wait(0.5)

        # 3. Evolutionary Cycle
        for cycle in range(2):
            # Define Children positions
            child_positions = [LEFT * 4 + DOWN, DOWN, RIGHT * 4 + DOWN]
            children = VGroup(*[
                self.create_node(f"C{i+1}", BLUE) for i in range(3)
            ])
            for i, child in enumerate(children):
                child.move_to(child_positions[i])

            # Branching lines
            lines = VGroup(*[
                Line(parent_node.get_bottom(), child.get_top(), buff=0.1)
                for child in children
            ])

            self.play(
                LaggedStart(*[Create(line) for line in lines], lag_ratio=0.2),
                LaggedStart(*[Create(child) for child in children], lag_ratio=0.2),
                run_time=1.0
            )

            # 4. Bar Charts (Win Rates)
            win_rates = [0.45, 0.62, 0.88] if cycle == 0 else [0.72, 0.94, 0.81]
            bars = VGroup()
            bar_labels = VGroup()

            for i, rate in enumerate(win_rates):
                bar = Rectangle(
                    width=0.8, 
                    height=rate * 2, 
                    fill_opacity=0.8, 
                    color=WHITE,
                    fill_color=WHITE
                )
                bar.next_to(children[i], DOWN, buff=0.4)
                # Align bottom of bars to a common floor
                bar.align_to(children[i].get_bottom() + DOWN * 2.5, DOWN)
                
                label = DecimalNumber(rate, num_decimal_places=2).scale(0.6)
                label.next_to(bar, UP, buff=0.1)
                
                bars.add(bar)
                bar_labels.add(label)

            # Animate bars growing
            self.play(
                *[GrowFromPoint(bar, bar.get_bottom()) for bar in bars],
                *[FadeIn(label) for label in bar_labels]
            )
            self.wait(0.5)

            # 5. Highlight Winner
            winner_idx = int(np.argmax(win_rates))
            winner_node = children[winner_idx]
            winner_bar = bars[winner_idx]
            
            self.play(
                winner_node[0].animate.set_color(GOLD),
                winner_bar.animate.set_color(GOLD),
                Indicate(winner_node),
                Indicate(winner_bar)
            )
            self.wait(0.5)

            # 6. Selection Transition: Keep Winner, remove others
            others = VGroup(
                *[c for i, c in enumerate(children) if i != winner_idx],
                *[b for i, b in enumerate(bars) if i != winner_idx],
                *[l for i, l in enumerate(bar_labels) if i != winner_idx],
                lines,
                parent_node,
                bar_labels[winner_idx],
                winner_bar
            )

            # Create new parent from winner and move to center
            new_parent_center = UP * 1.5
            
            self.play(
                FadeOut(others),
                winner_node.animate.move_to(new_parent_center).set_color(YELLOW)
            )
            
            # Update parent reference for next loop
            parent_node = winner_node
            
            if cycle == 0:
                self.wait(0.5)

        self.wait(2)

    def create_node(self, label_text, color):
        circle = Circle(radius=0.5, color=color, fill_opacity=0.2)
        label = Text(label_text, font_size=24).move_to(circle.get_center())
        return VGroup(circle, label)
