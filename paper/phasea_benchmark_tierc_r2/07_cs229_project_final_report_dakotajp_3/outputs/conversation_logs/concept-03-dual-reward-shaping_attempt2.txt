from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Setup Title and Environment
        title = Text("Dual Reward Shaping").to_edge(UP)
        self.add(title)

        target_zone = Circle(radius=1.5, color=YELLOW, fill_opacity=0.2).to_edge(RIGHT, buff=1)
        target_label = Text("Target Zone", color=YELLOW).scale(0.6).next_to(target_zone, UP)
        
        pegs = VGroup(*[Dot(color=BLUE) for _ in range(6)])
        pegs.arrange_in_grid(rows=3, cols=2, buff=0.5).to_edge(LEFT, buff=1)
        
        self.add(target_zone, target_label, pegs)

        # 2. Centroid and Line Visualization
        # Use explicit names for lambda parameters to avoid linter confusion
        centroid_dot = Dot(color=RED)
        centroid_dot.add_updater(lambda mobject: mobject.move_to(
            np.mean([p.get_center() for p in pegs], axis=0)
        ))
        
        centroid_text = Text("Centroid", color=RED).scale(0.4)
        centroid_text.add_updater(lambda mobject: mobject.next_to(centroid_dot, UP, buff=0.1))

        dashed_line = always_redraw(lambda: DashedLine(
            centroid_dot.get_center(), 
            target_zone.get_center(), 
            color=GRAY_A
        ))

        self.add(dashed_line, centroid_dot, centroid_text)

        # 3. Reward UI
        # w_2 part uses MAGENTA instead of PINK to ensure compatibility
        formula = MathTex(
            "R = ", "w_1(n_{in})^2", " + ", "w_2(d_{target})^{-1}",
            arg_separator=0.1
        ).scale(0.8).to_edge(DOWN, buff=1)
        
        formula[1].set_color(GREEN)
        formula[3].set_color(MAGENTA)
        
        reward_val = DecimalNumber(0, num_decimal_places=2, color=YELLOW).scale(1.2)
        reward_display = VGroup(Text("Total Reward: "), reward_val).arrange(RIGHT).next_to(formula, UP)

        self.add(formula, reward_display)

        # Constants for calculation
        w1_weight = 1.0
        w2_weight = 2.0
        
        def calculate_total_reward():
            # Distance from target center
            target_pos = target_zone.get_center()
            in_count = 0
            for peg in pegs:
                if np.linalg.norm(peg.get_center() - target_pos) < 1.5:
                    in_count += 1
            
            # Distance from centroid to target center
            dist_to_target = np.linalg.norm(centroid_dot.get_center() - target_pos)
            dist_to_target = max(dist_to_target, 0.1) # Avoid division by zero
            
            return w1_weight * (in_count**2) + w2_weight * (1.0 / dist_to_target)

        # Updater for the reward counter
        reward_val.add_updater(lambda mobject: mobject.set_value(calculate_total_reward()))

        # 4. Animation Sequence
        self.play(Write(formula), Write(reward_display))
        self.wait(1)

        # Move pegs towards the target one by one
        for i, peg_to_move in enumerate(pegs):
            # Generate a target position within the circle
            angle_offset = np.random.uniform(0, 2 * PI)
            radius_offset = np.random.uniform(0, 1.1)
            final_pos = target_zone.get_center() + np.array([
                radius_offset * np.cos(angle_offset), 
                radius_offset * np.sin(angle_offset), 
                0
            ])
            
            self.play(peg_to_move.animate.move_to(final_pos), run_time=1.2)
            
            # Show the squared bonus text pop-up (n^2)
            pop_up = MathTex(f"{i+1}^2", color=GREEN).scale(1.2).next_to(peg_to_move, UP, buff=0.2)
            self.play(FadeIn(pop_up, shift=UP), run_time=0.3)
            self.play(FadeOut(pop_up), run_time=0.3)

        self.wait(2)
