from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Setup Board and Target
        self.setup_board()
        
        target_pos = RIGHT * 4
        target_zone = Circle(radius=1.5, color=YELLOW, fill_opacity=0.2).move_to(target_pos)
        target_label = Text("Target Zone", font_size=24).next_to(target_zone, UP)
        self.add(target_zone, target_label)

        # 2. Setup Pegs (the agent's pieces)
        peg_start_positions = [
            LEFT * 3 + UP * 1.5,
            LEFT * 4 + UP * 0.5,
            LEFT * 3.5 + DOWN * 0.5,
            LEFT * 4.5 + DOWN * 1.5
        ]
        pegs = VGroup(*[Dot(pos, color=BLUE, radius=0.15) for pos in peg_start_positions])
        self.add(pegs)

        # 3. Setup Centroid (Average position of all pegs)
        centroid = Dot(color=RED)
        centroid.add_updater(lambda m: m.move_to(np.mean([p.get_center() for p in pegs], axis=0)))
        
        centroid_label = Text("Centroid", color=RED, font_size=20)
        centroid_label.add_updater(lambda m: m.next_to(centroid, UP, buff=0.1))
        
        # Dashed line from Centroid to Target
        line = always_redraw(lambda: DashedLine(centroid.get_center(), target_pos, color=WHITE))
        
        self.add(centroid, centroid_label, line)

        # 4. UI: Reward components and Formula
        pegs_in_count = ValueTracker(0)
        
        formula = MathTex(
            r"R = w_1(\text{pegs}_{in})^2 + w_2(\text{dist}(\text{centroid, target}))^{-1}",
            font_size=30
        ).to_edge(UP)
        
        reward_val = DecimalNumber(0, num_decimal_places=2, color=GREEN)
        reward_display = VGroup(Text("Total Reward: ", font_size=30), reward_val).arrange(RIGHT)
        reward_display.to_edge(DOWN, buff=0.5)
        
        def update_reward(m):
            dist = np.linalg.norm(centroid.get_center() - target_pos)
            w1, w2 = 1.0, 5.0
            n_in = pegs_in_count.get_value()
            # Formula: w1 * n^2 + w2 * (1/dist)
            current_reward = w1 * (n_in**2) + w2 * (1.0 / max(dist, 0.1))
            m.set_value(current_reward)
            
        reward_val.add_updater(update_reward)
        
        self.add(formula, reward_display)

        # 5. Animation: Move pegs and update metrics
        for i, peg in enumerate(pegs):
            # Calculate a point inside the target zone for this peg
            destination = target_pos + np.array([
                np.random.uniform(-0.6, 0.6), 
                np.random.uniform(-0.6, 0.6), 
                0
            ])
            
            # Use lambda t: t as a safe replacement for the linear rate function
            self.play(
                peg.animate.move_to(destination),
                run_time=2,
                rate_func=lambda t: t
            )
            
            # Increment count of pegs in target
            new_count = i + 1
            pegs_in_count.set_value(new_count)
            
            # Show "Squared Bonus" pop-up
            bonus_text = MathTex(f"{new_count}^2", color=GREEN).scale(1.2).next_to(peg, UR, buff=0.1)
            bonus_label = Text("Bonus!", font_size=20, color=GREEN).next_to(bonus_text, UP, buff=0.05)
            bonus_group = VGroup(bonus_text, bonus_label)
            
            self.play(FadeIn(bonus_group, shift=UP * 0.3), run_time=0.4)
            self.play(FadeOut(bonus_group), run_time=0.4)
            self.wait(0.2)

        self.wait(2)

    def setup_board(self):
        # Create a background grid of faint dots to represent the board
        board_dots = VGroup()
        for x in range(-7, 8):
            for y in range(-4, 5):
                dot = Dot(np.array([x, y, 0]), radius=0.02, color=GRAY, fill_opacity=0.3)
                board_dots.add(dot)
        self.add(board_dots)
