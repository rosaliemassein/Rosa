from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Setup the environment components
        # Target Zone on the right
        target_center = RIGHT * 4 + DOWN * 0.5
        target_zone = Circle(radius=1.5, color=YELLOW, fill_opacity=0.2).move_to(target_center)
        target_label = Text("Target Zone", font_size=24).next_to(target_zone, UP)
        
        # A cluster of blue pegs on the left
        pegs = VGroup(*[
            Dot(color=BLUE).shift(LEFT * 4 + UP * np.random.uniform(-2, 2) + RIGHT * np.random.uniform(-1, 1))
            for _ in range(6)
        ])
        
        # 2. Centroid calculation and visual marker
        centroid_dot = Dot(color=GREEN)
        # Updater to keep the dot at the average position of all pegs
        centroid_dot.add_updater(lambda d: d.move_to(
            np.mean([p.get_center() for p in pegs], axis=0)
        ))
        
        centroid_label = Text("Centroid", color=GREEN, font_size=20)
        centroid_label.add_updater(lambda l: l.next_to(centroid_dot, UP, buff=0.1))
        
        # Dashed line from Centroid to Target Center
        centroid_line = always_redraw(lambda: DashedLine(
            centroid_dot.get_center(), 
            target_zone.get_center(), 
            color=WHITE,
            dash_length=0.1
        ))

        # 3. Reward Display and Formula
        # Formula at the top
        formula = MathTex(
            r"R = w_1(\text{pegs}_{in})^2 + w_2(\text{dist}(\text{centroid, target}))^{-1}", 
            font_size=32
        ).to_edge(UP)
        
        # Manual Counter UI (Avoiding 'Variable' and 'Integer' classes)
        pi_label = Text("Pegs In: ", font_size=24).to_corner(UL).shift(DOWN)
        pi_val = DecimalNumber(0, num_decimal_places=0).next_to(pi_label, RIGHT)
        
        # Update Pegs In count based on distance to target zone
        pi_val.add_updater(lambda m: m.set_value(
            sum(1 for p in pegs if np.linalg.norm(p.get_center() - target_zone.get_center()) < 1.5)
        ))
        
        reward_label = Text("Reward: ", font_size=24).next_to(pi_label, DOWN, aligned_edge=LEFT)
        reward_val = DecimalNumber(0, num_decimal_places=1).next_to(reward_label, RIGHT)
        
        # Update Reward value using the provided formula
        def reward_logic_update(m):
            count = pi_val.get_value()
            dist = np.linalg.norm(centroid_dot.get_center() - target_zone.get_center())
            # R = w1 * count^2 + w2 * (1/dist)
            # w1=10, w2=5 as weights
            current_r = 10 * (count**2) + 5 / (max(dist, 0.1))
            m.set_value(current_r)
        
        reward_val.add_updater(reward_logic_update)
        
        ui_group = VGroup(pi_label, pi_val, reward_label, reward_val)

        # 4. Animation Sequence
        self.add(target_zone, target_label, pegs, centroid_dot, centroid_label, centroid_line, formula, ui_group)
        
        # Narrative hint at the bottom
        narrative = Text(
            "Constant 'gravity' pulls the team, with bonuses for finishing.",
            font_size=20, color=GRAY
        ).to_edge(DOWN)
        self.add(narrative)

        # Move each peg to the target zone one by one
        for i, peg in enumerate(pegs):
            # Calculate a unique spot within the target circle
            angle = np.random.uniform(0, 2 * PI)
            r = np.random.uniform(0, 0.9)
            dest = target_center + np.array([r * np.cos(angle), r * np.sin(angle), 0])
            
            # Animate movement
            self.play(
                peg.animate.move_to(dest),
                run_time=1.2
            )
            
            # Show the "Squared Bonus" pop-up for entering the zone
            current_count = int(pi_val.get_value())
            bonus_text = MathTex(f"{current_count}^2", color=YELLOW).scale(0.8).next_to(peg, UP, buff=0.1)
            self.play(FadeIn(bonus_text, shift=UP*0.2), run_time=0.3)
            self.play(FadeOut(bonus_text), run_time=0.3)

        self.wait(2)
