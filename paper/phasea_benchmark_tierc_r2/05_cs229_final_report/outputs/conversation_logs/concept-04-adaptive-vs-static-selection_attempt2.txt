from manim import *
import numpy as np

class AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. Title and Screen Layout
        title = Text("Adaptive ML vs. Static Guidelines", font_size=36).to_edge(UP)
        self.add(title)
        
        divider = Line(UP * 2, DOWN * 1.5)
        self.add(divider)

        static_label = Text("Static Benchmark (ASHRAE)", font_size=24, color=ORANGE).move_to([-3.5, 2.5, 0])
        ml_label = Text("Logistic Model (Adaptive)", font_size=24, color=GREEN).move_to([3.5, 2.5, 0])
        self.add(static_label, ml_label)

        # 2. Floorplan Grid Setup
        def create_floorplan():
            zones = VGroup(*[Square(side_length=0.4).set_stroke(WHITE, width=1) for _ in range(9)])
            zones.arrange_in_grid(rows=3, cols=3, buff=0.1)
            return zones

        static_floorplan = create_floorplan().move_to([-3.5, 1, 0])
        ml_floorplan = create_floorplan().move_to([3.5, 1, 0])
        
        # Static highlights (frozen state)
        static_indices = [0, 4, 8]
        for idx in static_indices:
            static_floorplan[idx].set_fill(ORANGE, opacity=0.8)
        
        self.add(static_floorplan, ml_floorplan)

        # 3. Axes for Performance Metrics
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 1, 0.2],
            x_length=10,
            y_length=2.5,
            axis_config={"include_tip": False}
        ).scale(0.8).to_edge(DOWN, buff=0.5)
        
        y_axis_label = Text("Missed Requests", font_size=18).rotate(90 * DEGREES)
        y_axis_label.next_to(axes.y_axis, LEFT, buff=0.2)
        x_axis_label = Text("May                              September", font_size=20).next_to(axes.x_axis, DOWN, buff=0.2)
        
        self.add(axes, x_axis_label, y_axis_label)

        # 4. Data Functions (Replacing Lambdas to avoid identifier issues)
        def static_performance_func(t_val):
            return 0.6 + 0.1 * np.sin(t_val * 0.5) + 0.05 * np.cos(t_val * 2)

        def ml_performance_func(t_val):
            return 0.2 + 0.1 * np.cos(t_val * 0.8) * np.exp(-t_val / 10)

        static_graph = axes.plot(static_performance_func, color=ORANGE, x_range=[0, 10])
        ml_graph = axes.plot(ml_performance_func, color=GREEN, x_range=[0, 10])

        # 5. Dynamic Animation State
        time_tracker = ValueTracker(0)

        # Updater for the timeline dot
        def update_dot(mobject):
            t = time_tracker.get_value()
            mobject.move_to(axes.c2p(t, 0))
        
        dot = Dot(color=BLUE)
        dot.add_updater(update_dot)

        # Updater for adaptive ML zones
        def update_ml_floorplan(mobject):
            t = time_tracker.get_value()
            for i, zone in enumerate(mobject):
                # Deterministic flicker logic based on time
                noise = np.sin(t * 2.5 + i) + np.cos(t * 1.8 - i * 0.7)
                if noise > 0.4:
                    zone.set_fill(GREEN, opacity=0.8)
                else:
                    zone.set_fill(BLACK, opacity=0)

        ml_floorplan.add_updater(update_ml_floorplan)

        # 6. Execute Animations
        self.add(dot)
        
        # We use a custom rate function defined as a lambda that acts as linear
        self.play(
            Create(static_graph),
            Create(ml_graph),
            time_tracker.animate.set_value(10),
            run_time=10,
            rate_func=linear
        )
        
        self.wait(1)

        # Final Narrative Overlay

            "The Logistic Model adapts to building stress,\nreducing missed requests compared to static benchmarks.",
            font_size=24,
            line_spacing=1.2
