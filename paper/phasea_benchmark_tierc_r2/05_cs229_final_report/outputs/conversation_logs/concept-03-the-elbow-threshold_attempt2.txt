from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Create Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)

        x_label = axes.get_x_axis_label("Fraction of Zones", edge=DOWN, direction=DOWN, buff=0.3).scale(0.6)
        y_label = axes.get_y_axis_label("Recall", edge=LEFT, direction=LEFT, buff=0.3).scale(0.6)

        # 2. Plot Recall Curve (Concave shape)
        # Function: y = x^(1/4) creates a sharp rise then levels off (representing validation results)
        curve = axes.plot(lambda x: x**(0.25) if x > 0 else 0, x_range=[0, 1], color=BLUE)
        
        # 3. Floorplan Side-panel (Grid of squares)
        grid_group = VGroup(*[
            Square(side_length=0.4, stroke_width=2, color=GRAY) 
            for _ in range(16)
        ]).arrange_in_grid(4, 4, buff=0.1)
        
        floorplan_title = Text("Building Floorplan", font_size=24)
        floorplan_container = VGroup(
            floorplan_title,
            grid_group
        ).arrange(DOWN, buff=0.5).to_edge(RIGHT, buff=0.7)

        # 4. Interactive Components (Tracker and Dot)
        tracker = ValueTracker(0.1)
        
        # Dot on curve
        dot = always_redraw(lambda: Dot(
            axes.c2p(tracker.get_value(), tracker.get_value()**(0.25)), 
            color=YELLOW
        ))
        
        # Vertical and Horizontal projection lines
        v_line = always_redraw(lambda: axes.get_vertical_line(dot.get_center(), color=WHITE, stroke_width=2))
        h_line = always_redraw(lambda: axes.get_horizontal_line(dot.get_center(), color=WHITE, stroke_width=2))

        # 5. Updater for Floorplan Squares
        def update_grid(grid):
            val = tracker.get_value()
            # As val (fraction of zones) goes from 0 to 1, highlight 0 to 16 squares
            num_highlighted = int(val * 16)
            for i, square in enumerate(grid):
                if i < num_highlighted:
                    square.set_fill(YELLOW, opacity=0.8)
                else:
                    square.set_fill(BLACK, opacity=0)

        grid_group.add_updater(update_grid)

        # 6. Animation Sequence
        self.play(Create(axes), Write(x_label), Write(y_label))
        self.play(Create(curve), run_time=2)
        self.play(FadeIn(floorplan_container))
        
        self.add(dot, v_line, h_line)
        
        # Show trade-off by moving the tracker (Linear progression)
        self.play(tracker.animate.set_value(0.9), run_time=4, rate_func=lambda t: t)
        self.wait(0.5)
        self.play(tracker.animate.set_value(0.2), run_time=2)

        # 7. Highlight the 'Elbow'
        # For x^(0.25), the elbow/sweet spot is visually around 0.15-0.25
        elbow_x = 0.2
        elbow_y = elbow_x**(0.25)
        elbow_coord = axes.c2p(elbow_x, elbow_y)
        
        elbow_circle = Circle(radius=0.4, color=RED).move_to(elbow_coord)
        elbow_circle.set_stroke(width=8)
        glow = elbow_circle.copy().set_stroke(width=20, opacity=0.3)
        
        elbow_label = Text("Elbow Point", font_size=24, color=RED).next_to(elbow_circle, UR)
        
        self.play(tracker.animate.set_value(elbow_x))
        self.play(Create(elbow_circle), FadeIn(glow), Write(elbow_label))
        self.wait(1)

        # 8. Formula Display
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=36)
        formula.to_corner(UL, buff=0.5).set_background_stroke(color=BLACK, width=4, opacity=1)
        
        self.play(Write(formula))
        self.wait(3)

        # Clean up
        grid_group.remove_updater(update_grid)
