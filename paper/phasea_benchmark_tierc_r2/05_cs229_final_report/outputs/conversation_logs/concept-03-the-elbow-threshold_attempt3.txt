from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Create Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        labels = axes.get_axis_labels(
            x_label=Text("Fraction Flagged", font_size=20),
            y_label=Text("Recall", font_size=20)
        )

        # 2. Define the Curve (Concave shape)
        def curve_func(x):
            if x <= 0: return 0
            return x**(0.35)
            
        curve = axes.plot(curve_func, x_range=[0, 1], color=BLUE)
        
        # 3. Floorplan on the right (4x4 grid of zones)
        grid = VGroup(*[
            Square(side_length=0.4, stroke_width=2).set_fill(GRAY_E, opacity=0.5)
            for _ in range(16)
        ]).arrange_in_grid(rows=4, cols=4, buff=0.1).to_edge(RIGHT, buff=1)
        
        grid_title = Text("Building Zones", font_size=24).next_to(grid, UP)

        # 4. Value Tracker and Interactive Components
        tracker = ValueTracker(0.05)
        
        # Moving dot on curve
        moving_dot = Dot(color=YELLOW)
        moving_dot.add_updater(
            lambda m: m.move_to(axes.c2p(tracker.get_value(), curve_func(tracker.get_value())))
        )
        
        # Dynamic lines for recall/fraction
        h_line = always_redraw(lambda: axes.get_horizontal_line(moving_dot.get_center(), color=YELLOW, stroke_width=2))
        v_line = always_redraw(lambda: axes.get_vertical_line(moving_dot.get_center(), color=YELLOW, stroke_width=2))

        # 5. Floorplan highlighting logic
        def update_grid(g):
            val = tracker.get_value()
            # Calculate how many squares to light up based on x-axis value (0 to 1)
            num_to_light = int(val * 16)
            for i, sq in enumerate(g):
                if i < num_to_light:
                    sq.set_fill(RED, opacity=0.8)
                else:
                    sq.set_fill(GRAY_E, opacity=0.5)

        grid.add_updater(update_grid)

        # 6. Formula
        formula = MathTex(r"\text{Recall} = \frac{TP}{TP + FN}", font_size=32).to_corner(UL, buff=0.5)
        
        # 7. Animation Sequence
        self.play(Create(axes), Write(labels), Write(grid_title))
        self.play(Create(grid))
        self.play(Create(curve), Write(formula))
        self.add(moving_dot, h_line, v_line)
        
        # Move tracker across the curve to show trade-off
        # Using lambda t: t instead of 'linear' to avoid potential undefined reference errors
        self.play(tracker.animate.set_value(0.9), run_time=4, rate_func=lambda t: t)
        self.wait(0.5)
        self.play(tracker.animate.set_value(0.2), run_time=2)
        
        # 8. Highlight the Elbow Point (Approx 0.2 on x-axis)
        elbow_x = 0.2
        elbow_point = axes.c2p(elbow_x, curve_func(elbow_x))
        
        elbow_highlight = Dot(elbow_point, color=WHITE).scale(2)
        elbow_label = Text("Optimal 'Elbow'", color=YELLOW, font_size=24).next_to(elbow_point, UR, buff=0.5)
        arrow = Arrow(elbow_label.get_bottom(), elbow_point, color=YELLOW, buff=0.1)

        # Define a pulsing circle for the "glow" effect
        ping = Circle(radius=0.1, color=WHITE, stroke_width=2).move_to(elbow_point)

        self.play(
            FadeIn(elbow_highlight),
            Create(arrow),
            Write(elbow_label),
            tracker.animate.set_value(elbow_x)
        )
        
        # Pulsing animation for the elbow
        self.play(
            ping.animate.scale(10).set_opacity(0),
            run_time=1,
            rate_func=lambda t: t
        )

        # Final Narration Text
        narration = Text(
            "Find the 'elbow' where recall is high\nbut flagged zones remain low.", 
            font_size=24,
            color=WHITE
        ).to_edge(DOWN, buff=0.3)
        
        self.play(Write(narration))
        self.wait(3)
