from manim import *
import numpy as np

class Concept03TheElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True, "stroke_width": 2},
        ).to_edge(LEFT, buff=0.7).shift(DOWN * 0.5)

        x_label = Text("Fraction of Zones Flagged", font_size=18).next_to(axes.x_axis, DOWN)
        y_label = Text("Recall", font_size=18).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)
        
        # 2. Plot Concave Curve (Recall vs Fraction)
        # y = 1 - (1-x)^3 creates the 'elbow' effect
        curve = axes.plot(lambda x: 1 - (1 - x)**3, x_range=[0, 1], color=GREEN)
        
        # 3. Value Tracker for interaction
        tracker = ValueTracker(0.1)
        
        # 4. Interactive elements (Dot and lines)
        def get_dot_pos():
            x = tracker.get_value()
            y = 1 - (1 - x)**3
            return axes.c2p(x, y)

        dot = always_redraw(lambda: Dot(get_dot_pos(), color=RED, radius=0.08))
        
        v_line = always_redraw(lambda: DashedLine(
            start=get_dot_pos(),
            end=axes.c2p(tracker.get_value(), 0),
            color=RED, stroke_width=2, opacity=0.6
        ))
        
        # 5. Building Floorplan (Side Panel)
        panel_title = Text("Building Floorplan", font_size=24).to_edge(RIGHT, buff=1.0).shift(UP * 2.5)
        
        # Create a grid of "Zones"
        zones = VGroup(*[
            Square(side_length=0.5, stroke_width=2, color=WHITE)
            for _ in range(16)
        ]).arrange_in_grid(4, 4, buff=0.1).next_to(panel_title, DOWN, buff=0.5)
        
        def zone_updater(m):
            threshold = tracker.get_value()
            num_to_highlight = int(threshold * 16)
            for i, sq in enumerate(m):
                if i < num_to_highlight:
                    sq.set_fill(YELLOW, opacity=0.7)
                    sq.set_stroke(YELLOW, opacity=1.0)
                else:
                    sq.set_fill(BLACK, opacity=0)
                    sq.set_stroke(WHITE, opacity=0.5)

        zones.add_updater(zone_updater)

        # 6. Narration text (Voice)
        voice_str = (
            "We decide on a threshold. Too low: too many zones flagged (inefficient).\n"
            "Too high: we miss critical cooling. The 'elbow' is the sweet spot\n"
            "maximizing catch rate while keeping targeted zones minimal."
        )
        voice_text = Text(voice_str, font_size=16, line_spacing=1.5).to_edge(DOWN, buff=0.3)
        
        # 7. Goal and Formula
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=32).next_to(axes, UP, buff=0.5)
        goal_text = Text("Goal: Balance sensitivity & selectivity", font_size=20, color=BLUE).next_to(formula, RIGHT, buff=1)

        # 8. Execution
        self.add(axes, x_label, y_label, curve, panel_title, zones)
        self.play(Write(formula), Write(goal_text))
        self.add(dot, v_line)
        self.play(Write(voice_text), run_time=2)

        # Move tracker across the curve
        # High threshold (Low fraction flagged, Low recall)
        self.play(tracker.animate.set_value(0.05), run_time=1)
        self.wait(1)
        
        # Moving to low threshold (High fraction flagged, High recall)
        self.play(tracker.animate.set_value(0.95), run_time=4, rate_func=lambda t: t)
        self.wait(1)
        
        # Moving to the 'Elbow'
        elbow_val = 0.3
        self.play(tracker.animate.set_value(elbow_val), run_time=2, rate_func=lambda t: t)
        
        # Highlight the Elbow
        elbow_point = axes.c2p(elbow_val, 1 - (1 - elbow_val)**3)
        glowing_circle = Circle(radius=0.4, color=YELLOW, stroke_width=5).move_to(elbow_point)
        elbow_label = Text("Sweet Spot (Elbow)", font_size=20, color=YELLOW).next_to(glowing_circle, UR, buff=0.1)
        
        self.play(
            Create(glowing_circle),
            Write(elbow_label),
            glowing_circle.animate.scale(1.2).set_stroke(opacity=0.5),
            rate_func=lambda t: np.sin(PI * t), # custom there_and_back
            run_time=1.5
        )
        
        self.wait(3)

        # Final cleanup for nice ending
        self.play(
            FadeOut(voice_text),
            FadeOut(glowing_circle),
            FadeOut(elbow_label),
            goal_text.animate.scale(1.5).move_to(ORIGIN)
        )
        self.wait(1)
