from manim import *
import numpy as np
import random

class AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. Background Layout Setup
        # Divide screen into two halves for the floorplans and a bottom area for the graph
        left_origin = LEFT * 3.5 + UP * 1
        right_origin = RIGHT * 3.5 + UP * 1
        
        static_label = Text("Static Benchmark (ASHRAE)", font_size=28).move_to(left_origin + UP * 2.2)
        ml_label = Text("Logistic Model (Adaptive)", font_size=28).move_to(right_origin + UP * 2.2)
        
        # Floorplan mockups
        static_bg = Rectangle(width=4.5, height=3.5, color=BLUE_E, fill_opacity=0.1).move_to(left_origin)
        ml_bg = Rectangle(width=4.5, height=3.5, color=GREEN_E, fill_opacity=0.1).move_to(right_origin)
        
        floorplan_text_l = Text("Building Floorplan", font_size=16, color=GRAY_B).move_to(static_bg.get_top() + DOWN * 0.3)
        floorplan_text_r = Text("Building Floorplan", font_size=16, color=GRAY_B).move_to(ml_bg.get_top() + DOWN * 0.3)
        
        self.add(static_bg, ml_bg, static_label, ml_label, floorplan_text_l, floorplan_text_r)
        
        # 2. Zone Highlights
        # Left side: Static dots (frozen)
        static_dots = VGroup(*[
            Dot(left_origin + p, color=RED, radius=0.12)
            for p in [UL*1.1, ORIGIN, DR*0.9, LEFT*0.8]
        ])
        
        # Right side: Potential dots for adaptive logic
        ml_dots_coords = [
            np.array([x, y, 0]) 
            for x in np.linspace(-1.5, 1.5, 5) 
            for y in np.linspace(-1, 1, 4)
        ]
        ml_dots = VGroup(*[
            Dot(right_origin + p, color=RED, radius=0.12)
            for p in ml_dots_coords
        ])
        
        self.add(static_dots, ml_dots)
        
        # 3. Timeline and Graph (Bottom)
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 10, 2],
            x_length=10,
            y_length=2.5,
            axis_config={"color": WHITE, "include_tip": False}
        ).to_edge(DOWN, buff=0.4)
        
        # Month labels
        months = ["May", "Jun", "Jul", "Aug", "Sep"]
        month_group = VGroup()
        for i, name in enumerate(months):
            m_text = Text(name, font_size=20).move_to(axes.c2p(i * 2.5, -1))
            month_group.add(m_text)
            
        graph_header = Text("Missed Requests (Recall Error)", font_size=24, color=WHITE).next_to(axes, UP, buff=0.1)
        self.add(axes, month_group, graph_header)
        
        # 4. Animation Logic
        time_tracker = ValueTracker(0)
        
        # Graph lines
        # Static line stays high
        static_line = always_redraw(lambda: axes.plot(
            lambda x: 7 + 0.6 * np.sin(x * 2),
            x_range=[0, max(0.01, time_tracker.get_value())],
            color=BLUE
        ))
        
        # ML line stays low
        ml_line = always_redraw(lambda: axes.plot(
            lambda x: 2 + 0.4 * np.cos(x * 3),
            x_range=[0, max(0.01, time_tracker.get_value())],
            color=GREEN
        ))
        
        # Legend
        leg_s = Line(LEFT*0.3, RIGHT*0.3, color=BLUE).move_to(axes.c2p(8, 9))
        leg_s_t = Text("Static", font_size=18).next_to(leg_s, RIGHT)
        leg_m = Line(LEFT*0.3, RIGHT*0.3, color=GREEN).move_to(axes.c2p(8, 8))
        leg_m_t = Text("Adaptive", font_size=18).next_to(leg_m, RIGHT)
        
        self.add(static_line, ml_line, leg_s, leg_s_t, leg_m, leg_m_t)
        
        # Adaptive dot updater: changing count and flickering
        def update_ml_dots(dots):
            t = time_tracker.get_value()
            # The "heat" intensity determines how many dots are flagged
            intensity = (np.sin(t * 0.8) + 1) / 2 # ranges 0 to 1
            for i, dot in enumerate(dots):
                # Unique frequency for each dot based on its index
                seed = (i * 1.5 + t * 4)
                if (np.sin(seed) + 1) / 2 < (intensity * 0.7 + 0.2):
                    dot.set_opacity(1)
                else:
                    dot.set_opacity(0)
                    
        ml_dots.add_updater(update_ml_dots)
        
        # Indicator line moving across timeline
        indicator = always_redraw(lambda: Line(
            axes.c2p(time_tracker.get_value(), 0),
            axes.c2p(time_tracker.get_value(), 10),
            color=YELLOW, stroke_width=2
        ))
        self.add(indicator)
        
        # 5. Execute
        self.play(
            time_tracker.animate.set_value(10),
            run_time=12,
            rate_func=lambda t: t # linear rate function
        )
        self.wait(2)

# Helper function if specifically required by the logic, though axes.plot is preferred
def get_custom_polyline(x_coords, y_coords):
    points = [np.array([x, y, 0]) for x, y in zip(x_coords, y_coords)]
    return Polyline(*points)
