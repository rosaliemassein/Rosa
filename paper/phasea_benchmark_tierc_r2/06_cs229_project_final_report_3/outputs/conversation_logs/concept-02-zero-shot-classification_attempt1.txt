from manim import *
import numpy as np

class ZeroShotClassification(Scene):
    def construct(self):
        # 1. Image Representation
        # Use a placeholder square if image is missing, otherwise ImageMobject
        try:
            image_obj = ImageMobject("img-8.jpeg").scale(1.2).to_edge(LEFT, buff=0.5)
        except:
            image_obj = Rectangle(color=WHITE, height=3, width=3).to_edge(LEFT, buff=0.5)
            image_obj.add(Text("Image", font_size=20))
        
        self.play(FadeIn(image_obj))
        
        # 2. Vector v
        v_label = MathTex(r"\mathbf{v}", color=BLUE).next_to(image_obj, RIGHT, buff=0.3)
        # Manually construct matrix look using MathTex to avoid 'Matrix' identifier issues
        v_bracket = MathTex(r"\begin{bmatrix} 0.12 \\ 0.85 \\ -0.4 \\ \vdots \end{bmatrix}").next_to(v_label, RIGHT)
        v_group = VGroup(v_label, v_bracket)
        
        self.play(Write(v_group))
        self.wait(1)

        # 3. Matrix X (Simplified)
        x_label = MathTex(r"\mathbf{X}").shift(UP * 2.5 + RIGHT * 3.5)
        
        countries = ["USA", "France", "Japan", "Brazil", "India"]
        # Create matrix rows: Text + Vector brackets
        x_rows = VGroup()
        for i, name in enumerate(countries):
            row_text = Text(name, font_size=24)
            row_vec = MathTex(r"\left[ \dots \text{embedding} \dots \right]").scale(0.6)
            row = VGroup(row_text, row_vec).arrange(RIGHT, buff=0.2)
            x_rows.add(row)
        
        x_rows.arrange(DOWN, buff=0.3).next_to(x_label, DOWN)
        x_dots = MathTex(r"\vdots").next_to(x_rows, DOWN, buff=0.1)
        matrix_container = VGroup(x_label, x_rows, x_dots).scale(0.8)
        
        self.play(FadeIn(matrix_container))
        self.wait(1)

        # 4. Dot Product Pulsing Highlight
        # Highlight France (index 1) and the vector v
        highlight_rect_x = SurroundingRectangle(x_rows[1], color=YELLOW)
        highlight_rect_v = SurroundingRectangle(v_bracket, color=YELLOW)
        
        self.play(Create(highlight_rect_x), Create(highlight_rect_v))
        # Pulse effect using simple scale animation
        self.play(
            highlight_rect_x.animate.scale(1.1), 
            highlight_rect_v.animate.scale(1.1),
            rate_func=lambda t: np.sin(PI * t), # custom pulse
            run_time=1.5
        )
        self.play(FadeOut(highlight_rect_x), FadeOut(highlight_rect_v))

        # 5. MathTex Formula
        formula = MathTex(
            r"h(\mathbf{v};\mathbf{X})=\frac{\exp(\mathbf{X}\mathbf{v})}{\sum_{j=1}^{N}\exp(\mathbf{X}_{j}^{T}\mathbf{v})}",
            font_size=34
        ).to_edge(DOWN, buff=0.5)
        
        self.play(Write(formula))
        self.wait(2)

        # 6. Bar Chart (Manual Construction using Axes and Rectangles)
        self.play(
            FadeOut(matrix_container),
            FadeOut(v_group),
            FadeOut(image_obj)
        )
        
        axes = Axes(
            x_range=[0, 5, 1],
            y_range=[0, 1, 0.2],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": False}
        ).shift(UP * 0.5)
        
        labels = VGroup(*[
            Text(countries[i], font_size=18).next_to(axes.c2p(i + 0.5, 0), DOWN)
            for i in range(5)
        ])
        
        # Initial low bars
        bar_heights = [0.1, 0.2, 0.15, 0.05, 0.1]
        bars = VGroup(*[
            Rectangle(
                width=0.5, 
                height=axes.get_top()[1] - axes.get_bottom()[1], 
                fill_opacity=0.8,
                color=BLUE
            ).scale(h, about_edge=DOWN).move_to(axes.c2p(i + 0.5, 0), DOWN)
            for i, h in enumerate(bar_heights)
        ])
        
        self.play(Create(axes), Write(labels))
        self.play(FadeIn(bars))
        self.wait(1)
        
        # Update bars (Softmax prediction: France shoots up)
        # France is index 1
        new_heights = [0.05, 0.9, 0.02, 0.01, 0.02]
        
        animations = []
        for i, bar in enumerate(bars):
            # Target height in coordinate units
            target_h = new_heights[i] * (axes.y_axis.get_length())
            if i == 1: # France
                animations.append(bar.animate.stretch_to_fit_height(target_h, about_edge=DOWN).set_color(GOLD))
            else:
                animations.append(bar.animate.stretch_to_fit_height(target_h, about_edge=DOWN))
                
        self.play(*animations, run_time=2)
        
        # Final prediction text
        pred_text = Text("Prediction: France", color=GOLD).next_to(axes, UP)
        self.play(Write(pred_text))
        self.wait(3)
