from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Formula display
        formula = MathTex(r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})")
        formula.to_edge(UP)
        self.add(formula)

        # 2. Neural Network Blocks (Encoders)
        image_encoder = VGroup(
            Rectangle(height=1.5, width=2.5, color=GREEN),
            Text("Image Encoder", font_size=24)
        ).shift(LEFT * 4 + UP * 1.5)
        
        text_encoder = VGroup(
            Rectangle(height=1.5, width=2.5, color=BLUE),
            Text("Text Encoder", font_size=24)
        ).shift(LEFT * 4 + DOWN * 1.5)

        self.play(Create(image_encoder), Create(text_encoder))

        # 3. Inputs (Street View and Text)
        street_view_img = Square(side_length=0.8, fill_opacity=0.8, color=GRAY).next_to(image_encoder, LEFT, buff=0.5)
        img_label = Text("Street View", font_size=16).next_to(street_view_img, DOWN)
        
        france_text = Text("'A Street View photo in France'", color=RED, font_size=18).next_to(text_encoder, LEFT, buff=0.5)

        self.play(FadeIn(street_view_img, img_label), FadeIn(france_text))

        # 4. Shared Embedding Space (2D Axes)
        axes = Axes(
            x_range=[-1.2, 1.2, 1], y_range=[-1.2, 1.2, 1],
            x_length=4, y_length=4,
            axis_config={"include_tip": True}
        ).shift(RIGHT * 3)
        axes_label = Text("Shared Embedding Space", font_size=20).next_to(axes, UP)
        self.play(Create(axes), Write(axes_label))

        # 5. Vectors and ValueTrackers
        # We track the angles of the image and text vectors
        angle_image = ValueTracker(70 * DEGREES)
        angle_text = ValueTracker(20 * DEGREES)
        similarity_val = ValueTracker(0.4) # Starting similarity

        def get_embedding_vector(angle_val, vector_color):
            return Arrow(
                axes.get_origin(),
                axes.c2p(np.cos(angle_val), np.sin(angle_val)),
                buff=0, 
                color=vector_color,
                stroke_width=6
            )

        # Creating the vector objects
        image_vec = get_embedding_vector(angle_image.get_value(), GREEN)
        text_vec = get_embedding_vector(angle_text.get_value(), BLUE)

        # Similarity Score Display
        score_label = Text("Similarity:", font_size=24).to_edge(DOWN).shift(RIGHT * 2)
        score_num = DecimalNumber(similarity_val.get_value(), color=YELLOW, num_decimal_places=2).next_to(score_label, RIGHT)

        # Add Updaters for continuous movement
        image_vec.add_updater(lambda mob: mob.become(get_embedding_vector(angle_image.get_value(), GREEN)))
        text_vec.add_updater(lambda mob: mob.become(get_embedding_vector(angle_text.get_value(), BLUE)))
        score_num.add_updater(lambda mob: mob.set_value(similarity_val.get_value()))

        # Show initial vectors and score
        self.play(
            FadeIn(image_vec),
            FadeIn(text_vec),
            FadeIn(score_label),
            FadeIn(score_num)
        )
        self.wait(0.5)

        # 6. Animate Alignment (Rotation and Score Increase)
        self.play(
            angle_image.animate.set_value(45 * DEGREES),
            angle_text.animate.set_value(45 * DEGREES),
            similarity_val.animate.set_value(1.00),
            run_time=2
        )
        self.wait(1)

        # 7. Contrastive Step: Show "wrong" label
        japan_text = Text("'A Street View photo in Japan'", color=ORANGE, font_size=18).move_to(france_text)
        
        self.play(
            ReplacementTransform(france_text, japan_text),
            similarity_val.animate.set_value(0.5)
        )

        # 8. Push "wrong" vector away
        self.play(
            angle_text.animate.set_value(-30 * DEGREES),
            similarity_val.animate.set_value(-0.25),
            run_time=2
        )
        self.wait(2)

        # Clean up
        image_vec.clear_updaters()
        text_vec.clear_updaters()
        score_num.clear_updaters()
