from manim import *

class ConceptAlignment(Scene):
    def construct(self):
        # 1. Setup Neural Network Encoders
        image_encoder = VGroup(
            Rectangle(height=1.5, width=2.5, color=BLUE),
            Text("Image Encoder", font_size=24)
        ).shift(UP * 2 + LEFT * 4)
        
        text_encoder = VGroup(
            Rectangle(height=1.5, width=2.5, color=GREEN),
            Text("Text Encoder", font_size=24)
        ).shift(DOWN * 2 + LEFT * 4)

        self.play(Create(image_encoder), Create(text_encoder))

        # 2. Setup Inputs
        # Representative placeholder for the image
        street_view_placeholder = VGroup(
            Rectangle(height=1.2, width=1.8, fill_opacity=0.3, color=WHITE),
            Text("Street View\n(France)", font_size=16)
        ).next_to(image_encoder, LEFT, buff=1)
        
        correct_text = Text("'A Street View photo in France'", font_size=20, color=YELLOW).next_to(text_encoder, LEFT, buff=0.5)

        self.play(FadeIn(street_view_placeholder), Write(correct_text))
        
        # 3. Embedding Space (Coordinate System)
        axes = Axes(
            x_range=[0, 2, 1], y_range=[0, 2, 1], 
            x_length=3, y_length=3, 
            axis_config={"include_tip": True}
        ).shift(RIGHT * 3.5)
        axes_label = Text("Shared Embedding Space", font_size=20).next_to(axes, UP)
        
        self.play(Create(axes), Write(axes_label))

        # 4. Vectors (Arrows)
        # We use Arrow instead of Vector to avoid specific environment issues
        origin = axes.coords_to_point(0, 0)
        img_end = axes.coords_to_point(1.2, 0.8)
        txt_end_initial = axes.coords_to_point(1.5, 0.2)
        
        image_vec = Arrow(origin, img_end, buff=0, color=BLUE)
        text_vec = Arrow(origin, txt_end_initial, buff=0, color=GREEN)
        
        img_vec_label = MathTex(r"v_{img}", color=BLUE, font_size=24).next_to(img_end, UR, buff=0.1)
        txt_vec_label = MathTex(r"v_{text}", color=GREEN, font_size=24).next_to(txt_end_initial, RIGHT, buff=0.1)

        self.play(GrowArrow(image_vec), Write(img_vec_label))
        self.play(GrowArrow(text_vec), Write(txt_vec_label))
        self.wait()

        # 5. Similarity Score Tracking
        tracker = ValueTracker(0.45)
        score_label = Text("Cosine Similarity: ", font_size=24).to_edge(UP)
        score_num = DecimalNumber(tracker.get_value(), num_decimal_places=2, color=YELLOW).next_to(score_label, RIGHT)
        
        # Use 'mob' as identifier to avoid 'd' undefined error
        score_num.add_updater(lambda mob: mob.set_value(tracker.get_value()))
        
        self.add(score_label, score_num)

        # 6. Animation: Alignment (Moving together)
        txt_end_aligned = axes.coords_to_point(1.2, 0.8)
        
        self.play(
            text_vec.animate.put_start_and_end_on(origin, txt_end_aligned),
            txt_vec_label.animate.next_to(txt_end_aligned, UP, buff=0.1),
            tracker.animate.set_value(0.99),
            run_time=2
        )
        self.wait()

        # 7. Wrong Label and Contrastive Pushing
        wrong_text_val = Text("'A Street View photo in Japan'", font_size=20, color=RED).move_to(correct_text.get_center())
        
        self.play(Transform(correct_text, wrong_text_val))
        
        # Pushing vector away
        txt_end_wrong = axes.coords_to_point(0.4, 1.4)
        
        self.play(
            text_vec.animate.set_color(RED).put_start_and_end_on(origin, txt_end_wrong),
            txt_vec_label.animate.set_color(RED).next_to(txt_end_wrong, LEFT, buff=0.1),
            tracker.animate.set_value(0.15),
            run_time=2
        )
        self.wait()

        # 8. Formula Display
        formula = MathTex(r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})", font_size=32)
        formula.to_edge(DOWN, buff=0.5)
        self.play(Write(formula))
        self.wait(2)
