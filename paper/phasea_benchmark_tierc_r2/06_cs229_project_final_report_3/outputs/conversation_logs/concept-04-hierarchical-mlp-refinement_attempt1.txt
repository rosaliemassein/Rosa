from manim import *
import numpy as np

class HierarchicalMLPRefinement(Scene):
    def construct(self):
        # 1. Define Data from JSON
        voice_str = "The final step is the refinement stage. Once CLIP tells us we are in a specific country..."
        goal_str = "Understand the two-stage inference pipeline."
        remarks_str = "Hierarchical approach: Global to local."
        formula_str = r"c^{(i)}:=\arg\min_{j}\|x^{(i)}-\mu_{j}\|^{2}"
        coord_str = r"(\text{lat}, \text{lon})"

        # 2. Input Stage: Image and CLIP
        image_box = Square(color=WHITE, fill_opacity=0.1).scale(0.7)
        image_label = Text("Input Image", font_size=20).next_to(image_box, UP)
        image_group = VGroup(image_box, image_label).to_edge(LEFT, buff=0.8).shift(UP * 1.5)

        clip_box = Rectangle(height=0.8, width=1.5, color=BLUE)
        clip_text = Text("CLIP", font_size=24, color=BLUE).move_to(clip_box)
        clip_group = VGroup(clip_box, clip_text).next_to(image_group, RIGHT, buff=1)

        arrow_to_clip = Arrow(image_group.get_right(), clip_group.get_left(), buff=0.1)

        # 3. Refinement Stage: MLP
        # Create 3 layers of neurons
        l1 = VGroup(*[Circle(radius=0.1, color=YELLOW, fill_opacity=0.8) for _ in range(4)]).arrange(DOWN, buff=0.15)
        l2 = VGroup(*[Circle(radius=0.1, color=YELLOW, fill_opacity=0.8) for _ in range(5)]).arrange(DOWN, buff=0.15)
        l3 = VGroup(*[Circle(radius=0.1, color=YELLOW, fill_opacity=0.8) for _ in range(3)]).arrange(DOWN, buff=0.15)
        mlp_layers = VGroup(l1, l2, l3).arrange(RIGHT, buff=0.7)
        
        mlp_rect = SurroundingRectangle(mlp_layers, color=GRAY, buff=0.2)
        mlp_label = Text("Country MLP", font_size=20).next_to(mlp_rect, UP)
        mlp_group = VGroup(mlp_rect, mlp_layers, mlp_label).next_to(clip_group, RIGHT, buff=1.2)

        arrow_to_mlp = Arrow(clip_group.get_right(), mlp_group.get_left(), buff=0.1)

        # Connections for the MLP
        connections = VGroup()
        for layer_a, layer_b in [(l1, l2), (l2, l3)]:
            for n1 in layer_a:
                for n2 in layer_b:
                    line = Line(n1.get_center(), n2.get_center(), stroke_width=1, stroke_opacity=0.3)
                    connections.add(line)

        # 4. Map Stage
        map_box = Rectangle(width=5, height=2.5, color=DARK_GREEN).to_edge(DOWN, buff=0.4)
        map_text = Text("Regional Clusters", font_size=18).next_to(map_box, UP, buff=0.1)
        
        # Centroids
        centroids = VGroup(*[
            Dot(map_box.get_center() + np.array([x, y, 0]), radius=0.06, color=WHITE)
            for x, y in [(-1.2, 0.4), (0.8, -0.5), (1.5, 0.2), (-0.4, -0.6), (0.1, 0.7)]
        ])
        target_centroid = centroids[1]

        # GPS Pin
        pin = VGroup(
            Line(ORIGIN, UP * 0.4, color=RED, stroke_width=4),
            Dot(UP * 0.4, color=RED, radius=0.08)
        )
        pin.move_to(target_centroid.get_center(), aligned_edge=DOWN)

        # 5. Formula and Animation
        formula_tex = MathTex(formula_str).scale(0.7).to_corner(UR, buff=0.5)
        coord_tex_obj = MathTex(coord_str).scale(0.8).move_to(pin.get_top() + UP * 0.4)

        # Animation Sequence
        self.play(FadeIn(image_group), Write(formula_tex))
        self.wait(0.5)

        self.play(Create(arrow_to_clip), FadeIn(clip_group))
        self.play(Create(arrow_to_mlp), Create(mlp_rect), FadeIn(mlp_layers), Write(mlp_label))
        self.play(Create(connections))

        # Signals flowing
        flow = connections.copy().set_color(WHITE).set_stroke(width=2, opacity=0.8)
        self.play(Create(flow), run_time=1)
        self.play(FadeOut(flow))

        self.play(Create(map_box), Write(map_text), Create(centroids))
        
        # Pointer to centroid
        pointer = Arrow(mlp_group.get_right(), target_centroid.get_center(), color=YELLOW, buff=0.1)
        self.play(Create(pointer))
        self.play(target_centroid.animate.set_color(YELLOW).scale(1.5))

        # Pin drop
        pin.shift(UP * 1.5)
        self.play(pin.animate.shift(DOWN * 1.5), rate_func=rush_into, run_time=0.6)
        
        # Transition from formula to coordinates
        self.play(ReplacementTransform(formula_tex, coord_tex_obj))
        
        # Goal/Voice text display at bottom
        info_text = Text(goal_str, font_size=22, color=BLUE).to_edge(UP, buff=0.2)
        self.play(Write(info_text))
        
        self.wait(2)
