from manim import *

class ContrastiveAlignment(Scene):
    def construct(self):
        # 1. Setup UI elements
        img_encoder = VGroup(
            Rectangle(height=1.2, width=2.2, color=BLUE),
            Text("Image Encoder", font_size=18)
        ).shift(LEFT * 4 + UP * 2.5)
        
        txt_encoder = VGroup(
            Rectangle(height=1.2, width=2.2, color=GREEN),
            Text("Text Encoder", font_size=18)
        ).shift(RIGHT * 4 + UP * 2.5)

        # Image placeholder (since img-7.jpeg might not exist in current dir)
        try:
            street_view_img = ImageMobject("img-7.jpeg").scale(0.5).next_to(img_encoder, LEFT, buff=0.5)
        except:
            street_view_img = Square(side_length=1.2, fill_opacity=0.5, color=GREY).next_to(img_encoder, LEFT, buff=0.5)
            street_view_img.add(Text("Photo", font_size=14).move_to(street_view_img))

        label_france = Text("A Street View photo in France", font_size=16, color=BLUE).next_to(txt_encoder, RIGHT, buff=0.5)
        label_japan = Text("A Street View photo in Japan", font_size=16, color=YELLOW).next_to(txt_encoder, RIGHT, buff=0.5)

        # Embedding Space
        axes = Axes(
            x_range=[-2, 2, 1],
            y_range=[-2, 2, 1],
            x_length=4,
            y_length=4,
            axis_config={"include_tip": True}
        ).shift(DOWN * 0.5)
        space_label = Text("Shared Embedding Space", font_size=18).next_to(axes, UP)

        # Vectors at Origin
        origin = axes.get_origin()
        image_vec = Vector(axes.c2p(1.2, 0.4) - origin, color=RED).shift(origin)
        text_vec = Vector(axes.c2p(0.4, 1.2) - origin, color=BLUE).shift(origin)
        
        # Similarity Value
        sim_tracker = ValueTracker(0.32)
        sim_box = VGroup(
            Text("Cosine Similarity:", font_size=20),
            DecimalNumber(0.32, num_decimal_places=2, color=GREEN)
        ).arrange(RIGHT).to_edge(LEFT, buff=0.5).shift(DOWN * 3)
        sim_box[1].add_updater(lambda d: d.set_value(sim_tracker.get_value()))

        # Formula
        formula = MathTex(r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})", font_size=24).to_edge(RIGHT, buff=0.5).shift(DOWN * 3)

        # 2. Narration/Instruction Box
        narration = Text(
            "Aligning visual features with geographic concepts using CLIP.",
            font_size=18
        ).to_edge(BOTTOM, buff=0.3)

        # Start Animation
        self.add(img_encoder, txt_encoder, space_label, axes)
        self.play(FadeIn(street_view_img), FadeIn(label_france), FadeIn(narration))
        self.wait(1)

        # 3. Feeding into encoders
        self.play(
            street_view_img.animate.scale(0.1).move_to(img_encoder.get_center()).set_opacity(0),
            label_france.animate.scale(0.1).move_to(txt_encoder.get_center()).set_opacity(0),
            run_time=1
        )

        # 4. Generate Vectors
        self.play(
            GrowArrow(image_vec), 
            GrowArrow(text_vec), 
            FadeIn(sim_box), 
            FadeIn(formula),
            narration.animate.become(Text("Vectors represent mathematical descriptions in a shared space.", font_size=18).to_edge(BOTTOM))
        )
        self.wait(1)

        # 5. Rotate to Align (Maximize similarity)
        aligned_dir = axes.c2p(1.1, 1.1) - origin
        self.play(
            image_vec.animate.put_start_and_end_on(origin, origin + aligned_dir),
            text_vec.animate.put_start_and_end_on(origin, origin + aligned_dir),
            sim_tracker.animate.set_value(1.00),
            narration.animate.become(Text("Training maximizes cosine similarity for matching pairs.", font_size=18).to_edge(BOTTOM)),
            run_time=2
        )
        self.wait(1)

        # 6. Push away "Wrong" label
        self.play(
            FadeOut(label_france),
            FadeIn(label_japan.scale(1.0).next_to(txt_encoder, RIGHT, buff=0.5)),
            narration.animate.become(Text("Pushing away incorrect labels like 'Japan'.", font_size=18).to_edge(BOTTOM))
        )
        
        wrong_dir = axes.c2p(-1.2, 0.2) - origin
        self.play(
            text_vec.animate.put_start_and_end_on(origin, origin + wrong_dir),
            sim_tracker.animate.set_value(-0.55),
            run_time=2
        )
        self.wait(2)
