from manim import *

class YOLOLocalizationLoss(Scene):
    def construct(self):
        # 1. Background Image Placeholder
        try:
            trash_image = ImageMobject("img-11.jpeg").scale(2)
        except:
            trash_image = Rectangle(width=10, height=7, color=DARK_GRAY, fill_opacity=0.5)
            trash_image.add(Text("Image: Multiple Trash Items", font_size=24))
        self.add(trash_image)

        # 2. 7x7 Grid (NumberPlane)
        grid = NumberPlane(
            x_range=[-3.5, 3.5, 1],
            y_range=[-3.5, 3.5, 1],
            x_length=7,
            y_length=7,
            background_line_style={
                "stroke_color": WHITE,
                "stroke_width": 1,
                "stroke_opacity": 0.4
            }
        )
        self.play(Create(grid))

        # 3. Highlight specific cell
        # Grid center at (0,0), cell (1,1) is top-right quadrant
        cell_coords = grid.coords_to_point(1, 1)
        highlight_cell = Rectangle(width=1, height=1, color=YELLOW, fill_opacity=0.3)
        highlight_cell.move_to(cell_coords)
        
        self.play(Create(highlight_cell))
        self.wait(0.5)

        # 4. Bounding Boxes
        # Ground Truth (Dashed via multiple DashedLines for compatibility)
        gt_center = grid.coords_to_point(1.4, 0.8)
        gt_w, gt_h = 2.4, 1.6
        gt_rect = Rectangle(width=gt_w, height=gt_h).move_to(gt_center)
        
        gt_dashed = VGroup(
            DashedLine(gt_rect.get_corner(UL), gt_rect.get_corner(UR), color=RED),
            DashedLine(gt_rect.get_corner(UR), gt_rect.get_corner(DR), color=RED),
            DashedLine(gt_rect.get_corner(DR), gt_rect.get_corner(DL), color=RED),
            DashedLine(gt_rect.get_corner(DL), gt_rect.get_corner(UL), color=RED)
        )
        
        # Predicted Box (Solid)
        pred_center = grid.coords_to_point(0.7, 1.3)
        pred_w, pred_h = 1.6, 2.2
        pred_box = Rectangle(width=pred_w, height=pred_h, color=BLUE).move_to(pred_center)
        
        self.play(Create(gt_dashed), Create(pred_box))

        # 5. Coordinate and Dimension Arrows
        # Center difference (x, y)
        arrow_x = Arrow(pred_box.get_center(), [gt_center[0], pred_center[1], 0], color=GREEN, buff=0)
        arrow_y = Arrow([gt_center[0], pred_center[1], 0], gt_center, color=ORANGE, buff=0)
        
        # Width difference indicator (Double-headed arrow using add_tip)
        arrow_w = Arrow(pred_box.get_right(), gt_rect.get_right(), color=PURPLE, buff=0)
        arrow_w.add_tip(at_start=True)

        self.play(Create(arrow_x), Create(arrow_y))
        self.play(Create(arrow_w))
        self.wait(1)

        # 6. LaTeX Loss Formula
        formula_str = r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right) + \sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)"
        loss_formula = MathTex(formula_str).scale(0.55).to_edge(UP, buff=0.3)
        
        # Background for text legibility
        formula_bg = Rectangle(
            width=loss_formula.width + 0.5, 
            height=loss_formula.height + 0.4, 
            fill_color=BLACK, 
            fill_opacity=0.8, 
            stroke_width=0
        ).move_to(loss_formula)
        
        self.play(FadeIn(formula_bg), Write(loss_formula))

        # 7. Mock Training Animation (Aligning boxes)
        self.play(
            pred_box.animate.move_to(gt_center).stretch_to_fit_width(gt_w).stretch_to_fit_height(gt_h),
            FadeOut(arrow_x), 
            FadeOut(arrow_y), 
            FadeOut(arrow_w),
            run_time=2.5
        )
        self.wait(1)

        # 8. Narration Text
        narration_str = (
            "YOLOv10 divides the image into a grid. Cells predict boxes.\n"
            "Loss penalizes errors in (x,y) and square roots of (w,h).\n"
            "Sqrt ensures small object errors are penalized more heavily."
        )
        narration = Text(narration_str, font_size=20, color=WHITE).to_edge(DOWN, buff=0.5)
        narration_bg = Rectangle(
            width=narration.width + 0.5, 
            height=narration.height + 0.4, 
            fill_color=BLACK, 
            fill_opacity=0.8, 
            stroke_width=0
        ).move_to(narration)

        self.play(FadeIn(narration_bg), Write(narration))
        
        # Highlight coordinate terms in formula (approximate position)
        highlight_rect = SurroundingRectangle(loss_formula, color=YELLOW, buff=0.1)
        self.play(Create(highlight_rect))
        self.wait(3)
