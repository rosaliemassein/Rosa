from manim import *
import random

class Concept02SyntheticPileGeneration(Scene):
    def construct(self):
        # Create a blank white Rectangle representing the canvas
        canvas = Rectangle(color=WHITE, width=6, height=6)
        canvas.set_fill(WHITE, opacity=0.1)
        canvas_label = Text("Synthetic Canvas", font_size=24).next_to(canvas, UP)
        self.add(canvas, canvas_label)

        # 5 Segmented trash icons (prototypes) floating on the left
        prototypes = VGroup(
            Circle(color=RED, fill_opacity=0.8),
            Square(color=BLUE, fill_opacity=0.8),
            Triangle(color=GREEN, fill_opacity=0.8),
            RoundedRectangle(color=ORANGE, fill_opacity=0.8, corner_radius=0.1),
            Star(color=PURPLE, fill_opacity=0.8)
        ).scale(0.3)

        for i, icon in enumerate(prototypes):
            icon.move_to(LEFT * 5 + UP * (2 - i))
        
        self.add(prototypes)

        # ValueTrackers to display the current sampled parameters
        scale_tracker = ValueTracker(1.0)
        rot_tracker = ValueTracker(0.0)

        scale_text = Text("Scale: ", font_size=20)
        scale_num = DecimalNumber(1.0, num_decimal_places=2).next_to(scale_text, RIGHT)
        scale_num.add_updater(lambda d: d.set_value(scale_tracker.get_value()))
        
        rot_text = Text("Rotation: ", font_size=20)
        rot_num = DecimalNumber(0.0, num_decimal_places=1, unit="^\\circ").next_to(rot_text, RIGHT)
        rot_num.add_updater(lambda d: d.set_value(rot_tracker.get_value()))
        
        tracker_ui = VGroup(
            VGroup(scale_text, scale_num),
            VGroup(rot_text, rot_num)
        )
        tracker_ui[0].arrange(RIGHT)
        tracker_ui[1].arrange(RIGHT)
        tracker_ui.arrange(DOWN, aligned_edge=LEFT)
        tracker_ui.to_corner(UR).shift(LEFT * 0.5)
        self.add(tracker_ui)

        # Process: Generate 9 objects by sampling from prototypes
        random.seed(42)
        pile = VGroup()
        for i in range(9):
            # Select a prototype icon
            proto_index = i % len(prototypes)
            item = prototypes[proto_index].copy()
            
            # Sample random transformation parameters
            sampled_scale = random.uniform(0.6, 1.5)
            sampled_rotation = random.uniform(0, 360)
            sampled_position = [
                random.uniform(-2.2, 2.2),
                random.uniform(-2.2, 2.2),
                0
            ]
            
            # Update the UI trackers to reflect sampling
            self.play(
                scale_tracker.animate.set_value(sampled_scale),
                rot_tracker.animate.set_value(sampled_rotation),
                run_time=0.25
            )
            
            # Move the item onto the canvas with transforms
            # Set Z-index to ensure correct layering/overlap
            item.set_z_index(i)
            self.play(
                item.animate.move_to(sampled_position)
                .scale(sampled_scale)
                .rotate(sampled_rotation * DEGREES),
                run_time=0.6
            )
            pile.add(item)

        self.wait(2)
        
        # Highlight the finished pile
        self.play(pile.animate.scale(1.1), run_time=0.5)
        self.play(pile.animate.scale(1/1.1), run_time=0.5)
        self.wait(1)
