import numpy as np
from manim import *

class YOLOV10LocalizationLoss(Scene):
    def construct(self):
        # 1. Formula Definition
        formula_text = r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right) + \sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)"
        
        # 2. Title
        title = Text("YOLOv10 Localization Loss").scale(0.8).to_edge(UP)
        self.play(Write(title))

        # 3. Image Placeholder and Grid
        try:
            trash_image = ImageMobject("img-11.jpeg")
        except:
            trash_image = Rectangle(width=6, height=4, color=GRAY, fill_opacity=0.3)
        
        trash_image.scale_to_fit_height(4).shift(LEFT * 3)
        
        # 7x7 Grid
        grid = NumberPlane(
            x_range=[0, 7, 1],
            y_range=[0, 7, 1],
            x_length=trash_image.width,
            y_length=trash_image.height,
            background_line_style={"stroke_color": WHITE, "stroke_width": 1, "stroke_opacity": 0.5}
        ).move_to(trash_image)

        self.play(FadeIn(trash_image), Create(grid))

        # 4. Highlight specific cell
        cell_w = trash_image.width / 7
        cell_h = trash_image.height / 7
        cell_origin = trash_image.get_corner(DL)
        target_cell_center = cell_origin + np.array([4.5 * cell_w, 3.5 * cell_h, 0])
        
        highlight_cell = Rectangle(
            width=cell_w, height=cell_h, 
            color=YELLOW, stroke_width=4
        ).move_to(target_cell_center)
        
        self.play(Create(highlight_cell))

        # 5. Boxes (Predicted and Ground Truth)
        gt_box = Rectangle(
            width=cell_w * 1.5, height=cell_h * 2.2, 
            color=BLUE
        ).set_stroke(dash_pattern=[0.1, 0.1]).move_to(target_cell_center + RIGHT * 0.2 + UP * 0.1)
        
        pred_proxy = Dot(target_cell_center - RIGHT * 0.3 - DOWN * 0.2, fill_opacity=0)
        predicted_box = SurroundingRectangle(pred_proxy, color=RED, buff=0.4)
        predicted_box.stretch_to_fit_width(cell_w * 1.3).stretch_to_fit_height(cell_h * 1.8)

        self.play(Create(predicted_box), Create(gt_box))

        # 6. Arrows showing distance (x, y)
        arrow_dist = Arrow(
            predicted_box.get_center(), 
            gt_box.get_center(), 
            color=YELLOW, buff=0, stroke_width=3
        )
        label_dist = MathTex(r"dx, dy", color=YELLOW).scale(0.5).next_to(arrow_dist, UP, buff=0.05)
        
        self.play(Create(arrow_dist), Write(label_dist))

        # 7. Loss Formula
        loss_formula = MathTex(
            r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}", 
            r"\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right)", 
            r"+", 
            r"\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}", 
            r"\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)"
        ).scale(0.5).to_edge(RIGHT, buff=0.5)

        self.play(FadeIn(loss_formula))
        self.wait(1)

        # 8. Mock Training Animation
        self.play(Indicate(loss_formula[1], color=YELLOW))
        self.play(
            predicted_box.animate.move_to(gt_box.get_center()),
            FadeOut(arrow_dist),
            FadeOut(label_dist),
            run_time=1.5
        )

        self.play(Indicate(loss_formula[4], color=BLUE))
        self.play(
            predicted_box.animate.stretch_to_fit_width(gt_box.width).stretch_to_fit_height(gt_box.height),
            run_time=1.5
        )

        self.wait(2)

        full_formula = MathTex(formula_text).scale(0.5).move_to(loss_formula)
        self.play(Transform(loss_formula, full_formula))
        self.wait()

        self.play(FadeOut(grid), FadeOut(trash_image), FadeOut(highlight_cell), FadeOut(predicted_box), FadeOut(gt_box))
        self.wait(1)
