from manim import *

class YOLOCoordinateLoss(Scene):
    def construct(self):
        # 1. Setup Image and Grid
        # Using a placeholder for the image
        try:
            image = ImageMobject("img-11.jpeg")
        except:
            image_rect = Rectangle(width=6, height=6, color=GRAY, fill_opacity=0.2)
            image_text = Text("Trash Image", color=WHITE).scale(0.8)
            image = Group(image_rect, image_text)
        
        image.scale_to_fit_height(6)
        image.to_edge(LEFT, buff=0.5)

        # 7x7 NumberPlane (Grid)
        grid = NumberPlane(
            x_range=[0, 7, 1],
            y_range=[0, 7, 1],
            x_length=image.get_width(),
            y_length=image.get_height(),
            background_line_style={
                "stroke_color": WHITE,
                "stroke_width": 2,
                "stroke_opacity": 0.5
            }
        )
        grid.move_to(image.get_center())

        # 2. Highlight Specific Cell
        # Choosing cell (4, 3) - 0 indexed from bottom left
        cell_width = grid.get_width() / 7
        cell_height = grid.get_height() / 7
        highlighted_cell = Rectangle(
            width=cell_width, 
            height=cell_height, 
            color=YELLOW, 
            fill_opacity=0.3, 
            stroke_width=4
        )
        highlighted_cell.move_to(grid.c2p(4.5, 3.5))

        # 3. Ground Truth and Predicted Boxes
        # Ground Truth: Dashed green (using DashedVMobject for compatibility)
        gt_box_base = Rectangle(
            width=cell_width * 2.5, 
            height=cell_height * 1.8, 
            color=GREEN
        )
        gt_box = DashedVMobject(gt_box_base)
        gt_box.move_to(grid.c2p(4.7, 3.6))

        # Predicted Box: Solid red
        pred_box = Rectangle(
            width=cell_width * 3.2, 
            height=cell_height * 2.2, 
            color=RED
        )
        pred_box.move_to(grid.c2p(4.2, 3.2))

        # 4. Arrows for (x, y) and (w, h)
        # Using Line with add_tip at both ends as a replacement for DoubleArrow
        center_arrow = Line(
            gt_box.get_center(), 
            pred_box.get_center(), 
            color=WHITE, 
            stroke_width=2
        ).add_tip(tip_length=0.15).add_tip(at_start=True, tip_length=0.15)
        
        # Dimension difference indicators
        w_arrow = Line(
            [gt_box.get_right()[0], gt_box.get_right()[1], 0],
            [pred_box.get_right()[0], gt_box.get_right()[1], 0],
            color=ORANGE, stroke_width=2
        ).add_tip(tip_length=0.1).add_tip(at_start=True, tip_length=0.1)
        
        h_arrow = Line(
            [gt_box.get_top()[0], gt_box.get_top()[1], 0],
            [gt_box.get_top()[0], pred_box.get_top()[1], 0],
            color=ORANGE, stroke_width=2
        ).add_tip(tip_length=0.1).add_tip(at_start=True, tip_length=0.1)

        # 5. Formula
        # Splitting terms into parts to highlight them later
        formula = MathTex(
            r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right)",
            r" + \sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)",
            color=WHITE
        ).scale(0.5)
        formula.to_edge(RIGHT, buff=0.3).shift(UP * 2)

        # 6. Animation Sequence
        self.play(FadeIn(image), Create(grid))
        self.play(Create(highlighted_cell))
        self.wait(1)
        
        self.play(Create(gt_box), Create(pred_box))
        self.play(Create(center_arrow))
        self.play(Create(w_arrow), Create(h_arrow))
        self.wait(1)

        self.play(Write(formula))
        self.wait(1)

        # Mock Training: Align boxes and highlight coordinate terms
        self.play(
            formula[0].animate.set_color(YELLOW),
            center_arrow.animate.scale(0.01),
            pred_box.animate.move_to(gt_box.get_center()),
            run_time=2
        )
        self.remove(center_arrow)
        
        # Align size and highlight size terms
        self.play(
            formula[0].animate.set_color(WHITE),
            formula[1].animate.set_color(ORANGE),
            w_arrow.animate.scale(0.01),
            h_arrow.animate.scale(0.01),
            pred_box.animate.stretch_to_fit_width(gt_box_base.get_width()).stretch_to_fit_height(gt_box_base.get_height()),
            run_time=2
        )
        self.remove(w_arrow, h_arrow)
        
        self.play(formula[1].animate.set_color(WHITE))
        self.wait(2)
        
        # Conclusion
        self.play(
            FadeOut(image),
            FadeOut(grid),
            FadeOut(highlighted_cell),
            FadeOut(gt_box),
            FadeOut(pred_box),
            formula.animate.scale(1.5).move_to(ORIGIN)
        )
        self.wait(2)
        self.play(FadeOut(formula))
        self.wait(1)
