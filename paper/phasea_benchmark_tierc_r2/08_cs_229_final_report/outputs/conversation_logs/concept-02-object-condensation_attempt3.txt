from manim import *
import numpy as np

class ObjectCondensation(Scene):
    def construct(self):
        # 1. Variables and Setup
        q_min = 0.1
        num_hits = 60
        voice_text = "Object condensation: high-beta hits act as condensation points."
        
        # 2. Display plane and formula
        # q_i = arctanh(beta_i) + q_min
        plane = NumberPlane().set_opacity(0.1)
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}", font_size=36)
        formula.to_edge(UP)
        self.add(plane, formula)
        
        # 3. Create hits (dots)
        # Create specific centers for "particles"
        center_locs = [np.array([-3, 1, 0]), np.array([2, -1.5, 0]), np.array([4, 2, 0])]
        hits = VGroup()
        
        # Background/Noise hits (low beta)
        for _ in range(num_hits):
            pos = np.array([np.random.uniform(-6, 6), np.random.uniform(-3, 3), 0])
            dot = Dot(point=pos, radius=0.07, color=PURPLE)
            dot.beta = np.random.uniform(0.05, 0.4)
            hits.add(dot)
        
        # Condensation point hits (high beta)
        condensation_points = VGroup()
        for loc in center_locs:
            cp = Dot(point=loc, radius=0.07, color=PURPLE)
            cp.beta = 0.95
            hits.add(cp)
            condensation_points.add(cp)
        
        self.play(FadeIn(hits))
        self.wait(0.5)
        
        # 4. Show Beta Charge Transformation
        # Each hit changes color and size based on its predicted beta
        charge_anims = []
        for h in hits:
            # Color mapping: Purple (low beta) to Yellow (high beta)
            target_color = interpolate_color(PURPLE, YELLOW, h.beta)
            # Size mapping: Larger for higher beta
            target_scale = 1.0 + (h.beta * 2.5)
            charge_anims.append(h.animate.set_color(target_color).scale(target_scale))
        
        self.play(*charge_anims, run_time=2)
        self.wait(0.5)
        
        # 5. Gravity Wells
        # Show potential wells centered at the condensation points
        wells = VGroup()
        for cp in condensation_points:
            well = VGroup(*[
                Circle(radius=r, color=YELLOW, stroke_width=2).set_opacity(1.0 - (r / 2.0))
                for r in [0.4, 0.8, 1.2, 1.6]
            ])
            well.move_to(cp.get_center())
            wells.add(well)
        
        self.play(FadeIn(wells, scale=0.5), run_time=1.5)
        
        # 6. Clustering Animation
        # Pull low-beta dots toward the nearest condensation point
        pull_anims = []
        for h in hits:
            # We only pull hits that aren't already condensation points
            if h.beta < 0.5:
                pos_h = h.get_center()
                # Find the nearest condensation point
                best_dist = float("inf")
                target_pos = pos_h
                for cp in condensation_points:
                    d = np.linalg.norm(pos_h - cp.get_center())
                    if d < best_dist:
                        best_dist = d
                        target_pos = cp.get_center()
                
                # If hit is within a reasonable "pull range"
                if best_dist < 4.5:
                    # Move 80% of the way to the center
                    new_pos = pos_h + 0.8 * (target_pos - pos_h)
                    pull_anims.append(h.animate.move_to(new_pos))
        
        # Using a smooth transition for the "gravity" effect
        self.play(*pull_anims, run_time=3, rate_func=smooth)
        
        # 7. Narration and Wrap-up
        voice_label = Text(voice_text, font_size=20).to_edge(1.5 * DOWN)
        self.play(Write(voice_label))
        self.wait(2)
