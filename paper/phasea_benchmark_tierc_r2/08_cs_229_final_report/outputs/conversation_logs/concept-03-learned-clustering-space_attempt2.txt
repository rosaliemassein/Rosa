from manim import *
import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Planes (Split Screen)
        physical_plane = NumberPlane(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5,
            y_length=5,
            background_line_style={"stroke_opacity": 0.4}
        ).to_edge(LEFT, buff=0.7)
        
        learned_plane = NumberPlane(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5,
            y_length=5,
            background_line_style={"stroke_opacity": 0.4}
        ).to_edge(RIGHT, buff=0.7)

        phys_title = Text("Physical Space", font_size=24).next_to(physical_plane, UP)
        learn_title = Text("Learned Space", font_size=24).next_to(learned_plane, UP)

        # 2. Define Data Points
        # Physical space: Overlapping clusters
        pts_a_phys = [
            [0.1, 0.2, 0], [-0.3, 0.4, 0], [0.4, -0.1, 0], 
            [-0.2, -0.3, 0], [0.1, -0.5, 0], [0.5, 0.3, 0]
        ]
        pts_b_phys = [
            [0.2, 0.1, 0], [-0.1, -0.2, 0], [0.6, 0.2, 0], 
            [0.0, 0.5, 0], [-0.4, 0.1, 0], [0.3, -0.4, 0]
        ]

        # Learned space: Separated circular distributions
        pts_a_learn = [
            [-1.5 + 0.4*np.cos(t), 0.4*np.sin(t), 0] 
            for t in np.linspace(0, 2*PI, 6, endpoint=False)
        ]
        pts_b_learn = [
            [1.5 + 0.5*np.cos(t), 0.5*np.sin(t), 0] 
            for t in np.linspace(0, 2*PI, 6, endpoint=False)
        ]

        # 3. Create Mobjects
        dots_a = VGroup(*[
            Dot(physical_plane.coords_to_point(*p), color=BLUE, radius=0.08) 
            for p in pts_a_phys
        ])
        dots_b = VGroup(*[
            Dot(physical_plane.coords_to_point(*p), color=RED, radius=0.08) 
            for p in pts_b_phys
        ])

        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=28
        ).to_edge(DOWN, buff=0.3)

        # 4. Manual Dashed Circle Helper (to avoid DashedVMobject issues)
        def get_manual_dashed_circle(radius, color, center):
            dashed_circle = VGroup()
            num_dashes = 16
            dash_angle = (2 * PI) / (2 * num_dashes)
            for i in range(num_dashes):
                start_angle = i * 2 * dash_angle
                arc = Arc(radius=radius, start_angle=start_angle, angle=dash_angle, color=color)
                dashed_circle.add(arc)
            dashed_circle.move_to(center)
            return dashed_circle

        # 5. Animation Sequence
        self.add(physical_plane, phys_title)
        self.play(Create(dots_a), Create(dots_b))
        self.wait(1)

        self.play(
            Create(learned_plane),
            Write(learn_title),
            Write(formula)
        )
        self.wait(1)

        # Define target dots in learned space
        target_a = VGroup(*[
            Dot(learned_plane.coords_to_point(*p), color=BLUE, radius=0.08)
            for p in pts_a_learn
        ])
        target_b = VGroup(*[
            Dot(learned_plane.coords_to_point(*p), color=RED, radius=0.08)
            for p in pts_b_learn
        ])

        # Transform dots from physical plane to learned plane
        self.play(
            Transform(dots_a, target_a),
            Transform(dots_b, target_b),
            run_time=2.5,
            path_arc=0.6
        )
        self.wait(1)

        # Create dashed classification boundaries
        circle_a = get_manual_dashed_circle(0.8, BLUE, learned_plane.coords_to_point(-1.5, 0))
        circle_b = get_manual_dashed_circle(1.0, RED, learned_plane.coords_to_point(1.5, 0))

        self.play(Create(circle_a), Create(circle_b))
        
        # Conclusion text
        success_label = Text("Classification Successful", font_size=20, color=GREEN).next_to(formula, UP, buff=0.2)
        self.play(Write(success_label))
        
        self.wait(3)
