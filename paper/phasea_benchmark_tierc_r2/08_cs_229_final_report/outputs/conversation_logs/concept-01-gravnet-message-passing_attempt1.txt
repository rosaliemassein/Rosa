from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Define the Voice/Narration text
        voice_text = (
            "The LDMX calorimeter has an irregular geometry, so we treat every hit as a node in a graph. "
            "GravNet creates a local neighborhood around each hit. The connection strength isn't fixed; "
            "it's weighted by a potential depending on distance in a learned feature space."
        )

        # 2. Create the irregular scatter plot of dots (2D simulation of ECal cells)
        dots = VGroup()
        for i in range(-5, 6):
            for j in range(-3, 4):
                # Filter to create a hexagonal-like cluster
                if abs(i) + abs(j) < 7:
                    # Irregular positioning to mimic detector geometry
                    pos = np.array([
                        i * 0.7 + np.random.uniform(-0.2, 0.2), 
                        j * 0.7 + np.random.uniform(-0.2, 0.2), 
                        0
                    ])
                    dot = Dot(point=pos, radius=0.08, color=GRAY)
                    dots.add(dot)
        
        # 3. Highlight the central dot
        # Choosing a dot near the center of the VGroup
        central_dot = dots[len(dots) // 2]
        
        # 4. Formula display (Reference: V_jk = exp(-d_jk^2))
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL).scale(0.8)
        
        self.add(dots)
        self.play(Create(dots), Write(formula))
        self.play(central_dot.animate.set_color(BLUE).scale(1.4))
        self.wait(0.5)

        # 5. Define the weighting function
        def get_weight(dist):
            # V_jk = exp(-d^2)
            # Dividing by a constant to make the visual transition smoother in 2D space
            return np.exp(-(dist**2) / 1.5)

        # 6. Draw and animate arrows/lines from central dot to neighbors
        connections = VGroup()
        for dot in dots:
            if dot == central_dot:
                continue
            
            # Initial distance check for local neighborhood
            d_init = np.linalg.norm(dot.get_center() - central_dot.get_center())
            if d_init < 2.5:
                line = Line(central_dot.get_center(), dot.get_center(), color=GREEN)
                
                # Add updater to handle dynamic shifting and weight scaling
                # Using default argument d_obj=dot to capture the current dot in the loop
                line.add_updater(lambda m, d_obj=dot: m.put_start_and_end_on(
                    central_dot.get_center(), d_obj.get_center()
                ).set_stroke(
                    opacity=get_weight(np.linalg.norm(central_dot.get_center() - d_obj.get_center())),
                    width=get_weight(np.linalg.norm(central_dot.get_center() - d_obj.get_center())) * 12
                ))
                connections.add(line)

        self.play(Create(connections), run_time=2)
        self.wait(1)

        # 7. Animate dots moving to simulate "learned features" updating
        # The updaters will automatically adjust the connections
        move_animations = []
        for dot in dots:
            # Shift dots to new positions
            target_shift = np.array([
                np.random.uniform(-0.7, 0.7),
                np.random.uniform(-0.7, 0.7),
                0
            ])
            move_animations.append(dot.animate.shift(target_shift))
        
        self.play(*move_animations, run_time=4, rate_func=smooth)
        self.wait(1)

        # 8. Display the narration text as a footer
        # Using VGroup of Text to simulate Paragraph/multiline wrapping
        footer_line1 = Text("LDMX calorimeter: treating hits as graph nodes.", font_size=18)
        footer_line2 = Text("Dynamic weighting based on learned feature distance.", font_size=18)
        footer = VGroup(footer_line1, footer_line2).arrange(DOWN, buff=0.2).to_edge(DOWN)
        
        bg_rect = SurroundingRectangle(footer, color=BLACK, fill_opacity=0.8, fill_color=BLACK, stroke_width=0)
        
        self.play(FadeIn(bg_rect), Write(footer))
        self.wait(3)
