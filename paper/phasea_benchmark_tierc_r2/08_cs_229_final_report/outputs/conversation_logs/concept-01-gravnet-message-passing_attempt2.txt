from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Create a scatter plot of dots representing ECal cells in an irregular grid
        dots = VGroup()
        for x in range(-5, 6):
            for y in range(-3, 4):
                # Add randomness to positions to create an irregular layout
                pos = np.array([
                    x * 0.8 + np.random.uniform(-0.3, 0.3),
                    y * 0.8 + np.random.uniform(-0.3, 0.3),
                    0
                ])
                dot = Dot(point=pos, radius=0.07, color=GRAY)
                dots.add(dot)

        # Highlight a central dot
        center_dot = dots[len(dots) // 2]
        center_dot.set_color(YELLOW)

        # 2. Add formula and description to the frame
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)", color=YELLOW).to_edge(UP)
        description = Text(
            "Connection strength depends on distance in learned space",
            font_size=20
        ).to_edge(DOWN)
        
        self.add(dots, formula, description)

        # 3. Create arrows (Lines) with updaters to scale opacity and thickness based on distance
        connections = VGroup()
        for d in dots:
            if d == center_dot:
                continue
            
            # Create a local neighborhood based on initial distance
            dist_initial = np.linalg.norm(d.get_center() - center_dot.get_center())
            if dist_initial < 2.2:
                # Use a Line to represent the connection
                line = Line(center_dot.get_center(), d.get_center(), buff=0)
                line.set_color(BLUE)
                
                # Custom attribute to keep track of the specific neighbor dot in the updater
                line.neighbor = d
                
                def update_connection(l):
                    p1 = center_dot.get_center()
                    p2 = l.neighbor.get_center()
                    # Calculate Euclidean distance
                    d_jk = np.linalg.norm(p1 - p2)
                    # Formula: V_jk = exp(-d^2)
                    strength = np.exp(-(d_jk**2))
                    # Update geometry and appearance
                    l.put_start_and_end_on(p1, p2)
                    l.set_stroke(opacity=strength, width=strength * 12)
                
                line.add_updater(update_connection)
                connections.add(line)

        # Initial animation of connections appearing
        self.play(Create(connections), run_time=2)
        self.wait(1)

        # 4. Animate dots moving to simulate updates in the 'learned feature space'
        # Updaters handle the dynamic recalculation of connection weights
        move_anims = []
        for d in dots:
            # Generate a target offset to show shifting relationships
            target_pos = d.get_center() + np.array([
                np.random.uniform(-1.2, 1.2),
                np.random.uniform(-1.2, 1.2),
                0
            ])
            move_anims.append(d.animate.move_to(target_pos))

        self.play(*move_anims, run_time=5, rate_func=smooth)
        self.wait(1)

        # Move the central dot specifically to emphasize the shifting connections
        self.play(
            center_dot.animate.shift(RIGHT * 2),
            run_time=2,
            rate_func=there_and_back
        )
        self.wait(2)
