from manim import *
import numpy as np

class LearnedClusteringSpaceAnimation(Scene):
    def construct(self):
        # 1. Setup Split Screen Planes
        left_title = Text("Physical Space", font_size=24).to_edge(UP).shift(LEFT * 3.5)
        right_title = Text("Learned Space", font_size=24).to_edge(UP).shift(RIGHT * 3.5)

        left_plane = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-4, 4, 1],
            x_length=5.5,
            y_length=5.5,
            background_line_style={"stroke_opacity": 0.3}
        ).shift(LEFT * 3.5)

        right_plane = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-4, 4, 1],
            x_length=5.5,
            y_length=5.5,
            background_line_style={"stroke_opacity": 0.3}
        ).shift(RIGHT * 3.5)

        self.add(left_title, right_title, left_plane, right_plane)

        # 2. Setup Dots (Overlapping in Physical Space)
        np.random.seed(42)
        
        # Cluster A (Electron 1) - Centralized
        dots_a = VGroup(*[
            Dot(left_plane.c2p(np.random.normal(0.2, 0.5), np.random.normal(0.2, 0.5)), 
                color=BLUE, radius=0.06) for _ in range(30)
        ])
        # Cluster B (Electron 2) - Heavily overlapping Cluster A
        dots_b = VGroup(*[
            Dot(left_plane.c2p(np.random.normal(-0.2, 0.5), np.random.normal(-0.2, 0.5)), 
                color=RED, radius=0.06) for _ in range(30)
        ])

        self.play(FadeIn(dots_a), FadeIn(dots_b))
        self.wait(1)

        # 3. Create Targets in Learned Space (Separated Blobs)
        # We move them relative to the right_plane
        target_a = VGroup()
        for i in range(len(dots_a)):
            target_dot = dots_a[i].copy().move_to(
                right_plane.c2p(np.random.normal(1.8, 0.4), np.random.normal(1.8, 0.4))
            )
            target_a.add(target_dot)

        target_b = VGroup()
        for i in range(len(dots_b)):
            target_dot = dots_b[i].copy().move_to(
                right_plane.c2p(np.random.normal(-1.8, 0.4), np.random.normal(-1.8, 0.4))
            )
            target_b.add(target_dot)

        # 4. Formula at the bottom
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=32
        ).to_edge(DOWN, buff=0.4)

        # 5. Animation sequence
        self.play(Write(formula))
        self.wait(0.5)

        # Transformation of dots from physical to learned space
        self.play(
            Transform(dots_a, target_a),
            Transform(dots_b, target_b),
            run_time=3,
            rate_func=smooth
        )

        # 6. Classification markers (Dashed Circles)
        # Using DashedVMobject on a Circle
        circle_a = DashedVMobject(Circle(radius=1.2, color=BLUE)).move_to(right_plane.c2p(1.8, 1.8))
        circle_b = DashedVMobject(Circle(radius=1.2, color=RED)).move_to(right_plane.c2p(-1.8, -1.8))
        
        label_a = Text("Cluster 1", font_size=18, color=BLUE).next_to(circle_a, UP, buff=0.1)
        label_b = Text("Cluster 2", font_size=18, color=RED).next_to(circle_b, DOWN, buff=0.1)

        self.play(
            Create(circle_a),
            Create(circle_b),
            Write(label_a),
            Write(label_b)
        )

        # 7. Descriptive narration label
        narration_label = Text(
            "Unwarping energy deposits into distinct, separable blobs",
            font_size=20,
            slant="ITALIC"
        ).next_to(formula, UP, buff=0.3)
        
        self.play(FadeIn(narration_label))
        self.wait(3)

        # Fade out everything
        self.play(
            *[FadeOut(mob) for mob in self.mobjects]
        )
        self.wait(1)
