from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup a 2D plane
        grid = NumberPlane(
            x_range=(-8, 8, 1), 
            y_range=(-5, 5, 1), 
            background_line_style={"stroke_color": BLUE, "stroke_opacity": 0.2}
        )
        self.add(grid)

        # 2. Header and Formula
        explanation = Text("Object Condensation", color=WHITE).scale(0.8)
        explanation.to_edge(UP, buff=0.5)
        
        formula = MathTex(
            r"q_i = \text{arctanh}(\beta_i) + q_{min}", 
            color=WHITE
        ).scale(0.8).next_to(explanation, DOWN)
        
        self.add(explanation, formula)

        # 3. Create a group of dots
        num_dots = 25
        dots = VGroup()
        np.random.seed(42)
        
        condensation_indices = [0, 1, 2]
        
        for i in range(num_dots):
            pos = np.array([
                np.random.uniform(-6, 6),
                np.random.uniform(-3, 2),
                0
            ])
            
            # Simulated beta: High for condensation points, low for others
            if i in condensation_indices:
                beta = np.random.uniform(0.85, 0.98)
            else:
                beta = np.random.uniform(0.05, 0.4)
            
            dot = Dot(point=pos, radius=0.08, color=PURPLE)
            dot.beta = beta
            dots.add(dot)

        self.play(Create(dots), run_time=1)
        self.wait(0.5)

        # 4. Animate growth and color change based on beta
        grow_animations = []
        condensation_points = []
        
        for dot in dots:
            # Manually interpolate color to avoid potential identifier issues
            # Purple: [128, 0, 128] / 255 -> [0.5, 0, 0.5]
            # Yellow: [255, 255, 0] / 255 -> [1, 1, 0]
            alpha = dot.beta
            target_color = interpolate_color(PURPLE, YELLOW, alpha)
            target_scale = 1.0 + alpha * 3.0
            
            grow_animations.append(
                dot.animate.set_color(target_color).scale(target_scale)
            )
            if alpha > 0.8:
                condensation_points.append(dot)
        
        self.play(*grow_animations, run_time=1.5)
        self.wait(0.5)

        # 5. Create 'Gravity Wells' (Gradient circles)
        wells = VGroup()
        for cp in condensation_points:
            well = VGroup()
            # Create a manual gradient effect using concentric circles
            for r in np.linspace(0.1, 2.2, 10):
                circle = Circle(
                    radius=r, 
                    stroke_width=0, 
                    fill_opacity=0.18 * (1 - r/2.2), 
                    fill_color=YELLOW
                ).move_to(cp.get_center())
                well.add(circle)
            wells.add(well)
        
        self.play(FadeIn(wells, scale=0.8), run_time=1)
        self.wait(0.5)

        # 6. Pull neighboring dots toward condensation points
        pull_anims = []
        for dot in dots:
            if dot in condensation_points:
                continue
            
            # Find the closest condensation point
            target_cp = condensation_points[0]
            min_dist = np.linalg.norm(dot.get_center() - target_cp.get_center())
            for cp in condensation_points[1:]:
                dist = np.linalg.norm(dot.get_center() - cp.get_center())
                if dist < min_dist:
                    min_dist = dist
                    target_cp = cp
            
            if min_dist < 4.5:
                # Calculate manual interpolation for position
                start_pos = dot.get_center()
                end_pos = target_cp.get_center()
                # Move 75% of the way to the center
                destination = start_pos + 0.75 * (end_pos - start_pos)
                pull_anims.append(dot.animate.move_to(destination))

        if pull_anims:
            self.play(*pull_anims, run_time=2, rate_func=smooth)
        self.wait(1)

        # 7. Final explanation
        final_note = Tex(
            r"Charge $q_i$ acts like gravity, clustering hits together", 
            color=YELLOW
        ).scale(0.8).to_edge(DOWN, buff=0.5)
        
        self.play(Write(final_note))
        self.play(Indicate(formula, color=YELLOW, scale_factor=1.2))
        self.wait(2)
