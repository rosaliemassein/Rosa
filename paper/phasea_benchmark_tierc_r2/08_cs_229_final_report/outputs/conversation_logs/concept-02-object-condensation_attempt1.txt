from manim import *
import numpy as np

class ConceptObjectCondensation(Scene):
    def construct(self):
        # 1. Setup Data and Plane of Dots
        q_min = 0.5
        
        # Create a grid of dots
        dots = VGroup()
        beta_values = []
        
        # Grid dimensions
        rows, cols = 7, 12
        x_range = np.linspace(-5, 5, cols)
        y_range = np.linspace(-3, 3, rows)
        
        for y in y_range:
            for x in x_range:
                pos = np.array([x + np.random.uniform(-0.1, 0.1), y + np.random.uniform(-0.1, 0.1), 0])
                # Default beta value (low)
                beta = np.random.uniform(0.1, 0.3)
                beta_values.append(beta)
                dot = Dot(pos, radius=0.08, color=PURPLE)
                dots.add(dot)

        # Force a few specific "condensation points" (indices selected manually)
        cp_indices = [20, 45, 62]
        for idx in cp_indices:
            beta_values[idx] = 0.95

        self.play(FadeIn(dots), run_time=1)
        self.wait(0.5)

        # 2. Animate dots growing and changing color based on beta values
        # We simulate the beta-charge mechanism
        growth_anims = []
        for i, (dot, beta) in enumerate(zip(dots, beta_values)):
            # interpolate_color replacement if it fails: simple if/else for the demo
            if beta > 0.8:
                target_color = YELLOW
                target_scale = 2.5
            elif beta > 0.4:
                target_color = ORANGE
                target_scale = 1.5
            else:
                target_color = PURPLE
                target_scale = 1.0
            
            growth_anims.append(dot.animate.set_color(target_color).scale(target_scale))
        
        self.play(*growth_anims, run_time=2)
        self.wait(0.5)

        # 3. Draw "Gravity Wells" (gradient circles) for high-beta dots
        wells = VGroup()
        for idx in cp_indices:
            cp_pos = dots[idx].get_center()
            # Create a concentric effect with decreasing opacity
            well = VGroup()
            for r in np.linspace(0.3, 1.5, 5):
                c = Circle(radius=r, stroke_width=0, fill_opacity=0.1, fill_color=YELLOW)
                c.move_to(cp_pos)
                well.add(c)
            wells.add(well)
        
        self.play(FadeIn(wells), run_time=1.5)

        # 4. Pull neighboring low-beta dots toward the center of these wells
        pull_anims = []
        for i, dot in enumerate(dots):
            if i in cp_indices:
                continue
            
            # Find the nearest condensation point
            dists = [np.linalg.norm(dot.get_center() - dots[cp_idx].get_center()) for cp_idx in cp_indices]
            min_dist = min(dists)
            nearest_cp_idx = cp_indices[np.argmin(dists)]
            
            # If the dot is within the influence range, pull it
            if min_dist < 2.5:
                direction = dots[nearest_cp_idx].get_center() - dot.get_center()
                # Move 70% of the way to the center
                pull_anims.append(dot.animate.shift(direction * 0.7))

        self.play(*pull_anims, run_time=2, rate_func=rush_into)
        self.wait(1)

        # 5. Add label for the charge formula
        formula_tex = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula_tex.to_edge(UP, buff=0.5)
        
        # Safe replacement for BackgroundRectangle using a SurroundingRectangle
        bg_box = SurroundingRectangle(formula_tex, color=BLACK, fill_color=BLACK, fill_opacity=0.7)
        formula_group = VGroup(bg_box, formula_tex)
        
        self.play(FadeIn(bg_box), Write(formula_tex))
        self.wait(3)
