from manim import *
import numpy as np

class Concept01GravNetMessagePassing(Scene):
    def construct(self):
        # 1. Create a 2D irregular hexagonal-like grid of dots
        # Using Scene instead of ThreeDScene as requested by constraints
        dots = VGroup()
        np.random.seed(42)
        
        # Create a hexagonal layout with jitter
        for x in range(-5, 6):
            for y in range(-3, 4):
                # Horizontal offset for hexagonal grid pattern
                offset_x = 0.5 if y % 2 == 1 else 0
                pos = np.array([
                    (x + offset_x) * 1.1 + np.random.uniform(-0.2, 0.2),
                    y * 0.9 + np.random.uniform(-0.2, 0.2),
                    0
                ])
                # Filter points to stay within a reasonable viewing area
                if np.linalg.norm(pos) < 6:
                    dots.add(Dot(point=pos, radius=0.08, color=BLUE))
        
        # 2. Identify and highlight a central dot
        # We find the dot closest to the origin (0,0,0)
        center_dot = dots[0]
        min_dist = np.linalg.norm(center_dot.get_center())
        for d in dots:
            dist_to_origin = np.linalg.norm(d.get_center())
            if dist_to_origin < min_dist:
                min_dist = dist_to_origin
                center_dot = d
        
        center_dot.set_color(YELLOW)
        center_dot.set_z_index(10) # Ensure it renders above connection lines

        # 3. Create arrows representing potential connections from the center dot
        connections = VGroup()
        for dot in dots:
            if dot == center_dot:
                continue
            
            # Create an arrow that will be updated dynamically
            arrow = Arrow(
                center_dot.get_center(), 
                dot.get_center(), 
                buff=0, 
                stroke_width=2,
                max_tip_length_to_length_ratio=0.15,
                color=WHITE
            )
            
            # Updater function to calculate strength based on V = exp(-d^2)
            def update_arrow(obj, target_dot=dot):
                p_start = center_dot.get_center()
                p_end = target_dot.get_center()
                distance = np.linalg.norm(p_end - p_start)
                
                # Formula: V_jk = exp(-d_jk^2)
                # We divide distance by a scaling factor (2.2) to make the visual falloff look natural
                weight = np.exp(-(distance / 2.2)**2)
                
                # Update the arrow position and visual properties
                obj.put_start_and_end_on(p_start, p_end)
                obj.set_stroke(opacity=weight, width=weight * 12)
                # Hide the arrow tip if the connection is very weak
                if weight < 0.1:
                    obj.set_stroke(opacity=0)
            
            arrow.add_updater(update_arrow)
            connections.add(arrow)

        # 4. Display the formula
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)", color=YELLOW)
        formula.to_edge(UP, buff=0.5)
        
        # Context label
        title = Text("GravNet Dynamic Weights", font_size=24, color=GRAY).to_edge(DOWN)

        # 5. Animation Sequence
        self.play(FadeIn(dots), Write(formula), FadeIn(title))
        self.wait(0.5)
        self.play(Indicate(center_dot, scale_factor=1.5, color=YELLOW))
        self.play(Create(connections), run_time=2)
        self.wait(1)

        # 6. Feature Space Update: Dots move slightly to represent learned relationship shifts
        # The updaters on the arrows will automatically recalculate the weights as dots move
        move_anims = []
        for d in dots:
            # Random drift vector
            drift = np.array([
                np.random.uniform(-1.2, 1.2),
                np.random.uniform(-1.2, 1.2),
                0
            ])
            move_anims.append(d.animate.shift(drift))

        # Perform the dynamic transition

        self.wait(2)

        # Subtle secondary adjustment

        for d in dots:
            drift = np.array([
                np.random.uniform(-0.5, 0.5),
                np.random.uniform(-0.5, 0.5),
                0
            ])
            second_move.append(d.animate.shift(drift))
            
        self.play(*second_move, run_time=3, rate_func=smooth)
        self.wait(3)
