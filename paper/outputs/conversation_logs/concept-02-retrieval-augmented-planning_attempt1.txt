from manim import *
import random

class RetrievalAugmentedPlanningConcept(Scene):
    def construct(self):
        # --- Data Strings (from JSON) ---
        voice_text = "Standard AI planning often relies on a single, fixed path. But here, we use Retrieval-Augmented Planning, or RAP."
        goal_text = "Explain how external knowledge retrieval enables the exploration of diverse, optimal planning paths."
        formula_str = r"\mathbf{P} = \mathcal{A}_{mgr}(\mathrm{RAP}(R))"
        
        # --- 1. Search Bar Phase ---
        search_bar = RoundedRectangle(corner_radius=0.1, width=6, height=0.7, color=WHITE)
        search_bar.to_edge(TOP, buff=0.5)
        
        # Create a simple magnifying glass icon using shapes (avoids LaTeX dependency)
        glass_circle = Circle(radius=0.15, color=WHITE).move_to(search_bar.get_left() + RIGHT * 0.4)
        glass_handle = Line(
            glass_circle.get_bottom() + LEFT*0.05, 
            glass_circle.get_bottom() + RIGHT*0.15 + DOWN*0.15, 
            color=WHITE
        )
        search_icon = VGroup(glass_circle, glass_handle)
        
        search_label = Text("Image Classification", font_size=24, color=GRAY)
        search_label.next_to(search_icon, RIGHT, buff=0.3)
        
        search_group = VGroup(search_bar, search_icon, search_label)
        
        self.play(Create(search_bar))
        self.play(Write(search_label), Create(search_icon))
        self.wait(0.5)

        # --- 2. External Documents Phase ---
        # Represent "the cloud" as a region in the top right
        docs = VGroup()
        for i in range(3):
            doc_rect = Rectangle(width=0.9, height=1.2, fill_opacity=0.8, fill_color=BLUE_D, stroke_color=WHITE)
            # Add small lines to look like text on a document
            lines = VGroup(*[Line(LEFT*0.3, RIGHT*0.3, stroke_width=1).shift(UP*(0.3 - 0.2*j)) for j in range(4)])
            doc = VGroup(doc_rect, lines)
            doc.move_to(RIGHT*5 + UP*2) # Start from cloud
            docs.add(doc)

        # Fly documents in to center-left
        self.play(
            LaggedStart(*[
                d.animate.move_to(LEFT*2 + UP*(1 - i)) 
                for i, d in enumerate(docs)
            ], lag_ratio=0.3)
        )
        self.wait(0.5)

        # --- 3. Tree Structure Transformation ---
        # Root node
        root = VGroup(
            Dot(color=RED, radius=0.15),
            Text("Parsed Requirement", font_size=20).next_to(ORIGIN, DOWN * 0.5)
        ).move_to(LEFT * 4)

        # Plan nodes
        plans = VGroup()
        plan_names = ["Plan A", "Plan B", "Plan C"]
        y_positions = [1.5, 0, -1.5]
        
        for name, y in zip(plan_names, y_positions):
            p_node = VGroup(
                Dot(color=GREEN),
                Text(name, font_size=22).next_to(ORIGIN, RIGHT * 0.5)
            ).move_to(RIGHT * 2 + UP * y)
            plans.add(p_node)

        # Tree Connections
        branches = VGroup(*[
            Line(root[0].get_center(), p[0].get_center(), color=GRAY_A, stroke_width=2)
            for p in plans
        ])

        # Morph documents into the diverse plans
        self.play(
            FadeOut(search_group),
            FadeIn(root),
            ReplacementTransform(docs, plans),
            Create(branches)
        )
        self.wait(0.5)

        # --- 4. Successful Path Highlight ---
        # Let's say Plan B is the most optimal
        successful_path = branches[1]
        successful_node = plans[1]
        
        self.play(
            successful_path.animate.set_stroke(color=YELLOW, width=6),
            successful_node.animate.scale(1.2).set_color(YELLOW),
        )
        self.play(Indicate(successful_node, color=YELLOW, scale_factor=1.4))
        
        # --- 5. Formula & Conclusion ---
        formula = MathTex(formula_str, font_size=42)
        formula.to_edge(BOTTOM, buff=1.2)
        
        goal_display = Text(goal_text, font_size=18, slant=ITALIC, color=LIGHT_GRAY)
        goal_display.to_edge(BOTTOM, buff=0.5)

        self.play(Write(formula))
        self.play(FadeIn(goal_display))
        
        self.wait(3)
