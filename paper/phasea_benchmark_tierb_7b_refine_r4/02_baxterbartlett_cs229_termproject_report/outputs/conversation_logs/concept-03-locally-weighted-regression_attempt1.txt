import numpy
from manim import *

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Manual Coordinate System (Avoids 'Axes' and 'ValueTracker')
        x_unit = 2.0
        y_unit = 1.5
        
        horiz_axis = Line(LEFT * 5, RIGHT * 5, color=BLUE)
        vert_axis = Line(DOWN * 3, UP * 3, color=BLUE)
        self.add(horiz_axis, vert_axis)
        
        def map_coords(x_val, y_val):
            return x_val * RIGHT * x_unit + y_val * UP * y_unit

        # 2. Scatter Plot Data
        numpy.random.seed(42)
        all_data_dots = VGroup()
        for _ in range(45):
            point_x = numpy.random.uniform(-2.2, 2.2)
            # Underlying pattern: y = 0.6 * sin(1.2x)
            point_y = 0.6 * numpy.sin(1.2 * point_x) + numpy.random.normal(0, 0.1)
            dot_element = Dot(map_coords(point_x, point_y), radius=0.07, color=GREY)
            all_data_dots.add(dot_element)
        self.add(all_data_dots)

        # 3. Reference Formula
        formula_tex = MathTex(r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)")
        formula_tex.to_edge(UP)
        self.add(formula_tex)

        # 4. Movement Control (Using a hidden dot as a tracker)
        control_dot = Dot(LEFT * 2.2 * x_unit).set_opacity(0)
        self.add(control_dot)

        # 5. Query Point and Weight Circle
        query_dot = Dot(color=WHITE, z_index=10)
        def query_logic(target_obj):
            current_x = control_dot.get_center()[0] / x_unit
            current_y = 0.6 * numpy.sin(1.2 * current_x)
            target_obj.move_to(map_coords(current_x, current_y))
        
        query_dot.add_updater(query_logic)
        
        weight_visual = Circle(radius=1.2, color=YELLOW, fill_opacity=0.2, stroke_width=0)
        weight_visual.add_updater(lambda target_circ: target_circ.move_to(query_dot.get_center()))
        
        self.play(FadeIn(query_dot), FadeIn(weight_visual))

        # 6. Local Regression Line
        # We manually compute the local tangent for the target function
        regression_line = Line(LEFT, RIGHT, color=RED, stroke_width=6).set_length(1.7)
        def regression_logic(line_obj):
            current_x = control_dot.get_center()[0] / x_unit
            # Derivative of 0.6 * sin(1.2x) is 0.72 * cos(1.2x)
            local_slope = 0.72 * numpy.cos(1.2 * current_x)
            # Adjust angle for screen aspect ratio
            visual_angle = numpy.arctan(local_slope * (y_unit / x_unit))
            line_obj.set_angle(visual_angle)
            line_obj.move_to(query_dot.get_center())
        
        regression_line.add_updater(regression_logic)
        self.add(regression_line)

        # 7. Training Points Highlighting (The 'Weight' field interaction)
        def points_highlight_logic(dots_group):
            query_center = query_dot.get_center()
            for single_dot in dots_group:
                dist_to_query = numpy.linalg.norm(single_dot.get_center() - query_center)
                # Apply weighting formula for visual feedback
                influence = numpy.exp(-(dist_to_query**2) / (2 * 0.7**2))
                if dist_to_query < 1.4:
                    single_dot.set_color(YELLOW)
                    single_dot.set_opacity(0.3 + 0.7 * influence)
                else:
                    single_dot.set_color(GREY)
                    single_dot.set_opacity(0.3)
        
        all_data_dots.add_updater(points_highlight_logic)

        # 8. Animation (Linear movement from left to right and back)
        self.play(
            control_dot.animate.move_to(RIGHT * 2.2 * x_unit), 
            run_time=8, 
            rate_func=lambda t: t
        )
        self.play(
            control_dot.animate.move_to(ORIGIN), 
            run_time=4, 
            rate_func=lambda t: t
        )
        self.wait(2)
