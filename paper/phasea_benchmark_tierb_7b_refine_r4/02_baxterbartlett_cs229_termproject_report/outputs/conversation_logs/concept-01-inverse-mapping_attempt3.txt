from manim import *
import numpy as np

class Concept01InverseMapping(Scene):
    def construct(self):
        # 1. Setup Input and Output Groups (using basic primitives)
        input_box = Rectangle(height=2.2, width=4.2, color=BLUE)
        input_label = Text("Desired Behavior", font_size=24, color=BLUE).next_to(input_box, UP)
        rt_text = Text("Rise Time", font_size=18)
        sse_text = Text("Steady State Error", font_size=18)
        st_text = Text("Stability", font_size=18)
        input_items = VGroup(rt_text, sse_text, st_text).arrange(DOWN, buff=0.2).move_to(input_box.get_center())
        input_group = VGroup(input_box, input_label, input_items).shift(LEFT * 3.5 + UP * 1.5)

        output_box = Rectangle(height=2.2, width=4.2, color=RED)
        output_label = Text("PID Constants", font_size=24, color=RED).next_to(output_box, UP)
        kp_text = MathTex("K_p")
        ki_text = MathTex("K_i")
        kd_text = MathTex("K_d")
        output_items = VGroup(kp_text, ki_text, kd_text).arrange(DOWN, buff=0.2).move_to(output_box.get_center())
        output_group = VGroup(output_box, output_label, output_items).shift(RIGHT * 3.5 + UP * 1.5)

        # 2. Arrow connecting them
        arrow = Arrow(input_box.get_right(), output_box.get_left(), color=WHITE)
        diagram = VGroup(input_group, output_group, arrow)

        # 3. Intro animation
        self.play(Create(input_group), Create(output_group), GrowArrow(arrow))
        self.wait(1)

        # 4. "Flip the script" - Rotate using animate.rotate to avoid Rotate class
        self.play(diagram.animate.rotate(PI), run_time=2)
        self.wait(1)

        # 5. Formula reference
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=24)
        formula.to_corner(UL)
        self.play(Write(formula))

        # 6. Motor Velocity Plot using manual Lines and VMobject
        # Origin for plot
        origin = DOWN * 2.5 + LEFT * 4
        x_axis = Line(origin, origin + RIGHT * 8, color=WHITE)
        y_axis = Line(origin, origin + UP * 3, color=WHITE)
        
        x_label = Text("Time", font_size=16).next_to(x_axis, RIGHT)
        y_label = Text("RPM", font_size=16).next_to(y_axis, UP)
        
        # Dashed line manual implementation
        setpoint_val = 2.0
        setpoint_y = origin + UP * setpoint_val
        dashes = VGroup()
        for i in range(20):
            d_start = setpoint_y + RIGHT * (i * 0.4)
            d_end = setpoint_y + RIGHT * (i * 0.4 + 0.2)
            dashes.add(Line(d_start, d_end, color=GRAY, stroke_width=2))

        # Motor behavior curve manual implementation
        curve_points = []
        target_val = 1.8 # Lower than setpoint to show steady state error
        for t_step in np.linspace(0, 8, 100):
            # f(t) = target * (1 - e^(-0.6t) * cos(2.5t))
            val = target_val * (1 - np.exp(-0.6 * t_step) * np.cos(2.5 * t_step))
            curve_points.append(origin + RIGHT * t_step + UP * val)
        
        motor_plot = VMobject(color=YELLOW).set_points_as_corners(curve_points)

        self.play(Create(x_axis), Create(y_axis), Write(x_label), Write(y_label))
        self.play(Create(dashes))
        self.play(Create(motor_plot), run_time=2)

        # 7. Measurements using simple Lines/Ticks instead of Brace

        rt_end = origin + RIGHT * 0.7 + DOWN * 0.2
        rt_line = Line(rt_start, rt_end, color=BLUE)
        rt_tick1 = Line(rt_start + UP*0.1, rt_start + DOWN*0.1, color=BLUE)
        rt_tick2 = Line(rt_end + UP*0.1, rt_end + DOWN*0.1, color=BLUE)
        rt_tag = Text("Rise Time", font_size=14, color=BLUE).next_to(rt_line, DOWN, buff=0.1)

        sse_tick1 = Line(sse_bottom + LEFT*0.1, sse_bottom + RIGHT*0.1, color=RED)
        sse_tick2 = Line(sse_top + LEFT*0.1, sse_top + RIGHT*0.1, color=RED)
        sse_tag = Text("Steady State Error", font_size=12, color=RED).next_to(sse_line, RIGHT, buff=0.1)
        sse_meas = VGroup(sse_line, sse_tick1, sse_tick2, sse_tag)
