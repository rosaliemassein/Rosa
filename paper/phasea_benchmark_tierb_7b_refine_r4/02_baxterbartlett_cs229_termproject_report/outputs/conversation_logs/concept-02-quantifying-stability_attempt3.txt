from manim import *
from collections import Counter
import numpy as np

class QuantifyingStability(Scene):
    def construct(self):
        # 1. Setup Axes for Velocity Plot
        # "Show a velocity plot where the motor is fluctuating around 30 RPM"
        axes = Axes(
            x_range=[0, 10, 2],
            y_range=[25, 35, 5],
            x_length=6,
            y_length=4,
            axis_config={"include_numbers": False, "include_tip": False}
        ).to_edge(LEFT, buff=0.5).shift(UP * 0.5)
        
        x_label = MathTex("t").next_to(axes.x_axis, RIGHT)
        y_label = MathTex("v (RPM)").next_to(axes.y_axis, UP)
        
        # Velocity Data - Fluctuating around 30 RPM
        # Creating a specific set of data to ensure mode is 30
        velocities = [30, 31, 29, 30, 32, 28, 30, 31, 30, 30, 29, 30, 30, 30, 31, 29, 30, 30, 30, 30]
        times = np.linspace(0, 10, len(velocities))
        
        v_graph_data = axes.plot_line_graph(
            x_values=times, 
            y_values=velocities, 
            add_vertex_dots=True, 
            vertex_dot_style={"radius": 0.04, "color": YELLOW},
            line_color=YELLOW
        )

        # 2. Highlight Last 20% (x = 8 to 10)
        # "Highlight the last 20% of the x-axis with a different background color (e.g., light blue)"
        rect_width = axes.x_axis.get_unit_size() * 2
        rect_height = axes.y_axis.get_length()
        highlight_rect = Rectangle(
            width=rect_width, 
            height=rect_height,
            fill_color=BLUE_A, 
            fill_opacity=0.3, 
            stroke_width=0
        ).move_to(axes.c2p(9, 30))
        
        highlight_text = Text("Final 20%", font_size=18, color=BLUE_A).next_to(highlight_rect, UP, buff=0.1)

        # 3. Histogram on the Side
        # "Show a 'Histogram' appearing on the side, where the 'bin' at 30 RPM is significantly taller than the others"
        counts = Counter(velocities)
        bins = [28, 29, 30, 31, 32]
        
        hist_axes = Axes(
            x_range=[27, 33, 1],
            y_range=[0, 15, 5],
            x_length=4,
            y_length=3,
            axis_config={"include_tip": False, "include_numbers": False}
        ).to_edge(RIGHT, buff=0.5).shift(UP * 1.5)
        
        hist_label = Text("Histogram", font_size=24).next_to(hist_axes, UP)
        
        bars = VGroup()
        for b in bins:
            count = counts.get(b, 0)
            # Scale height according to hist_axes y range
            bar_height = count * (hist_axes.y_axis.get_length() / 15)
            bar = Rectangle(
                width=0.5, 
                height=bar_height, 
                fill_color=GREEN if b == 30 else GRAY,
                fill_opacity=0.8,
                stroke_color=WHITE,
                stroke_width=1
            ).move_to(hist_axes.c2p(b, 0), DOWN)
            bars.add(bar)

        # 4. Stability Formula
        # "Animate a fraction bar where the numerator is the 'count of most common speed' and the denominator is 'other counts'"
        formula = MathTex(
            r"\text{Stability}", "=", r"\frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=40
        ).to_edge(DOWN, buff=1)

        # Animations
        self.play(Create(axes), Write(x_label), Write(y_label))
        self.play(Create(v_graph_data), run_time=2)
        self.wait(0.5)

        self.play(FadeIn(highlight_rect), Write(highlight_text))
        self.wait(1)

        self.play(Create(hist_axes), Write(hist_label))
        self.play(LaggedStart(*[GrowFromEdge(bar, DOWN) for bar in bars], lag_ratio=0.1))
        self.wait(1)

        self.play(Write(formula))
        
        # Highlight numerator part
        self.play(
            Indicate(bars[2], scale_factor=1.2, color=GREEN), 
            formula[2][0:13].animate.set_color(GREEN) # "counts at mode"
        )
        self.wait(1)
        
        # Highlight denominator part
        others = VGroup(*[bars[i] for i in [0, 1, 3, 4]])
        self.play(
            Indicate(others, scale_factor=1.1, color=GRAY_B),
            formula[2][14:].animate.set_color(GRAY_B) # "counts not at mode"
        )
        self.wait(3)
