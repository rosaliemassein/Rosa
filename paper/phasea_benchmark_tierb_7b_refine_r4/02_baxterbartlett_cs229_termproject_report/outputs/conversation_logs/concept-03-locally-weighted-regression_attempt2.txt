from manim import *
import math
import random

class LocallyWeightedRegressionScene(Scene):
    def construct(self):
        # 1. Manual Axes (Replacing Axes which was disallowed)
        x_ax = Line(LEFT * 5, RIGHT * 5, color=BLUE)
        y_ax = Line(DOWN * 3, UP * 3, color=BLUE)
        self.add(x_ax, y_ax)
        
        # 2. Formula
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)")
        formula.to_edge(UP)
        self.add(formula)
        
        # 3. Data Points (Synthetic Sine Wave with Noise)
        data_x = []
        data_y = []
        dots = VGroup()
        for i in range(40):
            x = -4.0 + i * 0.2
            # Target: y = 1.5 * sin(x)
            y = 1.5 * math.sin(x) + random.uniform(-0.2, 0.2)
            data_x.append(x)
            data_y.append(y)
            dot = Dot(point=[x, y, 0], radius=0.08, color=GREY, fill_opacity=0.4)
            dots.add(dot)
        self.add(dots)
        
        # 4. Query Point, Weight Field (Visual Circle), and Local Line
        query_dot = Dot(point=[-4, 0, 0], color=WHITE)
        weight_field = Circle(radius=0.8, color=YELLOW, fill_opacity=0.1, stroke_opacity=0.3)
        local_line = Line(color=YELLOW, stroke_width=6)
        self.add(weight_field, query_dot, local_line)
        
        # 5. Regression Calculation & Visual Update Logic
        def update_lwr(mobj):
            qx = query_dot.get_x()
            tau = 0.5  # Bandwidth
            
            # Manual Weighted Linear Regression: y = mx + b
            sum_w = 0
            sum_wx = 0
            sum_wy = 0
            sum_wxx = 0
            sum_wxy = 0
            
            for i in range(len(data_x)):
                xi = data_x[i]
                yi = data_y[i]
                
                # Formula: w = exp(-(xi - qx)^2 / (2 * tau^2))
                dist_sq = (xi - qx)**2
                weight = math.exp(-dist_sq / (2 * (tau**2)))
                
                # Visuals: Glowing dots nearby
                dots[i].set_opacity(0.2 + 0.8 * weight)
                if weight > 0.4:
                    dots[i].set_color(RED)
                else:
                    dots[i].set_color(GREY)
                
                # Stats for Weighted Least Squares
                sum_w += weight
                sum_wx += weight * xi
                sum_wy += weight * yi
                sum_wxx += weight * xi * xi
                sum_wxy += weight * xi * yi
            
            if sum_w > 1e-5:
                # Calculate Weighted Means
                mean_x = sum_wx / sum_w
                mean_y = sum_wy / sum_w
                
                # Calculate Slope (m) and Intercept (b)
                numerator = sum_wxy - mean_x * sum_wy
                denominator = sum_wxx - mean_x * sum_wx
                
                slope = numerator / denominator if abs(denominator) > 1e-7 else 0
                intercept = mean_y - slope * mean_x
                
                # Set Query Dot's vertical position based on regression
                qy = slope * qx + intercept
                query_dot.move_to([qx, qy, 0])
                weight_field.move_to([qx, qy, 0])
                
                # Update Local Regression Line span
                lx1, lx2 = qx - 0.8, qx + 0.8
                ly1, ly2 = slope * lx1 + intercept, slope * lx2 + intercept
                local_line.put_start_and_end_on([lx1, ly1, 0], [lx2, ly2, 0])
        
        # Assign the updater to the line so it recalculates every frame
        local_line.add_updater(update_lwr)
        
        # 6. Animation: Move query point across the domain
        self.play(
            query_dot.animate.shift(RIGHT * 8),
            run_time=10,
            rate_func=lambda t: t # Linear movement
        )
        
        self.wait(2)
