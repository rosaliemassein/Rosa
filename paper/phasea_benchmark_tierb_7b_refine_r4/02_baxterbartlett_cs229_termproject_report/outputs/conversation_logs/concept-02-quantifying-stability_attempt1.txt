from manim import *
import numpy as np

class StabilityExplanation(Scene):
    def construct(self):
        # 1. Manual Coordinate System for Velocity Plot (Replacing Axes)
        # Using Line instead of Axes to comply with constraints
        graph_origin = LEFT * 4 + DOWN * 1.5
        horiz_axis = Line(graph_origin, graph_origin + RIGHT * 5, color=WHITE)
        vert_axis = Line(graph_origin, graph_origin + UP * 4, color=WHITE)
        
        horiz_label = Text("Time", font_size=20).next_to(horiz_axis, DOWN)
        vert_label = Text("Velocity (RPM)", font_size=20).rotate(90 * DEGREES).next_to(vert_axis, LEFT)
        
        self.play(Create(horiz_axis), Create(vert_axis))
        self.play(Write(horiz_label), Write(vert_label))

        # 2. Velocity Plot (Fluctuating around 30 RPM)
        # Generate points manually to avoid 'x' variable issues in lambda
        t_vals = np.linspace(0, 5, 100)
        curve_points = []
        for t_step in t_vals:
            # Base height 2.5 (representing ~30 RPM) + fluctuations
            y_fluctuation = 2.5 + 0.3 * np.sin(t_step * 5) + 0.2 * np.cos(t_step * 11)
            curve_points.append(graph_origin + RIGHT * t_step + UP * y_fluctuation)
        
        velocity_curve = VMobject(color=YELLOW)
        velocity_curve.set_points_as_corners(curve_points)
        
        plot_title = Text("Motor Behavior", font_size=24).next_to(vert_axis, UP)
        self.play(Create(velocity_curve), Write(plot_title))
        self.wait(0.5)

        # 3. Highlight last 20% of the x-axis
        # The axis is 5 units wide, so 20% is 1 unit.
        highlight_rect = Rectangle(
            width=1.0, 
            height=3.8, 
            fill_color=BLUE, 
            fill_opacity=0.3, 
            stroke_width=0
        ).move_to(graph_origin + RIGHT * 4.5 + UP * 1.9)
        
        highlight_text = Text("Final 20%", font_size=18, color=BLUE).next_to(highlight_rect, UP)
        self.play(FadeIn(highlight_rect), Write(highlight_text))
        self.wait()

        # 4. Histogram on the side
        hist_origin = RIGHT * 2.5 + DOWN * 1.5
        hist_h_line = Line(hist_origin, hist_origin + RIGHT * 4, color=WHITE)
        
        # Defining bar heights: the middle one is the mode (tallest)
        bar_heights = [0.7, 1.2, 3.2, 1.5, 0.8] 
        bars = VGroup()
        for i, h in enumerate(bar_heights):
            bar = Rectangle(
                width=0.5, 
                height=h,
                fill_color=YELLOW if i == 2 else BLUE,
                fill_opacity=0.7
            ).move_to(hist_origin + RIGHT * (0.6 + i * 0.7) + UP * (h / 2))
            bars.add(bar)
        
        hist_title = Text("Histogram", font_size=24).next_to(bars, UP)
        mode_label = Text("30 RPM", font_size=16).next_to(bars[2], DOWN)
        
        self.play(Create(hist_h_line))
        self.play(Create(bars), Write(hist_title), FadeIn(mode_label))
        self.wait()

        # 5. Stability Formula
        stability_formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=32
        ).to_edge(DOWN, buff=0.5)
        
        # Background for formula visibility
        formula_bg = SurroundingRectangle(stability_formula, color=BLACK, fill_opacity=0.8, stroke_width=0)
        
        self.play(FadeIn(formula_bg), Write(stability_formula))
        
        # Simple animation highlighting parts of the fraction
        num_box = SurroundingRectangle(bars[2], color=YELLOW)
        den_bars = VGroup(*[bars[j] for j in [0, 1, 3, 4]])
        den_box = SurroundingRectangle(den_bars, color=BLUE)
        
        self.play(Create(num_box))
        self.wait(0.5)
        self.play(ReplacementTransform(num_box, den_box))
        self.wait(0.5)
        self.play(FadeOut(den_box))
        
        self.wait(2)
