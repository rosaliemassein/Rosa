import math

class InverseMapping(Scene):

        output_rect = Rectangle(width=3.5, height=1.5, color=GREEN)
        output_title = Text("PID Constants", font_size=24).move_to(output_rect.get_center())
        output_group = VGroup(output_rect, output_title).shift(UP * 2 + RIGHT * 3)

        arrow = Line(input_group.get_right(), output_group.get_left(), color=YELLOW).add_tip()
        
        diagram = VGroup(input_group, output_group, arrow)

        # 2. Formula Section
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=36).shift(UP * 0.5)

        # 3. Plot Section (Manual construction to avoid restricted identifiers)
        # Origin for the plot
        origin_point = DOWN * 3 + LEFT * 4
        x_axis = Line(origin_point, origin_point + RIGHT * 8, color=WHITE)

        x_label = Text("Time", font_size=16).next_to(x_axis, DOWN)
        y_label = Text("RPM", font_size=16).next_to(y_axis, LEFT)

        setpoint_y = origin_point[1] + 1.5
        setpoint_line = VGroup(*[
            Line(
                [origin_point[0] + i * 0.2, setpoint_y, 0], 
                [origin_point[0] + i * 0.2 + 0.1, setpoint_y, 0],
                color=GRAY
            ) for i in range(40)
        ])
        setpoint_text = Text("Setpoint", font_size=14).move_to([origin_point[0] + 7.5, setpoint_y + 0.2, 0])

        # Curve: Damped oscillation manually calculated
        curve = VMobject(color=RED)
        points = []

            # Formula: 1.5 * (1 - exp(-0.7*t) * cos(3*t))
            val = 1.5 * (1.0 - math.exp(-0.7 * t_val) * math.cos(3.0 * t_val))
            points.append([origin_point[0] + t_val, origin_point[1] + val, 0])
        curve.set_points_as_corners(points)

        rise_label = Text("Rise Time", font_size=16, color=BLUE).next_to(rise_line, DOWN, buff=0.1)
        rise_group = VGroup(rise_line, rise_tick1, rise_tick2, rise_label)

        # Steady State Error (difference at the end)

        error_line = Line([error_x + 0.2, origin_point[1] + 1.5, 0], [error_x + 0.2, points[-1][1], 0], color=GREEN)
