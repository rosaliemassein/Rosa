import numpy as np
from manim import *

class Concept02QuantifyingStability(Scene):
    def construct(self):
        # 1. Manual Coordinate System (Since Axes/NumberPlane are restricted)
        origin = LEFT * 5 + DOWN * 2
        x_len = 6
        y_len = 4
        
        x_axis = Line(origin, origin + RIGHT * x_len, color=WHITE)
        y_axis = Line(origin, origin + UP * y_len, color=WHITE)
        
        x_label = Text("Time", font_size=20).next_to(x_axis, DOWN)
        y_label = Text("RPM", font_size=20).next_to(y_axis, LEFT)
        
        # Grid lines/Ticks for reference
        y_30_pos = origin + UP * (y_len * 0.5) # 30 is midpoint between 20 and 40
        tick_30 = Line(y_30_pos + LEFT * 0.1, y_30_pos + RIGHT * 0.1)
        label_30 = Text("30", font_size=16).next_to(tick_30, LEFT)

        self.add(x_axis, y_axis, x_label, y_label, tick_30, label_30)

        # 2. Fluctuating Velocity Line
        # Mapping: X[0,10] -> [0, 6], Y[20, 40] -> [0, 4]
        def get_pos(t, rpm):
            return origin + RIGHT * (t * x_len / 10) + UP * ((rpm - 20) * y_len / 20)

        points = []
        for t in np.linspace(0, 10, 100):
            rpm = 30 + 0.8 * np.sin(t * 5) + 0.4 * np.cos(t * 12)
            points.append(get_pos(t, rpm))
        
        velocity_curve = VMobject(color=YELLOW)
        velocity_curve.set_points_as_corners(points)
        
        self.play(Create(velocity_curve), run_time=2)
        self.wait(0.5)

        # 3. Highlight last 20% (Time 8 to 10)
        p_start = get_pos(8, 20)
        p_end = get_pos(10, 40)
        
        highlight = Rectangle(
            width=p_end[0] - p_start[0],
            height=y_len,
            fill_color=BLUE,
            fill_opacity=0.3,
            stroke_width=0
        ).move_to(origin + RIGHT * (x_len * 0.9) + UP * (y_len / 2))
        
        sse_text = Text("Steady State Region", font_size=16, color=BLUE).next_to(highlight, UP)
        
        self.play(FadeIn(highlight), Write(sse_text))
        self.wait(1)

        # 4. Histogram on the right side
        hist_center = RIGHT * 3.5 + DOWN * 1
        bar_values = [0.5, 1.2, 3.5, 1.0, 0.4]
        bars = VGroup()
        for i, val in enumerate(bar_values):
            b = Rectangle(
                width=0.4,
                height=val,
                fill_opacity=0.8,
                fill_color=GREEN if i == 2 else WHITE,
                stroke_width=1
            )
            bars.add(b)
        
        bars.arrange(RIGHT, buff=0.1, aligned_edge=DOWN).move_to(hist_center, aligned_edge=DOWN)
        hist_title = Text("Histogram", font_size=20).next_to(bars, UP, buff=0.5)
        mode_tag = Text("Mode (30)", font_size=14, color=GREEN).next_to(bars[2], DOWN)
        
        self.play(
            FadeIn(hist_title),
            LaggedStart(*[Create(b) for b in bars], lag_ratio=0.1),
            Write(mode_tag)
        )
        self.wait(1)

        # 5. Stability Formula
        formula = MathTex(
            r"\text{Stability} = \frac{\text{counts at mode}}{\text{counts not at mode}}",
            font_size=30
        ).to_edge(UP, buff=0.5)
        
        rect = SurroundingRectangle(formula, color=WHITE, buff=0.2)
        
        self.play(Create(rect), Write(formula))
        self.wait(1)

        # 6. Conclusion
        explanation = Text(
            "Quantifying qualitative behavior into features for regression.",
            font_size=18,
            color=GRAY
        ).to_edge(DOWN, buff=0.4)
        
        self.play(FadeIn(explanation))
        self.wait(2)
