from manim import *
import numpy as np

class CosineSimilarityScene(Scene):
    def construct(self):
        # 1. Custom Axes using Line objects
        x_axis = Line(LEFT * 2, RIGHT * 2, color=BLUE)
        y_axis = Line(DOWN * 2, UP * 2, color=BLUE)
        axes_group = VGroup(x_axis, y_axis).to_edge(LEFT, buff=1)
        origin = axes_group.get_center()

        # Vector A (Fixed)
        vec_a = Line(origin, origin + RIGHT * 1.5, color=RED)
        
        # Vector B (Rotating)
        vec_b = Line(origin, origin + RIGHT * 1.5, color=GREEN)
        
        # Use a hidden Mobject as a tracker (since ValueTracker is disallowed)
        # tracker.get_x() will represent the angle in degrees
        tracker = Dot(radius=0).move_to(RIGHT * 60) 

        def update_vector_b(mob):
            angle_rad = tracker.get_x() * DEGREES
            # Start from origin, point in direction of angle
            direction = np.array([np.cos(angle_rad), np.sin(angle_rad), 0])
            mob.put_start_and_end_on(origin, origin + direction * 1.5)

        vec_b.add_updater(update_vector_b)

        # 2. Angle Arc
        # Using Arc instead of Angle (since Angle is disallowed)
        arc = Arc(radius=0.5, start_angle=0, angle=60 * DEGREES, arc_center=origin, color=WHITE)
        
        def update_arc(mob):
            angle_rad = tracker.get_x() * DEGREES
            new_arc = Arc(
                radius=0.5, 
                start_angle=0, 
                angle=max(0.001, angle_rad), 
                arc_center=origin, 
                color=WHITE
            )
            mob.become(new_arc)

        arc.add_updater(update_arc)

        # 3. Text and Formula
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=24
        ).to_edge(UP)
        
        sim_label = MathTex(r"\cos(\theta) = ", font_size=36).shift(UP * 2 + RIGHT * 3)
        sim_val_text = MathTex("0.50", font_size=36).next_to(sim_label, RIGHT)

        def update_sim_text(mob):
            angle_rad = tracker.get_x() * DEGREES
            val = np.cos(angle_rad)
            new_tex = MathTex(f"{val:.2f}", font_size=36).next_to(sim_label, RIGHT)
            mob.become(new_tex)

        sim_val_text.add_updater(update_sim_text)

        # Initial Scene Layout
        self.add(axes_group, vec_a, vec_b, arc, formula, sim_label, sim_val_text)
        self.wait(1)

        # Animate alignment (theta -> 0)
        self.play(tracker.animate.move_to(ORIGIN), run_time=3)
        self.wait(1)
        
        # Animate divergence (theta -> 90)
        self.play(tracker.animate.move_to(RIGHT * 90), run_time=3)
        self.wait(1)

        # 4. Transition to Matrix
        # Clean up updaters before fading out
        vec_b.remove_updater(update_vector_b)
        arc.remove_updater(update_arc)
        sim_val_text.remove_updater(update_sim_text)

        self.play(
            FadeOut(axes_group),
            FadeOut(vec_a),
            FadeOut(vec_b),
            FadeOut(arc),
            FadeOut(sim_label),
            FadeOut(sim_val_text)
        )

        # Create 5x5 Similarity Matrix
        matrix_group = VGroup()
        cell_size = 0.6
        for i in range(5):
            for j in range(5):
                # Using basic Square and setting fill manually
                sq = Square(side_length=cell_size)
                sq.set_stroke(WHITE, width=1)
                sq.set_fill(BLUE, opacity=0.3)
                sq.move_to(np.array([j * 0.7 - 1.4, -i * 0.7 + 1.0, 0]))
                matrix_group.add(sq)
        
        self.play(Create(matrix_group))
        self.wait(0.5)

        # Matrix Glow/Heatmap simulation
        # Highlight cells by changing fill color and opacity
        for _ in range(2):
            glow_anims = []
            for sq in matrix_group:
                # Randomly determine similarity intensity
                intensity = np.random.random()
                if intensity > 0.7:
                    color = YELLOW
                elif intensity > 0.4:
                    color = WHITE
                else:
                    color = BLUE
                glow_anims.append(sq.animate.set_fill(color, opacity=intensity))
            
            self.play(AnimationGroup(*glow_anims), run_time=1)
            self.wait(0.1)

        # Final match highlight
        final_match = matrix_group[12] # middle cell
        self.play(
            final_match.animate.set_fill(YELLOW, opacity=1).scale(1.2),
            run_time=1
        )
        self.wait(2)
