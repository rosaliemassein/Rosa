from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # Using a hidden mobject for tracking since ValueTracker is disallowed
        tracker = Dot(point=[PI / 3, 0, 0]).hide()
        
        # Manual Axes using Lines to replace disallowed Axes mobject
        origin_pt = [-3.5, -1.5, 0]
        x_axis = Line(origin_pt, [origin_pt[0] + 4, origin_pt[1], 0], color=WHITE)
        y_axis = Line(origin_pt, [origin_pt[0], origin_pt[1] + 4, 0], color=WHITE)
        x_label = MathTex("x").next_to(x_axis, RIGHT)
        y_label = MathTex("y").next_to(y_axis, UP)
        self.add(x_axis, y_axis, x_label, y_label)

        # Vector A (Stationary)
        vec_a = Arrow(origin_pt, [origin_pt[0] + 3, origin_pt[1], 0], color=RED, buff=0)
        self.add(vec_a)
        
        # Vector B (Dynamic) - using Arrow instead of Vector
        vec_b = Arrow(origin_pt, [origin_pt[0] + 3, origin_pt[1], 0], color=BLUE, buff=0)
        def update_vec_b(mob):
            theta = tracker.get_center()[0]
            end_pos = [
                origin_pt[0] + 3 * np.cos(theta), 
                origin_pt[1] + 3 * np.sin(theta), 
                0
            ]
            mob.put_start_and_end_on(origin_pt, end_pos)
        
        vec_b.add_updater(update_vec_b)
        self.add(vec_b)

        # Angle Theta represented by an Arc (replacing disallowed Angle mobject)
        angle_arc = Arc(radius=0.6, start_angle=0, angle=PI/3, arc_center=origin_pt, color=YELLOW)
        def update_arc(mob):
            # Ensure angle is at least a small positive value to avoid Arc errors
            theta = max(tracker.get_center()[0], 0.001)
            new_arc = Arc(radius=0.6, start_angle=0, angle=theta, arc_center=origin_pt, color=YELLOW)
            mob.become(new_arc)
        
        angle_arc.add_updater(update_arc)
        self.add(angle_arc)
        
        # Formula display
        formula = MathTex(r"\text{similarity} = \cos(\theta)", font_size=36).to_edge(UP)
        self.add(formula)
        
        # Similarity value update using MathTex instead of DecimalNumber
        sim_val_text = MathTex("0.50", font_size=36).next_to(formula, DOWN)
        def update_sim_val(mob):
            val = np.cos(tracker.get_center()[0])
            # Construct a simple text string for the number
            new_val = MathTex(f"{val:.2f}", font_size=36).next_to(formula, DOWN)
            mob.become(new_val)
        
        sim_val_text.add_updater(update_sim_val)
        self.add(sim_val_text)

        # 5x5 Similarity Matrix Construction
        matrix = VGroup()
        for i in range(5):
            row = VGroup()
            for j in range(5):
                # Start with randomized opacity for effect
                sq_opacity = 0.1 + (i + j) * 0.05
                sq = Square(side_length=0.55, fill_opacity=sq_opacity, fill_color=BLUE, stroke_width=1, color=GRAY_A)
                row.add(sq)
            row.arrange(RIGHT, buff=0.1)
            matrix.add(row)
        
        matrix.arrange(DOWN, buff=0.1).to_edge(RIGHT, buff=1)
        matrix_label = MathTex(r"\text{Similarity Matrix}", font_size=30).next_to(matrix, UP)
        self.add(matrix, matrix_label)
        
        # Specific cell interaction (middle cell 2,2)
        target_cell = matrix[2][2]
        def update_target_cell(mob):
            sim = np.cos(tracker.get_center()[0])
            # Map similarity to color manually since interpolate_color failed
            if sim > 0.9:
                mob.set_fill(YELLOW, opacity=1.0)
            elif sim > 0.6:
                mob.set_fill(WHITE, opacity=sim)
            elif sim > 0.3:
                mob.set_fill(BLUE, opacity=0.5)
            else:
                mob.set_fill(BLACK, opacity=0.7)
        
        target_cell.add_updater(update_target_cell)

        # Animation Sequence
        # Rotate B to align with A (similarity -> 1)
        self.play(tracker.animate.move_to([0.05, 0, 0]), run_time=3)
        self.wait(1)
        
        # Rotate B away (similarity -> 0)
        self.play(tracker.animate.move_to([PI / 2, 0, 0]), run_time=3)
        self.wait(2)
