from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # We use a Dot's x-coordinate to track the angle value (in radians)
        # to avoid disallowed ValueTracker and potential undefined identifier issues.
        # Initial angle ~70 degrees
        angle_tracker = Dot(point=[1.2, 0, 0]).hide()
        self.add(angle_tracker)

        # Manual Coordinate System (Replacement for Axes)
        center_origin = LEFT * 3 + DOWN * 0.5
        x_line = Line(center_origin, center_origin + RIGHT * 3, color=WHITE)
        y_line = Line(center_origin, center_origin + UP * 3, color=WHITE)
        self.add(x_line, y_line)

        # Vector A - Fixed along the X-axis
        vector_a = Line(center_origin, center_origin + RIGHT * 2.5, color=BLUE, stroke_width=6)
        label_a = MathTex(r"\mathbf{A}", color=BLUE).next_to(vector_a, DOWN, buff=0.1)
        self.add(vector_a, label_a)

        # Vector B - Dynamic rotation
        vector_b = Line(center_origin, center_origin + RIGHT * 2.5, color=RED, stroke_width=6)
        def update_vector_b(mob):
            current_angle = angle_tracker.get_center()[0]
            end_pos = center_origin + np.array([np.cos(current_angle), np.sin(current_angle), 0]) * 2.5
            mob.put_start_and_end_on(center_origin, end_pos)
        
        vector_b.add_updater(update_vector_b)
        
        label_b = MathTex(r"\mathbf{B}", color=RED)
        def update_label_b(mob):
            mob.next_to(vector_b.get_end(), UR, buff=0.1)
        label_b.add_updater(update_label_b)
        
        self.add(vector_b, label_b)

        # Angle visualization (Replacement for Angle)
        arc_repr = Arc(radius=0.6, start_angle=0, angle=1.2, arc_center=center_origin, color=WHITE)
        def update_arc_repr(mob):
            current_angle = angle_tracker.get_center()[0]
            new_arc = Arc(radius=0.6, start_angle=0, angle=current_angle, arc_center=center_origin, color=WHITE)
            mob.become(new_arc)
        
        arc_repr.add_updater(update_arc_repr)
        self.add(arc_repr)

        # Similarity Score Text and Formula
        formula_tex = MathTex(r"\text{similarity} = \cos(\theta)").to_edge(UP, buff=0.5)
        score_tex = MathTex("= 0.36", color=YELLOW).next_to(formula_tex, DOWN)
        
        def update_score_tex(mob):
            current_angle = angle_tracker.get_center()[0]
            val = np.cos(current_angle)
            # Use format to avoid potential f-string issues
            new_tex = MathTex("= {:.2f}".format(val), color=YELLOW).next_to(formula_tex, DOWN)
            mob.become(new_tex)
        
        score_tex.add_updater(update_score_tex)
        self.add(formula_tex, score_tex)

        # Similarity Matrix (5x5 grid of squares)
        matrix_group = VGroup()
        squares_list = []
        grid_start = RIGHT * 2 + UP * 1
        for i in range(5):
            for j in range(5):
                sq = Square(side_length=0.6, fill_opacity=1, stroke_width=2, color=GREY)
                sq.move_to(grid_start + DOWN * i * 0.65 + RIGHT * j * 0.65)
                # Diagonals represent self-similarity (1.0)
                if i == j:
                    sq.set_fill(WHITE, opacity=0.9)
                else:
                    sq.set_fill(BLUE, opacity=0.2)
                matrix_group.add(sq)
                squares_list.append(sq)
        
        self.add(matrix_group)

        # Link a specific cell to the vector alignment visualization
        # Cell at row 1, col 2
        correlation_cell = squares_list[7]
        cell_highlight = SurroundingRectangle(correlation_cell, color=YELLOW, buff=0)
        
        def update_cell_visuals(mob):
            current_angle = angle_tracker.get_center()[0]
            sim_val = np.cos(current_angle)
            if sim_val > 0.8:
                mob.set_fill(YELLOW, opacity=sim_val)
            elif sim_val > 0.4:
                mob.set_fill(WHITE, opacity=sim_val)
            else:
                mob.set_fill(BLUE, opacity=max(0.2, sim_val))
        
        correlation_cell.add_updater(update_cell_visuals)
        self.add(cell_highlight)

        # Animation Sequence
        # 1. Vectors align (theta -> 0)
        self.play(angle_tracker.animate.move_to([0.05, 0, 0]), run_time=4)
        self.wait(1)
        # 2. Vectors diverge (theta -> 90 degrees/1.5 rad)
        self.play(angle_tracker.animate.move_to([1.5, 0, 0]), run_time=4)
        self.wait(2)
