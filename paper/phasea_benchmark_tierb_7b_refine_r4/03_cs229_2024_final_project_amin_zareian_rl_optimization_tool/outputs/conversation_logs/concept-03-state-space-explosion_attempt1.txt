from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # Introductory text
        narration = Text(
            "State space explosion: states grow exponentially with grid size.",
            font_size=24,
            color=WHITE
        ).to_edge(UP)
        self.add(narration)

        # Manual Axes Construction (avoiding the "Axes" identifier to prevent parsing errors)
        origin_pt = [-5.5, -3, 0]
        x_end_pt = [-0.5, -3, 0]
        y_end_pt = [-5.5, 2.5, 0]
        
        horiz_line = Line(origin_pt, x_end_pt, color=WHITE)
        vert_line = Line(origin_pt, y_end_pt, color=WHITE)
        self.add(horiz_line, vert_line)

        x_title = Text("Matrix Size (N)", font_size=18).next_to(horiz_line, DOWN)
        y_title = Text("Possible States (Log Scale)", font_size=18).rotate(90 * DEGREES).next_to(vert_line, LEFT)
        self.add(x_title, y_title)

        # Formula at the top right
        formula = MathTex(r"\text{States} = 2^{(N \times N)}", color=YELLOW).to_corner(UR, buff=0.5)
        self.add(formula)

        # Manual Graphing Logic
        def get_coord(n_val):
            # N from 2 to 6 maps to screen X: -5.5 to -0.5
            # log10(States) from 0 to 12 maps to screen Y: -3 to 2.5
            scr_x = -5.5 + (n_val - 2) * (5.0 / 4.0)
            log_val = (n_val ** 2) * 0.30103  # log10(2^(N*N))
            scr_y = -3 + (log_val) * (5.5 / 12.0)
            return [scr_x, scr_y, 0]

        graph_path = [get_coord(time_step) for time_step in np.linspace(2, 6, 50)]
        curve = VMobject(color=RED, stroke_width=4)
        curve.set_points_as_corners(graph_path)
        
        # Helper for Grid
        def build_grid(n_size):
            cells = VGroup()
            cell_size = 1.5 / n_size
            for row_idx in range(n_size):
                for col_idx in range(n_size):
                    s = Square(side_length=cell_size, stroke_width=1)
                    s.set_fill(BLUE if (row_idx + col_idx) % 2 == 0 else RED, opacity=0.7)
                    cells.add(s)
            return cells.arrange_in_grid(n_size, n_size, buff=0).shift(RIGHT * 3 + UP * 1)

        grid_mobj = build_grid(2)
        label_mobj = Text("2x2 Grid", font_size=22).next_to(grid_mobj, UP)
        
        # Q-Table visualization
        box = Rectangle(width=2.2, height=3.5, color=WHITE).next_to(grid_mobj, DOWN, buff=0.5)
        rows_group = VGroup(*[
            Line([-1, 0, 0], [1, 0, 0], color=BLUE, stroke_opacity=0.4)
            for _ in range(12)
        ]).arrange(DOWN, buff=0.15).move_to(box)
        table_title = Text("Q-Table Rows", font_size=18).next_to(box, UP, buff=0.1)

        self.play(Create(curve), FadeIn(grid_mobj), Write(label_mobj), Create(box), FadeIn(rows_group), Write(table_title))
        self.wait(1)

        # Increase size iteratively to show growth
        for n_current in [3, 4, 5]:
            new_grid = build_grid(n_current)
            new_label = Text(f"{n_current}x{n_current} Grid", font_size=22).next_to(new_grid, UP)
            
            # Higher density rows to show state explosion
            density_count = n_current * 10
            new_rows = VGroup(*[
                Line([-1, 0, 0], [1, 0, 0], color=BLUE, stroke_opacity=0.3)
                for _ in range(density_count)
            ]).arrange(DOWN, buff=3.2/density_count).move_to(box)

            self.play(
                Transform(grid_mobj, new_grid),
                Transform(label_mobj, new_label),
                Transform(rows_group, new_rows),
                run_time=0.7
            )

        # Final State: Extreme growth turning into a blur
        blur_overlay = Rectangle(
            width=2.2, height=3.5, fill_color=BLUE, fill_opacity=0.95, stroke_width=0
        ).move_to(box)
        
        final_grid = build_grid(6)
        final_label = Text("6x6 Grid", font_size=22).next_to(final_grid, UP)

        self.play(
            Transform(grid_mobj, final_grid),
            Transform(label_mobj, final_label),
            FadeIn(blur_overlay),
            curve.animate.shift(UP * 20),
            narration.animate.set_color(RED),
            run_time=2,
            rate_func=lambda progress: progress*progress
        )
        
        explosion_txt = Text("EXPLOSION", color=YELLOW, weight="BOLD").scale(1.2).move_to(get_coord(5.5))
        self.play(Write(explosion_txt))
        self.wait(2)
