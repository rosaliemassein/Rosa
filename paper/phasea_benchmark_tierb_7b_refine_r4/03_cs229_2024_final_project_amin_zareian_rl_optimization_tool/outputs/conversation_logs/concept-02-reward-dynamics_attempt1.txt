from manim import *

class RewardDynamics(Scene):
    def construct(self):
        # 1. Setup 3x3 Grid
        # Initial state: center is RED, top neighbor is BLUE, others RED
        colors = [
            RED, BLUE, RED,
            RED, RED,  RED,
            RED, RED,  RED
        ]
        grid = VGroup(*[
            Square(side_length=1).set_fill(colors[i], opacity=0.8).set_stroke(GRAY, 1)
            for i in range(9)
        ]).arrange_in_grid(rows=3, cols=3, buff=0)
        grid.shift(LEFT * 3)

        center_pixel = grid[4]
        # Neighbors: top(1), left(3), right(5), bottom(7)
        neighbors = VGroup(grid[1], grid[3], grid[5], grid[7])

        # 2. Manual Bar Chart using Axes and Rectangles
        axes = Axes(
            x_range=[0, 4, 1],
            y_range=[0, 10, 2],
            x_length=4,
            y_length=3,
            axis_config={"include_tip": False}
        ).shift(RIGHT * 3.5 + DOWN * 0.5)
        
        chart_label = Text("Cumulative Reward", font_size=24).next_to(axes, UP)
        
        # Bars
        bar1 = Rectangle(width=0.5, height=0.5, fill_opacity=1, color=BLUE).move_to(axes.c2p(1, 0), BOTTOM)
        bar2 = Rectangle(width=0.5, height=1.0, fill_opacity=1, color=BLUE).move_to(axes.c2p(2, 0), BOTTOM)
        bar3 = Rectangle(width=0.5, height=1.0, fill_opacity=1, color=BLUE).move_to(axes.c2p(3, 0), BOTTOM)
        bars = VGroup(bar1, bar2, bar3)

        # 3. Formula
        formula = MathTex(r"R_t = \text{Border}_{t} - \text{Border}_{t-1}").to_edge(UP)

        self.add(grid, axes, chart_label, formula, bars)
        self.wait(1)

        # 4. Highlight target and neighbors
        focus_rect = SurroundingRectangle(center_pixel, color=YELLOW, buff=0)
        neighbor_highlights = VGroup(*[
            SurroundingRectangle(n, color=WHITE, buff=-0.05).set_stroke(width=2) 
            for n in neighbors
        ])
        
        self.play(Create(focus_rect))
        self.play(FadeIn(neighbor_highlights))
        self.wait(1)

        # 5. Animate Flip and Rewards
        # Visualizing the net change:
        # Before: 1 boundary (Center Red | Top Blue)
        # After flip to Blue: 3 boundaries (Center Blue | Left Red, Right Red, Bottom Red)
        # Net: +2
        
        # Define manual Cross (X mark)
        cross_line1 = Line(ORIGIN, 0.4*UR).shift(grid[1].get_bottom())
        cross_line2 = Line(ORIGIN, 0.4*UL).shift(grid[1].get_bottom())
        manual_cross = VGroup(cross_line1, cross_line2).set_color(RED).set_stroke(width=6)
        
        # Define Green Border lines
        g1 = Line(center_pixel.get_corner(DL), center_pixel.get_corner(UL)).set_color(GREEN).set_stroke(width=6)
        g2 = Line(center_pixel.get_corner(DR), center_pixel.get_corner(UR)).set_color(GREEN).set_stroke(width=6)
        g3 = Line(center_pixel.get_corner(DL), center_pixel.get_corner(DR)).set_color(GREEN).set_stroke(width=6)
        green_borders = VGroup(g1, g2, g3)

        reward_text = MathTex("+2", color=GREEN).next_to(center_pixel, UR, buff=0.1)

        self.play(
            center_pixel.animate.set_fill(BLUE),
            FadeOut(focus_rect),
            FadeOut(neighbor_highlights)
        )
        
        self.play(Create(manual_cross))
        self.play(Create(green_borders))
        self.play(Write(reward_text), reward_text.animate.shift(UP * 0.5))

        # 6. Update Bar Chart
        # Grow the 3rd bar to represent the new cumulative total
        new_bar3 = Rectangle(width=0.5, height=2.0, fill_opacity=1, color=BLUE).move_to(axes.c2p(3, 0), BOTTOM)
        
        self.play(Transform(bar3, new_bar3))
        self.wait(2)

        # Cleanup
        self.play(FadeOut(reward_text), FadeOut(manual_cross), FadeOut(green_borders))
        self.wait(1)
