from manim import *
import numpy as np

class GeneticSelectionAnimation(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex(r"P_{t+1} = \arg\max_{c \in \text{Children}(P_t)} \text{WinRate}(c)")
        formula.to_edge(DOWN)
        self.add(formula)

        # 2. Initial Parent
        parent_circle = Circle(radius=0.5, color=WHITE)
        parent_text = Text("Parent", font_size=24)
        parent = VGroup(parent_circle, parent_text).shift(UP * 2.5)
        self.play(Create(parent))

        # Evolution cycles
        for cycle in range(2):
            # 3. Create Children nodes
            children_labels = ["Child 1", "Child 2", "Child 3"]
            children = VGroup()
            lines = VGroup()
            
            # Positioning logic for branching
            for i in range(3):
                child_node = VGroup(
                    Circle(radius=0.4, color=WHITE),
                    Text(children_labels[i], font_size=20)
                )
                child_node.move_to(UP * 0.5 + LEFT * 4 + RIGHT * i * 4)
                line = Line(parent.get_bottom(), child_node.get_top(), buff=0.1)
                children.add(child_node)
                lines.add(line)
            
            self.play(Create(lines), Create(children))

            # 4. Create manual Bar Charts for each child (Axes + Rectangles)
            rates = [0.3, 0.8, 0.5] if cycle == 0 else [0.9, 0.4, 0.7]
            bars = VGroup()
            axes_group = VGroup()
            
            for i in range(3):
                # Simple axes
                ax = Axes(
                    x_range=[0, 1, 1],
                    y_range=[0, 1, 0.5],
                    x_length=0.8,
                    y_length=1.2,
                    axis_config={"include_tip": False, "stroke_width": 2}
                ).next_to(children[i], DOWN, buff=0.3)
                
                # Single bar as a rectangle
                bar = Rectangle(
                    width=0.4, 
                    height=0.01, # Start small for animation
                    fill_opacity=1, 
                    color=BLUE,
                    stroke_width=0
                ).move_to(ax.c2p(0.5, 0), aligned_edge=DOWN)
                
                axes_group.add(ax)
                bars.add(bar)

            self.play(Create(axes_group))
            
            # Animate bars growing
            growth_animations = []
            for i in range(3):
                target_height = rates[i] * 1.2 # scaled to y_length
                growth_animations.append(
                    bars[i].animate.stretch_to_fit_height(target_height).move_to(axes_group[i].c2p(0.5, 0), aligned_edge=DOWN)
                )
            self.play(*growth_animations)
            self.wait(0.5)
            
            # 5. Identify and Highlight Winner
            winner_idx = int(np.argmax(rates))
            gold_color = "#FFD700"
            
            self.play(
                children[winner_idx].animate.set_color(gold_color),
                bars[winner_idx].animate.set_color(gold_color)
            )
            self.wait(1)

            # 6. Evolutionary Transition
            # Group elements to fade out
            losers_and_axes = VGroup(
                parent, lines, axes_group,
                *[children[i] for i in range(3) if i != winner_idx],
                *[bars[i] for i in range(3)]
            )
            
            target_pos = UP * 2.5
            new_parent_label = Text("Parent", font_size=24).move_to(target_pos)
            
            self.play(
                FadeOut(losers_and_axes),
                children[winner_idx][0].animate.move_to(target_pos).set_color(WHITE),
                children[winner_idx][1].animate.become(new_parent_label).move_to(target_pos).set_color(WHITE)
            )
            
            # Reset parent reference
            parent = children[winner_idx]
            self.wait(0.5)

        # Final state
        self.play(parent.animate.scale(1.2).set_color(YELLOW))
        self.wait(2)
