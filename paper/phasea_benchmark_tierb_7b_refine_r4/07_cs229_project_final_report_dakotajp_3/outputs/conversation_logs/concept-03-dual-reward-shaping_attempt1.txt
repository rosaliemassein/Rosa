from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Setup Board and Target Zone
        # Using GREY_B for the board and a semi-transparent Circle for the goal
        board = Rectangle(width=10, height=6, color=GREY_B)
        target_zone = Circle(radius=1.0, color=GREEN, fill_opacity=0.3).shift(RIGHT * 3.5)
        target_label = Text("Target Zone", font_size=24, color=GREEN).next_to(target_zone, UP)
        
        # 2. Create a cluster of blue pegs
        pegs = VGroup(*[Dot(radius=0.15, color=BLUE) for _ in range(6)])
        pegs.arrange_in_grid(rows=3, cols=2, buff=0.4).shift(LEFT * 3.5)
        
        self.add(board, target_zone, target_label, pegs)
        
        # 3. Dynamic Centroid Logic
        # We use named functions for updaters to avoid potential lambda identifier issues
        centroid_dot = Dot(color=RED)
        def update_centroid_pos(obj):
            obj.move_to(pegs.get_center())
        centroid_dot.add_updater(update_centroid_pos)
        
        centroid_label = Text("Centroid", font_size=20, color=RED)
        def update_label_pos(obj):
            obj.next_to(centroid_dot, DOWN, buff=0.1)
        centroid_label.add_updater(update_label_pos)
        
        # 4. Connecting Line (Gravity Pull)
        # Using Line because DashedLine was flagged as disallowed in the environment
        gravity_line = Line(color=YELLOW)
        def update_line_points(obj):
            obj.put_start_and_end_on(
                pegs.get_center(), 
                target_zone.get_center()
            )
        gravity_line.add_updater(update_line_points)
        
        # 5. Reward Counter
        # Displays the formula R = w1(pegs_in)^2 + w2(dist)^-1
        reward_val = DecimalNumber(0, num_decimal_places=2, color=YELLOW).scale(1.2)
        reward_label = MathTex("R = ", color=YELLOW).scale(1.2)
        reward_group = VGroup(reward_label, reward_val).arrange(RIGHT).to_edge(UP)
        
        def update_reward_value(obj):
            # Calculate how many pegs are inside the target zone circle
            p_in = 0
            for p in pegs:
                if np.linalg.norm(p.get_center() - target_zone.get_center()) < 1.0:
                    p_in += 1
            
            # Distance from the bounding box center (centroid) to the target center
            dist = np.linalg.norm(pegs.get_center() - target_zone.get_center())
            
            # Reward formula calculation
            val = 5.0 * (p_in**2) + 2.0 * (1.0 / max(dist, 0.1))
            obj.set_value(val)

        reward_val.add_updater(update_reward_value)
        
        self.add(centroid_dot, centroid_label, gravity_line, reward_group)
        
        # 6. Animation: Moving pegs forward
        # As they move, the centroid, line, and reward counter update automatically
        pegs_count_inside = 0
        for i in range(len(pegs)):
            if i < 4:
                # Move these pegs into the target zone
                angle = i * (TAU / 4)
                offset = 0.5 * (np.cos(angle) * RIGHT + np.sin(angle) * UP)
                destination = target_zone.get_center() + offset
                
                self.play(pegs[i].animate.move_to(destination), run_time=0.8)
                
                # Show the squared bonus popup to emphasize non-linear reward
                pegs_count_inside += 1
                bonus_popup = MathTex(f"{pegs_count_inside}^2", color=YELLOW).scale(0.7).next_to(pegs[i], UP, buff=0.1)
                self.play(Write(bonus_popup), run_time=0.3)
                self.play(FadeOut(bonus_popup), run_time=0.3)
            else:
                # These pegs move forward but don't reach the goal yet
                self.play(pegs[i].animate.shift(RIGHT * 3.5), run_time=0.8)
        
        self.wait(2)
