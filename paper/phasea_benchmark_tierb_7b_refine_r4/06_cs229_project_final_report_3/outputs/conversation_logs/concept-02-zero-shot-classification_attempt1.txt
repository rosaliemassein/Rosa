from manim import *

class ZeroShotClassification(Scene):
    def construct(self):
        # 1. Image vector v representation
        v_rect = Rectangle(height=2.5, width=0.5, color=GREEN, fill_opacity=0.2)
        v_elements = VGroup(*[Line(LEFT*0.2, RIGHT*0.2, stroke_width=2) for _ in range(8)]).arrange(DOWN, buff=0.2).move_to(v_rect)
        v_label = MathTex(r"\mathbf{v}").next_to(v_rect, UP)
        v_group = VGroup(v_rect, v_elements, v_label).to_edge(LEFT, buff=1.0)

        # 2. Matrix X of country names (rows)
        countries = ["USA", "France", "Japan", "Brazil", "India"]
        X_rows = VGroup()
        for name in countries:
            r = Rectangle(height=0.45, width=2.2, stroke_width=1, color=BLUE)
            t = Text(name, font_size=18).move_to(r.get_center())
            X_rows.add(VGroup(r, t))
        X_rows.arrange(DOWN, buff=0.1)
        
        X_title = MathTex(r"\mathbf{X}").next_to(X_rows, UP)
        X_group = VGroup(X_rows, X_title).next_to(v_group, RIGHT, buff=1.2)

        self.play(FadeIn(v_group), FadeIn(X_group))
        self.wait(0.5)

        # 3. Dot product operation with pulsing highlight
        for i in range(len(countries)):
            pulse = X_rows[i][0].copy().set_stroke(YELLOW, 4).set_fill(YELLOW, opacity=0.3)
            self.play(
                v_rect.animate.set_stroke(color=YELLOW, width=6),
                FadeIn(pulse, scale=1.05),
                run_time=0.15
            )
            self.play(
                v_rect.animate.set_stroke(color=GREEN, width=2),
                FadeOut(pulse),
                run_time=0.1
            )

        # 4. Math formula at the top
        formula = MathTex(
            r"h(\mathbf{v};\mathbf{X})=\frac{\exp(\mathbf{X}\mathbf{v})}{\sum_{j=1}^{N}\exp(\mathbf{X}_{j}^{T}\mathbf{v})}",
            font_size=34
        ).to_edge(UP, buff=0.5)
        self.play(Write(formula))

        # 5. Manual Bar Chart implementation
        chart_origin = RIGHT * 1.5 + DOWN * 2
        x_axis = Line(chart_origin, chart_origin + RIGHT * 5, color=WHITE)
        
        bar_labels = VGroup()
        bars = VGroup()
        
        for i, name in enumerate(countries):
            # Label
            lbl = Text(name, font_size=14).next_to(chart_origin + RIGHT * (i * 1.0 + 0.5), DOWN, buff=0.2)
            bar_labels.add(lbl)
            
            # Initial tiny bar
            bar = Rectangle(
                width=0.6, 
                height=0.1, 
                fill_opacity=0.8, 
                color=BLUE_B,
                stroke_width=1
            )
            bar.move_to(chart_origin + RIGHT * (i * 1.0 + 0.5), aligned_edge=DOWN)
            bars.add(bar)

        self.play(Create(x_axis), Create(bar_labels), Create(bars))
        self.wait(0.5)

        # 6. Real-time update of Softmax probabilities (France is index 1)
        # Using specific values: France shoots up, others stay low
        softmax_probs = [0.12, 0.78, 0.04, 0.03, 0.03]
        
        bar_animations = []
        for i, p in enumerate(softmax_probs):
            target_height = p * 3.5 + 0.1 # Scaling height
            target_bar = bars[i].copy()
            target_bar.stretch_to_fit_height(target_height)
            target_bar.move_to(chart_origin + RIGHT * (i * 1.0 + 0.5), aligned_edge=DOWN)
            
            if i == 1: # France prediction
                target_bar.set_color(GREEN)
                target_bar.set_fill(GREEN, opacity=0.8)
            
            bar_animations.append(Transform(bars[i], target_bar))

        # Use a lambda for rate_func to avoid 'linear' identifier issues
        self.play(*bar_animations, run_time=2, rate_func=lambda t: t)
        
        # 7. Final highlight
        self.play(
            Indicate(bars[1], color=YELLOW),
            X_rows[1][0].animate.set_stroke(GREEN, 4),
            run_time=1.5
        )

        self.wait(2)
