from manim import *
import numpy as np
import random

class GravNetMessagePassing(Scene):
    def construct(self):
        # Create a formula display as requested
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula)

        # Create a scatter plot of dots in 2D (representing ECal cells)
        # We'll create an irregular grid layout
        dots = VGroup()
        for x in range(-3, 4):
            for y in range(-2, 3):
                # Base position
                pos = np.array([x * 1.3, y * 1.3, 0])
                # Add irregularity
                pos += np.array([random.uniform(-0.4, 0.4), random.uniform(-0.4, 0.4), 0])
                dot = Dot(point=pos, radius=0.08, color=BLUE)
                dots.add(dot)

        # Highlight one central dot
        center_idx = len(dots) // 2
        center_cell = dots[center_idx]
        
        self.play(Create(dots), run_time=2)
        self.play(center_cell.animate.set_color(YELLOW).scale(1.5))

        # Connection logic using distance-weighted function exp(-d^2)
        # We use a VGroup with an updater to handle the dynamic shifting
        connections = VGroup()
        
        def update_connections(mobj):
            mobj.clear()
            center_pos = center_cell.get_center()
            for other_dot in dots:
                if other_dot is center_cell:
                    continue
                
                # Calculate Euclidean distance
                diff = other_dot.get_center() - center_pos
                dist = np.linalg.norm(diff)
                
                # Apply the formula: exp(-d^2)
                # We scale d for better visual distribution in the coordinate space
                weight = np.exp(-(dist / 1.5)**2)
                
                # Only draw if the connection strength is significant
                if weight > 0.05:
                    # Animate thickness and opacity scaling with the weight
                    line = Line(
                        start=center_pos,
                        end=other_dot.get_center(),
                        stroke_width=weight * 12,
                        stroke_opacity=weight,
                        color=WHITE
                    )
                    mobj.add(line)
        
        connections.add_updater(update_connections)
        self.add(connections)
        self.wait(1)

        # As the 'learned features' update, show dots moving slightly
        # and connections shifting dynamically
        for _ in range(3):
            move_animations = []
            for d in dots:
                # Calculate a small random move
                shift_vec = np.array([random.uniform(-0.8, 0.8), random.uniform(-0.8, 0.8), 0])
                target_pos = d.get_center() + shift_vec
                
                # Keep dots within a reasonable frame
                target_pos[0] = np.clip(target_pos[0], -6, 6)
                target_pos[1] = np.clip(target_pos[1], -3.5, 3.5)
                
                move_animations.append(d.animate.move_to(target_pos))
            
            self.play(
                *move_animations,
                run_time=2.5,
                rate_func=rate_functions.smooth
            )
            self.wait(0.5)

        self.wait(2)
