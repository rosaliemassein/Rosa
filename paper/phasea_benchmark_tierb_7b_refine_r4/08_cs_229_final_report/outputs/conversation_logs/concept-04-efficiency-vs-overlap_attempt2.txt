from manim import *
import numpy as np

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 1.2, 0.2],
            x_length=9,
            y_length=5,
            axis_config={"include_tip": False}
        ).shift(DOWN * 0.5)
        
        # Labels using MathTex
        x_label = MathTex(r"\text{Distance between particles}").scale(0.7)
        x_label.next_to(axes.x_axis, DOWN)
        y_label = MathTex(r"\text{Efficiency}").scale(0.7).rotate(90 * DEGREES)
        y_label.next_to(axes.y_axis, LEFT)

        # 2. Define Curves using named functions to avoid lambda-related static analysis issues
        def ml_efficiency_func(x):
            return 0.9

        def standard_efficiency_func(x):
            # Sigmoid-like drop-off that falls to 0 at low x
            if x < 0: return 0
            return 0.9 / (1 + np.exp(-1.5 * (x - 3)))

        ml_curve = axes.plot(ml_efficiency_func, x_range=[0, 10], color=BLUE)
        std_curve = axes.plot(standard_efficiency_func, x_range=[0, 10], color=RED)
        
        ml_label = MathTex(r"\text{ML Model}", color=BLUE).scale(0.6).move_to(axes.c2p(8, 1.0))
        std_label = MathTex(r"\text{Standard Algorithm}", color=RED).scale(0.6).move_to(axes.c2p(8, 0.2))

        self.add(axes, x_label, y_label)
        self.play(Create(ml_curve), Create(std_curve), Write(ml_label), Write(std_label))
        self.wait(1)

        # 3. Animation Setup
        tracker = ValueTracker(9)

        # Vertical tracking line
        # Using a function and become() to update position
        v_line = Line(axes.c2p(9, 0), axes.c2p(9, 1.1), color=YELLOW, stroke_width=2)
        
        def update_v_line(mobj):
            current_x = tracker.get_value()
            new_line = Line(axes.c2p(current_x, 0), axes.c2p(current_x, 1.1), color=YELLOW, stroke_width=2)
            mobj.become(new_line)
        
        v_line.add_updater(update_v_line)

        # 4. Detector Icons Overlay
        detector_box = Rectangle(width=3.5, height=2, color=GRAY, fill_opacity=0.1).to_corner(UP + RIGHT)
        detector_text = MathTex(r"\text{Detector View}").scale(0.5).next_to(detector_box, UP, buff=0.1)
        box_center = detector_box.get_center()

        dot1 = Dot(color=WHITE, radius=0.08)
        dot2 = Dot(color=WHITE, radius=0.08)
        glow1 = Circle(radius=0.2, color=BLUE, stroke_opacity=0.3, fill_opacity=0.1)
        glow2 = Circle(radius=0.2, color=BLUE, stroke_opacity=0.3, fill_opacity=0.1)

        def update_p1(mobj):
            val = tracker.get_value()
            mobj.move_to(box_center + LEFT * (val * 0.12))

        def update_p2(mobj):
            val = tracker.get_value()
            mobj.move_to(box_center + RIGHT * (val * 0.12))

        def update_glow1(mobj):
            mobj.move_to(dot1.get_center())

        def update_glow2(mobj):
            mobj.move_to(dot2.get_center())

        dot1.add_updater(update_p1)
        dot2.add_updater(update_p2)
        glow1.add_updater(update_glow1)
        glow2.add_updater(update_glow2)

        self.add(v_line, detector_box, detector_text, dot1, dot2, glow1, glow2)

        # 5. Execute Animation
        # Moving from high distance to near-zero distance
        def linear_rate(t):
            return t

        self.play(
            tracker.animate.set_value(0.2),
            run_time=8,
            rate_func=linear_rate
        )
        self.wait(2)

        # Clean up updaters
        dot1.clear_updaters()
        dot2.clear_updaters()
        v_line.clear_updaters()
        glow1.clear_updaters()
        glow2.clear_updaters()
