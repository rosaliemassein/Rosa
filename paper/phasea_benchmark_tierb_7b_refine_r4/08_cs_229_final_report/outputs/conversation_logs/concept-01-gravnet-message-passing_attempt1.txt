from manim import *
import numpy as np

class GravNetConcept(Scene):
    def construct(self):
        # 1. Define positions for nodes (simulating irregular calorimeter cells)
        # Using [x, y, 0] coordinates to represent positions in feature space
        pts_coords = [
            [0, 0, 0],         # Center node (Index 0)
            [2.2, 0.4, 0],     # East
            [1.2, 2.1, 0],     # North-East
            [-1.5, 1.7, 0],    # North-West
            [-2.3, -0.3, 0],   # West
            [-0.7, -2.2, 0],   # South-West
            [1.6, -1.9, 0]     # South-East
        ]
        
        # 2. Create the nodes as Dots
        nodes = VGroup(*[Dot(p, color=GRAY) for p in pts_coords])
        central_node = nodes[0]
        central_node.set_color(GREEN).scale(1.5)
        
        # 3. Reference Formula for dynamic weighting
        formula = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        
        self.add(formula)
        self.play(FadeIn(nodes))
        self.wait(0.5)
        
        # 4. Helper function to generate an arrow with distance-weighted properties
        # This encapsulates the GravNet logic: potential depends on distance in feature space
        def get_weighted_arrow(start_node, end_node):
            p1 = start_node.get_center()
            p2 = end_node.get_center()
            dist = np.linalg.norm(p1 - p2)
            # The formula: weight = exp(-d^2). 
            # We scale dist by 0.5 for better visual contrast in the scene.
            weight = np.exp(-(dist * 0.5)**2)
            
            # Create a line with an arrow tip
            line = Line(p1, p2, color=BLUE, buff=0.1)
            line.add_tip(tip_length=0.15)
            
            # Dynamically set thickness and opacity based on the GravNet potential formula
            line.set_stroke(width=weight * 12, opacity=weight)
            return line

        # 5. Create connections and add updaters to handle dynamic motion
        links = VGroup()
        for i in range(1, len(nodes)):
            target = nodes[i]
            
            # Initialize the connection mob
            arrow_mob = get_weighted_arrow(central_node, target)
            
            # Define an updater function. We use t_node=target as a default argument
            # to capture the specific node instance for this closure.
            def update_link(m, t_node=target):
                new_arrow = get_weighted_arrow(central_node, t_node)
                m.become(new_arrow)
            
            arrow_mob.add_updater(update_link)
            links.add(arrow_mob)
        
        # Display the neighborhood and connections
        self.play(Create(links), lag_ratio=0.15, run_time=2)
        self.wait(1)
        
        # 6. Animate transition of dots into a new "learned feature space"
        # As dots move, the updaters will automatically recalculate weights and update the visuals.
        new_pts_coords = [
            [0.5, 0.3, 0],    # Central node shifts slightly
            [1.3, 0.8, 0],    # Move closer (becomes stronger)
            [-3.8, 2.5, 0],   # Move further (becomes weaker)
            [0.9, 0.4, 0],    # Move closer
            [-4.2, -1.8, 0],  # Move further
            [-1.2, -3.0, 0],  # Move further
            [1.4, -0.2, 0]    # Move closer
        ]
        
        # Prepare list of movement animations
        move_anims = []
        for i in range(len(nodes)):
            move_anims.append(nodes[i].animate.move_to(new_pts_coords[i]))
        
        # Play the transformation to emphasize the dynamic nature of GravNet connections
        self.play(*move_anims, run_time=5, rate_func=smooth)
        self.wait(2)
