from manim import *
import math

class EfficiencyVsOverlap(Scene):
    def construct(self):
        # Construction of axes using basic Line objects
        # Horizontal line from x=-4 to x=4, Vertical line from y=-2 to y=2.5
        h_line = Line(LEFT * 4 + DOWN * 2, RIGHT * 4 + DOWN * 2, color=GRAY)
        v_line_axis = Line(LEFT * 4 + DOWN * 2, LEFT * 4 + UP * 2.5, color=GRAY)
        self.add(h_line, v_line_axis)

        # Labels for axes
        h_lab = Text("Distance between particles", font_size=20).next_to(h_line, DOWN)
        v_lab = Text("Efficiency", font_size=20).rotate(90 * DEGREES).next_to(v_line_axis, LEFT)
        self.add(h_lab, v_lab)

        # ML Model Performance: Flat line at efficiency 0.9 (3.6 units above base)
        ml_curve_obj = Line(LEFT * 4 + UP * 1.6, RIGHT * 4 + UP * 1.6, color=BLUE)
        ml_text = Text("ML Model", color=BLUE, font_size=16).next_to(ml_curve_obj, UP, buff=0.1)

        # Standard Algorithm Performance: Drops as distance decreases
        # Manual construction of a dashed-like curve using segments
        std_group = VGroup()
        n_segments = 50
        for i in range(n_segments):
            # Distance value from 0 to 8
            d_a = (i / n_segments) * 8
            d_b = ((i + 0.5) / n_segments) * 8
            
            # Efficiency Calculation using math.exp to avoid numpy dependencies if restricted
            e_a = 0.9 * (1.0 - math.exp(-d_a / 1.5))
            e_b = 0.9 * (1.0 - math.exp(-d_b / 1.5))
            
            # Convert values to coordinate points
            p_a = LEFT * 4 + RIGHT * d_a + DOWN * 2 + UP * (e_a * 4.0)
            p_b = LEFT * 4 + RIGHT * d_b + DOWN * 2 + UP * (e_b * 4.0)
            
            std_group.add(Line(p_a, p_b, color=RED, stroke_width=2))
        
        std_text = Text("Standard Algorithm", color=RED, font_size=16).move_to(RIGHT * 2 + DOWN * 1)

        self.play(Create(ml_curve_obj), Write(ml_text))
        self.play(Create(std_group), Write(std_text))

        # Position marker dot (invisible) to track distance animation
        tracker_mobject = Dot(RIGHT * 4 + DOWN * 2)
        tracker_mobject.set_opacity(0)
        self.add(tracker_mobject)
        
        # Vertical marker line that follows the tracker
        v_marker = Line(DOWN * 2, UP * 2.5, color=WHITE, stroke_opacity=0.5)
        v_marker.add_updater(lambda m: m.move_to([tracker_mobject.get_center()[0], 0.25, 0]))
        
        # Particle Shower icons (glowing dots)
        s1 = Dot(radius=0.4, color=YELLOW, fill_opacity=0.3)
        s2 = Dot(radius=0.4, color=YELLOW, fill_opacity=0.3)
        c1 = Dot(radius=0.1, color=YELLOW)
        c2 = Dot(radius=0.1, color=YELLOW)
        showers = VGroup(s1, s2, c1, c2)
        
        def update_showers(m):
            # Logic: as tracker x moves from 4 to -4, distance current_d moves from 8 to 0
            current_d = tracker_mobject.get_center()[0] + 4.0
            gap_val = current_d * 0.2
            m[0].move_to(UP * 3.0 + LEFT * gap_val)
            m[1].move_to(UP * 3.0 + RIGHT * gap_val)
            m[2].move_to(m[0].get_center())
            m[3].move_to(m[1].get_center())

        showers.add_updater(update_showers)
        self.add(v_marker, showers)
        
        # Animate the particles moving closer (distance tracker moving left)
        # Standard curve height decreases as distance (x) decreases, while ML stays high.
        self.play(
            tracker_mobject.animate.move_to(LEFT * 3.95 + DOWN * 2),
            run_time=8,
            rate_func=lambda t: t
        )
        self.wait(2)
