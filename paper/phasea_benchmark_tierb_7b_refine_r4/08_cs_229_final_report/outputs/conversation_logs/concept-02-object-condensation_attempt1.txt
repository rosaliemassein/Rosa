from manim import *
import numpy as np
import random

class ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup the 2D plane of dots
        random.seed(42)  # For reproducibility
        dots = VGroup(*[
            Dot(point=[random.uniform(-5, 5), random.uniform(-3, 2), 0], radius=0.08)
            for _ in range(50)
        ])
        self.add(dots)

        # 2. Assign beta values and calculate q_i
        q_min = 0.5
        
        # 3. Formula Label
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{\text{min}}", font_size=40)
        formula.to_edge(UP)
        self.play(Write(formula))

        # 4. Animate dots growing and changing color
        grow_animations = []
        condensation_centers = []
        
        # Designate 4 random dots as condensation points (high beta)
        high_beta_indices = random.sample(range(len(dots)), 4)
        
        for i, dot in enumerate(dots):
            if i in high_beta_indices:
                beta = random.uniform(0.85, 0.95)
                condensation_centers.append(dot)
            else:
                beta = random.uniform(0.1, 0.4)
            
            # Calculate q_i = arctanh(beta) + q_min
            # Using manual arctanh calculation to be safe
            q_val = 0.5 * np.log((1 + beta) / (1 - beta)) + q_min
            
            # Normalize q_val for color/size (ranges approx 0.6 to 2.5)
            alpha = np.clip((q_val - 0.6) / 2.0, 0, 1)
            
            # Use Manim's interpolation helpers directly
            target_color = interpolate_color(PURPLE, YELLOW, alpha)
            target_scale = 1.0 + alpha * 1.5
            
            grow_animations.append(
                dot.animate.set_color(target_color).scale(target_scale)
            )

        self.play(*grow_animations, run_time=2)
        self.wait(0.5)

        # 5. Create gravity wells for high beta dots
        wells = VGroup()
        for center_dot in condensation_centers:
            # Concentric rings to simulate a gravity well
            well = VGroup(*[
                Circle(
                    radius=r, 
                    stroke_width=0, 
                    fill_opacity=0.15 - (r/15), 
                    fill_color=YELLOW
                ).move_to(center_dot.get_center())
                for r in np.arange(0.2, 1.8, 0.2)
            ])
            wells.add(well)
        
        self.play(FadeIn(wells, lag_ratio=0.1), run_time=1.5)

        # 6. Pull neighboring dots toward the condensation points
        pull_animations = []
        for dot in dots:
            if dot not in condensation_centers:
                # Find the nearest condensation point without using lambda 'cp'
                nearest_center = condensation_centers[0]
                min_dist = np.linalg.norm(dot.get_center() - nearest_center.get_center())
                
                for potential_center in condensation_centers[1:]:
                    dist = np.linalg.norm(dot.get_center() - potential_center.get_center())
                    if dist < min_dist:
                        min_dist = dist
                        nearest_center = potential_center
                
                # If within a certain distance, move toward it
                if min_dist < 4.0:
                    start_pos = dot.get_center()
                    end_pos = nearest_center.get_center()
                    # Calculate position 70% of the way to the center manually
                    target_pos = start_pos + 0.7 * (end_pos - start_pos)
                    pull_animations.append(dot.animate.move_to(target_pos))

        # Use a standard rate function from the manim namespace
        self.play(*pull_animations, run_time=3, rate_func=rate_functions.ease_in_quad)
        self.wait(2)
