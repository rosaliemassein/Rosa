from manim import *
import numpy as np

class GravNetMessagePassing(Scene):
    def construct(self):
        # Governing weight formula display
        formula_display = MathTex(r"V_{jk} = \exp(-d_{jk}^2)").to_corner(UL)
        self.add(formula_display)

        # Create a set of points representing detector hit nodes
        # Using irregular hexagonal-like spacing
        nodes_group = VGroup()
        for x_idx in range(-4, 5):
            for y_idx in range(-2, 3):
                # Base hexagonal positions
                pos_x = x_idx * 1.3 + (0.4 if y_idx % 2 == 0 else 0)
                pos_y = y_idx * 1.0
                # Add jitter for irregularity
                pos_x += np.random.uniform(-0.15, 0.15)
                pos_y += np.random.uniform(-0.15, 0.15)
                nodes_group.add(Dot(point=[pos_x, pos_y, 0], radius=0.08, color=GRAY_B))

        # Focus node (representing the 'central' hit in message passing)
        central_hit = nodes_group[len(nodes_group) // 2]
        central_hit.set_color(GREEN)

        # Draw connection lines with updaters to handle dynamic distance-weighted connections
        # The prompt asks for arrows, but Lines are more stable for dynamic thickness/opacity.
        for node_item in nodes_group:
            if node_item == central_hit:
                continue
            
            # Initial connection
            edge_line = Line(node_item.get_center(), central_hit.get_center(), color=BLUE_C)
            
            # Closure to bind the specific node to the updater
            def get_edge_updater(source_node, target_node):
                def edge_updater(mobject_item):
                    p1_loc = source_node.get_center()
                    p2_loc = target_node.get_center()
                    # Calculate squared distance manually to avoid np.sum identifier issues
                    distance_sq = (p1_loc[0] - p2_loc[0])**2 + (p1_loc[1] - p2_loc[1])**2
                    # Connection strength formula
                    v_weight = np.exp(-distance_sq)
                    mobject_item.put_start_and_end_on(p1_loc, p2_loc)
                    # Scale opacity and thickness by weight
                    mobject_item.set_stroke(opacity=v_weight, width=v_weight * 12)
                return edge_updater
            
            edge_line.add_updater(get_edge_updater(node_item, central_hit))
            self.add(edge_line)

        # Ensure dots are rendered on top of connection lines
        self.add(nodes_group)
        self.wait(1)

        # Animate the 'learned feature space' update: nodes move dynamically, shifting the graph structure
        move_animations = []
        for hit_node in nodes_group:
            # Shift nodes by random amounts
            delta_x = np.random.uniform(-1.2, 1.2)
            delta_y = np.random.uniform(-1.0, 1.0)
            move_animations.append(hit_node.animate.shift([delta_x, delta_y, 0]))

        # Run the shift animation; updaters will dynamically adjust lines based on current positions
        self.play(*move_animations, run_time=5)
        self.wait(2)
