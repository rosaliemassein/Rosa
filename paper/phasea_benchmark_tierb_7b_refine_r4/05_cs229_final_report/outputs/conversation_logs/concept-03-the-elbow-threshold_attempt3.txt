from manim import *
import math

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=36)
        formula.to_edge(UP)
        self.add(formula)

        # 2. Manual Coordinate System
        # We define a custom coordinate system since Axes was flagged as disallowed.
        origin_x = -5
        origin_y = -2
        graph_width = 6
        graph_height = 4
        
        x_axis = Line([origin_x, origin_y, 0], [origin_x + graph_width, origin_y, 0], color=WHITE)
        y_axis = Line([origin_x, origin_y, 0], [origin_x, origin_y + graph_height + 0.5, 0], color=WHITE)
        
        label_x = Tex("Fraction of Zones Flagged", font_size=24).next_to(x_axis, DOWN)
        label_y = Tex("Recall", font_size=24).next_to(y_axis, LEFT).rotate(90 * DEGREES)
        
        self.add(x_axis, y_axis, label_x, label_y)

        # 3. Curve Construction (y = 1 - e^-5x)
        # Defining a function for the concave curve logic
        def get_y_value(input_val):
            # Scale y range from [0, 1] to [0, graph_height]
            return (1.0 - math.exp(-5.0 * input_val)) * graph_height

        curve_points = []
        for i in range(101):
            prog = i / 100.0
            px = origin_x + prog * graph_width
            py = origin_y + get_y_value(prog)
            curve_points.append([px, py, 0])
        
        curve_line = VMobject(color=BLUE)
        curve_line.set_points_as_corners(curve_points)
        self.add(curve_line)

        # 4. Building Floorplan (6x6 Grid of Zones)
        zones = VGroup()
        for row_idx in range(6):
            for col_idx in range(6):
                sq_zone = Square(side_length=0.4, stroke_width=1, color=WHITE)
                sq_zone.move_to([3 + col_idx * 0.5, -1 + row_idx * 0.5, 0])
                zones.add(sq_zone)
        
        zones_label = Tex("Building Floorplan", font_size=30).next_to(zones, UP)
        self.add(zones, zones_label)

        # 5. Tracking Logic (Replacing ValueTracker with an animated Dot)
        # The x-coordinate of 'tracker_mobject' will represent our threshold value [0, 1]
        tracker_mobject = Dot(radius=0, color=BLACK).move_to([0.1, 0, 0])
        
        indicator_dot = Dot(color=YELLOW)
        horizontal_line = Line(color=GREY, stroke_width=1)
        vertical_line = Line(color=GREY, stroke_width=1)

        def update_indicator(mob):
            val = tracker_mobject.get_center()[0]
            px = origin_x + val * graph_width
            py = origin_y + get_y_value(val)
            mob.move_to([px, py, 0])

        def update_h_line(mob):
            val = tracker_mobject.get_center()[0]
            px = origin_x + val * graph_width
            py = origin_y + get_y_value(val)
            mob.set_points_as_corners([[origin_x, py, 0], [px, py, 0]])

        def update_v_line(mob):
            val = tracker_mobject.get_center()[0]
            px = origin_x + val * graph_width
            py = origin_y + get_y_value(val)
            mob.set_points_as_corners([[px, origin_y, 0], [px, py, 0]])

        def update_zones(mob):
            val = tracker_mobject.get_center()[0]
            # Map 0.0-1.0 to 0-36 zones
            active_count = int(val * 36)
            for idx, zone_sq in enumerate(mob):
                if idx < active_count:
                    zone_sq.set_fill(GREEN, opacity=0.7)
                else:
                    zone_sq.set_fill(BLACK, opacity=0)

        indicator_dot.add_updater(update_indicator)
        horizontal_line.add_updater(update_h_line)
        vertical_line.add_updater(update_v_line)
        zones.add_updater(update_zones)

        self.add(indicator_dot, horizontal_line, vertical_line)

        # 6. Animation Sequence
        # Sweep to high threshold
        self.play(tracker_mobject.animate.move_to([0.95, 0, 0]), run_time=3)
        self.wait(0.5)
        # Sweep to low threshold
        self.play(tracker_mobject.animate.move_to([0.05, 0, 0]), run_time=2)
        self.wait(0.5)
        # Target the 'elbow' point (around 0.3)
        self.play(tracker_mobject.animate.move_to([0.3, 0, 0]), run_time=2)
        
        # Highlight the Elbow area
        elbow_coords = [origin_x + 0.3 * graph_width, origin_y + get_y_value(0.3), 0]
        glow = Circle(radius=0.4, color=GOLD, stroke_width=4).move_to(elbow_coords)
        elbow_note = Tex("The Elbow", color=GOLD, font_size=32).next_to(glow, UR)
        
        self.play(Create(glow), Write(elbow_note))
        self.wait(3)
