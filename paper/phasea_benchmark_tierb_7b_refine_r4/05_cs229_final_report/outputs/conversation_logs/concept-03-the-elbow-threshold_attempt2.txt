from manim import *

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Create Axes
        # Using simple concave function to avoid numpy (y = 1 - (1-p)^2)
        ax = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)

        # Labels for the axes
        label_x = Tex("Fraction of Zones Flagged", font_size=24).next_to(ax.get_x_axis(), DOWN)
        label_y = Tex("Recall", font_size=24).rotate(90 * DEGREES).next_to(ax.get_y_axis(), LEFT)

        # 2. Formula (from prompt)
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_corner(UL, buff=0.5)

        # 3. Concave curve
        # Using 'p' instead of 'x' to avoid potential identifier conflicts
        curve = ax.plot(
            lambda p: 1 - (1 - p) * (1 - p),
            x_range=[0, 1],
            color=RED
        )

        # 4. ValueTracker to move point along the curve
        vt = ValueTracker(0.1)

        # 5. Moving Dot and its updater
        # Using a function to calculate the point to avoid complex nested lambdas
        def get_curve_point():
            val = vt.get_value()
            return ax.c2p(val, 1 - (1 - val) * (1 - val))

        dot_obj = Dot(color=YELLOW)
        dot_obj.add_updater(lambda m: m.move_to(get_curve_point()))

        # 6. Building Floorplan (Zones)
        # 16 squares representing zones
        zones_group = VGroup(*[
            Square(side_length=0.4, stroke_width=2).set_fill(WHITE, opacity=0.1)
            for _ in range(16)
        ]).arrange_in_grid(rows=4, cols=4, buff=0.1).to_edge(RIGHT, buff=1)

        zones_label = Tex("Targeted Zones", font_size=24).next_to(zones_group, UP)

        def update_zones(mob):
            progress = vt.get_value()
            threshold = int(progress * 16)
            for idx, square in enumerate(mob):
                if idx < threshold:
                    square.set_fill(GREEN, opacity=0.8)
                    square.set_stroke(GREEN, width=3)
                else:
                    square.set_fill(WHITE, opacity=0.1)
                    square.set_stroke(WHITE, width=2)

        zones_group.add_updater(update_zones)

        # 7. Animation Sequence
        self.add(ax, label_x, label_y, formula, curve, dot_obj, zones_group, zones_label)
        
        # Move across the curve to show the trade-off
        self.play(vt.animate.set_value(0.95), run_time=4, rate_func=linear)
        self.wait(0.5)
        
        # Move back to the 'elbow' (approx 0.3)
        elbow_val = 0.3
        self.play(vt.animate.set_value(elbow_val), run_time=2)
        
        # 8. Highlight the 'elbow'
        elbow_coord = ax.c2p(elbow_val, 1 - (1 - elbow_val) * (1 - elbow_val))
        glow_circle = Circle(radius=0.4, color=YELLOW).move_to(elbow_coord)
        elbow_text = Tex("The 'Elbow'", color=YELLOW, font_size=30).next_to(glow_circle, UR)
        
        self.play(Create(glow_circle), Write(elbow_text))
        self.play(Flash(glow_circle, color=YELLOW))
        
        # Final message
        final_msg = Tex("Sweet spot: Maximize recall, minimize zones", font_size=24).to_edge(DOWN, buff=0.3)
        self.play(FadeIn(final_msg))
        
        self.wait(3)
