from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Coordinate System for Sigmoid (Right side)
        axes = Axes(
            x_range=[-6, 6, 2],
            y_range=[0, 1.2, 0.5],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(RIGHT, buff=0.5)
        
        sigmoid_curve = axes.plot(lambda x: 1 / (1 + np.exp(-x)), color=YELLOW)
        sigmoid_label = MathTex(r"P(y=1) = \sigma(z)", color=YELLOW, font_size=30).next_to(axes, UP)
        
        self.play(Create(axes), Create(sigmoid_curve), Write(sigmoid_label))
        self.wait(0.5)

        # 2. Sliding Bars and Trackers (Left side)
        # Using ValueTrackers to drive the bar heights and dot position
        x1_tracker = ValueTracker(0.2)
        x30_tracker = ValueTracker(0.3)
        
        bar_bg_1 = Rectangle(width=0.6, height=3, stroke_color=WHITE).shift(LEFT * 4.5 + DOWN * 0.5)
        bar_bg_30 = Rectangle(width=0.6, height=3, stroke_color=WHITE).shift(LEFT * 2.5 + DOWN * 0.5)
        
        label_1 = MathTex("X_{1d}", font_size=28).next_to(bar_bg_1, DOWN)
        label_30 = MathTex("X_{30d}", font_size=28).next_to(bar_bg_30, DOWN)
        title_1 = Text("1-Day T-CSP", font_size=16).next_to(bar_bg_1, UP)
        title_30 = Text("30-Day T-CSP", font_size=16).next_to(bar_bg_30, UP)
        
        # Redraw bars based on tracker values (mapped 0-1 to height 0-3)
        bar_1 = always_redraw(lambda: Rectangle(
            width=0.6, 
            height=max(0.01, x1_tracker.get_value() * 3), 
            fill_color=BLUE, 
            fill_opacity=0.7,
            stroke_width=0
        ).move_to(bar_bg_1.get_bottom(), aligned_edge=DOWN))
        
        bar_30 = always_redraw(lambda: Rectangle(
            width=0.6, 
            height=max(0.01, x30_tracker.get_value() * 3), 
            fill_color=GREEN, 
            fill_opacity=0.7,
            stroke_width=0
        ).move_to(bar_bg_30.get_bottom(), aligned_edge=DOWN))
        
        self.play(
            Create(bar_bg_1), Create(bar_bg_30),
