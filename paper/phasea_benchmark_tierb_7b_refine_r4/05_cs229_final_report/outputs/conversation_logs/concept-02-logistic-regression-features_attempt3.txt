from manim import *

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # Sigmoid helper function (no external library imports used)
        def get_sig(val_z):
            return 1.0 / (1.0 + (2.7182818 ** (-val_z)))

        # 1. UI Elements - Right Side (Plot)
        # Create axes manually as NumberPlane/Axes are disallowed
        line_horiz = Line([1.0, -1.5, 0], [5.0, -1.5, 0])
        line_vert = Line([1.0, -1.5, 0], [1.0, 1.5, 0])
        self.add(line_horiz, line_vert)

        # Create Sigmoid Curve manually using VMobject
        pts_data = []
        for i in range(51):
            val_z_in = -4.0 + (i * 0.16)
            val_p_out = get_sig(val_z_in)
            # Map math values to screen coordinates
            # z from -4 to 4 maps to screen horizontal from 1.0 to 5.0
            # p from 0 to 1 maps to screen vertical from -1.5 to 1.5
            pos_h = 3.0 + (val_z_in * 0.5)
            pos_v = -1.5 + (val_p_out * 3.0)
            pts_data.append([pos_h, pos_v, 0])
        
        curve_obj = VMobject(color=BLUE)
        curve_obj.set_points_as_corners(pts_data)
        self.add(curve_obj)

        # Labels for axes
        label_prob = MathTex(r"P(y=1)").move_to([1.0, 1.8, 0])
        label_z_axis = MathTex(r"z").move_to([5.2, -1.5, 0])
        self.add(label_prob, label_z_axis)

        # 2. Dot on curve
        # Initial position z=0, p=0.5
        moving_dot = Dot(color=YELLOW).move_to([3.0, 0, 0])
        self.add(moving_dot)

        # 3. UI Elements - Left Side (Bars)
        # Instead of ValueTracker/always_redraw, we use explicit animations
        base_line_v = -2.5
        pos_bar1_h = -4.5
        pos_bar2_h = -2.5
        
        # Initial bars (nearly zero height)
        rect_1d = Rectangle(width=0.8, height=0.1, fill_opacity=0.8, color=RED)
        rect_1d.move_to([pos_bar1_h, base_line_v + 0.05, 0])
        
        rect_30d = Rectangle(width=0.8, height=0.1, fill_opacity=0.8, color=GREEN)
        rect_30d.move_to([pos_bar2_h, base_line_v + 0.05, 0])
        
        text_1d = Text("1-Day T-CSP", font_size=20).next_to(rect_1d, DOWN)
        text_30d = Text("30-Day T-CSP", font_size=20).next_to(rect_30d, DOWN)
        self.add(rect_1d, rect_30d, text_1d, text_30d)

        # 4. Formula Display
        main_eqn = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=24
        ).move_to([-3.5, 2.0, 0])
        self.add(main_eqn)

        # 5. Animation sequence
        # Move 1: Increase inputs
        new_z1 = 2.0
        new_p1 = get_sig(new_z1)
        h1 = 2.5
        h2 = 1.5
        self.play(
            rect_1d.animate.stretch_to_fit_height(h1).move_to([pos_bar1_h, base_line_v + h1/2, 0]),
            rect_30d.animate.stretch_to_fit_height(h2).move_to([pos_bar2_h, base_line_v + h2/2, 0]),
            moving_dot.animate.move_to([3.0 + (new_z1 * 0.5), -1.5 + (new_p1 * 3.0), 0]),
            run_time=2
        )
        self.wait(1)

        # Transform formula to show parameter values
        param_eqn = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(0.5 + 0.8 X_{1d} + 0.4 X_{30d})}}",
            font_size=24
        ).move_to([-3.5, 2.0, 0])
        self.play(Transform(main_eqn, param_eqn))
        self.wait(0.5)

        # Move 2: Decrease inputs
        new_z2 = -2.5
        new_p2 = get_sig(new_z2)
        h3 = 0.4
        h4 = 0.6
        self.play(
            rect_1d.animate.stretch_to_fit_height(h3).move_to([pos_bar1_h, base_line_v + h3/2, 0]),
            rect_30d.animate.stretch_to_fit_height(h4).move_to([pos_bar2_h, base_line_v + h4/2, 0]),
            moving_dot.animate.move_to([3.0 + (new_z2 * 0.5), -1.5 + (new_p2 * 3.0), 0]),
            run_time=2
        )
        self.wait(2)
