from manim import *

class ElbowThreshold(Scene):
    def construct(self):
        # 1. Coordinate System
        axes = Axes(
            x_range=[0, 1.05, 0.2],
            y_range=[0, 1.05, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_numbers": True}
        ).to_edge(LEFT, buff=1.0)
        
        x_label = Tex("Fraction of Zones Flagged").scale(0.6).next_to(axes.x_axis, DOWN)
        y_label = Tex("Recall").scale(0.6).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)
        
        # 2. Formula
        formula = MathTex(r"\text{Recall} = \frac{TP}{TP + FN}").scale(0.8).to_edge(UP)
        
        # 3. Curve (Concave curve: y = 1 - (1-x)^2)
        def curve_func(x):
            return 1 - (1 - x)**2

        curve = axes.plot(curve_func, x_range=[0, 1], color=BLUE)
        
        # 4. Value Tracker and Moving Point
        # Using a ValueTracker to control the threshold position
        tracker = ValueTracker(0.1)
        
        moving_dot = Dot(color=YELLOW)
        
        def update_dot(mob):
            val = tracker.get_value()
            mob.move_to(axes.c2p(val, curve_func(val)))
            
        moving_dot.add_updater(update_dot)
        
        # 5. Side Panel Floorplan (Zones)
        side_panel = Rectangle(width=4, height=5, color=GREY).shift(RIGHT * 4)
        panel_title = Text("Floorplan Zones", font_size=24).next_to(side_panel, UP)
        
        # Grid of squares representing zones
        zones = VGroup(*[
            Square(side_length=0.5, fill_opacity=0.1, color=WHITE)
            for _ in range(12)
        ]).arrange_in_grid(rows=4, cols=3, buff=0.1).move_to(side_panel.get_center())
        
        def update_zones(group):
            val = tracker.get_value()
            # Highlight zones based on the current threshold
            threshold_count = int(val * 12)
            for i, zone in enumerate(group):
                if i < threshold_count:
                    zone.set_fill(RED, opacity=0.8)
                else:
                    zone.set_fill(WHITE, opacity=0.1)
                    
        zones.add_updater(update_zones)
        
        # 6. Static Elements for the "Elbow"
        elbow_x = 0.35
        elbow_point = axes.c2p(elbow_x, curve_func(elbow_x))
        elbow_highlight = Circle(radius=0.4, color=YELLOW).move_to(elbow_point)
        elbow_label = Text("The 'Elbow'", color=YELLOW, font_size=20).next_to(elbow_highlight, UR)
        
        # 7. Animation Sequence
        self.add(formula, axes, x_label, y_label, curve, moving_dot, side_panel, panel_title, zones)
        
        # Animate threshold movement
        # Move to the left (few zones flagged)
        self.play(tracker.animate.set_value(0.05), run_time=1)
        self.wait(0.5)
        
        # Move to the right (many zones flagged)
        self.play(tracker.animate.set_value(0.95), run_time=3)
        self.wait(0.5)
        
        # Return to the "Elbow"
        self.play(tracker.animate.set_value(elbow_x), run_time=1.5)
        self.play(Create(elbow_highlight), Write(elbow_label))
        self.play(Indicate(elbow_highlight))
        
        self.wait(2)
