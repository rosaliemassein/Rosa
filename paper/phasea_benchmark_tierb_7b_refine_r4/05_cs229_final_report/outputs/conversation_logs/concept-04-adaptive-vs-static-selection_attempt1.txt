import numpy as np
from manim import *

class AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. Split-screen Layout
        # Using Rectangles as floorplan placeholders to ensure compilation
        left_bg = Rectangle(width=5.5, height=3.5, color=GRAY_B).shift(LEFT * 3.5 + UP * 1)
        right_bg = Rectangle(width=5.5, height=3.5, color=GRAY_B).shift(RIGHT * 3.5 + UP * 1)
        
        left_title = Text("Static Benchmark", font_size=28, color=WHITE).next_to(left_bg, UP)
        right_title = Text("Logistic Model (ML)", font_size=28, color=WHITE).next_to(right_bg, UP)
        
        left_fp_text = Text("Building Floorplan", font_size=16, color=GRAY_A).move_to(left_bg)
        right_fp_text = Text("Building Floorplan", font_size=16, color=GRAY_A).move_to(right_bg)
        
        self.add(left_bg, right_bg, left_title, right_title, left_fp_text, right_fp_text)

        # 2. Static Highlighted Zones (Left)
        # These represent traditional ASHRAE "rules of thumb" that never change
        static_zones = VGroup(
            Circle(radius=0.2, color=YELLOW, fill_opacity=0.8).move_to(left_bg.get_center() + LEFT*1.8 + UP*0.6),
            Circle(radius=0.2, color=YELLOW, fill_opacity=0.8).move_to(left_bg.get_center() + RIGHT*1.2 + DOWN*0.8),
            Circle(radius=0.2, color=YELLOW, fill_opacity=0.8).move_to(left_bg.get_center() + LEFT*0.4 + DOWN*1.0)
        )
        self.add(static_zones)

        # 3. Timeline Setup (Manual construction)
        timeline_y = -1.5
        timeline_line = Line(LEFT * 5.5 + UP * timeline_y, RIGHT * 5.5 + UP * timeline_y, color=WHITE)
        months = ["May", "Jun", "Jul", "Aug", "Sep"]
        timeline_markers = VGroup()
        for i, m in enumerate(months):
            pos = timeline_line.point_from_proportion(i/4)
            tick = Line(pos + UP*0.1, pos + DOWN*0.1, color=WHITE)
            label = Text(m, font_size=20).next_to(pos, DOWN)
            timeline_markers.add(tick, label)
        
        self.add(timeline_line, timeline_markers)

        # 4. Graph Construction (Manual Axes)
        graph_origin = LEFT * 5.5 + DOWN * 3.5
        xaxis = Line(graph_origin, graph_origin + RIGHT * 11, color=WHITE)
        yaxis = Line(graph_origin, graph_origin + UP * 1.5, color=WHITE)
        ylabel = Text("Missed Requests", font_size=16).next_to(yaxis, LEFT, buff=0.1).rotate(90 * DEGREES)
        self.add(xaxis, yaxis, ylabel)

        # 5. Dynamic Elements
        # Using a hidden Dot as a time tracker (replaces ValueTracker)
        tracker = Dot(radius=0).move_to(LEFT * 5.5).set_opacity(0)
        self.add(tracker)
        
        ml_zones = VGroup() # Adaptive zones on the right
        static_graph_line = VMobject(color=YELLOW) # High error line
        ml_graph_line = VMobject(color=RED) # Low error line
        cursor = Triangle(fill_opacity=1, color=WHITE).scale(0.12).rotate(180 * DEGREES)
        
        self.add(ml_zones, static_graph_line, ml_graph_line, cursor)

        # Updater function to simulate dynamic behavior over time
        def update_scene(mob):
            # Alpha is progress from 0 to 1 based on tracker x-position
            alpha = (tracker.get_center()[0] + 5.5) / 11
            alpha = max(0, min(1, alpha))
            
            # Update Timeline Cursor
            cursor.move_to(timeline_line.point_from_proportion(alpha) + UP * 0.25)
            
            # Update Graph Curves
            s_points = []
            m_points = []
            resolution = 40
            for step in range(int(resolution * alpha) + 1):
                p_alpha = step / resolution
                # Static error curve (Yellow): Higher, fluctuating
                s_y_val = 1.0 + 0.15 * np.sin(p_alpha * 12)
                # ML error curve (Red): Significantly lower, optimized
                m_y_val = 0.4 - 0.2 * p_alpha + 0.1 * np.cos(p_alpha * 18)
                
                s_points.append(graph_origin + RIGHT * (p_alpha * 11) + UP * s_y_val)
                m_points.append(graph_origin + RIGHT * (p_alpha * 11) + UP * m_y_val)
            
            if len(s_points) > 1:
                static_graph_line.set_points_as_corners(s_points)
                ml_graph_line.set_points_as_corners(m_points)
            
            # Update Adaptive Zones (Flickering/Changing on the right)
            # Use alpha-based seed to make flicker consistent but dynamic
            seed = int(alpha * 100)
            rng = np.random.default_rng(seed)
            # Seasonality logic: flag more zones during heatwaves (middle of timeline)
            zone_count = int(2 + 5 * np.sin(alpha * 3.14159))
            
            # Clear and rebuild ML zones
            ml_zones.submobjects = []
            for _ in range(max(1, zone_count)):
                random_pos = right_bg.get_center() + np.array([
                    rng.uniform(-2.2, 2.2),
                    rng.uniform(-1.2, 1.2),
                    0
                ])
                ml_zones.add(Circle(radius=0.2, color=RED, fill_opacity=0.8).move_to(random_pos))

        tracker.add_updater(update_scene)

        # 6. Animation Progression
        # Move the tracker across the screen to trigger the updaters
        self.play(
            tracker.animate.move_to(RIGHT * 5.5),
            run_time=12,
            rate_func=lambda t: t # Linear progression
        )
        
        # Hold the final state
        self.wait(2)
