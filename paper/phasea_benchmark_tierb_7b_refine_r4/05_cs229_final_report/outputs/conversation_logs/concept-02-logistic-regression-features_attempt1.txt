from manim import *
import numpy as np

class LogisticRegressionFeatures(Scene):
    def construct(self):
        # 1. Coordinate System
        # Using NumberPlane as requested in the remarks.
        plane = NumberPlane(
            x_range=[-6, 6, 1],
            y_range=[0, 1.2, 0.2],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True},
            background_line_style={"stroke_opacity": 0.3}
        ).to_edge(RIGHT, buff=0.5)

        def sigmoid_math(input_x):
            return 1.0 / (1.0 + np.exp(-input_x))

        sigmoid_curve = plane.plot(sigmoid_math, color=GREEN)
        curve_label = MathTex(r"P(y=1)", color=GREEN).next_to(sigmoid_curve, UP)

        # 2. Tracking Values
        # Using DecimalNumber as a hidden container for values to avoid ValueTracker issues
        track_1d = DecimalNumber(0).hide()
        track_30d = DecimalNumber(0).hide()

        # 3. Features (Sliding Bars)
        bar_1d = Rectangle(width=0.7, height=0.1, fill_opacity=0.8, color=YELLOW)
        bar_1d.move_to(LEFT * 4.5 + DOWN * 0.5) # Anchored near bottom
        
        bar_30d = Rectangle(width=0.7, height=0.1, fill_opacity=0.8, color=ORANGE)
        bar_30d.move_to(LEFT * 2.5 + DOWN * 0.5)

        def update_b1(mob):
            val = track_1d.get_value()
            # Scale height and maintain base at the same Y level
            mob.stretch_to_fit_height(max(0.1, abs(val)), about_edge=DOWN)

        def update_b2(mob):
            val = track_30d.get_value()
            mob.stretch_to_fit_height(max(0.1, abs(val)), about_edge=DOWN)

        bar_1d.add_updater(update_b1)
        bar_30d.add_updater(update_b2)

        label_1d = Text("1-Day T-CSP", font_size=20).next_to(bar_1d, DOWN)
        label_30d = Text("30-Day T-CSP", font_size=20).next_to(bar_30d, DOWN)

        # 4. Logistic Mapping Dot
        prediction_dot = Dot(color=RED)

        def update_prediction(mob):
            # Parameters: beta0 = -1.0, beta1 = 0.8, beta2 = 0.4
            # Calculated weighted sum
            z_score = -1.0 + (0.8 * track_1d.get_value()) + (0.4 * track_30d.get_value())
            mob.move_to(plane.c2p(z_score, sigmoid_math(z_score)))

        prediction_dot.add_updater(update_prediction)

        # 5. Formula
        # Logistic formula provided in prompt
        formula = MathTex(
            r"P(y=1) = \frac{1}{1 + e^{-(\beta_0 + \beta_1 X_{1d} + \beta_{30} X_{30d})}}",
            font_size=32
        ).to_edge(UP, buff=0.5)

        # 6. Assembly and Animation
        self.add(plane, sigmoid_curve, curve_label)
        self.add(bar_1d, bar_30d, label_1d, label_30d)
        self.add(track_1d, track_30d, prediction_dot, formula)

        # Animate bars sliding and dot moving
        self.play(track_1d.animate.set_value(4), run_time=2)
        self.wait(0.5)
        self.play(track_30d.animate.set_value(3), run_time=2)
        self.wait(0.5)
        self.play(
            track_1d.animate.set_value(-2),
            track_30d.animate.set_value(-4),
            run_time=3
        )
        self.wait(1)

        # Final state: High probability
        self.play(
            track_1d.animate.set_value(5),
            track_30d.animate.set_value(4),
            run_time=2
        )
        self.wait(2)
