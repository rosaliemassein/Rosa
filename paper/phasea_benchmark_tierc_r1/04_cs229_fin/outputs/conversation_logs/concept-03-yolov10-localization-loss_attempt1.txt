from manim import *
import numpy as np

class YOLOCoordinateLoss(Scene):
    def construct(self):
        # 1. Background Grid & Image Logic
        # Fallback for image if file not found
        try:
            image = ImageMobject("img-11.jpeg")
            image.scale_to_fit_height(6)
            self.add(image)
        except:
            bg_rect = Rectangle(width=10, height=7, fill_opacity=0.2, color=GREY)
            self.add(bg_rect)

        # 7x7 Grid
        grid = NumberPlane(
            x_range=[-3.5, 3.5, 1],
            y_range=[-3.5, 3.5, 1],
            x_length=7,
            y_length=7,
            background_line_style={"stroke_color": WHITE, "stroke_width": 2, "stroke_opacity": 0.4}
        )
        self.add(grid)

        # 2. Highlight Specific Cell
        # Using index 1,1 (center at grid.c2p(1,1))
        cell_center_point = grid.c2p(1, 1)
        highlight_cell = Square(side_length=1, color=YELLOW, fill_opacity=0.3)
        highlight_cell.move_to(cell_center_point)
        self.play(Create(highlight_cell))

        # 3. Bounding Boxes
        # Ground Truth (Dashed - constructed from DashedLines for safety)
        gt_center = grid.c2p(1.3, 1.2)
        gt_w, gt_h = 2.4, 1.6
        gt_rect = Rectangle(width=gt_w, height=gt_h).move_to(gt_center)
        gt_box = VGroup(*[
            DashedLine(gt_rect.get_vertices()[i], gt_rect.get_vertices()[(i+1)%4], color=GREEN)
            for i in range(4)
        ])
        
        # Prediction
        pred_center = grid.c2p(0.7, 0.8)
        pred_w, pred_h = 1.8, 1.0
        pred_box = Rectangle(width=pred_w, height=pred_h, color=RED).move_to(pred_center)

        self.play(Create(gt_box), Create(pred_box))

        # 4. Arrows showing differences
        # Center difference (x, y)
        arrow_center = Arrow(pred_box.get_center(), gt_center, buff=0, color=YELLOW)
        
        # Dimension difference (w, h)
        arrow_w = Arrow(
            pred_box.get_right(), 
            pred_box.get_right() + RIGHT * (gt_w - pred_w) / 2, 
            buff=0, color=BLUE
        )
        arrow_h = Arrow(
            pred_box.get_top(), 
            pred_box.get_top() + UP * (gt_h - pred_h) / 2, 
            buff=0, color=BLUE
        )

        self.play(Create(arrow_center))
        self.play(Create(arrow_w), Create(arrow_h))
        self.wait(1)

        # 5. LaTeX formula
        # Slicing the LaTeX string carefully to allow coloring
        formula = MathTex(
            r"\lambda_{coord}\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((x_{i}-\hat{x}_{i})^{2}+(y_{i}-\hat{y}_{i})^{2}\right)",
            r"+",
            r"\sum_{i=0}^{S^{2}}\sum_{j=0}^{B}1_{ij}^{obj}\left((\sqrt{w_{i}}-\sqrt{\hat{w}_{i}})^{2}+(\sqrt{h_{i}}-\sqrt{\hat{h}_{i}})^{2}\right)",
            font_size=30
        )
        formula.to_edge(DOWN)
        # Add background rectangle for readability
        formula_bg = BackgroundRectangle(formula, color=BLACK, fill_opacity=0.8, buff=0.1)
        self.play(FadeIn(formula_bg), Write(formula))

        # 6. Mock Training Animation
        # Align prediction to ground truth and highlight relevant terms
        self.play(
            pred_box.animate.move_to(gt_center).stretch_to_fit_width(gt_w).stretch_to_fit_height(gt_h),
            formula[0].animate.set_color(YELLOW),
            formula[2].animate.set_color(BLUE),
            FadeOut(arrow_center),
            FadeOut(arrow_w),
            FadeOut(arrow_h),
            run_time=3
        )

        self.wait(2)
