import numpy as np

class YOLOv10LocalizationLoss(Scene):

        image_group = VGroup(image_rect, image_label)
        self.play(FadeIn(image_group))

        # Overlay a 7x7 grid
        grid = NumberPlane(
            x_range=[0, 7, 1],
            y_range=[0, 7, 1],
            x_length=6,
            y_length=6,
            background_line_style={
                "stroke_color": WHITE,
                "stroke_width": 2,
                "stroke_opacity": 0.4

        self.play(Create(grid))

        cell_hl.set_fill(YELLOW, opacity=0.3)
        cell_hl.set_stroke(YELLOW, width=2)
        self.play(DrawBorderThenFill(cell_hl))

        # 3. Predicted Box and Ground Truth Box

        pred_center = grid.c2p(3.8, 3.2)

            pass
        # Using a simple yellow rectangle with lower opacity as a fallback for "dashed" visual
        gt_box = Rectangle(width=gt_w, height=gt_h, color=YELLOW).move_to(gt_center).set_stroke(opacity=0.5)

        self.play(Create(pred_box))
        self.play(Create(gt_box))

        # 4. Coordinate Error Arrows

            [gt_center[0], gt_center[1], 0],
