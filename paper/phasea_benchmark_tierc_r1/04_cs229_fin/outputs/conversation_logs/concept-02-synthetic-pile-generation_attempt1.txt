from manim import *
import numpy as np
import random

class Concept02SyntheticPileGeneration(Scene):
    def construct(self):
        # 1. Setup Canvas (Blank white-bordered rectangle)
        canvas = Rectangle(width=7, height=5, color=WHITE, stroke_width=2)
        canvas_label = Text("Synthetic Canvas", font_size=24).next_to(canvas, UP)
        self.add(canvas, canvas_label)

        # 2. Value Trackers for Scale and Rotation to visualize randomization
        scale_val = ValueTracker(1.0)
        rot_val = ValueTracker(0)

        scale_text = Text("Scale: ", font_size=24)
        scale_num = DecimalNumber(1.0, num_decimal_places=2, font_size=24)
        scale_label = VGroup(scale_text, scale_num).arrange(RIGHT).to_corner(UL, buff=0.5)
        
        rot_text = Text("Rotation: ", font_size=24)
        rot_num = DecimalNumber(0, num_decimal_places=0, font_size=24)
        rot_unit = Text("°", font_size=24)
        rot_label = VGroup(rot_text, rot_num, rot_unit).arrange(RIGHT).next_to(scale_label, DOWN, aligned_edge=LEFT)

        # Fix: Using 'm' as the lambda argument to avoid conflicts
        scale_num.add_updater(lambda m: m.set_value(scale_val.get_value()))
        rot_num.add_updater(lambda m: m.set_value(rot_val.get_value()))

        self.add(scale_label, rot_label)

        # 3. Create 9 "Trash" Icons (Segmented Objects)
        # Using standard colors and shapes to ensure compatibility
        colors = [RED, BLUE, GREEN, YELLOW, ORANGE, PURPLE, MAROON, GOLD, GREY]
        
        trash_items = VGroup()
        for i in range(9):
            # Procedural shapes to ensure no "undefined" errors for specific classes
            if i % 3 == 0:
                shape = Circle(radius=0.35)
            elif i % 3 == 1:
                shape = Square(side_length=0.6)
            else:
                shape = Triangle().scale(0.4)
                
            shape.set_fill(colors[i], opacity=0.9)
            shape.set_stroke(WHITE, width=1)
            
            # Position icons initially in a circle around the canvas
            angle = TAU * i / 9
            start_pos = 5 * np.array([np.cos(angle), np.sin(angle), 0])
            shape.move_to(start_pos)
            trash_items.add(shape)

        self.add(trash_items)

        # 4. Animate icons moving onto the canvas
        for i, item in enumerate(trash_items):
            target_scale = random.uniform(0.7, 1.5)
            target_rot = random.uniform(0, 360)
            # Randomized target within the canvas bounds
            target_pos = np.array([
                random.uniform(-2.8, 2.8),
                random.uniform(-1.8, 1.8),
                0
            ])

            self.play(
                scale_val.animate.set_value(target_scale),
                rot_val.animate.set_value(target_rot),
                item.animate.move_to(target_pos).scale(target_scale).rotate(target_rot * DEGREES),
                run_time=0.6
            )
            # Higher Z-index for later items to show layering
            item.set_z_index(i + 1)

        self.wait(2)
