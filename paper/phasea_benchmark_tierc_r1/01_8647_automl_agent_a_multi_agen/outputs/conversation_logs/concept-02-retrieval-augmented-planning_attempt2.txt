from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Formula at the top
        formula = MathTex(r"\mathbf{P} = \mathcal{A}_{mgr}(\mathrm{RAP}(R))").to_edge(UP)

        # 2. Agent Manager (Central)
        agent_mgr_circle = Circle(radius=1, color=BLUE, fill_opacity=0.3)
        agent_mgr_label = Text("Agent\nManager", font_size=24)
        agent_manager = VGroup(agent_mgr_circle, agent_mgr_label).move_to(ORIGIN)

        # 3. Knowledge Base (Cloud icon on the left)
        kb_blobs = VGroup(
            Circle(radius=0.5).shift(0.3 * LEFT),
            Circle(radius=0.5).shift(0.3 * RIGHT),
            Circle(radius=0.5).shift(0.3 * UP),
            Circle(radius=0.3).shift(0.2 * LEFT + 0.2 * UP),
        ).set_fill(GREY, opacity=0.5).set_stroke(WHITE, 2)
        kb_text = Text("Knowledge\nBase", font_size=20)
        knowledge_base = VGroup(kb_blobs, kb_text).shift(LEFT * 4.5)

        # 4. Plan Icons (Right side)
        # Plan A: CNN (Grid)
        plan_a_icon = VGroup(*[Square(side_length=0.2) for _ in range(4)]).arrange_in_grid(rows=2, cols=2, buff=0.1)
        plan_a = VGroup(plan_a_icon, Text("Plan A: CNN", font_size=18).next_to(plan_a_icon, DOWN)).set_color(YELLOW)
        
        # Plan B: Transformer (Attention/Nodes)
        plan_b_icon = VGroup(Dot(), Dot(), Dot()).arrange(RIGHT, buff=0.3)
        plan_b_lines = VGroup(Line(plan_b_icon[0].get_center(), plan_b_icon[1].get_center()), 
                              Line(plan_b_icon[1].get_center(), plan_b_icon[2].get_center()), 
                              Line(plan_b_icon[0].get_center(), plan_b_icon[2].get_center()))
        plan_b_icon.add(plan_b_lines)
        plan_b = VGroup(plan_b_icon, Text("Plan B: Transformer", font_size=18).next_to(plan_b_icon, DOWN)).set_color(GREEN)
        
        # Plan C: Other Strategy (Diamond)
        plan_c_icon = RegularPolygon(n=4).rotate(45 * DEGREES).scale(0.3)
        plan_c = VGroup(plan_c_icon, Text("Plan C: RNN", font_size=18).next_to(plan_c_icon, DOWN)).set_color(RED)

        plans = VGroup(plan_a, plan_b, plan_c).arrange(DOWN, buff=1).shift(RIGHT * 4.5)

        # Animations
        self.play(Write(formula))
        self.play(FadeIn(agent_manager), FadeIn(knowledge_base))
        self.wait(0.5)

        # Animate arrow pulling Knowledge into the Manager
        arrow_kb_to_mgr = Arrow(knowledge_base.get_right(), agent_manager.get_left(), color=WHITE)
        self.play(GrowArrow(arrow_kb_to_mgr))
        
        # Knowledge transfer particles (Moving dots instead of MoveAlongPath for safety)
        particles = VGroup(*[Dot(radius=0.08, color=WHITE) for _ in range(3)])
        for p in particles:
            p.move_to(knowledge_base.get_right())

        self.play(
            particles[0].animate.move_to(agent_manager.get_left()),
            rate_func=linear
        )
        self.play(
            particles[1].animate.move_to(agent_manager.get_left()),
            rate_func=linear
        )
        self.play(
            particles[2].animate.move_to(agent_manager.get_left()),
            rate_func=linear
        )
        
        self.play(FadeOut(particles), agent_mgr_circle.animate.set_fill(BLUE, opacity=0.7))
        self.wait(0.5)

        # Branching out into three paths
        branch_arrows = VGroup(*[
            Arrow(agent_manager.get_right(), p.get_left(), color=WHITE, buff=0.2) 
            for p in plans
        ])

        # Create branches and plans
        self.play(
            AnimationGroup(
                Create(branch_arrows),
                TransformFromCopy(agent_manager, plan_a),
                TransformFromCopy(agent_manager, plan_b),
                TransformFromCopy(agent_manager, plan_c),
                lag_ratio=0.2
            ),
            run_time=2
        )

        # Final explanation text (using string for slant to avoid undefined identifier)
        explanation = Text(
            "External data informs parallel strategy exploration", 
            font_size=20, 
            slant="ITALIC"
        ).to_edge(DOWN)
        self.play(Write(explanation))
        
        self.wait(2)
