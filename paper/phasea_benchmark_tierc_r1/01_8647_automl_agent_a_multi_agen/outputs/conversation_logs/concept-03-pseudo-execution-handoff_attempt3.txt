from manim import *

class Concept03PseudoExecutionHandoff(Scene):
    def construct(self):
        # 1. Setup Titles and Background
        title = Text("Pseudo-Execution Information Handoff", font_size=36).to_edge(UP)
        
        # 2. Agents Layout (Split Screen Concept)
        # Data Agent at the top
        data_agent_label = Text("Data Agent (A_d)", color=BLUE, font_size=32)
        data_box = RoundedRectangle(height=1.8, width=4.5, color=BLUE)
        data_agent_group = VGroup(data_box, data_agent_label).shift(UP * 2 + LEFT * 2)
        
        # Model Agent at the bottom
        model_agent_label = Text("Model Agent (A_m)", color=GREEN, font_size=32)
        model_box = RoundedRectangle(height=1.8, width=4.5, color=GREEN)
        model_agent_group = VGroup(model_box, model_agent_label).shift(DOWN * 1.5 + LEFT * 2)
        
        # Formula at the very bottom
        formula = MathTex(
            r"s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)", 
            font_size=34,
            color=WHITE
        ).to_edge(DOWN, buff=0.5)

        # 3. Search Efficiency Bar (Right Side)
        efficiency_tracker = ValueTracker(5)
        efficiency_title = Text("Search Efficiency", font_size=24).shift(RIGHT * 4 + UP * 1.5)
        
        bar_bg = Rectangle(width=0.8, height=4, color=GREY_D, fill_opacity=0.5).next_to(efficiency_title, DOWN)
        bar_fill = always_redraw(lambda: Rectangle(
            width=0.8,
            height=efficiency_tracker.get_value() / 100 * 4,
            color=YELLOW,
            fill_opacity=0.8,
            stroke_width=0
        ).align_to(bar_bg, DOWN))
        
        efficiency_percent = always_redraw(lambda: DecimalNumber(
            efficiency_tracker.get_value(),
            unit="%",
            num_decimal_places=0,
            font_size=28
        ).next_to(bar_bg, UP, buff=0.2))
        
        efficiency_ui = VGroup(efficiency_title, bar_bg, bar_fill, efficiency_percent)

        # 4. Animation Sequence
        self.add(title, data_agent_group, model_agent_group, efficiency_ui, formula)
        self.wait(1)

        # Step 1: Data Agent generates Insights (O_d)
        insight_label = Text("O_d: Data Insights", font_size=22, color=YELLOW)
        insight_label.move_to(data_agent_group.get_center())
        
        self.play(FadeIn(insight_label, shift=DOWN))
        self.play(Indicate(insight_label))
        
        # Step 2: Handoff to Model Agent
        handoff_arrow = Arrow(
            data_agent_group.get_bottom(), 
            model_agent_group.get_top(), 
            buff=0.1, 
            color=WHITE
        )
        
        self.play(
            Create(handoff_arrow),
            insight_label.animate.next_to(handoff_arrow, RIGHT),
            run_time=1.5
        )
        
        # Step 3: Model Agent receives input and expands sub-tasks
        # We simulate sub-task expansion by showing s_m growing
        sub_task_rect = Rectangle(width=1.2, height=0.7, color=WHITE, fill_opacity=0.2).move_to(model_agent_group)
        sub_task_text = MathTex("s_m", font_size=24).move_to(sub_task_rect)
        sub_task_group = VGroup(sub_task_rect, sub_task_text)
        
        self.play(
            FadeOut(insight_label),
            FadeOut(handoff_arrow),
            FadeIn(sub_task_group)
        )
        
        # Step 4: Pseudo-execution search (training-free)
        # Efficiency bar increases as agents "reason" without GPU training
        self.play(
            sub_task_group.animate.scale(2.2).shift(RIGHT * 0.5),
            efficiency_tracker.animate.set_value(92),
            formula.animate.set_color(YELLOW),
            run_time=3,
            rate_func=rate_functions.smooth
        )
        
        # Highlight the specific term in the formula
        self.play(Circumscribe(formula[0][14:])) # Target O_i^d
        
        self.wait(2)
