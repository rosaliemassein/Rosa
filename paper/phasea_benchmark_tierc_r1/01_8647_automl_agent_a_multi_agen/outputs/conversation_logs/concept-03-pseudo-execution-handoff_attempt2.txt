from manim import *

class Concept03PseudoExecutionHandoff(Scene):
    def construct(self):
        # 1. Formula Header
        formula = MathTex(r"s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)").to_edge(UP, buff=0.5)
        self.play(Write(formula))

        # Split screen indicator (visual line)
        divider = Line(LEFT * 7, RIGHT * 7, color=GRAY, stroke_opacity=0.3)
        self.add(divider)

        # 2. Data Agent Setup (Top Half)
        data_agent_label = MathTex(r"\text{Data Agent } (\mathcal{A}_d)", color=GREEN).scale(0.8).shift(UP * 2.5 + LEFT * 4)
        data_block = VGroup(
            RoundedRectangle(height=1.2, width=2.2, corner_radius=0.2, color=BLUE, fill_opacity=0.2),
            Text("Raw Data", font_size=22)
        ).shift(UP * 1.5)

        self.play(FadeIn(data_agent_label), Create(data_block))

        # 3. Pseudo-analysis Animation
        pseudo_label = Text("Pseudo-analysis...", font_size=20, color=YELLOW).next_to(data_block, UP, buff=0.2)
        insight_od = MathTex("O_d", color=YELLOW).scale(1.2).move_to(data_block)
        
        self.play(Write(pseudo_label))
        self.play(
            data_block[0].animate.set_color(YELLOW),
            ReplacementTransform(data_block[1], insight_od),
            run_time=1.5
        )
        self.wait(0.5)

        # 4. Model Agent Setup (Bottom Half)
        model_agent_label = MathTex(r"\text{Model Agent } (\mathcal{A}_m)", color=RED).scale(0.8).shift(DOWN * 0.5 + LEFT * 4)
        # Task block (sm) starts small
        task_block_sm = Rectangle(height=0.8, width=0.5, color=ORANGE, fill_opacity=0.4).shift(DOWN * 1.5)
        sm_label = MathTex("s_m").next_to(task_block_sm, UP, buff=0.1)

        self.play(FadeIn(model_agent_label), Create(task_block_sm), Write(sm_label))

        # 5. Information Hand-off
        handoff_arrow = Arrow(UP * 0.8, DOWN * 1.0, color=WHITE, buff=0)
        self.play(GrowArrow(handoff_arrow), FadeOut(pseudo_label))
        
        # Move Od (Insights) down to the model agent section
        self.play(
            insight_od.animate.scale(0.6).next_to(task_block_sm, LEFT, buff=0.3),
            run_time=1.5
        )

        # 6. Search Efficiency UI
        eff_label = Text("Search Efficiency", font_size=20).to_edge(RIGHT, buff=1).shift(UP * 0.5)
        eff_bg = Rectangle(width=3, height=0.3, color=WHITE).next_to(eff_label, DOWN, buff=0.2)
        
        # Use ValueTracker for the efficiency bar animation
        eff_tracker = ValueTracker(0.01)
        eff_fill = always_redraw(lambda: Rectangle(
            width=max(0.01, eff_tracker.get_value() * 3),
            height=0.3,
            fill_color=GREEN,
            fill_opacity=0.8,
            stroke_width=0
        ).align_to(eff_bg, LEFT))

        self.play(FadeIn(eff_label), Create(eff_bg), Create(eff_fill))

        # 7. Final Sequence: Expansion and Search Efficiency Increase
        self.play(
            task_block_sm.animate.stretch_to_fit_width(4, about_edge=LEFT),
            eff_tracker.animate.set_value(1.0),
            insight_od.animate.shift(RIGHT * 0.2),
            sm_label.animate.shift(RIGHT * 1.5),
            run_time=3,
            rate_func=smooth
        )

        # Add a "Training-Free" tag
        tag = Text("Training-Free Search", font_size=24, color=GREEN).next_to(eff_bg, DOWN, buff=0.5)
        self.play(Write(tag))

        self.wait(2)
