from manim import *

class PseudoExecutionAnimation(Scene):
    def construct(self):
        # 1. Setup Layout
        divider = Line(LEFT * 7, RIGHT * 7, color=GRAY)
        self.play(Create(divider))
        
        data_agent_label = Text("Data Agent (A_d)", font_size=24).to_edge(UP).shift(LEFT * 4)
        model_agent_label = Text("Model Agent (A_m)", font_size=24).to_edge(DOWN).shift(LEFT * 4)
        self.play(Write(data_agent_label), Write(model_agent_label))

        # 2. Data Agent Pseudo-Analysis
        data_block = Rectangle(width=2, height=1.5, color=BLUE, fill_opacity=0.2).shift(UP * 1.5 + LEFT * 2)
        data_label = Text("Data Block", font_size=20).move_to(data_block)
        
        self.play(Create(data_block), Write(data_label))
        self.wait(0.5)
        
        # Pseudo-analysis transition
        insight_rect = MathTex(r"O_d", color=YELLOW).scale(1.2).shift(UP * 1.5 + RIGHT * 2)
        insight_label = Text("Data Insights", font_size=18).next_to(insight_rect, UP)
        
        analysis_arrow = Arrow(data_block.get_right(), insight_rect.get_left(), color=WHITE)
        
        self.play(GrowArrow(analysis_arrow))
        self.play(FadeIn(insight_rect), Write(insight_label))
        self.wait(1)

        # 3. Handoff to Model Agent
        handoff_arrow = Arrow(insight_rect.get_bottom(), DOWN * 0.5 + RIGHT * 2, color=YELLOW)
        
        model_task_block = Rectangle(width=3, height=1, color=GREEN, fill_opacity=0.2).shift(DOWN * 1.5 + RIGHT * 2)
        model_task_label = MathTex(r"s_i^m", font_size=30).move_to(model_task_block)
        
        self.play(
            insight_rect.animate.move_to(model_task_block.get_top() + UP * 0.5),
            GrowArrow(handoff_arrow),
            run_time=1.5
        )
        self.play(Create(model_task_block), Write(model_task_label))

        # 4. Formula Display
        formula = MathTex(r"s_i^m = \mathrm{PD}(R, \mathcal{A}_m, \mathbf{p}_i, O_i^d)", font_size=36)
        formula.shift(DOWN * 1.5 + LEFT * 2)
        self.play(Write(formula))
        self.wait(1)

        # 5. Search Efficiency Bar
        efficiency_tracker = ValueTracker(0)
        
        efficiency_label = Text("Search Efficiency", font_size=20).to_edge(RIGHT).shift(UP * 0.5)
        
        # Draw bar container
        bar_bg = Rectangle(width=4, height=0.4, color=WHITE).next_to(efficiency_label, DOWN)
        bar_fill = always_redraw(lambda: Rectangle(
            width=max(0.001, efficiency_tracker.get_value() * 4),
            height=0.4,
            fill_color=YELLOW,
            fill_opacity=0.8,
            stroke_width=0
        ).align_to(bar_bg, LEFT))
        
        percentage_label = always_redraw(lambda: Text(
            f"{int(efficiency_tracker.get_value() * 100)}%",
            font_size=24
        ).next_to(bar_bg, RIGHT))

        self.play(Create(bar_bg), FadeIn(efficiency_label))
        self.add(bar_fill, percentage_label)
        
        # Animate progress
        self.play(
            efficiency_tracker.animate.set_value(0.95),
            model_task_block.animate.scale(1.2).set_color(GOLD),
            run_time=4,
            rate_func=smooth
        )
        
        final_text = Text("Training-Free Search Complete", color=GOLD, font_size=24).to_edge(DOWN, buff=0.5)
        self.play(Write(final_text))
        self.wait(2)
