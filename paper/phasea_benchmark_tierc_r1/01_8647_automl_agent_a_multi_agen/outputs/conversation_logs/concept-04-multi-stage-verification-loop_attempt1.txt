from manim import *

class MultiStageVerificationLoop(Scene):
    def construct(self):
        # 1. Setup Formula and Title
        formula = MathTex(
            r"\mathcal{M}^{\star} = \mathcal{A}_{o}(I^{\star}) \text{ s.t. } \mathrm{ImpVer}(\mathcal{M}^{\star}) = \text{Pass}"
        ).scale(0.8).to_edge(UP)
        self.add(formula)

        # 2. Define Main Positions
        planning_pos = LEFT * 5
        gate_x_coords = [-2, 0, 2]
        deployment_pos = RIGHT * 5

        # 3. Create Visual Elements
        # Planning Node
        planning_box = Rectangle(width=2.2, height=1.2, color=BLUE).move_to(planning_pos)
        planning_label = Text("Planning", font_size=20).move_to(planning_pos)
        planning_group = VGroup(planning_box, planning_label)

        # Verification Gates
        gates = VGroup()
        gate_labels = VGroup()
        gate_names = ["ReqVer", "ExecVer", "ImpVer"]
        for i, name in enumerate(gate_names):
            x = gate_x_coords[i]
            gate_line = Line(UP * 1.5, DOWN * 1.5, color=GREY).move_to([x, 0, 0])
            label = Text(name, font_size=18).next_to(gate_line, UP)
            gates.add(gate_line)
            gate_labels.add(label)

        # Deployment Node
        deployment_icon = Star(color=GOLD, fill_opacity=1).scale(0.4).move_to(deployment_pos)
        deployment_label = Text("Deployment", font_size=18).next_to(deployment_icon, UP)
        deployment_group = VGroup(deployment_icon, deployment_label)

        self.add(planning_group, gates, gate_labels, deployment_group)

        # 4. Create the 'Plan' Object
        plan = Square(side_length=0.4, fill_opacity=1, color=YELLOW).move_to(planning_pos)
        self.play(FadeIn(plan))

        # --- ANIMATION SEQUENCE ---

        # Sequence A: Pass first gate, fail second
        # Move to Gate 1 (ReqVer)
        self.play(plan.animate.move_to([gate_x_coords[0], 0, 0]), run_time=1)
        self.play(plan.animate.set_color(GREEN))
        self.wait(0.2)

        # Move to Gate 2 (ExecVer)
        self.play(plan.animate.move_to([gate_x_coords[1], 0, 0]), run_time=1)
        self.play(plan.animate.set_color(RED))
        self.wait(0.3)

        # Feedback Loop: Use a multi-point line to simulate a curved feedback path
        # This avoids CurvedArrow or ArcBetweenPoints which caused identifier errors
        p_start = [gate_x_coords[1], -0.5, 0]
        p_mid1 = [gate_x_coords[1], -2.0, 0]
        p_mid2 = [planning_pos[0], -2.0, 0]
        p_end = [planning_pos[0], -0.6, 0]
        
        feedback_path = VMobject(color=RED)
        feedback_path.set_points_as_corners([p_start, p_mid1, p_mid2, p_end])
        feedback_path.add_tip()

        self.play(Create(feedback_path))
        self.play(
            plan.animate.move_to(planning_pos).set_color(YELLOW),
            FadeOut(feedback_path),
            run_time=1.5
        )
        self.wait(0.5)

        # Sequence B: Successful path through all three gates
        for i in range(3):
            target_x = gate_x_coords[i]
            self.play(plan.animate.move_to([target_x, 0, 0]), run_time=0.7)
            self.play(plan.animate.set_color(GREEN), run_time=0.2)
        
        # Reach Deployment
        self.play(plan.animate.move_to(deployment_pos), run_time=1)
        self.play(
            Flash(deployment_icon, color=GOLD),
            plan.animate.scale(0),
            run_time=1
        )

        self.wait(2)
