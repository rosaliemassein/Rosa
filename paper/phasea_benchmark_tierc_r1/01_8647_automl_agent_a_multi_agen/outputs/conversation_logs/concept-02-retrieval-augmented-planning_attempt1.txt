from manim import *

class Concept02RetrievalAugmentedPlanning(Scene):
    def construct(self):
        # 1. Setup Icons
        # Agent Manager (Central Icon)
        manager_circle = Circle(radius=0.8, color=YELLOW, fill_opacity=0.2)
        manager_label = Text("Agent\nManager", font_size=24)
        manager = VGroup(manager_circle, manager_label).move_to(ORIGIN)

        # Knowledge Base (Icon on the left)
        kb_circles = VGroup(
            Circle(radius=0.4, color=BLUE, fill_opacity=0.5).shift(LEFT * 0.2),
            Circle(radius=0.4, color=BLUE, fill_opacity=0.5).shift(RIGHT * 0.2),
            Circle(radius=0.4, color=BLUE, fill_opacity=0.5).shift(UP * 0.3)
        )
        kb_label = Text("Knowledge\nBase", font_size=20)
        knowledge_base = VGroup(kb_circles, kb_label).to_edge(LEFT, buff=1.5)

        # 2. Initial Animation
        self.play(FadeIn(knowledge_base), FadeIn(manager))
        self.wait(0.5)

        # 3. Knowledge Retrieval Animation
        # Animate arrows pulling Knowledge into the Manager
        arrows = VGroup(
            Arrow(knowledge_base.get_right(), manager.get_left(), buff=0.1),
            Arrow(knowledge_base.get_right(), manager.get_left(), buff=0.1).shift(UP * 0.5),
            Arrow(knowledge_base.get_right(), manager.get_left(), buff=0.1).shift(DOWN * 0.5)
        )
        
        # Use Dots to represent knowledge flow (manually animated for compatibility)
        particles = VGroup(*[
            Dot(radius=0.08, color=BLUE).move_to(knowledge_base.get_right()) 
            for _ in range(6)
        ])
        
        self.play(Create(arrows))
        self.play(
            LaggedStart(*[
                p.animate.move_to(manager.get_center())
                for p in particles
            ], lag_ratio=0.2),
            run_time=2,
            rate_func=lambda t: t  # Safe alternative to 'linear'
        )
        self.play(FadeOut(particles), FadeOut(arrows))

        # 4. Branching into Parallel Plans
        # Define the three distinct strategy paths
        plan_a_icon = Square(side_length=0.8, color=GREEN, fill_opacity=0.3)
        plan_a_label = Text("Plan A: Transformer", font_size=18).next_to(plan_a_icon, RIGHT)
        plan_a = VGroup(plan_a_icon, plan_a_label).shift(RIGHT * 3 + UP * 2)

        plan_b_icon = Circle(radius=0.4, color=RED, fill_opacity=0.3)
        plan_b_label = Text("Plan B: CNN", font_size=18).next_to(plan_b_icon, RIGHT)
        plan_b = VGroup(plan_b_icon, plan_b_label).shift(RIGHT * 3)

        plan_c_icon = Triangle(color=PURPLE, fill_opacity=0.3).scale(0.5)
        plan_c_label = Text("Plan C: Hybrid", font_size=18).next_to(plan_c_icon, RIGHT)
        plan_c = VGroup(plan_c_icon, plan_c_label).shift(RIGHT * 3 + DOWN * 2)

        # Connecting lines
        conn_a = Line(manager.get_right(), plan_a.get_left(), color=GRAY_B)
        conn_b = Line(manager.get_right(), plan_b.get_left(), color=GRAY_B)
        conn_c = Line(manager.get_right(), plan_c.get_left(), color=GRAY_B)

        # Animation: Branching out using TransformMatchingShapes
        self.play(
            Create(conn_a), Create(conn_b), Create(conn_c),
            TransformMatchingShapes(manager.copy(), plan_a),
            TransformMatchingShapes(manager.copy(), plan_b),
            TransformMatchingShapes(manager.copy(), plan_c),
            run_time=2
        )
        self.wait(1)

        # 5. Formula Display
        formula = MathTex(r"\mathbf{P} = \mathcal{A}_{mgr}(\mathrm{RAP}(R))")
        formula.to_edge(DOWN, buff=1)
        bg_box = SurroundingRectangle(formula, color=YELLOW, buff=0.2)
        
        self.play(Write(formula), Create(bg_box))
        self.wait(2)

        # Cleanup/Summary Text
        summary = Text("Retrieval-Augmented Planning (RAP)", font_size=24, color=YELLOW).to_edge(UP)
        self.play(FadeIn(summary))
        self.wait(2)
