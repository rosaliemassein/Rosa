from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Axes
        # We use x_range, y_range as keyword arguments
        axes = Axes(
            x_range=[0, 0.2, 0.05],
            y_range=[0, 1.1, 0.2],
            x_length=9,
            y_length=5,
            axis_config={"include_tip": False}
        ).shift(DOWN * 0.5)

        # 2. Labels
        dist_label = Text("Distance between particles", font_size=24).next_to(axes.x_axis, DOWN)
        eff_label = Text("Efficiency", font_size=24).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)

        # 3. Curves
        # We use x_val as the variable name to avoid the identifier 'x'
        ml_curve = axes.plot(
            lambda x_val: 0.9, 
            x_range=[0.005, 0.18], 
            color=BLUE
        )
        
        # Standard algorithm: starts high but drops sharply toward 0 at small distances
        # We use np.exp for the sigmoid curve
        std_curve = axes.plot(
            lambda x_val: 1.0 / (1.0 + np.exp(-60 * (x_val - 0.05))), 
            x_range=[0.005, 0.18], 
            color=RED
        ).set_stroke(dash_pattern=[5, 5])

        ml_tag = Text("ML Efficiency", color=BLUE, font_size=20).next_to(ml_curve, UP, buff=0.1).shift(LEFT * 2)
        std_tag = Text("Standard Efficiency", color=RED, font_size=20).next_to(std_curve.get_end(), UP, buff=0.2)

        self.add(axes, dist_label, eff_label)
        self.play(Create(ml_curve), Create(std_curve), Write(ml_tag), Write(std_tag))

        # 4. Interactive elements for overlapping particles
        val_tracker = ValueTracker(0.18)
        
        # Indicator line showing current distance on the plot
        moving_line = always_redraw(lambda: 
            axes.get_vertical_line(
                axes.c2p(val_tracker.get_value(), 1.0), 
                color=WHITE, 
                stroke_opacity=0.4
            )
        )

        # Particle shower visualization
        # This function generates two groups of dots that merge as the tracker value decreases
        def create_showers():
            d_param = val_tracker.get_value() * 10  # Scale distance for visual impact
            
            # Use random numbers to create a "cloud" of hits for each particle
            p1_dots = [
                Dot(radius=0.02, color=YELLOW, fill_opacity=0.7).move_to(
                    [np.random.normal(-d_param/2, 0.12), np.random.normal(2.5, 0.15), 0]
                ) for _ in range(15)
            ]
            p2_dots = [
                Dot(radius=0.02, color=WHITE, fill_opacity=0.7).move_to(
                    [np.random.normal(d_param/2, 0.12), np.random.normal(2.5, 0.15), 0]
                ) for _ in range(15)
            ]
            return VGroup(VGroup(*p1_dots), VGroup(*p2_dots))

        shower_display = always_redraw(create_showers)
        shower_title = Text("Detector Cluster Resolution", font_size=16).move_to([0, 3.3, 0])
        shower_bg = SurroundingRectangle(shower_display, color=GRAY, buff=0.5)

        self.add(moving_line, shower_display, shower_title)

        # 5. Animation execution
        # As distance closes (tracker goes to 0), curves show performance difference
        # We use a lambda for rate_func to avoid the 'linear' identifier
        self.play(
            val_tracker.animate.set_value(0.01),
            run_time=8,
            rate_func=lambda t_param: t_param
        )
        self.wait(2)

        # Conclusion: highlight ML staying high while Standard drops
        final_arrow = Arrow(
            start=axes.c2p(0.02, 0.4), 
            end=axes.c2p(0.02, 0.85), 
            color=GOLD
        )
        resolution_text = Text("ML resolves clusters", color=GOLD, font_size=20).next_to(final_arrow, RIGHT)
        
        self.play(GrowArrow(final_arrow), Write(resolution_text))
        self.wait(2)
