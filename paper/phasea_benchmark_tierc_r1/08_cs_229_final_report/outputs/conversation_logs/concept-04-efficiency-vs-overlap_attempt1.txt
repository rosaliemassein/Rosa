from manim import *
import numpy as np

class Concept04EfficiencyVsOverlap(Scene):
    def construct(self):
        # Setup Axes
        # x-axis: distance between particles (0 to 10)
        # y-axis: efficiency (0 to 1.2)
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[0, 1.2, 0.2],
            x_length=9,
            y_length=5,
            axis_config={"include_tip": False}
        ).to_edge(DOWN, buff=0.5)
        
        x_lab = axes.get_x_axis_label("Distance between particles", edge=DOWN, direction=DOWN, buff=0.3)
        y_lab = axes.get_y_axis_label("Efficiency", edge=LEFT, direction=LEFT, buff=0.3)

        # ML Model: Stays flat around 0.9 efficiency
        # Using 'v' as the lambda variable to avoid potential linter issues with 'x'
        ml_curve = axes.plot(lambda v: 0.9, x_range=[0, 10], color=BLUE)
        ml_tag = Text("ML Approach", color=BLUE, font_size=24).move_to(axes.c2p(8, 1.0))

        # Standard Algorithm: Drops sharply to zero as distance 'v' decreases
        # Modeled with an exponential: 0.9 * (1 - exp(-0.8 * v))
        std_curve_base = axes.plot(
            lambda v: 0.9 * (1 - np.exp(-0.8 * v)),
            x_range=[0, 10],
            color=RED
        )
        
        # We manually handle the DashedVMobject to satisfy environments that might lack the identifier
        try:
            std_curve = DashedVMobject(std_curve_base, num_dashes=40)
        except:
            # Fallback if DashedVMobject is not available
            std_curve = std_curve_base
            std_curve.set_stroke(opacity=0.7)
            
        std_tag = Text("Standard Algorithm", color=RED, font_size=24).move_to(axes.c2p(8, 0.3))

        # Title
        title = Text("Efficiency vs. Particle Overlap", font_size=32).to_edge(UP, buff=0.3)

        # ValueTracker to simulate particle distance closing
        # Starts at 10 (far apart) and moves toward 0 (overlap)
        dist_tracker = ValueTracker(10)

        # Vertical line tracking the current distance on the graph
        v_line = always_redraw(lambda: 
            Line(
                axes.c2p(dist_tracker.get_value(), 0),
                axes.c2p(dist_tracker.get_value(), 1.1),
                color=WHITE,
                stroke_width=2,
                stroke_opacity=0.6
            )
        )

        # Particle shower icons merging at the top of the scene
        def get_shower_icons():
            d = dist_tracker.get_value()
            # visual_offset scales the tracker distance for scene positioning
            visual_offset = d * 0.15
            center_pos = axes.c2p(5, 1.3) # Position above the graph
            
            p1 = Circle(radius=0.2, color=YELLOW, fill_opacity=0.8).set_stroke(width=0)
            p2 = Circle(radius=0.2, color=YELLOW, fill_opacity=0.8).set_stroke(width=0)
            
            # Add outer "glow" circles
            g1 = Circle(radius=0.4, color=YELLOW, fill_opacity=0.2).set_stroke(width=0)
            g2 = Circle(radius=0.4, color=YELLOW, fill_opacity=0.2).set_stroke(width=0)
            
            shower1 = VGroup(g1, p1).move_to(center_pos).shift(LEFT * visual_offset)
            shower2 = VGroup(g2, p2).move_to(center_pos).shift(RIGHT * visual_offset)
            
            return VGroup(shower1, shower2)

        shower_animation = always_redraw(get_shower_icons)

        # --- Animation Sequence ---
        
        # 1. Initialize Graph
        self.add(axes, x_lab, y_lab, title)
        self.play(Create(ml_curve), Write(ml_tag), run_time=1.5)
        self.play(Create(std_curve), Write(std_tag), run_time=1.5)
        self.wait(1)

        # 2. Add dynamic elements
        self.add(v_line, shower_animation)
        
        # 3. Animate distance closing (10 to near 0)
        # Using lambda t: t for rate_func to avoid potential "linear" identifier issues
        self.play(
            dist_tracker.animate.set_value(0.1),
            run_time=8,
            rate_func=lambda t: t
        )
        
        self.wait(2)
        
        # Highlight the efficiency gap at low distance
        indicator_rect = SurroundingRectangle(VGroup(ml_tag, std_tag), color=WHITE, buff=0.2)
        self.play(Create(indicator_rect))
        self.play(FadeOut(indicator_rect))
        self.wait(1)
