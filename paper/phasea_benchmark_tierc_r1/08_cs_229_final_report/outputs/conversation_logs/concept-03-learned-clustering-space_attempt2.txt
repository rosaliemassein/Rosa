import numpy as np

class LearnedClusteringSpace(Scene):
    def construct(self):

            y_length=6,
            background_line_style={"stroke_opacity": 0.3}
        ).shift(LEFT * 3.5)
        
        right_plane = NumberPlane(
            x_range=[-4, 4, 1], 
            y_range=[-4, 4, 1],

            background_line_style={"stroke_opacity": 0.3}
        ).shift(RIGHT * 3.5)
        
        phys_label = Text("Physical Space", color=BLUE).scale(0.7).next_to(left_plane, UP)
        learn_label = Text("Learned Space", color=GREEN).scale(0.7).next_to(right_plane, UP)
        
        # Formula provided in the prompt
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=34
        ).to_edge(DOWN, buff=0.4)
        
        # 2. Generate Clusters

        num_dots = 20
        
        phys_dots_red = VGroup()
        phys_dots_yellow = VGroup()
        
        # We'll define target positions for the transformation
        learn_dots_red_targets = VGroup()
        learn_dots_yellow_targets = VGroup()
        
        for _ in range(num_dots):
            # Physical Space: Overlapping distributions near the center of the left plane

            dot_red = Dot(left_plane.c2p(px1, py1), color=RED, radius=0.06)
            phys_dots_red.add(dot_red)
            
            px2, py2 = np.random.normal(0.2, 0.6), np.random.normal(0, 0.6)
            dot_yellow = Dot(left_plane.c2p(px2, py2), color=YELLOW, radius=0.06)
            phys_dots_yellow.add(dot_yellow)
            
            # Learned Space: Separated circular distributions on the right plane

            learn_dots_red_targets.add(Dot(right_plane.c2p(lx1, ly1), color=RED, radius=0.06))
            
            lx2, ly2 = np.random.normal(1.8, 0.5), np.random.normal(-1.5, 0.5)
            learn_dots_yellow_targets.add(Dot(right_plane.c2p(lx2, ly2), color=YELLOW, radius=0.06))
