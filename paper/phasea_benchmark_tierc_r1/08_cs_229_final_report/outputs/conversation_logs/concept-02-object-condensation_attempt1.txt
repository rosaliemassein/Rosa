from manim import *
import numpy as np

class Concept02ObjectCondensation(Scene):
    def construct(self):
        # 1. Setup Plane and Formula
        plane = NumberPlane(
            x_range=(-7, 7, 1), 
            y_range=(-4, 4, 1), 
            background_line_style={"stroke_opacity": 0.15}
        )
        self.add(plane)

        # q_i = arctanh(beta_i) + q_min
        charge_eq = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        charge_eq.to_edge(UP).fix_in_frame()
        charge_eq.add_background_rectangle()
        self.play(Write(charge_eq))

        # 2. Initialize dots (particles)
        num_dots = 45
        dots = VGroup()
        
        # We'll designate a few points as condensation centers
        center_points = [
            np.array([-3, 1, 0]),
            np.array([2, -1.5, 0]),
            np.array([4, 2, 0])
        ]
        
        condensation_dots = []
        noise_dots = []

        for i in range(num_dots):
            pos = np.array([np.random.uniform(-6, 6), np.random.uniform(-3.5, 3), 0])
            dot = Dot(point=pos, radius=0.06, color=PURPLE)
            
            # Determine if this dot is close to a center to simulate a high beta
            dist_to_centers = [np.linalg.norm(pos - c) for c in center_points]
            min_dist = min(dist_to_centers)
            
            if min_dist < 0.8:
                dot.beta = np.random.uniform(0.85, 0.98)
                condensation_dots.append(dot)
            else:
                dot.beta = np.random.uniform(0.05, 0.4)
                noise_dots.append(dot)
            
            dots.add(dot)

        self.play(FadeIn(dots, lag_ratio=0.05))
        self.wait(0.5)

        # 3. Transition: Beta values grow and dots change color
        # We simulate q = arctanh(beta) visually by color and size
        # Use color_gradient for safe color interpolation
        colors_lookup = color_gradient([PURPLE, YELLOW], 100)
        
        grow_anims = []
        for dot in dots:
            # Color from purple (low beta) to yellow (high beta)
            idx = int(dot.beta * 99)
            target_color = colors_lookup[idx]
            # Radius grows with beta
            target_scale = 1.0 + (dot.beta * 3.5)
            grow_anims.append(dot.animate.set_color(target_color).scale(target_scale))
            
        self.play(*grow_anims, run_time=2)

        # 4. Create "Gravity Wells" for high-beta dots
        wells = VGroup()
        for cp in condensation_dots:
            # Concentric circles to represent potential well
            well = VGroup(*[
                Circle(radius=r, stroke_width=0, fill_opacity=0.12 - (r/20), color=YELLOW)
                for r in np.linspace(0.1, 1.8, 10)
            ]).move_to(cp.get_center())
            wells.add(well)
            
        self.play(FadeIn(wells, lag_ratio=0.2))
        self.wait(0.5)

        # 5. Pull noise dots toward the nearest high-beta condensation point
        pull_anims = []
        for dot in noise_dots:
            # Find the nearest condensation dot
            dists = [np.linalg.norm(dot.get_center() - cp.get_center()) for cp in condensation_dots]
            if dists:
                nearest_cp_idx = np.argmin(dists)
                if dists[nearest_cp_idx] < 4.5:
                    target_pos = condensation_dots[nearest_cp_idx].get_center()
                    # Calculate point 75% of the way to center for visualization
                    start_pos = dot.get_center()
                    interp_pos = start_pos + 0.75 * (target_pos - start_pos)
                    pull_anims.append(dot.animate.move_to(interp_pos))

        # rush_into acts as an acceleration/gravity-like rate function
        self.play(*pull_anims, run_time=3, rate_func=rush_into)
        self.wait(2)
