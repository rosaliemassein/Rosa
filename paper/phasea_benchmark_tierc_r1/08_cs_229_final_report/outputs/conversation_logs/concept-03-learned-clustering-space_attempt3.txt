from manim import *

class LearnedClusteringSpace(Scene):
    def construct(self):
        # 1. Setup Planes (Split Screen)
        phys_plane = NumberPlane(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5.5,
            y_length=5.5,
            background_line_style={"stroke_opacity": 0.3}
        ).to_edge(LEFT, buff=0.5)

        learn_plane = NumberPlane(
            x_range=[-3, 3, 1],
            y_range=[-3, 3, 1],
            x_length=5.5,
            y_length=5.5,
            background_line_style={"stroke_opacity": 0.3}
        ).to_edge(RIGHT, buff=0.5)

        phys_title = Text("Physical Space", font_size=28).next_to(phys_plane, UP)
        learn_title = Text("Learned Space", font_size=28, color=YELLOW).next_to(learn_plane, UP)

        # 2. Define Data Points
        # Physical space: overlapping clusters near center
        p1_phys_coords = [[0.4, 0.5, 0], [0.8, -0.2, 0], [-0.2, 0.6, 0], [0.5, 0.1, 0], [0.1, -0.4, 0], [0.6, 0.7, 0]]
        p2_phys_coords = [[-0.4, -0.5, 0], [-0.8, 0.2, 0], [0.2, -0.6, 0], [-0.5, -0.1, 0], [-0.1, 0.4, 0], [-0.6, -0.7, 0]]

        # Learned space: separated clusters
        p1_learn_coords = [[1.5, 1.5, 0], [1.8, 1.2, 0], [1.2, 1.7, 0], [1.6, 1.0, 0], [1.3, 1.3, 0], [1.9, 1.6, 0]]
        p2_learn_coords = [[-1.5, -1.5, 0], [-1.8, -1.2, 0], [-1.2, -1.7, 0], [-1.6, -1.0, 0], [-1.3, -1.3, 0], [-1.9, -1.6, 0]]

        dots1 = VGroup(*[Dot(phys_plane.c2p(*c), color=RED, radius=0.08) for c in p1_phys_coords])
        dots2 = VGroup(*[Dot(phys_plane.c2p(*c), color=BLUE, radius=0.08) for c in p2_phys_coords])
        
        # 3. Targets for Transformation
        targets1 = VGroup(*[Dot(learn_plane.c2p(*c), color=RED, radius=0.08) for c in p1_learn_coords])
        targets2 = VGroup(*[Dot(learn_plane.c2p(*c), color=BLUE, radius=0.08) for c in p2_learn_coords])

        # 4. Formula
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=32
        ).to_edge(DOWN, buff=0.3)

        # 5. Animation
        self.add(phys_plane, learn_plane, phys_title, learn_title)
        self.play(Create(dots1), Create(dots2))
        self.wait(1)
        
        self.play(Write(formula))
        self.wait(0.5)

        # Transform dots from physical to learned coordinates
        self.play(
            Transform(dots1, targets1),
            Transform(dots2, targets2),
            run_time=3,
            path_arc=0.3
        )
        self.wait(0.5)

        # 6. Classification indicator (Dashed Circles)
        # Using DashedVMobject which is built into Manim Community
        circle1 = DashedVMobject(Circle(radius=0.9, color=RED).move_to(learn_plane.c2p(1.5, 1.5, 0)))
        circle2 = DashedVMobject(Circle(radius=0.9, color=BLUE).move_to(learn_plane.c2p(-1.5, -1.5, 0)))

        self.play(Create(circle1), Create(circle2))
        self.wait(2)
