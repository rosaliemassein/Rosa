from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Split Screen / Planes
        left_origin = 3.5 * LEFT
        right_origin = 3.5 * RIGHT
        
        plane_left = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-4, 4, 1],
            x_length=6,
            y_length=6,
            background_line_style={"stroke_opacity": 0.3}
        ).move_to(left_origin)
        
        plane_right = NumberPlane(
            x_range=[-4, 4, 1],
            y_range=[-4, 4, 1],
            x_length=6,
            y_length=6,
            background_line_style={"stroke_opacity": 0.3}
        ).move_to(right_origin)

        left_label = Text("Physical Space", font_size=24).next_to(plane_left, UP)
        right_label = Text("Learned Space", font_size=24).next_to(plane_right, UP)

        self.add(plane_left, plane_right, left_label, right_label)

        # 2. Create Hits (Dots) - Overlapping in Physical Space
        np.random.seed(42)
        num_dots = 15
        
        # Cluster 1 (Physical)
        c1_phys_dots = VGroup(*[
            Dot(point=plane_left.c2p(np.random.normal(0.4, 0.5), np.random.normal(0, 0.5), 0), 
                color=BLUE, radius=0.07)
            for _ in range(num_dots)
        ])
        
        # Cluster 2 (Physical) - Heavily overlapping
        c2_phys_dots = VGroup(*[
            Dot(point=plane_left.c2p(np.random.normal(-0.4, 0.5), np.random.normal(0, 0.5), 0), 
                color=GREEN, radius=0.07)
            for _ in range(num_dots)
        ])

        self.add(c1_phys_dots, c2_phys_dots)

        # 3. Define Target Dots - Separated in Learned Space
        c1_learned_dots = VGroup(*[
            Dot(point=plane_right.c2p(np.random.normal(-1.5, 0.4), np.random.normal(1.2, 0.4), 0), 
                color=BLUE, radius=0.07)
            for _ in range(num_dots)
        ])

        c2_learned_dots = VGroup(*[
            Dot(point=plane_right.c2p(np.random.normal(1.5, 0.4), np.random.normal(-1.2, 0.4), 0), 
                color=GREEN, radius=0.07)
            for _ in range(num_dots)
        ])

        # 4. Formula and Narration
        formula = MathTex(
            r"L_V = \frac{1}{N} \sum q_j \sum (\mathbb{1} \check{V}_k + (1-\mathbb{1}) \hat{V}_k)",
            font_size=28
        ).to_edge(DOWN, buff=0.3)
        
        narration_text = Text(
            "Mapping hits to a multi-dimensional 'learned coordinate' space.",
            font_size=18, color=YELLOW
        ).next_to(formula, UP, buff=0.2)

        # 5. Animation
        self.wait(1)
        self.play(Write(formula))
        self.wait(1)

        # Transform dots from physical to learned space
        self.play(
            ReplacementTransform(c1_phys_dots, c1_learned_dots),
            ReplacementTransform(c2_phys_dots, c2_learned_dots),
            run_time=3,
            path_arc=0.6
        )

        # 6. Manual dashed circles to avoid undefined identifier errors
        def get_dashed_circle(center, radius, color):
            dashed_circle = VGroup()
            num_segments = 24
            for i in range(num_segments):
                if i % 2 == 0:
                    start_angle = i * (TAU / num_segments)
                    dashed_circle.add(Arc(
                        radius=radius, 
                        start_angle=start_angle, 
                        angle=TAU/num_segments, 
                        color=color,
                        stroke_width=2
                    ))
            return dashed_circle.move_to(center)

        circle1 = get_dashed_circle(c1_learned_dots.get_center(), 1.1, BLUE)
        circle2 = get_dashed_circle(c2_learned_dots.get_center(), 1.1, GREEN)

        self.play(
            Create(circle1),
            Create(circle2),
            FadeIn(narration_text),
            run_time=2
        )
        
        self.wait(3)
