from manim import *
import numpy as np

class ObjectCondensation(Scene):
    def construct(self):
        # Configuration
        V_NUM_DOTS = 30
        np.random.seed(42)
        q_min_val = 0.5
        
        # 1. Formula Display
        formula = MathTex(r"q_i = \text{arctanh}(\beta_i) + q_{min}")
        formula.to_edge(UP)
        self.play(Write(formula))
        self.wait(0.5)

        # 2. Create the Plane and Dots
        # dot_positions will be random 2D points
        dot_positions = [
            np.array([np.random.uniform(-5, 5), np.random.uniform(-3, 1.5), 0])
            for _ in range(V_NUM_DOTS)
        ]
        
        # Assign beta values: most low (noise), a few high (condensation points)
        betas = np.random.uniform(0.1, 0.4, size=V_NUM_DOTS)
        condensation_indices = [4, 12, 25] # Specific indices to be "condensation points"
        for idx in condensation_indices:
            betas[idx] = np.random.uniform(0.85, 0.98)
            
        # Initial state: Small purple dots
        dots = VGroup(*[
            Dot(pos, radius=0.06, color=PURPLE)
            for pos in dot_positions
        ])
        
        self.play(FadeIn(dots, lag_ratio=0.05))
        self.wait(1)
        
        # 3. Animate Growth and Color Shift (Beta Charge)
        # We manually compute the target color and scale based on beta
        grow_animations = []
        for i, dot in enumerate(dots):
            # Beta determines the size and color shift
            beta = betas[i]
            # interpolate_color(PURPLE, YELLOW, beta)
            target_color = interpolate_color(PURPLE, YELLOW, beta)
            target_scale = 1 + beta * 2.5
            
            grow_animations.append(
                dot.animate.set_color(target_color).scale(target_scale)
            )
            
        self.play(*grow_animations, run_time=2)
        self.wait(1)

        # 4. Create Gravity Wells (Concentric Circles/Mesh)
        wells = VGroup()
        for idx in condensation_indices:
            center_dot = dots[idx]
            # A gravity well represented by expanding gradient circles
            well = VGroup(*[
                Circle(radius=r, stroke_width=0, fill_opacity=0.15 - (r * 0.08), fill_color=YELLOW)
                .move_to(center_dot.get_center())
                for r in [0.3, 0.6, 1.0, 1.5]
            ])
            wells.add(well)
            
        self.play(FadeIn(wells, lag_ratio=0.2))
        self.wait(0.5)

        # 5. Physics Pull: Animate low-beta hits being pulled to centers
        pull_animations = []
        for i, dot in enumerate(dots):
            if i in condensation_indices:
                continue
            
            # Find the closest condensation point index
            target_idx = condensation_indices[0]
            min_dist = np.linalg.norm(dot.get_center() - dots[target_idx].get_center())
            for c_idx in condensation_indices[1:]:
                dist = np.linalg.norm(dot.get_center() - dots[c_idx].get_center())
                if dist < min_dist:
                    min_dist = dist
                    target_idx = c_idx
            
            # Move the dot toward the nearest well center
            start_pos = dot.get_center()
            end_pos = dots[target_idx].get_center()
            # We use manual interpolation for the target position (move 75% of the way)
            move_target = start_pos + 0.75 * (end_pos - start_pos)
            
            pull_animations.append(dot.animate.move_to(move_target))
            
        # Use smooth rate function to simulate physical attraction
        self.play(*pull_animations, run_time=3, rate_func=slow_into)
        self.wait(2)
        
        # Cleanup
        self.play(FadeOut(dots), FadeOut(wells), FadeOut(formula))
        self.wait(1)
