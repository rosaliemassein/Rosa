from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # Math function for graph: y = log10(2^(n*n))
        def calc_states_log(n_input):
            return (n_input**2) * np.log10(2)

        # 1. Setup Axes
        # Matrix Size N from 2 to 6
        # States (log scale) from 10^0 to 10^10
        axes = Axes(
            x_range=[2, 6, 1],
            y_range=[0, 10, 2],
            x_length=6,
            y_length=5,
            axis_config={"include_numbers": True},
            tips=True
        ).to_edge(LEFT, buff=0.7)

        # Labels for axes
        y_lab = MathTex(r"\text{States } (10^n)").scale(0.8).rotate(90*DEGREES).next_to(axes.y_axis, LEFT)
        x_lab = Text("Matrix Size (N)", font_size=24).next_to(axes.x_axis, DOWN)
        formula_display = MathTex(r"\text{Total States} = 2^{(N \times N)}", color=YELLOW).to_corner(UR, buff=0.5)

        # 2. Q-Table visual frame
        q_table_rect = Rectangle(width=2.5, height=4, color=WHITE).to_edge(RIGHT, buff=1.0)
        q_table_title = Text("Q-Table", font_size=28).next_to(q_table_rect, UP)

        # 3. State Tracker for current matrix size N
        n_val_tracker = ValueTracker(2)

        # 4. Exponential Curve
        # The plot will update as the tracker increases
        state_plot = always_redraw(lambda: axes.plot(
            calc_states_log,
            x_range=[2, min(n_val_tracker.get_value(), 5.8)],
            color=BLUE
        ))

        # 5. Dynamic Table Rows (scrolling effect)
        def create_moving_rows():
            current_n = n_val_tracker.get_value()
            
            # Number of rows increases exponentially to simulate the "explosion"
            count_rows = int(12 + (current_n - 2)**3.5 * 10)
            count_rows = min(count_rows, 300) # Performance cap
            
            # Speed of scrolling increases as matrix size grows
            scroll_vel = (current_n - 1) * 4
            row_buff = 3.8 / max(count_rows, 1)
            
            # Create a group of lines representing Q-table entries
            rows_vg = VGroup(*[
                Line(
                    q_table_rect.get_left() + [0.1, 0, 0],
                    q_table_rect.get_right() - [0.1, 0, 0],
                    stroke_width=max(0.4, 1.5 - (current_n - 2) * 0.5),
                    color=YELLOW,
                    stroke_opacity=0.7
                ) for _ in range(count_rows)
            ]).arrange(DOWN, buff=row_buff).move_to(q_table_rect.get_center())
            
            # Apply scrolling offset based on time and current speed
            rows_vg.shift(UP * ((self.time * scroll_vel) % max(row_buff, 0.01)))
            
            # Add a red blur overlay when state space becomes massive (N > 4.5)
            if current_n > 4.5:
                blur_opacity = min((current_n - 4.5) * 1.5, 0.85)
                blur_zone = Rectangle(
                    width=q_table_rect.get_width(),
                    height=q_table_rect.get_height(),
                    fill_color=RED,
                    fill_opacity=blur_opacity,
                    stroke_width=0
                ).move_to(q_table_rect.get_center())
                return VGroup(rows_vg, blur_zone)
            
            return rows_vg

        q_rows_moving = always_redraw(create_moving_rows)

        # 6. Animation Execution
        self.add(axes, x_lab, y_lab, formula_display, q_table_rect, q_table_title)
        
        # Initial reveal
        self.play(
            Create(state_plot),
            FadeIn(q_rows_moving),
            run_time=2
        )
        self.wait(1)
        
        # Main transition: N increases from 2 to near 6
        # The curve will shoot off the top as N approaches 6
        self.play(
            n_val_tracker.animate.set_value(5.8),
            run_time=10,
            rate_func=ease_in_quad
        )
        self.wait(2)
