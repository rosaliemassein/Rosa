from manim import *
import numpy as np

class StateSpaceExplosion(Scene):
    def construct(self):
        # 1. Formula and Title
        formula_tex = r"\text{Total States} = 2^{(N \times N)}"
        formula = MathTex(formula_tex).to_edge(UP)
        
        # 2. Axes representing Logarithmic Scale
        # Since 2^(N*N) grows too fast for a linear scale, we plot log10(states)
        # log10(2^(N^2)) = N^2 * log10(2)
        # For N=2, y = 1.2. For N=5, y = 7.5. For N=6, y = 10.8.
        axes = Axes(
            x_range=[2, 6.5, 1],
            y_range=[0, 10, 2],
            x_length=7,
            y_length=5,
            axis_config={"include_numbers": True, "include_tip": True}
        ).shift(DOWN * 0.5)
        
        x_label = axes.get_x_axis_label("N")
        y_label = axes.get_y_axis_label(r"\log_{10}(\text{States})").scale(0.8)

        self.add(formula, axes, x_label, y_label)

        # Function to define growth
        def get_log_states(n_val):
            return (n_val**2) * 0.30103 # log10(2) is approx 0.30103

        # Plot the curve
        graph = axes.plot(get_log_states, x_range=[2, 6.2], color=GREEN)

        # 3. Table Visuals (Initial state for 2x2 grid)
        table_rect = Rectangle(height=4, width=2.5, color=WHITE).to_edge(LEFT, buff=0.5)
        table_title = Text("Q-Table", font_size=24).next_to(table_rect, UP)
        
        # Represent rows as a group of lines
        rows = VGroup(*[
            Line(LEFT, RIGHT, stroke_width=2, color=BLUE).set_width(2.1)
            for _ in range(16)
        ]).arrange(DOWN, buff=0.15).move_to(table_rect)
        
        self.play(Create(table_rect), Write(table_title))
        self.play(Create(rows))
        self.wait(1)

        # 4. Interactive Trackers
        n_tracker = ValueTracker(2)
        
        # Label that updates with N
        n_display = always_redraw(lambda: 
            MathTex(f"N = {n_tracker.get_value():.1f}", color=YELLOW).next_to(axes, UP, buff=0.1)
        )
        
        # Dot that moves along the graph
        graph_dot = always_redraw(lambda: 
            Dot(color=YELLOW).move_to(
                axes.c2p(n_tracker.get_value(), get_log_states(n_tracker.get_value()))
            )
        )

        # Blur effect mobjects
        scrolling_lines = VGroup(*[
            Line(LEFT, RIGHT, stroke_width=1, color=BLUE).set_width(2.1)
            for _ in range(40)
        ]).arrange(DOWN, buff=0.05).move_to(table_rect)
        
        solid_blur = Rectangle(
            height=4, width=2.5, 
            fill_color=BLUE, fill_opacity=0, 
            stroke_width=0
        ).move_to(table_rect)

        # 5. The "Explosion" Animation
        self.add(n_display, graph_dot)
        
        # We use a custom rate_func lambda t: t for a linear progression
        self.play(
            n_tracker.animate.set_value(6.2),
            Create(graph),
            rows.animate.shift(UP * 10).set_opacity(0),
            FadeIn(scrolling_lines),
            scrolling_lines.animate.shift(UP * 20),
            solid_blur.animate.set_fill(opacity=0.8),
            run_time=6,
            rate_func=lambda t: t
        )

        # 6. Final state transition - color turns to red to signify failure/explosion
        self.play(
            solid_blur.animate.set_fill(color=RED, opacity=1),
            formula.animate.set_color(RED).scale(1.1),
            run_time=1
        )
        
        self.wait(2)
