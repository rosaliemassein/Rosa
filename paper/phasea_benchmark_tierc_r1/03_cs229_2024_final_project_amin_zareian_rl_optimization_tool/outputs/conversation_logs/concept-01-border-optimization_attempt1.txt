from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Create 4x4 VGroup of Square objects
        n = 4
        squares = VGroup(*[
            Square(side_length=1.0, fill_opacity=1, stroke_width=2, stroke_color=GRAY_D)
            for _ in range(n * n)
        ]).arrange_in_grid(rows=n, cols=n, buff=0)

        # 2. Initialize left half red and right half blue
        # Columns 0, 1 = RED; Columns 2, 3 = BLUE
        for i in range(n):
            for j in range(n):
                idx = i * n + j
                if j < n // 2:
                    squares[idx].set_fill(RED)
                else:
                    squares[idx].set_fill(BLUE)

        # Display the formula reference
        formula = MathTex(r"B = \sum_{i,j \in \text{Neighbors}} [P_i \neq P_j]").to_edge(UP)
        self.add(formula)

        # 3. Dynamic Border Logic
        # Function to find and create lines between pixels of different colors
        def get_current_borders():
            lines = VGroup()
            for r in range(n):
                for c in range(n):
                    idx = r * n + c
                    current_color = str(squares[idx].get_fill_color()).upper()
                    
                    # Check neighbor to the right
                    if c < n - 1:
                        ridx = r * n + (c + 1)
                        if current_color != str(squares[ridx].get_fill_color()).upper():
                            p1 = squares[idx].get_corner(UR)
                            p2 = squares[idx].get_corner(DR)
                            lines.add(Line(p1, p2, color=GREEN, stroke_width=8).set_z_index(10))
                    
                    # Check neighbor below
                    if r < n - 1:
                        bidx = (r + 1) * n + c
                        if current_color != str(squares[bidx].get_fill_color()).upper():
                            p1 = squares[idx].get_corner(DL)
                            p2 = squares[idx].get_corner(DR)
                            lines.add(Line(p1, p2, color=GREEN, stroke_width=8).set_z_index(10))
            return lines

        # Initialize the border lines with an updater
        border_lines = get_current_borders()
        border_lines.add_updater(lambda mob: mob.become(get_current_borders()))

        # 4. Counter in the corner
        counter_label = Text("Border Adjacencies: ", font_size=28).to_edge(RIGHT).shift(UP * 2)
        # Using DecimalNumber instead of Integer for broad compatibility
        count_val = DecimalNumber(
            len(border_lines), 
            num_decimal_places=0, 
            color=GREEN
        ).next_to(counter_label, DOWN)
        
        # Updater for the counter
        count_val.add_updater(lambda mob: mob.set_value(len(get_current_borders())))

        self.add(squares, border_lines, counter_label, count_val)
        self.wait(1)

        # 5. Animation to flip colors to checkerboard pattern
        # Target state: (row + col) % 2 == 0 is RED, otherwise BLUE
        for r in range(n):
            for c in range(n):
                idx = r * n + c
                target_color = RED if (r + c) % 2 == 0 else BLUE
                
                # Compare current color to target color
                if str(squares[idx].get_fill_color()).upper() != str(target_color).upper():
                    self.play(
                        squares[idx].animate.set_fill(target_color),
                        run_time=0.25
                    )

        self.wait(2)
