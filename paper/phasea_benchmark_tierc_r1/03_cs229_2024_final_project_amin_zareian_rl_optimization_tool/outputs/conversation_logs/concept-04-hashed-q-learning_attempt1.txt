from manim import *

class HashedQLearningConcept(Scene):
    def construct(self):
        # 1. Title and Objective
        title = Text("Concept: Hashed Q-Learning", font_size=36).to_edge(UP)
        goal_text = Text(
            "Memory efficiency vs optimization accuracy.",
            font_size=20, color=GRAY
        ).next_to(title, DOWN)
        
        formula = MathTex(
            r"\text{Index} = \text{Hash}(\text{MatrixState}) \pmod L",
            font_size=28
        ).to_edge(DOWN, buff=0.5)

        self.add(title, goal_text, formula)

        # 2. Helper to create 4x4 grids
        def create_grid_pattern(pattern_type):
            grid = VGroup()
            for r in range(4):
                for c in range(4):
                    if pattern_type == 0: # Checkered
                        color = RED if (r + c) % 2 == 0 else BLUE
                    elif pattern_type == 1: # Vertical Stripes
                        color = RED if c % 2 == 0 else BLUE
                    else: # Bottom-heavy
                        color = RED if r > 1 else BLUE
                    
                    sq = Square(side_length=0.15, fill_opacity=1, fill_color=color, stroke_width=0.5)
                    sq.move_to([c * 0.18, -r * 0.18, 0])
                    grid.add(sq)
            return grid

        # 3. Create Left Grids
        grid1 = create_grid_pattern(0).shift(LEFT * 5 + UP * 1.5)
        grid2 = create_grid_pattern(1).shift(LEFT * 5)
        grid3 = create_grid_pattern(2).shift(LEFT * 5 + DOWN * 1.5)
        grids = VGroup(grid1, grid2, grid3)

        # 4. Hash Function Box
        hash_box = VGroup(
            Rectangle(width=2.5, height=1.5, color=BLUE),
            Text("Hash Function", font_size=24)
        ).move_to(LEFT * 0.5)

        # 5. Q-Table (Target)
        q_rows = VGroup(*[
            Rectangle(width=3, height=0.4, stroke_width=1)
            for _ in range(6)
        ]).arrange(DOWN, buff=0.08).shift(RIGHT * 4)
        
        table_label = Text("Compact Q-Table", font_size=24).next_to(q_rows, UP)
        table_group = VGroup(q_rows, table_label)

        # 6. Animation
        self.play(FadeIn(grids), Create(hash_box), Create(table_group))
        self.wait(1)

        # Mapping visualization
        # G1 maps to Row index 1
        # G2 maps to Row index 4
        # G3 maps to Row index 1 (Collision)
        
        mappings = [
            (grid1, q_rows[1], WHITE, YELLOW),
            (grid2, q_rows[4], WHITE, YELLOW),
            (grid3, q_rows[1], WHITE, RED)
        ]

        for i, (start_grid, target_row, color_in, color_out) in enumerate(mappings):
            a_in = Arrow(start_grid.get_right(), hash_box.get_left(), color=color_in, buff=0.1)
            a_out = Arrow(hash_box.get_right(), target_row.get_left(), color=color_out, buff=0.1)
            
            self.play(GrowArrow(a_in))
            self.play(GrowArrow(a_out))
            
            if i == 2: # Collision Case
                self.play(Indicate(target_row, color=RED))
                collision_text = Text("COLLISION!", color=RED, font_size=20).next_to(target_row, RIGHT)
                
                # Highlight the confusion
                flash_rect = target_row.copy().set_fill(RED, opacity=0.5).set_stroke(RED, 3)
                self.play(FadeIn(flash_rect))
                self.play(flash_rect.animate.set_opacity(0), rate_func=there_and_back, run_time=1)
                self.play(Write(collision_text))
            else:
                self.play(target_row.animate.set_fill(BLUE, opacity=0.3))
            
            self.wait(0.5)

        self.wait(2)
