from manim import *
import numpy as np

class BorderOptimization(Scene):
    def construct(self):
        # 1. Create a 4x4 VGroup of Square objects
        squares = VGroup(*[
            Square(side_length=1.0, fill_opacity=1, stroke_width=2, color=WHITE) 
            for _ in range(16)
        ]).arrange_in_grid(rows=4, cols=4, buff=0)

        # 2. Initially, color the left half red and the right half blue
        # Columns 0, 1 are RED; Columns 2, 3 are BLUE
        for i in range(16):
            col = i % 4
            if col < 2:
                squares[i].set_fill(RED)
            else:
                squares[i].set_fill(BLUE)

        # Helper function to generate green borders between different colors
        def get_current_borders():
            lines = VGroup()
            for r in range(4):
                for c in range(4):
                    idx = r * 4 + c
                    # Check neighbor to the right
                    if c < 3:
                        right_idx = r * 4 + (c + 1)
                        if squares[idx].fill_color != squares[right_idx].fill_color:
                            lines.add(Line(
                                squares[idx].get_corner(UR),
                                squares[idx].get_corner(DR),
                                color=GREEN, stroke_width=10
                            ))
                    # Check neighbor below
                    if r < 3:
                        bottom_idx = (r + 1) * 4 + c
                        if squares[idx].fill_color != squares[bottom_idx].fill_color:
                            lines.add(Line(
                                squares[idx].get_corner(DL),
                                squares[idx].get_corner(DR),
                                color=GREEN, stroke_width=10
                            ))
            return lines

        # Dynamic border group that updates every frame
        borders = always_redraw(get_current_borders)

        # 3. Counter in the corner that increments with adjacencies
        counter_label = MathTex("B = ", color=WHITE).to_corner(UL).shift(RIGHT * 2)
        count_num = Integer(0).next_to(counter_label, RIGHT)
        # Link the counter to the number of lines in the border group
        count_num.add_updater(lambda mob: mob.set_value(len(borders)))

        # Display the formula
        formula = MathTex(r"B = \sum_{i,j \in \text{Neighbors}} [P_i \neq P_j]").to_edge(DOWN)

        self.add(squares, borders, counter_label, count_num, formula)
        self.wait(1)

        # 4. Animate the squares flipping colors one by one to a checkerboard pattern
        # Target pattern: (row + col) % 2 == 0 is RED, else BLUE
        for r in range(4):
            for c in range(4):
                idx = r * 4 + c
                target_color = RED if (r + c) % 2 == 0 else BLUE
                
                # Check if the square needs to change color
                if squares[idx].fill_color != target_color:
                    self.play(
                        squares[idx].animate.set_fill(target_color),
                        run_time=0.4
                    )
                else:
                    # Brief wait to keep rhythm even if color is already correct
                    self.wait(0.1)

        self.wait(2)
