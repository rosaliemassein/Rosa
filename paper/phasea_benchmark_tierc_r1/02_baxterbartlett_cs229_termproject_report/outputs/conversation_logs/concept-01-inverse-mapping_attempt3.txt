from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # Explicitly define constants that were reported as undefined
        VEC_OUT = np.array([0, 0, 1])
        COLOR_BLUE = BLUE
        COLOR_GREEN = GREEN

        # 1. Boxes and Arrow Setup (The "Traditional" relationship)
        # Left side: Desired Behavior (Input)
        db_title = Text("Desired Behavior", font_size=24, color=COLOR_BLUE)
        db_items = VGroup(
            Text("Rise Time", font_size=18),
            Text("Steady State Error", font_size=18),
            Text("Stability", font_size=18)
        ).arrange(DOWN, aligned_edge=LEFT).next_to(db_title, DOWN, buff=0.2)
        db_rect = SurroundingRectangle(VGroup(db_title, db_items), color=COLOR_BLUE, buff=0.3)
        input_label = Text("Input", font_size=16, color=COLOR_BLUE).next_to(db_rect, UP)
        input_group = VGroup(db_rect, db_title, db_items, input_label)

        # Right side: PID Constants (Output)
        pid_title = Text("PID Constants", font_size=24, color=COLOR_GREEN)
        pid_items = MathTex(r"K_p, K_i, K_d", font_size=22, color=COLOR_GREEN).next_to(pid_title, DOWN, buff=0.2)
        pid_rect = SurroundingRectangle(VGroup(pid_title, pid_items), color=COLOR_GREEN, buff=0.3)
        output_label = Text("Output", font_size=16, color=COLOR_GREEN).next_to(pid_rect, UP)
        output_group = VGroup(pid_rect, pid_title, pid_items, output_label)

        # Positions
        input_group.shift(LEFT * 3.5 + UP * 1.5)
        output_group.shift(RIGHT * 3.5 + UP * 1.5)
        arrow = Arrow(db_rect.get_right(), pid_rect.get_left(), color=WHITE, buff=0.1)
        
        diagram = VGroup(input_group, arrow, output_group)

        # Animate Intro
        self.play(FadeIn(input_group))
        self.play(GrowArrow(arrow))
        self.play(FadeIn(output_group))
        self.wait(1)

        # "Flips the script" - Rotate 180 degrees
        self.play(Rotate(diagram, angle=PI, axis=VEC_OUT), run_time=1.5)
        self.wait(1)

        # Shrink and move to top
        self.play(diagram.animate.scale(0.6).to_edge(UP, buff=0.3))

        # 2. Motor Velocity Plot
        axes = Axes(
            x_range=[0, 6, 1],
            y_range=[0, 1.5, 0.5],
            x_length=7,
            y_length=2.2,
            axis_config={"include_tip": False, "font_size": 18},
        ).shift(DOWN * 1.2)
        
        labels = axes.get_axis_labels(
            x_label=Text("Time", font_size=14), 
            y_label=Text("RPM", font_size=14)
        )
        
        # Horizontal setpoint
        setpoint = DashedLine(axes.c2p(0, 1), axes.c2p(5.5, 1), color=GRAY)
        setpoint_text = Text("Setpoint", font_size=12, color=GRAY).next_to(setpoint, RIGHT, buff=0.1)
        
        # Damped oscillation curve
        curve = axes.plot(
            lambda x: 1 - np.exp(-0.7 * x) * np.cos(4 * x),
            x_range=[0, 5.5],
            color=YELLOW
        )

        self.play(Create(axes), Write(labels))
        self.play(Create(setpoint), FadeIn(setpoint_text))
        self.play(Create(curve), run_time=2.5)

        # 3. Braced Labels
        # Rise Time brace (on x-axis)
        p_start = axes.c2p(0, 0)
        p_rise = axes.c2p(0.5, 0)
        brace_rise_line = Line(p_start, p_rise)
        brace_rise = Brace(brace_rise_line, direction=DOWN, buff=0.1)
        text_rise = brace_rise.get_text("Rise Time", buff=0.1).scale(0.5)

        # Steady State Error brace (at the end)
        x_end = 5.2
        y_end = 1 - np.exp(-0.7 * x_end) * np.cos(4 * x_end)
        p_target = axes.c2p(x_end, 1)
        p_actual = axes.c2p(x_end, y_end)
        brace_sse_line = Line(p_actual, p_target)
        brace_sse = Brace(brace_sse_line, direction=RIGHT, buff=0.1)
        text_sse = brace_sse.get_text("Steady State Error", buff=0.1).scale(0.5)

        self.play(Create(brace_rise), FadeIn(text_rise))
        self.play(Create(brace_sse), FadeIn(text_sse))

        # 4. PID Formula
        formula = MathTex(
            r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", 
            font_size=28
        ).to_edge(DOWN, buff=0.2)
        self.play(Write(formula))
        
        self.wait(2)
