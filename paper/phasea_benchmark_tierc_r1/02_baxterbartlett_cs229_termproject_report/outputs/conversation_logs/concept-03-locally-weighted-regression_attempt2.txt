from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Setup Narration and Formula
        voice_text = """The most successful method was Locally Weighted Regression. Instead of trying to fit one rigid line to all the data, it looks at your desired behavior and gives more 'weight' to training examples that look similar. It's like a local expert that ignores irrelevant data points far from your target."""
        
        formula = MathTex(
            r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)",
            font_size=36
        ).to_edge(UP, buff=0.5)

        # 2. Create Axes and Scatter Plot
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-2, 2, 1],
            x_length=10,
            y_length=5,
            axis_config={"include_tip": False}
        ).shift(DOWN * 0.5)

        # Generate sample data: y = sin(x) + noise
        np.random.seed(42)
        n_points = 60
        x_data = np.sort(np.random.uniform(-4.5, 4.5, n_points))
        y_data = np.sin(x_data) + np.random.normal(0, 0.15, n_points)
        
        # Dots initially hidden (opacity 0)
        dots = VGroup(*[
            Dot(axes.c2p(x, y), radius=0.08, color=BLUE) 
            for x, y in zip(x_data, y_data)
        ])

        # 3. Interactive Elements setup
        query_x_tracker = ValueTracker(-4)
        tau = 0.6  # Bandwidth parameter

        query_dot = always_redraw(lambda: 
            Dot(axes.c2p(query_x_tracker.get_value(), 0), color=WHITE).set_z_index(10)
        )
        
        # Weight Field: Visual representation of the 'influence' area
        weight_field = always_redraw(lambda:
            Circle(
                radius=1.5, 
                stroke_width=0, 
                fill_opacity=0.2, 
                color=YELLOW
            ).move_to(axes.c2p(query_x_tracker.get_value(), 0))
        )

        # Updater to make dots glow based on the LWR weight formula
        def update_dots(mobs):
            q_x = query_x_tracker.get_value()
            for i, dot in enumerate(mobs):
                # Calculate weight
                dist = abs(x_data[i] - q_x)
                weight = np.exp(-(dist**2) / (2 * tau**2))
                
                # Visual weighting: points far away fade out, close points glow yellow
                if weight > 0.1:
                    dot.set_color(YELLOW)
                    dot.set_opacity(weight)
                    dot.set_stroke(WHITE, width=weight * 4, opacity=weight)
                else:
                    dot.set_color(BLUE)
                    dot.set_opacity(0.2)
                    dot.set_stroke(width=0)

        dots.add_updater(update_dots)

        # Local Regression Line Calculation
        def get_local_line():
            q_x = query_x_tracker.get_value()
            weights = np.exp(-((x_data - q_x)**2) / (2 * tau**2))
            
            # Weighted Least Squares
            W = np.diag(weights)
            X = np.column_stack((np.ones(n_points), x_data))
            Y = y_data
            
            try:
                XTW = X.T @ W
                beta = np.linalg.inv(XTW @ X) @ XTW @ Y
                
                # Define a short local segment
                l_x = np.array([q_x - 1.0, q_x + 1.0])
                l_y = beta[0] + beta[1] * l_x
                
                return Line(
                    axes.c2p(l_x[0], l_y[0]),
                    axes.c2p(l_x[1], l_y[1]),
                    color=RED,
                    stroke_width=8
                ).set_z_index(5)
            except:
                return Line(axes.c2p(0,0), axes.c2p(0,0), stroke_width=0)

        local_line = always_redraw(get_local_line)

        # 4. Animation Sequence
        self.play(Write(formula))
        self.play(Create(axes), FadeIn(dots), run_time=2)
        
        voice_label = Text(
            voice_text, 
            font_size=20,
            line_spacing=1.2
        ).next_to(formula, DOWN, buff=0.2).set_width(12)
        
        self.play(FadeIn(voice_label))
        self.wait(1)
        
        self.play(FadeIn(query_dot), FadeIn(weight_field))
        self.add(local_line)
        self.wait(1)
        
        # Move query point across the domain
        # Use lambda t: t for a linear rate if 'linear' is undefined
        self.play(
            query_x_tracker.animate.set_value(4), 
            run_time=10, 
            rate_func=lambda t: t
        )
        self.wait(2)
