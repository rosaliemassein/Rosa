from manim import *
import numpy as np

class InverseMappingConcept(Scene):
    def construct(self):
        # 1. Diagram Setup: Split Screen Boxes
        # Desired Behavior Box (Input)
        input_title = Text("Desired Behavior", font_size=24)
        b1 = Text("• Rise Time", font_size=18)
        b2 = Text("• Steady State Error", font_size=18)
        b3 = Text("• Stability", font_size=18)
        bullets = VGroup(b1, b2, b3).arrange(DOWN, aligned_edge=LEFT, buff=0.15)
        input_rect = Rectangle(width=4, height=2.5, color=BLUE)
        input_content = VGroup(input_title, bullets).arrange(DOWN, buff=0.3)
        input_box = VGroup(input_rect, input_content).shift(LEFT * 3.5 + UP * 1.5)

        # PID Constants Box (Output)
        output_title = Text("PID Constants", font_size=24)
        output_math = MathTex(r"K_p, K_i, K_d", font_size=30)
        output_rect = Rectangle(width=4, height=2.5, color=GREEN)
        output_content = VGroup(output_title, output_math).arrange(DOWN, buff=0.4)
        output_box = VGroup(output_rect, output_content).shift(RIGHT * 3.5 + UP * 1.5)

        # Connection Arrow
        arrow = Arrow(input_box.get_right(), output_box.get_left(), buff=0.2)
        diagram = VGroup(input_box, arrow, output_box)
        
        # Reference Formula at the top
        formula = MathTex(r"U = K_p e + K_i \int e dt + K_d \frac{de}{dt}", font_size=28).to_edge(UP, buff=0.2)

        # 2. Motor Velocity Plot Setup
        axes = Axes(
            x_range=[0, 8, 1],
            y_range=[0, 1.5, 0.5],
            x_length=7,
            y_length=2.5,
            axis_config={"include_tip": False}
        ).to_edge(DOWN, buff=0.5)
        
        y_lab = Text("RPM", font_size=18).next_to(axes.y_axis, UP, buff=0.1)
        x_lab = Text("Time", font_size=18).next_to(axes.x_axis, RIGHT, buff=0.1)

        # Define function using t_val to avoid 'x' identifier issues
        def motor_fn(t_val):
            # Settles at 0.85 to show steady state error against 1.0 setpoint
            return 0.85 * (1.0 - np.exp(-0.8 * t_val) * np.cos(3.0 * t_val))
        
        curve = axes.plot(motor_fn, x_range=[0, 7], color=YELLOW)
        setpoint = DashedLine(axes.c2p(0, 1.0), axes.c2p(7, 1.0), color=WHITE)
        setpoint_lab = Text("Setpoint", font_size=14).next_to(setpoint, RIGHT, buff=0.1)

        # 3. Animation Sequence
        self.add(formula)
        self.play(FadeIn(input_box), FadeIn(output_box), Create(arrow))
        self.wait(1)

        # "Flips the script": Rotate the entire diagram 180 degrees
        # We also rotate the text back so it's not upside down
        text_mobjects = [input_title, b1, b2, b3, output_title, output_math]
        self.play(
            diagram.animate.rotate(PI, axis=[0, 0, 1]),
            *[m.animate.rotate(PI) for m in text_mobjects],
            run_time=2
        )
        self.wait(1)

        # Animate the motor velocity plot
        self.play(Create(axes), Write(y_lab), Write(x_lab))
        self.play(Create(setpoint), Write(setpoint_lab))
        self.play(Create(curve), run_time=3)

        # Braced labels for Rise Time and Steady State Error
        # Rise time (approx first intersection near 0.8s)
        rt_brace = Brace(Line(axes.c2p(0, 0), axes.c2p(0.8, 0)), DOWN, buff=0.1)
        rt_text = rt_brace.get_text("Rise Time", font_size=16)

        # Steady State Error (difference at end of curve)
        sse_brace = Brace(Line(axes.c2p(7, 0.85), axes.c2p(7, 1.0)), RIGHT, buff=0.1)
        sse_text = sse_brace.get_text("Steady State Error", font_size=16)

        self.play(
            Create(rt_brace), Write(rt_text),
            Create(sse_brace), Write(sse_text)
        )
        self.wait(2)
