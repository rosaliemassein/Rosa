from manim import *
import numpy as np

class LocallyWeightedRegression(Scene):
    def construct(self):
        # 1. Setup Axes
        ax = Axes(
            x_range=[-5, 5, 1], 
            y_range=[-3, 3, 1], 
            x_length=10,
            y_length=6,
            axis_config={"include_tip": False, "color": GREY}
        ).shift(DOWN * 0.3)
        
        # 2. Generate Scatter Plot
        np.random.seed(42)
        x_data = np.random.uniform(-4, 4, 100)
        y_data = np.sin(x_data) + np.random.normal(0, 0.2, 100)
        
        dots = VGroup(*[
            Dot(ax.c2p(x, y), radius=0.06, color=BLUE) 
            for x, y in zip(x_data, y_data)
        ])
        
        # 3. Query Point and Value Tracker
        x_tracker = ValueTracker(-4)
        
        # Create the local regression line with always_redraw
        def get_regression_line():
            x_target = x_tracker.get_value()
            tau = 0.5
            dists = np.abs(x_data - x_target)
            weights = np.exp(-(dists**2) / (2 * tau**2))
            
            # Weighted Linear Regression math
            X = np.vstack([np.ones_like(x_data), x_data]).T
            W = np.diag(weights)
            
            try:
                # theta = (X^T W X)^-1 X^T W y
                xtwx = X.T @ W @ X + 1e-4 * np.eye(2)
                xtwy = X.T @ W @ y_data
                beta = np.linalg.solve(xtwx, xtwy)
                
                # Define line segment endpoints
                x_start, x_end = x_target - 0.7, x_target + 0.7
                y_start = beta[0] + beta[1] * x_start
                y_end = beta[0] + beta[1] * x_end
                
                return Line(
                    ax.c2p(x_start, y_start), 
                    ax.c2p(x_end, y_end), 
                    color=YELLOW, 
                    stroke_width=6
                )
            except:
                return Line(ax.c2p(0,0), ax.c2p(0,0), stroke_width=0)

        regression_line = always_redraw(get_regression_line)
        
        # 4. Query Dot and Weight Circle
        query_dot = always_redraw(lambda: 
            Dot(ax.c2p(x_tracker.get_value(), np.sin(x_tracker.get_value())), color=WHITE).set_z_index(10)
        )
        
        weight_circle = always_redraw(lambda:
            Circle(radius=1.2, color=WHITE, stroke_width=1, fill_opacity=0.1)
            .move_to(query_dot.get_center())
        )

        # 5. Dot Highlighting logic
        def update_dots(group):
            x_target = x_tracker.get_value()
            tau = 0.5
            for i, dot in enumerate(group):
                dist = abs(x_data[i] - x_target)
                weight = np.exp(-(dist**2) / (2 * tau**2))
                dot.set_opacity(0.2 + 0.8 * weight)
                if weight > 0.5:
                    dot.set_color(WHITE)
                else:
                    dot.set_color(BLUE)

        dots.add_updater(update_dots)

        # 6. Labels
        formula = MathTex(r"w^{(i)} = \exp\left(-\frac{||x^{(i)} - x||^2}{2\tau^2}\right)").to_edge(UP)
        narration = Text(
            "Locally Weighted Regression: Predicting by looking at local neighbors.",
            font_size=24
        ).to_edge(DOWN)

        # 7. Animation
        self.add(ax, dots, formula, narration)
        self.play(Create(query_dot), Create(weight_circle))
        self.add(regression_line)
        self.wait(1)
        
        # Move the query point across the dataset
        self.play(
            x_tracker.animate.set_value(4), 
            run_time=10, 
            rate_func=linear
        )
        self.wait(2)
