from manim import *
import numpy as np

class IterativeClustering(Scene):
    def construct(self):
        # Set seed for reproducible random dot placement
        np.random.seed(42)
        
        # Create 27 dots randomly placed
        dots = VGroup(*[
            Dot(point=[np.random.uniform(-5, 5), np.random.uniform(-3, 3), 0]) 
            for _ in range(27)
        ])
        self.add(dots)
        
        unassigned_dots = list(dots)
        cluster_colors = [BLUE, GREEN, RED]
        
        # Logic for 3 clusters
        for cluster_idx in range(3):
            current_color = cluster_colors[cluster_idx]
            
            # 1. Find the two most similar (closest) unassigned dots to seed a cluster
            min_dist = float('inf')
            seed_pair = [None, None]
            
            for i in range(len(unassigned_dots)):
                for j in range(i + 1, len(unassigned_dots)):
                    d = np.linalg.norm(unassigned_dots[i].get_center() - unassigned_dots[j].get_center())
                    if d < min_dist:
                        min_dist = d
                        seed_pair = [unassigned_dots[i], unassigned_dots[j]]
            
            # Highlight seed pair with bright rings
            rings = VGroup(
                Circle(radius=0.2, color=YELLOW).move_to(seed_pair[0].get_center()),
                Circle(radius=0.2, color=YELLOW).move_to(seed_pair[1].get_center())
            )
            
            self.play(Create(rings), run_time=0.5)
            self.play(
                seed_pair[0].animate.set_color(current_color),
                seed_pair[1].animate.set_color(current_color),
                FadeOut(rings),
                run_time=0.5
            )
            
            current_cluster = [seed_pair[0], seed_pair[1]]
            unassigned_dots.remove(seed_pair[0])
            unassigned_dots.remove(seed_pair[1])
            
            # The expansion center for the "search circle"
            expansion_origin = VGroup(*seed_pair).get_center()
            
            # 2. Greedily pull in the next most similar dots until we hit 9 per cluster
            while len(current_cluster) < 9:
                best_dist = float('inf')
                closest_dot = None
                
                # Find the unassigned dot closest to any member of the current cluster
                for dot in unassigned_dots:
                    for member in current_cluster:
                        d = np.linalg.norm(dot.get_center() - member.get_center())
                        if d < best_dist:
                            best_dist = d
                            closest_dot = dot
                
                # Calculate required radius for search circle to reach the next dot
                # The circle expands from the seed center
                search_radius = np.linalg.norm(closest_dot.get_center() - expansion_origin)
                
                search_circle = Circle(radius=0.01, color=current_color, stroke_opacity=0.5)
                search_circle.move_to(expansion_origin)
                
                self.add(search_circle)
                # Using lambda t: t for linear rate to avoid undefined identifier errors
                self.play(
                    search_circle.animate.set_width(search_radius * 2),
                    run_time=0.3,
                    rate_func=lambda t: t
                )
                
                self.play(
                    closest_dot.animate.set_color(current_color),
                    FadeOut(search_circle),
                    run_time=0.2
                )
                
                current_cluster.append(closest_dot)
                unassigned_dots.remove(closest_dot)
                
        self.wait(2)
