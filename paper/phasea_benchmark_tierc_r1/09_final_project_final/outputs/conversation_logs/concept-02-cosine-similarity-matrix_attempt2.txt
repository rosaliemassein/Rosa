from manim import *
import numpy as np

class Concept02CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Setup Axes
        # Use a simple Axes configuration consistent with standard Manim
        axes = Axes(
            x_range=[0, 5, 1],
            y_range=[0, 5, 1],
            x_length=5,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)

        # 2. ValueTracker for angle theta
        theta_tracker = ValueTracker(60 * DEGREES)
        
        # 3. Vector A (Static, White)
        # Using Line and add_tip as "Vector" was reported as undefined
        line_a = Line(axes.c2p(0, 0), axes.c2p(3.5, 0), color=WHITE)
        line_a.add_tip()
        label_a = MathTex(r"\mathbf{A}").next_to(line_a.get_end(), DOWN)
        
        # 4. Vector B (Dynamic, Red)
        # Create a Line with an updater to simulate a rotating vector
        line_b = Line(axes.c2p(0, 0), axes.c2p(3.5, 0), color=RED)
        line_b.add_tip()
        line_b.rotate(60 * DEGREES, about_point=axes.c2p(0, 0))
        
        def update_b(mob):
            val = theta_tracker.get_value()
            new_b = Line(axes.c2p(0, 0), axes.c2p(3.5, 0), color=RED).add_tip()
            new_b.rotate(val, about_point=axes.c2p(0, 0))
            mob.become(new_b)
        
        line_b.add_updater(update_b)
        
        # Label B
        label_b = MathTex(r"\mathbf{B}", color=RED)
        def update_label_b(mob):
            mob.next_to(line_b.get_end(), UR, buff=0.1)
        label_b.add_updater(update_label_b)
        
        # 5. Angle Arc
        # Using Arc instead of Angle class to ensure compatibility
        angle_arc = Arc(radius=0.7, start_angle=0, angle=60 * DEGREES, color=YELLOW, arc_center=axes.c2p(0, 0))
        
        def update_arc(mob):
            val = theta_tracker.get_value()
            # Ensure angle is at least slightly positive for Arc rendering
            mob.become(Arc(
                radius=0.7, 
                start_angle=0, 
                angle=max(0.001, val), 
                color=YELLOW, 
                arc_center=axes.c2p(0, 0)
            ))
        
        angle_arc.add_updater(update_arc)

        # 6. Similarity Formula and Dynamic Value
        # Full formula from the prompt
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}", 
            color=YELLOW
        ).scale(0.7).to_edge(UP, buff=0.5)
        
        sim_val_label = MathTex(r"\text{Value: }", color=YELLOW)
        sim_val_num = MathTex("0.50", color=YELLOW)
        sim_display = VGroup(sim_val_label, sim_val_num).arrange(RIGHT).next_to(formula, DOWN)
        
        def update_sim_val(mob):
            val = np.cos(theta_tracker.get_value())
            # Replace the number part with the updated value
            mob.become(MathTex(f"{val:.2f}", color=YELLOW).next_to(sim_val_label, RIGHT))
        
        sim_val_num.add_updater(update_sim_val)

        # 7. Similarity Matrix
        # 5x5 grid of squares
        matrix_group = VGroup(*[
            Square(side_length=0.6, fill_opacity=0).set_stroke(BLUE, 2)
            for _ in range(25)
        ]).arrange_in_grid(5, 5, buff=0.1).to_edge(RIGHT, buff=0.5)
        
        matrix_title = MathTex(r"\text{Similarity Matrix}", color=BLUE).scale(0.8).next_to(matrix_group, UP)

            val = np.cos(theta_tracker.get_value())

            mob.set_fill(YELLOW, opacity=max(0, val))
        
        target_cell.add_updater(update_cell)
