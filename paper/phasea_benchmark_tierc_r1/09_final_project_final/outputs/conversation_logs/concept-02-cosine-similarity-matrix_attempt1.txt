from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # Custom hex colors to avoid undefined identifier issues in some environments
        COLOR_DARK_BLUE = "#1C758A"
        COLOR_LIGHT_BLUE = "#58C4DD"
        
        # 1. Title and Formula Setup
        title = Text("Cosine Similarity", font_size=36).to_edge(UP)
        formula = MathTex(
            r"\text{similarity} = \cos(\theta) = \frac{\mathbf{A} \cdot \mathbf{B}}{\|\mathbf{A}\| \|\mathbf{B}\|}",
            font_size=30
        ).next_to(title, DOWN)
        
        # 2. Geometry Visualization Components
        axes = Axes(
            x_range=[-1, 2, 1], y_range=[-1, 2, 1], 
            x_length=4, y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=1)
        
        theta_tracker = ValueTracker(60 * DEGREES)
        
        # Using Arrow as a robust alternative to Vector
        vec_a = Arrow(
            axes.coords_to_point(0, 0), 
            axes.coords_to_point(1.5, 0), 
            color=RED, 
            buff=0
        )
        label_a = MathTex(r"\mathbf{A}", color=RED).next_to(vec_a.get_end(), RIGHT)
        
        # Vector B rotates dynamically
        vec_b = always_redraw(lambda: Arrow(
            axes.coords_to_point(0, 0), 
            axes.coords_to_point(
                1.5 * np.cos(theta_tracker.get_value()), 
                1.5 * np.sin(theta_tracker.get_value())
            ),
            color=BLUE, 
            buff=0
        ))
        label_b = always_redraw(lambda: MathTex(r"\mathbf{B}", color=BLUE).next_to(vec_b.get_end(), UR, buff=0.1))
        
        # Using Arc as a robust alternative to Angle
        angle_arc = always_redraw(lambda: Arc(
            radius=0.5, 
            start_angle=0, 
            angle=theta_tracker.get_value(),
            arc_center=axes.coords_to_point(0, 0),
            color=WHITE
        ))
        
        theta_label = always_redraw(lambda: MathTex(r"\theta").move_to(
            Arc(
                radius=0.8, 
                start_angle=0, 
                angle=max(theta_tracker.get_value(), 0.001), # Avoid zero angle for label placement
                arc_center=axes.coords_to_point(0, 0)
            ).point_from_proportion(0.5)
        ))

        # Similarity value display logic
        sim_text = Text("Similarity: ", font_size=24, color=YELLOW)
        sim_val = always_redraw(lambda: DecimalNumber(
            np.cos(theta_tracker.get_value()), 
            num_decimal_places=2, 
            color=YELLOW
        ).next_to(sim_text, RIGHT))
        sim_group = VGroup(sim_text, sim_val).to_edge(DOWN, buff=1.5).shift(LEFT * 3)

        self.add(title, formula, axes, vec_a, label_a, vec_b, label_b, angle_arc, theta_label, sim_group)
        
        # Animate the vectors aligning and diverging
        self.play(theta_tracker.animate.set_value(10 * DEGREES), run_time=2)
        self.wait(0.5)
        self.play(theta_tracker.animate.set_value(90 * DEGREES), run_time=2)
        self.wait(0.5)
        self.play(theta_tracker.animate.set_value(30 * DEGREES), run_time=1)
        
        # 3. Transition to the Similarity Matrix
        self.play(
            FadeOut(axes, vec_a, label_a, vec_b, label_b, angle_arc, theta_label, sim_group),
            formula.animate.scale(0.8).to_edge(UP)
        )

        # Create a 5x5 Matrix of squares
        matrix_size = 5
        squares = VGroup()
        for i in range(matrix_size):
            for j in range(matrix_size):
                # Identity diagonal = 1.0 similarity, others are random
                val = 1.0 if i == j else np.random.uniform(0.1, 0.7)
                sq = Square(side_length=0.6)
                sq.set_stroke(WHITE, width=1)
                
                # Colors based on alignment intensity
                if i == j:
                    sq.set_fill(YELLOW, opacity=0.9)
                else:
                    sq.set_fill(COLOR_DARK_BLUE, opacity=val)
                squares.add(sq)
        
        squares.arrange_in_grid(rows=5, cols=5, buff=0.1).move_to(ORIGIN)
        matrix_label = Text("Similarity Matrix", font_size=30).next_to(squares, UP)
        
        self.play(Write(matrix_label))
        self.play(Create(squares), run_time=2.5)
        
        # Final highlight to emphasize the "glow" of alignment
        self.play(Indicate(squares[12], color=WHITE)) # Center cell
        self.wait(2)
