from manim import *
import numpy as np

class CosineSimilarityMatrix(Scene):
    def construct(self):
        # 1. Setup Axes and Vectors
        axes = Axes(
            x_range=[-1, 5, 1],
            y_range=[-1, 5, 1],
            x_length=5,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)

        # Vector A is fixed at a specific angle (approx 18.4 degrees)
        angle_a = 18.4 * DEGREES
        vector_a = Arrow(axes.c2p(0, 0), axes.c2p(3.5 * np.cos(angle_a), 3.5 * np.sin(angle_a)), buff=0, color=YELLOW)
        label_a = MathTex(r"\mathbf{A}", color=YELLOW).next_to(vector_a.get_end(), RIGHT)

        # Vector B is controlled by a ValueTracker (angle in degrees)
        theta_tracker = ValueTracker(80)
        
        def get_vector_b():
            angle_b = theta_tracker.get_value() * DEGREES
            return Arrow(axes.c2p(0, 0), axes.c2p(3.5 * np.cos(angle_b), 3.5 * np.sin(angle_b)), buff=0, color=GREEN)

        vector_b = always_redraw(get_vector_b)
        label_b = always_redraw(lambda: MathTex(r"\mathbf{B}", color=GREEN).next_to(vector_b.get_end(), UP))

        # 2. Angle Arc and Formula
        # Using Angle as shown in Reference Example 2
        def get_angle_mobject():
            line1 = Line(axes.c2p(0, 0), axes.c2p(1, 0))
            angle_rad = theta_tracker.get_value() * DEGREES
            line2 = Line(axes.c2p(0, 0), axes.c2p(np.cos(angle_rad), np.sin(angle_rad)))
            return Angle(line1, line2, radius=0.6, color=WHITE)
            
        angle_arc = always_redraw(get_angle_mobject)
        
        theta_tex = always_redraw(lambda: MathTex(r"\theta").move_to(
            Angle(
                Line(axes.c2p(0, 0), axes.c2p(1, 0)),
                Line(axes.c2p(0, 0), axes.c2p(np.cos(theta_tracker.get_value() * DEGREES), np.sin(theta_tracker.get_value() * DEGREES))),
                radius=0.9
            ).point_from_proportion(0.5)
        ))

        # Similarity calculation display
        sim_formula = MathTex(r"\text{similarity} = \cos(\theta) = ", color=WHITE).to_edge(UP, buff=0.5)
        sim_value = always_redraw(lambda: Text(
            f"{np.cos((theta_tracker.get_value() * DEGREES) - angle_a):.2f}",
            font_size=36,
            color=ORANGE
        ).next_to(sim_formula, RIGHT))

        # 3. Similarity Matrix (Grid)
        matrix_group = VGroup()
        for _ in range(25):
            sq = Square(side_length=0.5, fill_opacity=0.8, stroke_width=1, color=BLUE)
            matrix_group.add(sq)
        
        matrix_group.arrange_in_grid(5, 5, buff=0.1).to_edge(RIGHT, buff=1)
        matrix_label = Text("Similarity Matrix", font_size=24).next_to(matrix_group, UP)

        # Updater for the middle cell to react to alignment
        # We manually change color/opacity since interpolate_color was problematic
        active_cell = matrix_group[12] 
        def update_cell(cell):
            similarity = np.cos((theta_tracker.get_value() * DEGREES) - angle_a)
            if similarity > 0.95:
                cell.set_color(WHITE)
                cell.set_fill(WHITE, opacity=1.0)
            elif similarity > 0.8:
                cell.set_color(YELLOW)
                cell.set_fill(YELLOW, opacity=0.8)
            elif similarity > 0.5:
                cell.set_color(ORANGE)
                cell.set_fill(ORANGE, opacity=0.5)
            else:
                cell.set_color(BLUE)
                cell.set_fill(BLUE, opacity=0.3)

        active_cell.add_updater(update_cell)

        # 4. Construction and Animation
        self.add(axes, vector_a, label_a, sim_formula)
        self.play(Create(vector_b), Write(label_b), Create(angle_arc), Write(theta_tex), Write(sim_value))
        self.play(Create(matrix_group), Write(matrix_label))
        self.wait(1)

        # Transition 1: Align vector B with vector A
        self.play(theta_tracker.animate.set_value(18.4), run_time=3)
        self.wait(1)

        # Transition 2: Move vector B away
        self.play(theta_tracker.animate.set_value(110), run_time=3)
        self.wait(2)
