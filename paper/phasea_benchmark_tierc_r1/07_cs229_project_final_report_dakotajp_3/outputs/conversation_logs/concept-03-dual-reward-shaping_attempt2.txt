from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Setup Board and Target
        grid = VGroup(*[
            Dot(point=[x, y, 0], color=GRAY, radius=0.05)
            for x in np.arange(-5, 6, 1)
            for y in np.arange(-3, 4, 1)
        ]).set_opacity(0.3)
        
        target_zone = Circle(radius=1.2, color=GREEN, fill_opacity=0.2).shift(RIGHT * 4)
        target_label = Text("Target Zone", font_size=24, color=GREEN).next_to(target_zone, UP)
        
        # 2. Setup Pegs
        pegs = VGroup(*[
            Dot(point=[x, y, 0], color=BLUE, radius=0.15)
            for x, y in [(-4, 1), (-4, -1), (-5, 0), (-3, 0)]
        ])
        
        # 3. Centroid and Line
        centroid_dot = Dot(pegs.get_center(), color=YELLOW)
        centroid_label = Text("Centroid", font_size=20, color=YELLOW)
        centroid_label.add_updater(lambda m: m.next_to(centroid_dot, UP, buff=0.1))
        
        centroid_line = DashedLine(centroid_dot.get_center(), target_zone.get_center(), color=YELLOW)
        
        def update_centroid(m):
            m.move_to(pegs.get_center())
            
        def update_line(m):
            # Dynamic dashed line from centroid to target
            m.become(DashedLine(centroid_dot.get_center(), target_zone.get_center(), color=YELLOW, dash_length=0.1))

        centroid_dot.add_updater(update_centroid)
        centroid_line.add_updater(update_line)
        
        # 4. Reward Counters and UI
        pegs_in_count = ValueTracker(0)
        
        formula = MathTex(
            "R = w_1(\\text{pegs}_{in})^2 + w_2(\\text{dist})^{-1}",
            font_size=36
        ).to_edge(UP)
        
        reward_display = VGroup(
            Text("In: ", font_size=24),
            DecimalNumber(0, num_decimal_places=0, font_size=24, color=BLUE),
            Text("Dist: ", font_size=24),
            DecimalNumber(0, num_decimal_places=2, font_size=24, color=YELLOW),
            Text("Reward: ", font_size=24),
            DecimalNumber(0, num_decimal_places=2, font_size=24, color=RED)
        ).arrange(RIGHT, buff=0.3).to_edge(DOWN)
        
        pegs_in_num = reward_display[1]
        dist_num = reward_display[3]
        total_reward_num = reward_display[5]
        
        # Updaters for UI numbers
        pegs_in_num.add_updater(lambda m: m.set_value(pegs_in_count.get_value()))
        
        def get_dist():
            return np.linalg.norm(centroid_dot.get_center() - target_zone.get_center())
            
        dist_num.add_updater(lambda m: m.set_value(get_dist()))
        
        def get_total_reward():
            p_in = pegs_in_count.get_value()
            d = get_dist()
            # Conceptual reward calculation
            return (p_in**2) + (10 / (d + 0.1))
            
        total_reward_num.add_updater(lambda m: m.set_value(get_total_reward()))
        
        # 5. Animation Sequence
        self.add(grid, target_zone, target_label, centroid_line, centroid_dot, centroid_label, pegs, formula, reward_display)
        
        self.wait(1)
        
        # Define destinations inside the target zone
        destinations = [
            target_zone.get_center() + LEFT*0.3 + UP*0.3,
            target_zone.get_center() + RIGHT*0.3 + DOWN*0.3,
            target_zone.get_center() + LEFT*0.4 + DOWN*0.2,
            target_zone.get_center() + RIGHT*0.2 + UP*0.4
        ]
        
        for i, peg in enumerate(pegs):
            # Move peg to destination
            self.play(
                peg.animate.move_to(destinations[i]),
                run_time=1.5,
                rate_func=lambda t: t # Use lambda instead of 'linear' identifier
            )
            
            # Update piece count
            pegs_in_count.increment_value(1)
            
            # Visual feedback for entering zone
            bonus_val = int(pegs_in_count.get_value())
            bonus_text = MathTex(f"{bonus_val}^2", color=RED_A, font_size=42).next_to(peg, UP)
            self.play(
                Write(bonus_text),
                bonus_text.animate.shift(UP * 0.7).set_opacity(0),
                run_time=0.7
            )
            self.remove(bonus_text)
            
        self.wait(2)
