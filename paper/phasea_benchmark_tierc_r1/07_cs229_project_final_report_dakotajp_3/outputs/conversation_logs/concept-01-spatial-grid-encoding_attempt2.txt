from manim import *
import numpy as np

class Concept01SpatialGridEncoding(ThreeDScene):
    def construct(self):
        # 1. Define the board parameters
        board_radius = 2
        n = 4 * board_radius + 1  # Formula: n = 4 * radius + 1
        
        # 2. Create the hexagonal grid of circles
        hex_grid = VGroup()
        # Axial coordinates for a hexagon
        for r in range(-board_radius, board_radius + 1):
            for q in range(-board_radius, board_radius + 1):
                if abs(r + q) <= board_radius:
                    # Convert axial to pixel coordinates
                    x = 0.5 * (1.5 * q)
                    y = 0.5 * (np.sqrt(3) * (r + q / 2))
                    circle = Circle(radius=0.2, color=WHITE, stroke_width=2)
                    circle.move_to([x, y, 0])
                    hex_grid.add(circle)
        
        hex_grid.center()

        # 3. Initial Display
        self.play(Create(hex_grid), run_time=1.5)
        self.wait(0.5)

        # 4. Animate tilting into a 3D perspective
        self.move_camera(phi=65 * DEGREES, theta=-35 * DEGREES, run_time=2)
        self.wait(0.5)

        # 5. Morph into a square grid of cells
        # Create an n x n grid of squares (n = 9 if radius = 2)
        square_grid = VGroup(*[
            Square(side_length=0.35, stroke_width=1, fill_opacity=0.1, fill_color=WHITE)
            for _ in range(n * n)
        ]).arrange_in_grid(rows=n, cols=n, buff=0.05)
        square_grid.move_to(ORIGIN)

        self.play(ReplacementTransform(hex_grid, square_grid), run_time=1.5)
        self.wait(0.5)

        # 6. Split into three separate, semi-transparent layers floating vertically (along Z-axis)
        layer_spacing = 1.0
        
        # Layer 1: Self (Blue)
        layer1 = square_grid.copy().set_color(BLUE).shift(OUT * layer_spacing)
        # Layer 2: Target (Green)
        layer2 = square_grid.copy().set_color(GREEN).shift(ORIGIN)
        # Layer 3: Opponent (Red)
        layer3 = square_grid.copy().set_color(RED).shift(IN * layer_spacing)
        
        # Add some "pieces" to the layers to demonstrate encoding
        for layer, color, positions in zip(
            [layer1, layer2, layer3], 
            [BLUE, GREEN, RED],
            [[12, 13, 22], [40, 41, 42], [65, 75, 76]]
        ):
            layer.set_fill(color, opacity=0.2)
            for idx in positions:
                if idx < len(layer):
                    layer[idx].set_fill(color, opacity=0.9)

        layers_group = VGroup(layer1, layer2, layer3)
        
        self.play(
            ReplacementTransform(square_grid, layers_group),
            run_time=2
        )
        self.wait(1)

        # 7. Labels and Annotations (Fixed in Frame)
        formula = MathTex("n = 4 \\times \\text{board radius} + 1", font_size=32).to_corner(UL)
        dim_label = MathTex("n \\times n \\times 3", font_size=42, color=YELLOW).to_edge(RIGHT, buff=1)
        
        layer_labels = VGroup(
            Text("Layer 1: Current Player (Self)", font_size=18, color=BLUE),
            Text("Layer 2: Target Destination", font_size=18, color=GREEN),
            Text("Layer 3: Opponent Pieces", font_size=18, color=RED)
        ).arrange(DOWN, aligned_edge=LEFT, buff=0.4).to_edge(LEFT, buff=0.5)

        narration = Text(
            "Transforming hex-geometry into a\nmulti-layered CNN grid.",
            font_size=22,
            line_spacing=1.5
        ).to_edge(DOWN, buff=0.5)

        self.add_fixed_in_frame_mobjects(formula, dim_label, layer_labels, narration)
        
        self.play(
            Write(formula),
            Write(dim_label),
            FadeIn(layer_labels, shift=RIGHT),
            FadeIn(narration, shift=UP)
        )
        
        self.wait(3)
