from manim import *

class Concept02OpponentAbstraction(Scene):
    def construct(self):
        # 1. Title and Formula
        title = Text("Opponent Abstraction", font_size=40).to_edge(UP)
        formula = MathTex(r"S \in \mathbb{B}^{n \times n \times 3}").next_to(title, DOWN)
        self.play(Write(title), Write(formula))
        self.wait(1)

        # Helper function for a manual cross-out
        def get_custom_cross(mobject):
            l1 = Line(mobject.get_corner(UL), mobject.get_corner(DR), color=RED, stroke_width=6)
            l2 = Line(mobject.get_corner(UR), mobject.get_corner(DL), color=RED, stroke_width=6)
            return VGroup(l1, l2)

        # 2. 2-Player Board Representation
        def create_tensor_stack(colors, spacing=0.2):
            stack = VGroup()
            for i, color in enumerate(colors):
                plane = Square(side_length=2, fill_opacity=0.8, fill_color=color, stroke_color=WHITE)
                plane.shift(i * spacing * (UP + RIGHT))
                stack.add(plane)
            return stack

        two_player_colors = [BLUE, RED, GREY]
        two_player_stack = create_tensor_stack(two_player_colors).shift(LEFT * 3)
        two_p_label = Text("2-Player Game", font_size=24).next_to(two_player_stack, DOWN)
        
        me_label = Text("Me", color=BLUE, font_size=20).next_to(two_player_stack, LEFT)
        opp_label = Text("Opponent", color=RED, font_size=20).next_to(two_player_stack, UP, buff=0.1)

        self.play(FadeIn(two_player_stack), Write(two_p_label), Write(me_label), Write(opp_label))
        self.wait(1)

        # 3. 6-Player Board Representation
        # 5 distinct opponent colors (Using MAGENTA instead of PINK)
        six_player_opponents = [ORANGE, MAGENTA, YELLOW, GREEN, PURPLE]
        
        # Build 6-player stack: Me (1) + Opponents (5) + Board (1) = 7 layers
        six_player_stack = VGroup()
        # Me
        six_player_stack.add(Square(side_length=2, fill_opacity=0.8, fill_color=BLUE, stroke_color=WHITE))
        # Opponents
        for i, c in enumerate(six_player_opponents):
            six_player_stack.add(Square(side_length=2, fill_opacity=0.8, fill_color=c, stroke_color=WHITE).shift((i+1) * 0.2 * (UP + RIGHT)))
        # Board/Environment
        six_player_stack.add(Square(side_length=2, fill_opacity=0.8, fill_color=GREY, stroke_color=WHITE).shift(6 * 0.2 * (UP + RIGHT)))
        
        six_player_stack.shift(RIGHT * 3)
        six_p_label = Text("6-Player Game", font_size=24).next_to(six_player_stack, DOWN)
        
        self.play(FadeIn(six_player_stack), Write(six_p_label))
        self.wait(1)

        # 4. Cross-out individual identities of the 5 opponents
        crosses = VGroup(*[get_custom_cross(six_player_stack[i+1]) for i in range(5)])
        self.play(Create(crosses))
        self.wait(1)

        # 5. Merging 5 colors into a single 'Red' intensity layer
        # Calculate destination for the unified layer based on the stack center
        unified_opp_layer = Square(side_length=2, fill_opacity=0.8, fill_color=RED, stroke_color=WHITE).move_to(six_player_stack[0].get_center() + 0.2 * (UP + RIGHT))
        
        # New background layer for the final 3-layer view
        final_board_layer = six_player_stack[-1].copy().move_to(six_player_stack[0].get_center() + 0.4 * (UP + RIGHT))

        self.play(
            FadeOut(crosses),
            ReplacementTransform(six_player_stack[1:6], unified_opp_layer),
            six_player_stack[6].animate.move_to(final_board_layer.get_center()),
            six_p_label.animate.become(Text("Unified 'Opponent Layer'", font_size=24).next_to(six_p_label, ORIGIN))
        )

        # 6. Check mark over the unified 'Opponent Layer'
        check = Tex(r"\checkmark", color=GREEN).scale(2).move_to(unified_opp_layer.get_center())
        self.play(Write(check))
        self.wait(1)

        # 7. Final remark
        abstraction_text = Text(
            "Me vs The Field", 
            font_size=28, 
            color=YELLOW
        ).to_edge(DOWN, buff=0.5)
        
        self.play(Write(abstraction_text))
        self.play(Indicate(two_player_stack), Indicate(VGroup(six_player_stack[0], unified_opp_layer, six_player_stack[6])))
        self.wait(3)
