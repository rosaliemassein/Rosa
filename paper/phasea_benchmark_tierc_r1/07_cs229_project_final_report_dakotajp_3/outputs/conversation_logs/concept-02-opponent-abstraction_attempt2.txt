from manim import *

class Concept02OpponentAbstraction(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex(r"S \in \mathbb{B}^{n \times n \times 3}").to_edge(UP)
        self.add(formula)

        # 2. Represent 2-player game board (Me: Blue vs Opponent: Red)
        two_p_title = Text("2-Player Game", font_size=24).shift(UP * 2.5 + LEFT * 3.5)
        two_p_board = Rectangle(width=2, height=2, color=GRAY).shift(UP * 1 + LEFT * 3.5)
        p1_2p = Dot(two_p_board.get_center() + LEFT * 0.4, color=BLUE, radius=0.2)
        p2_2p = Dot(two_p_board.get_center() + RIGHT * 0.4, color=RED, radius=0.2)
        two_p_group = VGroup(two_p_title, two_p_board, p1_2p, p2_2p)

        # 3. Represent 6-player game board (Me: Blue vs 5 distinct colors)
        six_p_title = Text("6-Player Game", font_size=24).shift(UP * 2.5 + RIGHT * 3.5)
        six_p_board = Rectangle(width=2, height=2, color=GRAY).shift(UP * 1 + RIGHT * 3.5)
        me_6p = Dot(six_p_board.get_center() + UP * 0.5, color=BLUE, radius=0.2)
        
        # Opponent colors (using standard constants, avoiding PINK if problematic)
        opp_colors = [RED, YELLOW, GREEN, PURPLE, ORANGE]
        opponents = VGroup(*[
            Dot(color=c, radius=0.15) for c in opp_colors
        ]).arrange(RIGHT, buff=0.15).move_to(six_p_board.get_center() + DOWN * 0.3)
        six_p_group = VGroup(six_p_title, six_p_board, me_6p, opponents)

        self.play(FadeIn(two_p_group), FadeIn(six_p_group))
        self.wait(1)

        # 4. Neural Network Input Layers (The abstraction layers)
        # Representing the n x n x 3 tensor as a stack of 3 layers
        layers = VGroup()
        layer_names = ["'Me' Layer", "Board Layer", "Opponent Layer"]
        for i in range(3):
            l = Rectangle(width=2.5, height=2.5)
            l.set_stroke(WHITE, 1)
            l.set_fill(WHITE, opacity=0.05)
            # Offset to create 3D stack effect
            l.shift(DOWN * 1.5 + RIGHT * (i * 0.2 - 0.2) + UP * (i * 0.2 - 0.2))
            layers.add(l)
        
        labels = VGroup(*[
            Text(layer_names[i], font_size=18).next_to(layers[i], DOWN, buff=0.2)
            for i in range(3)
        ])

        self.play(Create(layers), Write(labels))
        self.wait(1)

        # 5. Cross-out animation over individual identities of the 5 opponents
        crosses = VGroup()
        for opp in opponents:
            # Manually drawing lines for the cross to avoid Cross mobject dependency
            c1 = Line(opp.get_corner(UL), opp.get_corner(DR), color=WHITE, stroke_width=2)
            c2 = Line(opp.get_corner(UR), opp.get_corner(DL), color=WHITE, stroke_width=2)
            crosses.add(VGroup(c1, c2))
        
        self.play(Create(crosses))
        self.wait(0.5)

        # 6. Merge the 5 colors into a single 'Red' intensity in the third layer
        # This illustrates the "Opponent Layer" abstraction
        self.play(
            FadeOut(crosses),
            opponents.animate.move_to(layers[2].get_center()).scale(0.1).set_opacity(0),
            layers[2].animate.set_fill(RED, opacity=0.7),
            run_time=2
        )

        # 7. Check mark over the unified 'Opponent Layer'
        # Constructing a check mark manually using lines
        p1 = layers[2].get_center() + LEFT * 0.4 + DOWN * 0.1
        p2 = layers[2].get_center() + DOWN * 0.4
        p3 = layers[2].get_center() + RIGHT * 0.5 + UP * 0.4
        check_mark = VMobject(color=GREEN, stroke_width=10).set_points_as_corners([p1, p2, p3])
        
        self.play(Create(check_mark))
        self.play(Indicate(check_mark, color=GREEN), Flash(layers[2], color=GREEN))
        
        # Summary text
        summary = Text("Model sees 'Me' vs 'The Field'", font_size=24, color=YELLOW).to_edge(DOWN)
        self.play(Write(summary))
        self.wait(3)
