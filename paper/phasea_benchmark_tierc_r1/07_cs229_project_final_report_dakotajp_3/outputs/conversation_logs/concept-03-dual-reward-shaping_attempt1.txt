from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Setup the Target Zone
        goal_circle = Circle(radius=1.5, color=YELLOW, fill_opacity=0.2).shift(RIGHT * 4)
        goal_label = Text("Target Zone", font_size=24).next_to(goal_circle, UP)
        self.play(Create(goal_circle), Write(goal_label))

        # 2. Setup the Pegs
        pegs = VGroup(*[
            Circle(radius=0.25, color=BLUE, fill_opacity=0.8) 
            for _ in range(4)
        ])
        pegs.arrange(DOWN, buff=0.4).shift(LEFT * 4)
        self.play(Create(pegs))

        # 3. Dynamic Centroid and Gravity Line
        # We use always_redraw to avoid updater arguments like 'm'
        centroid_dot = always_redraw(lambda: Dot(
            average_point([p.get_center() for p in pegs]), 
            color=WHITE
        ))
        
        centroid_text = always_redraw(lambda: Text(
            "Centroid", font_size=18
        ).next_to(centroid_dot, UP, buff=0.1))

        gravity_line = always_redraw(lambda: DashedLine(
            centroid_dot.get_center(),
            goal_circle.get_center(),
            color=GRAY_A
        ))

        self.add(gravity_line, centroid_dot, centroid_text)

        # 4. Reward Logic and HUD
        # Formula Display
        formula = MathTex(
            r"R = w_1(\text{pegs}_{in})^2 + w_2(\text{dist}(\text{centroid, target}))^{-1}", 
            font_size=28
        ).to_edge(UP)
        self.play(Write(formula))

        # We define a function to calculate the current reward for the counter
        def calculate_current_reward():
            # Count pegs inside the radius of the goal zone
            p_in = 0
            for p in pegs:
                dist_to_center = np.linalg.norm(p.get_center() - goal_circle.get_center())
                if dist_to_center < 1.5:
                    p_in += 1
            
            # Centroid distance component
            c_pos = average_point([p.get_center() for p in pegs])
            dist_to_goal = np.linalg.norm(c_pos - goal_circle.get_center())
            
            # Weights w1=10, w2=5
            val = 10 * (p_in**2) + 5 * (1.0 / max(dist_to_goal, 0.5))
            return val

        reward_label = Text("Total Reward:", font_size=24).to_corner(UL).shift(DOWN * 0.5)
        reward_counter = always_redraw(lambda: 
            DecimalNumber(calculate_current_reward(), num_decimal_places=2)
            .next_to(reward_label, RIGHT)
        )
        
        self.add(reward_label, reward_counter)

        # 5. Simulation Loop
        # Move pegs one by one to simulate progress
        for i in range(len(pegs)):
            # Define target point inside the circle
            target_pos = goal_circle.get_center() + np.array([
                np.random.uniform(-0.6, 0.6),
                np.random.uniform(-0.6, 0.6),
                0
            ])
            
            # Animate the movement
            self.play(
                pegs[i].animate.move_to(target_pos),
                run_time=1.5
            )
            
            # Check current count of pegs in
            current_in = sum(1 for p in pegs if np.linalg.norm(p.get_center() - goal_circle.get_center()) < 1.5)
            
            # Show "Squared Bonus" popup
            bonus_popup = MathTex(f"{current_in}^2", color=YELLOW).next_to(pegs[i], UP, buff=0.2)
            self.play(
                FadeIn(bonus_popup, shift=UP * 0.2),
                Flash(pegs[i], color=YELLOW, line_length=0.2),
                run_time=0.5
            )
            self.play(FadeOut(bonus_popup), run_time=0.3)

        self.wait(2)
