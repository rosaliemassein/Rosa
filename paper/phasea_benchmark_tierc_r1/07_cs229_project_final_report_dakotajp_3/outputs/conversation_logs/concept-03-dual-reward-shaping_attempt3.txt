from manim import *
import numpy as np

class DualRewardShaping(Scene):
    def construct(self):
        # 1. Create a board with a cluster of blue pegs
        pegs = VGroup(*[
            Dot(color=BLUE, radius=0.2)
            for _ in range(10)
        ])
        pegs.arrange_in_grid(rows=5, cols=2, buff=0.4).shift(LEFT * 4)
        
        target_pos = RIGHT * 4
        target_zone = Circle(radius=1.5, color=YELLOW, fill_opacity=0.2).move_to(target_pos)
        target_dot = Dot(target_pos, color=YELLOW)
        target_label = Text("Target Zone", font_size=24).next_to(target_zone, UP)

        self.add(target_zone, target_dot, target_label)
        self.play(Create(pegs))

        # Helper function for centroid calculation
        def get_centroid_pos():
            return sum([p.get_center() for p in pegs]) / len(pegs)

        # 2. Calculate the centroid and mark it
        centroid_dot = Dot(get_centroid_pos(), color=RED)
        
        def update_centroid_label(label_obj):
            label_obj.next_to(centroid_dot, UP, buff=0.1)
            
        centroid_label = Text("Centroid", color=RED, font_size=20)
        centroid_label.add_updater(update_centroid_label)
        
        def update_centroid_dot(dot_obj):
            dot_obj.move_to(get_centroid_pos())
            
        centroid_dot.add_updater(update_centroid_dot)
        
        self.play(FadeIn(centroid_dot), Write(centroid_label))

        # 3. Draw a dashed line from the Centroid to the Target Zone
        dist_line = always_redraw(lambda: DashedLine(
            centroid_dot.get_center(), 
            target_pos, 
            color=GRAY
        ))
        
        self.play(Create(dist_line))

        # 4. Counters and Formula
        reward_val = ValueTracker(0)
        
        reward_title = Text("Reward: ", font_size=32)
        reward_num = DecimalNumber(0, num_decimal_places=2, include_sign=False)
        reward_display = VGroup(reward_title, reward_num).arrange(RIGHT).to_edge(UP)
        
        def update_reward_num(num_obj):
            num_obj.set_value(reward_val.get_value())
            
        reward_num.add_updater(update_reward_num)
        
        formula = MathTex(
            r"R = w_1(\text{pegs}_{in})^2 + w_2(\text{dist}(\text{centroid, target}))^{-1}", 
            font_size=30, color=YELLOW
        ).to_edge(DOWN)
        
        self.add(reward_display, formula)

        # 5. Animate the pegs moving forward
        w1, w2 = 2.0, 5.0
        pegs_in_count = 0
        
        # Use a simple lambda for linear rate to avoid identifier issues
        linear_rate = lambda t: t
        
        for i in range(len(pegs)):
            peg = pegs[i]
            # Random destination within target zone
            destination = target_pos + np.array([
                np.random.uniform(-0.7, 0.7),
                np.random.uniform(-0.7, 0.7),
                0
            ])
            
            # Animate the movement
            self.play(
                peg.animate.move_to(destination),
                rate_func=linear_rate,
                run_time=0.6
            )
            
            pegs_in_count += 1
            
            # Calculate updated reward
            current_dist = np.linalg.norm(get_centroid_pos() - target_pos)
            # Avoid division by zero
            safe_dist = max(current_dist, 0.1)
            new_reward_total = w1 * (pegs_in_count**2) + w2 * (1.0 / safe_dist)
            
            # Squared Bonus pop-up
            bonus_tex = Text(f"{pegs_in_count}^2", color=GREEN, font_size=30).next_to(peg, UP, buff=0.1)
            
            self.play(
                reward_val.animate.set_value(new_reward_total),
                FadeIn(bonus_tex, shift=UP * 0.3),
                run_time=0.3
            )
            self.play(FadeOut(bonus_tex), run_time=0.1)

        self.wait(2)
