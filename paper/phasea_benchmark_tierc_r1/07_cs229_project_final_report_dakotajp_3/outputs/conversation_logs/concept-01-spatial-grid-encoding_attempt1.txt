from manim import *
import numpy as np

class SpatialGridEncoding(Scene):
    def construct(self):
        # 1. Configuration
        board_radius = 2
        n = 4 * board_radius + 1  # n = 9
        cell_size = 0.3
        
        # 2. Hexagonal Board Construction (2D)
        hex_board = VGroup()
        for r in range(-board_radius * 2, board_radius * 2 + 1):
            for q in range(-board_radius * 2, board_radius * 2 + 1):
                if abs(r + q) <= board_radius * 2:
                    # Hex coordinate to Cartesian
                    x = 0.35 * (q + r / 2)
                    y = 0.35 * (r * np.sqrt(3) / 2)
                    circle = Circle(radius=0.12, color=GRAY_B, fill_opacity=0.2, stroke_width=2)
                    circle.move_to([x, y, 0])
                    hex_board.add(circle)
        
        hex_board.move_to(ORIGIN)
        self.play(Create(hex_board), run_time=1.5)
        self.wait(0.5)

        # 3. Simulate perspective tilt using 2D transformations (Shear and Scale)
        # Instead of 3DScene, we use 2D effects to represent the perspective
        perspective_hex = hex_board.copy().apply_matrix([
            [1, 0.5, 0],
            [0, 0.5, 0],
            [0, 0, 1]
        ]).set_stroke(opacity=0.5)
        
        self.play(Transform(hex_board, perspective_hex), run_time=1.5)
        self.wait(0.5)

        # 4. Transform into a square grid (Represented in 2D perspective)
        square_grid_base = VGroup()
        for i in range(n):
            for j in range(n):
                sq = Square(side_length=cell_size, stroke_width=1, color=GRAY_C)
                sq.move_to([(i - n/2) * cell_size, (j - n/2) * cell_size, 0])
                square_grid_base.add(sq)
        
        # Apply the same 2D perspective transform
        square_grid_base.apply_matrix([
            [1, 0.5, 0],
            [0, 0.5, 0],
            [0, 0, 1]
        ]).move_to(ORIGIN)

        self.play(Transform(hex_board, square_grid_base), run_time=1.5)
        self.remove(hex_board)
        self.add(square_grid_base)

        # 5. Split into 3 semi-transparent layers floating "vertically" in 2D space
        # We shift them along the visual Y-axis to simulate height/depth
        
        # Layer 1: Self pieces (Blue)
        layer_self = square_grid_base.copy().set_color(BLUE).set_stroke(opacity=0.6)
        # Highlight some cells as "pieces"
        for k in [10, 11, 19, 20, 28]: 
            layer_self[k].set_fill(BLUE, opacity=0.8)
        
        # Layer 2: Target zone (Green)
        layer_target = square_grid_base.copy().set_color(GREEN).set_stroke(opacity=0.6)
        for k in [60, 61, 62, 70, 71]: 
            layer_target[k].set_fill(GREEN, opacity=0.8)
            
        # Layer 3: Opponent pieces (Red)
        layer_opponent = square_grid_base.copy().set_color(RED).set_stroke(opacity=0.6)
        for k in [40, 41, 42, 49, 50]: 
            layer_opponent[k].set_fill(RED, opacity=0.8)

        # Define 2D offset for visual stacking
        stack_offset = UP * 1.8
        
        self.play(
            layer_self.animate.shift(stack_offset),
            square_grid_base.animate.set_stroke(opacity=0.2), # Fade the base
            layer_opponent.animate.shift(-stack_offset),
            run_time=2
        )
        # Rename base to layer_target for conceptual clarity
        layer_target.move_to(square_grid_base.get_center())
        self.remove(square_grid_base)
        self.add(layer_target)

        # 6. Labels and Formula
        formula = MathTex(r"n = 4 \times \text{board radius} + 1 = 9", font_size=32)
        formula.to_edge(TOP).to_edge(LEFT)
        
        tensor_shape = MathTex(r"n \times n \times 3", font_size=48, color=YELLOW)
        tensor_shape.next_to(formula, DOWN, aligned_edge=LEFT)

        l1_label = Text("Layer 1: Current Player", color=BLUE, font_size=20).next_to(layer_self, RIGHT)
        l2_label = Text("Layer 2: Target Zone", color=GREEN, font_size=20).next_to(layer_target, RIGHT)
        l3_label = Text("Layer 3: Opponent Pieces", color=RED, font_size=20).next_to(layer_opponent, RIGHT)

        self.play(
            Write(formula),
            Write(tensor_shape),
            FadeIn(l1_label),
            FadeIn(l2_label),
            FadeIn(l3_label)
        )

        # 7. Final Narration/Summary
        narration = Text(
            "The hexagonal board is mapped to a square coordinate system\nfor standard Convolutional Neural Networks.",
            font_size=24,
            line_spacing=1.2
        ).to_edge(DOWN)
        
        self.play(Write(narration))
        self.wait(3)
