from manim import *
import numpy as np

class AdaptiveVsStaticSelection(Scene):
    def construct(self):
        # 1. Labels and Layout Setup
        title_static = Text("Static Benchmark", color=RED).scale(0.7)
        title_ml = Text("Logistic Model", color=BLUE).scale(0.7)
        
        title_static.to_edge(UP).shift(LEFT * 3.5)
        title_ml.to_edge(UP).shift(RIGHT * 3.5)
        
        divider = Line(UP * 4, DOWN * 1, color=WHITE)
        self.add(title_static, title_ml, divider)

        # 2. Floorplan Representation
        def create_floorplan_layout():
            floor = VGroup()
            for row in range(3):
                for col in range(3):
                    rect = Rectangle(width=0.8, height=0.6, stroke_width=1)
                    rect.shift(RIGHT * col * 0.9 + UP * row * 0.7)
                    floor.add(rect)
            return floor.center()

        # Left side floorplan (Static)
        left_floor = create_floorplan_layout().shift(LEFT * 3.5 + UP * 0.5)
        static_indices = [1, 4, 7]
        static_highlights = VGroup(*[
            left_floor[idx].copy().set_fill(RED, opacity=0.6).set_stroke(RED, 2)
            for idx in static_indices
        ])
        self.add(left_floor, static_highlights)

        # Right side floorplan (Adaptive)
        right_floor = create_floorplan_layout().shift(RIGHT * 3.5 + UP * 0.5)
        adaptive_highlights = VGroup(*[
            right_floor[i].copy().set_fill(BLUE, opacity=0.1).set_stroke(BLUE, 2)
            for i in range(len(right_floor))
        ])
        self.add(right_floor, adaptive_highlights)

        # 3. Timeline and Graph
        axes = Axes(
            x_range=[5, 9, 1],
            y_range=[0, 10, 2],
            x_length=10,
            y_length=2.5,
            axis_config={"include_tip": False}
        ).to_edge(DOWN, buff=0.5)
        
        month_names = ["May", "Jun", "Jul", "Aug", "Sep"]
        for i, name in enumerate(month_names):
            lbl = Text(name).scale(0.35).next_to(axes.x_axis.n2p(i + 5), DOWN)
            self.add(lbl)

        graph_label = Text("Missed Requests (Recall Error)", color=WHITE).scale(0.4).next_to(axes, UP, buff=0.1)
        self.add(axes, graph_label)

        # 4. Animation State
        time_tracker = ValueTracker(5)

        # Define explicit plotting functions to avoid parameter name issues
        def static_plot_func(t_val):
            return 7.5 + 0.4 * np.sin(t_val * 2.5)

        def ml_plot_func(t_val):
            return 2.5 + 0.5 * np.cos(t_val * 3.0)

        # 5. Updaters
        # Red line for Static
        static_line = always_redraw(lambda: axes.plot(
            static_plot_func,
            x_range=[5, time_tracker.get_value()],
            color=RED
        ))

        # Blue line for ML
        ml_line = always_redraw(lambda: axes.plot(
            ml_plot_func,
            x_range=[5, time_tracker.get_value()],
            color=BLUE
        ))

        # Flickering ML zones
        def update_adaptive_zones(mobs):
            current_time = time_tracker.get_value()
            for k, room in enumerate(mobs):
                # Pseudo-random flicker based on time and room index
                is_active = np.sin(current_time * 6.0 + k * 1.5) > -0.2
                room.set_fill(BLUE, opacity=0.7 if is_active else 0.05)

        adaptive_highlights.add_updater(update_adaptive_zones)

        # 6. Run Animation
        self.add(static_line, ml_line)
        
        self.play(
            time_tracker.animate.set_value(9),
            run_time=12,
            rate_func=rate_functions.linear
        )
        
        self.wait(2)
        adaptive_highlights.clear_updaters()
        self.wait()
