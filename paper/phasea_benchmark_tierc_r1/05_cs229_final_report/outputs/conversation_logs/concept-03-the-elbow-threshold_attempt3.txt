from manim import *
import numpy as np

class TheElbowThreshold(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)

        x_label = MathTex(r"\text{Fraction of Zones Flagged}").scale(0.6).next_to(axes.x_axis, DOWN)
        y_label = MathTex(r"\text{Recall}").scale(0.6).rotate(90 * DEGREES).next_to(axes.y_axis, LEFT)
        
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}").scale(0.8).to_corner(UL)
        self.add(axes, x_label, y_label, formula)

        # 2. Curve Function (Concave curve: rises sharply then levels off)
        def curve_func(x):
            return x**(0.35) if x > 0 else 0

        curve = axes.plot(curve_func, x_range=[0, 1], color=BLUE)
        self.play(Create(curve), run_time=2)

        # 3. Value Tracker and Moving Point
        vt = ValueTracker(0.01)
        
        dot = always_redraw(lambda: Dot(color=YELLOW).move_to(
            axes.c2p(vt.get_value(), curve_func(vt.get_value()))
        ))
        
        # 4. Building Floorplan (Side Panel)
        grid_rows, grid_cols = 4, 4
        squares = VGroup()
        for i in range(grid_rows):
            for j in range(grid_cols):
                sq = Square(side_length=0.4, stroke_width=2, color=WHITE)
                sq.move_to(RIGHT * 4 + UP * 1.5 + RIGHT * j * 0.5 + DOWN * i * 0.5)
                squares.add(sq)
        
        building_label = Text("Building Zones", font_size=24).next_to(squares, UP)
        self.add(squares, building_label)

        # Updater to highlight squares based on the fraction of zones flagged
        def update_squares(m):
            threshold = vt.get_value()
            total_squares = len(m)
            num_to_highlight = int(threshold * total_squares)
            for idx, sq in enumerate(m):
                if idx < num_to_highlight:
                    sq.set_fill(YELLOW, opacity=0.8)
                else:
                    sq.set_fill(BLACK, opacity=0)

        squares.add_updater(update_squares)
        self.add(dot)

        # 5. Animation sequence
        # Move point and update building (using lambda for linear rate to avoid identifier errors)
        self.play(vt.animate.set_value(0.2), run_time=3, rate_func=lambda t: t)
        
        # Highlight the 'elbow'
        elbow_point = axes.c2p(0.2, curve_func(0.2))
        glow_circle = Circle(radius=0.3, color=YELLOW, stroke_width=5).move_to(elbow_point)
        glow_circle.set_fill(YELLOW, opacity=0.2)
        elbow_text = Text("The Elbow", font_size=20, color=YELLOW).next_to(glow_circle, UR, buff=0.1)
        
        self.play(Create(glow_circle), Write(elbow_text))
        self.wait(1)
        
        # Move further right to show the trade-off (diminishing returns)
        self.play(vt.animate.set_value(0.95), run_time=3, rate_func=lambda t: t)
        self.wait(1)
        
        # Return to the sweet spot
        self.play(
            vt.animate.set_value(0.2),
            FadeOut(glow_circle),
            FadeOut(elbow_text),
            run_time=2
        )
        
        # Final visual confirmation of the elbow
        final_flash = Flash(axes.c2p(0.2, curve_func(0.2)), color=YELLOW, flash_radius=0.5)
        self.play(final_flash)
        self.wait(2)
