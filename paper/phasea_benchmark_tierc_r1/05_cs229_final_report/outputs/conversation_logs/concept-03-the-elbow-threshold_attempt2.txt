from manim import *
import numpy as np

class Concept03TheElbowThreshold(Scene):
    def construct(self):
        # 1. Formula setup
        formula = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=36).to_corner(UL)
        
        # 2. Axes setup
        axes = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            axis_config={"include_tip": False}
        ).to_edge(LEFT, buff=0.5).shift(DOWN * 0.5)
        
        labels = axes.get_axis_labels(
            x_label=Text("Fraction of Zones Flagged", font_size=18),
            y_label=Text("Recall", font_size=18)
        )

        # 3. Curve setup: f(x) = 1 - (1-x)^3 creates a concave curve rising sharply then leveling
        def curve_func(x):
            return 1 - (1 - x)**3

        plot_curve = axes.plot(curve_func, x_range=[0, 1], color=RED)

        # 4. Interactive elements
        tracker = ValueTracker(0.1)
        dot = always_redraw(lambda: Dot(
            axes.c2p(tracker.get_value(), curve_func(tracker.get_value())), 
            color=YELLOW
        ))

        # 5. Side-panel Building Floorplan (4x4 grid)
        num_zones = 16
        zones = VGroup(*[
            Square(side_length=0.6, stroke_color=WHITE, stroke_width=2)
            for _ in range(num_zones)
        ]).arrange_in_grid(rows=4, cols=4, buff=0.1).to_edge(RIGHT, buff=1)
        
        floor_label = Text("Building Zones", font_size=24).next_to(zones, UP)

        def update_zones(mob):
            val = tracker.get_value()
            clamped_val = max(0, min(1, val))
            threshold_index = int(clamped_val * num_zones)
            for i, zone in enumerate(mob):
                if i < threshold_index:
                    zone.set_fill(YELLOW, opacity=0.8)
                else:
                    zone.set_fill(YELLOW, opacity=0.1)
        
        zones.add_updater(update_zones)

        # 6. Text boxes for narration
        narration_1 = Text(
            "Low threshold: too many zones flagged.\nHigh threshold: we miss requests.",
            font_size=22, color=BLUE
        ).to_edge(DOWN)

        narration_2 = Text(
            "The 'elbow' maximizes recall\nwhile minimizing flagged zones.",
            font_size=22, color=GREEN
        ).to_edge(DOWN)

        # 7. Animation Sequence
        self.add(axes, labels, formula, plot_curve, zones, floor_label, dot)
        
        # Narration Part 1
        self.play(Write(narration_1))
        
        # Move right (towards low threshold / many zones flagged)
        self.play(tracker.animate.set_value(0.9), run_time=3, rate_func=lambda t: t)
        self.wait(1)
        
        # Move left (towards high threshold / few zones flagged)
        self.play(tracker.animate.set_value(0.05), run_time=2, rate_func=lambda t: t)
        self.wait(1)

        # Narration Part 2
        self.play(FadeOut(narration_1))
        self.play(Write(narration_2))

        # Highlight the 'elbow'
        elbow_x = 0.35
        self.play(tracker.animate.set_value(elbow_x), run_time=2)
        
        elbow_pos = axes.c2p(elbow_x, curve_func(elbow_x))
        elbow_circle = Circle(radius=0.4, color=YELLOW).move_to(elbow_pos)
        
        self.play(Create(elbow_circle))
        self.play(Flash(elbow_pos, color=YELLOW, flash_radius=0.5))
        
        self.wait(3)

        # Final Cleanup
        self.play(
            FadeOut(axes), FadeOut(labels), FadeOut(formula), 
            FadeOut(plot_curve), FadeOut(zones), FadeOut(floor_label), 
            FadeOut(dot), FadeOut(narration_2), FadeOut(elbow_circle)
        )
