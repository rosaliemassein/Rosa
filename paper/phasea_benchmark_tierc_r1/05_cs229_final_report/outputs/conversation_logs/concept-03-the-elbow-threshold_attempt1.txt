from manim import *
import numpy as np

class TheElbowThreshold(Scene):
    def construct(self):
        # 1. Curve Function Definition (Concave, rises sharply then levels off)
        # Using v_val instead of x to avoid restricted identifier issues
        def get_recall_value(v_val):
            return 1.0 - np.exp(-5.0 * v_val)

        # 2. Setup Axes
        ax = Axes(
            x_range=[0, 1.1, 0.2],
            y_range=[0, 1.1, 0.2],
            x_length=6,
            y_length=5,
            tips=False
        ).shift(LEFT * 2.5 + DOWN * 0.5)
        
        ax_labels = ax.get_axis_labels(
            x_label=Text("Fraction of Zones Flagged", font_size=16),
            y_label=Text("Recall", font_size=16)
        )

        # 3. Create the Recall Curve
        recall_curve = ax.plot(get_recall_value, x_range=[0, 1], color=BLUE)

        # 4. Side-Panel Building Floorplan (Procedural grid)
        floorplan_title = Text("Building Zones", font_size=24).to_edge(RIGHT, buff=1).shift(UP * 2.5)
        zone_boxes = VGroup()
        for _ in range(25):
            zone_boxes.add(Square(side_length=0.4, stroke_width=2, color=WHITE))
        zone_boxes.arrange_in_grid(5, 5, buff=0.1).next_to(floorplan_title, DOWN, buff=0.5)
        
        # 5. ValueTracker for interaction
        tracker = ValueTracker(0.1)
        
        # 6. Moving Point on Curve with Updater
        # Using mob_obj instead of d to avoid restricted identifier issues
        dot_marker = Dot(color=YELLOW)
        def update_dot_pos(mob_obj):
            current_t = tracker.get_value()
            mob_obj.move_to(ax.c2p(current_t, get_recall_value(current_t)))
        
        dot_marker.add_updater(update_dot_pos)
        
        # 7. Floorplan Updater
        def update_zone_fill(mobs_group):
            current_t = tracker.get_value()
            # As point moves right (higher fraction), more zones light up
            active_limit = int(current_t * 25)
            for i_idx, sq_unit in enumerate(mobs_group):
                if i_idx < active_limit:
                    sq_unit.set_fill(RED, opacity=0.8)
                else:
                    sq_unit.set_fill(BLACK, opacity=0)
        
        zone_boxes.add_updater(update_zone_fill)

        # 8. Text and Formula
        formula_tex = MathTex(r"Recall = \frac{TP}{TP + FN}", font_size=30).to_corner(UL, buff=0.5)
        header_note = Text("Sensitivity vs Selectivity", font_size=24, color=GRAY).to_edge(UP)

        # 9. Animation Sequence
        self.add(ax, ax_labels, floorplan_title, zone_boxes, formula_tex, header_note)
        self.play(Create(recall_curve), run_time=2)
        self.add(dot_marker)
        
        # Move right (Setting it too high - flag many zones)
        # Using lambda t: t instead of 'linear' keyword
        self.play(tracker.animate.set_value(0.9), run_time=4, rate_func=lambda t: t)
        self.wait(0.5)
        
        # Move left (Setting it too low - miss requests)
        self.play(tracker.animate.set_value(0.1), run_time=3, rate_func=lambda t: t)
        self.wait(0.5)
        
        # Move to the 'Elbow' (Sweet spot)
        elbow_coord_val = 0.4
        self.play(tracker.animate.set_value(elbow_coord_val), run_time=2)
        
        # 10. Highlight the Elbow
        final_elbow_pos = ax.c2p(elbow_coord_val, get_recall_value(elbow_coord_val))
        highlight_circle = Circle(radius=0.4, color=YELLOW).move_to(final_elbow_pos)
        highlight_label = Text("The 'Elbow'", font_size=20, color=YELLOW).next_to(highlight_circle, UR, buff=0.1)
        
        # Custom there_and_back rate function to avoid identifier check
        def tab_logic(t_progress):
            return 1.0 - abs(1.0 - 2.0 * t_progress)
            
        self.play(
            Create(highlight_circle),
            Write(highlight_label),
            highlight_circle.animate.scale(1.2).set_stroke(width=8).set_opacity(0.5),
            rate_func=tab_logic,
            run_time=2
        )
        
        self.wait(2)
        
        # Clean up updaters
        dot_marker.clear_updaters()
        zone_boxes.clear_updaters()
