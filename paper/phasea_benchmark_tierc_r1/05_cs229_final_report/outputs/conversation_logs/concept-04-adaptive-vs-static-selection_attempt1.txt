from manim import *
import numpy as np

class AdaptiveVSStaticSelection(Scene):
    def construct(self):
        # Define a linear rate function locally to avoid potential undefined identifier issues
        func_linear = lambda t: t
        
        # 1. Background/Layout Setup
        divider = Line(UP * 4, DOWN * 1, color=GRAY_A)
        self.add(divider)

        title_static = Text("Static Benchmark", color=BLUE).scale(0.7).to_edge(UP).shift(LEFT * 3.5)
        title_logistic = Text("Logistic Model", color=GREEN).scale(0.7).to_edge(UP).shift(RIGHT * 3.5)
        self.add(title_static, title_logistic)

        # 2. Timeline and Axes (May to September)
        # May=5, June=6, July=7, August=8, September=9
        axes = Axes(
            x_range=[5, 9, 1],
            y_range=[0, 100, 20],
            x_length=10,
            y_length=3,
            axis_config={"include_numbers": False, "color": WHITE},
            tips=False
        ).to_edge(DOWN, buff=0.5)
        
        month_names = ["May", "Jun", "Jul", "Aug", "Sep"]
        month_labels = VGroup(*[
            Text(month_names[idx], font_size=20).next_to(axes.c2p(5 + idx, 0), DOWN)
            for idx in range(5)
        ])
        graph_label = Text("Missed Requests (Recall Error)", font_size=22, color=WHITE).next_to(axes, UP, buff=0.2)
        
        self.add(axes, month_labels, graph_label)

        # 3. Floorplan Creation
        def create_floorplan(center_point):
            # Creates a 2x3 grid of rooms
            rooms = VGroup()
            for i in range(6):
                room = Rectangle(width=0.9, height=0.7, stroke_color=WHITE)
                room.move_to(center_point + RIGHT * (i % 3 - 1) * 1.0 + DOWN * (i // 3 - 0.5) * 0.8)
                rooms.add(room)
            return rooms

        fp_left = create_floorplan(LEFT * 3.5 + UP * 1.5)
        fp_right = create_floorplan(RIGHT * 3.5 + UP * 1.5)
        self.add(fp_left, fp_right)

        # 4. Static Highlights (Frozen throughout)
        static_highlights = VGroup(*[
            Rectangle(width=0.9, height=0.7, fill_color=BLUE, fill_opacity=0.6, stroke_width=0)
            .move_to(fp_left[room_idx]) for room_idx in [0, 2, 4] 
        ])
        self.add(static_highlights)

        # 5. Dynamic Setup with ValueTracker
        time_tracker = ValueTracker(5) 

        # Logistic Highlights (Dynamic flickering and count changing)
        def get_logistic_highlights():
            curr_t = time_tracker.get_value()
            # Intensity peaks mid-summer (July/August)
            intensity = np.sin((curr_t - 5) / 4 * PI)
            num_zones = int(2 + 4 * intensity) 
            
            # Use high frequency sine for flickering
            flicker_val = 0.4 + 0.3 * np.sin(curr_t * 20)
            
            highlights_group = VGroup()
            for room_idx in range(min(num_zones, 6)):
                highlight_rect = Rectangle(
                    width=0.9, 
                    height=0.7, 
                    fill_color=GREEN, 
                    fill_opacity=flicker_val, 
                    stroke_width=0
                ).move_to(fp_right[room_idx])
                highlights_group.add(highlight_rect)
            return highlights_group

        logistic_highlights = always_redraw(get_logistic_highlights)
        self.add(logistic_highlights)

        # 6. Graphs for Missed Requests
        static_line = always_redraw(lambda: axes.plot(
            lambda val_x: 70 + 8 * np.sin(val_x * 1.2), 
            x_range=[5, time_tracker.get_value()], 
            color=BLUE
        ))
        
        logistic_line = always_redraw(lambda: axes.plot(
            lambda val_x: 20 + 12 * np.cos(val_x * 0.8), 
            x_range=[5, time_tracker.get_value()], 
            color=GREEN
        ))
        
        self.add(static_line, logistic_line)

        # Tag labels for the lines
        static_tag = Text("Static", color=BLUE, font_size=16)
        logistic_tag = Text("Logistic", color=GREEN, font_size=16)
        
        static_tag.add_updater(lambda mob: mob.next_to(
            axes.c2p(time_tracker.get_value(), 70 + 8 * np.sin(time_tracker.get_value() * 1.2)), 
            UR, buff=0.1
        ))
        logistic_tag.add_updater(lambda mob: mob.next_to(
            axes.c2p(time_tracker.get_value(), 20 + 12 * np.cos(time_tracker.get_value() * 0.8)), 
            DR, buff=0.1
        ))
        
        self.add(static_tag, logistic_tag)

        # 7. Play Animation
        self.play(
            time_tracker.animate.set_value(9),
            run_time=12,
            rate_func=func_linear
        )
        self.wait(1)

        # Conclusion Text
        conclusion = Text("Logistic Model adapts to seasonal heat, reducing errors", font_size=24, color=YELLOW).to_edge(UP, buff=0.2)
        self.play(FadeIn(conclusion, shift=DOWN))
        self.wait(2)
