from manim import *
import numpy as np

class KMeansSpatialQuantization(Scene):
    def construct(self):
        # 1. Background and Outline
        plane = NumberPlane().set_opacity(0.1)
        self.add(plane)

        # Simplified Australia-like polygon points
        points = [
            [0, 2, 0], [1.5, 1.8, 0], [3, 1, 0], [3.5, -1, 0], 
            [2.5, -2.5, 0], [1, -2, 0], [-1, -2.5, 0], [-3, -2, 0],
            [-3.5, 0, 0], [-2, 1.5, 0]
        ]
        country_outline = Polygon(*points, color=BLUE, stroke_width=4)
        country_outline.set_fill(BLUE, opacity=0.2)
        
        self.play(Create(country_outline))

        # 2. Data Points (Randomly placed within a bounds, filtered)
        np.random.seed(42)
        data_coords = []
        while len(data_coords) < 120:
            p = [np.random.uniform(-3.5, 3.5), np.random.uniform(-2.5, 2.2), 0]
            # Simple bounding check or polygon point-in-shape check
            if country_outline.point_is_inside(p):
                data_coords.append(p)
        
        dots = VGroup(*[Dot(p, radius=0.05, color=GRAY) for p in data_coords])
        self.play(FadeIn(dots, lag_ratio=0.05))

        # 3. Initial Centroids (Creating "Cross" manually to avoid undefined identifiers)
        def make_cross(color):
            return VGroup(
                Line(0.2 * UL, 0.2 * DR, stroke_width=5),
                Line(0.2 * UR, 0.2 * DL, stroke_width=5)
            ).set_color(color)

        num_k = 4
        colors = [RED, GREEN, YELLOW, PINK]
        initial_positions = [[1, 1, 0], [-1.5, 1.2, 0], [-1.2, -1, 0], [2, -1.2, 0]]
        
        centroids = VGroup()
        for i in range(num_k):
            c = make_cross(colors[i])
            c.move_to(initial_positions[i])
            centroids.add(c)
            
        self.play(FadeIn(centroids))
        self.wait(1)

        # 4. K-means Iteration Logic
        for iteration in range(2):
            # Assignment Phase
            assign_anims = []
            clusters = {i: [] for i in range(num_k)}
            
            for dot in dots:
                dists = [np.linalg.norm(dot.get_center() - c.get_center()) for c in centroids]
                idx = np.argmin(dists)
                clusters[idx].append(dot)
                assign_anims.append(dot.animate.set_color(colors[idx]))
            
            self.play(*assign_anims, run_time=1.5)

            # Update Phase (Centroids move to mean)
            move_anims = []
            for i in range(num_k):
                if clusters[i]:
                    new_mean = np.mean([d.get_center() for d in clusters[i]], axis=0)
                    move_anims.append(centroids[i].animate.move_to(new_mean))
            
            self.play(*move_anims, run_time=1.5)
            self.wait(0.5)

        # 5. Formula and Final Annotations
        formula = MathTex(
            r"\mu_{j}:=\frac{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}x^{(i)}}{\sum_{i=1}^{n}\mathbf{1}\{c^{(i)}=j\}}",
            font_size=36
        ).to_edge(UP).add_background_rectangle()
        
        self.play(Write(formula))
        
        # Narration text (truncated for display)
        narration = Text(
            "Regression → Classification",
            font_size=24, color=WHITE
        ).to_edge(DOWN, buff=1.0)
        
        goal = Text(
            "Goal: Discretize continuous space into regions",
            font_size=20, color=GRAY_A
        ).next_to(narration, DOWN)

        self.play(FadeIn(narration), FadeIn(goal))
        
        # Highlight final centroids
        highlights = VGroup(*[Circle(radius=0.3, color=WHITE).move_to(c) for c in centroids])
        self.play(Create(highlights))
        self.wait(3)
