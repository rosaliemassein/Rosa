from manim import *
import numpy as np

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Encoders
        img_encoder = Rectangle(width=3, height=1.5, color=BLUE).shift(LEFT * 4 + UP * 1.5)
        img_enc_label = Text("Image Encoder", font_size=24).move_to(img_encoder)
        
        text_encoder = Rectangle(width=3, height=1.5, color=GREEN).shift(LEFT * 4 + DOWN * 1.5)
        text_enc_label = Text("Text Encoder", font_size=24).move_to(text_encoder)
        
        # Inputs
        # Using a representative square for the street view image
        img_input = Square(side_length=1.2, fill_opacity=0.5, color=BLUE_E).next_to(img_encoder, LEFT, buff=0.8)
        img_input_text = Text("Street View\n(France)", font_size=16).move_to(img_input)
        
        text_input = Text("'A photo in France'", font_size=18).next_to(text_encoder, LEFT, buff=0.8)

        # 2. Embedding Space (2D Coordinate System)
        axes = Axes(
            x_range=[-1.5, 1.5, 1],
            y_range=[-1.5, 1.5, 1],
            x_length=4,
            y_length=4,
            axis_config={"include_tip": True}
        ).shift(RIGHT * 3)
        space_label = Text("Shared Embedding Space", font_size=20).next_to(axes, UP)

        # 3. Vectors and Similarity Logic
        # We'll use a tracker for the angle of the text vector
        # Image vector is fixed at 45 degrees
        img_angle = 45 * DEGREES
        txt_angle_tracker = ValueTracker(150 * DEGREES)
        
        # Image vector (fixed)
        img_vec = Arrow(
            axes.c2p(0, 0, 0), 
            axes.c2p(np.cos(img_angle), np.sin(img_angle), 0), 
            buff=0, color=BLUE
        )
        img_vec_label = Text("Image", font_size=18, color=BLUE).next_to(img_vec.get_end(), UR, buff=0.1)

        # Text vector (dynamic)
        txt_vec = always_redraw(lambda: Arrow(
            axes.c2p(0, 0, 0),
            axes.c2p(np.cos(txt_angle_tracker.get_value()), np.sin(txt_angle_tracker.get_value()), 0),
            buff=0, color=GREEN
        ))
        
        txt_vec_label = always_redraw(lambda: Text(
            "Text", font_size=18, color=GREEN
        ).next_to(txt_vec.get_end(), 
                  direction=np.array([np.cos(txt_angle_tracker.get_value()), np.sin(txt_angle_tracker.get_value()), 0]), 
                  buff=0.1))

        # Similarity Score (Cosine Similarity = cos(theta1 - theta2))
        sim_tracker = ValueTracker(0)
        sim_label = Text("Cosine Similarity: ", font_size=24).to_edge(DOWN).shift(LEFT * 0.5)
        sim_number = DecimalNumber(0, num_decimal_places=3).next_to(sim_label, RIGHT)
        
        def update_sim_number(mobj):
            angle_diff = txt_angle_tracker.get_value() - img_angle
            mobj.set_value(np.cos(angle_diff))
        
        sim_number.add_updater(update_sim_number)

        # 4. Animation Sequence
        self.play(Create(img_encoder), Create(text_encoder), Write(img_enc_label), Write(text_enc_label))
        self.play(FadeIn(img_input), Write(img_input_text), Write(text_input))
        self.wait()

        # Transition to Shared Space
        self.play(
            img_input.animate.scale(0.2).move_to(img_encoder),
            text_input.animate.scale(0.5).move_to(text_encoder),
            FadeOut(img_input_text)
        )
        self.play(Create(axes), Write(space_label))
        self.play(GrowArrow(img_vec), Write(img_vec_label))
        self.play(GrowArrow(txt_vec), Write(txt_vec_label))
        self.add(sim_label, sim_number)
        self.wait()

        # Aligining (Positive Training)
        self.play(
            txt_angle_tracker.animate.set_value(img_angle),
            run_time=3,
            rate_func=bezier([0, 0, 1, 1])
        )
        self.wait()

        # Pushing away (Contrastive - Wrong Label)
        wrong_text = Text("'A photo in Japan'", font_size=18, color=RED).move_to(text_encoder)
        self.play(FadeOut(text_input), FadeIn(wrong_text))
        self.play(
            txt_angle_tracker.animate.set_value(220 * DEGREES),
            txt_vec.animate.set_color(RED),
            run_time=2
        )
        self.wait()

        # Final Formula
        formula = MathTex(r"\mathcal{L}_{CLIP} = 0.5 \cdot (\mathcal{L}_{Text} + \mathcal{L}_{Images})")
        formula.to_edge(UP).shift(LEFT * 2)
        self.play(Write(formula))
        self.wait(2)
