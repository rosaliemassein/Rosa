from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Formula Display
        formula = MathTex(
            r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})",
            font_size=36
        ).to_edge(UP)
        self.add(formula)

        # 2. Player Icons
        # Simple representation of 10 players
        players = VGroup(*[
            VGroup(
                Circle(radius=0.15, color=WHITE),
                Line(ORIGIN, DOWN * 0.25),
                Line(LEFT * 0.15, RIGHT * 0.15).shift(DOWN * 0.1)
            )
            for _ in range(10)
        ]).arrange(RIGHT, buff=0.5).shift(DOWN * 2)

        player_labels = VGroup(*[
            Text(str(i + 1), font_size=18).next_to(players[i], DOWN, buff=0.1)
            for i in range(10)
        ])

        self.play(Create(players), Write(player_labels))

        # 3. Bars representing Average Coefficients
        # Simulated values: 1st is high, 5th and 8th have peaks relative to neighbors
        height_values = [3.2, 2.4, 1.0, 0.8, 1.9, 0.7, 0.6, 1.5, 0.4, 0.2]
        bars = VGroup()
        for i, h in enumerate(height_values):
            bar = Rectangle(
                width=0.3, 
                height=h, 
                fill_opacity=0.8, 
                color=WHITE, 
                stroke_width=1
            )
            # Position bar base on the player icon
            bar.next_to(players[i], UP, buff=0.1)
            # Align bottom of the rectangle to the reference point
            bar.shift(UP * (h/2)) 
            # Correcting alignment: set_y and align_to is cleaner
            bar.align_to(players[i].get_top() + UP*0.1, DOWN)
            bars.add(bar)

        # Animate bars growing from their respective bottoms
        self.play(
            AnimationGroup(
                *[GrowFromPoint(bar, bar.get_bottom()) for bar in bars],
                lag_ratio=0.1
            )
        )
        self.wait(1)

        # 4. Highlight 1st and 2nd (Stars)
        stars_highlight = VGroup(*[
            SurroundingRectangle(bars[i], color=GOLD, buff=0.05)
            for i in [0, 1]
        ])
        self.play(
            bars[0].animate.set_color(GOLD),
            bars[1].animate.set_color(GOLD),
            Create(stars_highlight)
        )
        self.wait(1)

        # 5. Highlight 5th and 8th (Crucial depth)
        depth_highlight_5 = SurroundingRectangle(bars[4], color=BLUE, buff=0.05)
        depth_highlight_8 = SurroundingRectangle(bars[7], color=BLUE, buff=0.05)

        self.play(
            bars[4].animate.set_color(BLUE),
            bars[7].animate.set_color(BLUE),
            Create(depth_highlight_5),
            Create(depth_highlight_8)
        )

        # Pulsing effect for 5 and 8
        for _ in range(2):
            self.play(
                Indicate(bars[4], color=BLUE, scale_factor=1.2),
                Indicate(bars[7], color=BLUE, scale_factor=1.2),
                run_time=1.0
            )

        self.wait(2)
