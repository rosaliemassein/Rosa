from manim import *
import numpy as np

class RegularizationTradeoff(Scene):
    def construct(self):
        # 1. Setup Axes
        axes = Axes(
            x_range=[1, 11, 1],
            y_range=[0, 11, 2],
            x_length=6,
            y_length=4,
            axis_config={"include_tip": True}
        ).to_edge(LEFT, buff=0.5)
        
        # Explicit labels to avoid keyword confusion
        x_lab = Text("Number of Players", font_size=20)
        y_lab = Text("MSE", font_size=20)
        axes_labels = axes.get_axis_labels(x_label=x_lab, y_label=y_lab)

        # 2. Define Curves using named functions to avoid identifier 'x' issues
        def knn_func(val):
            # Overfitting: starts high, dips, then rises sharply
            # Using a parabola shifted: 0.4(val-4)^2 + 1.5
            return 0.4 * (val - 4)**2 + 1.5

        def regularization_func(val):
            # Steady decline: 8 / sqrt(val)
            return 8 / (np.sqrt(val))

        knn_curve = axes.plot(knn_func, x_range=[1, 10], color=RED)
        knn_label = Text("KNN", color=RED, font_size=18).next_to(knn_curve.get_end(), UP)

        reg_curve = axes.plot(regularization_func, x_range=[1, 10], color=GREEN)
        reg_label = Text("Lasso/Ridge", color=GREEN, font_size=18).next_to(reg_curve.get_end(), RIGHT)

        # 3. Formula
        formula = MathTex(
            r"J(\theta) = \|y - X\theta\|^2 + \lambda \|\theta\|_1",
            font_size=34
        ).to_edge(UP, buff=0.5)
        
        lambda_highlight = SurroundingRectangle(formula[0][12:13], color=YELLOW, buff=0.1)
        penalty_text = Text("Penalty", color=YELLOW, font_size=20).next_to(lambda_highlight, UP)

        # 4. Coefficient Bar Chart
        num_coeffs = 150
        np.random.seed(42)
        # Create random initial values for coefficients
        initial_coeffs = np.random.uniform(0.1, 1.8, num_coeffs)
        
        lambda_tracker = ValueTracker(0)

        def get_coefficient_bars():
            current_lambda = lambda_tracker.get_value()
            bars = VGroup()
            # Total width of chart is 3.5 units
            total_w = 3.5
            single_w = total_w / num_coeffs
            
            for i in range(num_coeffs):
                # Lasso effect: Soft-thresholding
                # h = sign(initial) * max(0, |initial| - lambda)
                # Since all initial are positive here for visual simplicity:
                h = max(0.001, initial_coeffs[i] - current_lambda)
                
                # Change color as they hit zero
                bar_color = BLUE if h > 0.001 else GREY_D
                
                bar = Rectangle(
                    width=single_w,
                    height=h,
                    fill_opacity=0.8,
                    fill_color=bar_color,
                    stroke_width=0
                )
                bars.add(bar)
            
            bars.arrange(RIGHT, buff=0, aligned_edge=DOWN)
            bars.to_edge(RIGHT, buff=0.8).shift(DOWN * 1.2)
            return bars

        coefficient_group = always_redraw(get_coefficient_bars)
        bar_title = Text("Coefficients (θ)", font_size=20).next_to(get_coefficient_bars(), UP)

        # 5. Animation
        self.add(axes, axes_labels)
        self.play(Create(knn_curve), Write(knn_label))
        self.wait(0.5)
        self.play(Create(reg_curve), Write(reg_label))
        self.wait(1)
        
        self.play(Write(formula))
        self.play(Create(lambda_highlight), Write(penalty_text))
        self.wait(0.5)
        
        self.play(FadeIn(bar_title))
        self.add(coefficient_group)
        self.wait(1)
        
        # Increase lambda to show coefficients vanishing
        # rate_functions.linear is used to ensure compatibility
        self.play(
            lambda_tracker.animate.set_value(2.0),
            run_time=6,
            rate_func=rate_functions.linear
        )
        
        self.wait(2)
