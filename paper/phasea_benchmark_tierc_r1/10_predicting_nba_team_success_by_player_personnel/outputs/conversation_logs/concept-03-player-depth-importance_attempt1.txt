from manim import *

class ConceptAnimation(Scene):
    def construct(self):
        # 1. Setup Titles and Formula
        title = Text("Player Rank vs. Team Success", font_size=32).to_edge(UP)
        formula = MathTex(
            r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})",
            font_size=36
        ).next_to(title, DOWN)

        # 2. Create Player Icons (10 Basketball Player Representations)
        player_icons = VGroup(*[
            VGroup(
                Circle(radius=0.25, color=WHITE, fill_opacity=0.1),
                Text(str(i + 1), font_size=18)
            ) for i in range(10)
        ]).arrange(RIGHT, buff=0.5).shift(DOWN * 2)

        # Labels for the axis
        x_axis_label = Text("Player Rank (by Minutes)", font_size=20).next_to(player_icons, DOWN, buff=0.3)

        # 3. Create Average Coefficient Bars
        # Representative data for the visualization
        coeffs = [3.0, 2.5, 1.2, 1.1, 1.8, 0.8, 0.7, 1.4, 0.5, 0.4]
        bars = VGroup()
        for i, val in enumerate(coeffs):
            bar = Rectangle(
                width=0.4,
                height=val,
                fill_color=BLUE_D,
                fill_opacity=0.7,
                stroke_width=1
            )
            bar.next_to(player_icons[i], UP, buff=0.1, aligned_edge=DOWN)
            bars.add(bar)

        # 4. Animation Sequence
        self.play(Write(title))
        self.wait(0.5)

        # Draw player icons
        self.play(
            Create(player_icons),
            Write(x_axis_label),
            run_time=2
        )
        self.wait(0.5)

        # Animate bars growing
        self.play(
            AnimationGroup(
                *[GrowFromEdge(bar, DOWN) for bar in bars],
                lag_ratio=0.1
            ),
            run_time=2
        )
        self.wait(1)

        # Highlight Stars (1st and 2nd bars) with Gold Glow
        star_highlight = VGroup(bars[0], bars[1])
        star_circles = VGroup(player_icons[0], player_icons[1])
        
        self.play(
            star_highlight.animate.set_color(GOLD).set_opacity(1),
            Indicate(star_circles, color=GOLD, scale_factor=1.2),
            run_time=1.5
        )
        self.wait(1)

        # Highlight Depth (5th and 8th bars) with Pulsing Blue Light
        # Indices 4 and 7 correspond to 5th and 8th players
        depth_bars = VGroup(bars[4], bars[7])
        depth_icons = VGroup(player_icons[4], player_icons[7])

        self.play(
            depth_bars.animate.set_color(PURE_BLUE).set_opacity(1),
            depth_icons.animate.set_color(PURE_BLUE),
        )

        # Pulsing effect
        for _ in range(3):
            self.play(
                depth_bars.animate(rate_func=there_and_back).set_stroke(WHITE, width=4),
                depth_icons.animate(rate_func=there_and_back).scale(1.2),
                run_time=0.8
            )

        # Final Formula Reveal
        self.play(Write(formula))
        self.play(
            formula.animate.set_color_by_tex("beta_1", GOLD)
            .set_color_by_tex("beta_5", PURE_BLUE)
            .set_color_by_tex("beta_8", PURE_BLUE)
        )
        
        self.wait(3)
