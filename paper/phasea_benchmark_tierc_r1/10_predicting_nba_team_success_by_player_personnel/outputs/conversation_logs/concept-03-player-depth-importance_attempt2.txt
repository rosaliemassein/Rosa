from manim import *

class PlayerDepthImportance(Scene):
    def construct(self):
        # Step 1: Create player icons (represented by Dots with numbers inside)
        player_dots = [Dot(radius=0.25, color=GRAY_B) for _ in range(10)]
        player_nums = [Text(str(i + 1), font_size=14).move_to(player_dots[i].get_center()) for i in range(10)]
        player_icons = VGroup(*[VGroup(player_dots[i], player_nums[i]) for i in range(10)])
        player_icons.arrange(RIGHT, buff=0.6).to_edge(DOWN, buff=1.5)

        # Labels for 1st, 2nd... 10th below the icons
        label_texts = ["1st", "2nd", "3rd", "4th", "5th", "6th", "7th", "8th", "9th", "10th"]
        labels = VGroup(*[
            Text(label_texts[i], font_size=16).next_to(player_icons[i], DOWN, buff=0.2)
            for i in range(10)
        ])

        # Step 2: Create vertical bars for Average Coefficients
        # Simulated heights emphasizing the 1st, 2nd, 5th, and 8th positions
        heights = [3.2, 2.6, 1.2, 1.1, 2.1, 0.9, 0.8, 1.8, 0.6, 0.4]
        bars = VGroup(*[
            Rectangle(
                width=0.4, 
                height=h, 
                fill_opacity=0.8, 
                fill_color=WHITE, 
                stroke_width=1
            ).next_to(player_icons[i], UP, buff=0.2)
            for i, h in enumerate(heights)
        ])

        # Formula at the top
        formula = MathTex(
            r"\text{Wins} \approx \beta_1(\text{1st\_PIE}) + \beta_5(\text{5th\_PIE}) + \dots + \beta_8(\text{8th\_PIE})",
            font_size=34
        ).to_edge(UP, buff=0.7)

        # Step 3: Animate the players and labels
        self.play(Create(player_icons), Write(labels))
        self.wait(0.5)
        
        # Animate bars growing using GrowFromPoint to simulate growth from the base
        self.play(
            AnimationGroup(
                *[GrowFromPoint(bar, bar.get_edge(DOWN)) for bar in bars],
                lag_ratio=0.1,
                run_time=2
            )
        )
        self.wait(0.5)

        # Step 4: Highlight the top two players with a gold color and glow
        star_highlights = VGroup(*[
            SurroundingRectangle(bars[i], color=GOLD, buff=0.1, stroke_width=4)
            for i in [0, 1]
        ])
        
        self.play(
            bars[0].animate.set_fill(GOLD),
            bars[1].animate.set_fill(GOLD),
            Create(star_highlights),
            run_time=1
        )
        self.wait(0.5)

        # Step 5: Highlight the 5th and 8th bars with pulsing blue light
        # Indices are 4 and 7
        depth_indices = [4, 7]
        
        self.play(
            *[bars[i].animate.set_fill(BLUE) for i in depth_indices],
            *[Flash(bars[i], color=BLUE, line_length=0.2, num_lines=10) for i in depth_indices]
        )

        # Pulsing effect using Indicate
        for _ in range(2):
            self.play(
                Indicate(bars[4], color=BLUE, scale_factor=1.15),
                Indicate(bars[7], color=BLUE, scale_factor=1.15),
                run_time=0.8
            )

        # Step 6: Reveal the formula and link to highlights
        self.play(Write(formula))
        
        # Boxes for specific formula terms matching the bar highlights
        box1 = SurroundingRectangle(formula[0][6:13], color=GOLD)
        box5 = SurroundingRectangle(formula[0][15:22], color=BLUE)
        box8 = SurroundingRectangle(formula[0][26:33], color=BLUE)
        
        self.play(Create(box1))
        self.play(Create(box5), Create(box8))
        
        self.wait(3)
